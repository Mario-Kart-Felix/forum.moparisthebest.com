procedure MouseHuman(X, Y, XR, YR: integer; click, left: Boolean);
var
xx, yy: Integer;
begin
xx := X + Random(XR);
yy := Y + Random(YR);
MoveMouseSmoothEx(xx, yy, 3, 8, 20, 20, 10);
if click = True then
begin
Wait(20 + Random(50));
HoldMouse(xx, yy, left);
Wait(20 + Random(50));
ReleaseMouse(xx, yy, left);
end;
end;

procedure FeatherBuy;
var
xx, yy: Integer;
