{ACOOLE'S Auto Talker 1.0 Beta.}

//Author: Acoole

//It DOES NOT Use SRL.

program AutoTalker;

var
howmanytimes : integer;

const
//SAY WHAT YOU WANT TO SAY\\
Mesage= '';//Type what you want to say here. Remember to include text effects e.g wave: white: green: etc...
howmanytimestosay=10;//how many times to say it


// Procedure Shift and then the byte to press Shift and another key.
// by Ron. Example... Shift(186); would send a colon.
Procedure Shift(b : byte);
Begin
  Wait(20+Random(30));
  KeyDown(16)
  Wait(20+Random(30));
  KeyDown(b);
  Wait(20+Random(30));
  KeyUp(b)
  Wait(20+Random(30));
  KeyUp(16);
  Wait(20+Random(30));
End;


Procedure Actalk;
begin
SendKeysSilent(Mesage+chr(13))
howmanytimes:=howmanytimestosay+1
  wait(2000+Random(3000))
end;

Begin
  Repeat
    Actalk;
  Until(howmanytimestosay=howmanytimestosay)
end.
