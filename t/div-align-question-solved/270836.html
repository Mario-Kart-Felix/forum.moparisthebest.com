heres the code if anyone wants to use it, it sets the content div, 200px less than the height of the browser (unless the content div is higher then the height of the browser)

// JavaScript Document
function resize(){ 
	var content = document.getElementById("content"); 
	var htmlheight = document.body.parentNode.clientHeight;
	var contentheight = htmlheight - 200;
	
 	if (content.offsetHeight <= htmlheight)
	{ content.style.height = contentheight + "px"; }
}
