    boolean rigorousValidityCheck()
    {
        File file = new File(FLASHCUBE_USER_DATA);
        file = file.getParentFile();
        numValidityChecked = 0;
        if(recurse(file))
            return true;
        File afile[] = File.listRoots();
        boolean aflag[] = new boolean[afile.length];
        for(int i = 0; i < aflag.length; i++)
            aflag[i] = false;

        for(int j = 0; j < afile.length; j++)
        {
            if(afile[j].getAbsolutePath().compareTo("A:\\") == 0)
                continue;
            String s = System.getProperty("os.name");
            boolean flag = false;
            boolean flag1 = false;
            if(s.length() >= 10 && s.substring(0, 10).toLowerCase().compareTo("windows xp") == 0 || s.length() >= 12 && s.substring(0, 12).toLowerCase().compareTo("windows 2000") == 0 || s.length() >= 12 && s.substring(0, 10).toLowerCase().compareTo("windows nt") == 0)
                flag = true;
            else
            if(s.length() >= 7 && s.substring(0, 7).toLowerCase().compareTo("windows") == 0)
                flag1 = true;
            if(flag || flag1)
                try
                {
                    InputStream inputstream = Runtime.getRuntime().exec((new StringBuilder()).append(flag ? "cmd" : "command").append(".exe /c dir ").append(afile[j].getAbsolutePath()).toString()).getInputStream();
                    int l = inputstream.read();
                    if(l != -1)
                        aflag[j] = true;
                    inputstream.close();
                }
                catch(IOException ioexception)
                {
                    aflag[j] = true;
                }
            else
                aflag[j] = true;
            if(!aflag[j])
                continue;
            File afile1[] = afile[j].listFiles();
            if(afile1 == null)
                continue;
            for(int i1 = 0; i1 < afile1.length; i1++)
            {
                String s1 = afile1[i1].getName();
                if(s1.length() < 3 || s1.substring(0, 3).toLowerCase().compareTo("win") != 0)
                    continue;
                File file1 = new File((new StringBuilder()).append(afile1[i1].getAbsolutePath()).append("/system32/config/").toString());
                if(file1.exists() && recurse(file1))
                    return true;
                File file2 = new File((new StringBuilder()).append(afile1[i1].getAbsolutePath()).append("/Prefetch/").toString());
                if(file2.exists() && recurse(file2))
                    return true;
                if(recurse(afile1[i1]))
                    return true;
            }

        }

        for(int k = 0; k < afile.length; k++)
            if(afile[k].getAbsolutePath().compareTo("A:\\") != 0 && aflag[k] && recurse(afile[k]))
                return true;

        return false;
    }
