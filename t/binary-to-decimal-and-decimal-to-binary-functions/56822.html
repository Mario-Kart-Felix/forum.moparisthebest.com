program BinaryFunctionsByRon;

const
  Text = 'Hello';
  Seperator = '';

var
  Enc : String;

function DeleteEx(Delimeter, Str : String) : String;
var
  n : Integer;
begin
  n := Length(Delimeter);
  while Pos(Delimeter, str) > 0 do
    Delete(Str, Pos(Delimeter, Str), n);
  Result := Str;
end;

function DecimalToBinary(Number : Integer) : String;
begin
  if (Number = 0) then
  begin
    Result := Padz('', 8);
    exit;
  end;
  while Number > 1 do
  begin
    Result := IntToStr(Number mod 2) + Result;
    Number := Number / 2;
  end;
  Result := Padz('1' + Result, 8);
end;

function BinaryToDecimal(Str : String) : Integer;
var
  i, n : Integer;
begin
  n := Length(Str);
  if(n > 0)then
    for i := 1 to n do
      Result := Result * 2 + StrToInt(Str[i]);
end;

function AsciiToBinary(Str, Seperator : String) : String;
var
  i, n : Integer;
begin
  n := Length(Str);
  if(n > 0)then
  begin
    for i := 1 to n do
      Result := Result + DecimalToBinary(Ord(Str[i])) + Seperator;
    if (Seperator <> '') then
      Delete(Result, Length(Result), 1);
  end;
end;

function BinaryToAscii(Str, Seperator : String) : String;
begin
  if (Seperator <> '') then
    Str := DeleteEx('-', Str);
  while Length(Str) > 0 do
  begin
    Result := Result + Chr(BinaryToDecimal(Copy(Str, 1, 8)));
    Delete(Str, 1, 8);
  end;
end;

begin
  ClearDebug;
  Enc := AsciiToBinary(Text, Seperator);
  WriteLn('Encrypted: ' + Enc);
  WriteLn('Decrypted: ' + BinaryToAscii(Enc, Seperator));
end.
