IF YOU KNOW WHAT YOU ARE DOING these might be useful.

public boolean hasAny(int ... itemList) { for (int item : itemList) { if (hasItem(item)) return true; } return false; }

public boolean hasAll(int ... itemList) { for (int item : itemList) if (!hasItem(item)) return false; return true; }
