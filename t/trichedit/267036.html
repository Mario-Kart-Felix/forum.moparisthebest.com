Nashi July 31, 2016, 6:09am #2

Unfortunately, TStrings are gay so you’ll pretty much have to do this:

[code]var
Buffer: TStringArray;

begin
SafeInitForm;
SafeShowFormModal;
SetArrayLength(Buffer, RichEdit1.Lines.Count);
for i := 0 to High(Buffer) do
Buffer[i] := RichEdit1.Lines.Strings[i];
for i := 0 to High(Buffer) do
Writeln(Buffer[i]);
FreeForm(frmDesign);
end.[/code]
