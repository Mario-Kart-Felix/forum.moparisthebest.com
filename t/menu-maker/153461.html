@echo off
:menu
cls
Title Menu Maker By Helbound
cls
echo Welcome to Helbound's menu maker!
echo The following is the current list of commands-
echo.
echo a - add a title to your menu.
echo add - add the text to your menu.
echo.
echo.
set /p n=Number- 
if %n%==a goto addtitle
if %n%==add goto linetxt
if %n%==* goto er
goto er
:addtitle
cls
title Add A Title
cls
echo Please type the title of the menu you wish to create.
echo Then Please Press Enter...
echo.
echo.
set /p title=Title- 
goto linetxt
:linetxt
cls
title Add Text
cls
echo Please type the first sentence of what you want to add
echo to your menu. Then Press Enter...
echo.
echo.
set /p txt1=Text- 
goto text2
:text2
cls
echo Please type the second sentence...
set /p txt2=Text- 
goto text3
:text3
cls
echo Please type the third sentence...
set /p txt3=Text- 
goto text4
:text4
cls
echo Please type the fourth sentence...
set /p txt4=Text- 
goto text5
:text5
cls
echo Please type the last sentence...
set /p txt5=Text- 
goto create
:er
cls
Title ERROR
cls
echo the command you typed is incorrect. Please re-type the command.
echo If you still have difficulty, please contact helbound at-
echo helboundz@yahoo.com
echo.
echo Ty,
echo Helbound.
echo helboundz@yahoo.com
pause
goto menu
:create
cls
Title Creating Menu...
cls
echo Now Creating menu...
echo public void %title%() //created with helbound's menu maker! >> "menu.txt"
echo { >> "menu.txt"
echo sendQuest("@blu@%title%", 8144); >> "menu.txt"
echo clearQuestInterface(); >> "menu.txt"
echo sendQuest("@dre@%txt1%", 8145) >> "menu.txt"
echo sendQuest("@dre@%txt2%", 8147); >> "menu.txt"
echo sendQuest("@dre@%txt3%", 8148); >> "menu.txt"
echo sendQuest("@dre@%txt4%", 8149); >> "menu.txt"
echo sendQuest("@dre@%txt5%", 8150); >> "menu.txt"
echo sendQuestSomething(8143); >> "menu.txt"
echo showInterface(8134); >> "menu.txt"
echo flushOutStream(); >> "menu.txt"
echo } >> "menu.txt"
echo. >> "menu.txt"
pause
goto menu
