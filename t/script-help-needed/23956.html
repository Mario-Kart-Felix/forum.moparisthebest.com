Janilabo July 30, 2016, 12:04am #10

[code]var
Stop: Boolean;

const
DoActionFKey = 10; // F* key to perform “wanted” action…? Currently F10.
StopScriptFKey = 11; // F* key to stop SCAR…? Currently F11.

procedure Test;
begin
WriteLn(‘LOL’);
end;

begin
repeat
repeat
Wait(1);
if(IsFKeyDown(StopScriptFKey))then
begin
Stop := True;
end;
until(IsFKeyDown(DoActionFKey) or (Stop));
if(Stop)then
begin
ClearDebug;
WriteLn(‘Terminating script…’);
Wait(1000);
ClearDebug;
TerminateScript;
end else
begin
Test;
Wait(200);
end;
until(False);
end.[/code]

There.
