	private boolean checkEmpty(Player player){
		for(int i = 0; i < player.playerEquipment.length; i++){
			if(player.playerEquipment[i] != -1)
				return false;
		}
		for(int i = 0; i < player.playerItems.length; i++){
			if(player.playerItems[i] != 0)
				return false;
		}
		return true;
	}
