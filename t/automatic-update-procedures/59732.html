////////////////////////////////////////////////////////////
// usage: NeedUpdate(‘http://www.google.com’ , ‘1.0’) //

Function NeedUpdate(URL, CurrentVer : String) : Boolean;
Var
WebPage : String;
begin
WebPage := GetPage( URL );
if not (WebPage = CurrentVer) then
Result := True;
end;[/code]
