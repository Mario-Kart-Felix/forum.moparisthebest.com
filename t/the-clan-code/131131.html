quote from clanhandler.java [code]import java.io.*;

public class ClanHandler {

public void createClan(String clanName, String owner) {
try {
    BufferedWriter bw = new BufferedWriter(new FileWriter("./clans/"+clanName+".txt"));
    bw.write(owner);
    bw.newLine();
    bw.flush();
    bw.close();
} catch(IOException e) {
}
}

public void appendToClans(String clanName) {
try {
    BufferedWriter bw = new BufferedWriter(new FileWriter("./clans/clanlist.txt", true));
    bw.write(clanName);
    bw.newLine();
    bw.flush();
    bw.close();
} catch(IOException e) {
    System.err.println("Error adding to the clanlist.txt file");
}
}

public void deleteClan(String clanName) {
try {
    File file = new File("./clans/"+clanName+".txt");
    file.delete();
    BufferedReader br = new BufferedReader(new FileReader("./clans/clanlist.txt"));
    BufferedWriter bw = new BufferedWriter(new FileWriter("./clans/tempclanlist.txt", true));
    String data = null;
    for(int i = 0; i < "./clans/clanlist.txt".length(); i++) {
	while((data = br.readLine()) != null) {
	    if(data.equalsIgnoreCase(clanName)) continue;
	    bw.write(data);
	    bw.newLine();
	    bw.flush();
	}
    }
    bw.close();
    br.close();
    File list = new File("./clans/clanlist.txt");
    list.delete();
    BufferedWriter bb = new BufferedWriter(new FileWriter("./clans/clanlist.txt"));
    BufferedReader bn = new BufferedReader(new FileReader("./clans/tempclanlist.txt"));
    String data2 = null;
    for(int i = 0; i < "./clans/tempclanlist.txt".length(); i++) {
	while((data2 = bn.readLine()) != null) {
	    bb.write(data2);
	    bb.newLine();
	    bb.flush();
	}
    }
    bb.close();
    bn.close();
    File tlist = new File("./clans/tempclanlist.txt");
    tlist.delete();
} catch(Exception e) {
}
}

public boolean checkClanExist(String clanName) {
try {
    BufferedReader br = new BufferedReader(new FileReader("./clans/"+clanName+".txt"));
    br.close();
    return true;
} catch(IOException e) {
}
return false;
}

public void addMember(String clanName, String member) {
try {
    BufferedWriter bw = new BufferedWriter(new FileWriter("./clans/"+clanName+".txt", true));
    bw.write(member);
    bw.newLine();
    bw.flush();
    bw.close();
} catch(IOException e) {
}
}

public boolean checkClanMember(String clanName, String name) {
try {
    BufferedReader br = new BufferedReader(new FileReader("./clans/"+clanName+".txt"));
    String userName = null;
    while((userName = br.readLine()) != null) {
	if(userName.equalsIgnoreCase(name)) {
	    return true;
	}
    }
    br.close();
} catch(IOException e) {
    System.err.println("Error checking if this player is in given clan");
}
return false;
}
