Why don’t you use the command packet, it’s easy to do, here you have it:

[code]
package com.phoenix.runescape.network.packet.receive;

import com.phoenix.runescape.item.Item;
import com.phoenix.runescape.mobile.player.Player;
import com.phoenix.runescape.network.packet.IncomingPacket;
import com.phoenix.runescape.network.packet.IncomingPacketListener;
import com.phoenix.runescape.network.packet.IncomingPacketOpcode;

@IncomingPacketOpcode( 103 )

public final class CommandPacketListener extends IncomingPacketListener {

@Override
public void operate(Player player, IncomingPacket packet) {
	int messageSize = packet.getLength() - 1;
	byte message[] = packet.readBytes(messageSize);
	String command = new String(message);
	
	if (command.equalsIgnoreCase("test")) {
		player.getInventoryContainer().addItem(new Item(995, 1000));
	}
}

}[/code]
