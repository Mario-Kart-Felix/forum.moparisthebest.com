Something like this?

Program WeirdoScript;

Const
     Color1 = 10485759; // Color1 of whatever.
     Color2 = 10527138; // Color2 of whatever.
     Color3 = 8160396;  // Color3 of whatever.

Var
   x, y : Integer;

Procedure ClickColor(color : Integer);
Begin
     If (FindColor(x, y, color, 1000, 1000, 1000, 1000)) Then
     Begin
          MoveMouseSmooth(x, y);
          ClickMouse(x, y, true);
          Wait(20+Random(50));
          ClickMouse(x, y, true);
     End;
End;

Procedure ClickLoop;
Begin
     ClickColor(color1);
     ClickColor(color2);
     ClickColor(color3);
End;

Begin
     Repeat
           ClickLoop;
     Until(False)
End.
