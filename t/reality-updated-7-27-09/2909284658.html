[code=java]package net.rly.srv.quests;

import net.rly.srv.client;

/**

  • @author Newty
  • Quest - Find osama, recieve GP after caught (?)
    */

public class OsamaHunt extends QuestSys {

//To check if Osama has been caught or not
boolean osamaCaught = false;

/**
 * Start quest, finish quest
 */
public static void Prince_Ali(client c) {
	if(c.qStage[3] == 1) {
		c.qStage[3] = 2;
		npcDiag(c, "Prince_Ali", "Well hello there "+c.playerName+
				"there is a problem here in Al Karid! A terrorist named Sadam is attacking!"
				"If you want to help please check the Al Karid duel arena!");
	} else if(c.qStage[3] == 0) {
		npcDiag(c, "Prince_Ali", "Well thank you "+c.playerName+
					"I am happy that you caught this bastard!"
					"Here's your reward for saving Al Karid");
		c.addItem(995, 1000000);
		
}

/**
 * Tells you Osama is at Al Karid mining area.
 */
public static void DuelGuard(client c) {
	if(c.qStage[3] == 2) {
		c.qStage[3] == 3;
		npcDiag(c, "Guard", "Well, so Prince Ali has sent you eh? Well I heard"
					"the terrorist was attacking in the mines of Al Karid"
					"affending the miners...maybe you can check there?"
					"Oh by the way...maybe a rope will catch him!");
	} else {
		npcDiag(c, "Guard", "The duel arena is a really nice place. Check it out!");
}


/**
  * use rope on adamant ore you get Osama caught? /win
  */
public static void catchOsama(client c) {
	if(c.IsItemInBag(954) && useOnObject == 2258) {
		c.sM("You look around the rock....you see Osama.");
		c.sM("You catch Osama...");
		osamaCaught = true;
	}
}

public static void Osama(client c) {
	if(c.qStage[3] == 3) {
		c.qStage[3] = 4;
		if(!catchOsama()) {
			npcDiag(c, "Osama", "Hehe, soon Al Karid will be mine!");
			c.sM("This isn't a good idea, what should I do?");
		} else if(catchOsama()) {
			c.sM("Maybe I should talk to Prince Ali!");
	}
}

}[/code]
