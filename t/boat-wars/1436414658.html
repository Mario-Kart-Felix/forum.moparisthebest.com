import java.awt.*;
import java.applet.Applet;
public class Boat extends Applet
{
	public void init() {
		repaint();
	}
	int[] x = new int[]{75,100,200,225};
	int[] y = new int[]{150,200,200,150};
	int[] x1 = new int[]{150,200,150};
	int[] y1 = new int[]{125,75,50};
	public void paint( Graphics g ) {
		g.setColor(Color.cyan);
		g.fillRect(0, 0,300,300);
		g.setColor(Color.darkGray);
		g.fillPolygon(x, y, x.length);
		g.setColor(Color.BLACK);
		g.drawString("S.S. JavaBoat!", 90,175);
		g.drawLine(150, 150, 150, 50);
		g.setColor(Color.WHITE);
		g.fillPolygon(x1, y1, x1.length);
		g.setColor(Color.BLUE);
		g.fillRect(0, 180,300,150);
		
	}
}
