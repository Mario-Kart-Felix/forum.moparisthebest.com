public class JarPreparator {
    private Vector<ClassGen> classGen = new Vector<ClassGen>();
    private HashMap<String, ClassGen> loaderCrap = new HashMap<String, ClassGen>();
    private Vector<ClassGen> classArray = new Vector<ClassGen>();
    private URLClassLoader classLoader;
    private String signlinkName;

    public JarPreparator(JarFile file) throws IOException {
        Vector<ClassGen> classes = new Vector<ClassGen>();
        Enumeration<JarEntry> entries = file.entries();
        while (entries.hasMoreElements()) {
            JarEntry entry = entries.nextElement(); //Load the next element in our jar file.
            if (entry.getName().endsWith(".class")) {
                JavaClass clazz = new ClassParser(file.getInputStream(entry), entry.getName()).parse();
                classes.add(new ClassGen(clazz));
            }
        }
        JarFile loaderfile = new JarFile("C:/loader.jar");
        Enumeration<JarEntry> entriez = loaderfile.entries();
        while (entriez.hasMoreElements()) {
            JarEntry entry = entriez.nextElement(); //Load the next element in our jar file.
            if (entry.getName().endsWith(".class")) {
                JavaClass clazz = new ClassParser(loaderfile.getInputStream(entry), entry.getName()).parse();
                loaderCrap.put(clazz.getClassName(), new ClassGen(clazz));
            }
        }
        System.out.println("Loaded " + (classes.size() + loaderCrap.size()) + " potential classes.");
        this.classGen = classes;
        runUpdates();
        classLoader = new URLClassLoader(new URL[]{new URL("file:C:/loader.jar")});
    }

    public void runUpdates() {
        int replaced = 0;
        Vector<ClassGen> vg = new Vector<ClassGen>();
        //System.out.println("Starting internal updater.");
        for (ClassGen cg : classGen) {
            if (loaderCrap.containsKey(cg.getClassName()) && cg.getClassName().length() == 2) {
                vg.add(loaderCrap.get(cg.getClassName()));
                //System.out.println("\tReplaced clone class: " + cg.getClassName());
                replaced++;
            } else {
                vg.add(cg);
            }
        }
        for (ClassGen cg : vg) {
            if (cg.getSuperclassName().contains("Canvas")) {
                cg.getConstantPool().setConstant(cg.getSuperclassNameIndex(), new ConstantClass(cg.getConstantPool().addUtf8("com/reflect0r/core/listeners/CanvasUpdateListener")));
                //System.out.println("\tCanvas class: " + cg.getClassName() + " => com.reflect0r.core.listeners.CanvasUpdateListener");
            }
            if (cg.getConstantPool().lookupUtf8("setFocusTraversalKeysEnabled") != -1) {
                signlinkName = cg.getClassName();
                System.out.println("\tSignlink class: "+ cg.getClassName() +" => Detected.");
            }
        }
        classArray = vg;
    }

    public Class loadClass(String name) {
        try {
            return classLoader.loadClass(name);
        } catch (ClassNotFoundException e) {
            System.out.println("ClassLoader: " + name + " does not exist.");
        }
        return null;
    }

    public String getSignlink() {
        return signlinkName;
    }
}
