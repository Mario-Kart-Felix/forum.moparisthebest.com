[code=Java]
import java.util.Scanner;
import java.util.stream.Stream;

public class LoginChecker {

private static User[] users = { new User("Jim", "12345"),
		new User("Carl", "carls-password"), new User("Mopar", "or-no-car") };

public static void main(String... args) {
	Scanner scanner = new Scanner(System.in);

	System.out.println("Please enter a username.");
	String name = scanner.nextLine();
	System.out.println("Please enter the password for this account: " + name);
	String password = scanner.nextLine();
	System.out.println("Attempting to login: " + name + ", with password: " + password + ".");
	scanner.close();

	String answer = validateUser(new User(name, password)) ? "Sucessful login!" : "Login failed.";
	System.out.println(answer);
	System.exit(0);
}

private static boolean validateUser(User user) {
	return Stream.of(users).anyMatch(u -> u.equals(user));
}

private static class User {

	private String name;
	private String password;

	public User(String name, String password) {
		this.name = name;
		this.password = password;
	}

	public String getName() {
		return name;
	}

	public String getPassword() {
		return password;
	}

	@Override
	public boolean equals(Object object) {
		User user = (User) object;
		return user.getName().equalsIgnoreCase(name) && user.getPassword().equals(password);
	}
}
