begin
lvlcount:=0
SleepAdjust:=0
repeat
if (lvlcount=17) then
SleepAdjust:= some integer you’d like to decrease the wait time by… say… 30

sleep(the current wait time-SleepAdjust)
more code that is sending keys
lvlcount:=lvlcount+1
until(whatever it does)
[/code]
