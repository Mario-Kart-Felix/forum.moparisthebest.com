hmm

you can make it drop the ore using the srl mining include…

I modified your script, should work good now :rolleyes:

[code]program PowerMiner;

{.include SRL/SRL.scar}
{.include SRL/SRL/skill/Mining.scar}

const
RockColour = 2174282 ; //Colour of rock
WaitTime = 3000 ; //Timing interval betwee nclickin each rock
TimesToRun = 1 ; //How many times to run

var
i : integer;
loads:integer;

Procedure minerock; //The mine rock procedure
var x, y : integer;

begin;
repeat
findcolorspiral(x,y,RockColour,3,3,515,336) //Find the colour of rock
mmouse(x,y,3,3) //Move mouse to rock
istextat2(9,9,‘Mine’,20) //Make sure there is text, Mine
mouse(x,y,2,2,true)// Click the rock
wait(WaitTime+random(300)) //Wait for set time and + random 300 milisecs
until(invfull) //Mine until ivn full
if (invfull = true) then
loads := loads + 1; //+1 to the loads
end;

Procedure dropore; //Drop whole inv procedure
begin;
DropAll; //Will drop EVERYTHING in inventory
end;

procedure ProgressReport; //Progrress report
begin;
WriteLn(’ ‘);
WriteLn(’… Progress Report:…’);
WriteLn(’ ‘);
WriteLn(‘Mined and Dropped ’ + inttostr(loads) + ’ loads.’);
WriteLn(’ ');
WriteLn('Made by ')if (i >=TimesToRun)then //If done all jobs then…
WriteLn(‘Done all jobs’) //Write in, done all jobs
end;

begin
loads:=0; //Defult the loads to 0
SetupSRL; //Setup SRL
SetupSRLMining; //Setup the SRL mining
BenMouse:= False; //Make benmouse = false
repeat //Repeat all of the above
minerock; //Begin mining
DisguiseScar('Done ’ + inttostr(loads) + ‘loads’) //Disguise scar as how many loads done
dropore; //Drop the ore
ProgressReport; //Print Progress report
i :=i+1;
until(i >=TimesToRun); //If done all jobs
TerminateScript; //Terminate script
end.[/code]

I commented the script to show you how it works

Happy scripting
