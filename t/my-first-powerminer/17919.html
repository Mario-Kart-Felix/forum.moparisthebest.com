[scar]{--------------------Arr0w’s Powerminer--------------------}
{ To start this scipt fill the colors of the rock in the setup}
{ Then start next to the rock you want to mine.}

program autoMiner;
{.include SRL/SRL.scar}
{.include SRL/SRL/Skill/Mining.Scar}
const

rockcolor = 0 ; // the color of the rock
rockcolor1 = 0 ; // try get a different color of the rock

procedure miner;
begin
repeat
wait(1000+random(10))
if(findcolorspiral(x,y,rockcolor,3,3,515,336)) or
(findcolorspiral(x,y,rockcolor1,3,3,515,336)) then
Mmouse(x,y,3,3)
wait(300+random(200))
if (istextat2(x,y,‘Mine’,5) = true) then
Wait(1000+random(500))
Mouse(x,y,2,2,true)
wait(1000+random(1000))
until(invfull)
end;

procedure FFight;
begin
if (FindFight) then
begin
Writeln(‘Found Fight’);
RunAwayDirection(‘N’);
writeln(‘Running Away’)
Wait(8000 + random(3000));
RunBack;
Writeln(‘Running Back’)
end;
if (FindNewBox) then
begin
writeln(‘Found Strange Box Trying To Solve…’)
OpenBox;
SolveBox;
wait(1000);
end;
if (FindNewBox) then
begin
GambleNewBox;
end;
if (FindNewBox) then
begin
LogOut;
Writeln(‘Unable To Solve Box…Terminating’);
TerminateScript;
end;
end;

procedure DropOre;
begin
MouseSpeed:= 13+random(7);
DropTo(2, 14);
DropTo(15, 28);
end;

begin
SetupSRL;
SetupSrlMining;
FindNormalRandoms;
SRLRandomsReport;
FindPickHeadColor;
FindPick;
ChatsOff;
SetRun(true);
repeat
FindNormalRandoms;
FFight;
Miner;
DropOre;
Writeln(‘Thank you for using arr0ws powerminer’)
until(false)
end.[/scar]
theres that’s my first real script.
