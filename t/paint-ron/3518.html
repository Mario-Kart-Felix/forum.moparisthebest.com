Hey I was bored so I found out how to draw my name in Paint using clickmouse. Not that hard… just a little fun lol. Just copy and paste the script at the bottom and then target the entire screen. Make sure you have a big paint thing open, and click on any of the paint tools, then run the script.

Change the WaitT at the bottom to whatever number. Default is 500 (half second). Thats the speed of each click.

program RON;

Var WaitT : Integer;

Procedure R;
Begin
     ClickMouse(100, 100, true);
     Wait(WaitT);
     ClickMouse(100, 110, true);
     Wait(WaitT);
     ClickMouse(100, 120, true);
     Wait(WaitT);
     ClickMouse(100, 130, true);
     Wait(WaitT);
     ClickMouse(100, 140, true);
     Wait(WaitT);
     ClickMouse(100, 150, true);
     Wait(WaitT);
     ClickMouse(110, 100, true);
     Wait(WaitT);
     ClickMouse(120, 100, true);
     Wait(WaitT);
     ClickMouse(130, 110, true);
     Wait(WaitT);
     ClickMouse(120, 120, true);

     ClickMouse(110, 120, true);
     Wait(WaitT);
     ClickMouse(120, 130, true);
     Wait(WaitT);
     ClickMouse(130, 140, true);
     Wait(WaitT);
     ClickMouse(140, 150, true);
     Wait(WaitT);
End;

Procedure O;
Begin
     ClickMouse(170, 100, true);

End;

Procedure N;
Begin
     ClickMouse(220, 150, true);
