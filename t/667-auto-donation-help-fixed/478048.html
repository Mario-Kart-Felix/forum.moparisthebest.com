public static ResultSet query(String s) throws SQLException {
	try {
		if (s.toLowerCase().startsWith("select")) {
			ResultSet rs = stmt.executeQuery(s);
			return rs;
		} else {
			stmt.executeUpdate(s);
		}
		return null;
	} catch (Exception e) {
		destroyConnection();
	}
	return null;
}

public static void destroyConnection() {
	try {
		stmt.close();
		con.close();
	} catch (Exception e) {
	}
}
