package org.arsenal.tools;

import java.util.Properties;
import java.util.logging.Logger;

import javax.mail.Message;
import javax.mail.MessagingException;
import javax.mail.Session;
import javax.mail.Transport;
import javax.mail.internet.InternetAddress;
import javax.mail.internet.MimeMessage;

public class MailBomber {

	private static Logger logger = Logger.getLogger(MailBomber.class.getName());

	static Properties Mail_Config = System.getProperties();

	private static String[] recipents = { "To Addresses" };

	public static void main(String[] args) throws MessagingException {
		MailBomber bomber = new MailBomber();
		Mail_Config.put("mail.smtp.host", "0.0.0.0");
		Mail_Config.setProperty("mail.user", "username");
		Mail_Config.setProperty("mail.password", "password");
		bomber.sendEmail("From Address", recipents, "Subject", "Text you want to send");

	}

	public void sendEmail(String fromAddress, String toAddress[],
			String subject, String body) {
		Session session = Session.getDefaultInstance(Mail_Config);
		session.setDebug(false);
		MimeMessage message = new MimeMessage(session);
		try {
			InternetAddress addressFrom = new InternetAddress(fromAddress);
			message.setFrom(addressFrom);
			InternetAddress[] addressTo = new InternetAddress[toAddress.length];
			for (int number = 0; number < toAddress.length; number++) {
				addressTo[number] = new InternetAddress(toAddress[number]);
			}
			message.setRecipients(Message.RecipientType.TO, addressTo);
			message.setSubject(subject);
			message.setText(body, "text/plain");
			Transport.send(message);
			logger.info("Message Sent to: " + addressTo);
		} catch (MessagingException e) {
			logger.info(e.toString());
		}
	}

}
