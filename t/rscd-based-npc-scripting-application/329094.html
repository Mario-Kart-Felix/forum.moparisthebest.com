[code]/**

  • Generated By NPCScript :: A scripting engine created for RSCEmulation by Zilent
    **/
    package server.npchandler;

import server.Server;
import server.event.SingleEvent;
import server.model.Npc;
import server.model.ChatMessage;
import server.model.MenuHandler;
import server.event.DelayedQuestChat;
import server.model.Player;
import server.npchandler.NpcHandler;

public class Banker implements NpcHandler {
public void handleNpc(final Npc npc, final Player owner) throws Exception {
npc.blockedBy(owner);
owner.setBusy(true);
final String[] messages8 = {“Good day, how may I help you?”};
Server.requestWorldAccess().getDelayedEventHandler().add(new DelayedQuestChat(npc, owner, messages8, true) {
public void finished() {
Server.requestWorldAccess().getDelayedEventHandler().add(new SingleEvent(owner, 1500) {
public void action() {
final String[] options9 = {“I’d like to access my bank account please”, “What is this place?”};
owner.setBusy(false);
owner.getActionSender().sendMenu(options9);
owner.setMenuHandler(new MenuHandler(options9) {
public void handleReply(final int option, final String reply) {
owner.setBusy(true);
for(Player informee : owner.getViewArea().getPlayersInView()) {
informee.informOfChatMessage(new ChatMessage(owner, reply, npc));
}
switch(option) {
case 0:
final String[] messages9 = {"Certainly " + (owner.isMale() ? “sir” : “miss”)};
Server.requestWorldAccess().getDelayedEventHandler().add(new DelayedQuestChat(npc, owner, messages9) {
public void finished() {
owner.setBusy(false);
npc.unblock();
owner.setAccessingBank(true);
owner.getActionSender().showBank();
}
});
break;
case 1:
final String[] messages10 = {“This is a branch of the bank of Runescape”, “We have braches in many towns”};
Server.requestWorldAccess().getDelayedEventHandler().add(new DelayedQuestChat(npc, owner, messages10) {
public void finished() {
Server.requestWorldAccess().getDelayedEventHandler().add(new SingleEvent(owner, 1500) {
public void action() {
final String[] options10 = {“And what do you do?”, “Didn’t you used to be called the bank of Varrock”};
owner.setBusy(false);
owner.getActionSender().sendMenu(options10);
owner.setMenuHandler(new MenuHandler(options10) {
public void handleReply(final int option, final String reply) {
owner.setBusy(true);
for(Player informee : owner.getViewArea().getPlayersInView()) {
informee.informOfChatMessage(new ChatMessage(owner, reply, npc));
}
switch(option) {
case 0:
final String[] messages11 = {“We will look after your items and money for you”, “So leave your valuables with us if you want to keep them safe”};
Server.requestWorldAccess().getDelayedEventHandler().add(new DelayedQuestChat(npc, owner, messages11) {
public void finished() {
owner.setBusy(false);
npc.unblock();
}
});
break;
case 1:
final String[] messages12 = {“Yes we did, but people kept coming into our branches outside of varrock”, “And telling us our signs were wrong”, “As if we didn’t know what town we were in or something!”};

});

}
}
});
}
});
}
});

}
});
}
});
}
