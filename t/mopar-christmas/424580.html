import java.applet.Applet;
import java.awt.*;
import java.util.ArrayList;

public class Christmas extends Applet implements Runnable{
	Thread t;
	Image Buffer;
	Graphics gBuffer;
	final int MAXFLAKES = 300;
	ArrayList<SnowFlake> flakes  = new ArrayList<SnowFlake>();

	
	public void init() {
		setSize(800,600);
		setBackground(Color.BLACK);
        Buffer=createImage(getWidth(),getHeight());
        gBuffer=Buffer.getGraphics();
              
        

	}
	
	public void start() {
		if(t == null) {
			t = new Thread(this);
			t.start();
		}
	}
	
	public void addFlakes() {
		try {
			t.sleep(50);
		} catch (Exception e) {
			//nothing to see here
		}
		
		
		flakes.add(new SnowFlake(800,600));
	}
	
	@Override
	public void run() {
		while(true) {
			try {
				t.sleep(10);
			} catch (Exception e) {
				// nothin to see here
			}
			
			if(flakes.size() < MAXFLAKES) {
				addFlakes();
			}
			
			for(int i = 0; i < flakes.size(); i++) {
				flakes.get(i).move();
			}
			
			repaint();
		}
		
	}
	
    public void update(Graphics g)
    {
            paint(g);
    }
	
	public void paint(Graphics g) {
		Image img2 = Toolkit.getDefaultToolkit().getImage("mopar.jpg");
		gBuffer.drawImage(img2, 0, 0, 800, 600, this);
		
		Font font = new Font("Arial",Font.BOLD,30);
		gBuffer.setFont(font);
		gBuffer.setColor(Color.BLACK);
		gBuffer.drawString("Merry Christmas Mopar!", 220,40);

		gBuffer.setColor(Color.WHITE);
		for(int i = 0; i < flakes.size(); i++) {
			flakes.get(i).paint(gBuffer);
		}
		
		g.drawImage(Buffer,0,0,this);
		
	}

}

import java.awt.*;
import java.util.Random;

public class SnowFlake {
	private int width, height, x, y, distance, velX, velY;
	Graphics g;
	Font font;

	public SnowFlake(int width, int height) {
		this.width = width;
		this.height = height;
		
		//set flake at a random distance
		Random r = new Random();
		distance = r.nextInt(5);

		velY = distance; //the further the way it is the slower it falls
		
		r = new Random();
		//set flake at random velocity dependent on distance
		velX = r.nextInt(2) * (distance - 2);
		

		
		//change size of flake depending on distance
		font = new Font("Arial",Font.BOLD,distance * 4 + 2); // 2..18
		
		//place randomly on x axis
    	x = r.nextInt(800);
    	
 
    	
	}
	
	/*
	 *  Moves the snowflake
	 */
	public void move() {
		if (x > width || y - distance - 5 > height || x < -distance) {
			//set it back to it's starting position
	    	Random r = new Random();
	    	y = 0;
	    	x = r.nextInt(width);
		} else {
			x += velX;
			//so it falls
			y += velY;
		}
		
	}
	
	public void paint(Graphics g) {

		g.setFont(font);
		this.g = g;
		g.drawString("*", x, y);
	}
	
}
