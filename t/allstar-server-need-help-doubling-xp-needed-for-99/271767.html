use this one:

public int getLevelForXP(int exp) {
    int points = 0;
    int output = 0;
	if (exp > 13000000)
		return 99;

    for (int lvl = 1; lvl <= 99; lvl++) {
        points += Math.floor(
                (double) lvl + 150.0 * Math.pow(2.0, (double) lvl / 7.0));
        output = (int) Math.floor(points / 4);
        if (output >= exp) {
            return lvl;
        }
    }
    return 99;
}
