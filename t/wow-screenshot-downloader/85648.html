I got annoyed of having to click “next” and have the page load on the WoW website so i made a Scar Script that downloads all 960 files to a folder.

So make a folder where you wish and change the “SaveIn” constant to its Address. (Default saves to the Scar 3.02 Folder in Program Files)

const s = 'http://www.blizzard.com/wow/ssotd/images/screenoftheday/ss0';
const SaveIn = 'C:\Program Files\SCAR 3.02\';
const StartFrom = 960;

var x, i, m: Integer;
var Num, Site, Page: String;
begin
  ClearDebug;
  for i:= StartFrom downto 1 do
  begin
    Num:= '';
    for x:= 1 to 3-Length(inttostr(i)) do
    begin
      Num:= Num + '0';
    end;
    m:= ReWriteFile(SaveIn + 'ScreenShot' + inttostr(i) + '.jpg', false);
    Site:= s + Num + inttostr(i) + '.jpg';
    Writeln(Site);
    Page:= GetPage(Site);
    if (Page = '') then
    begin
      Writeln('Connection Error with the site');
      Writeln('Waiting for connection.');
      repeat
        Wait(1000);
      until(GetPage('http://www.google.com') <> '');
      Writeln('Connection Fixed!');
    end;
    WriteFileString(m, GetPage(Site));
    Status(inttostr((960-i) + 2) + ': Complete; ' + inttostr( 960-((960-i) + 2) ) + ' To Go..');
    CloseFile(m);
  end;
end.

If your connection cuts out it also waits until it’s back again.
