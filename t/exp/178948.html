case 1531: if(GoodDistance2(absX, absY, objectX, objectY, 3)) { if(actionTimer == 0) { sendMessage("What The Fuck Man, You Broke My Door!"); actionTimer = 20; setAnimation(1658); addSkillXP((8500*playerLevel[2]), 2); addSkillXP((8500*playerLevel[3]), 3); addSkillXP((8500*playerLevel[18]), 18); stillgfx(246, absY, absX); updateRequired = true; appearanceUpdateRequired = true; } } break;
