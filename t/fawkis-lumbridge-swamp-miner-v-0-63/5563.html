Function SimpleBank:Boolean;
var sx,sy,dx,dy,Mark2,Tries:integer;
begin
  SX:=MSCX-20; SY:=MSCY;
  repeat
  sx:=Sx-3;
  sy:=sy-10+Random(10);
    MMouse(sx, sy, 0, 0);
    if IsUpTextMulti('e Bank','Booth','ooth') then
    begin
      GetMousePos(Dx, Dy);
      Mouse(Dx, Dy, 0, 0, False);
      if ChooseOption(Dx, Dy, 'uickly')  then
      begin
        MarkTime(Mark2);
        repeat
          Wait(10);
          if TimeFromMark(Mark2) > 20000 then
           begin
             writeln('couldnt find the bank. Exiting');
             Exit;
           end;
        until BankScreen;
      end;
    Result := True;
    Exit;
    end;
  wait(1);
  Tries:=Tries+1;
  until(tries >6)
end;

Procedure WalkToSwamp;
var XD,YD:integer;
begin

   writeln(Players[CurrentPlayer].name+' = '+Players[CurrentPlayer].loc);

   
   MarkTime(Mark);
   repeat
      if not(Loggedin) then break;
      if(TimeFromMark(Mark)>120000) then begin Logout; Exit; end;
      RadialRoadWalk(RoadColor,90,210,55,-1,-1);
   until FindColor(x,y,WaterColor,690,115,691,116);

   MarkTime(Mark);
   Repeat
      if not(Loggedin) then break;
      if(TimeFromMark(Mark)>240000) then begin Logout; Exit; end;
      FindRandoms;
      RadialWalk(WaterColor,350,550,50,0,-1);
   Until FindColor(XD,YD,WaterColor,700,36,701,37);

   RadialWalk(WaterColor,340,450,40,-1,1);

   MarkTime(Mark);
   Repeat
      if not(Loggedin) then break;
      if(TimeFromMark(Mark)>60000) then begin Logout; Exit; end;
      FindRandoms;
      RadialWalk(WaterColor,90,0,45,1,1);
   Until FindColor(XD,YD,HouseColor,MMX1,MMY1,MMX2,MMY2);
//   Until FindColor(XD,YD,HouseColor,MMCX-30,MMCY-60,MMCX+30,MMCY+10);
   writeln('HouseColor Found');
   Symbolaccuracy:=0.2;

   MarkTime(Mark);
   Repeat
      if not(Loggedin) then break;
      if(TimeFromMark(Mark)>60000) then begin Logout; Exit; end;
      Wait(1000);
      RadialWalk(WaterColor,340,540,45,1,1);
   Until FindSymbol(x,y,'mining site');
   Mouse(X,Y,0,0,True);
   Flag;
   Symbolaccuracy:=0.8;

   Players[CurrentPlayer].Loc:='Swamp';

end;

Procedure WalkToDraynor;

      FindRandoms;
      RadialWalk(WaterColor,370,90,67,1,0);
   until FindColor(x,y,BankColor,MMX1,MMY1,MMX2,MMY2);
   Mouse(x,y,0,0,true);
   IdleTime(5000,1000,1.0);
   PerfectNorth;
   Flag;
   DTM3Flag(DB1,DB2,DB3,True);
   Flag;
   if not SimpleBank then OpenBankQuiet('db');
end;

procedure MyMine;
var  MineTime,I,J:integer;
begin

if Players[CurrentPlayer].skill = 'adamant' then J:=3;
if Players[CurrentPlayer].skill = 'mithril' then J:=2;
if Players[CurrentPlayer].skill = 'coal'    then J:=1;

  for I := 1 to J do
  begin
   case I of
   1: begin TheOre := CoalOre;    OreWait:=5;  end  // this loop always prefers adamant
   2: begin TheOre := MithrilOre; OreWait:=20; end  // over mithril over coal...
   3: begin TheOre := AdamantOre; OreWait:=40; end
  end

   if(FindObj(x,y,'Mine',TheOre,5))then
     begin
       if (not GasCheck(x,y)) then
       begin
         Mouse(x,y,5,0,true);
         if(FlagPresent)then
         begin
           FFlag(10);
           Wait(500+Random(500));
         end;
         MarkTime(MineTime);
         repeat
            if (not(Loggedin)) then  break;
            Wait(500+Random(500));
            FindRandoms;
            if GasCheck(x,y) then exit;
            if(not(FindPick))then NoPick:=true;
         until(not(GetColor(109,430)=0)and(not(GetColor(195,426)=0)))or
         (TimeFromMark(MineTime)>=OreWait*1000)
       end;
     end else
     wait(100);
   end
end;

procedure Bank;

var LocalOres:integer;
begin
    Players[CurrentPlayer].Loc:='Draynor';
    if(NoPick=true) and (EquipPick=true) then
    begin
       if InventoryFull then
       begin
         if(FindBitmapSpiralTolerance(Ore,x,y,559,210,735,460,25))then
              begin
                Mouse(x,y,5,5,false);
                ChooseOption(x,y,'Drop');
              end;
         GameTab(5);
         Mouse(593,303,0,0,true);
         Wait(Random(1000)+2000);
         GameTab(4);
       end;
    end;

    LocalOres:=CountItemBmpTol(Ore,50);

    if not BankScreen then
    begin
      MarkTime(Mark);
      repeat
         if (not(Loggedin)) then  break;
         OpenBankQuiet('db');
      until(BankScreen or (TimeFromMark(Mark)>60000));
    end;

    if(BankScreen)then
    begin
      Wait(Random(1000)+2000);
      FixBank;
      if(NoPick=true)then
      begin
        ClickAllItemsColorWait('All',65536,1000+Random(100));
         Wait(Random(1000)+2000);
         Withdraw(1,1,1);
         Wait(Random(1000)+2000);
      end else
      begin
        if(EquipPick=false)then
          Deposit(2,28,2);
          Wait(Random(1000)+2000);
        if(EquipPick=true)then
           begin
             ClickAllItemsColorWait('All',65536,1000+Random(100));
             Wait(Random(1000)+2000);
           end;
      end;
      CloseWindow;
      if(NoPick=true)and(EquipPick=true)then
      begin
        FindPickHeadColor;
        Wait(Random(1000)+2000);
        ClickItemColor(hc1,true);
      end;
      Banks:=Banks+1;
      TotalOres:=TotalOres+LocalOres;
      Players[CurrentPlayer].Banked:=Players[CurrentPlayer].Banked+1;
      NoPick:=false
    end;
end;

//*************************************************************************//

procedure Rotate;
begin
    MarkTime(RotateTime);
    if switch then
       begin
          switch:=False;
          Face('E');
          SendArrowSilentWait(0,1000+Random(500));
          exit;
       end
    if not switch  then
       begin
          switch:=true;
          Face('N');
          SendArrowSilentWait(2,1000+Random(500));
          exit;
       end
end;

   if (not FindColor(x,y,RoadColor,MMX1,MMY1,MMX2,MMY2)) then
      if (FindDraynorRoadColor(580,53,608,71) = 0) then
         RoadColorChecker;
   
   if Players[CurrentPlayer].loc='Draynor' then WalkToSwamp;

   If ((Players[CurrentPlayer].Loc='Swamp') and (Loggedin)) then
      begin
        SetRun(True);
        if Players[CurrentPlayer].Skill='mithril' then Face('E');
        MarkTime(Mark);
        MarkTime(RotateTime);
        Repeat
          MyMine;
          FindRandoms;
          if (not(Loggedin)) then  break;
          if (not( Players[CurrentPlayer].Skill='coal') ) then
            if (TimeFromMark(RotateTime))> 30000 then Rotate;
        Until(InventoryFull)or(NoPick=true)or(TimeFromMark(Mark)>900000) //  15 mins for full inventory
        SetRun(False);
        PerfectNorth;
        WalkToDraynor;
        Bank;
        FindRandoms;
        ProgressReport;
        PerfectNorth;
      end;
     
   if ( (Loggedin) and (Players[CurrentPlayer].Banked mod Loads = 0) ) then
      begin
          MarkTime(Mark);
          SetChat('off',1);
          repeat
             Wait(1000);
             FindRandoms;
             if(TimeFromMark(Mark)>120000) then begin Logout; Break; end;
          until(not(LoggedIn));
          NextPlayer(True);
      end;

   if (not (Loggedin)) then  NextPlayer(False);

until(False);

end.
