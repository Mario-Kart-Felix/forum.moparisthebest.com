One common way is drawing to a ‘BufferedImage’ and then drawing the ‘BufferedImage’ on your canvas.

Here is a example.

DemoBuffer.java

import java.awt.Graphics;
import java.awt.Graphics2D;
import java.awt.Image;
import java.awt.image.BufferedImage;

import javax.swing.JFrame;


/**
 * 
 * @author Fsig
 *
 */
public class DemoBuffer extends JFrame{
	private DemoCanvas canvas = new DemoCanvas();
	private static Image buffer = new BufferedImage(300,200, BufferedImage.TYPE_INT_ARGB);
	
	public static void main(String[] args){
		new DemoBuffer();
	}
	
	/**
	 * Launch a JFrame which contains our canvas.
	 */
	public DemoBuffer(){
		this.setTitle("EG: Buffer");
		this.setSize(300, 200);
		this.setLocationRelativeTo(null);
		
		add(canvas);
		
		drawToImage("Fsig is GOD...");
		
		this.setVisible(true);
	}
	
	/**
	 * Draw to the image.
	 * 
	 * @param text
	 */
	public static void drawToImage(String text){
		Graphics g = buffer.getGraphics();
		g.drawString(text, 100, 80);
		g.dispose();
	}
	
	/**
	 * Paint the image onto the canvas.
	 * 
	 * @param g
	 */
	public static void gameDraw(Graphics2D g){
		g.drawImage(buffer, 0, 0, 300, 200, null);
	}

}

DemoCanvas.java

import java.awt.Color;
import java.awt.Graphics;
import java.awt.Graphics2D;

import javax.swing.JPanel;

/**
 * 
 * @author Fsig
 *
 */
public class DemoCanvas extends JPanel{
	private static final long serialVersionUID = 1L;

	public DemoCanvas(){
		setFocusable(true);
		requestFocus();
	}
	
	@Override
	public void paint(Graphics g){
		Graphics2D g2d = (Graphics2D)g;
        super.paintComponent(g2d);
        setBackground(Color.BLACK);

        DemoBuffer.gameDraw(g2d);
	}

}
