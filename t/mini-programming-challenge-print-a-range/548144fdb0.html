public class Riddle {

    public static void main(String[] args) {
        Riddle riddle = new Riddle();
        riddle.f(13);
    }

    void f(int i) {
        if (i > 0) {
            if (i % 2 == 0) {
                f(-i);
                f(-(i - 1));
            } else if (i % 2 == 1) {
                f(-(i - 1));
                System.out.println(i);
                System.out.println(i-1);
                f(-(i-2));
            }
        } else if (i < 0) {
            if (i % 2 == 0) {
                f(i + 2);
                System.out.println(-(i + 1));
                System.out.println(-i);
            } else if (i % 2 == -1) {
                System.out.println(-i);
                System.out.println(-(i + 1));
                if (i == -3)
                    System.out.println(1);
                else
                    f(i + 2);
            }
        }
    }
}
