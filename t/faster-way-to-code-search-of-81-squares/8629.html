Mad cow you forget the ab section. I wrote one, too.

var Allc: array [1..81] of Integer;
var ab, ac: Integer;

procedure LoadColors;
begin
  Allc[1]:= 1804768;
  Allc[2]:= 1385384;
  Allc[3]:= 14803428;
  Allc[4]:= 11016981;
  Allc[5]:= 936558;
  Allc[6]:= 13705240;
  Allc[7]:= 1184277;
  Allc[8]:= 1812704;
  Allc[9]:= 14689051;
  Allc[10]:= 498170;
  Allc[11]:= 114921;
  
  //Add more here.
end;

procedure FindColors;
var i, u, x, y: Integer;
var Foundb, Foundc: Boolean;
begin
  LoadColors;
  for i:= 1 to 81 do
  begin
    if (Allc[i] <= -1) then
    begin
      u:= i;
      break;
    end;
  end;
  for i:= 1 to u do
  begin
    if (FindColor(x, y, Allc[i], 118, 44, 147, 76)) and (not(Foundb)) then
    begin
      Foundb:= True;
      Ab:= i;
    end;
    if (FindColor(x, y, Allc[i], 117, 81, 149, 112)) and (not(Foundc)) then
    begin
      Foundc:= True;
      Ac:= i;
    end;
  end;
end;

begin
  FindColors;
end.
