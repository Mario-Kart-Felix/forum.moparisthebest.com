In more details added a task system wich dosnt need any requierements for it to execute, Basic code example:

package org.astrect.rs2.actor.task.impl;

import org.astrect.rs2.actor.task.GameTask;
import org.astrect.rs2.actor.task.TaskHandler;

/**
 * Created with Eclipse. Date: 13/08/2013 Time: 3:42:58
 *
 * @author Sebastian <juan.2114@hotmail.com>
 * @see java.lang.Object 
 */
public class Sysout extends GameTask {

	private final String message;
	
	public Sysout(String message) {
		this.message = message;
	}
	
	@TaskHandler
	public void exeucte() {
		System.out.println(message);
	}
}
