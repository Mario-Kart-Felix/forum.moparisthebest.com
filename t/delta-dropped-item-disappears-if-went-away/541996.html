[code=java5]import java.util.ArrayList;
import java.util.List;

public class FloorItem {

private static List<FloorItem> floorItems = new ArrayList<FloorItem>();

public static void loadItems() {
	for(FloorItem floorItem : floorItems) {
		addItem(floorItem.getX(), floorItem.getY(), floorItem.getZ(), floorItem.getId());
	}
}

private int x;
private int y;
private int z;
private int id;

public FloorItem(int x, int y, int z, int id) {
	this.x = x;
	this.y = y;
	this.z = z;
	this.id = id;
}

public int getX() {
	return x;
}

public int getY() {
	return y;
}

public int getZ() {
	return z;
}

public int getId() {
	return id;
}

public static List<FloorItem> getFloorItems() {
	return floorItems;
}

}
