import pygame
import random

pygame.init()
font = pygame.font.Font(None, 17)
score = 0

text = font.render(“Score %d” % score, True, (0, 0, 0), (255, 255, 255))
width = 400
height = 400
screen = pygame.display.set_mode((width, height))
clock = pygame.time.Clock()
running = True
gameover = False

class Snake:
def init(self, surface, x, y):
self.surface = surface
self.x = x
self.y = y
self.speed = 1
self.dirx = 0
self.diry = -self.speed
self.length = 40
self.crashed = False
self.body = []

def keyEvent(self, event):
    if event.key == pygame.K_UP:
        self.dirx = 0
        self.diry = -self.speed
    elif event.key == pygame.K_DOWN:
        self.dirx = 0
        self.diry = self.speed
    elif event.key == pygame.K_LEFT:
        self.dirx = -self.speed
        self.diry = 0
    elif event.key == pygame.K_RIGHT:
        self.dirx = self.speed
        self.diry = 0
    
def move(self):
    self.x += self.dirx
    self.y += self.diry
    
    if (self.x, self.y) in self.body or self.x <= 0 or self.x+5 >= width or self.y <= 0 or self.y+5 >= height:
        self.crashed = True
        
    self.body.insert(0, (self.x, self.y))
    
    if len(self.body) > self.length:
        self.body.pop()
        
def draw(self):
    for x, y in self.body:
        pygame.draw.rect(self.surface, (0, 0, 100), (x, y, 5, 5), 0)

class Food:
def init(self, surface):
self.surface = surface
self.x = random.randint(1,width-5)
self.y = random.randint(1,height-5)

def draw(self):
    pygame.draw.rect(self.surface, (0, 100, 0), (self.x, self.y, 5, 5), 0)

def collision(self, x, y):
    if x < self.x-5 or x > self.x+5:
        return False
    elif y < self.y-5 or y > self.y+5:
        return False
    else:
        return True

fA = Food(screen)
fB = Food(screen)
s = Snake(screen, width/2, height/2)

while running:
screen.fill((255, 255, 255))
for event in pygame.event.get():
if event.type == pygame.QUIT:
running = False
elif event.type == pygame.KEYDOWN:
s.keyEvent(event)
screen.blit(text, (width/2, 10))
if gameover == False:
s.move()
s.draw()
fA.draw()
fB.draw()
if (fA.collision(s.x, s.y)):
fA = Food(screen)
s.length += 2
s.speed += 0.1
score += 1
text = font.render(“Score %d” % score, True, (0, 0, 0), (255, 255, 255))
if (fB.collision(s.x, s.y)):
fB = Food(screen)
s.length += 2
s.speed += 0.1
score += 1
text = font.render(“Score %d” % score, True, (0, 0, 0), (255, 255, 255))
if s.crashed:
gameover = True
pygame.display.flip()
clock.tick(100)
print “Score: %d” % score[/code]

As I said on IRC, you should reuse objects, learn proper OOP design, and stop hard-coding sizes and such.

The “reuse objects and learn proper OOP design” suggestions are actually connected. Instead of reallocating each food object (‘fA’ and ‘fB’) once collision occurs, you should simply have a function in the class that re-randomizes its location. Also, you should have the ‘Food’ class invoke a callback function upon collision detection that changes the game state (increases snake’s velocity, its length, and the game score).

The article I gave you on breaking out of nested loops was a proposal submission (a PEP); I mistakenly thought Python provided support for this (how shitty that it doesn’t – even Java supports this crap).

Other than that, it’s ok.
