     * The active script states in this executor.
     */
    static ScriptState[] states;

     * The current states stack pointer offset.
     */
    static int statesStackPointer;

    /**
     * The global integer variables array.
     */

    /**
     * The integer stack array.
     */
    static int[] intStack;

     * The current integer stack pointer offset.
     */
    static int intStackPointer;

    /**
     * Executes a script from a preloaded source.
     * @param script The script to execute.
     */
    private static void execute(Script script) {
        int[] insnOpcodes = script.insnOpcodes;
        int[] intOperands = script.intOperands;
        int scriptCounter = -1;
        int insnOpcode = -1;
        for(;;) {
            insnOpcode = insnOpcodes[++scriptCounter];
            if(insnOpcode == 0) {
                intStack[intStackPointer++] = intOperands[scriptCounter];
            } else if(insnOpcode == 3) {
                --intStackPointer;
            } else if(insnOpcode == 6) {
                intStackPointer -= 2;
                if(intStack[intStackPointer] == intStack[intStackPointer + 1])
                    scriptCounter += intOperands[scriptCounter];
            } else if(insnOpcode == 7) {
                intStackPointer -= 2;
                if(intStack[intStackPointer] != intStack[intStackPointer + 1])
                    scriptCounter += intOperands[scriptCounter];
            } else if(insnOpcode == 8) {
                intStackPointer -= 2;
                if(intStack[intStackPointer] > intStack[intStackPointer + 1])
                    scriptCounter += intOperands[scriptCounter];
            } else if(insnOpcode == 9) {
                intStackPointer -= 2;
                if(intStack[intStackPointer] < intStack[intStackPointer + 1])
                    scriptCounter += intOperands[scriptCounter];
            } else if(insnOpcode == 10) {
                intStackPointer -= 2;
                if(intStack[intStackPointer] >= intStack[intStackPointer + 1])
                    scriptCounter += intOperands[scriptCounter];
            } else if(insnOpcode == 11) {
                intStackPointer -= 2;
                if(intStack[intStackPointer] <= intStack[intStackPointer + 1])
                    scriptCounter += intOperands[scriptCounter];
            } else if(insnOpcode == 17) {
                globalIntVars[intOperands[scriptCounter]] = intStack[--intStackPointer];
            } else if(insnOpcode == 21) {
                if(statesStackPointer == 0)
                    return;
            }
        }
    }

    /**
     * The main entry point for the program.
     * @param args The command line arguments.
     */
    public static void main(String[] args) {
        globalIntVars = new int[GLOBALIVARS];
        intStack = new int[ISTACKSIZE];

          0x00, 0x15
        };

          0x00, 0x00
