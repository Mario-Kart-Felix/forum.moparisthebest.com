//This script belongs to Westside Rep And only Westside Rep
//complete the setup
//You need SRL v.3.0+

program typesomestuff;
{.include srl\srl.scar}
{.include srl\srl\extended\xtext.scar}
var t: integer;
const
//----Setup----//
type1= ‘’;//what u wanna say
type2= ‘’;//what u wanna say
type3= ‘’;//what u wanna say
type4= ‘’;//what u wanna say
type5= ‘’;//what u wanna say
howmanytimes= 5;//how manny times do u wanna say it?
waittime= 5000;//how much time in between saying things (milliseconds
//-End-Of-Setup-//

procedure autotalk;
begin
sendmytext(type1)
wait(waittime);
sendmytext(type2)
wait(waittime);
sendmytext(type3)
wait(waittime);
sendmytext(type4)
wait(waittime);
sendmytext(type5)
wait(waittime);
end;

begin
setupsrl;
activateclient;
repeat;
autotalk;
t:=t+1
until (t=howmanytimes);
writeln (‘thank you for using this script’)
end.

Oh and you were asking how to make it display in a random order. Well here is how you do it. I commented the spot where I created the procedure RandomOrder and explained the whole thing there. Oh and I also added some other things for the hell of it.

// Written by Westside Rep.
// Randomized Talk, beautified, & progress report by Ron.

Program TypeSomeStuff;
{.include srl\srl.scar}
{.include srl\srl\extended\xtext.scar}

Const
  Text1 = '';       // Place what you want to say here.
  Text2 = '';       // Place what you want to say here.
  Text3 = '';       // Place what you want to say here.
  Text4 = '';       // Place what you want to say here.
  Text5 = '';       // Place what you want to say here.
  HowManyTimes = 5; // How many times do you want to say all 5 things?
  WaitTime = 5000;  // How long you want to wait in milliseconds.

Var
  t : Integer;
  ROrder : Array[0..4] of String;

// The way to make a random order being sent is to first setup an array
// of the Const's Text and then further down below write in...
// SendMyText(ROrder[Random(4)]);
// That will use the SendMyText function and send out the the array
// ROrder then make a Random of 4 each time.
Procedure RandomOrder;
Begin
  ROrder[0] := Text1;
  ROrder[1] := Text2;
  ROrder[2] := Text3;
  ROrder[3] := Text4;
  ROrder[4] := Text5;
End;

Begin
  SetupSRL;
  ActivateClient;
  RandomOrder;
  Repeat;
    SendMyText(ROrder[Random(4)]);
    ClearDebug;
    t := t + 1;
    WriteLn('AutoTyped '+IntToStr(t)+' time(s).');
  Until (t = HowManyTimes)or(IsFKeyDown(12))
  WriteLn('Thank you for using TypeSomeStuff by Westside Rep.');
End.
