package com.rs2;

import com.rs2.players.Player;
import java.net.ServerSocket;
import java.net.Socket;

/**
 * This class handles creating the server, accepting incoming
 * connections, and creating players for the connections.
 * @author Andrew
 */
public class Server {

    /**
     * The server socket. This is used to accept connections,
     * and grabbing the socket, which will be used at later
     * dates (for reading and writing).
     */
    private ServerSocket server;

    /**
     * Creates a new Server class and starts the server.
     * @param args The command line arguments.
     */
    public static void main(String args[]) {
        try {
            /**
             * Attempts to create a new server class and start
             * the server. This may return an IOException, and
             * if it does, we want to end the application.
             */
            new Server().startServer();
        } catch (Exception ex) {
            /**
             * Failed to start the server.
             */
            ex.printStackTrace();
        }
    }

    /**
     * This attempts to start the server socket, on the default
     * port used by RuneScape (43594). This throws an exception,
     * which is caught in the 'main' method, if an exception
     * does occur. This allows us to stop the server without
     * attempting to accept any connections, etc.
     * @throws Exception Port already in use?
     */
    private void startServer() throws Exception {
        server = new ServerSocket(43594);
        System.out.println("Server Binded to Port, Online!");
        process();
    }

    /**
     * This method accepts and create players from incoming
     * connections.
     */
    private void process() {
        /**
         * The socket only has to be initialized once, since we
         * will be making a new client, with the socket as a
         * parameter, eliminating the need to create a new socket
         * each time the block of code is ran.
         */
        Socket client;
        /**
         * While true basically means, 'while the thread in which
         * the while loop is on, is alive, do...'.
         */
        while (true) {
            /**
             * The try statement is used here, rather then outside
             * of the while loop, because if an error does occur,
             * we still want to be accepting clients.
             */
            try {
                /**
                 * Attempt to set the socket (client) with the
                 * incoming connection. The 'setTpcNoDelay' sets
                 * whether Nagle's algorithm is being used.
                 */
                client = server.accept();
                client.setTcpNoDelay(true);
                /**
                 * We create a new thread, which runs a new player
                 * with the arguments of the socket of the player.
                 * A thread will use about 256kb per player (doing
                 * nothing). 100 players should use about 25mb
                 * of memory, which isn't really too bad.
                 */
                new Thread(new Player(client)).start();
                /**
                 * Lets sleep a little, limiting the amount of
                 * connections that can be accepted. This will
                 * stop a possible huge flood of people connecting
                 * at once, which may cause problems.
                 */
                Thread.sleep(100);
            } catch (Exception ex) {
                ex.printStackTrace();
            }
        }
    }
}
