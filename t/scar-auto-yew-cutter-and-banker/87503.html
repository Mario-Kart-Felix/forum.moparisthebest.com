[code]{----------------------Mike’s Yew Cutter----------------------|
| Only my second script, far from done, same for my miner. |
| I am autoing to sixty woodcut now to see if it works. |

program MrlyYewCutter;
{.include SRL/SRL.Scar}
{.include SRL/SRL/Skill/WoodCutting.Scar}
{.include SRL/SRL/Extended/xMapWalk.Scar}

const
LoadsToDo = 0; // Pick the number of loads to do.
WaitTime = 0; //Pick the time to click.
YewColor = 0; //Pick a tree color here.
YewColorTwo = 0; //Pick A second tree color here.
BankSymbolColor = 0; //Pick the bank symbol color here.
YewTreeSymbolColor = 0; //Pick the bank symbol color here.

var
Loads:Integer;

Procedure DeclarePlayers;
begin
HowManyPlayers :=4;
NumberOfPlayers(HowManyPlayers);
CurrentPlayer:=0;

 Players[0].Name :='';
 Players[0].Pass :='';
 Players[0].Nick :='';
 Players[0].Loc  :='';
 Players[0].Skill:='';
 Players[0].Active:=True;

 Players[1].Name :='';
 Players[1].Pass :='';
 Players[1].Nick :='';
 Players[1].Loc  :='';
 Players[1].Skill:='';
 Players[1].Active:=True;

 Players[2].Name :='';
 Players[2].Pass :='';
 Players[2].Nick :='';
 Players[2].Loc  :='';
 Players[2].Skill:='';
 Players[2].Active:=True;

 Players[3].Name :='';
 Players[3].Pass :='';
 Players[3].Nick :='';
 Players[3].Loc  :='';
 Players[3].Skill:='';
 Players[3].Active:=True;

end;

procedure SetUp;
begin
DeclarePlayers;
LoginPlayer;
HighestAngle;
end;

procedure FindNubRandoms;
begin
FindNormalRandoms;
FindEnt(YewColor);
FindHead;
AttachHead;
TalkToRand;
DwarfItem;
FindLamp(‘WoodCutting’);
FindMime;
FindMaze;
FindScapeRune;
SolvePinBall;
FindPlant;
SolveFrog;
end;

procedure Fight;
begin
if(FindFight)then
begin
RunAwayDirection(‘N’);
Wait(9342+Random(5423))
RunBack;
end;
end;

procedure Death;
begin
if(FindDead)then
LogOut;
NextPlayer(False);
end;

procedure CutYews;
begin
if FindMSColorTol(x,y,YewColor,2) or
FindMSColorTol(x,y,YewColorTwo,2) then
begin
MMouse(x,y,3,3)
If IsUpTextMulti(‘Chop’,‘down’,‘Yew’) then
Mouse(x,y,2,2,True)
Wait(WaitTime+Random(1312))
repeat
until(InventoryFull)
end;
end;

procedure BankYews;
begin
If FindMMColorTol(x,y,BankSymbolColor,2) then
MMouse(x,y,3,3)
Mouse(x,y,2,2,True)
Flag;
Wait(3422+Random(756))
OpenBankQuiet(‘Eb’);
if (BankScreen) then
Deposit(2,28,2)
CloseBank;
Loads:=Loads+1;
end;

procedure BackToYews;
begin
FindMMColorTol(x,y,YewTreeSymbolColor,2)
MMouse(x,y,3,3)
Mouse(x,y,2,2,True)
Flag;
Wait(2654+Random(1232))
end;

procedure ProgressReport;
begin
WriteLn(’-----------------Mrly Yew Cutter-----------------’)
WriteLn(’ ‘)
WriteLn(‘You have cut and banked’ +inttostr(Loads)+ ‘loads’)
WriteLn(’ ‘)
WriteLn(’-----------------Mrly Yew Cutter-----------------’)
end;

begin
SetUpSRL;
SetUp;
FindAxeHeadColor;
repeat
FindNubRandoms;
Fight;
Death;
CutYews;
BankYews;
BackToYews;
until(Loads = LoadsToDo)
ProgressReport;
if(Loads = LoadsToDo)then
LogOut;
NextPlayer(True);
end.[/code]
