heres one i quickly wrote in java

[code]import java.util.ArrayList;

public class Sort {
String s = “122326239903939391834071275”;
public static void main(String args[]) {
new Sort();
}

public Sort() {
	char[] c = s.toCharArray();
	ArrayList[] numbers = new ArrayList[10];
	for(int i = 0; i < numbers.length; i++)
		numbers[i] = new ArrayList();
	for(int i = 0; i < c.length; i++)
		numbers[(int)c[i] - 48].add(c[i]);
	char[] c2 = new char[c.length];
	int offs = 0;
	for(int i = 0; i < numbers.length; i++)
		for(int x = 0; x < numbers[i].size(); x++)
			c2[offs++] = (Character)numbers[i].get(x);
	System.out.println(new String(c2));
}

}[/code]
