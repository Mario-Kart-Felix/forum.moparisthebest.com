public class RandomElimination {

	private static String[] names;
	
	private static String remove(int idx) {
		String remove = names[idx];
		String[] newNames = new String[names.length - 1];
		if(idx == 0)
			System.arraycopy(names, 1, newNames, 0, newNames.length);
		else if(idx == names.length - 1)
			System.arraycopy(names, 0, newNames, 0, newNames.length);
		else {
			System.arraycopy(names, 0, newNames, 0, idx);
			System.arraycopy(names, idx + 1, newNames, idx, newNames.length - idx);			
		}
		names = newNames;
		return remove;
	}
	
	public static void main(String[] args) {
		if(args.length < 1) {
			System.err.println("Usage: RandomElimination contestant [...]");
			return;
		}
		names = args;
		int count = 1;
		System.out.printf("Starting random elimination with %d contestants!\n", names.length);
		while(names.length > 1) {
			int rand = (int) (Math.random() * names.length);
			String removedName = remove(rand);
			System.out.printf("Aww, %s lost at round %d\n", removedName, count);
			count++;
		}
		System.out.printf("%s won!", names[0]);
	}
}
