These are all the variables for the new ranking system we are going to make.

int pkPoints - The PK Points.
int killed - The current killstreak.
boolean hasCongratulated - Has the player been congratulated. Used for checking to say the “Congratulations! You have upgraded a rank!” message.
int mostKilled - The most kills in a row the player has had.
int exp - The amount of exp the player has.
String rankName - The player’s rank name (its a very messy String, but it does the job).
enum RankNames - The list of the rank names. It also includes the minimum and maximum experience range for the rank to be displayed.

[ul][li]int expRequired1 - The minimum exp required to have this rank displayed near your name.[/li]
[li]int expRequired2 - The maximum exp required to have this rank displayed near your name.[/li][/ul]

package com.rs2.model.players.content.ranking;

import com.rs2.model.players.Client;
import com.rs2.model.players.PlayerHandler;
import com.rs2.model.players.content.ranking.RankingSystemConstants.RankNames;

/**
 * The ranking system itself.
 * 
 * @author Arham 4
 * 
 */
public final class RankingSystem {

	/**
	 * Handles the death of the player (ranking-wise).
	 * 
	 * @param casualty
	 *            The person killed.
	 */
	public void handleDeath(Client casualty) {
		Client killer = (Client) PlayerHandler.players[casualty.KillerId];
		casualty.getRankingConstants().killed = 0;
		casualty.getPacketSender()
				.sendMessage(
						"Your killstreak has been reset.");
		killer.getRankingConstants().killed++;
		killer.getPacketSender().sendMessage(
				"Your killstreak has been increased! Current killstreak: "
						+ killer.getRankingConstants().killed + ".");
		if (hasBeatRecord(killer)) {
			killer.getRankingConstants().mostKilled = killer
					.getRankingConstants().killed;
			killer.getPacketSender().sendMessage(
					"Impressive! You have a new killstreak record now!");
		}
		killer.getRankingConstants().exp += 10;
		killer.getPacketSender().sendMessage(
				"Your ranking experience has risen by 10! Current ranking experience: "
						+ killer.getRankingConstants().exp + ".");
		handleBonus(killer);
		handlePromotion(killer);
	}

	/**
	 * Handles the bonus exp given.
	 * 
	 * @param killer
	 *            The killer.
	 */
	public void handleBonus(Client killer) {
		int killed = killer.getRankingConstants().killed;
		int bonus = 0;
		String message = "You have recieved a bonus of " + bonus
				+ " ranking exp for having a " + killed
				+ " killstreak! Current ranking experience: "
				+ killer.getRankingConstants().exp + ".";
		if (killed == 3)
			bonus = 20;
		else if (killed == 5)
			bonus = 30;
		else if (killed == 10)
			bonus = 40;
		else if (killed == 15)
			bonus = 50;
		else if (killed == 20)
			bonus = 60;
		else if (killed == 30)
			bonus = 70;
		else if (killed == 50)
			bonus = 100;
		killer.getRankingConstants().exp += bonus;
		if (killed >= 3)
			killer.getPacketSender().sendMessage(message);
		else
			return;
	}

	/**
	 * Handles the promotion of the player's rank.
	 * 
	 * @param killer
	 *            The killer.
	 */
	public void handlePromotion(Client killer) {
		for (RankNames names : RankNames.values()) {
			if (killer.getRankingConstants().exp >= names.expRequired1
					&& killer.getRankingConstants().exp <= names.expRequired2
					&& !killer.getRankingConstants().hasCongratulated)
				killer.getPacketSender().sendMessage(
						"Congratulations! You are now a: " + names.toString());
			else if (killer.getRankingConstants().exp >= names.expRequired1
					&& killer.getRankingConstants().exp <= names.expRequired2
					&& killer.getRankingConstants().hasCongratulated)
				return;
		}
	}

	/**
	 * Tells if the player has beat their killstreak record.
	 * 
	 * @param killer
	 *            The killer.
	 * @return The result of the check.
	 */
	public boolean hasBeatRecord(Client killer) {
		if (killer.getRankingConstants().mostKilled < killer
				.getRankingConstants().killed) {
			killer.getRankingConstants().hasCongratulated = true;
			return true;
		}
		return false;
	}

}
