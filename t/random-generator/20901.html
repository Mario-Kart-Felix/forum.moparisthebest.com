[scar]program RandomNumberGenerator;
var
randomomg, i, ii : integer;
writetest : string;

procedure writeln1;
begin
writeln(writetest);
writeln(‘Random number between 0-10 = ‘+inttostr(randomomg)+’ (’+(writetest)+’)’);
writeln(’ ');
end;

begin
repeat
cleardebug;
randomomg:= random(11);
if randomomg>5 then writetest:=‘Above 5’ else writetest:=‘Below 6’;
writeln1;
i:= i + 1;
//wait(40)
ii:= ii + randomomg;
until(i = 100)
writeln('Value of 100 random Nmbers = '+floattostr(ii));
writeln(‘Average = ‘+Floattostr((ii/(100))));
writeln(’’);
end.[/scar]
