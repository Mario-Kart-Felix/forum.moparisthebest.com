[center][size=24pt][tt]Ixtlis’s[/tt] New Quest System[/size][/center]

[size=18pt]Overview[/size] [size=6pt]of what this system does…[/size]
First, I made an abstract class named Quest.java. Then, I made a new package in it named impl. All quests are in that impl package, and all the classes in there implement Quest.java. Next, I made QuestManager.java. QuestManager.java consists of two arrays. One is a huge list of all the frames. The other is the frames that are to be modified. Also, inside is a void named findQuestComplete. Inside this, you add (a) switch statement(s). They will be based upon the quest’s progress. After that, I (Well, AtomicInt_) made a new package named questn. Inside it there was one class, Quest.java. Inside this Quest.java were ‘modifiers’ or methods that modify what the player sees (quest name, what’s in the interface of steps, what’s in the rewards interface, etc).

[size=18pt]Overview[/size] [size=6pt]of what you need to do…[/size]
Added a quest is as easy as 1-2-3 in this quest system!
[anchor=howto]1. All you need to do is go to the impl package and make a class of your quest’s (or quests’) name. Inside it, you’d implement Quest.java (the one in the original package, not the one in questn).
2. After you have done whatever you needed to do, you go to QuestManager.java and in the 2nd array (framesDisplayed), you add your quest like this:

	public static Object[][] framesDisplayed = { {
			CooksAssistant.getFrameToOverride(), CooksAssistant.getQuestName(),
			CooksAssistant.getQuestSteps(player) } };
  1. In the void findQuestComplete, you add a switch statement and in the switch statement, add the ints in a case and put the text’s state. Like this:
	public static void findQuestComplete(Player player) {
		switch (CooksAssistant.cooksAssistantProgress) {
		case 1:
			getQuest();
			Quest.sendState(player, CooksAssistant.getFrameToOverride(),
					CooksAssistant.getQuestName(),
					server.content.quests.questn.Quest.State.STARTED);
			break;
		case 2:
			getQuest();
			Quest.sendState(player, CooksAssistant.getFrameToOverride(),
					CooksAssistant.getQuestName(),
					server.content.quests.questn.Quest.State.COMPLETED);
			break;
		}
	}

You can forget about adding case 0 as staying red is by default.

[size=18pt]How to add[/size] [size=6pt]this simple quest system…[/size]

Go to your content folder (or whatever folder you see would be right) and make Quest.java. In it, add this:

[code=java]
package server.content.quests;

import server.model.Player;

/**

  • Everything in server.content.quests.impl extends this. This is all that is
  • needed to make a full fledged quest.
  • @author Arham 4/RelentlessPK

*/
public interface Quest {

/**
 * The frame ID to override. Call it like, frames[0]. Used for custom
 * quests. If it's for a custom quest, then just return 0. NOTE: To call an
 * array, you do 'the actual section - 1'. So, if it's like: public int[]
 * frames = {432, 243, 56}; And I wanna call 243, then I'd call it:
 * frames[1]
 * 
 * @return Return the frame to override.
 */
public int frameToOverride();

/**
 * The quest name.
 * 
 * @return Return the quest name.
 */
public String questName();

/**
 * The quest's steps.
 * 
 * @return Return the quest's steps.
 */
public String[] questSteps(Player player);

}[/code]

package server.content.quests.questn;

import server.model.Player;

public final class Quest {

	public static enum State {
		NOT_STARTED, STARTED, COMPLETED
	}

	public static void sendInterface(Player player, String quest,
			String... lines) {
		try {
			int line = 8147;
			for (int i = 8144; i < 8195; i++) {
				player.getActionSender().sendString(i, "");
			}
			player.getActionSender().sendString(8144, quest);
			for (int i = 0; i < lines.length; i++) {
				player.getActionSender().sendString(++line, lines[i]);
			}
			player.getActionSender().sendInterface(8134);
		} catch (Exception e) {
			e.printStackTrace();
		}
	}

	public static void sendRewardInterface(Player player, String name,
			int totalPoints, String... lines) {
		if (lines.length > 6) {
			throw new IllegalArgumentException(
					"Amount of lines exceeds maximum length.");
		}
		player.getActionSender().sendInterface(12140);
		player.getActionSender().sendString(12144,
				"You have completed: " + name + ".");
		for (int i = 0; i < lines.length; i++) {
			player.getActionSender().sendString(12150 + i, lines[i]);
		}
		player.getActionSender().sendString(12147, "" + totalPoints);
	}

	public static void sendState(Player player, int line, String name,
			State state) {
		switch (state) {
		default:
		case NOT_STARTED:
			player.getActionSender().sendString(line, name);
			break;
		case STARTED:
			player.getActionSender().sendString(line, "@yel@" + name);
			break;
		case COMPLETED:
			player.getActionSender().sendString(line, "@gre@" + name);
			break;
		}
	}

}

[size=18pt]An Example[/size]

package server.content.quests.impl;

import server.content.quests.Quest;
import server.content.quests.QuestManager;
import server.model.Player;

public class CooksAssistant implements Quest {

	public static int cooksAssistantProgress;

	public static int getFrameToOverride() {
		return cooksAssistant.frameToOverride();
	}

	@Override
	public int frameToOverride() {
		return QuestManager.frames[1];
	}

	public static String getQuestName() {
		return getCooksAssitant().questName();
	}

	@Override
	public String questName() {
		return "Cook's Assistant";
	}

	private static CooksAssistant cooksAssistant = new CooksAssistant();

	public static CooksAssistant getCooksAssitant() {
		return cooksAssistant;
	}

	public static String[] getQuestSteps(Player player) {
		return getCooksAssitant().questSteps(player);
	}

	@Override
	public String[] questSteps(Player player) {
		switch (cooksAssistantProgress) {
		case 0:
			String[] i = { "I can start this quest ny talking to the cook.",
					"He is located in the lumbridge castle." };
			return i;
		case 1:
			String[] i1 = {
					"@str@I can start this quest by talking to the cook.",
					"@str@He is located in the lumbridge castle.",
					"The cook has asked you to get the following items:",
					"@blu@A Bucket of Milk", "@blu@A Pot of Flour",
					"@blu@1 egg" };
			return i1;
		case 2:
			String[] i2 = {
					"@str@I can start this quest by talking to the cook.",
					"@str@He is located in the lumbridge castle.",
					"@str@The cook has asked you to get the following items:",
					"@blu@@str@A Bucket of Milk", "@blu@@str@A Pot of Flour",
					"@blu@@str@1 egg", "@red@QUEST COMPLETED!" };
			return i2;
		}
		for (int i = 0; i < questSteps(player).length; i++)
			server.content.quests.questn.Quest.sendInterface(player,
					"Cook's Assistant", questSteps(player)[i]);
		return null;
	}

}
	public static Object[][] framesDisplayed = { {
			CooksAssistant.getFrameToOverride(), CooksAssistant.getQuestName(),
			CooksAssistant.getQuestSteps(player) } };

public static void findQuestComplete(Player player) { switch (CooksAssistant.cooksAssistantProgress) { case 1: getQuest(); Quest.sendState(player, CooksAssistant.getFrameToOverride(), CooksAssistant.getQuestName(), server.content.quests.questn.Quest.State.STARTED); break; case 2: getQuest(); Quest.sendState(player, CooksAssistant.getFrameToOverride(), CooksAssistant.getQuestName(), server.content.quests.questn.Quest.State.COMPLETED); break; } }
