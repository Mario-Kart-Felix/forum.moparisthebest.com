if (UseBow || UseCrossBow || UseRing) { TotalExp = (double)(800*hitDiff); TotalExp = (double)(TotalExp*CombatExpRate); addSkillXP((int)(TotalExp), 4); } else if (FightType != 3) { TotalExp = (double)(800*hitDiff); TotalExp = (double)(TotalExp*CombatExpRate); addSkillXP((int)(TotalExp), SkillID); } else { TotalExp = (double)(800*hitDiff); TotalExp = (double)(TotalExp*CombatExpRate);

if(playerEquipment[playerWeapon] == 4151 && specialAmount > 49){ AttackingOn2.specGFX(341); specialAmount -= 49; hitDiff = misc.random(playerMaxHit) + misc.random(StrPrayer) + misc.random(voidmelee); lastSpecial = System.currentTimeMillis(); setAnimation(1658); actionInterval = getbattleTimer(); lastAction = System.currentTimeMillis(); specOn = false; }

The special GFX, when using items with a spec

Change those numbers to your desired exp rate.
