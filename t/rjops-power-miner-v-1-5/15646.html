Procedure ChangeReport(Reason : String);
var
   wMinutes,wSeconds,RHours,RMinutes,RSeconds,avtime:LongInt;
  Time:string;
begin
  wSeconds:=(GetSystemTime-ST) div 1000;
  wMinutes:=wSeconds div 60;
  RHours:=wMinutes div 60;
  Time:=inttostr(wSeconds)+' Seconds';
  if(DroppedOres>0)then
    avtime:=wSeconds/DroppedOres;
  if wMinutes<>0 then
  begin
    RSeconds:=wSeconds Mod (wMinutes*60);
    Time:=inttostr(wMinutes)+' Minutes and '+inttostr(RSeconds)+' Seconds';
  end;
  if RHours<>0 then
  begin
    RMinutes:=wMinutes Mod (RHours*60);
    RSeconds:=wSeconds Mod (wMinutes*60);
    Time:=inttostr(RHours)+' Hours, '+inttostr(RMinutes)+
    ' Minutes and '+inttostr(RSeconds)+' Seconds';
  end;
  
  Status(Reason);
  ClearReport;
  AddToReport('Rjop455''s Auto Miner');
  AddToReport('*********************');
  AddToReport('Currently: ' + Reason);
  AddToReport('Ores Mined: ' + inttostr(OresMined) );
  AddToReport('XP gained ' + inttostr(XP));
  AddToReport('Worked for: ' + Time);
End;

procedure Mining;
var r, count, SysTime, ExtraXP :Integer;
GotOre :boolean;
begin
  SetRandomOreColor;
  ChangeReport('Mining...')
  GotOre:=false;
  count:=InvCount;
  if(FindObj(x,y,'Mine',OreColor,5))then
  begin
    if(not(FindGas(x,y)))then
    begin
      Mouse(x,y,r,r,true);
      if(FlagPresent)then
      begin
        Flag;
        Wait(500+random(500));
      end;
      MarkTime(SysTime);
      repeat
      if(not(FindColorSpiral(x,y,OreColor,210,130,300,220)))then
        break;
        Wait(100+random(200));
      if(not(GetColor(109,430)=0)and(not(GetColor(195,426)=0)))then
      begin
        If(InvCount >= count +1)Then
        begin
          OresMined:=OresMined+1
          XP:=XP + XPRock;
          GotOre:=true
        if(XPRock = 17)Then
          begin
            ExtraXP:=ExtraXP+1;
          if(ExtraXP = 2)then
          begin
           XP:=XP + 1;
           ExtraXP:= 0;
          end;

          end;

        end;
      end;
      if(TimeFromMark(SysTime)>=MaxWait*1000 = true)then
     begin
     GotOre:=true
     end;
      until(GotOre=true)
    end;
  end;
end;
