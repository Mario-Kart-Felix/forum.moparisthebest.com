[code]public final class IrrationalShift extends TreeAdapter {
    @SuppressWarnings("unchecked")
    public <T extends Node> T transform(T obj) {
        return (T) obj.accept(this, VisitOrder.PREORDER);
    }

    private static Expression rationaliseArg(BinaryShiftExpression shift) {
        Expression ex = shift.right();
        if (ex.sort() == INT_EXPR) {
            int value = (Integer) ex.value();
            int mask = (shift.type() == Type.LONG_TYPE) ? 63 : 31;
            return new IntegerExpression(value & mask);
        }
        return null;
    }

    public Expression visitShiftLeftExpression(ShiftLeftExpression expr) {
        Expression ex = rationaliseArg(expr);
        return ex == null ? null : new ShiftLeftExpression(expr.left(), ex);
    }

    public Expression visitShiftRightExpression(ShiftRightExpression expr) {
        Expression ex = rationaliseArg(expr);
        return ex == null ? null : new ShiftRightExpression(expr.left(), ex);
    }

    public Expression visitShiftUnsignedRightExpression(ShiftUnsignedRightExpression expr) {
        Expression ex = rationaliseArg(expr);
        return ex == null ? null : new ShiftUnsignedRightExpression(expr.left(), ex);
    }
}
