import java.util.*;

public class Test {
	private class Pair<V,K> {
		public V first;
		public K second;

		public Pair(V first, K second) {
			this.first = first;
			this.second = second;
		}
	}

	private ArrayList<Pair<Integer,String>> multiples;

	public Test() {
		multiples = new ArrayList<Pair<Integer,String>>();
	}

	public boolean addMultiple(int i, String word) {
		return multiples.add(new Pair<Integer,String>(i,word));
	}

	public void printTest(int n) {
		for(int i = 1; i < n; i++) {
			System.out.println(getText(i));
		}
	}

	private String getText(int i) {
		String number = "";
		for(Pair<Integer,String> multiple : multiples) {
			if((i % multiple.first) == 0) {
				number+=multiple.second;
			}
		}
		if(number.equals("")) {
			number = Integer.toString(i);
		}
		return number;
	}

	public static void main(String[] args) {
		Test t = new Test();
		t.addMultiple(3,"Fizz");
		t.addMultiple(5,"Buzz");
		t.addMultiple(7,"Bazz");
		t.printTest(100);
	}
}
