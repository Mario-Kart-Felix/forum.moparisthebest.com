public void loginMessage() { for (final Player player : PlayerHandler.players) { if (player != null) { final Client c2 = (Client) player; if (playerName.equalsIgnoreCase("Phil") || playerName.equalsIgnoreCase("Phil")) { c2.sendMessage("<shad=7675309>[Main Owner] " + Misc.optimizeText(playerName) + " has logged in!</col>"); } else if (playerRights == 3) { c2.sendMessage("<shad=6081134>[Co-Owner] " + Misc.optimizeText(playerName) + " has logged in! Feel free to ask him for help!</col>"); } if (playerRights == 2) { c2.sendMessage("<shad=1166775>[Administrator] " + Misc.optimizeText(playerName) + " has logged in! Feel free to ask him for help!</col>"); } if (playerRights == 1) { c2.sendMessage("<shad=265709>[Moderator] " + Misc.optimizeText(playerName) + " has logged in! Feel free to ask him for help!</col>"); } if (playerName.equalsIgnoreCase("HEAD MOD") || playerName.equalsIgnoreCase("HEAD MOD")) { c2.sendMessage("<shad=7675309>[Head Moderator] " + Misc.optimizeText(playerName) + " has logged in!</col>"); } if (playerRights == 4 && isDonator == 1) { c2.sendMessage("<shad=15695415>[Donator] " + Misc.optimizeText(playerName) + " has logged in!</col>"); } } } }
