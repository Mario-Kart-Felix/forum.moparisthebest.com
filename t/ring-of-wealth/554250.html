    {1926, 525, 6, 60},
    {1326, 525, 4, 60},

static ArrayList<int[]> findNpcDrops(int npcId) {
    ArrayList<int[]> dropList = new ArrayList<>();
    
    for (int i = 0; i < RARE_DROPS.length; i++) {
        if (npcId == RARE_DROPS[i][0])
        dropList.add(RARE_DROPS[i]);
    }
    
    return dropList;
}
