public static boolean isValidNumber(long value) {
	List<Integer> addUps = new ArrayList<Integer>();
	char[] characters = String.valueOf(value).toCharArray();
	for (int i = 1; i < characters.length + 1; i++) {
		char c = characters[i - 1];
		if (i % 2 == 0) {
			int convert = Integer.parseInt("" + c) * 2;
			int v = 0;
			for (char c2 : String.valueOf(convert).toCharArray())
				v += Integer.parseInt("" + c2);
			addUps.add(v);
		} else addUps.add(Integer.parseInt("" + c));
	}
	int complete = 0;
	for (int i : addUps) complete += i;
	return complete % 10 == 0;
}

public static int[][] generateChromosomes(int length) {
	int[][] values = new int[length][25];
	for (int i = 0; i < length; i++) {
		for (int j = 0; j < 25; j++) {
			values[i][j] += new Random().nextBoolean() ? 0 : 1;
		}
	}
	return values;
}

public static List<Object[]> conceiveChild(int length) {
	int[][] parents1 = generateChromosomes(length);
	int[][] parents2 = generateChromosomes(length);
	int[][][][] children = new int[2][2][length][25];
	
	for (int i = 0; i < length; i++) {
		for (int j = 0; j < 25; j++) {
			if (j != 0 && j != 24) {
				if (parents1[i][j] != parents2[i][j]) { //if the genes on the index are not the same
					int old = parents1[i][j];
					children[0][0][i][j] = parents2[i][j]; //swaps the genes
					children[1][0][i][j] = old; //swaps the genes
				}
			}
			if (new Random().nextBoolean()) {
				children[0][1][i][j] = new Random().nextBoolean() ? 0 : 1;
				children[1][1][i][j] = new Random().nextBoolean() ? 0 : 1;
			}
		}
	}
	List<Object[]> conceived = new ArrayList<Object[]>();
	for (int i = 0; i < length; i++) {
		int mutations = 0;
		String childBinary = "";
		for (int j = 0; j < 25; j++) {
			int originalGene = children[0][0][i][j];
			int mutatedGene = children[0][1][i][j];
			if (originalGene != mutatedGene) {
				mutations++;
			}
			childBinary += children[0][0][i][j];
		}
		double chance = (double) (mutations) / 25;
		conceived.add(new Object[] {childBinary, chance});
	}
	return conceived;
}
