This is my yell, you’ll probably have to adjust yours because I’ve added restrictions such as muted players and if I have the yell system disabled.

[code=java]
if(command.startsWith(“yell”) && command.length() > 5 ) {
if (!server.yellSystem && playerRights == 0) {
sM(“The yell system has been turned off.”);
return;
}
if(!muted){
String titles = “”;

				if (playerName.equalsIgnoreCase("mista pnut")){
			titles = "[@blu@The Boss@bla@] -";
			}
			else if (playerRights == 1) {
                titles = "[@blu@Moderator@bla@] -";
            }
            else if (playerRights == 2) {
                titles = "[@blu@Admin@bla@] -";
            }
            else if (playerRights == 3) {
                titles = "[@blu@Owner@bla@] -";
            }
            else if (donator == 1 && playerRights == 0) {
                titles = "[@red@Donator@bla@] -";
            }
            else if (donator == 2) {
                titles = "[@red@Super Donator@bla@]-";
            }
			else if (playerRights == 0) {
				titles = "[@blu@Player@bla@] -";
			}
            yell(titles + "" + playerName + ": "
                    + command.substring(5));
		}

}[/code]
