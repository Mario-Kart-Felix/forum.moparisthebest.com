import java.net.*;
import java.io.*;

public class client {
	
	public String InternetProtocol = "messenger.hotmail.com";
	public int PortNumber = 1863;

	public String emailaddr = "notgoingtoexist@hotmail.com";
	public String password = "yesitis";

	public Socket MessengerSocket = null;

	public BufferedReader receive = null;
	public PrintWriter send = null;

	public boolean debug = false;

	public static void main(String args[]) {
		new client();
	}

	public client() {
		establishConnection();
	}

	public void establishConnection() {
		if(connect()) {
			try {
				send("VER 0 MSNP9 MSNP8 MSNP7 CVR0");
				while(MessengerSocket != null) {
					String rec = receive.readLine();
					parse(rec);	
				}
			} catch(IOException e) {
				System.out.println(e);
			}
		} else {
			System.out.println("god hates you =]");
		}
	}

	public boolean connect() {
		try {
			MessengerSocket = new Socket(InternetProtocol, PortNumber);
			send = new PrintWriter(MessengerSocket.getOutputStream(), true);
			receive = new BufferedReader(new InputStreamReader(MessengerSocket.getInputStream()));
			debug("connected to " + InternetProtocol + ":" + PortNumber + "\n");
			return true;
		} catch(IOException e) {
			System.out.println(e);
			return false;
		}
	}

	public boolean close() {
		try {
			send.close();
			receive.close();
			MessengerSocket.close();
			MessengerSocket = null;
			return true;
		} catch(IOException e) {
			System.out.println(e);
			return false;
		}
	}

	public void parse(String data) {
		debug("<<< " + data);
		if(data.startsWith("VER")) {
			send("CVR 1 0x0409 winnt 5.1 i386 MSNMSGR 6.0.0602 MSMSGS " + emailaddr);
		} else if(data.startsWith("CVR")) {
			send("USR 2 TWN I " + emailaddr);
		} else if(data.startsWith("XFR")) {
			close();
			setAddr(getNotifServer(data));
			establishConnection();
		} else if(data.startsWith("USR 2")) {
			String challenge = data.split(" ")[4];
			send("USR 3 TWN S " + tweenerAuth(challenge));
		} else if(data.startsWith("USR 3")) {
			send("SYN 4 6");
			//send("CHG 5 NLN 0");
		} else if(data.startsWith("LST")) {
			System.out.println(data.split(" ")[1]);
		} else if(data.startsWith("CHL")) {
			//sorry no challenges, I said it was simple :/
		}
	}
	
	public void send(String msg) {
		send.println(msg);
		debug(">>> " + msg);
	}

	public String getNotifServer(String data) {
		String full = data.split(" ")[3];

		int end = full.indexOf(':', 0);

		return full.substring(0, end);
	}

	public String getLoginServer() {
		try {
			URL nexus = new URL("https://nexus.passport.com/rdr/pprdr.asp");
		
			URLConnection nexusC = nexus.openConnection();		
			String header = nexusC.getHeaderField("PassportURLs");

			int start = header.indexOf("DALogin=") + 8;
			int end = header.indexOf(',', start);
			
			return "https://" + header.substring(start, end);
		} catch(Exception ex) {
			System.out.println(ex);
			return null;
		}
	}

	public String tweenerAuth(String challenge) {
		try {
			URL nexus = new URL(getLoginServer());

			URLConnection nexusC = nexus.openConnection();

			nexusC.setRequestProperty("Authorization", 
				"Passport1.4 OrgVerb=GET,OrgURL=http%3A%2F%2Fmessenger%2Emsn%2E com, sign-in=" + 
				emailaddr.replace("@", "%40") + ",pwd=" + URLEncoder.encode(password, "UTF-8") + "," + challenge);
			nexusC.setRequestProperty("User-Agent", "MSMSGS");

			String ret = nexusC.getHeaderField("Authentication-Info");

			int start = ret.indexOf("from-PP='") + 9;
			int end = ret.indexOf("',", start);

			return ret.substring(start, end);
		} catch(IOException e) {
			System.out.println(e);
			return null;
		}
	}

	public void setAddr(String addr) {
		InternetProtocol = addr;
	}
	
	public void setPort(int port) {
		PortNumber = port;
	}

	public void debug(String s) {
		if(debug)
			System.out.println(s);
	}
}
