procedure CursorToMouse;
Var
   W, H, x1, y1 : integer;
begin

   DisplayDebugImgWindow(500, 400);
   Buffer:= BitmapFromString(500, 400, '');

        SetTransparentColor(Icon ,  clblack);
        If(FindWindowTitlePart('Debug', False))Then
          Begin
                GetClientDimensions(W, H);
                writeln('Successfully complied')
                end;

              repeat

              getmousepos(x,y)
              X1:= X - 10;
              y1:= y - 10;

              FastDrawSizeTransparent(x1, y1, x1+22, y1+22, Icon, Buffer)

         SafeDrawBitmap(Buffer, GetDebugCanvas, 0, 0);
         FastDrawClear(Buffer, clblack);
         until(false)
end;
