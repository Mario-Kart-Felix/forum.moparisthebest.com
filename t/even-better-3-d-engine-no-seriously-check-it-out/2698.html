ONE LAST THING THAT MAKES MY SCAR 3D ENGINE THE BEST!!!!

Its possible to make your own "world". Under the Define procedure, you will see a whole bunch of arrays and numbers.
The arrays are structured as follows, World --> Object --> Face --> Coordanate. This means for every face, there are
a number of coordanates, for every object there are a number of faces, and the world array is made up of a number of
objects. At the moment, there is only a cube which I have created, but feel free to make your own. Also, you'll have
to set the length of each array using the setarraylength(varable array, lenght) procedure.

function fact(i : integer) : integer;
var
   holdi,res : integer;
begin
     holdi := i;
     res := 1;
     for i := 1 to holdi do
     begin
          res := i * res
     end;
     result := res;
end;

function arctan(e : extended) : extended;
var
   count : integer;
   res : extended;
   inv : boolean;
begin
     if(abs(e) > 1)then
     begin
          e := pow(e,-1);
          inv := true;
     end else
     begin
          inv := false;
     end;
     for count := 0 to 6 do
     begin
          res := res + ( ( pow(2,2 * count) * pow(fact(count),2) ) / fact(2 * count + 1) ) *  ( ( pow(e,2*count+1) ) / ( pow(1 + pow(e,2),count + 1) ) );
     end;
     if(inv)then
     begin
          res := 1.5707963267 - res;
     end;
     result := res;
end;

procedure equate;
var
   count : array [0..3] of integer;
begin
     setarraylength(world2, getarraylength(world));
     for count[0] := 0 to getarraylength(world2) - 1 do
     begin
          setarraylength(world2[count[0]], getarraylength(world[count[0]]));
          for count[1] := 0 to getarraylength(world2[count[0]]) - 1 do
          begin
               setarraylength(world2[count[0]][count[1]], getarraylength(world[count[0]][count[1]]));
          end;
     end;
end;

procedure translate;
var
   count : array [0..3] of integer;
begin
     for count[0] := 0 to getarraylength(world2) - 1 do
     begin
          for count[1] := 0 to getarraylength(world2[count[0]]) - 1 do
          begin
               for count[2] := 0 to getarraylength(world2[count[0]][count[1]]) - 1 do
               begin
                    for count[3] := 0 to 2 do
                    begin
                         world2[count[0]][count[1]][count[2]][count[3]] := world[count[0]][count[1]][count[2]][count[3]] + trans[count[3]];
                    end;
               end;
          end;
     end;
end;

procedure rot;
var
   count : array [0..2] of integer;
   rotate2 : array [0..1] of extended;
   dist,ang,ele,flatdist : extended;
begin
     rotate2[0] := rotate[0] / (180/pi);
     rotate2[1] := rotate[1] / (180/pi);
     for count[0] := 0 to getarraylength(world2) - 1 do
     begin
          for count[1] := 0 to getarraylength(world2[count[0]]) - 1 do
          begin
               for count[2] := 0 to getarraylength(world2[count[0]][count[1]]) - 1 do
               begin
                    x := world2[count[0]][count[1]][count[2]][0];
                    y := world2[count[0]][count[1]][count[2]][1];
                    z := world2[count[0]][count[1]][count[2]][2];
                    dist := sqrt(pow(x,2) + pow(y,2) + pow(z,2));
                    if(x = 0) then
                    begin
                         x := 1;
                    end;
                    if(y = 0) then
                    begin
                         y := 1;
                    end;
                    if(z = 0) then
                    begin
                         z := 1;
                    end;
                    ang := arctan(x/z) + pi;
                    flatdist := abs(x / sin(ang));
                    ele := arctan(y/flatdist) + pi;
                    ang := ang + rotate2[0];
                    ele := ele + rotate2[1];
                    world2[count[0]][count[1]][count[2]][1] := round( sin(ele) * abs(dist) );
                    flatdist := abs(cos(ele) * dist);
                    world2[count[0]][count[1]][count[2]][2] := round( cos(ang) * flatdist );
                    world2[count[0]][count[1]][count[2]][0] := round( sin(ang) * flatdist );
               end;
          end;
     end;
end;

procedure draw;
begin
     precanvas.Rectangle(-1, -1, width+1, height+1);
     for i[0] := 0 to getarraylength(world2) - 1 do
     begin
         for i[1] := 0 to getarraylength(world2[i[0]]) - 1 do
         begin
              x := world2[i[0]][i[1]][0][0];
              y := -1 * world2[i[0]][i[1]][0][1];
              z := world2[i[0]][i[1]][0][2];
              if(z = 0)then
              begin
                   z := 1;
              end;
              x1 := round( x / z * zoom + width / 2);
              y1 := round( y / z * zoom + height / 2);
              precanvas.moveto(x1,y1);
              for i[2] := 0 to getarraylength(world2[i[0]][i[1]]) - 1 do
              begin
                   x := world2[i[0]][i[1]][i[2]][0];
                   y := -1 * world2[i[0]][i[1]][i[2]][1];
                   z := world2[i[0]][i[1]][i[2]][2];
                   if(z = 0)then
                   begin
                        z := 1;
                   end;
                   x1 := round( x / z * zoom + width / 2);
                   y1 := round( y / z * zoom + height / 2);                   precanvas.lineto(x1,y1);
              end;
              x := world2[i[0]][i[1]][0][0];
              y := -1 * world2[i[0]][i[1]][0][1];
              z := world2[i[0]][i[1]][0][2];
              if(z = 0)then
              begin
                   z := 1;
              end;
              x1 := round( x / z * zoom + width / 2);
              y1 := round( y / z * zoom + height / 2);
              precanvas.lineto(x1,y1);
         end;
     end;
     copycanvas(precanvas,canvas,0,0,width,height,0,0,width,height);
end;

procedure listen;
var
   mx,my : integer;
begin
     getmousepos(mx,my);
     if(mx>100 + sensitivity)then
     begin
          rotate[0] := rotate[0] - looksensetivity;
     end;
     if(mx<100 - sensitivity)then
     begin
          rotate[0] := rotate[0] + looksensetivity;
     end;
     getmousepos(mx,my);
     if(my>100 + sensitivity)then
     begin
          rotate[1] := rotate[1] + looksensetivity;
     end;
     if(my<100 - sensitivity)then
     begin
          rotate[1] := rotate[1] - looksensetivity;
     end;
     movemouse(100,100);
     if(iskeydown(chr(32)))then
     begin
          trans[1] := trans[1] - 10;
     end;
     if(isfunctionkeydown(1))then
     begin
          trans[1] := trans[1] + 10;
     end;
     if(iskeydown('a'))then
     begin
          trans[0] := trans[0] + 10;
     end;
     if(iskeydown('d'))then
     begin
          trans[0] := trans[0] - 10;
     end;
     if(iskeydown('w'))then
     begin
          trans[2] := trans[2] - 10;
     end;
     if(iskeydown('s'))then
     begin
          trans[2] := trans[2] + 10;
     end;
     if(iskeydown('q'))then
     begin
          terminatescript;
     end;
end;

begin
     define;
     setup;
     repeat
           translate;
           rot;
           draw;
           listen;
           wait(10);
     until(false);
end.

Your engine is better. And you are right, arctangent is not present, and it would be bloody useful, but we don’t have it. Anyways, there’s a misconception as far as 3d graphics go: We are merely providing a lustrous illusion using an Euclidian plane.

I also realized scar is a nastily written scripting engine. Oh, and it’s written in delphi, which handles memory about as well as aryan updates. Anyways, the problem you are having is dealing with your buffering.

Thing is, you’re not. -sigh-
canvas := GetDebugCanvas;
precanvas := canvas;
So you made a pointer canvas, which points to the debug canvas. Your buffer canvas, precanvas, isn’t even a buffer.

All you have done is made a pointer to an existing object (Debug), then duplicated the pointer, or precanvas.

Try this: “precanvas:= GetBitmapCanvas(BitmapFromString(width,height,’’));”

Yep, that creates an actual buffer.

You’re also going to either need to clear, or recreate your buffer each time you generate a frame. Otherwise you’ll paint on top of your previous frame, and it will become more like an acid trip for about .3 seconds when it becomes a useless blob of pixels.
