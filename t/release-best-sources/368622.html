public final class Jailcheck extends Thread {

	public void CheckPrisoners() {

			for(Player p : world.getPlayers()) {
				if(p.getLocation().inJail() && p.getJailTimeleft() != 0) {
				if(p.getLocation().inJail() && p.getJailTimeleft() > 0) {
