import java.util.Random;
import java.util.Scanner;

public class UncreativeProgram{
	public static void main(String[] args){
	
		System.out.println("Welcome to UncreativeProgram.");

		Scanner scanner = new Scanner(System.in);
		Random random = new Random();
	
		while(true){			
			int ran = random.nextInt(3);

			System.out.println("Please enter a question, or type 'exit' to exit.");
	
			String s = scanner.nextLine();
			
			if(s.equalsIgnoreCase("exit")){
				System.out.println("UncreativeProgram will now exit.");
				break;
			}
			else{
	
				switch(ran){
					case 0: System.out.println("@");
						break;
					case 1: System.out.println("$");
						break;
					case 2: System.out.println("0");
						break;
					default: System.out.println("OH NOES AN ERROR!");
						break;
				}
			}
		}
	}
}
