    public abstract class Dialogue {

    /**

    • The text
      */
      private String[] text;

    /**

    • The player in dialogue
      */
      private Player player;

    /**

    • If the dialogue should continue
      */
      private boolean continuing;

    • @param text to the text
    • @param continuing to continuing

      public Dialogue(Player player, String[] text, boolean continuing) {
      this.text = text;
      this.player = player;
      this.continuing = continuing;
      }

    /**

    • Executes the dialogue
      */
      public abstract void execute();

    /**

    • The formula of the dialogue
      */

    • @return text

      return text;

    • @return player

    package org.hyperion.rs2.model.dialogue;

import java.util.HashMap;
import java.util.Map;

import org.hyperion.rs2.model.Player;

public abstract class DialogueHandler {

/**
 * Map of all dialogues
 */
private static Map<Integer, DialogueHandler> dialogues = new HashMap<Integer, DialogueHandler>();

/**
 * Gets the dialogue
 * @param player The player who is going through dialogue
 */
public abstract void getDialogue(Player player);

/**
 * Starts up the dialogue
 * @param player The player who is starting dialogue
 */
public void start(Player player) {
	
	/** Removes all the interfaces */
	player.getActionSender().sendRemoveAllInterfaces();
	
	/** Gets the dialogue */
	getDialogue(player);
	
	/** Executes the dialogue */
	player.getDialogue().peek().execute();
}

/**
 * @param player The player who clicked option
 * @param button The button clicked
 */
public void handleOptions(Player player, int button) {
	
	/** Removes all interfaces*/
	player.getActionSender().sendRemoveAllInterfaces();
	
	/** NOTE: Since all dialogues do not have options I decided not to force
	 * a dialogue option method in each one to handle options though you just
	 * override this method manually in the class and at the end call <code>handleOptions(player, button)</code> 
	 * PLEASE DO NOT CRAM The clicking buttons class with all different options this is better */
}

/**
 * Alternates the dialogue
 * @param player The player who is alternating the dialogue
 */
public void alternate(Player player) {
	
	/** The past dialogue */
	Dialogue pastDialogue = player.getDialogue().poll();
	
	/** The future dialogue */
	Dialogue futureDialogue = player.getDialogue().peek();
	
	/** Removes all current interfaces */
	player.getActionSender().sendRemoveAllInterfaces();
	
	/** If the dialogue should still continue */
	if(pastDialogue.isContinuing()) {
		futureDialogue.execute();
		return;
	}
	
	/** If not then clear the dialogue list */
	player.getDialogue().clear();
	
}

/**
 * @return dialogues
 */
public static Map<Integer, DialogueHandler> getDialogues() {
	return dialogues;
}

/**

  • An information dialogue

    /**

    • @param player The player going through dialogue

    • @param text The text of the dialogue
      */
      public InfoDialogue(Player player, boolean continuing, String… text) {
      super(player, text, continuing);
      }

    /**

    • (non-Javadoc)
    • @see org.hyperion.rs2.model.dialogue.Dialogue#execute()
      */
      @Override
      public void execute() {
      getPlayer().getActionSender().sendChatInterface(getFormula(getText().length));
      getPlayer().getActionSender().sendString(getFormula(getText().length) + 2, “”);
      for (int i = 0; i < getText().length; i++) {
      getPlayer().getActionSender().sendString(getFormula(getText().length) + 3 + i, getText()[i]);
      }
      }

    • @see org.hyperion.rs2.model.dialogue.Dialogue#getFormula(int)
      */

      public int getFormula(int i) {
      return 968 + (i - 1) + (i * i) + i / 2 - 1 - i / 4 - i / 4;

  • The Mob Dialogue Action
    */
    public class MobDialogue extends EntityDialogue {

    /**

    • The mob id
      */
      private int mobId;

    /**

    • @param player The player going through dialogue
    • @param mobId the id of the mob talking to
    • @param emotion The emotion mob will use
    • @param continuing If the dialogue should continue
    • @param text The text of the dialogue

    getPlayer().getActionSender().sendDialogueAnimation(getEmotion().getEmoteId(), getFormula(getText().length) + 1);
    getPlayer().getActionSender().sendNPCDialogueHead(mobId, getFormula(getText().length) + 1);
    for (int i = 0; i < getText().length; i++) {
    getPlayer().getActionSender().sendString(getFormula(getText().length) + 3 + i, getText()[i]);

     }

    package org.hyperion.rs2.model.dialogue.xml;

import org.hyperion.rs2.model.dialogue.DialogueHandler;
import org.hyperion.rs2.util.xml.XMLHandler;

import com.thoughtworks.xstream.XStream;
import com.thoughtworks.xstream.annotations.XStreamImplicit;

public class XMLDialogue {

/**
 * The type of dialogue
 */
public enum DialogueType {
	
	/**
	 * Player dialogue type
	 */
	PLAYER,
	
	/**
	 * Option dialogue type
	 */
	OPTION,
	
	/**
	 * The info dialogue type
	 */
	INFO,
	
	/**
	 * The mob dialogue type
	 */
	MOB;
}


/**
 * The id of the mob
 */
private int id;

/**
 * The type of dialogue
 */
private String[] types;

/**
 * The emotions of the player
 */
private String[] emotions;

/**
 * If the dialogue is continuing
 */
private boolean[] continuing;

/**
 * The text of the dialogue
 */
@XStreamImplicit
private String[][] text;

/**
 * @return id
 */
public int getId() {
	return id;
}

/**
 * @return types
 */
public String[] getTypes() {
	return types;
}

/**
 * @return emotions
 */
public String[] getEmotions() {
	return emotions;
}

/**
 * @return continuing
 */
public boolean[] isContinuing() {
	return continuing;
}

/**
 * @return text
 */
public String[][] getText() {
	return text;
}


/**
 * The Dialogue loader
 */
public static class DialogueLoader extends XMLHandler {

	/**
	 * (non-Javadoc)
	 * @see org.hyperion.rs2.util.xml.XMLHandler#getKey()
	 */
	@Override
	public String getKey() {
		return "dialogue";
	}

	/**
	 * (non-Javadoc)
	 * @see org.hyperion.rs2.util.xml.XMLHandler#getClassName()
	 */
	@Override
	public Class<?> getClassName() {
		return XMLDialogue.class;
	}

	/**
	 * (non-Javadoc)
	 * @see org.hyperion.rs2.util.xml.XMLHandler#getPath()
	 */
	@Override
	public String getPath() {
		return "dialogues.xml";
	}

	/**
	 * (non-Javadoc)
	 * @see org.hyperion.rs2.util.xml.XMLHandler#load(com.thoughtworks.xstream.XStream)
	 */
	@Override
	public void load(XStream xstream) {
		xstream.addImplicitCollection(XMLDialogue.class, "text");		
	}

	/**
	 * (non-Javadoc)
	 * @see org.hyperion.rs2.util.xml.XMLHandler#bind(java.lang.Object, int)
	 */
	@Override
	public void bind(Object object, int i) {
		XMLDialogue dialogue = (XMLDialogue) object;
		
		/** The handler */
		XMLDialogueHandler handler = new XMLDialogueHandler(dialogue);
		
		/** Adds dialogue to the map */
		DialogueHandler.getDialogues().put(dialogue.getId(), handler);
		
	}
	
}

    package org.hyperion.rs2.model.dialogue.xml;

import org.hyperion.rs2.model.Player;
import org.hyperion.rs2.model.dialogue.DialogueEmotion;
import org.hyperion.rs2.model.dialogue.DialogueHandler;
import org.hyperion.rs2.model.dialogue.type.*;

/**

  • Handles all xml dialogues
    */
    public class XMLDialogueHandler extends DialogueHandler {

    /**

    • The dialogue
      */
      private XMLDialogue dialogue;

    /**

    • @param dialogue to the dialogue
      */
      public XMLDialogueHandler(XMLDialogue dialogue) {
      this.dialogue = dialogue;
      }

    /**

    • (non-Javadoc)

    • @see org.hyperion.rs2.model.dialogue.DialogueHandler#getDialogue(org.hyperion.rs2.model.Player)
      */
      @Override
      public void getDialogue(Player player) {

      /** The loop that will be used to grab all data out of dialogue */
      for(int i = 0; i < dialogue.getTypes().length; i++) {

       /** The type of dialogue */
 XMLDialogue.DialogueType type = XMLDialogue.DialogueType.valueOf(dialogue.getTypes()[i]);
 
 /** The emotion of the dialogue*/
 DialogueEmotion emotion = DialogueEmotion.valueOf(dialogue.getEmotions()[i]);
 
 /** If the dialogue is continuing */
 boolean continuing = dialogue.isContinuing()[i];
 
 /** The text of the dialogue */
 String[] text = dialogue.getText()[i];

 /**
  * switches through type of dialogues
  */
 switch(type) {
 
 /**
  * If the dialogue is a player dialogue
  */
 case PLAYER:
 	player.getDialogue().add(new PlayerDialogue(player, emotion, continuing, text));
 	break;
 	
 /**
  * If the dialogue is a mob dialogue
  */	
 case MOB:
 	player.getDialogue().add(new MobDialogue(player, dialogue.getId(), emotion, continuing, text));
 	break;
 	
 /**
  * If the dialogue is a info dialogue
  */	
 case INFO:
 	player.getDialogue().add(new InfoDialogue(player, continuing, text));
 	break;
 
 /**
  * If the dialogue is a option dialogue
  */
 case OPTION:
 	player.getDialogue().add(new OptionDialogue(player, text));
 	break;
 
 
 }

      }

    }
