public void applyDead() {

	if(c.duelStatus != 6) {
		c.killerId = findKiller();

			c.playerKilled = c.playerId;

			}
			if (Server.playerHandler.players[c.playerId].connectedFrom != o.lastKilled && c.duelStatus == 0) {
