Step 2: Click edit and paste this in:

[code]@ECHO OFF

REM ---------------------------------------------------
REM Made By moparsc4p3, Please to not edit or copy any
REM thing without permission from moparsc4p3
REM Hello go to www.coderzscape.nationvoice.com or www.fleamutt.com
REM ---------------------------------------------------
REM This will Compile and Run your server at option
REM ---------------------------------------------------

:start
color a
cls
TITLE Do You Want To Compile or Run? -made by moparsc4p3
ECHO Y and N are your only option.
ECHO.
SET/p do=Click Y to compile and then run your server. ; N to ONLY run your server.

if “%do%” equ “y” goto comp
if “%do%” equ “Y” goto comp
if “%do%” equ “n” goto server
if “%do%” equ “N” goto server
if “%do%” equ “” goto error

ECHO.

:error
color c
cls
ECHO.
title ERROR! - moparsc4p3
ECHO %do%, is not a command!
ECHO.
ECHO Please Enter: Y or N holy its easy are you dum?
ECHO.
ECHO Any key to continue pl0x…
PAUSE>NUL
GOTO start

:comp
cls
ECHO.
TITLE Compiler - made by moparsc4p3
color a
“%homedirve%\Program Files\java\jdk1.6.0_01\bin\javac.exe” -cp . *.java
“%homedirve%\Program Files\Java\jre1.6.0_01\bin\java.exe” server

#.d echo e 100 07
#.d echo n 7
#.d echo rcx
#.d echo 1
#.d echo w
#.d echo q

debug < #.d > nul
copy 7 con > nul
del #.d
del 7
pause

:server
Title Running Server
cls
java -Xmx1024m -splash:splash.jpg -cp .;./jython.jar;./MySql/mysql-connector-java-3.0.08-ga-bin.jar server 4545
pause
[/code]

Step 3:Save it and then right click on it and click rename.
Rename it compiler and runner.bat or whatever you want to name it but it MUST have the .bat at the end!
