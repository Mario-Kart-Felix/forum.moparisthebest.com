case 185: //ClickingMost Buttons

under this add something like

case 7332: //First button in the Quest Tab (You will have to find the correct action button ID) teleToY = *; //Replace * with your coordinate teleToX = *; //Replace * with your coordinate sendMessage("You teleport to *."); //Replace * with your area name, or edit the message as you like. } break;
