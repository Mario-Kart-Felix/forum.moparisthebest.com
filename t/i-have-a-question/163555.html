mod_taharok July 30, 2016, 5:34pm #2

Of course, here is an example of something doing quite that:

import javax.swing.JPanel;
import java.awt.Color;
import jav.awt.Graphics;
import java.awt.event.MouseAdapter;
import java.awt.event.MouseEvent;

public class Example extends JPanel {

	final private int TILE_SIZE = 16;
	private int selectionX = 0, selectionY = 0;
	final private Color FILL_COLOR = new Color(0, 255, 0, 80);

	public Example() {
		super();
		addListeners();
	}

	private void addListeners() {
		this.addMouseListener(new MouseAdapter() {
			public void mouseClicked(MouseEvent e) {
				selectionX = Math.round(e.getX() / TILE_SIZE) * TILE_SIZE;
				selectionY = Math.round(e.getY() / TILE_SIZE) * TILE_SIZE;
				repaint();
			}
		});
	}

	public void paint(Graphics g) {
		g.setColor(FILL_COLOR);
		g.fillRect(selectionX, selectionY, TILE_SIZE, TILE_SIZE);
	}
}

And the main class:

import javax.swing.JFrame;

public class Main extends JFrame {

	public Main() {
		super("Example");
		setSize(300, 300);
		add(new Example());
	}

	public static void main(String[] args) {
		new Main().setVisible(true);
	}
}

Undoubtedly one could elaborate further than that, but I think that is a good start. It will allow you to select tiles anywhere on the canvas.
