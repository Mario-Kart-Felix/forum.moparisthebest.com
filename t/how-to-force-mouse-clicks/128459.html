[code]import java.awt.Robot;
import java.awt.AWTException;
import java.awt.event.InputEvent;

public class Clicker
{
private Robot robertTheRobot;

public Clicker()
{
    try{
        robertTheRobot = new Robot();
    }catch(AWTException e) {}
}
public void click(int button)
{
    if(button == InputEvent.BUTTON1_MASK) {
        System.out.println("Left clicking");
    } else if(button == InputEvent.BUTTON3_MASK) {
        System.out.println("Right Clicking");
    }
    try {
        robertTheRobot.mousePress(button);
        Thread.sleep(61); // just a random amount of time
        robertTheRobot.mouseRelease(button);
    } catch(IllegalArgumentException e) {
        System.out.println("i like ur button, rly... but java doesnt");
    } catch(InterruptedException e) {
        System.out.println("interrupted ffs");
    }        
}

}
[/code]
