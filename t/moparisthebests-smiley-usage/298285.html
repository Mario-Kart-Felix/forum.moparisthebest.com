If you ever happen to look at Moparisthebest’s older posts, they all have something in common: the excessive use of “smileys” or “emoticons”.
Today’s project was to create a python script that would crawl every single one of mopar’s posts, and generate a CSV representation of the number of smilies and the date.
Here is a rendering of the CSV data:

As you can the count decreases significantly as time progresses.
Here is the python script, one of my poorer works I will admit, but it’s has little need to be versatile (that means its a HACK)

import urllib2
import urllib
import cookielib
import time

class SmileyCounter():
    url = "http://www.moparisthebest.com/smf/index.php?action=profile;u=1;sa=showPosts;start=";
    start_newest = 0;
    start_oldest = 6220;
    step = -20;
    total_smilies = 0;
    smiley_root = "http://www.moparisthebest.com/smf/Smileys/vb/";
    smiley_files = [
    "smile.gif",
    "wink.gif",
    "biggrin.gif",
    "mad.gif",
    "frown.gif",
    "eek.gif",
    "cool.gif",
    "rolleyes.gif",
    "tongue.gif",
    "redface.gif",
    "confused.gif",
    "shifty.gif",
    "winkgrin.gif",
    ];
    stats = [];
    date_search = '<td class="middletext" align="right" style="padding: 0 1ex; white-space: nowrap;">';

    def __init__(self):
        print "Logging In";
        opener = self.Login();
        print "Downloading pages";
        for start in range(self.start_oldest, self.start_newest, self.step):
            page = opener.open(self.url + str(start));
            data = page.read();
            count = self.CountSmilies(data);
            self.total_smilies += count[0];
            self.stats.append(count);

    def CountSmilies(self, html):
        count = 0;
        for smiley in self.smiley_files:
             count += html.count(smiley);
        dateStartIdx = html.find(self.date_search) + len(self.date_search);
        dateEndIdx = html.find("</td>", dateStartIdx);
        date = html[dateStartIdx:dateEndIdx].strip().replace('on: ', '');
        date = date[0:date.find(',')];
        return count, date;
    def Login(self):
        yesCookie = cookielib.Cookie(None, "mpr_agree2", "yes", None, None, ".moparisthebest.com", None, None, "/", None, False, time.time() + 60 * 60, False, None, None, None);
        loginCookie = cookielib.Cookie(None, "moparsmf", "GETURCOOKIEDATA", None, None, ".moparisthebest.com", None, None, "/", None, False, time.time() + 60 * 60, False, None, None, None);
        loginCookie2 = cookielib.Cookie(None, "SMFCookie17", "GETURCOOKIEDATA", None, None, ".moparisthebest.com", None, None, "/", None, False, time.time() + 60 * 60, False, None, None, None);
        cj = cookielib.CookieJar();
        cj.set_cookie(yesCookie);
        cj.set_cookie(loginCookie);
        cj.set_cookie(loginCookie2);
        opener = urllib2.build_opener(urllib2.HTTPCookieProcessor(cj));
        return opener;
    def DumpStats(self):
        for stat in self.stats:
            print stat[1] + '\t' + str(stat[0]);

sc = SmileyCounter();
sc.DumpStats();

I removed the hardcoded cookie contents, as I’d rather not have just anyone stealing my session. You can easily grab your own from firefox.
