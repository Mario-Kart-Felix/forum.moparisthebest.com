try making a new file in player named NPCCpmbat.java, start off from there, and if you already have it, just add npc and do the melee method

	public int CombatXPRate = 100;
      public void GiveMeleeXP(Player p) {
      if(p.AttackStyle == 1) {
  		p.addSkillXP(500 * p.MeleeDamage * CombatXPRate ,0);
      } else if(p.AttackStyle == 2) {
   		p.addSkillXP(500 * p.MeleeDamage * CombatXPRate ,2);
      } else if(p.AttackStyle == 3) {
   		p.addSkillXP(500 * p.MeleeDamage * CombatXPRate ,1);
      } else if(p.AttackStyle == 4) {
   		p.addSkillXP((500 * p.MeleeDamage * CombatXPRate)/3 ,0);
   		p.addSkillXP((500 * p.MeleeDamage * CombatXPRate)/3 ,1);
   		p.addSkillXP((500 * p.MeleeDamage * CombatXPRate)/3 ,2);
      }
      p.addSkillXP((5 * p.MeleeDamage * CombatXPRate)/3 ,3);
   }
