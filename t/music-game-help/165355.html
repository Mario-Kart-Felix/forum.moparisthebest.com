Not sure if this would be to any help, but here is a SoundHandler class that I made a while ago:

[code]/** SoundHandler.java */

import javax.sound.sampled.AudioFormat;
import javax.sound.sampled.AudioSystem;
import javax.sound.sampled.LineUnavailableException;
import javax.sound.sampled.SourceDataLine;

public abstract class SoundHandler {
private static final float SAMPLE_RATE = 8000f;

 private SoundHandler() {
      
 }
 
 
 
 public static void play(int hertz, int millis) {
      try {
           byte[] byteBuffer = new byte[1];
           
           AudioFormat audioFormat = new AudioFormat(SAMPLE_RATE, 8, 1, true, false);
           
           SourceDataLine sourceDataLine = AudioSystem.getSourceDataLine(audioFormat);
           sourceDataLine.open(audioFormat);
           sourceDataLine.start();
           
           for(int i = 0; i < millis * 8; i++) {
                double angle = i / (SAMPLE_RATE / hertz) * 2.0 * Math.PI;
                
                byteBuffer[0] = (byte)(Math.sin(angle) * 110.0);
                
                sourceDataLine.write(byteBuffer, 0, 1);
           }
           
           sourceDataLine.drain();
           sourceDataLine.stop();
           sourceDataLine.close();
      }
      catch(LineUnavailableException e) {
           
      }
 }

}[/code]
