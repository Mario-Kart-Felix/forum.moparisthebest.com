        public bool WalkMiniMap(ScionTile Tile)
        {
	    //Thanks to RSBot Dev Team.
	    private static int[] CURVESIN = new int[2048];
	    private static int[] CURVECOS = new int[2048];

            for (int i = 0; i < 2048; i++)
            {
                    CURVESIN[i] = (int)(65536.0 * Math.Sin(i * 0.0030679615));
                    CURVECOS[i] = (int)(65536.0 * Math.Cos(i * 0.0030679615));
            }

            int mmHeight = 152;
            int mmWidth = 152;
            int mmMasterX = 516;
            int mmMasterY = 0;
            int mmGetX = 35;
            int mmGetY = 9;

            double x = Tile.getX();
            double y = Tile.getY();

            x -= GetBaseX();
            y -= GetBaseY();

            int calculatedX = ((int)(x * 4 + 2)) - GetPlayerX2() * 4;
            int calculatedY = ((int)(y * 4 + 2)) - GetPlayerY2() * 4;

            int angle = stub.botClient.GetCompassAngle() + stub.botClient.GetMapOffsetAngle() & 0x7ff;
            int actDistSq = calculatedX * calculatedX + calculatedY * calculatedY;

            int mmDist = Math.Max(mmHeight / 2, mmWidth / 2);
            if (mmDist * mmDist >= actDistSq)
            {
                int cs = CURVESIN[angle];
                int fact = 256 + stub.botClient.GetMiniMapScale();
                cs = 256 * cs / fact;
                int cc = CURVECOS[angle];
                cc = 256 * cc / fact;
                int i_25_ = -(calculatedX * cs) + calculatedY * cc >> 16;
                int i_26_ = calculatedX * cc + calculatedY * cs >> 16;

                int screenx = (mmGetX + mmMasterX) + mmWidth / 2 + i_26_;
                int screeny = -i_25_ + mmHeight / 2 + (mmGetY + mmMasterY);
                stub.mouse.ClickMouse(screenx, screeny, true);
                return true;
            }
            else
            {
                return false;//not on minimap
            }

        }

If you look at rsbot, you see it’s initialized in the static method of the client.
This means those arrays are only calculated once. In your case it’s calculated every call.

Change your class to this: (I’ll take the name Calculations as class name in this example)

class Calculations
{
   private static int[] CURVESIN = new int[2048];
   private static int[] CURVECOS = new int[2048]
   
   static Calculations
   {
      for (int i = 0; i < 2048; i++)
      {
         CURVESIN[i] = (int)(65536.0 * Math.Sin(i * 0.0030679615));
         CURVECOS[i] = (int)(65536.0 * Math.Cos(i * 0.0030679615));
      }
   }

   public bool WalkMiniMap(ScionTile Tile)
   {
      int mmHeight = 152;
      int mmWidth = 152;
      int mmMasterX = 516;
      int mmMasterY = 0;
      int mmGetX = 35;
      int mmGetY = 9;

      double x = Tile.getX();
      double y = Tile.getY();

      x -= GetBaseX();
      y -= GetBaseY();

      int calculatedX = ((int)(x * 4 + 2)) - GetPlayerX2() * 4;
      int calculatedY = ((int)(y * 4 + 2)) - GetPlayerY2() * 4;

      int angle = stub.botClient.GetCompassAngle() + stub.botClient.GetMapOffsetAngle() & 0x7ff;
      int actDistSq = calculatedX * calculatedX + calculatedY * calculatedY;

      int mmDist = Math.Max(mmHeight / 2, mmWidth / 2);
      if (mmDist * mmDist >= actDistSq)
      {
         int cs = CURVESIN[angle];
         int fact = 256 + stub.botClient.GetMiniMapScale();
         cs = 256 * cs / fact;
         int cc = CURVECOS[angle];
         cc = 256 * cc / fact;
         int i_25_ = -(calculatedX * cs) + calculatedY * cc >> 16;
         int i_26_ = calculatedX * cc + calculatedY * cs >> 16;

         int screenx = (mmGetX + mmMasterX) + mmWidth / 2 + i_26_;
         int screeny = -i_25_ + mmHeight / 2 + (mmGetY + mmMasterY);
         stub.mouse.ClickMouse(screenx, screeny, true);
         return true;
      }
      else
      {
         return false;//not on minimap
       }
   }
}
