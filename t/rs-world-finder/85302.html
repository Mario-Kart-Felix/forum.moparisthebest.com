This will either List all the open worlds (not full) or return the URL of a world you specify (As long as it’s not full)

Syntax:
java URLFinder 15 this will find the URL of world 15
OR
java URLFinder all this will list all the non-full worlds, and the URL for it.

preview of it listing all worlds: http://rafb.net/p/SEo6SK86.html

preview of it listing 1 world:

World URL: http://se5.runescape.com/l1,p0,j0
World: 65

[code=Java]import java.io.DataInputStream;
import java.io.IOException;
import java.net.URL;
import java.net.URLConnection;
import java.util.Scanner;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class URLFinder {
public static void main(String… args) {
Scanner s = new Scanner(System.in);
System.out.print(“What world? (“all” for a list of worlds): “);
String world = s.next();
Pattern p = null;
if(world.contains(“all”))
p = Pattern.compile(”<a href=”(.)">World (.)");
else
p = Pattern.compile("<a href="(.*)">World " + Integer.parseInt(world) + “”);

	String toSearch = getContent();
	Matcher m = p.matcher(toSearch);

	while(m.find()) {
		String found = m.group();
		
		int quote = found.indexOf("\"") + 1;
		int quote2 = found.indexOf("\"", quote);
		
		String urlForWorld = found.substring(quote, quote2);
		int World = found.indexOf("World") + 1;
		World = found.indexOf(" ", World);
		
		int end = found.indexOf("</a>", World + 1);
		int theWorld = Integer.parseInt(found.substring(World, end).trim());
		System.out.println("World URL: " + urlForWorld);
		System.out.println("World: " + theWorld + "\n\n");	
	}
}

public static String getContent() {
	try {
		URL url = new URL("http://www.runescape.com/sln.ws?lores.x=1&plugin=0");
		URLConnection urlc = url.openConnection();
		DataInputStream dis = new DataInputStream(urlc.getInputStream());
		byte[] b = new byte[urlc.getContentLength()];
		dis.readFully(b);
		dis.close();
		
		return new String(b);
	} catch(IOException e) {
		e.printStackTrace();
		return "";
	}
}

public final class WorldInfo {

public final static List getWorlds(char[] buffer) throws IOException {

if(buffer == null)
  buffer = new char[0xffff];

InputStreamReader reader = new InputStreamReader(new URL(WORLD_SELECTION).openStream());
int length = 0;
for(int space = buffer.length, chunk; (chunk = reader.read(buffer, length, space)) != -1; ) {
  length += chunk;
  space -= chunk;
}
reader.close();

return parseWorlds(new Segment(buffer, 0, length));

}

public final static List parseWorlds(CharSequence s) {

LinkedList<WorldInfo> list = new LinkedList<WorldInfo>();

for(Matcher matcher = Pattern.compile(FUNCTION_PATTERN).matcher(s); matcher.find(); )
  list.add(new WorldInfo(matcher));

return list;

}
