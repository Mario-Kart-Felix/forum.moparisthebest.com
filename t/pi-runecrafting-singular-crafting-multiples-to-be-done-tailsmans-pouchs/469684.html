My concept for this was taking use of arrays and the use of java. Java is superior because most languages require you to rewrite the code instead of using voids (java) or defs (ruby) to call them, thus, making the file bigger. The concept is fully written in CraftRunes.java.

Step 9) Now, make a file named CraftRunes.java. Inside it, put this:

package server.content.skill.runecrafting;

import server.content.skill.SkillConstants;
import server.model.players.Client;

/**
 * Handles the action of crafting runes in the skill Runecrafting.
 * 
 * @author Arham 4
 * 
 */
public class CraftRunes extends RunecraftingData {

	/**
	 * Allows us to call Client.
	 */
	Client c;

	/**
	 * If the server fails to find any essence.
	 */
	private boolean failure;

	/**
	 * The constructor of the class.
	 * 
	 * @param c
	 *            Allows us to call Client
	 */
	public CraftRunes(Client c) {
		this.c = c;
	}

	private int RC_ANIM = 791; // The animation for runecrafting.
	private int howMany = 0; // Storer.

	/**
	 * Step one: Check the alters to see which rune to craft.
	 * 
	 * @param objectId
	 *            Detects the object id of the clicked object
	 */
	public void checkAlters(int objectId) {
		switch (objectId) {
		/*
		 * If the object is an Air Obelisk then craft an air rune.
		 */
		case 2478:
			craftAir();
			break;
		/*
		 * If the object is a Mind Obelisk then craft a mind rune.
		 */
		case 2479:
			craftMind();
			break;
		/*
		 * If the object is a Water Obelisk then craft a water rune.
		 */
		case 2480:
			craftWater();
			break;
		/*
		 * If the object is a Earth Obelisk then craft an earth rune.
		 */
		case 2481:
			craftEarth();
			break;
		/*
		 * If the object is a Fire Obelisk then craft a fire rune.
		 */
		case 2482:
			craftFire();
			break;
		/*
		 * If the object is a Body Obelisk then craft a body rune.
		 */
		case 2483:
			craftBody();
			break;
		/*
		 * If the object is a Cosmic Obelisk then craft a cosmic rune.
		 */
		case 2484:
			craftCosmic();
			break;
		/*
		 * If the object is a Chaos Obelisk then craft a chaos rune.
		 */
		case 2487:
			craftChaos();
			break;
		/*
		 * If the object is a Nature Obelisk then craft a nature rune.
		 */
		case 2486:
			craftNature();
			break;
		/*
		 * If the object is a Law Obelisk then craft a law rune.
		 */
		case 2485:
			craftLaw();
			break;
		/*
		 * If the object is a Death Obelisk then craft a death rune.
		 */
		case 2488:
			craftDeath();
			break;
		/*
		 * If the object is a Blood Obelisk then craft a blood rune.
		 */
		case 3624:
			craftBlood();
			break;
		}
	}

	/**
	 * Step two: Do the actions for each rune.
	 */
	private void craftBlood() {
		removeEssence(true);
		if (failure)
			returnMessage(true);
		else if (!failure)
			startActions(12, false);
	}

	private void craftDeath() {
		removeEssence(true);
		if (failure)
			returnMessage(true);
		else if (!failure)
			startActions(11, false);
	}

	private void craftLaw() {
		removeEssence(true);
		if (failure)
			returnMessage(true);
		else if (!failure)
			startActions(10, false);
	}

	private void craftNature() {
		removeEssence(true);
		if (failure)
			returnMessage(true);
		else if (!failure)
			startActions(9, true);
	}

	private void craftChaos() {
		removeEssence(true);
		if (failure)
			returnMessage(true);
		else if (!failure)
			startActions(8, true);
	}

	private void craftCosmic() {
		removeEssence(true);
		if (failure)
			returnMessage(true);
		else if (!failure)
			startActions(7, true);
	}

	private void craftBody() {
		removeEssence(false);
		if (failure)
			returnMessage(true);
		else if (!failure)
			startActions(6, true);
	}

	private void craftFire() {
		removeEssence(false);
		if (failure)
			returnMessage(false);
		else if (!failure)
			startActions(5, true);
	}

	private void craftEarth() {
		removeEssence(false);
		if (failure)
			returnMessage(false);
		else if (!failure)
			startActions(4, true);
	}

	private void craftWater() {
		removeEssence(false);
		if (failure)
			returnMessage(false);
		else if (!failure)
			startActions(3, true);
	}

	private void craftMind() {
		removeEssence(false);
		if (failure)
			returnMessage(false);
		else if (!failure)
			startActions(2, true);
	}

	private void craftAir() {
		removeEssence(false);
		if (failure)
			returnMessage(false);
		else if (!failure)
			startActions(0, true);
	}

	private void returnMessage(boolean both) {
		if (both)
			c.sendMessage("You need to have either a rune essence or a pure essence to craft this rune.");
		else if (!both)
			c.sendMessage("You need to have a rune essence to craft this rune.");
	}

	/**
	 * Step 2.1: Remove the essence based on if the rune can use pure essence or
	 * not.
	 * 
	 * @param both
	 *            See if rune can be crafted by both essence (Rune and Pure)
	 */
	private void removeEssence(boolean both) {
		boolean oneFalse = false;
		boolean twoFalse = false;
		if (both) {
			if (c.getItems().playerHasItem(1436)) {
				while (c.getItems().playerHasItem(1436)) {
					c.getItems().deleteItem(1436, 1);
					howMany++;
				}
			} else
				oneFalse = true;
			if (c.getItems().playerHasItem(7936)) {
				while (c.getItems().playerHasItem(7936)) {
					c.getItems().deleteItem(7936, 1);
					howMany++;
				}
			} else
				twoFalse = true;
			if (oneFalse && twoFalse)
				failure = true;
		} else if (!both) {
			if (c.getItems().playerHasItem(1436)) {
				while (c.getItems().playerHasItem(1436)) {
					c.getItems().deleteItem(1436, 1);
					howMany++;
				}
			} else
				oneFalse = true;
			if (oneFalse)
				failure = true;
		}
	}

	/**
	 * Step 2.2: Start the actions for crafting.
	 * 
	 * @param section
	 *            What crafting section should the server look at?
	 * @param canMultiply
	 *            Can the rune be a multiple?
	 */
	private void startActions(int section, boolean canMultiply) {
		if (canMultiply) {
			if (c.playerLevel[SkillConstants.RUNECRAFTING] >= multipleEssence[section][0]) {
				c.startAnimation(RC_ANIM);
				c.getItems().addItem(multipleEssence[section][2],
						howMany * multipleEssence[section][3]);
				c.getPA().addSkillXP(multipleEssence[section][1],
						SkillConstants.RUNECRAFTING);
				howMany = 0;
			} else
				startActions(section, false);
		} else if (canMultiply == false) {
			if (c.playerLevel[SkillConstants.RUNECRAFTING] <= singularEssence[section][0]
					&& singularEssence[section][0] < multipleEssence[section][0]) {
				c.startAnimation(RC_ANIM);
				c.getItems().addItem(singularEssence[section][2], howMany);
				c.getPA().addSkillXP(singularEssence[section][1],
						SkillConstants.RUNECRAFTING);
				howMany = 0;
			} else
				c.sendMessage("You need a runecrafting level of "
						+ singularEssence[section][0] + " to craft this rune.");
		}
	}

	/**
	 * FINISHED!
	 */

}

This handles crafting runes.
