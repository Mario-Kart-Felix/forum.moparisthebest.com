package org.evelus.fs.entries;

import java.io.IOException;
import java.nio.ByteBuffer;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import org.evelus.fs.FileEntry;
import org.evelus.fs.Packable;
import org.evelus.fs.archives.FileContainer;

/**
 * ChildContainer.java
 * @version 1.0.0
 * @author RuneTekk Development (SiniSoul)
 */
public final class ChildContainer implements FileEntry<FileContainer> {
    
    /**
     * The {@link ChildContainerEntry} list for this {@link ChildContainer}.
     */
    private List<ChildContainerEntry> entries;
    
    /**
     * The default amount of entries to decode this {@link ChildContainer}.
     */
    private int defaultEntries;
    
    /**
     * Represent an entry for this child container.
     */
    public static class ChildContainerEntry implements Packable {
        
        /**
         * The payload array for this {@link ChildContainerEntry}.
         */
        private byte[] payload;
        
        /**
         * Packs a byte array into the payload.
         * @param src The source array.
         * @throws IOException An IOException was thrown while packing.
         */
        public void pack(byte[] src) throws IOException {
            pack(src, 0, src.length);
        }

        /**
         * Packs the bytes back into the payload array.
         * @param src The source array.
         * @param offset The offset in the source array.
         * @param length The amount of bytes to pack.
         * @throws IOException An IOException was thrown while packing.
         */
        @Override
        public void pack(byte[] src, int offset, int length) throws IOException {
            payload = new byte[length - offset];
            System.arraycopy(src, offset, payload, 0, length);
        }

        /**
         * Nothing to unpack.
         * @return The child container entry array.
         * @throws IOException An IOException was thrown while unpacking. 
         *                     Although, this never should happen.
         */
        @Override
        public byte[] unpack() throws IOException {
            return payload;
        }
    }

    @Override
    public void decode(FileContainer t) throws IOException {
        entries.clear();
        ByteBuffer buffer = ByteBuffer.wrap(t.unpack());
        int pos = buffer.capacity() - 1;
        buffer.position(pos);
        int off = buffer.get();
        pos -= 4 * (off * defaultEntries);
        buffer.position(pos);
        int[] entryOffsets = new int[defaultEntries];      
        int entryOffset = 0;
        for (int i = 0; i < entryOffsets.length; i++) {
            entryOffset += buffer.getInt();
            entryOffsets[i] += entryOffset;
        }
        ChildContainerEntry[] childEntries = new ChildContainerEntry[defaultEntries];
        for (int i = 0; i < childEntries.length; i++) {
            childEntries[i] = new ChildContainerEntry();
            childEntries[i].payload = new byte[entryOffsets[i]];
            entryOffsets[i] = 0;
        }
        /* Although most of this is unneeded I am thinking that this might be 
         * updated in the future and it'd be safe to closely emulate how the 
         * client parses the children containers 
         */
        int entryLength = 0, srcOffset = 0;
        buffer.position(pos);
        for (int i = 0; i < childEntries.length; i++) { 
            entryLength += buffer.getInt();
            System.arraycopy(buffer.array(), srcOffset, childEntries[i].payload, entryOffsets[i], entryLength);
            srcOffset += entryLength;
            entryOffsets[i] += entryLength;
        }
        entries.addAll(Arrays.asList(childEntries));
    }

    @Override
    public void encode(FileContainer t) throws IOException {
        int bufferSize = 1 + 4 * entries.size();
        for(ChildContainerEntry entry : entries)
            bufferSize += entry.payload.length;
        ByteBuffer buffer = ByteBuffer.allocate(bufferSize);
        for(int i = 0; i < entries.size(); i++) {
            ChildContainerEntry entry = entries.get(i);
            if(entry == null)
                continue;
            if(entry.payload != null)
                buffer.put(entry.payload);
            else 
                throw new IOException("Error: Null payload in child container entry.");
        }
        int lastValue = 0;
        for(int i = 0; i < entries.size(); i++) {
            ChildContainerEntry entry = entries.get(i);
            if(entry == null)
                continue;
            int offset = entry.payload.length - lastValue;
            buffer.putInt(offset);
            lastValue = entry.payload.length;
        }
        buffer.put((byte) 1);
        t.pack(buffer.array(), 0, buffer.position());
        this.defaultEntries = entries.size();
    }
    
    /**
     * Sets the amount of default entries for decoding.
     * @param entries The default amount of entries to load.
     */
    public void setDefaultEntries(int entries) {
        this.defaultEntries = entries;
    }
    
    /**
     * Gets a {@link ChildContainerEntry} from the entries list.
     * @param index The index of the {@link ChildContainerEntry} to get.
     * @return The entry.
     */
    public ChildContainerEntry getEntry(int index) {
        return entries.get(index);
    }
    
    /**
     * Constructs a new {@link ChildContainer};
     * @param defaultEntries The default amount of entries in the child container.
     */
    public ChildContainer(int defaultEntries) {
        entries = new ArrayList<ChildContainerEntry>();
        this.defaultEntries = defaultEntries;
    }
}
