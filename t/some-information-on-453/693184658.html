public XY toMiniMap() throws InterruptedException {
		AryanTile myPlayer = theReflectionMethods.getLocation();
		int xDiff = x - myPlayer.x;
		int yDiff = y - myPlayer.y;
		XY aPoint = new XY(xDiff * 4, yDiff * 4);
		XY aRotatedPoint = CompassMethods.rotate(aPoint, Math
				.toRadians(theReflectionMethods.getCompass()));
		XY thePointOnVirtualMiniMap = new XY(MiniMap.centerX + aRotatedPoint.x,
				MiniMap.centerY - aRotatedPoint.y);
		if (MiniMap.inMiniMap(thePointOnVirtualMiniMap)) {
			return thePointOnVirtualMiniMap;
		}
		return null;
	}
