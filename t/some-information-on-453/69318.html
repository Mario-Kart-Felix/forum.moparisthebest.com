 public void doPath(Tile[] path) throws InterruptedException {
        int loops = 0;
        Tile end = path[path.length - 1];
        while (distanceTo(end) > 1 && loops < 100) {
            Tile next = findTileAlongInvalidPath(path);
            if (next == null)
                return;
            XY screen = TileToMinimap(next);
            if (screen != null) {
                mouseClickLeft(screen);
                sleep(walkTime(next));
            }
            loops++;
        }
    }
public final Tile findTileAlongInvalidPath(Tile[] path) {
        int MostNear = 0;
        for (int i = 0; i < path.length; i++) {
            if (distanceTo(path[i]) < distanceTo(path[MostNear])) {
                MostNear = i;
            }
        }
        int SecMostNear = MostNear;
        if (SecMostNear < path.length - 1) {
            SecMostNear++;
        }
        while (SecMostNear < path.length - 1
                && distanceTo(path[SecMostNear + 1]) < 11) {
            SecMostNear++;
        }
        if (distanceTo(path[SecMostNear]) < 11) {
            return path[SecMostNear];
        } else {
            Tile challenger1 = path[MostNear];
            Tile challenger2 = path[SecMostNear];
            Tile winner = challenger1;
            int winnerDist = distanceTo(winner);
            int distanceBetweenChall = distance(challenger1, challenger2);
            int StartX = challenger1.x;
            int StartY = challenger1.y;
            int dx = StartX - challenger2.x;
            int dy = StartY - challenger2.y;
            for (int i = 0; i <= distanceBetweenChall; i++) {
                Tile Challenger = new Tile(StartX + dx
                        * (i / distanceBetweenChall), StartY + dy
                        * (i / distanceBetweenChall));
                int challengerDist = distanceTo(Challenger);
                if (challengerDist < winnerDist) {
                    winnerDist = challengerDist;
                    winner = Challenger;
                }
            }
            return findTileOnWayToTile(winner);
        }
    }
/**
     * Finds a tile in the direction of the Destionation
     * 
     * @param Destination
     *            The tile to walk to
     * @return Tile in the direction of the Destination, otherwise null.
     */
    public final Tile findTileOnWayToTile(Tile Destination) {
        if (Destination == null)
            return null;
        Tile next = null;
        int x_difference = Destination.x - getTile(getPlayer()).x;
        int y_difference = Destination.y - getTile(getPlayer()).y;
        int abs_x_difference = Math.abs(x_difference);
        int abs_y_difference = Math.abs(y_difference);
        int multiple;
        int x;
        int y;
        int x_change = 0;
        int y_change = 0;
        int size = 0;
        if (distanceTo(Destination) > 9 || !(canReach(Destination))) {
            multiple = 7;
            x = (getTile(getPlayer()).x + multiple * x_difference
                    / (abs_x_difference + abs_y_difference));
            y = (getTile(getPlayer()).y + multiple * y_difference
                    / (abs_x_difference + abs_y_difference));
            while (!(canReach(new Tile(x, y)))) {
                if (size >= 8)
                    break;
                else if (y_change == size && x_change < size && x_change > 0)
                    x_change++;
                else if (x_change == 0 && y_change >= 0) {
                    y_change++;
                    x_change++;
                    size++;
                } else if (y_change == size && x_change < 0)
                    x_change++;
                else if (y_change == (-1 * size) && x_change > (-1 * size))
                    x_change--;
                else if (x_change == size)
                    y_change--;
                else if (x_change == -1 * size)
                    y_change++;
            }
            if (size < 8) {
                next = new Tile(x + x_change, y + y_change);
            }
        } else {
            next = Destination;
        }
        return next;
    }
