function addPost($str1, $str2, $str3) {
		global $db;
		global $errors;
		if(empty($str1) || empty($str2)) {
			$errors .= "You've left a field empty.";
		}
		if(strlen($str1) > 75) {
			$errors .= "The post title cannot be longer than 140 characters.";
		}
		if(strlen($str2) > 10000) {
			$errors .= "Your message cannot be greater than 10,000 characters.";
		}
		if($str3 != $_SESSION['admin']) {
			$errors .= "Invalid key.";
		}
		if(empty($errors)) {
			$stmt = $db->prepare("INSERT INTO posts (title, message, date) VALUES (:title, :message, :date)");
			$stmt->bindParam(":title", $str1, PDO::PARAM_STR);
			$stmt->bindParam(":message", $str2, PDO::PARAM_STR);
			$stmt->bindParam(":date", date("m-d-y"), PDO::PARAM_STR);
			$stmt->execute();
			header("Location: index.php?module=blog");
			die();
		}
	}
