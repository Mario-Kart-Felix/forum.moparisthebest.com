SETO, to print the bill ot a different place or file just do this (assuming you are writing ASCII instead of Binary):

try{
     File out = new File("bill.txt");
     FileOutputStream output = new FileOutPutStream(out);
     PrintWriter pw = new PrintWriter(output);
     
     for(int i=0; i<(array of items or what-have-you); i++){
           pw.println(i);
     }
     
    pw.close();
    output.close();
}
catch(Exception e){
    System.out.println();
}
