import org.arsenal.net.io.InputStream as stream
import org.arsenal.net.io.Session as session
import org.arsenal.rscache.Cache as cache
import org.arsenal.world.model.player.Player as player
import org.arsenal.util.Misc as misc
import org.arsenal.util.SerializableFilesManager as fileManager
import logging

class Login562:
    
    def decode(stream):
        session.setDecoder(-1)
        packetId = stream.readUnsignedByte()
        logging.info("Packet Id"+packetId)
        if packetId == 16 or packetId == 18 or packetId == 19:
            decodeWorldLogin(stream)
        else: 
            logging.info("PacketId: "+packetId)
            session.getChannel().close()
        
    def decodeWorldLogin(stream):
        packetSize = stream.readUnsignedByte()
        if packetSize != stream.getRemaining():
            session.getChannel().close()
            return
        if stream.readInt() != 562:
            logging.info("Invalid revision trying to connect")
            return
        stream.readByte()
        displayMode = stream.readUnsignedShort()
        screenWidth = stream.readUnsignedShort()
        screenHeight = stream.readUnsignedShort()
        for i in range(0, 24):
            stream.readInt()
        stream.readString()
        stream.readInt()
        stream.readInt()
        stream.readShort()
        for i in range(0, 29):
            if cache.getCacheFileManagers()[i] == null:
                crc = 0
            else:
                crc = cache.getCacheFileManagers()[i].getInformation().getInformationContainer().getCrc()
            if crc != stream.readInt():
                logging.info("Invalid Crc Keys")    
        if stream.readUnsignedByte() != 10:
            logging.info("Invalid rsa header")
            return
        for i in range(0, 4):
            stream.readInt()
        playerLongName = stream.readLong()
        password = stream.readString()
        username = misc.formatPlayerNameForProtocol(misc.longToString(playerLongName))
        if fileManager.containsPlayer(username):
            player = player(password)
        else:
            player = fileManager.loadPlayer(username)
            if player == null:
                return
        player.init(session, username, displayMode, screenWidth, screenHeight)
        session.setDecoder(3, player)
        session.setEncoder(2, player)
        player.start()
