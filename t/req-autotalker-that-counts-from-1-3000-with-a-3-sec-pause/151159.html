Var
I : Integer;
Begin
I:=1
Repeat
sendkeys(’::emote ’ + IntToStr(I)))
Wait(3000)
Until(I=3000)
End

or he could use this (fill in the typespeed with only slow or fast)

[code]program talker;
{.include srl/srl.scar}
const
typespeed = ‘’;
{
type slow for slow typing and fast for fast typing
in between the ‘’ (after typespeed = ) above
}
var
x : integer;
procedure report;
begin
cleardebug;
writeln(’------------------------------------------------------’);
writeln(‘You got up to ::emote ’ + inttostr(x) ’ befor terminating the script’);
writeln(‘Your Type Speed was ’ + typespeed);
writeln(’------------------------------------------------------’);
end;

procedure talk;
begin
activateclient;
cleardebug;
if typespeed = ‘slow’ then
begin
for x:= 1 to 3000 do
begin
typesend(’::emote ’ + inttostr(x));
report;
wait(3000)
end;
end;
if typespeed = ‘fast’ then
begin
for x:= 1 to 3000 do
begin
sendkeys(’::emote ’ + inttostr(x));
sendkeys(chr(13));
wait(3000)
report;
end;
end;
end;

begin
setupsrl;
talk;
end.[/code]
