import java.util.Scanner;
public class AgePredicterBigGrin {

    static class LanGuage {

        static class English {
            public static final String Question_One = "What Year Is It Now? ";
            public static final String Question_Two = "What Year Do You Want To Know Your Age? ";
            public static final String Question_Three = "How Old Are You Now? ";
            public static final String Prediction_Format = "You Will Be %d Years Old In %d";
        }
    }

    static class COnstants {

        public static final Integer New_String_Length_Constant = new Integer(0);
    }

    static class Scanner_Helper {

        public final String Helper_Name;
        public Scanner Scanner;

        public Scanner_Helper(String Name) {
            String Temporary_String = Name;
            Name = new String(new char[COnstants.New_String_Length_Constant.shortValue()]);
            Helper_Name = Temporary_String;
        }

        public void Get_Scanner(Scanner Scanner) {
            Scanner_Helper Scanner_Helper = this;
            Scanner_Helper.Scanner = Scanner;
        }

        public Scanner Give_Scanner() {
            Scanner_Helper Scanner_Helper = this;
            return Scanner_Helper.Scanner;
        }
    }

    public static void main(String[] args) throws IOException {
        Scanner_Helper Scanner_Helper = new Scanner_Helper("My Scanner :)");

        Scanner Scanner = new Scanner(System.in);

        Scanner_Helper.Get_Scanner(Scanner);

        System.out.print(LanGuage.English.Question_One);

        Scanner = Scanner_Helper.Give_Scanner();

        Integer First_Question_Answer = new Integer(Scanner.next());

        System.out.print(LanGuage.English.Question_Two);

        Scanner = Scanner_Helper.Give_Scanner();

        Integer Second_Question_Answer = new Integer(Scanner.next());

        System.out.print(LanGuage.English.Question_Three);

        Scanner = Scanner_Helper.Give_Scanner();

        Integer Third_Question_Answer = new Integer(Scanner.next());

        Integer New_Age = new Integer((Second_Question_Answer.intValue() - First_Question_Answer.intValue()) + Third_Question_Answer.intValue());

        System.out.printf(LanGuage.English.Prediction_Format, New_Age.intValue(), Second_Question_Answer.intValue());
    }
}
