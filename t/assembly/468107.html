Simple program made in Assembly that I started learning not long time ago. I wrote a lot of comments so any of you can understand :

.386						; We use the registries of the 80386 processor
.model flat, stdcall
option casemap : none

include \masm32\include\masm32rt.inc	; Includes the file


.code						; Defines that we start writing the code part

main :						; Starts a label

print "ch content : "		; Displays a message
mov ch, 40					; Moves the decimal value 40 in the ch registry (8 bits)
movzx eax, ch				; Extends the ch content to 32 bits so it fits in register eax
print str$ (eax), 13, 10	; Displays the value of eax registry

print "cl content : "		; Displays a message
mov cl, 35					; Moves the decimal value 35 in the cl registry (8 bits)
movzx eax, cl				; Extends the cl content to 32 bits so it fits in register eax
print  str$ (eax), 13, 10	; Displays the value of eax registry

inkey "Press any key to continue..."

exit  						; Closes the console
 
end main					; Ends the main label
