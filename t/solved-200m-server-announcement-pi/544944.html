public boolean addSkillXP(int amount, int skill) { int random3 = Misc.random(300); if (c.xpLock == true || c.Jail == true) { if (random3 == 1) { c.sendMessage("No XP added - reason: XP LOCK!"); } return false; } if (c.playerRights == 4 || c.playerRights == 5 || c.playerRights == 1) { c.playerXP[skill] += amount * 2.0; } else if (c.playerRights == 6) { c.playerXP[skill] += amount * 3.0; } else { c.playerXP[skill] += amount; } if (amount + c.playerXP[skill] < 0 || c.playerXP[skill] > 200000000) { if (c.playerXP[skill] > 200000000) { c.playerXP[skill] = 200000000; } return false; } if (c.wearingBrawlers() && c.playerLevel[7] > 50 && c.playerLevel[7] < 70 && c.playerSkilling[7] == true && c.inWild()) { amount += 5000; c.sendMessage("[COOKING BRAWLERS] 5K Bonus XP Added. Charges: " + c.brawlerFM + "/150."); c.brawlerFM += 1; handleDegradingBrawlersFM(); } if (c.wearingFMBrawlers() && c.playerLevel[7] > 70 && c.playerLevel[7] < 90 && c.playerIsFiremaking == true && c.inWild()) { amount += 8000; c.sendMessage("[FM BRAWLERS] 8K Bonus XP Added.
