	public static Sprite createSprite(BufferedImage img, String name) throws IOException
	{
		Sprite sprite = new Sprite(Unpacker.getCache(0).getFile(1).toArchive(), name);
		sprite.width = img.getWidth();
		sprite.height = img.getHeight();
		sprite.setPixels(new int[sprite.width * sprite.height]);
		sprite.setSpecialWidth(sprite.width);
		sprite.setSpecialHeight(sprite.height);
		sprite.setXOffset(0);
		sprite.setYOffset(0);
		ArrayList<Integer> palette = new ArrayList<Integer>();
		for(int x = 0; x < sprite.width; x++)
		{
			for(int y = 0; y < sprite.height; y++)
			{
				int color = img.getRGB(x, y);
				sprite.setPixel(x, y, color);
				if(!palette.contains(color))
					palette.add(color);
			}
		}
		sprite.setNumColors(palette.size());
		sprite.setPalette(convert(palette));
		return sprite;
	}
