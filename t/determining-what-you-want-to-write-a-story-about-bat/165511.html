Here it is as a .bat its V1 i will keep adding more ideas to it but for now i think its not to bad

[code]@echo off

echo This is the ultimate plan to make a story made by Ancient(V1).

echo. Ancientforums.smfforfree2.com

echo Choose one of the options below:

echo.

echo -plan

echo -website

echo -credits

echo -exit

set /p input=
if %input%==plan goto recipt
if %input%==exit goto exit
if %input%==website goto website
if %input%==credits goto credits

:recipt
echo First when is the story set?
echo Past: Present: Future

set /p input=
if %input%==present goto Opt1
if %input%==past goto Opt2
if %input%==future goto Opt3
echo -exit
pause
:1st
echo First or 3rd Person?
echo 1st:3rd
set /p input=
if %input%==1st goto main character 1
if %input%==3rd goto main character 1
echo -exit
pause
:Opt1
echo Theme?
echo Alien invasion(type alien):War:Dictatorship(type dict):Rebellion(type reb)
set /p input=
if %input%==alien goto 1st
if %input%==war goto 1st
if %input%==dict goto 1st
if %input%==reb goto 1st
echo -exit
pause
:Opt2
echo Theme?
echo RS sort of thing:Egyptian(type egypt):WWII(type ww2) or WWI(ww1): Ancient Greece/Rome(ancient)
set /p input=
if %input%==rs goto 1st
if %input%==egypt goto 1st
if %input%==ww2 goto 1st
if %input%==ww1 goto 1st
if %input%==ancient goto 1st
echo -exit
pause
:Opt3
echo Theme?
echo You’re on you own WAY to many options :D(type cont)
set /p input=
if %input%==cont goto 1st
echo -exit
pause
:main character 1
echo Main character(s)?
echo 1:Group
set /p input=
if %input%==1 goto main character 2
if %input%==group goto main character 2
echo -exit
pause
:main character 2
echo Main character?
echo Boy:Girl
set /p input=
if %input%==boy goto main character 3
if %input%==girl goto main character 3
echo -exit
pause
:main character 3
echo Main Character?
echo Special Powers(type special):Normal
set /p input=
if %input%==special goto plot
if %input%==normal goto plot
echo -exit
pause
:plot
echo Plot?
echo A Mystery:Out and Out BLOOD AND GORE :P(type gore)
set /p input=
if %input%==mystery goto main character 4
if %input%==gore goto main character 4
echo -exit
pause
:main character 4
echo Main character?
echo special(why?):Normal
set /p input=
if %input%==special goto end
if %input%==normal goto end
echo -exit
pause
:end
echo Now you have your idea for a story laid out i hope
echo you enjoyed the Ancient Story.bat file
:credits
echo All Ancient
echo Type exit to leave or type plan to start again
set /p input=
if %input%==exit goto exit
if %input%==plan goto recipt
pause
:website
echo ancientforums.smfforfree2.com
echo type exit to leave or type plan to start again
set /p input=
if %input%==exit goto exit
if %input%==plan goto recipt
pause

:exit
exit
pause [/code]
