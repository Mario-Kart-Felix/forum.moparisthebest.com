Keegan1 July 31, 2016, 10:08am #2

//X,Y Is where door is
//Door state 0 = closed, 1 = open.
//Face: 0= WEST | -1 = NORTH | -2 = EAST | -3 = SOUTH
//Face = Face Begin With
//FaceOpen = Face Opened Door
//FaceClosed = Face Closed Door
//Height 0 = Ground Level, 1 Is up ladder, 2 Is up up ladder, 3 = Underground I think.

// XCoord YCoord Id FaceO FaceC Face State Height
door = 2726 3368 1533 -2 -3 -3 1 0 //example

(Remember to use Tab to space it out)
