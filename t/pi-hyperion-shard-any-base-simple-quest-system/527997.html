I decided to create an API for programming quests so I don’t have to keep repeating the same code for each quest. Disclaimer: This is not a tutorial that teaches you how to implement a quest, but instead an API that will let you program quests alot more tidily and quicker. However, to get this working, you will need to change the variable names accordingly to your server. For example, to change the getLevel method, all you need to do is replace player.getSkill().getSkillLevel()[Skill.STRENGTH]; with player.playerLevel[2] (if you're using PI)

First of all, paste this class anywhere in ur source

package net.mikuframework.rs2.ltsmith.model.quest;

import net.mikuframework.rs2.ltsmith.model.Player;


/**
 * Represents a Quest
 * 
 * @author LT Smith
 * 
 */
public abstract class Quest {

	private Player player;

	private int id;

	private String name;

	private QuestStage progressIndicator;

	private int stage;

	public abstract String[] questJournalFrames();

	public abstract QuestStage getProgress();

	public abstract void formatQuestJournal();

	public static enum QuestStage {
		NOT_STARTED, IN_PROGRESS, FINISHED;
	}

	public abstract boolean hasRequirements();

	public Player getPlayer() {
		return player;
	}

	public Quest(Player thePlayer, int idNumber, String name) {
		this.setPlayer(thePlayer);
		this.setId(idNumber);
		this.setName(name);
	}

	public void updateStage(int newStage) {
		this.setStage(newStage);
		this.getPlayer().rewriteQuestListInterface();
	}

	public void setPlayer(Player player) {
		this.player = player;
	}

	public void setName(String name) {
		this.name = name;
	}

	public int getId() {
		return id;
	}

	public void setId(int id) {
		this.id = id;
	}

	@Override
	public String toString() {
		return name;
	}

	public int getStage() {
		return stage;
	}

	public void setStage(int stage) {
		this.stage = stage;
	}

	public QuestStage getProgressIndicator() {
		return progressIndicator;
	}

	public void setProgressIndicator(QuestStage progressIndicator) {
		this.progressIndicator = progressIndicator;
	}
}

Secondly (this is an optional stage, it is not exactly necessary. skip it if you think you’ll beable to understand the API without an example), you can implement an example quest

package net.mikuframework.rs2.ltsmith.model.quest.impl;

import net.mikuframework.rs2.ltsmith.model.Player;
import net.mikuframework.rs2.ltsmith.model.Skill;
import net.mikuframework.rs2.ltsmith.model.quest.Quest;
import net.mikuframework.rs2.ltsmith.model.quest.QuestConstants;

/**
 * Example quest implementation
 * 
 * @author LT Smith
 * 
 */
public class DragonLolzQuest extends Quest {

	private int dragonKills;

	public DragonLolzQuest(Player thePlayer, int idNumber, String name) {
		super(thePlayer, idNumber, name);
	}

	@Override
	public String[] questJournalFrames() {
		// TODO Auto-generated method stub
		return new String[] { "pancakes", "are", "sexy" };
	}

	@Override
	public void formatQuestJournal() {
		this.getPlayer().getActionSender()
				.sendInterface(QuestConstants.QUEST_JOURNAL_INTERFACE_ID);
		for (int lines : QuestConstants.COLLECTION_OF_QUEST_LINES)
			this.getPlayer().getActionSender().sendQuest("", lines);
		switch (this.getStage()) {
		case 0:
			this.getPlayer()
					.getActionSender()
					.sendQuest(this.toString(),
							QuestConstants.QUEST_TITLE_FRAME_ID);
			this.getPlayer()
					.getActionSender()
					.sendQuest(questJournalFrames()[0],
							QuestConstants.COLLECTION_OF_QUEST_LINES[0]);
			break;
		case 1:
			this.getPlayer()
					.getActionSender()
					.sendQuest(this.toString(),
							QuestConstants.QUEST_TITLE_FRAME_ID);
			this.getPlayer()
					.getActionSender()
					.sendQuest(questJournalFrames()[0],
							QuestConstants.COLLECTION_OF_QUEST_LINES[0]);
			break;
		case 2:
			this.getPlayer()
					.getActionSender()
					.sendQuest(this.toString(),
							QuestConstants.QUEST_TITLE_FRAME_ID);
			this.getPlayer()
					.getActionSender()
					.sendQuest(questJournalFrames()[0],
							QuestConstants.COLLECTION_OF_QUEST_LINES[0]);
			break;
		}

	}

	@Override
	public QuestStage getProgress() {
		if (this.getStage() >= 1 && this.getStage() <= 2)
			return QuestStage.IN_PROGRESS;
		else if (this.getStage() == 3)
			return QuestStage.FINISHED;
		else
			return QuestStage.NOT_STARTED;
	}

	@Override
	public boolean hasRequirements() {
		if (this.getPlayer().getSkills().getSkillLevel()[Skill.getStrength()] < 10)
			return false;
		return true;

	}

	public int getDragonKills() {
		return dragonKills;
	}

	public void setDragonKills(int dragonKills) {
		this.dragonKills = dragonKills;
	}
}

Thirdly, you will need to implement the quest constant class which is used to store important variables for each quest such as the ID and the position of the specific quest on the quest array.

package net.mikuframework.rs2.ltsmith.model.quest;

 * Important quest variables

public class QuestConstants {

	public static final int QUEST_JOURNAL_INTERFACE_ID = 8134;
	public static final int QUEST_TITLE_FRAME_ID = 8144;
	public static final int[] COLLECTION_OF_QUEST_LINES = { 8147, 8148, 8149,
			8150, 8151, 8152, 8153, 8154 };
	public static final int DRAGON_LOLZ_POSITION = 0;
	public static final int DRAGON_LOLZ_ID = 7332;

}

To rewrite the quest interface list frames (the id of a quest should always be the frame id of a quest line on the interface)

	public void rewriteQuestListInterface() {
		for (Quest allQuests : quests)
			if (allQuests.getProgress() == QuestStage.NOT_STARTED)
				this.getActionSender().sendQuest(
						"@red@" + allQuests.toString(), allQuests.getId());
			else if (allQuests.getProgress() == QuestStage.IN_PROGRESS)
				this.getActionSender().sendQuest(
						"@yel@" + allQuests.toString(), allQuests.getId());
			else if (allQuests.getProgress() == QuestStage.FINISHED)
				this.getActionSender().sendQuest(
						"@gre@" + allQuests.toString(), allQuests.getId());
	}
