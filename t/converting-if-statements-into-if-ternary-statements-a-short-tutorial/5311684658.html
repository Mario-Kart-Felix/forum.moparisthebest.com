[code=java]
import java.util.*;

public class Ternary {

/**
 * Ternary If's!
 */

public static void main(String[] args) {
	//Constructer for Scanner Class & Random Class
	Scanner i = new Scanner(System.in);
	Random r = new Random();
	
	//Welcome Message
	System.out.println("WELCOME TO MAGIC 8 BALL!\n");
	
	//Declared Variables
	int randomAnswer;
	String question;
	boolean keepRunning = false;
	
	//Loops While Variable KeepRunning = false
	do {
	//Output/Input Text
	System.out.printf("%-40s","\nEnter a question (Q to quit):");
	question = i.nextLine();
	
	//Checks Question Value To See If Returned To 'Y'
	if (question.equalsIgnoreCase("y")) {
		//Exits Program
		System.exit(0);
	}
	//Generate Random Number
	randomAnswer = r.nextInt(12) + 1;
	
	//Ternary If StateMent Start
	//Note: ? = (If Constructor), : = (Else/Else If Constructor)
	System.out.println(randomAnswer == 1 ? "Outlook not so good!" 
			: randomAnswer == 2 ? "Reply hazy, try again!"
			: randomAnswer == 3 ? "Most likely"
			: randomAnswer == 4 ? "Don’t count on it!"  
			: randomAnswer == 5 ? "Yes definitely!"
			: randomAnswer == 6 ? "You can count on it!"
		    : randomAnswer == 7 ? "My sources say no!"
			: randomAnswer == 8 ? "My reply is no!"
			: randomAnswer == 9 ? "Outlook good!"
			: randomAnswer == 10 ? "Without a doubt!"
			: randomAnswer == 11 ? "As I see it, yes!"
			: randomAnswer == 12 ? "Very doubtful!"
			: "ERROR RETRY!");
	//End Of Ternary Statement
	}
	while (keepRunning  = true);

}

}
