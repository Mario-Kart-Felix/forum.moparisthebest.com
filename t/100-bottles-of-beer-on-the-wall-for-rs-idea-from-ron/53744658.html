program New; var i,waitT: integer; liquid: string;

Procedure SendThis(Msg : String);
Begin
SendKeys(Msg);
SendKeys(Chr(13));
End;

begin
activateclient;
cleardebug;
liquid := ‘Gin An Juice’ //what do you want to drink? beer? milk? water? booze?
i := 100 //bottles of (liquid)
waitT := 5 //how many seconds between messages?
cleardebug;
sendthis(‘Lets Par-Tay’)
waitT := waitT*1000
sendthis(‘There are ’ + inttostr(i+1) + ’ bottles of ’ + (liquid) + ’ on the wall.’)
wait(waitT)
sendthis(‘Take one down, pass it around,’)
wait(waitT)
sendthis(inttostr(i) + ’ Bottles of ’ + (liquid) + ’ on the wall!’)
wait(waitT)
repeat
sendthis(‘Take one down, pass it around,’)
wait(waitT)
sendthis(inttostr(i-1) + ’ Bottles of ’ + (liquid) + ’ on the wall!’)
i := i-1
wait(waitT)
until(i=2)
begin
sendthis(‘Take one down, pass it around,’)
wait(waitT)
sendthis(‘1 more bottle of ’ + (liquid) + ’ on the wall!’)
wait(waitT)
sendthis(‘Take it down, pass it around,’)
wait(waitT)
sendthis(‘There are no more bottles of ’ + (liquid) + ’ on the wall!’)
end;
end.
