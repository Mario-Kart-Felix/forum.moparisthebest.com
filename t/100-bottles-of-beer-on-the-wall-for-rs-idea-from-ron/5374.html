[scar]program New;
{.include srl/srl.scar}
var
iTemp,WaitTemp,liquid: string;
i,waitT: integer;

procedure TypeText(text: string);
var
i: Integer;
begin
for i := 1 to Length(Text) do
begin
TypeByte(GetKeyCode(Text[i]));
Wait(50 + Random(10));
end;
end;

begin
setupsrl;
activateclient;
cleardebug;
liquid := readln(‘What do you want to drink? Beer? Booze? Milk? Water?’);
iTemp := readln(‘How many bottles do you want to drink?’);
waitTemp := readln(‘How long do you want to wait between messages (in seconds)? (Reccommended to be 1)’);
cleardebug;
begin
i := strtoint(itemp);
waitt := strtoint(waittemp);
end;
waitT := waitT*1000;
TypeSend(‘There are ’ + inttostr(i) + ’ bottles of ’ + (liquid) + ’ on the wall.’)
wait(waitT)
TypeSend(‘Take one down, pass it around,’)
wait(waitT)
TypeText(inttostr(i-1) + ’ Bottles of ’ + (liquid) + ’ on the wall’)
keydown(16)
typesend(’!’)
keyup(16)
wait(waitT)
repeat
TypeSend(‘Take one down, pass it around,’)
wait(waitT)

TypeSend(‘Take one down, pass it around,’)

keyup(16)
wait(waitT)

wait(waitT)

end.[/scar]
