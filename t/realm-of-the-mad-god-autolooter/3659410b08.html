[code]program ROTMGAutoGrabber;
Type
ROTMGItem = record
Autograb, Image, Tier : Integer;
Name: string;
end;
Type
ROTMGSlot = record
Image, Slot: Integer;
end;
Type
ROTMGItemList = record
List : Array[0…500] of ROTMGItem;
Length : Integer;
end;
Type
ROTMGSlotList = record
List : Array[0…7] of ROTMGSlot;
end;
var
ItemList : ROTMGItemList;
SlotList : ROTMGSlotList;
itemsGrabbed : Integer;
itemsMissed : Integer;
ii : Integer;
procedure LoadSlots;
var
i : integer;
begin
for i := 1 to 8 do
begin
SlotList.List[i-1].Slot := i;
SlotList.List[i-1].Image := LoadBitmap(IncludePath + ‘ROTMG\Slots’ + ‘slot’ + inttostr(i) + ‘.png’);
end;
end;

procedure EnableMasks(image : Integer);
var
a, b, w, h : Integer;
begin
GetBitmapSize(image, w, h);
for a := 0 to h-1 do
begin
if (a < 2) or (h - a - 1 < 2) then
begin
for b := 0 to w-1 do
begin
FastSetPixel(image,b,a, RGBtoColor(0,0,0));
end;
end else
begin
FastSetPixel(image,0,a,RGBtoColor(0,0,0));
FastSetPixel(image,1,a,RGBtoColor(0,0,0));
FastSetPixel(image,w-1,a,RGBtoColor(0,0,0));
FastSetPixel(image,w-2,a,RGBtoColor(0,0,0));
end;
end;
end;

function LoadItem(fileName : string) : boolean;
begin
if Strtoint(ReadIni(fileName,‘Autograb’,IncludePath + ‘ROTMG\Config.ini’)) > 0 then
begin
ItemList.List[ItemList.Length].Image := LoadBitmap(IncludePath + ‘ROTMG\Equipment’ + fileName);
ItemList.List[ItemList.Length].Name := Readini(fileName,‘Name’,IncludePath + ‘ROTMG\Config.ini’);
ItemList.List[ItemList.Length].Autograb := strtoint(Readini(fileName,‘Autograb’,IncludePath + ‘ROTMG\Config.ini’));
ItemList.List[ItemList.Length].Tier := strtoint(Readini(fileName,‘Tier’,IncludePath + ‘ROTMG\Config.ini’));
EnableMasks(ItemList.List[ItemList.Length].Image);
result := true;
exit;
end;
result := false;
end;

function LoadItems() : Integer;
var
x, itemCount, loaded : Integer;
fileNames : TStringArray;
begin
fileNames:= GetFiles(IncludePath + ‘ROTMG\Equipment’,‘png’);
itemCount:= length(fileNames);
for x:= 0 to itemCount-1 do
begin
if LoadItem(fileNames[x]) then
begin
inc(loaded);
inc(ItemList.Length);
end;
end;
result := loaded;
end;
procedure GrabItem(item : ROTMGItem; x, y : Integer);
var
i,a,b,w,h,c,d,n,m : Integer;
begin
for i := 0 to 7 do
begin
if item.Autograb = 2 then
begin
GetMousePos(n,m);
GetBitmapSize(item.Image,w,h);
MoveMouse(x + w / 2, y + h / 2);
KeyDown(16);
ClickMouse(x + w / 2, y + h / 2, mouse_left);
KeyUp(16);
MoveMouse(n,m);
inc(itemsGrabbed);
Writeln(‘Items grabbed: (’ + inttostr(itemsGrabbed) + ‘) Items Missed: (’ + inttostr(itemsMissed) + ‘)’);
Writeln(‘Grabbed item "’ + item.Name + ‘"’);
Wait(10);
exit;
end else
if FindBitmapToleranceIn(SlotList.List[i].Image,a,b,610,400,800,600,25) then
begin
GetBitmapSize(SlotList.List[i].Image,c,d);
GetBitmapSize(item.Image,w,h);
GetMousePos(n,m);
MoveMouse(x + w / 2, y + h / 2);
HoldMouse(x + w / 2,y + h / 2,mouse_Left);
MoveMouse(a + c / 2,b + d / 2);
ReleaseMouse(a + c / 2,b + d / 2,mouse_left);
MoveMouse(n,m);
inc(itemsGrabbed);
Writeln(‘Items grabbed: (’ + inttostr(itemsGrabbed) + ‘) Items Missed: (’ + inttostr(itemsMissed) + ‘)’);
Writeln(‘Grabbed item "’ + item.Name + ‘"’);
Wait(10);
exit;
end;
end;
inc(itemsMissed);
Writeln(‘Items grabbed: (’ + inttostr(itemsGrabbed) + ‘) Items Missed: (’ + inttostr(itemsMissed) + ‘)’);
Writeln(‘Missed item "’ + item.Name + ‘" (INVENTORY FULL)’);
end;

procedure ScanItem (item : ROTMGItem);
var
x,y : integer;
begin
if FindBitmapToleranceIn(item.Image,x,y,610,500,800,600,30) then
begin
Writeln(‘found item:’ + item.Name )
GrabItem(item,x,y);
end;
end;
procedure Initialize;
begin
ItemList.Length := 0;
itemsGrabbed := 0;
itemsMissed := 0;
Writeln(‘Initializing autograbber’);
Writeln(‘Loading Items…’);
Writeln(‘Loaded ’ + inttostr(LoadItems()) + ’ items’);
Writeln(‘Loading slots…’);
LoadSlots();
Writeln(‘Slots loaded’);
Writeln(‘Entering main loop…’);
end;
begin
Initialize();
while true do
begin
for ii := 0 to ItemList.Length do
begin
ScanItem(ItemList.List[ii]);

deaner August 3, 2016, 11:52pm #171

The error at line 62 problem is caused because you are missing the config.ini entry for a picture that is in your "includes\rotmg\equipment’ folder. Make sure all your config entries and .png files line up. There could also be a slight error or difference in filenames between your equipment folder and the config file. This will cause the same problem.

Search carefully… there is either an extra picture in your equipment folder or a missing/invalid config.ini entry.

If it comes right down to it just delete the equipment folder and config file and download them again. The config file that came with the original script is flawed, there is a working one a few posts back, use it instead.

Also: How make the whole script much faster and more reliable:

I replaced lines 105-107:


HoldMouse(x + w / 2,y + h / 2,mouse_Left);
MoveMouse(a + c / 2,b + d / 2);
ReleaseMouse(a + c / 2,b + d / 2,mouse_left);

with:


ClickMouse(x + w / 2,y + h / 2,mouse_Left);
Wait(10)
ClickMouse(x + w / 2,y + h / 2,mouse_left);

I’m not sure why it wasn’t written this way originally other than that maybe the double-click item pickup was a recent ‘innovation’ in one of the latest ROTMG builds??

Other Issues:

SCREEN MAGNIFICATION!!! (!!!) I wish I could capitalize exclamation marks… The ROTMG client MUST BE at 800 x 600 RES for the script to work. If the client is too small you will get a

type message in the debug window. If the client is too big (zoomed in) the script will run error free… but won’t pick anything up. This is because it is searching the wrong part of the client for your inventory slots and items.

To find out what your client resolution is open up simba in such a way that you can see both the bottom right corner of the ROTMG client AND the bottom left corner of the Simba window. Now left click and drag the green crosshair button (located near the right side of the Simba window toolbar, says, “select a client” when you hold your mouse pointer over it) to your ROTMG client and release. NOTICE THE GIANT RED SQUARE. With Simba as the active window, place the mouse pointer over the bottom right corner of the ROTMG client (Bottom right corner of the giant red square). DONT CLICK. Just look at the (x,y) coordinates on the lower left portion of the Simba window. It should read close to, if not exactly (800,600). If it is bigger than that you need to zoom out, smaller and you need to zoom in.

Quality!!! Some people may have issues because they changed the quality of the graphics in the flash settings (right click on ROTMG client… ‘Quality -> Low, Medium, High’) and the script no longer recognizes your empty inventory slots. This can return an “Item Missed - Inventory Full” message. I have lag issues and run my client at the low quality setting. To compensate for this I used the ‘snipping tool’ (Start Menu->All Programs->Accessories->Snipping Tool) and replaced the .png image files in the …ROTMG\Slots folder with samples that I snipped from my own ROTMG client. Just get the number and a little grey area around it. Less is more. The quality doesn’t appear to affect item recognition; the original image files work fine for me at any quality setting.

Adding new items!!! Use the snipping tool to grab .png files of items you want to be able to pick up, but can’t - like the RED SHADED items. Unfortunately the images from the equipment wiki dont seem to work so you need to snip the image from your own inventory. Then make a new entry in your config.ini file using the existing format for items.

THE NEW IMAGE FILENAME MUST EXACTLY MATCH THE CORRESPONDING NEW CONFIG ENTRY

Or say hello to line 62 errors…

DO NOT ENTER Tier=UT FOR UT ITEMS. IT MUST BE A NUMBER!!! NOT A LETTER!!!

I’m not sure what the purpose of the tier entry is… possibly to set pickup priority?? Haven’t tested that out.

I’m tired and I can’t think of anything else…

Other than that sometimes Simba is just a little whiney bitch and closing it and your browser and reolading both of them might solve your problems where nothing else will.

Oh yeah… for the love of all that’s holy once it’s up and running and working properly right click the script tab and set the script to “Read Only”. Numerous headaches will be averted by this simple procedure.
