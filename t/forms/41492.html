Weibs July 30, 2016, 3:19am #4

You could try using the Hide and Show Form properties, here’s an example script:

[code]program New;

var
frmDesign : TForm;
Button1 : TButton;
Button2 : TButton;
T: TTimer;
Hidden: Boolean;
x, y: Integer;

procedure Close(Sender: TObject);
begin
frmDesign.ModalResult := mrOk;
WriteLn(‘Ended’);
WriteLn(’’);
end;

procedure Hide(Sender: TObject);
begin
frmDesign.Hide;
Hidden := True;
WriteLn(‘Form is Hidden’);
WriteLn(’’);
end;

procedure UnHide(Sender: TObject);
begin
if IsFKeyDown(12) then
begin
if Hidden then
begin
frmDesign.Show;
Hidden := False;
WriteLn(‘Form is unhidden’);
WriteLn(’’);
end;
end;
end;

procedure Init;
begin
frmDesign := CreateForm;
frmDesign.Left := 250;
frmDesign.Top := 114;
frmDesign.Width := 164;
frmDesign.Height := 180;
frmDesign.Caption := ‘frmDesign’;
frmDesign.Color := clBtnFace;
frmDesign.Font.Color := clWindowText;
frmDesign.Font.Height := -11;
frmDesign.Font.Name := ‘MS Sans Serif’;
frmDesign.Font.Style := [];
frmDesign.Visible := False;
frmDesign.PixelsPerInch := 96;
Button1 := TButton.Create(frmDesign);
Button1.Parent := frmDesign;
Button1.OnClick := @Hide;
Button1.Left := 41;
Button1.Top := 32;
Button1.Width := 75;
Button1.Height := 25;
Button1.Caption := ‘Hide’;
Button1.TabOrder := 8;
Button2 := TButton.Create(frmDesign);
Button2.Parent := frmDesign;
Button2.OnClick := @Close;
Button2.Left := 41;
Button2.Top := 82;
Button2.Width := 75;
Button2.Height := 25;
Button2.Caption := ‘End’;
Button2.TabOrder := 8;
T := TTimer.Create(frmDesign);
T.Interval := 10;
T.OnTimer := @UnHide;
end;

procedure SafeInit;
var
v: TVariantArray;

begin
SetArrayLength(v, 0);
ThreadSafeCall(‘Init’, v);
end;

procedure ShowFormModal;
begin
frmDesign.ShowModal;
end;

procedure SafeShowModal;
var
v: TVariantArray;

begin
SetArrayLength(v, 0);
ThreadSafeCall(‘ShowFormModal’, v);
end;

begin
ClearDebug;
SafeInit;
SafeShowModal;
end.[/code]

Run the script, press the Hide button to hide the form, then press F12 to show it, and End to end the script.
