		player.addTemporary("COORDINATE_EVENT", new CoordinateEvent(location, 1) {

private static void woodcutting(final Player player, final int objectId, final Location location, final int level, final int log, final double xp) {
	if (level > player.getSkills().getLevel(8)) {
		player.getActionSender().sendMessage("You need a Woodcutting level of " + level + " to chop down this tree.");
		return;
	}
	if (player.getInventory().freeSlots() == 0) {
		player.getActionSender().sendMessage("Your inventory is too full to hold anymore "+ItemDefinition.forId(log).getName().toLowerCase()+".");
		player.getActionSender().sendSound(2277, 1, 0);
		return;
	}
	if (checkAxe(player) == -1) {
		player.getActionSender().sendMessage("You do not have a axe which you have the woodcutting level to use.");
		return;
	}
	final GameObject object = RegionManager.getObject(player, objectId, location);
	if (object == null) {
		return;
	}
	final GameObject tree = new GameObject(getId(objectId), location, object.getType(), object.getRotation());
	player.playAnimation(Animation.create(checkAxeAnimation(player)));
	player.getActionSender().sendMessage("You swing your axe at the tree.");
	ActionEvent event = new ActionEvent() {
		
		private Event event;

		@Override
		public void run() {
			event = new Event(1500) {
				
				private int delay = 0;

				@Override
				public void execute() {
					if (RegionManager.getRegionByLocation(location).getRemovedObjects().contains(tree)) {
						player.playAnimation(Animation.create(-1, 0));
						this.stop();
						return;
					}
					if (player.getInventory().freeSlots() == 0) {
						player.getActionSender().sendMessage("Your inventory is too full to hold any more "+ItemDefinition.forId(log).getName().toLowerCase()+".");
						player.getActionSender().sendSound(2277, 1, 0);
						this.stop();
						return;
					}	
					if (delay >= 1) {
						player.playAnimation(Animation.create(checkAxeAnimation(player)));
						delay = 0;
					} else {
						delay++;
					}
					final double formula = (((player.getSkills().getLevel(Skills.WOODCUTTING) * .25) - (level * 2.0)) + ((checkAxe(player) + 19/* safety */) * .15) + (level + player.getSkills().getLevel(Skills.WOODCUTTING) / 1.8) + 30); // 15% of axe level + 19. (This makes it even)
					if ((formula / 100) > Math.random()) {
						player.playAnimation(Animation.create(-1));
						player.getInventory().add(new Item(log));
						player.getActionSender().sendMessage("You get some "+ItemDefinition.forId(log).getName().toLowerCase()+".");
						player.getSkills().addExperience(8, xp);
						if (getChance(objectId) >= new Random().nextInt(100)) {
							player.playAnimation(Animation.create(-1, 0));
							RegionManager.removeObject(player, object);
							RegionManager.addObject(player, tree);
							World.getWorld().submit(new Event(getTimer(objectId)) {
								@Override
								public void execute() {
									RegionManager.removeObject(player, tree);
									RegionManager.addObject(player, object);
									this.stop();
								}
							});
							this.stop();
						}
					}
				}					
			};
			World.getWorld().submit(event);
		}

		@Override
		public void stop() {
			player.playAnimation(Animation.create(-1));
			event.stop();
		}
		
	};		
	player.addTemporary("ACTION_EVENT", event);
	event.run();
}
