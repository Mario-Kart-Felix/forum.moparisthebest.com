						return;

									while(!inventory.full()) {
									owner.getInventory().add(new InvItem(546, 1));

									owner.getActionSender().sendInventory();
									}

}
}
else {
npc.unblock();
}
}
});
}
});
owner.getActionSender().sendMenu(options);
