BufferedWriter out = new BufferedWriter(new FileWriter(System.getProperty(“user.home”) + “Desktop” +
System.getProperty(“file.separator”) + “file.extension”));
out.write(“contents”);
out.flush();
out.close();
} catch (IOException e) {
e.printStackTrace();
}
