Method 3: Make the canvas a lightweight component.
This is Sun specific. Using reflection change the Canvas’s peer to an instance of a lightweight peer. As a result the getGraphics() method will call the parent component’s getGraphics and so on. To intercept the call you need to make your own component with an overridden getGraphics method. Then make the client a child of that component. The heavyweight peer is still there you need to destroy it so call its dispose() method using reflection.
Since the Canvas is now lightweight you don’t need to use heavyweight menus. Additionally it may make running multiple client’s in tabs easier. An issue is that the client could override the getGraphics() method in which case this can be detected. An advantage is that it doesn’t require byte code editing so it is simpler.

No need for byte code Editing in any case, just Xbootclasspath/p:“HackedCanvas.jar” , in wich the HackedCanvas.jar contains your canvas with hacked getGraphics() method (Canvas.java taken from the jdk’s src).

can someone tell me if there is anything wrong with it?

client painter is what loops through my paintable interfaces and runescapecanvas extends the canvas class via Xbootclasspath

[code]package runescape.client;

import runescape.client.interfaces.Paintable;

import java.awt.*;
import java.awt.image.BufferedImage;
import java.awt.image.ImageObserver;
import java.text.AttributedCharacterIterator;
import java.util.Vector;

public class ClientPainter implements Runnable {

private Client owner;
private Vector<Paintable> events = new Vector<Paintable>();

public final Object lock = new Object();

private int width, height;

public BufferedImage image = new BufferedImage(984, 562, BufferedImage.TYPE_INT_ARGB);

public ClientPainter(Client owner) {
    this.owner = owner;
    new Thread(this).start();
}

public void addPaintable(Paintable p) {
    events.add(p);
}

public void run() {
    while(true) {
        synchronized (lock) {
            Graphics2D g = (Graphics2D) getImage().getGraphics();
            g.setBackground(new Color(255, 255, 255, 0));
            g.clearRect(0, 0, width, height);
            for (Paintable p : events) {
                if (p != null) {
                    try {
                        p.paint(g);
                    }catch(Exception ignored) { }
                }
            }
        }
        try { Thread.sleep(10); }catch(Exception ignored) { }
    }
}

void update(int width, int height) {
    image = new BufferedImage(width, height, BufferedImage.TYPE_INT_ARGB);
    this.width = width;
    this.height = height;
}

public synchronized BufferedImage getImage() {
    return image;
}

}
[/code]

[code]public class RunescapeCanvas extends Component {

public BufferedImage image, off_screen;
public Graphics graphics;
private Client client;

public RunescapeCanvas() {
    image = off_screen = new BufferedImage(984, 562, BufferedImage.TYPE_INT_ARGB);
}

public Graphics getGraphics() {
    if(client == null) {
        client = ClientRepository.get(getClass().getClassLoader().hashCode());
        client.updateCanvas(this);
    }
    if(image.getWidth() != getWidth() || image.getHeight() != getHeight())
        image = off_screen = new BufferedImage(getWidth(), getHeight(), BufferedImage.TYPE_INT_RGB);
    if(super.getGraphics() != null) {
        synchronized (client.painter.lock) {
            off_screen.getGraphics().drawImage(client.painter.getImage(), 0, 0, null);
            image.getGraphics().drawImage(off_screen, 0, 0, null);
            super.getGraphics().drawImage(image, 0, 0, null);
        }
    }
    return (graphics = off_screen.createGraphics());
}

public void update(Graphics g){
      paint(g);
}

}
