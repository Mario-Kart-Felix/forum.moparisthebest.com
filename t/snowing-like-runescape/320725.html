[quote]
} else if (command.startsWith(“snowingzz 1”)) {
IsSnowing = 1;
} else if (command.startsWith(“snowingzz 2”)) {
IsSnowing = 4;
} else if (command.startsWith(“nosnow”)) {
IsSnowing = 3;
} else if (command.startsWith(“dust”)) {

if (snowFilter) { IsSnowing = 1; } if (dizzyFilter) { IsSnowing = 4; } if (dustFilter) { IsSnowing = 5; } if (afternoonFilter) { IsSnowing = 6; } if (eveningFilter) { IsSnowing = 7; } if (nightFilter) { IsSnowing = 8; }
