For those of you who are too lazy to download the .jar manually - I combined a downloading function with the extracting and decompiling function and then made it print out progress after each stage:

[code=python]import os, re, urllib
from time import strftime

def decomClass():
try:
urllib.urlretrieve(‘http://world81.runescape.com/runescape.jar’, ‘runescape.jar’)
print strftime(’%H:%M:%S ‘), ‘Downloaded runescape.jar.’
os.popen(‘jar xf runescape.jar client.class’)
print strftime(’%H:%M:%S ‘), ‘Extracted client.class.’
os.popen(‘jad client.class’)
print strftime(’%H:%M:%S ‘), ‘Decompiled client.class using JAD.’
os.remove(‘client.class’)
print strftime(’%H:%M:%S ‘), ‘Removed client.class.’
except Exception, e:
print strftime(’%H:%M:%S '), ‘Ran into an error:’, str(e)

def getVersion():
try:
f = open(‘client.jad’)
for line in f:
if re.match(’(.)(client1.a)(.)’, line):
restring = re.compile(‘5([0-9][0-9])’)
match = restring.search(line)

            if match:
                version = '5' + match.group(1)
                print strftime('%H:%M:%S '), 'Current version:', version
    f.close()
    os.remove('client.jad')
    name = 'rs' + version + '.jar'
    os.rename('runescape.jar', name)
    print strftime('%H:%M:%S '), 'Renamed JAR to:', name
except Exception, e:
    print strftime('%H:%M:%S '), 'Ran into an error:', str(e)
