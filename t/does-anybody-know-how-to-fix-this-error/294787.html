// Writes recieved data to a specified file
// Makes use of the java.io functions, Buffered Writer, etc
void report(String msg) {

	BufferedWriter bw = null;

	try {
		bw = new BufferedWriter(new FileWriter("./reports/Reports.txt",
				true));
		bw.write(msg + " | On " + date + " At " + time);
		bw.newLine();
		bw.newLine();
		bw.flush();
	} catch (IOException ioe) {
		ioe.printStackTrace();
	} finally {
		if (bw != null)
			try {
				bw.close();
			} catch (IOException ioe2) {
				misc.println("Error saving server reports..");
			}
	}
}
