public boolean addSkillXP(int amount, int skill){ if (amount+c.playerXP[skill] < 0 || c.playerXP[skill] > 200000000) { if(c.playerXP[skill] > 200000000) { c.playerXP[skill] = 200000000; } return false; } if(playerRights >= 1) { amount *= Config.SERVER_EXP_BONUS; } else { int oldLevel = getLevelForXP(c.playerXP[skill]); c.playerXP[skill] += amount; if (oldLevel < getLevelForXP(c.playerXP[skill])) { if (c.playerLevel[skill] < c.getLevelForXP(c.playerXP[skill]) && skill != 3 && skill != 5) c.playerLevel[skill] = c.getLevelForXP(c.playerXP[skill]); levelUp(skill); c.gfx100(199); requestUpdates(); } setSkillLevel(skill, c.playerLevel[skill], c.playerXP[skill]); refreshSkill(skill); return true; } }
