[code=java5]import java.awt.Rectangle;
import java.awt.event.*;

public class Player extends Rectangle {

int dx, dy;
String dir;

public Player(int x, int y, int width, int height) {

	this.x = x;
	this.y = y;
	this.width = width;
	this.height = height;

	dir = "left";

}

public void move() {

	x += dx;
	y += dy;

}

public void keyPressed(KeyEvent e) {

	int key = e.getKeyCode();

	switch(key) {

		case KeyEvent.VK_UP:
			dy = -3;
			dir = "up";
			break;
		case KeyEvent.VK_DOWN:
			dy = 3;
			dir = "down";
			break;
		case KeyEvent.VK_RIGHT:
			dx = 3;
			dir = "right";
			break;
		case KeyEvent.VK_LEFT:
			dx = -3;
			dir = "left";
			break;
		case KeyEvent.VK_SPACE:
			Game.bullets.add(new Bullet(x, y, 10, 5, dir));
			break;

	}

}

public void keyReleased(KeyEvent e) {

	int key = e.getKeyCode();

	switch(key) {

		case KeyEvent.VK_UP:
			dy = 0;
			break;

		case KeyEvent.VK_DOWN:
			dy = 0;
			break;
		case KeyEvent.VK_RIGHT:
			dx = 0;
			break;
		case KeyEvent.VK_LEFT:
			dx = 0;
			break;

	}

}

[code=java5]import java.awt.Rectangle;

public class Bullet extends Rectangle {

int dy,dx;

String dir;

public Bullet(int x, int y, int width, int height, String lastDir) {

	this.x = x;
	this.y = y;
	this.width = width;
	this.height = height;

	dir = lastDir;

}

public void move() {

	if (dir.equalsIgnoreCase("up")) {

		dy = -10;
		y += dy;

	} else if (dir.equalsIgnoreCase("down")) {

		dy = 10;
		y += dy;

	} else if (dir.equalsIgnoreCase("left")) {

		dx = -10;
		x += dx;

	} else if (dir.equalsIgnoreCase("right")) {

		dx = 10;
		x += dx;

	}


}

[code=java5]import java.awt.Image;
import javax.swing.ImageIcon;
import java.awt.Rectangle;

public class Flower {

static Image img;
static Image blue;
static Image black;
static Image orange;
static Image purple;
static Image red;

int x, y, dx, dy;
boolean gameOver = false;

public Flower(int type, int x, int y) {

	blue = new ImageIcon("H:/Programming/Game/blue.png").getImage();
	black = new ImageIcon("H:/Programming/Game/black.png").getImage();
	orange = new ImageIcon("H:/Programming/Game/orange.png").getImage();
	purple = new ImageIcon("H:/Programming/Game/purple.png").getImage();
	red = new ImageIcon("H:/Programming/Game/red.png").getImage();

	switch(type) {

		case 1:
			img = blue;
			break;
		case 2:
			img = black;
			break;
		case 3:
			img = orange;
			break;
		case 4:
			img = purple;
			break;
		case 5:
			img = red;
			break;

	}

	this.x = x;
	this.y = y;


}

public Image getImage() {

	return img;

}

public void move(int x, int y) {

	if(x > this.x) {

		dx = 1;

	} else {

		dx = -1;

	}

	if(y > this.y) {

		dy = 1;

	} else {

		dy = -1;

	}

	this.x += dx;
	this.y += dy;

}

public int getX() {

	return x;

}

public int getY() {

	return y;

}

public void testCollision(Rectangle r, String type) {

	Rectangle test = new Rectangle(x,y,25,25);

	if(r.intersects(test) && (type.equals("bullet"))) {

		Game.removeEnemy(this);
		Game.score += 10;

	}

	if(r.intersects(test) && (type.equals("player"))) {


		gameOver = true;

	}		

}

public boolean getGameOver() {

	return gameOver;

}
