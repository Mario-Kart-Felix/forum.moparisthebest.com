program CowFighter;
{.include SRL/SRL.scar}
  var x,y: Integer;
Procedure FindCows;
 begin
  if not(findColor(x,y, 8948106, 6, 67, 503, 351))and 
  not(findColorTolerance(x,y, 9900, 11, 61, 169, 131, 30))then
   begin
    repeat
     KeyDown(VK_LEFT);
     Wait(50+random(50));
     KeyUp(VK_LEFT);
    until (findColor(x, y, 8948106, 6, 67, 503, 351))
   end;
 end;
 
Procedure AttackCow;
 begin
  if(findColor(x,y, 8948106, 6, 147, 503, 351))or
  (findColor(x,y, 8948106, 191, 64, 503, 351))then
   begin
    Mouse(x, y, 5, 5, true);
   end;
 end;



begin
SetupSRL;
repeat
Wait(100+random(100));
FindCows;
AttackCow;
until false;
end.
