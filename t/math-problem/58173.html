x - z = 7, x + y = 3, and z - y = 6

x = 7 + z
z = 6 + y
y = 3 - x

x = 7 + (6 + (3 - x))
x = 16 - x
x = 16/2
x = 8.

put x into eqn

y = 3 - 8
y = -5.

put y into eqn

z = 6 + (-5)
z = 1.
