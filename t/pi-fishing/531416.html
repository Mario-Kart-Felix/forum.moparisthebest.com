[code=java]package server.model.players.skills.impl;

import java.util.ArrayList;

import server.event.CycleEvent;
import server.event.CycleEventContainer;
import server.event.CycleEventHandler;
import server.model.players.Client;
import server.util.Misc;

public class Fishing {

private int[] NPCID = { 233, 234 };
ArrayList<Integer> npcid = new ArrayList<Integer>();

private int[] LEVELREQ = { 1, 10 };
ArrayList<Integer> levelreq = new ArrayList<Integer>();

private int[] FISHGIVEN = { 317, 333 };
ArrayList<Integer> fishgiven = new ArrayList<Integer>();

private int[] EXPGIVEN = { 10, 50 };
ArrayList<Integer> expgiven = new ArrayList<Integer>();

public void init(Client c, int id, int lvl) {
	loadData();

	if (lvl >= levelreq.get(npcid.indexOf(id))) {
		start(c, id);
	} else {
		c.sendMessage("You need a fishing level of at least "
				+ levelreq.get(npcid.indexOf(id)));
	}

}

private void start(final Client c, final int id) {

	CycleEventHandler.getSingleton().addEvent(this, new CycleEvent() {

		public void execute(CycleEventContainer container) {

			if (c.getItems().freeSlots() <= 0 || c.isMoving || !c.isActive) {
				c.sendMessage("STOP STOP SOP");
				container.stop();
			}

			if (c.fishingTimer > 0) {
				c.fishingTimer--;
				c.sendMessage("" + c.fishingTimer);
			}
			if (c.fishingTimer <= 0) {
				c.getItems().addItem(fishgiven.get(npcid.indexOf(id)), 1);
				c.fishingTimer = Misc.random(10);
			}
		}

		@Override
		public void stop() {
			c.startAnimation(-1);
		}
	}, 1);
}

private void loadData() {
	if (npcid.size() != NPCID.length) {
		for (int i = 0; i < NPCID.length; i++) {
			npcid.add(NPCID[i]);
			levelreq.add(LEVELREQ[i]);
			fishgiven.add(FISHGIVEN[i]);
			expgiven.add(EXPGIVEN[i]);
		}
	}

}

}
