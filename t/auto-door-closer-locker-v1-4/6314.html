program DoorLockerV1;

{Door Locker V1.4 By Zach1188____}
{}
{Instructions: }
{1. Drag crosshairs onto RS (REQUIRED) }
{2. Find an open door, and close it once (manually). }
{3. Wait for someone to open the door again. }
{4. Place your mouse over the door. }
{5. Press ctrl+alt+r (default start shortcut), or click the green }
{ arrow. }
{ }
{This script is still in the testing stage, or beta if you must. It }
{does not have anti-randoms, so if someone could add them for me I }
{will add them to the credits. Please do not modify any other }
{part of the script. }
{ }
{Credits: }
{- Zach1188 (Creator) }
{ }
{ }
{}

var
x2,y2,x,y,Cancel1Sec:integer;

Procedure IfDoorOpen;
begin
Cancel1Sec:=0;
repeat
wait(1)
Until(FindColorTolerance(x,y,13292049,35, 4,89, 24,35) or FindColorTolerance(x,y,12961540,35, 4,89, 24,35) or FindColorTolerance(x,y,13292050,35, 4,89, 24,35) or FindColorTolerance(x,y,12961540,35, 4,89, 24,35) or FindColorTolerance(x,y,12895236,35, 4,89, 24,35) or FindColorTolerance(x,y,13292050,35, 4,89, 24,35) or FindColorTolerance(x,y,15000611,35, 4,89, 24,35) or FindColorTolerance(x,y,13421577,35, 4,89, 24,35));

GetMousePos(x2,y2)
x2:=x2+1;
y2:=y2+1;
end;

Procedure CloseDoor;
begin
wait(random(5))
ClickMouse(x2,y2,true)
repeat
wait(1)
Cancel1Sec:=Cancel1Sec+1
If Cancel1Sec > 999 then IfDoorOpen
Until(FindColorTolerance(x,y,13292049,35, 4,89, 24,35) or FindColorTolerance(x,y,12961540,35, 4,89, 24,35) or FindColorTolerance(x,y,13292050,35, 4,89, 24,35) or FindColorTolerance(x,y,12961540,35, 4,89, 24,35) or FindColorTolerance(x,y,12895236,35, 4,89, 24,35) or FindColorTolerance(x,y,13292050,35, 4,89, 24,35) or FindColorTolerance(x,y,15000611,35, 4,89, 24,35) or FindColorTolerance(x,y,13421577,35, 4,89, 24,35)=false);
end;

begin
ActivateClient;
repeat
IfDoorOpen;
CloseDoor;
until(false)
end.
