{4. Place your mouse over the door. }
{5. Press ctrl+alt+r (default start shortcut), or click the green }
{ arrow. }
{ }
{This script is still in the testing stage, or beta if you must. It }

{ }

{.include srl/srl.scar}
var
x2,y2,x,y,Cancel1Sec:integer;
const
Nick = ‘’;//your nick ie. 3-4 letters of username

Procedure IfDoorOpen;
begin
Cancel1Sec:=0;
repeat
wait(1)
Until(FindColorTolerance(x,y,13292049,35, 4,89, 24,35) or
FindColorTolerance(x,y,12961540,35, 4,89, 24,35) or
FindColorTolerance(x,y,13292050,35, 4,89, 24,35) or
FindColorTolerance(x,y,12961540,35, 4,89, 24,35) or
FindColorTolerance(x,y,12895236,35, 4,89, 24,35) or
FindColorTolerance(x,y,13292050,35, 4,89, 24,35) or
FindColorTolerance(x,y,15000611,35, 4,89, 24,35) or
FindColorTolerance(x,y,13421577,35, 4,89, 24,35));
GetMousePos(x2,y2)
x2:=x2+1;
y2:=y2+1;
end;

Procedure CloseDoor;
begin
wait(random(5))
ClickMouse(x2,y2,true)
repeat
wait(1)
Cancel1Sec:=Cancel1Sec+1
If Cancel1Sec > 999 then IfDoorOpen
Until(FindColorTolerance(x,y,13292049,35, 4,89, 24,35) or
FindColorTolerance(x,y,12961540,35, 4,89, 24,35) or
FindColorTolerance(x,y,13292050,35, 4,89, 24,35) or
FindColorTolerance(x,y,12961540,35, 4,89, 24,35) or
FindColorTolerance(x,y,12895236,35, 4,89, 24,35) or
FindColorTolerance(x,y,13292050,35, 4,89, 24,35) or
FindColorTolerance(x,y,15000611,35, 4,89, 24,35) or
FindColorTolerance(x,y,13421577,35, 4,89, 24,35)=false);
end;
Procedure AntiRandom;
begin
SetScreenName(Nick);
FindNormalRandoms;
FindTalk;
FindLamp(‘strength’);
if FindFight then
begin
Writeln(‘Found a fight so i terminated the script’);
TerminateScript;
end;
end;

begin
ActivateClient;
repeat
IfDoorOpen;
CloseDoor;
until(false)
end.[/code]
