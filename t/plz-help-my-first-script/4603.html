I was bored and I made it sexy for you. I added a Functionc alled MyFC which shortenedyour code a little

///////////////////////////////////////////////////////////////////////////
/// Powerminer ///
/// By Uncle James ///
/// Instructions ///
/// + make sure you add all the colours ///
/// + mini-map facing north ///
/// + U must look from above in ///
/// + V-bright, 32bit colors ///
/// + U Need Scar version 2.0.3 ///
/// + start in any mine, any where ///
/// + make sure silent mouse is off ///
/// + make sure the server is low-detail ///
/// ///
///////////////////////////////////////////////////////////////////////////
Program PowerMiner;

Var DropColor, x, y : Integer;

Const
     (*==========Colors On Main Screen==========*)
     Ore1   = 536912; // color of first rock to mine
     Ore2   = 536912; // color of second rock to mine
     Ore3   = 536912; // color of third rock to mine
     Invore = 3966956; // color of rock in inventory

Function MyFC(bitmap : Integer): Boolean;
Begin
     FindColor(x, y, bitmap, 201, 113, 327, 220);
End;

Procedure MineRocks;
Begin
     If MyFC(ore1) or MyFC(ore2) or MyFC(ore3) Then
     Begin
          Wait(Random(50)+500);
          MoveMouseSmooth(x, y);
          Wait(Random(50)+100);
     End;
     If MyFC(ore1) or MyFC(ore2) or MyFC(ore3) Then
     Begin
          ClickMouse(x, y, true);
          Wait(random(1000)+5000);
          Wait(2000);
     End;
End;

Procedure DropOre;
Begin
     If FindColor(x, y, Invore, 699, 424, 742, 426) Then
     Begin
          MoveMouseSmooth(x,y);
          Wait(Random(50)+500);
          DropColor(invore)
     End;
End.
