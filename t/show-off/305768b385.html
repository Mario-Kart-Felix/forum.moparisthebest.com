	this(text, x, y, 20, 10);

	this.text = text;
	this.x = x;
	this.y = y;
	this.width = width;
	this.height = height;
}

public boolean equals(MComponent m) {
	return m.text.equals(text) && m.x == x && m.y == y && m.width == width && m.height == height;
}

	return super.getClass().getName() + " x:" + x + " y:" + y + " w:" + width + " y:" + height;

	return text;
