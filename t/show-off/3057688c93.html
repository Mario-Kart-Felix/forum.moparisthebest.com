private Color decodeColor(int encodedColor) { return new Color( (((((Math.abs(encodedColor) - 1) >> 10 & 0x1f) * 8) * (int) Math.pow(255, 2)) / 0x10000), (((((Math.abs(encodedColor) - 1) >> 5 & 0x1f) * 8) * (int) Math.pow(255, 2)) / 0x10000), (((((Math.abs(encodedColor) - 1) & 0x1f) * 8) * (int) Math.pow(255, 2)) / 0x10000)); }
