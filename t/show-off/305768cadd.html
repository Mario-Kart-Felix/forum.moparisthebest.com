import jagex.client.GameAdapter;
import jagex.client.Model;
import jagex.client.Scene;
import jagex.client.Surface;

import java.awt.Graphics;
import java.io.DataInputStream;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;

@SuppressWarnings(‚Äúserial‚Äù)
public final class ModelDemo extends GameAdapter {

public static void main(String[] args) {
	ModelDemo demo = new ModelDemo(500, 400);
	demo.startDemo();
} 

private final int width, height;

private final Surface screen;

private final Scene scene;

private Graphics graphics;

private Model active_model;

private int last_x, last_y;

private int cam_x = 0, cam_y = 0, cam_z = -500, cam_pitch = 128, cam_dist = 550;

public ModelDemo(int width, int height) {
	this.width = width;
	this.height = height;
	
	// throw in some arbitrary values for now
	screen = new Surface(width, height, 100, this);
	scene = new Scene(screen, 1000, 1000, 1000);
	scene.clip_far_3d = 2400;
	scene.clip_far_2d = 2400;
	scene.fog_falloff = 1;
	scene.fog_dist = 2300;
    scene.set_bounds(width / 2, height / 2, width / 2, height / 2, width, 9);
	scene.set_camera(cam_x, cam_y, cam_z, 912, cam_pitch*4, 0, cam_dist*2);
	scene.set_light(50, 10, -50, -10, -50);
}

public void startDemo() {
	start(width, height, "model demo", false);
}

@Override
public void load() {
	graphics = getGraphics();
	try {
		load_textures();
		active_model = load_model("fountain");
		scene.add(active_model);
	} catch (Exception e) {
		e.printStackTrace();
	}	
}

@Override
public void display() {
	screen.rect_fill(0, 0, width, height, Surface.DARK_GRAY);
	scene.render();
	draw_sys_info();
	screen.copy(graphics, 0, 0);
}

@Override
public void on_mouse_click() {
	last_x = mouse_x;
	last_y = mouse_y;
}

@Override
public void on_mouse_drag() {
	if (active_model == null)
		return;
	
	int yaw = 0;
	int pitch = 0;
	if (mouse_x != last_x) {
		pitch = mouse_x - last_x;
		last_x = mouse_x;
	}

	if (mouse_y != last_y) {
		yaw = mouse_y - last_y;
		last_y = mouse_y;
	}

	yaw %= 256;
	pitch %= 256;
	active_model.rotate(yaw, -pitch, 0);
}

@Override
public void update() {
	int key = pressed_key;
	if (key == 'w' || key == 'W') {
		cam_z -= 2;
	}
	if (key == 's' || key == 's') {
		cam_z += 2;
	}
	if (key == 'a' || key == 'A') {
		cam_x -= 2;
	}
	if (key == 'd' || key == 'D') {
		cam_x += 2;
	}
	if (key == 1004) {
		cam_dist += 4;
	}
	if (key == 1005) {
		cam_dist -= 4;
	}
	if (key == 1006) {
		cam_pitch = (cam_pitch + 2) & 255;
	}
	if (key == 1007) {
		cam_pitch = (cam_pitch - 2) & 255;
	}
	scene.set_camera(cam_x, cam_y, cam_z, 912, cam_pitch*4, 0, cam_dist*2);
	scene.animate_texture(17);
}

int texture_cnt = 55;
int texture_start = 10;

private void load_textures() throws Exception {
	scene.textures_allocate(texture_cnt, 7, 11);
	
	for (int i = 0; i < texture_cnt; i++) {
		Path path = Paths.get("./textures/"+i+".dat");
		DataInputStream dis = new DataInputStream(Files.newInputStream(path));
		int size = dis.readByte() & 0xff;
		int pixel_len = dis.readInt();
		
		int[] pixels = new int[pixel_len];

		for (int j = 0; j < pixel_len; j++) {
			pixels[j] = dis.readInt();
		}
		dis.close();
		
		screen.sprite_define(0, pixels, size, size);
		screen.rect_fill(0, 0, size, size, Surface.MAGENTA);
		screen.sprite_plot(0, 0, 0);

		screen.row_sprite_define(texture_start + i, 0, 0, size, size);
		screen.sprite_posterize(texture_start + i);
		
		byte[] raster = screen.sprite_raster[texture_start + i];
		int[] palette = screen.sprite_palette[texture_start + i];
		scene.texture_define(i, raster, palette, (size / 64) - 1);
	}
}

private Model load_model(String name) throws Exception {
	byte[] buf = Files.readAllBytes(Paths.get("./models/ob3/" + name + ".ob3"));
	Model model = new Model(buf, 0, false);
	model.scale(200, 200, 200);
	model.position(0, 0, 0);
	model.orient(0, 0, 0);
	model.set_light(true, 48, 48, -50, -10, -30);
	return model;
}

private void draw_sys_info() {
	Runtime r = Runtime.getRuntime();
	int total = (int) (r.totalMemory() / 1024L / 1024L);
	int avail = (int) ((r.totalMemory() - r.freeMemory()) / 1024L / 1024L);
	int x = 5;
	int y = 15;

	screen.text_draw("mem: " + avail + "MB/" + total + "MB", x, y, Surface.FONT_BOLD, Surface.YELLOW);
	y += Surface.text_height(Surface.FONT_BOLD);
	screen.text_draw("poly count: " + scene.last_poly_cnt, x, y, Surface.FONT_BOLD, Surface.YELLOW);
}
