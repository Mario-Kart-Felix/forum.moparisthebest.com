[code=java]// create a new thread so we don’t tie up
// resources that would otherwise slow
// down rendering
new Thread(new Runnable() {
// the initial starting camera position
// for this animation
int cX = 9300,
cY = 1000,
cZ = 4400,
cYaw = 932,
cPitch = 550,
cRoll = 0,
cDist = 1500;
@Override
public void run() {
while (true) {
if (cX > 300 && cX <= 5000) {
cX++;
}
if (cY > 1500 && cY <= 5000 && cX > 4300) {
cY++;
} else {
cY–;
}
if (cZ > 300 && cZ <= 5000 && cZ > 4300) {
cY++;
} else {
cY–;
}
if (cPitch != 700) {
// make the pitch progress more aggressively
cPitch += 2;
}

        // set the camera to the new calculated position
        aClass4.method146(cRoll, 0x30c - -100, cPitch, cY, cYaw, cZ, 64 ^ -20191, cX, cDist);

        // sleep so the animation doesn't happen within 2 frames
        Game.sleep(100L);
    }
}

}).start();[/code]
