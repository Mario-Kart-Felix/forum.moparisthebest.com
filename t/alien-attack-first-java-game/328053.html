Well I decided to make a game with Java. (Finally)

This game is one of those where the aliens come in rows and you need to shoot them before they pass the line. I also implemented some of the features of other games, such as the pong game where you try to hit all of the blocks with a ball bounced with a paddle. The game has power ups, and gets rather hard in the last levels. There is a score algorithm as well.
More information can be found here: Alien Attack HomePage

class Bullet extends Rectangle2D.Float {
	public int y_speed;
	private Enemy[] enemies;
	public int y;
	PaintSurface ps;
	int index;
	public boolean isBad;

	public Bullet(int x, int y, int y_speed, Enemy[] enemies, PaintSurface ps, int index) {
		super(x, y, 2, 4);
		this.x = x;
		this.enemies = enemies;
		this.ps = ps;
		this.index = index;
		this.y_speed = y_speed;
		if(y_speed > 0) {
			isBad = true;
		}
	}
	public void move() {
		Rectangle2D r = new Rectangle2D.Float(super.x, super.y, 2, 4);	
		if(!isBad) {			
			for (int i = 0; i < enemies.length; i ++) {
				if(enemies[i] != null && enemies[i].intersects(r)) {
					ps.totalHits++;
					if(!ps.powerUps[0]) {
						ps.destroyB(index);
					}
					enemies[i].destroy();
				}
			}
		}
		if(isBad) {
			Rectangle2D ship = new Rectangle2D.Float(ps.x_pos, ps.y_pos, 53, 51);
			Rectangle2D me = new Rectangle2D.Float(super.x, super.y, 2, 4);
			if(me.intersects(ship)) {
				ps.takeHit();
				ps.destroyB(index);
			}
		}
		super.y += y_speed;
		y = (int)(super.y);
		x = (int)(super.x);
	}
}

class Enemy extends Rectangle2D.Float {
	public double x_speed = 0;
	public int y;
	public double x;
	PaintSurface ps;
	int index;
	boolean isBonus = false;
	int bonusType; //0 for red bullet, 1 for MultiShot

	public Enemy(int x, int y, PaintSurface ps, int index, double speed) {
		super(x, y, 33,37);
		this.ps = ps;
		this.x = x;
		this.y = y;
		this.x_speed = speed;
		this.index = index;
	}
	public void move() {
		Rectangle2D r = new Rectangle2D.Float(super.x, super.y, 33, 37);
		super.x += x_speed;
		this.x = super.x;
		if(x < 2 && x_speed < 0 || x > 263 && x_speed > 0) {
			x_speed = -x_speed;
			super.y += 45;
			this.y = (int)(super.y);
		}
	}
	public void destroy() {
		if(isBonus) {
			ps.powerUps[bonusType] = true;
			ps.puCounter[bonusType] += 175;
			ps.bonusPoints += 50;
		}
		ps.destroyE(index);
	}
		
}
class AnimationThread extends Thread {
	JDialog gs;

	public AnimationThread(JDialog gs) {
		this.gs = gs;
	}

	public void run() {
		try {
			ps.started = true;
			ps.setDimensions(300, 400);
		}
		catch (Exception e) {
			e.printStackTrace();
		}
		while(true) {
			gs.repaint();
			try {
				Thread.sleep(20);
			} catch (Exception e) {
				e.printStackTrace();
			}
		}
	}
}
		
class PaintSurface extends JPanel {
	double x_speed = .000000000001;
	int y_pos, x_pos, totalShots, totalHits, bonusPoints, numberCounter, width, height = 0;
	Image plane, planemoving, planehit, bonus, bonus1, enemy;
	ImageObserver observer;
	Toolkit tk = Toolkit.getDefaultToolkit();
	public boolean started, turnRed, lost, won, moving = false;
	public int hitPoints = 50;
	Bullet[] bullets = new Bullet[900];
	Enemy[] enemies = new Enemy[100];
	double[] enemySpeeds = {-.5, -.6, -.8, -1.1, -1.5, -2, -2.6, -3.4, -4.2, -5.1, -6.1, -7.2, -8.4, -9.7, -11.1};
	double[] enemyNumber = {1, 2, 4, 7, 11, 16, 22, 29, 37, 46, 56, 67, 79, 92, 106};
	boolean[] powerUps = {false, false}; //0 is Power Bullets, 1 is Multi Shot, Others are open
	int[] puCounter = {0, 0};

	public void setDimensions(int width, int height) {
		loadImages();
		this.width=width-15;
		this.height = height-65;
		this.x_pos = width/2;
		this.y_pos = 315;
	}
	public void takeHit() {
		hitPoints--;
		turnRed = true;
		if(hitPoints < 1) {
			lost = true;
		}
	}
	public void destroyE(int index) {
		enemies[index] = null;
	}
	public void destroyB(int index) {
		bullets[index] = null;
	}	
	public void setLoose() {
		lost = true;
	}
	public int getEnemyBulletID() {
		int first = -1;
		for(int i = 0; i < 90; i++) {
			if(ps.bullets[i] != null) {
				continue;
			}
			first =  i;
		}
		return first;
	}
	public void loadImages() {
		bonus = tk.getImage("bonus.png");
		bonus1 = tk.getImage("bonus1.png");
		enemy = tk.getImage("enemy.png");
		plane = tk.getImage("plane.png");
		planehit = tk.getImage("planehit.png");
		planemoving = tk.getImage("planemoving.png");
	}
		
	public int getScore() {
		if(totalHits != 0 && totalShots != 0) {
			if ( ((int)(((double)(totalHits)/(double)(totalShots))*totalHits + bonusPoints)) < 1) {
				return 1;
			}	
			return (int)((((double)(totalHits)/(double)(totalShots)))*totalHits + bonusPoints);
		}
		return 0;
	}

	public void paint(Graphics g) {
		Graphics2D g2 = (Graphics2D)g;
		if(started && !lost && !won) {		
			g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
			g.drawImage(tk.getImage("background.png"), 0, 0, this);
			if(turnRed) {
				g.drawImage(planehit, x_pos, y_pos, this);
				turnRed = false;
			} 
			else if(!moving){
				g.drawImage(plane, x_pos, y_pos, this);
			}
			else {
				g.drawImage(planemoving, x_pos, y_pos, this);
			}
			if(x_pos < 0 && x_speed < 0) {
			}
			else if(x_pos > width - 50 && x_speed > 0) {
			}
			else { 
				x_pos += x_speed; 
			}
			for(int i = 0; i < puCounter.length; i++) {
				if(powerUps[i]) {
					puCounter[i]--;
				}
				if(puCounter[i] < 1) {
					powerUps[i] = false;
				}
			}
			for(int i = 0; i < bullets.length; i++) {
				if(bullets[i] != null) {
					if(powerUps[0] && !bullets[i].isBad) {
						bullets[i].y_speed = -7;
						g2.setColor(Color.RED);
					}
					else if(!powerUps[0] && !bullets[i].isBad) {
						bullets[i].y_speed = -5;
					}
					bullets[i].move();
					if(bullets[i] != null) {
						g2.fill(bullets[i]);
						g2.setColor(Color.BLACK);
						if(bullets[i].y < 0) {
							bullets[i] = null;
						}
						if(bullets[i] != null && bullets[i].y > 400) {
							bullets[i] = null;
						}
					}
				}
			}
			int counter = 0;
			for(int i = 0; i < enemies.length; i++) {
				if(enemies[i] != null) {
					enemies[i].move();
					int random = (int)(Math.random() * 200);
					if(random == 2) {
						int b = getEnemyBulletID();
						if(b != -1) {
							bullets[b] = new Bullet((int)(enemies[i].x) + 17, (int)(enemies[i].y) + 38 , 5, enemies, this, b);
						}	
					}
					g2.fill(enemies[i]);
					if(!enemies[i].isBonus) {
						g.drawImage(enemy, (int)(enemies[i].x), enemies[i].y, this);
					}
					else {
						switch(enemies[i].bonusType) {
							case 0:
								g.drawImage(bonus, (int)(enemies[i].x), enemies[i].y, this);
							break;
							case 1:
								g.drawImage(bonus1, (int)(enemies[i].x), enemies[i].y, this);
							break;
						}
					}
					if(enemies[i].y + 37 > 350) {
						setLoose();
					}
					counter++;
				}
			}
			if(counter < 1) {
				if(numberCounter > 13) {
					won = true;
				}
				for(int i =0; i < enemyNumber[numberCounter]; i++) {
					int random = (int) (Math.random() * 30);
					enemies[i] = new Enemy((width  + (39*i)) , 5, this, i, enemySpeeds[numberCounter]);
					if(random  == 2) {
						enemies[i].isBonus = true;
						int random2 = (int) (Math.random() * 2);
						if(random2 == 0) {
							enemies[i].bonusType = 0;
						}
						else {
							enemies[i].bonusType = 1;
						}
					}
				}
				numberCounter++;
			}
			g2.setFont(new Font("Times New Roman", Font.BOLD, 12));
			if(powerUps[1] && powerUps[0]) {
				g.drawString("Multi Shot: " + puCounter[1]/35, 5, 330);
				g.drawString("Fire Shot: " + puCounter[0]/35, 5, 320);
			}
			else if(powerUps[0]) {
				g.drawString("Fire Shot: " + puCounter[0]/35, 5, 330);
			}
			else if(powerUps[1]) {
				g.drawString("Multi Shot: " + puCounter[1]/35, 5, 330);
			}
			g.drawString("HP: " + hitPoints, 5, 340 );
			g.drawString("Enemies Left: " + (counter), 5, 350 );
			g.drawString("Level: " + numberCounter , 5, 360);
			Line2D line2 = new Line2D.Float(0, 310, width+20, 310);
			g2.draw(line2);
		}
		NumberFormat nf = NumberFormat.getNumberInstance();
		nf.setMaximumFractionDigits(1);
		if(lost) {
			g2.setFont(new Font("Impact", Font.PLAIN, 30));
			g2.drawString("GAME OVER: YOU LOST!", 15, 200);
			g2.setFont(new Font("Times New Roman", Font.PLAIN, 16));
			g2.drawString("Shots Fired: "+ totalShots, 15, 220);
			g2.drawString("Shots Hit: "+ totalHits, 15, 235);
			g2.drawString("Hit Percent: "+ nf.format(((double)(totalHits)/(double)(totalShots)*100)) + "%", 15, 250);
			g2.drawString("Bonus: PowerUps * 50 = "+ bonusPoints, 15, 265);
			g2.drawString("Your Final Score Was: " + getScore(), 15, 295);
			g2.drawString("You Ended With: " + hitPoints + "hp", 15, 345);
		}
		if(won) {
			g2.setFont(new Font("Impact", Font.PLAIN, 30));
			g2.drawString("GAME OVER: YOU WON!", 5, 100);
			g2.setFont(new Font("Times New Roman", Font.PLAIN, 16));
			g2.drawString("Shots Fired: "+ totalShots, 15, 220);
			bonusPoints += 200;
			g2.drawString("Shots Hit: "+ totalHits, 15, 235);
			g2.drawString("Hit Percent: "+ nf.format(((double)(totalHits)/(double)(totalShots)*100)) + "%", 15, 250);
			g2.drawString("Bonus: PowerUps * 50 + 200 for winning = "+ bonusPoints, 15, 265);
			g2.drawString("Your Final Score Was: " + getScore(), 15, 295);
			g2.drawString("You Ended With: " + hitPoints + "hp", 15, 345);
		}	
	}
}
