A JDialog for the contents of each tabbed pane? You’d want to use a JPanel instead.

So basically, just design any jpanel the way you normally would to add it to a frame. Use any layout manager you’d like, really (or none if you prefer).

JTabbedPane tabbedPane = nee JTabbedPane();

JPanel somePanel = new JPanel();
// ... Do whatever with the panel

tabbedPane.addTab("Tab Title", somePanel);

{
    settingsDialog = new JDialog(CommandWindow.frame, "EnurScape Settings");
    settingsDialog.setLayout(new BorderLayout());
    settingsDialog.setIconImage(CommandWindow.frame.getIconImage());
    JTabbedPane tabbedPane = new JTabbedPane();
    tabbedPane.setUI(new javax.swing.plaf.basic.BasicTabbedPaneUI());
    final String tabNames[] = new String[]{"Server", "Client", "Security", "Other"}; //These are the tab names
    JPanel tabs[] = new JPanel[tabNames.length]; //tabPanels
    final String sectionNames[][] = new String[][]{{"Basic Settings", "Logging"}, {"Command Access", "Client Security", "Default User Rights"}, {"DOS Prevention", "EnurScape Client Only"}, {"About EnurScape Private Server", "Bugs & Suggestions"}}; //These strings are what the titled borders are called
    JPanel sections[][] = new JPanel[tabs.length][3]; //These panels go into the tab panels to create a section with a titled border from the above string array
    for (int i = 0; i < tabs.length; i++) {
        tabs[i] = new JPanel();
        //tabs[i].setLayout(new GridLayout(sectionNames[i].length, 1));
        tabbedPane.addTab(tabNames[i], tabs[i]);
    }
    for (int i = 0; i < sections.length; i++) {
        for (int j = 0; j < sections[i].length; j++) {
            if (sectionNames[i].length <= j) sections[i][j] = null;
            else {
                sections[i][j] = new JPanel();
                sections[i][j].setLayout(new FlowLayout());
                sections[i][j].setBorder(BorderFactory.createTitledBorder(sectionNames[i][j])); // <---- This is where I create the section borders
                tabs[i].add(sections[i][j]);
            }
        }
    }
}
