You set the waittime variables, the tolerance, and the click positions for click one and click two, and how many times to repeat. This script just clicks in one position, using Mouse command, with the tolerance you set.

I use this to craft hard leather into hard bodies quickly, but you can also use for high alchemy, and about anything you can think of. Simple two click tasks.

(EASILY SETUP)
ClickTol:=5 //This is the tolerance it randomly adds in cordinants to the x and y of your clicks.
Click1X:=459 //x position for first click
Click1Y:=289 //y position for first click
Click2X:=475 //x position for second click
Click2Y:=324 //y position for second click
WaitTime:=25 //Time in MS between clicks
WaitTol:=50 //Time tolerance in MS, randomly adds to waittime.
ClickTimes:=10 //How many times it will repeat.

program BumClicker;
{.include SSI2.txt}
var
Click1x,Click1y,waittime,waittol,clicktol,
Click2x,Click2y,stop,ClickTimes: Integer;

Procedure Setup;
Begin
 ClickTol:=5      //This is the tolerance it randomly adds in cordinants to the x and y of your clicks.
 Click1X:=459     //x position for first click
 Click1Y:=289     //y position for first click
 Click2X:=475     //x position for second click
 Click2Y:=324     //y position for second click
 WaitTime:=25     //Time in MS between clicks
 WaitTol:=50      //Time tolerance in MS, randomly adds to waittime.
 ClickTimes:=10   //How many times it will repeat.
End;

Procedure Clicker;
Begin
Stop:=0
 Repeat
  Stop:=Stop+1
  Mouse(Click1X,Click1Y,ClickTol,ClickTol,true)
  Wait((WaitTime)+random(WaitTimeTol))
  Mouse(Click2X,Click2Y,ClickTol,ClickTol,true)
  Wait((WaitTime)+random(WaitTimeTol))
 Until(stop=ClickTimes)
End;

begin
Setup;
Clicker;
end.
