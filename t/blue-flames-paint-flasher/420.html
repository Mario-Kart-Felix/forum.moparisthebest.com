///Version 1.12 I added a new way to find the start button using dtm
program PaintFLASHER;

var flash:integer;PIcon:integer;x,y:integer;Pbucket:integer;Green:integer;
red:integer;Starticon:Integer;warning:string;warning2:string;warning3,warning4:string;
warning5,warning6,warning7:string;

const
{------------------------------------------------------------------}
Flashes = 300;//how many times to change colors
M0 = 1;//way to open start. 1=new way using dtm to find the start icon
//2=old way using coordinates (must have 1024X768 desktop)
{------------------------------------------------------------------}

procedure Load;
begin
PIcon := BitmapFromString(14, 2, ‘8BB7DF8ADB8A3D9F310167014’ +
‘05E14EBBB05D4A1018A691A4073C3708DC0A37E82AB98A0A7ADBE’ +

‘CAADDF0B8E3E9CAD6CA9FB6CBE4744BEBD7CF’);

‘CE6008016E180053DC00074036C7C0C0C9CDC98D10495B73B0998’ +
‘66C26A6F9D80100C76B430B’);

Green := BitmapFromString(12, 2, ‘z78DA33307073330082C14202’ +
‘00D1981F21’);

red := BitmapFromString(12, 2, ‘z78DA7373330002B741430200D9’ +
‘D81F21’);

Starticon := DTMFromString(‘78DA63146160607067624006AED11FC034231’ +
‘0836418C530D538C79BC2D58069714C35ABBA5851D40000127904’ +
‘C5’);
end;

procedure OpeningPaint;
begin
warning:=‘Dont use this if you have a chance of having a Seizure!’
warning2:=‘You have 5 seconds to stop this script’
warning3:=‘You have 4 seconds to stop this script’
warning4:=‘You have 3 seconds to stop this script’
warning5:=‘You have 2 seconds to stop this script’
warning6:=‘You have 1 seconds to stop this script’
warning7:=‘Too late :-(’

if(m0=1)then
begin
Writeln(warning);
Writeln(warning2);
wait(1000)
Writeln(warning3);
wait(1000)
Writeln(warning4);
wait(1000)
Writeln(warning5);
wait(1000)
Writeln(warning6);
wait(1000)
Writeln(warning7);
Load;
if(FindDTM(Starticon,x,y,0,0,22,751))then
Movemouse(x,y);
Wait(15);
Clickmouse(x,y, True);
Wait(45);
findbitmap(PIcon,x,y);
wait(2000);
movemouse(x,y);
wait(20);
clickmouse(x,y, True);
wait(1000);
findbitmap(Pbucket,x,y);
Wait(300);
Movemouse(x,y);
Wait(150);
Clickmouse(x,y, True);
Wait(200)
findbitmap(Green, x,y);
wait(150);
MoveMouse(x,y);
wait(150);
Clickmouse(x,y, True);
Wait(300);
findbitmap(red, x,y);
wait(150);
Movemouse(x,y);
Wait(150);
Clickmouse(x,y, False);
Wait(150);
Movemouse(425, 321);
Wait(400);
end;
if(m0=2)then
begin
Writeln(warning);
Writeln(warning2);
Wait(5000);
load;
Movemouse(38, 756);
Wait(150);
Clickmouse(38, 756, True);
Wait(450);
findbitmap(PIcon,x,y);
wait(2000);
movemouse(x,y);
wait(20);
clickmouse(x,y, True);
wait(1000);
findbitmap(Pbucket,x,y);
Wait(300);
Movemouse(x,y);
Wait(150);

wait(150);
Movemouse(x,y);
Wait(150);
Clickmouse(x,y, False);
Wait(150);
Movemouse(425, 321);
Wait(400);
end;
end;
procedure flashing;
begin
clickmouse(425, 321, True);
Wait(5)
Clickmouse(425, 321, False);
