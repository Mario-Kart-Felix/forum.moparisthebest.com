No problem. My server suite has many tools that will make operating a server much easier. One such example is implementing SUIT (my interface library) with XML to produce something like this:

<?xml version="1.0" encoding="utf-8"?>
<?xml-stylesheet href="interface.css"?>
<interface id="3233">
    <frame id="inventory">
        <scrollpane>
            <itemtable width="4" length="7" />
        </scrollpane>
    </frame>
</interface>
frame {
    visible: false;
}

#inventory {
    background: url("img/frame/background.jpg");
    visible: true;
}
