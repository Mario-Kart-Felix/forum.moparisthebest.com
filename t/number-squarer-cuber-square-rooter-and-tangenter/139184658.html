import java.math.BigInteger;
import java.math.BigDecimal;

public class FermatFactorizationMethod {
    public static final BigInteger RSA_MODULUS_202 = new BigInteger(
        "7162900525229798032761816791230527296329313291232324"+
        "2902378492635012082079728940539290656365223631636210"+
        "00728841182238772712427862772219676577293600221789"
    );
    public static final BigInteger ZERO = BigInteger.valueOf(0),
                                   ONE = BigInteger.valueOf(1),
                                   TWO = BigInteger.valueOf(2),
                                   FOUR = BigInteger.valueOf(4);
    
    
    public static void main(String[] args) {
        new FermatFactorizationMethod(RSA_MODULUS_202);
    }
    
    
    public FermatFactorizationMethod(final BigInteger i) {
        // Must be odd composite number
        if(i.mod(TWO).equals(ZERO))
            return;
        
        // Find k, the first value for which if squared is larger than i
        BigInteger maxKValue = i.add(ONE).divide(TWO);
        BigInteger kSearch = sqrt(new BigDecimal(i), 25).toBigInteger();
        while(kSearch.pow(2).compareTo(i) < 1 && kSearch.compareTo(maxKValue) == -1)
            kSearch = kSearch.add(ONE);
        
        
        System.out.println("k = "+ kSearch);
        
        
        // Run Fermat's Factorization method
        BigInteger iterationDifference;
        for(; kSearch.compareTo(maxKValue) == -1; kSearch = kSearch.add(ONE)) {
            try {
                BigInteger factorBase = sqrt(new BigDecimal(kSearch.pow(2).subtract(i)), 25).toBigIntegerExact();
                
                System.out.println(i +" = ("+ (kSearch.subtract(factorBase)) +")("+ (kSearch.add(factorBase)) +")");
                break;
            } catch(ArithmeticException e) {}
        }
    }
    
    
    
    /*
     * Algorithm used from http://www.merriampark.com/bigsqrt.htm
     */
    public static final BigDecimal sqrt(final BigDecimal originalValue, int divisionScale) {
        // Must be >= 1, otherwise doesn't run square root algorithm
        int maxIterations = 50;

        // Make sure originalValue is a non-negitive number
        if(originalValue.compareTo(new BigDecimal(ZERO)) == -1) {
            throw new ArithmeticException("Square root of a negative number.");
        }

        if(maxIterations < 1)
            return originalValue;

        BigDecimal lastValue = new BigDecimal(ZERO),
                   sqrtValue = getInitialApproximation(originalValue),
                   errorValue = null;

        int currentIteration = 0;
        boolean isCalculating = true;
        while(isCalculating) {
            lastValue = sqrtValue;
            sqrtValue = originalValue.divide(sqrtValue, divisionScale, BigDecimal.ROUND_HALF_UP);
            sqrtValue = sqrtValue.add(lastValue);
            sqrtValue = sqrtValue.divide(new BigDecimal(TWO), divisionScale, BigDecimal.ROUND_HALF_UP);
            errorValue = originalValue.subtract(sqrtValue.multiply(sqrtValue));

            if(++currentIteration == maxIterations) {
                isCalculating = false;
            } else if(lastValue.equals(sqrtValue)) {
                isCalculating = errorValue.abs().compareTo(new BigDecimal(ONE)) > -1;
            }
        }
        return sqrtValue;
    }

    private static final BigDecimal getInitialApproximation(final BigDecimal originalValue) {
        int totalIntegerDigits = originalValue.toBigInteger().toString().length();
        if ((totalIntegerDigits % 2) == 0) {
            totalIntegerDigits--;
        }
        totalIntegerDigits /= 2;

        return new BigDecimal(ONE).movePointRight(totalIntegerDigits);
    }
}

BobDole81 July 29, 2016, 10:08pm #28

It’s Fermat’s Factorization Method, which “brute forces” a relatively large integer by finding its form as a difference of two squares. Basically, it’s a method for finding the prime factors of odd, composite numbers. The actual method is not very difficult in terms of mathimatical complexity (you really only need a good knowledge of algebra to understand it), but it comes under the Number Theory category, which is why you’ve probably not heard of it.

Its sorta a brute force…but its not slow because its a brute forcer.

Fermat’s theorem runs on the fact that any odd composite integer is the difference of two squares, a^2 - b^2. Therefore, a^2 must be greater than i, the odd composite integer. The factorization method therefore keeps incrementing a until a^2 - i = b^2 some perfect square, which is the “offset” for the factors - a + b, and a - b.
