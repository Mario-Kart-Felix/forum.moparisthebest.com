I recently wrote a program out of boredom to log the online count every 15 minutes of two rsc servers out, one being RSCRevolution, the other RSCEmulation. I ran the loggers for nearly a week straight (i accidentally killed the process yesterday and forgot about it until today). It got me pretty interested in calculating statistics about how active rsc is right now, so with that being said, I wrote another program to parse the output of the first and generate some statistics.

P.S. Yeah I know this is such a trivial task, but still it was fun to see the result.

#!/usr/bin/ruby

require 'open-uri'
require 'rufus-scheduler'

RSCE_URL = 'http://www.rscemulation.org/include/ajax/online.php'
RSCR_URL = 'http://www.rscrevolution.com/style/newgame/includes/global.counter.php'

RSCE_FILE = 'rsce.txt'
RSCR_FILE = 'rscr.txt'

def log_online_count url, file
    begin
        count = open(url).read
    rescue
        count = 'Error retreiving online count'
    end
    File.open(file, 'a') do |f|
        f.puts "#{Time.new.strftime('%Y-%m-%d %H:%M:%S')}\t#{count}"
    end
end

scheduler = Rufus::Scheduler.new

scheduler.every '15m' do
    log_online_count RSCE_URL, RSCE_FILE
    log_online_count RSCR_URL, RSCR_FILE
end

scheduler.join

[code=c]#include <stdio.h>
#include <errno.h>

int main(int argc, const char * argv[])
{
FILE* file = fopen(argv[1], “r”);

if (!file)
{
    perror("Error opening file");
    return 1;
}

char *first_log, *last_log;
char high_time[20], low_time[20];
char buf[100];

count = total = high = 0;
low = -1;

while (fgets(buf, 100, file) != NULL)
{
    char date[11];
    char time[9];
    int online;
    sscanf(buf, "%s %s\t%d\n", date, time, &online);

    if (low == -1 || low > online)
    {
        low = online;
        sprintf(low_time, "%s %s", date, time);
    }
    if (online > high)
    {
        high = online;
        sprintf(high_time, "%s %s", date, time);
    }

    count += online;
    ++total;
}

fclose(file);
file = NULL;

printf("Average online count: %d\n", count / total);
printf("Highest player count %d occured at %s\n", high, high_time);
printf("Lowest player count %d occured at %s\n", low, low_time);
printf("Logs taken: %d\n", total);
printf("First log taken at: %s\n", first_log);
printf("Last log taken at: %s\n", last_log);
return 0;

}[/code]

c version of the logger:

[code=c]#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <string.h>
#include <threads.h>
#include <curl/curl.h>

#define RSCE_URL “http://www.rscemulation.org/include/ajax/online.php”
#define RSCR_URL “http://www.rscrevolution.com/style/newgame/includes/global.counter.php”

#define RSCE_FILE “rsce.txt”
#define RSCR_FILE “rscr.txt”

int log_online_count(char* url, char* log);
size_t fmt_time(char* buf, size_t len);
size_t curl_handle(char* str, size_t size, size_t nmemb, void* userdata);

int main(int argc, char * argv[])
{
time_t t1, t2;
int success;

while (1)
{
    time(&t1);
    success = log_online_count(RSCE_URL, RSCE_FILE);
    success |= log_online_count(RSCR_URL, RSCR_FILE);
    
    if (!success)
    {
        fprintf(stderr, "Fatal error logging online count!\n");
        exit(1);
    }
    
    time(&t2);
    xtime xt = { .sec = 900 - (t2 - t1) };
    thrd_sleep(&xt);
}
return 0;

}

int log_online_count(char* url, char* log)
{
FILE* file = fopen(log, “a”);
CURL* curl = curl_easy_init();
int success = file && curl;

if (success)
{
    curl_easy_setopt(curl, CURLOPT_URL, url);
    curl_easy_setopt(curl, CURLOPT_WRITEDATA, file);
    curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, curl_handle);
    curl_easy_perform(curl);
}

if (file) fclose(file);
if (curl) curl_easy_cleanup(curl);
return success;

}

size_t fmt_time(char* buf, size_t len)
{
time_t t;
time(&t);
return strftime(buf, len, “%Y:%m:%d %H:%M:%S”, gmtime(&t));
}

size_t curl_handle(char* str, size_t size, size_t nmemb, void* userdata)
{
char tm_buf[20];
fmt_time(tm_buf, 20);
fprintf((FILE*) userdata, “%s\t%d\n”, tm_buf, atoi(str));
return size;
