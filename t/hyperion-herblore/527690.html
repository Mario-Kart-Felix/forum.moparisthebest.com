package org.hyperion.rs2.model.content.skills.herblore;

import java.util.HashMap;

import org.hyperion.rs2.model.Item;
import org.hyperion.rs2.model.Player;
import org.hyperion.rs2.model.content.skills.SkillHandler;



/**
 * @author Tom (get on mehh)
 */

public class CleanHerbs {
	
	// Was is still identifying in 464?
	
	private static final String CLEAN_MESSAGE = "You clean the herb.";
	
	/**
	 * @author Tom (get on mehh)
	 */
	public enum HerbData {
		
		// TODO: Add the rest of the herbs.
		
		GUAM(199, 200, 1, 2.5);
		
		private HerbData(int grimy, int clean, int level, double xp) {
			this.grimy = grimy;
			this.clean = clean;
			this.level = level;
			this.xp = xp;
		}
		
		private int grimy, clean, level;
		private double xp;
		
		public int getGrimy() {
			return grimy;
		}
		public int getClean() {
			return clean;
		}
		public int getLevel() {
			return level;
		}
		public double getXp() {
			return xp;
		}
		
		static HashMap<Integer, HerbData> identifyData = new HashMap<Integer, HerbData>();
		
		static {
			for(HerbData herb : HerbData.values())
				identifyData.put(herb.getGrimy(), herb);
		}
	}
	
	public static boolean canClean(final Player player, int herb) {
		HerbData i = HerbData.identifyData.get(herb);
		if (i == null)
			return false;
		if (!SkillHandler.skillEnabled(player, SkillHandler.HERBLORE))
			return false;
		if (!SkillHandler.checkSkill(player, SkillHandler.HERBLORE, i.getLevel(), "clean this herb."))
			return false;
		cleanHerb(player, herb);
		return true;
	}
	
	private static void cleanHerb(final Player player, int herb) {
		HerbData i = HerbData.identifyData.get(herb);
		player.getInventory().remove(new Item(i.getGrimy()));
		player.getInventory().add(new Item(i.getClean()));
		player.getSkills().addExperience(SkillHandler.HERBLORE, i.getXp());
		player.getActionSender().sendMessage(CLEAN_MESSAGE);
		
	}
	
}

package org.hyperion.rs2.model.content.skills.herblore;

import org.hyperion.rs2.model.Animation;
import org.hyperion.rs2.model.Item;
import org.hyperion.rs2.model.Player;
import org.hyperion.rs2.model.content.skills.SkillHandler;


/**
 * @author Tom (get on mehh)
 */

public class Potions {

	private static final int VIAL_OF_WATER = 227;
	private static final int POTION_MAKING_ANIMATION = 363;
	
	/**
	 * @author Tom (get on mehh)
	 */
	public enum PotionData {
		//TODO: Add the rest...
		GUAM_POTION(new Item(VIAL_OF_WATER), new Item(249), new Item(91), 3, 12.5),
		ATTACK_POTION(new Item(91), new Item(221), new Item(121), 3, 12.5);
		
		PotionData(final Item vial, final Item ingredient, final Item potion, final int level, final double exp) {
			this.vial = vial;
			this.ingredient = ingredient;
			this.potion = potion;
			this.level = level;
			this.exp = exp;
		}
		
		final Item vial, ingredient, potion;
		final int level;
		final double exp;
		
		public Item getItem() {
			return vial;
		}
		
		public Item getIngredient() {
			return ingredient;
		}
		
		public Item getPotion() {
			return potion;
		}
		
		public int getLevel() {
			return level;
		}
		
		public double getExperience() {
			return exp;
		}
		
		public static PotionData isPotion(final Item item, final Item item2) {
			for(PotionData p : PotionData.values()) {
				if(p == null)
					continue;
				if((item.getId() == p.getItem().getId() && item2.getId() == p.getIngredient().getId()) || (item.getId() == p.getIngredient().getId() && item2.getId() == p.getItem().getId())) {
					return p;
				}
			}
			
			return null;
		}
	}
	
	public static boolean canMakePotion(final Player player, final Item usedItem, final Item usedWith) {
		PotionData potData = PotionData.isPotion(usedItem, usedWith);
		if (potData == null)
			return false;
		if (!SkillHandler.skillEnabled(player, SkillHandler.HERBLORE))
			return false;
		if (!SkillHandler.checkSkill(player, SkillHandler.HERBLORE, potData.getLevel(), "make this potion."))
			return false;
		makePotion(player, usedItem, usedWith);
		return true;
	}
	
	private static void makePotion(final Player player, final Item usedItem, final Item usedWith) {
		PotionData potData = PotionData.isPotion(usedItem, usedWith);
		Animation.create(POTION_MAKING_ANIMATION);
		player.getInventory().remove(usedItem);
		player.getInventory().remove(usedWith);
		player.getInventory().add(potData.getPotion());
		if (potData.getItem().getId() == VIAL_OF_WATER)
			player.getActionSender().sendMessage("You put the herb into the vial of water.");
		else
			player.getActionSender().sendMessage("You mix the ingredient into the unfinished potion and make a complete potion.");
	}
	
}

package org.hyperion.rs2.model.content.skills;

import org.hyperion.rs2.model.Player;

/**
 * @author Tom (get on mehh)
 */

public class SkillHandler {
	
	public static final int ATTACK = 0, DEFENCE = 1, STRENGTH = 2, HITPOINTS = 3, RANGED = 4, PRAYER = 5, MAGIC = 6, COOKING = 7, 
			WOODCUTTING = 8, FLETCHING = 9, FISHING = 10, FIREMAKING = 11, CRAFTING = 12, SMITHING = 13, MINING = 14, HERBLORE = 15, 
			AGILITY = 16, THIEVING = 17, SLAYER = 18, FARMING = 19, RUNECRAFTING = 20;
	
	public static final String[] SKILL_NAME = {"Attack", "Defence", "Strength", "Hitpoints", "Ranged", "Prayer", "Magic", "Cooking", 
		"Woodcutting", "Fletching", "Fishing", "Firemaking", "Crafting", "Smithing", "Mining", "Herblore", "Agility", "Thieving", 
		"Slayer", "Farming", "Runecrafting"};
	
	private static final boolean[] SKILL_ENABLED = {false, false, false, false, false, false, false, false, false, false, false, false,
		false, false, false, true, false, false, false, false, false};
	
	public static boolean checkSkill(Player player, int skill, int levelReq, String event) {
		if (!player.getSkills().hasLevel(skill, levelReq))
			player.getActionSender().sendMessage("You need "+levelReq+" "+SKILL_NAME[skill]+" to "+event);
		return player.getSkills().hasLevel(skill, levelReq) ? true : false;
	}
	
	public static boolean skillEnabled(Player player, int skill) {
		if (!SKILL_ENABLED[skill])
			player.getActionSender().sendMessage("This skill is currently disabled.");
		return SKILL_ENABLED[skill] ? true : false;
	}
	
	
}
