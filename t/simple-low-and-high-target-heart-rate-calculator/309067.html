import javax.swing.*;

public class thrcalc {
	
	public static int restingheartrate(int heartBeats){
		int RHR = heartBeats * 6;
		return RHR;
	}
	
	public static int targetheartrate(int age, int maxHeartRate) {

		int maxRateMinusAge = maxHeartRate - age;
		return maxRateMinusAge;
		
	}
		
	public static double lowrange(double lowHeartRangePercent, int maxRateMinusAge) {
		
		double lowHeartRange = maxRateMinusAge * lowHeartRangePercent;
		return lowHeartRange;
	}
	
	public static double highrange(double highHeartRateRangePercent, int maxRateMinusAge) {
		
		double highHeartRateRange = maxRateMinusAge * highHeartRateRangePercent;
		return highHeartRateRange;
	}
		
	public static void main(String[] args) {
		
		String sHeartBeats, sAge;
		int heartBeats, RHR, maxHeartRate, age, maxRateMinusAge;
		double lowHeartRateRange, lowHeartRangePercent, highHeartRateRange, highHeartRateRangePercent;
		
		sHeartBeats = JOptionPane.showInputDialog("count of heartbeats for 10 seconds");
				heartBeats = Integer.parseInt(sHeartBeats);
				
		sAge = JOptionPane.showInputDialog("How old are you?");
			age = Integer.parseInt(sAge);
			
			maxHeartRate = 220;
			
			lowHeartRangePercent = .65;
			highHeartRateRangePercent = .85;

		RHR = restingheartrate(heartBeats);
		
		maxRateMinusAge = targetheartrate(age, maxHeartRate);
		
		lowHeartRateRange = lowrange(lowHeartRangePercent, maxRateMinusAge);
		
		highHeartRateRange = highrange(highHeartRateRangePercent, maxRateMinusAge);
		
			JOptionPane.showMessageDialog(null, "your resting heart rate is: " + RHR);
			JOptionPane.showMessageDialog(null, "your low heart rate range is: " + lowHeartRateRange);
			JOptionPane.showMessageDialog(null, "your high heart rate range is: " + highHeartRateRange);
			
		
	}
}

why is int in a bracket, also where did my maxRateMinusAge var go?

public static int targetheartrate(int age, int maxHeartRate) {
    //int maxRateMinusAge = maxHeartRate - age; - See Comment B
    return (int)/*See Comment A*/(maxHeartRate - age);
}

Comment A:

The int is there to basically say, whatever is next it’s a integer, it’s a conversion.
If the return statement came back as a double (But in this case it wont as you doing int - int and that can’t result in a double), you would get runtime errors as you’ve defined this method to return a integer type, not a double.double.

Comment B:

The reason that line is not needed is because you can just do all the calculations in the return statement.
