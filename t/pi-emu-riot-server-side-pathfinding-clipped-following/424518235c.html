    public void findRoute(int i) {
	final int x = npcs[i].getX() + npcs[i].moveX;
	final int y = npcs[i].getY() + npcs[i].moveY;
	boolean canWalk = Region.getClipping(x, y, npcs[i].heightLevel, npcs[i].moveX, npcs[i].moveY);
	for (int p = 0; p < maxNPCs; p++) {
	    if (npcs[p] == null) {
		continue;
	    }
	    if (npcs[p].absX == npcs[i].absX + npcs[i].moveX && npcs[p].absY == npcs[i].absY + npcs[i].moveY) {
		npcs[i].moveX = 0;
		npcs[i].moveY = 0;
		break;
	    }
	}
	if (canWalk) {
	    npcs[i].getNextNPCMovement(i);
	} else {
	    npcs[i].moveX = 0;
	    npcs[i].moveY = 0;
	}
	npcs[i].updateRequired = true;
    }
