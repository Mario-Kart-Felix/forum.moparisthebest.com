Heres an old script I wrote while figuring out the updateserver protocol for my new engine private server:

import java.net.*;
import java.io.*;

public class VersionCheck {

	public static String 	rAdr = "au1.runescape.com";		// Remote Server
	public static int 	rPrt = 43594;					// Remote Port
	public static int	sVer = 420;							// Initial Client Version

	public static void main(java.lang.String args[]){
		System.out.println("Current Version [ SpuneRace Project ]");
		System.out.println("Version 1.0 (C)2006 Troy Osborne [ defyboy@gmail.com ]");
		System.out.println("----------------------");
		VersionCheck v = new VersionCheck();
		v.connect();
	}
	
	public void connect(){
		try{
			System.out.println("Connecting to server [ " + rAdr + ":" + rPrt + " ]...");
			while(sVer < 500){
				skt = new Socket(rAdr, rPrt);
				ssi = skt.getInputStream();
				sso = skt.getOutputStream();
				System.out.print("Checking Version: " + sVer);
				sso.write(15);
				sso.write(0);
				sso.write(0);
				sso.write(sVer >> 8);
				sso.write(sVer);
				sso.flush();
				int r = ssi.read();	// Response (0=OK, 6=Out of date)
				if(r == 0){
					System.out.println(" OK, Version found: " + sVer);
					System.out.println("----------------------");
					System.exit(0);
				}else{
					System.out.println(" Failed...");
					sVer++;
					sso.close();
					ssi.close();
					skt.close();
				}
			}
			System.out.println("----------------------");
			System.out.println("Couldnt find version.");
		}catch(Exception _ex){
			System.out.println("Failed!");
			System.out.println("An error occurred connecting to server, Check your settings.");
			System.exit(2);
		}
	}

	public Socket skt;
	public InputStream ssi;
	public OutputStream sso;
	
}
