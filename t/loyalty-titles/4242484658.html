public enum LoyaltyTitle {
    SIR(1000),
    LORD(1000),
    LIONHEART(4000),
    HELLRAISER(8000),
    DESPERADO(10000),
    COUNT(15000),
    OVERLORD(15000),
    DUKE(20000),
    KING(25000),
    BIG_CHEESE(25000),
    BIGWIG(25000),
    WUNDERKIND(50000);

    private final int cost;

    private LoyaltyTitle(final int cost) {
        this.cost = cost;
    }

    public int getCost() {
        return cost;
    }
}
