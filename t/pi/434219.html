public void loginMessage() {
for (int j = 0; j < Server.playerHandler.players.length; j++) {
if (Server.playerHandler.players[j] != null) {
Client c2 = (Client)Server.playerHandler.players[j];
switch (playerRights){
case 0: //Regular Player
c2.sendMessage("[PLAYER] “+playerName+” has logged in.");
break;
case 1: //Donator
c2.sendMessage("[DONATOR] “+playerName+” has logged in.");
break;
case 2:
c2.sendMessage("@gre@[Moderator] “+playerName+” has logged in.");
break;
case 3:
c2.sendMessage("[GLOBAL MODERATOR] “+playerName+” has logged in.");
break;
case 4:
c2.sendMessage("@yel@[Adminstrator “+playerName+” has logged in.");
break;
case 5:
c2.sendMessage("@red@[Owner] “+playerName+” has logged in.");
break;
}
}
}
}
