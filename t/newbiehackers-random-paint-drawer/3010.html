Wrote this while I was at school (I dont have paint at home)
open paint, select the whole window (I think), change the size of the drawing bit to maximum (but don’t let there be any scrollbars, then run it, each loop it chooses a random colour and then draws a new line.
also, choose the paint bucket tool to just make it flash real fast

here 'tis

{.Script Info:
# ScriptName  = Random line drawer
# Author      = newbiehacker
# Description = Draws multicoloured lines all over the canvas
# Version     = 1.0
# Date        = Unknown
# Comments    = Blah
/Script Info}
const
  topX = 3;
  topY = 3;
  bottomX = 960;
  bottomY = 612;

var
  x, y: Integer;
  
procedure chooseRandomColor;
begin
  x:= -11 + random(205 + 11);
  y:= 626 + random(654 - 626);
  moveMouse(x, y);
  clickMouse(x, y, true);
end;

  
begin
  repeat
    x:= topX + random(bottomX);
    y:= topY + random(bottomY);
    moveMouse(x, y);
    holdMouse(x, y, true);
    moveMouse(topX + random(bottomX - topX), topY + random(bottomY - topY));
    getMousePos(x, y);
    releaseMouse(x, y, true);
    chooseRandomColor;
    wait(10);
  until(false);
end.
