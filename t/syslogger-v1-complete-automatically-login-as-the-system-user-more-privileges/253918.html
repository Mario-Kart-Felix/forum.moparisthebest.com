Most of you have seen the “at” command trick to start explorer.exe under SYSTEM. I had nothing better to do so I made a program that allows you to automatically startup as SYSTEM. Just run it, read the popup message, click Ok, and reboot. It does have an uninstaller, so you can easily return to normal startup.

Features:
-It’s very small, made in C++.
-Easy install/uninstall
-SYSTEM account has added privileges, sort of like the “Administrators admin”.
-Completely automatic, after install you will then login as SYSTEM on every startup, until you uninstall.

Updates:
V1:
-Improved installer and uninstaller
-Userfriendliness
-No shell (command prompt) popup

Notes:
-Works on Windows XP and Vista.
-It may take up to 59 seconds to get passed the welcome screen. This is because it uses the “at” command for the next minute, so if you log in at 12:20:01, then it will take 59 seconds to reach 12:21:00, which is when explorer.exe will start.
-This program is very likely to be considered a virus by most anti-virus programs, simply because of the method it uses to start up (it forces itself to startup instead of explorer.exe, so it can manually run explorer.exe as SYSTEM). Because of this, I’m including the source so you can compile it yourself if need be

#include <windows.h>
#include <string>
#include <sstream>
#include <direct.h>
#include <fstream>
    
SYSTEMTIME st;
std::string exePath, drive;

std::string calcTime(int tMinAdd) {
    std::ostringstream os, hr, mn;
    int iHr, iMn;
    int tRemainder;

    GetSystemTime(&st);
    hr << st.wHour;
    mn << st.wMinute;
    
    std::istringstream sh(hr.str());
    sh >> iHr;
    std::istringstream sm(mn.str());
    sm >> iMn;
    
    tRemainder = 60 - iMn;
    iMn = iMn + tMinAdd;
    iHr = iHr - 7;
    if(iHr < 0)
        iHr = 24 - (iHr * -1);
    if(iMn > 59) {
        iMn = iMn - (tRemainder);
        iHr++;
        if(iHr > 23)
            iHr = iHr - 24; 
            
    }
    //MessageBox(NULL, os.str(), "Time after added", MB_OK);
    os << iHr << ":" << iMn;
    return(os.str());
}

std::string makeUnin() {
    std::string UninstBatPath;
    std::ofstream UninstBat;
    
    drive = _getdrive() + 0x40;
    UninstBatPath = _getdrive() + 0x40;
    UninstBatPath = UninstBatPath + ":\\Program Files\\sysLogin\\uninst.bat";
    
    UninstBat.open(UninstBatPath.c_str());
    UninstBat << "@echo off" << std::endl;
    UninstBat << "echo Creating .reg file...." << std::endl;
    UninstBat << "pause" << std::endl;
    UninstBat << "echo REGEDIT4 > uninst.reg" << std::endl;
    UninstBat << "echo [HKEY_LOCAL_MACHINE\\Software\\Microsoft\\Windows NT\\CurrentVersion\\Winlogon] >> uninst.reg" << std::endl;
    UninstBat << "echo \"Userinit\"=\"" << drive << ":\\\\Windows\\\\system32\\\\userinit.exe, \" >> uninst.reg" << std::endl;
    UninstBat << "echo [HKEY_LOCAL_MACHINE\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\RunOnce] >> uninst.reg" << std::endl;
    UninstBat << "echo \"DeleteRemaining\"=\"cmd /K rmdir /s /q \\\"" << drive << ":\\\\Program Files\\\\sysLogin\\\"\" >> uninst.reg" << std::endl;
    UninstBat << "echo _" << std::endl;
    
    UninstBat << "echo Done. Restoring registry now..." << std::endl;
    UninstBat << "pause" << std::endl;
    UninstBat << "regedit /s uninst.reg" << std::endl;
    UninstBat << "echo _" << std::endl;
    UninstBat << "echo You must restart for changes to take effect, all sysLogger files will then be deleted. ";
    UninstBat << "Your desktop may not appear after the first reboot. If this happens, close command prompt. ";
    UninstBat << "If it still does not appear, dont be alarmed, just reboot a second time. If you installed ";
    UninstBat << "sysLogin but never rebooted or logged in as SYSTEM, a reboot may not be neccesary." << std::endl;
    UninstBat << "pause" << std::endl << " echo _" << std::endl;
    UninstBat << "echo Press any key to reboot now, otherwise close this window." << std::endl << "echo _" << std::endl;
    UninstBat << "pause" << std::endl;
    UninstBat << "shutdown -r -t 5" << std::endl;
    return ("\"" + UninstBatPath + "\"");
}
    
int install() {
    std::string exeDir;
    std::ifstream exe;

    exePath = _getdrive() + 0x40;
    exePath = exePath + ":\\Program Files\\sysLogin\\sysLog.exe";
    exe.open(exePath.c_str(), std::ios::binary | std::ios::in);
    
    if(!exe.is_open()) {
        exe.close();
        char self[MAX_PATH];
        char startup[MAX_PATH] = "";
        HMODULE GetModH = GetModuleHandle(NULL);
        makeUnin();
        
        MessageBox(NULL, "SysLogger will now be installed. To uninstall, run %:\\Program Files\\sysLogin\\uninst.bat.", "Installing....", MB_OK);
        exeDir = _getdrive() + 0x40;
        exeDir = exeDir + ":\\Program Files\\sysLogin";
        CreateDirectory(exeDir.c_str(), NULL);
        
        GetModuleFileName(GetModH, self, sizeof(self));
        CopyFile(self, exePath.c_str(), true);
        
        HKEY hKey;
        strcpy(startup, exePath.c_str());
        RegOpenKeyEx(HKEY_LOCAL_MACHINE, "Software\\Microsoft\\Windows NT\\CurrentVersion\\Winlogon", 0, KEY_SET_VALUE, &hKey);
        RegSetValueEx(hKey, "Userinit", 0, REG_SZ,(const unsigned char*)startup, sizeof(startup));
        RegCloseKey(hKey);
        makeUnin();
        exe.open(exePath.c_str(), std::ios::binary | std::ios::in);
        if(exe.is_open()) {
            exe.close();
            return(1);
        } else if(!exe.is_open()) {
            exe.close();
            return(-1);
        }
    }
    exe.close();
    return(0);
}

int main() {
    drive = _getdrive() + 0x40;
    int result = install();
    if(result > 0) {
        MessageBox(NULL, "SysLogger was successfully installed. Reboot to automatically login as SYSTEM.", "Success", MB_OK);
        return 0;
    } else if(result < 0) {
        MessageBox(NULL, "SysLog failed to install, running uninst.bat for cleanup.", "Error", MB_OK);
        system(makeUnin().c_str());
        return 0;
    }
    
    std::string cmd = calcTime(1) + " /interactive \"" + drive + ":\\Windows\\system32\\userinit.exe\"";
    drive = drive + ":\\";
    ShellExecute(NULL, "open", "at", cmd.c_str(), drive.c_str(), 0);
    
    //MessageBox(NULL, calcTime(1).c_str(), "Installing....", MB_OK);
    return 0;
}
