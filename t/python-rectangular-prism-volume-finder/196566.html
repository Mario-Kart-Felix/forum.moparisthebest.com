	Volume = Width * Height * Depth
	print("Volume of the cube is: " + str(Volume))
def getIntInput(prompt):
	theInput = raw_input(prompt)
	if(isNumber(theInput)):
		return int(theInput)
	else:
		print "Invalid number: " + theInput
		return getIntInput(prompt)
def isNumber(n):
	import re
	if re.match("^[-+]?[0-9]+$", n):
		return True
	return False
main()
