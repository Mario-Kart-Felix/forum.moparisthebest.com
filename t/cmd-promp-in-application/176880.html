public class Test {
	public static void main(String[] args) {
		System.out.println("Test 1");
		javax.swing.JFrame frame = new javax.swing.JFrame("Test Frame");
		frame.setSize(300, 300);
		javax.swing.JTextArea area = new javax.swing.JTextArea();
		java.io.PrintStream stream = System.out;
		System.setOut(new TestPrintStream(area, stream));
		frame.add(new javax.swing.JScrollPane(area));
		frame.setVisible(true);
		System.out.println("Test 2");
		System.setOut(new TestPrintStream(null, stream));
		System.out.println("Test 3");
	}
}

private class TestPrintStream extends java.io.PrintStream {
	javax.swing.TextArea area;

	public TestPrintStream(javax.swing.TextArea area, java.io.PrintStream stream) {
		super(stream);
		this.area = area;
	}

	public void println(String str) {
		if (area != null)
			area.setText(area.getText() + str + "\n");
		else
			super.println(str);
	}
}
