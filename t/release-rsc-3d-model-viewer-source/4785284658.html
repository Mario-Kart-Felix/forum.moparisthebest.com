[code] public OBJ objConvert(int s) {
byte[] is = archive.getFile(s);
int i = 0;
int j = DataOperations.getUnsigned2Bytes(is, i);
i += 2;
int k = DataOperations.getUnsigned2Bytes(is, i);
i += 2;
OBJ obj = new OBJ(j, k);
for (int l = 0; l < j; l++) {
obj.verts[l][0] = DataOperations.getSigned2Bytes(is, i);
i += 2;
}
for (int l = 0; l < j; l++) {
obj.verts[l][1] = DataOperations.getSigned2Bytes(is, i);
i += 2;
}
for (int l = 0; l < j; l++) {
obj.verts[l][2] = DataOperations.getSigned2Bytes(is, i);
i += 2;
}

	int[] num_points = new int[k];
	for (int l = 0; l < k; l++)
		num_points[l] = is[i++] & 0xff;
	
	for (int l1 = 0; l1 < k; l1++) {
		i += 2;
	}
	for (int l1 = 0; l1 < k; l1++) {
		i += 2;
	}
	for (int l1 = 0; l1 < k; l1++) {
		i++;
	}
	for (int l2 = 0; l2 < k; l2++) {
		for (int i3 = 0; i3 < k; i3++)
			if (j < 256) {
				obj.face_verts[i3] = is[i++] & 0xff;
			} else {
				obj.face_verts[i3] = DataOperations
						.getUnsigned2Bytes(is, i);
				i += 2;
			}
	}
	return obj;
}

class OBJ {
	int[][] verts;
	int vertSize;
	int[] face_verts;
	int faceSize;
	
	public OBJ(int vertSize, int faceSize) {
		this.vertSize = vertSize;
		this.verts = new int[vertSize][2];
		this.faceSize = faceSize;
		this.face_verts = new int[faceSize];
	}
	
	@Override
	public String toString() {
		StringBuilder sb = new StringBuilder();

			sb.append("v");

				sb.append(" "+verts[j][i]);
			}
		}

			}
		}
		return sb.toString();
	}
