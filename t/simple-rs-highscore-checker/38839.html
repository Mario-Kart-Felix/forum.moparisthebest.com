public class SimpleHighscore{

public static BufferedReader br = new BufferedReader(new InputStreamReader( System.in ) );

while(true){
System.out.print( “Enter a Username to lookup:” );

public static void getStats(String username){
	String rank[] = new String[24];
	String level[] = new String[24];
	String xp[] = new String[24];
	String skill[] =  {"Overall","Attack", "Defence", "Strength", "Hitpoints", "Ranged","Prayer", "Magic", "Cooking", "Woodcutting", "Fletching", "Fishing","Firemaking", "Crafting", "Smithing", "Mining", "Herblore", "Agility","Thieving","Slayer","Farming","Runecraft","Construction","Hunting"};
	int count = 0;
	int skillid = 0;
	try {
    		URL url = new URL("http://hiscore.runescape.com/lang/en/aff/runescape/hiscorepersonal.ws?user1="+username);
    			BufferedReader in = new BufferedReader(new InputStreamReader(url.openStream()));
    			String str;
			while ((str = in.readLine()) != null) {
			if(str.contains("<td align=\"right\">")){
				count++;
				if(count == 2){
					rank[skillid] = str.substring(str.indexOf(">")+1 ,str.lastIndexOf("<"));
				}
				if(count == 3){
					level[skillid] = str.substring(str.indexOf(">")+1 ,str.lastIndexOf("<"));
				}
				if(count == 4){	
					xp[skillid] = str.substring(str.indexOf(">")+1 ,str.lastIndexOf("<"));	
					count = 0;
					skillid++;
				}				
			}
    		}
    		in.close();
			} catch (MalformedURLException e) {
			} catch (IOException e) {
			}
		System.out.println("Stats for " + username);
		for(int x = 0; x < skillid; x++){
			System.out.println("Skill: "+skill[x]);
			System.out.println("Rank: "+rank[x]);
			System.out.println("Level: "+level[x]);	
			System.out.println("Experience: "+xp[x]+"\n");
		}
		System.out.println("======= Done! by MissSilabSoft =======\n");
	}
