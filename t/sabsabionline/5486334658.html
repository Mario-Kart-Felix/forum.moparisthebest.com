It‚Äôs been a busy week with xmas and everything. I managed to write a equipment dumper (dumps the equip slots). We read the data from a text file (dont laugh plz). But yeah, this should ensure that no items equip the wrong way.

for people who needs this.

	public boolean isWieldable() {
		try {
			if (actions == null)
				return false;
			String menu = actions[1].toLowerCase();
			return (menu.equals("wear") || menu.equals("wield") || menu
					.equals("equip"));
		} catch (Exception e) {
			return false;
		}
	}

	public static void dumpEquipments() {
		try {
			FileWriter fw = new FileWriter("equips.txt");
			for (int j = 0; j < totalItems; j++) {
				ItemDef item = ItemDef.forID(j);
				if (item == null || item.name == null || !item.isWieldable())
					continue;
				String name = item.name.toLowerCase();
				int slot = 3;
				if (name.contains("body") || name.contains("torso")
						|| name.contains("robe top")
						|| name.contains("robetop") || name.contains("jacket")
						|| name.contains("tunic") || name.contains("shirt")
						|| name.contains("top") || name.contains("blouse")
						|| name.contains("gown") || name.contains("hauberk"))
					slot = 4;
				else if (name.contains("shield") || name.contains("defender")
						|| name.contains("book"))
					slot = 5;
				else if (name.contains("boots"))
					slot = 10;
				else if (name.contains("cape") || name.contains("backpack")
						|| name.contains("apparatus")
						|| name.equalsIgnoreCase("eye patch")
						|| name.contains("accumulator")
						|| name.contains("cloak"))
					slot = 1;
				else if (name.contains("hat") || name.contains("helm")
						|| name.contains("mask") || name.contains("bandana")
						|| name.contains("cavalier") || name.contains("beret")
						|| name.contains("coif") || name.contains("wreath")
						|| name.contains("hood") || name.contains("sallet")
						|| name.contains("cowl"))
					slot = 0;
				else if (name.contains("legs") || name.contains("skirt")
						|| name.contains("bottoms") || name.contains("pants")
						|| name.contains("leggin") || name.contains("shorts")
						|| name.contains("tasset") || name.contains("chaps")
						|| name.contains("druidic robe")
						|| name.contains("cuisse"))
					slot = 7;
				else if (name.contains("amulet") || name.contains("scarf")
						|| name.contains("necklace"))
					slot = 2;
				else if (name.contains("ring"))
					slot = 12;
				else if (name.contains("glove") || name.contains("vambrace"))
					slot = 9;
				else if (name.contains("arrow") || name.contains("bolt"))
					slot = 13;
				fw.write(j + " " + slot + "\r\n");
			}
			fw.close();
		} catch (IOException ioe) {
			ioe.printStackTrace();
		}
	}
