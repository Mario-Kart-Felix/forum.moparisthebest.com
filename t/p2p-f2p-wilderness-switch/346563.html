	/**
	 * Wilderness Switch @author Yong Min

	public boolean startWildernessSwitch() {
		if(wildSwitch != null) {
			return false;
		}
		wildSwitch = new SingleEvent(null, 66200) {
			public void action() {
				if(world.getWildernessType())
					world.setWildernessType(false);
				else {
					world.setWildernessType(true);
				}
				for(Player p : world.getPlayers()) {
					String wildernessType;
					if(world.getWildernessType())
						wildernessType = "@gre@P2P";
					else {
						wildernessType = "@gre@F2P";
					}
					p.getActionSender().sendMessage("The wilderness has been changed to: " + wildernessType);
					if(p.getLocation().wildernessLevel() > 0) {
						for(InvItem wieldedItem : p.getInventory().getItems()) {
							for(int items : world.getP2PItems()) {
								if((wieldedItem.getID() == items) && (wieldedItem.isWielded())) {
									wieldedItem.setWield(false);
									p.updateWornItems(wieldedItem.getWieldableDef().getWieldPos(), p.getPlayerAppearance().getSprite(wieldedItem.getWieldableDef().getWieldPos()));
									p.getActionSender().sendSound("click");
									p.getActionSender().sendEquipmentStats();
									p.getActionSender().sendInventory();
									world.setWildernessSwitching(false);
								}
							}
						}
						if(p.getCurStat(0) > p.getMaxStat(0)) {
							p.setCurStat(0, p.getMaxStat(0));
						}
						if(p.getCurStat(1) > p.getMaxStat(1)) {
							p.setCurStat(1, p.getMaxStat(1));
						}
						if(p.getCurStat(2) > p.getMaxStat(2)) {
							p.setCurStat(2, p.getMaxStat(2) + DataConversions.roundUp(p.getMaxStat(2) / 100.0D * 10.0D) + 2);
						}
						if(p.getCurStat(3) > p.getMaxStat(3)) {
							p.setCurStat(3, p.getMaxStat(3));
						}
						if(p.getCurStat(4) > p.getMaxStat(4)) {
							p.setCurStat(4, p.getMaxStat(4));
						}
						p.getActionSender().sendStats();
						wildSwitch = null;
					}
				};
			}
		};
		world.getDelayedEventHandler().add(wildSwitch);
		return true;
	}

		for(Player p : world.getPlayers())
			if((!(world.getWildernessType())) && (p.getLocation().wildernessLevel() > 0)) {
				for(InvItem currentItem : p.getInventory().getItems()) {
					for(int index : world.getP2PItems()) {
						if(((currentItem.getID() == index) && (currentItem.isWielded()))) {
							currentItem.setWield(false);
							p.updateWornItems(currentItem.getWieldableDef().getWieldPos(), p.getPlayerAppearance().getSprite(currentItem.getWieldableDef().getWieldPos()));

[b]MASTERFISHER UPDATE: This will check the current wilderness type and give a different fish to the players depending on the wilderness type. It is currently set to Lobsters for F2P and Sharks for P2P.

package org.rscdaemon.server.npchandler;

import org.rscdaemon.server.model.*;
import org.rscdaemon.server.event.ShortEvent;

public class MasterFisher implements NpcHandler {
	public static final World world = World.getWorld();
	
	public void handleNpc(final Npc npc, Player player) throws Exception {
		String fishName;
		int fishID;
		String wildernessType;
		
		if(!world.getWildernessType()) {
			fishID = 373;
			wildernessType = "F2P";
			fishName = "Lobster";
		} else {
			fishID = 546;
			wildernessType = "P2P";
			fishName = "Shark";
		}
		final int finalFishID = fishID;
		String finalFishName = fishName;
		player.informOfNpcMessage(new ChatMessage(npc, "The wilderness is " + wildernessType + ", would you like some " + finalFishName + "?", player));
		player.setBusy(true);
		world.getDelayedEventHandler().add(new ShortEvent(player) {
			public void action() {
				owner.setBusy(false);
				String[] options = { "Yes please" };
				owner.setMenuHandler(new MenuHandler(options) {
					public void handleReply(final int option, String reply) {
						if(owner.isBusy()) {
							return;
						}
						owner.informOfChatMessage(new ChatMessage(owner, reply, npc));
						owner.setBusy(true);
						world.getDelayedEventHandler().add(new ShortEvent(owner) {
							public void action() {
								switch(option) {
									case 0:
										world.getDelayedEventHandler().add(new ShortEvent(owner) {
											public void action() {
												do {
													if(owner.getInventory().size() >= 30) {
														return;
													}
													owner.setBusy(false);
													owner.getInventory().add(new InvItem(finalFishID, 1));
													owner.getActionSender().sendInventory();
													npc.unblock();
												}
												while(owner.getInventory().size() != 30);
												npc.unblock();
											}
										});
									}
									owner.setBusy(false);
									npc.unblock();
								}
							});
						}
					});
					owner.getActionSender().sendMenu(options);
				}
			});
			npc.blockedBy(player);
		}
	}
