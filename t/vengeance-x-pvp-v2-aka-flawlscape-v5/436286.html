public static final String findcachedir() { String s; String s1; File file; s = "c:/.YOURCLIENTNAME_Cache/"; s1 = ""; file = new File((new StringBuilder()).append(s1).append(s).toString()); if(file.exists() || file.mkdir()) { return (new StringBuilder()).append(s1).append(s).append("/").toString(); } return null; }
