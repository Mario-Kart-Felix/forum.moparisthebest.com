public TeleportHandler(Player p, Location location, int type) {
	handleTeleport(p, new Teleport(location, type));
}


private void handleTeleport(final Player player, final Teleport teleport) {
	if(System.currentTimeMillis() - player.getButtonDelay() < TeleportConstants.buttonDelay) return;
	player.playAnimation(Animation.create(TeleportConstants.teleportStyles[teleport.getType()].getAnim()));
	player.playGraphics(Graphic.create(TeleportConstants.teleportStyles[teleport.getType()].getGFX(), TeleportConstants.teleportStyles[teleport.getType()].getGFXDelay()));
	World.getWorld().submit(new Event(2000) {
		@Override
		public void execute() {
				player.playAnimation(Animation.create(TeleportConstants.teleportStyles[teleport.getType()].getAnim2()));
				player.setLocation(teleport.getLocation());				
				stop();
		}
		
	});
	
}
