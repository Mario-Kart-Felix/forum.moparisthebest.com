This? I just typed it up from scratch really quickly.

[code]program Dropper;
{.include SRL/SRL.scar}

const
ItemColor = 00000; //the color of the item in your inv
ItemTolerance = 3; //the higher, the less accurate the color needs ot be

Procedure DropIt;
begin
repeat
FindColorTolerance(x,y,ItemColor,MIX1, MIY1, MIX2,MIY2,ItemTolerance)
Wait(100+ random(100));
Mouse(x, y, 1, 1, False);
Wait(200 + random(100));
ChooseOption(‘rop’);
until not (FindColorTolerance(x,y,ItemColor,MIX1, MIY1, MIX2,MIY2,ItemTolerance))
end;

begin
SetupSRL;
DropIt;
end;[/code]

I fixed the script a little bit. Because colors are not dependable, I added the IsUpText.
Just make sure you don’t choose any capitol letters.

[code]program Dropper;
{.include SRL/SRL.scar}
var
dropped:Integer;

const
ItemColor = 1406123; //the color of the item in your inv
ItemTolerance = 2; //the higher, the less accurate the color needs ot be
ItemName = ‘rimp’; //change this to 3 letters out of the name in the top left. Don’t use capitols.

Procedure DropIt;
begin
Dropped:=0;
repeat
if(FindColorTolerance(x,y,ItemColor,MIX1, MIY1, MIX2,MIY2,ItemTolerance))then
begin
wait(100)
MMouse(x,y,2,2);
if(IsUpText(ItemName))then
begin
GetMousePos(x,y);
Mouse(x,y,1,1,false);
Wait(200+ random(100));
ChooseOption(x,y,‘rop’);
Dropped:=Dropped+1;
end;
end;
until not (FindColorTolerance(x,y,ItemColor,MIX1, MIY1, MIX2,MIY2,ItemTolerance))
Writeln(‘Dropped ’ +inttostr(Dropped) +’ "’+itemname +’"s. Terminating script.’)
end;

begin
SetupSRL;
DropIt;
TerminateScript;
end.[/code]
