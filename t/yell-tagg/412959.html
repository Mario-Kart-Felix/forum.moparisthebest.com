	public void yell(final String Message) {
		synchronized (Message) {
			try {
				for (final Player p : PlayerHandler.players) {
					if ((Message.indexOf("tradereq") > 0)
							|| (Message.indexOf("duelreq") > 0)) {
						return;
					}
					if ((p == null) || !p.isActive) {
						continue;
					}
					final Client temp = (Client) p;
					if ((temp.absX > 0) && (temp.absY > 0)) {
						if ((temp != null) && !temp.disconnected && p.isActive) {
							temp.sendMessage(Message);
						}
					}
				}
			} catch (Exception e) {
				e.printStackTrace();
			}
		}
	}
