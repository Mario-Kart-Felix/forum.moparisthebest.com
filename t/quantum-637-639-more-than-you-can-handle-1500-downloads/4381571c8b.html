Fight Caves snippet for anyone who wants (Recreated from Grumpy’s)

[code=java]package org.dementhium.content.minigames;

import org.dementhium.content.DialogueManager;
import org.dementhium.model.npc.NPC;
import org.dementhium.model.player.Player;
import org.dementhium.util.Misc;
import org.dementhium.model.World;
import org.dementhium.net.ActionSender;
import org.dementhium.model.Location;

public class FightCaves {

private static int waves = 0;

public static int getWave() {
	return waves;
}

public static void setWave(int waves) {
	waves = waves;
}

public static void startCaves(final Player player) {
	player.requestWalk(2438, 5168);
	player.teleport(2412, 5117, 0);
	player.requestWalk(2407, 5090);
	player.setAttribute("teleblock", 999999999);
	DialogueManager.sendDialogue(player, DialogueManager.CALM_TALK, 2617, -1, "You're on your own now, " + player.getUsername() + ".", "Prepare to fight for your life!");
	sendWave(player, 2734, Location.locate(Misc.random(2386, 2412), Misc.random(5075, 5105), 0), true); //Keep this here and add others in NPC.java
}

public static int Tokkul() {
	return getWave() * 251;
}

public static void addReward(Player player) {
	if (getWave() == 64) {
		player.getInventory().addItem(6570, 1);
		player.getInventory().addItem(6529, Tokkul());
	} else
		player.getInventory().addItem(6529, Tokkul());
}

public static void endCaves(final Player player ) {
	if (getWave() == 64) {
		player.teleport(2439, 5169, 0);
		player.getSkills().restoreEverything(player);
		DialogueManager.sendDialogue(player, DialogueManager.CALM_TALK, 2617, -1, "You have defeated TzTok-Jad, I am most impressed!", "Please accept this gift as a reward.");
		addReward(player);
	} else {
		player.removeAttribute("teleblock");
		DialogueManager.sendDialogue(player, DialogueManager.CALM_TALK, 2617, -1, "Do not feel bad human,", "I knew you couldn't do it.");
		player.teleport(2438, 5168, 0);
		addReward(player);
	}
}


public static void sendWave(final Player player, int npcId, Location location, boolean isNextWave) {
	NPC waveNpc = new NPC(npcId, location);
	if (isNextWave && getWave() == 64) {
		endCaves(player);
		return;
	}
	if (isNextWave) {
		setWave(getWave() + 1);
		player.sendMessage("Wave: " + getWave());
	}
	waveNpc.getCombatExecutor().setVictim(player);
}

}[/code]
