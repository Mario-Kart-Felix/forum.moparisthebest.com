[code]program OrangeClicker;

var
BitmapName, x, y: Integer;

procedure LoadBmp;
begin
BitmapName := BitmapFromString(5, 4, ‘F68F90F09C56EEA232DA9’ +
‘110EFA029DC825DEC9538E18B1EE0852A9F4200DA8250FCA049FA’ +
‘9C3AF99943E68649F29069DE7B42F39442F9A02CE99130’);
end;

procedure FindColor;
begin
if(FindBitmap(BitmapName, x, y)) then
MoveMouseSmoothEX(x,y +random(0),20,40,45,25,30);
ClickMouse(x,y,True);
Wait(300)
end;

begin
repeat
LoadBmp;
FindColor;
Until(False)
end.[/code]
