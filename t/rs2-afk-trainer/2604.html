I made this because i got sick of getting logged off while i was doing homework when i was fishing.

This will keep you logged in while your AFK.

[code]program AFK;
{.include SSI2.txt}
var
num:integer;

procedure operation(number:Integer);
begin
case number of
1:begin MMouse(560,210,180,210);wait(1000+Random(2500)) end;//Move Mouse Randomly in stats box
2:begin Mouse(5,5,510,335,false);wait(1000+Random(1000)) end;//Right click randomly in game window
100:begin Mouse(575,175,20,15,True);wait(Random(1000)) end;//Click on Stats Icon
200:begin GameTab(random(7));wait(150+random(500));GameTab(4) end;//Click on random icon then back to inventory
300:begin Mouse(630,175,20,15,True) end;//Click on inventory icon
end;
end;

Procedure AFK;
Begin
num:= Random(10)
if(num >=5)then
begin
operation(100)
while(num >= 5) do
begin
operation(Random(3))
num:= Random(10)
end
operation(300)
end
else
begin
operation(200)
end
End;

begin
Repeat
AFK;
Wait(20000+Random(10000))
Until(False)
end.[/code]
