Well I just looked through my old TeleScape server… and came across this method. It isn’t a command, but it spawns a Zamorak Monk. The Monk then says “Feel my vengeance!” and attacks the closest player to it.

public void zammyMonkAttack() {
for (Player p : server.playerHandler.players)
{

client person = (client)p;
 	if(p != null && person != null) 
	{
	for(int I = 0; I < server.npcHandler.npcs.length; I++)
	{
		if((server.npcHandler.npcs[I] != null) && (server.npcHandler.npcs[I].npcType == 1044))
		{
			server.npcHandler.npcs[I].updateRequired = true;
			server.npcHandler.npcs[I].textUpdateRequired = true;
			server.npcHandler.npcs[I].textUpdate = "Feel my vengeance!";
    			server.npcHandler.npcs[I].StartKilling = playerId;
    			server.npcHandler.npcs[I].RandomWalk = false;
    			server.npcHandler.npcs[I].IsUnderAttack = true;
				}
			}
		}
	}
}

That should do after a little bit of editing.
