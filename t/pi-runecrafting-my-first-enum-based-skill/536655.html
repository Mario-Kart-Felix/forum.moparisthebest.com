package server.model.players.skills.Runecrafting;

import server.Config;
import server.model.players.Client;
import server.util.Misc;

public class Runecrafting extends RuneData {

public static final int PURE_ESS = 7936;
public static final int RUNE_ESS = 1436;


public static void craftRunes(Client c, int objectId, int playerLevel, int essId){

	for (final RCData r : RCData.values()) {

		if (objectId == r.getObjectId()) {

			if(r.getRequiredLevel() > c.playerLevel[20]){
				c.sendMessage("You need "+r.getRequiredLevel()+" to craft "+
						c.getItems().getItemName(r.getRuneId())+"s.", false);
			} else {
				
				if(r.getEssId() == PURE_ESS && !c.getItems().playerHasItem(PURE_ESS)){
					c.sendMessage("You need Pure Essence to craft runes at this altar.", false);
				}
					
				int itemToDelete = 0;
				int itemToDelete2 = 0;
				if(r.getEssId() == RUNE_ESS){
					if(c.getItems().playerHasItem(PURE_ESS) && c.getItems().playerHasItem(RUNE_ESS)){
						itemToDelete = PURE_ESS;
						itemToDelete2 = RUNE_ESS;
					}else if(c.getItems().playerHasItem(PURE_ESS))
						itemToDelete = PURE_ESS;
					else if(c.getItems().playerHasItem(RUNE_ESS))
						itemToDelete = RUNE_ESS;
					else{
						c.sendMessage("You need Rune Essence to craft runes at this altar.", false);
						return;
					}
				} else if (r.getEssId() == PURE_ESS)
					itemToDelete = PURE_ESS;
				
				int amount = c.getItems().getItemAmount(itemToDelete+itemToDelete2);
				int exp = (r.getExpGain() * amount)* Config.RUNECRAFTING_EXP;
				int amountToCraft = amount * getRuneMultiplyer(c, r.getRuneId());
				
				if(amountToCraft == 0)
					amountToCraft = amount;
				
				//Subtract the opposite itemToDelete so it doesn't attempt to take more than the player has.
				c.getItems().deleteItem(itemToDelete, amount-itemToDelete2);
				if(itemToDelete2 != 0)
					c.getItems().deleteItem(itemToDelete2, amount-itemToDelete);
				c.gfx0(186);
				c.startAnimation(791, 0);
				c.getPA().addSkillXP(exp, 20);
				c.sendMessage("You craft "+ amountToCraft +" "+c.getItems().getItemName(r.getRuneId())+"s.", true);
				c.getItems().addItem(r.getRuneId(), amountToCraft);
				c.saveCharacter = true;
			}
		}else {}
	}
}

public static int getRuneMultiplyer(Client c, int runeId){
	for(final RCData r : RCData.values()) {
		if(runeId == r.getRuneId()) {
			if(r.getDivide() != -1){
				return c.getPA().getLevelForXP(c.playerXP[20]) / r.getDivide();
			}
			return 1;
		}
	}
	return 1;
}

public static void handleZMI(Client c){
	int essAmount = 0;
	
	int[] possibleRunes = {
		554, 555, 556, 557, 558, 559, 560,
		561, 562, 563, 564, 565, 566, 9075
	};
	
	if(!c.getItems().playerHasItem(PURE_ESS)){
		c.sendMessage("You need Pure Essence to use this altar.", false);
		return;
	}
	
	essAmount = c.getItems().getItemAmount(PURE_ESS);
	
	c.gfx0(186);
	c.startAnimation(791, 0);
	
	int[] runesToGive = new int[essAmount];
	int[] amountToGive = new int[essAmount];
	int[] expToGive = new int[essAmount];
	
	/*
	 * Calculating what runes you'll get and how many
	 */
	for(int i = 0; i < 28; i++){
		if(essAmount > 0 && i <= runesToGive.length-1){
			runesToGive[i] = possibleRunes[Misc.random(possibleRunes.length-1)];
			amountToGive[i] = Misc.random(essAmount);
			essAmount -= amountToGive[i];
		}else
			continue;
	}
	
	/*
	 * Calculating the XP gained from the runes (It's doubled because of the ZMI altar's effect.
	 */
	for (final RCData r : RCData.values()) {
		for(int q = 0; q < runesToGive.length-1; q++){
			if(runesToGive[q] == r.getRuneId())
				expToGive[q] = (((r.getExpGain() * amountToGive[q])*Config.RUNECRAFTING_EXP)*2);
		}
	}
	
	/*
	 * Removing the pure ess and adding the runes in place
	 */
	for(int j = 0; j < runesToGive.length-1;j++){
		c.getItems().deleteItem(PURE_ESS, amountToGive[j]);
		c.getItems().addItem(runesToGive[j], amountToGive[j]);
		c.getPA().addSkillXP(expToGive[j], 20);
	}
}

package server.model.players.skills.Runecrafting;

public class RuneData {

	public enum RCData {
		AIR(2478, 1, 5, 1436, 556, 10),
		MIND(2479, 2, 6, 1436, 558, 14),
		WATER(2480, 5, 6, 1436, 555, 19),
		EARTH(2481, 9, 7, 1436, 557, 26),
		FIRE(2482, 14, 7, 1436, 554, 35),
		BODY(2483, 20, 8, 1436, 553, 46),
		COSMIC(2484, 27, 8, 7936, 564, 59),
		CHAOS(2487, 35, 9, 7936, 562, 74),
		ASTRAL(17010, 40, 9, 7936, 9075, 82),
		NATURE(2486, 44, 9, 7936, 561, 91),
		LAW(2485, 54, 10, 7936, 563, -1),
		DEATH(2488, 65, 10, 7936, 560, -1),
		BLOOD(30624, 77, 11, 7936, 565, -1);

		private int objectId, requiredLevel, expGain, essId, runeId, divideBy;
		
		private RCData(int objectID, int requiredLevel, int expGain, int essId, int runeId, int divideBy){
			this.objectId = objectID;
			this.requiredLevel = requiredLevel;
			this.expGain = expGain;
			this.essId = essId;
			this.runeId = runeId;
			this.divideBy = divideBy;
		}

		public int getObjectId() {
			return objectId;
		}

		public int getRequiredLevel() {
			return requiredLevel;
		}

		public int getExpGain() {
			return expGain;
		}

		public  int getEssId() {
			return essId;
		}

		public  int getRuneId() {
			return runeId;
		}

		public  int getDivide() {
			return divideBy;
		}
	}
}
