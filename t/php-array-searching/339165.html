function below($needle, $haystack) {
  foreach($haystack as $key => $value) {
    if($value > $needle) {
      $foundKey = $key-1;
      break;
    }
  }
  if($needle >= end($haystack)) $foundKey = key($haystack);
  if(!isset($foundKey) || $foundKey == -1) $foundKey = false;
  return $foundKey;
}
