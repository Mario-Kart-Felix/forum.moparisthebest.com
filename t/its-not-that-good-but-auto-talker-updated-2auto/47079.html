Procedure Shift(b : byte);
Begin
  Wait(20+Random(30));
  KeyDown(16)
  Wait(20+Random(30));
  KeyDown(b);
  Wait(20+Random(30));
  KeyUp(b)
  Wait(20+Random(30));
  KeyUp(16);
  Wait(20+Random(30));
End;
