first off there is an easier way to do this.

'This makes a integer and a world variable to use for the detail and world
Dim intDetail As Integer = "0" 
Dim strWorld As String = ""

'If the "High Detail" Checkbox is checked then will switch to highdetail
If Me.chkHighDetail.Checked = True Then
intDetail = "1"
Else If Me.chkHighDetail.Checked = False Then
intDetail = "0"
End If

'This Chooses the world
Select Case Text2.Caption 'in .Net it will be Text2.text instead of Text.caption
Case "1"
strWorld  = "ul2"
Case "2"
strWorld  = "ul8"
Case "3"
strWorld = "po3"
Case "4"
strWorld = "po4"
Case "5"
strWorld = "po5"
Case "6"
strWorld = "po6"
Case "7"
strWorld = "above2"
Case "8"
strWorld = "above3"
Case "9"
strWorld = "above4"
Case "10"
strWorld = "jolt7"
Case "11"
strWorld = "jolt8"
Case "12"
strWorld = "jolt9"
Case "13"
strWorld = "nl3"
Case "14"
strWorld = "nl4"
Case "15"
strWorld = "uk2"

'Continue here and find the world URL of all worlds (http://img427.imageshack.us/img427/1707/asdasdasdas0bs.jpg)
End Select


'This Actually makes it go to the page
'Also changes out the World variable and Detail variable if needed
WebBrowser1.Navigate "http://" & intWorld & ".runescape.com/lang/en/aff/runescape/game.ws?lowmem=1&plugin=" & intDetail

Btw you need a new Checkbox named chkDetail.
this checkbox will control high and low detail
