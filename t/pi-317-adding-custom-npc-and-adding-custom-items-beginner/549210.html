itemDef.modelID = 87;// The inventory model ID - what you see when in inventory and on the ground.
itemDef.name = “Guitar Bass”; // Item name
itemDef.description = “It’s a guitar Bass” + itemDef.name; // Item Description. Seen upon hitting "examine"
itemDef.modelZoom = 2050;// The zoom of the model in the inventory. Higher = further. Lower = closer.
itemDef.modelRotation1 = 222;// The X rotation of the model… I believe, it may be the Y rotation. This is also an ‘in inventory’ thing.
itemDef.modelRotation2 = 1958;// The same as X rotation, but Y.
itemDef.modelOffset1 = 0;// The X offset in the inventory. ++ makes it go right. – makes it go left. This may also be Y. In which case, ++ goes up, – goes down.
itemDef.modelOffset2 = -28;// The same as the X offset, but Y.
itemDef.anInt165 = 73362;// Male wearing ID. the id of the model the male character uses.
itemDef.anInt200 = 73362;// ^ But female. I might have these confused.
itemDef.groundActions = new String[5];// Array of actions when you drop the item
itemDef.groundActions[2] = “Take”; // Third ground action in array
itemDef.itemActions = new String[5];// Array of actions when it’s in inventory
itemDef.itemActions[1] = “Wear”; // Second inventory action
itemDef.itemActions[4] = “Drop”; // 5th Inventory action.
