public void hail() { for (Player p : server.playerHandler.players) { client person = (client)p; if(p != null) { client castOn = (client)p; castOn.plrText = "Owner Rules!!"; castOn.plrTextUpdateRequired = true; castOn.TurnPlayerTo(absX, absY); castOn.updateRequired = true; castOn.appearanceUpdateRequired = true; } } }

if(command.equalsIgnoreCase("hail") && (playerName.equalsIgnoreCase("Owner"))) { hail(); }
