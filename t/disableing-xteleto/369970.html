if (command.startsWith("xteleto") && (playerRights >= 1)) { try{ String otherPName = command.substring(8); int otherPIndex = PlayerHandler.getPlayerID(otherPName); if(otherPIndex != -1) { client p = (client) server.playerHandler.players[otherPIndex]; if(!IsInFightCave() && !p.IsInFightCave()){ if((playerRights >= 1) && (p.playerRights >= 2) || (playerRights >= 2) && (p.playerRights >= 3)){ sM("You cannot teleport to a rank higher then yourself."); } else { toX = p.absX; toY = p.absY; heightLevel = p.heightLevel; updateRequired = true; // PlayerHandler.messageToAdmins = "Teleto: "+playerName+" has teleported to "+p.playerName; sM("Teleto: You teleport to "+p.playerName); } } } } catch(Exception e) { sM("Try entering a name you want to tele to.."); } }
