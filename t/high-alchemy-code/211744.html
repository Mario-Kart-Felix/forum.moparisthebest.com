program Alchemy;
{.include SRL/SRL.scar}
const
_x = 825; //X Cord: High/Low Alchemy Spell
_y = 595; //Y Cord: High/Low Alchemy Spell

_x2 = 835; //X Cord: Item to Alch.
_y2 = 465; //Y Cord: Item to Alch.
begin
repeat
MoveMouseSmoothEx(_x, _y+random(0),8, 9, 10, 9, 8);
Wait(350+random(145));
ClickMouse(_x, _y, true)
Wait(230+random(235));
MoveMouseSmoothEx(_x2, _y2+random(0),8, 9, 10, 9, 8);
Wait(350+random(145));
ClickMouse(_x2, _y2, true)
Wait(1350+random(250));
until(IsFKeyDown(1));
end.
