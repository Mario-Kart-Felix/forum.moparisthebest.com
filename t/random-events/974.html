public class StrangeBox extends RandomHandler implements InterfaceListener {
private int[] sideShape = new int[3];
private int[] sideNumber = new int[3];
private String question = null;
private String[] answer = new String[3];

public StrangeBox() {
	EventManager.addInterfaceListener(this);
}

public void newInterface(RSInterface rsi) {
	if(rsi.interfaceID == 15230 || rsi.interfaceID == 15231 || rsi.interfaceID == 16154)
		setShape(rsi.interfaceID, rsi.textureID);

	if(rsi.interfaceID == 16155 || rsi.interfaceID == 16156 || rsi.interfaceID == 16157)
		setNumber(rsi.interfaceID, rsi.textureID);
	if(rsi.interfaceID == 6561)
		question = rsi.message.toLowerCase();
	if(rsi.interfaceID == 6565 || rsi.interfaceID == 6566 || rsi.interfaceID == 6567)
	 setAnswer(rsi.interfaceID, rsi.message);
}

private void setAnswer(int i, String s) {
	switch(i) {
		case 6565:
			answer[0] = s.toLowerCase();
			break;
		case 6566:
			answer[1] = s.toLowerCase();
			break;
		case 6567:
			answer[2] = s.toLowerCase();
			break;
	}

}

public long run(long l)

{
if (getInventoryIndex(3062) == -1)
return -1;

  if (getInterface() != 6554)
  {
     if (getTab() != 3)
     {
        log("Selecting inventory");
        selectTab(3);
        return random(2000, 3000);
     }
     else
     {
        int box = Constants.ITEM_STRANGE_BOX;
		log("Opening box");
        atSpecialInventoryItem(box, getInventoryIndex(box), 1);
        return random(2000, 3000);
     }
  }
  else
  {
     if (getInterface() == 6554)
     {
        int answer = getStrangeBox();
        log("Answer is " + answer);
        if (answer == 0)
        {
           atInterface(6562, 1);
           return 2000;
        }
        else if (answer == 1)
        {
           atInterface(6563, 1);
           return 2000;
        }
        else if (answer == 2)
        {
           atInterface(6564, 1);
           return 2000;
        }
        else
        {
           log("error in strange box answer!");
           return -1;
        }
     }
     else
        return -1;
  }

}

final int CIRCLE = 0;
final int PENTAGON = 1;
final int SQUARE = 2;
final int STAR = 3;
final int TRIANGLE = 4;

private int getShape(int model)
{
switch (model)
{
case 7005:
case 7020:
case 7035:
return CIRCLE;
case 7006:
case 7021:
case 7036:
return PENTAGON;
case 7007:
case 7022:
case 7037:
return SQUARE;
case 7008:
case 7023:
case 7038:
return STAR;
case 7009:
case 7024:
case 7039:
return TRIANGLE;
default:
log(“MODEL IS NOT RECOGNISED?!”);
// System.exit(0);
return 0;
}
}

private int getNumber(int model)

{
if (model >= 7010 && model <= 7019)
return model - 7010;
if (model >= 7025 && model <= 7034)
return model - 7025;
if (model >= 7040 && model <= 7049)
return model - 7040;
log(“MODEL NOT RECOGNISED?!”);
// System.exit(0);
return 0;
}

private void setShape(int i, int j) {
	switch(i) {
		case 15230:
			sideShape[0] = getShape(j);
			break;
		case 15231:
			sideShape[1] = getShape(j);
			break;
		case 16154:
			sideShape[2] = getShape(j);
	}
}

private void setNumber(int i, int j) {
	switch(i) {
		case 16155:
			sideNumber[0] = getNumber(j);
			break;
		case 16156:
			sideNumber[1] = getNumber(j);
			break;
		case 16157:
			sideNumber[2] = getNumber(j);
			break;
	}
}

private int getStrangeBox()

{

  if (question.indexOf("what number is") > -1)
  {
     int[] answerInt = new int[3];
     for (int i = 0; i < 3; i++)
        answerInt[i] = Integer.parseInt(answer[i].substring(0, 1));
     int answerSide = -1;
     if (question.indexOf("circle") != -1)
        answerSide = findPos(sideShape, CIRCLE);
     if (question.indexOf("pentagon") != -1)
        answerSide = findPos(sideShape, PENTAGON);
     if (question.indexOf("square") != -1)
        answerSide = findPos(sideShape, SQUARE);
     if (question.indexOf("star") != -1)
        answerSide = findPos(sideShape, STAR);
     if (question.indexOf("triangle") != -1)
        answerSide = findPos(sideShape, TRIANGLE);
     return findPos(answerInt, sideNumber[answerSide]);
  }
  else
  {

	int lookingForNumber = Integer.parseInt(question.substring(question.lastIndexOf(" ") + 1, question.lastIndexOf(" ") + 2));
    return findPos(sideNumber, lookingForNumber);
  }

}

private int findPos(int[] array, int find)
{
for (int i = 0; i < array.length; i++)
if (array[i] == find)
return i;
return -1;
}
}
