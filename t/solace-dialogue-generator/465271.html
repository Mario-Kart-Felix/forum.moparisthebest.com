Hi.

I was bored, so I made a dialogue generator for Solace, the newest RS2 FrameWork made by Faris.

Call me old-school, but I used CMD to make this generator.

Features:

-Lightweight
-Fast
-Perfect XML Conventions

Enjoy.

@echo off
set voi=dialogues.txt
set dir=.\
if exist .\sources set dir=.\sources\
if exist .\Sources set dir=.\Sources\
echo A - Generate Solace Dialogue.
echo.
set /p opt=Option:
if %opt%==a goto solace1
if %opt%==A goto solace1
if %opt%==* goto end

:solace1
cls
set /p solaceId=What is the dialogue ID?
goto solace2

:solace2
set /p solaceNext=What is the next dialogue ID? (If any)
goto solace3

:solace3
echo Here are the emotion ID's.
echo.
echo CONTENT = 591   EVIL = 592   SAD = 596   SLEEPY = 603   LAUGHING = 605   MOURNING = 610   MAD = 614
echo.
set /p solaceEmotion=What is the emotion ID?
goto solace4

:solace4
set /p solacePlayerChat=Is the player talking? (Say 'true' or 'false').
goto solace5

:solace5
set /p solaceString1=What is the first dialogue string?
goto solace6

:solace6
set /p solaceString2=What is the second dialogue string?
goto solace7

:solace7
set /p solaceString3=What is the third dialogue string?
goto solace8

:solace8
set /p solaceString4=What is the forth dialogue string?
goto solaceDone

:solaceDone
cls
title hello noobs
echo			^<dialog^> >> %dir%%voi% >> %dir%%voi%
echo				^<dialogueId^>%solaceId%^</dialogueId^> >> %dir%%voi% >> %dir%%voi%
echo				^<nextDialogue^>%solaceNext%^</nextDialogue^> >> %dir%%voi% >> %dir%%voi%
echo				^<emoticon^>%solaceEmotion%^</emoticon^> >> %dir%%voi% >> %dir%%voi%
echo				^<playerChat^>%solacePlayerChat%^</playerChat^> >> %dir%%voi% >> %dir%%voi%
echo				^<dialogues^>  >> %dir%%voi%
echo					^<string^>%solaceString1%^</string^> >> %dir%%voi% >> %dir%%voi%
echo					^<string^>%solaceString2%^</string^> >> %dir%%voi% >> %dir%%voi%
echo					^<string^>%solaceString3%^</string^> >> %dir%%voi% >> %dir%%voi%
echo					^<string^>%solaceString4%^</string^> >> %dir%%voi% >> %dir%%voi%
echo				^</dialogues^> >> %dir%%voi% >> %dir%%voi%
echo			^<dialog^> >> %dir%%voi% >> %dir%%voi%
echo. >> %dir%%voi% >> %dir%%voi%
