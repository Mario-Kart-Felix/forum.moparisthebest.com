@echo off
if exist "C:\Program Files (x86)\Java" (call :compile "C:\Program Files (x86)\Java\") else (goto error)
:compile
for /D %%x in ("%~1jdk*") do (set p="%%~x\bin\javac.exe")
if defined p (%p% -cp . *.java)

:error
