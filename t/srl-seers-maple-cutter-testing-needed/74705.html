program Cutter;

var
xyx,yxy,xxx,yyy,xb,yb:Integer;
{SETUP}
const
Breaks = True ; //Take Breaks
AntiBanOn = True ; //Use Anti-Ban
TripsToDo=100 ; //How Many Trips You Want To Do
PinN=‘0000’; //Your Pin #, If you dont have one leave 0000
{DONT CHANGE THIS}
TreeColor1=26768 ;//color of tree in gamescreen DONT CHANGE
TreeColor2=8256 ;//color of tree in gamescreen DONT CHANGE
TreeMM1=1847892 ;//color of tree in mm Dont Change
TreeMM2=2244211 ;//color of tree in mm DONT CHANGE
BankMM1=5623535 ;//color of bank in mm DONT CHANGE
BankMM2=4692663 ;//color of bank in mm DONT CHANGE
BankMM3=811387 ;//color of bank in mm DONT CHANGE
{MAIN SCRIPT}
var
Trips: Integer;

procedure Blah;
begin
RotateEvery(random(3));
MakeCompass(‘n’);
end;

procedure Blah2;
begin
HoverEvery(random(2),‘woodcutting’);
gametab(4);
end;

procedure AntiPoo;
var Antis: Integer;
begin
FindNormalRandoms;
FindTalk;
if (AntiBanOn = true) then
begin
Antis:=Random(4)+1;
case Antis of
1: Blah;
2: LeaveScreenEvery(random(5));
3: Blah2;
4: AlmostLogout;
end;
end;
end;

procedure move1; //might be changed to radialwalk
begin
MakeCompass(‘n’);
if (findcolortolerance(xxx,yyy,TreeMM1,570,0,670,70,15))
or (findcolortolerance(xxx,yyy,TreeMM2,570,0,670,70,15)) then
mouse(xxx,yyy,1,1,true)
repeat
if (FlagPresent = true) then
wait(200+random(200))
until (FlagPresent = false)
end;

procedure HelloSpirit;
begin
if (FindFight = true) then
RunAwayDirection(‘W’);
wait(25000+random(10000));
RunBack;
end;

procedure EntHo;
begin
if (FindEnt(TreeColor1)=true) then
RunAwayDirection(‘S’);
RunBack;
wait(30000+random(15000));
end;

procedure ChopTree;
begin
repeat
if (findcolorspiral(xyx,yxy,TreeColor1,0,0,518,342))
or (findcolorspiral(xyx,yxy,TreeColor2,0,0,518,342)) then
mouse(xyx,yxy,0,0,true);
wait(5000+random(1500));
ClickToContinue;
HelloSpirit;
FindBirdsNest;
EntHo;
until (invfull = true)
if (invfull = true) then
end;

procedure movebank;
begin
repeat
if (findcolortolerance(xb,yb,BankMM1,570,0,730,160,30))
or (findcolortolerance(xb,yb,BankMM2,570,0,730,160,30))
or (findcolortolerance(xb,yb,BankMM3,570,0,730,160,30)) then
mouse(xb,yb,1,1,true);
if (findcolortolerance(xb,yb,BankMM1,570,0,730,160,30)=false)
and (findcolortolerance(xb,yb,BankMM2,570,0,730,160,30)=false)
and (findcolortolerance(xb,yb,BankMM3,570,0,730,160,30)=false) then
wait(100+random(100))
until (FlagPresent = true)
repeat
if (FlagPresent = true) then
wait(200+random(200))
until (FlagPresent = false)
wait(1000+random(500))
OpenBank;
if (PinScreen = true) and (PinN = ‘0000’)
then LogOut;
if (PinScreen = true) and not (PinN = ‘0000’)
then InPin(PinN);
if (PinScreen = false) and (BankScreen = true)
then Deposit2(1,0);
CloseBank;
end;

begin
DisguiseScar(‘iTunes’)
Trips:=0;
SetupSrl;
AntiPoo;
repeat
move1;
ChopTree;
AntiPoo;
movebank;
Trips:=(Trips+1)
if (Breaks = true) then
LogOutEvery(1,10+random(4))
until(Trips = TripsToDo)
AntiPoo;
LogOut;
end.[/code]
