[CODE=java]/*

  •    PestControl.java

  •  Created By Jason (Clawz Fury)
  • To handle the Pest Control minigame

public class PestControl {

	private client c;

	public PestControl(client client) 
	{
		this.c = client;
	}
	 
	public boolean portal1;
	public boolean portal2;
	public boolean portal3;
	public boolean portal4;
	public boolean inPc;

	public static int waitTimer = 300;
	
	public int[] objectIds = {14315, 14314};
	public int[] npcIds = {3781, 3789};
	public int[] portalIds = {3777, 3778, 3779, 3780};

	public int inGameX = 2658;
	public int inGameY = 2611;

	public int endGameX = 2657;
	public int endGameY = 2639;


       /**
	** Begins the Pest Control minigame...
 	**
 	**/

	public void initializePc()
	{
		if(inPcBoat() && !inPc)
		{
			inPc = true;
			resetPest();
			c.toX = inGameX + misc.random(1);
			c.toY = inGameY + misc.random(3);
			server.npcHandler.spawnANPC(3777, 2628, 2591,0);
			server.npcHandler.spawnANPC(3778, 2680, 2588,0);
			server.npcHandler.spawnANPC(3779, 2669, 2570,0);
			server.npcHandler.spawnANPC(3780, 2645, 2569,0);
			portal1 = false;
			portal2 = false;
			portal3 = false;
			portal4 = false;
			c.ResetAttackNPC();
		}
	}

       /**
	** Ends the minigame...
 	** 
 	**/

	public void endPc()
	{
		if(portal1 && portal2 && portal3 && portal4 && inPc)
		{
			inPc = false;
			resetPest();
			c.pcPoints++;
			c.toX = endGameX;
			c.toY = endGameY;
			waitTimer = 300;
			c.playerLevel[0] = c.getLevelForXP(c.playerXP[0]);
			c.playerLevel[1] = c.getLevelForXP(c.playerXP[1]);
			c.playerLevel[2] = c.getLevelForXP(c.playerXP[2]);
			c.playerLevel[4] = c.getLevelForXP(c.playerXP[4]);
			c.playerLevel[5] = c.getLevelForXP(c.playerXP[5]);
			c.playerLevel[6] = c.getLevelForXP(c.playerXP[6]);
			c.sM("You have defeated pest control! You now have "+c.pcPoints+" points!");
			c.ResetAttackNPC();
			c.refreshSkills();
			pcPanelOn(false);
		}
	}

       /**
	** Checks if all the portals have been defeated
 	** 
 	**/

	public void checkPc()
	{
		if(portal1 && portal2 && portal3 && portal4)
		{
			endPc();
		}
	}

       /**
	** Area checks
 	** 
 	**/
			
	public boolean inPcGame() 
	{
			if (c.absX >= 2624 && c.absX <= 2690 && c.absY >= 2550 && c.absY <=2619) 
		{
  				return true;

			} 
		else 
		{
  				return false;
			}
	}

	public boolean inPcBoat() 
	{
			if (c.absX >= 2660 && c.absX <= 2663 && c.absY >= 2638 && c.absY <=2643) 
		{
  				return true;
			}
		else 
		{
  				return false;
			}
	}

	public boolean isInPcGame() 
	{
	if((c.absX >= 2624 && c.absX <= 2686 && c.absY >= 2560 && c.absY <=2625))

			return true;
			else
			return false;
	}

       /**
	** Resets the NPC's
 	** 
 	**/

	public void resetPest() 
	{
		for (int i = 0; i < server.npcHandler.maxNPCs; i++) 
		{
			for (int portals : portalIds) 
			{
				if (server.npcHandler.npcs[i] != null) 
				{
					if(server.npcHandler.npcs[i].npcType == portals)
					{
						server.npcHandler.npcs[i].absX = 0;
						server.npcHandler.npcs[i].absY = 0;
					}
				}
			}
		}
	}

       /**
	** Handles the NPC's (Void Knight, Squire who lets you leave)
 	** 
 	**/

	public void handleNpcs(int npcId)
	{
		switch(npcId)
		{
			case 3781:
				inPc = false;
				c.toX = endGameX;
				c.toY = endGameY;
				pcPanelOn(false);
				break;
			case 3789:
				c.sendQuest("      "+c.pcPoints+"", 10379);
				c.showInterface(10294);
				break;
		}
	}

       /**
	** Handles timers
 	** 
 	**/

	public void process() 
	{
		if(waitTimer == 0)
		{
			waitTimer = 300;
			initializePc();
			updatePanel();
		}
		if(waitTimer > 0)
		{
			waitTimer -= 1;
			updatePanel();
		}
	}

       /**
	** Updates the Panel
 	** 
 	**/

	public void updatePanel() 
	{
		c.sendQuest(""+waitTimer+"", 15899);
	}

       /**
	** Ethier turns the Panel On or Off
 	** 
 	**/

	public void pcPanelOn(boolean on) 
	{
		if(on == false)
		{
           	 		c.setInterfaceWalkable(-1);
			c.sendQuest("", 15894);
			c.sendQuest("", 15895);
			c.sendQuest("", 15897);
			c.sendQuest("", 15898);
			c.sendQuest("", 15899);
			c.sendQuest("", 15900);
			c.sendQuest("", 15901);
			c.sendQuest("", 15896);
			c.sendQuest("", 15902);
			c.sendQuest("", 15903);
			c.sendQuest("", 15904);
			c.sendQuest("", 15905);
			c.sendQuest("", 15906);
		}
		if(on == true)
		{
			c.setInterfaceWalkable(15892);
			c.sendQuest("Pest Control", 15894);
			c.sendQuest("Game starts in:", 15895);
			c.sendQuest("", 15897);
			c.sendQuest("", 15898);
			c.sendQuest(""+waitTimer+"", 15899);
			c.sendQuest("Good Luck!", 15900);
			c.sendQuest("", 15901);
			c.sendQuest("", 15896);
			c.sendQuest("", 15902);
			c.sendQuest("", 15903);
			c.sendQuest("", 15904);
			c.sendQuest("", 15905);
			c.sendQuest("", 15906);
		}
	}

       /**
	** Handles Objects
 	** 
 	**/
		
	public void handleObjects(int objectid, int x, int y)
	{
		switch(objectid)
		{
			case 14315:
				if(System.currentTimeMillis() - c.lastAction > c.actionInterval)
				{
					c.sM("You enter the boat, "+waitTimer+" till next game startserver.");
					c.toX = 2660;
					c.toY = 2639;
					c.actionInterval = 2500;
					c.lastAction = System.currentTimeMillis();
					pcPanelOn(true);
				}
				break;
			case 14314:
				c.sM("You leave the boat.");
				c.toX = 2657;
				c.toY = 2639;
				pcPanelOn(false);
				break;
		}
	}

       /**
	** Action checks
 	** 
 	**/

	public boolean stopAction()
	{
		if(inPcGame() || inPcBoat() || isInPcGame())
		{
			c.sM("You cannot do this action!");

			return true;
		} else {
			return false;
		}
	}
