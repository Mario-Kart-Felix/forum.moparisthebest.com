Well, I’ve created my first SRL script. It says guard killer but I suppose it could be used on anything. Just edit the part that says ‘Al-Kharid Warrior’ to whatever you’re going to attack
I haven’t had any known errors with it yet, and it should work… has updated randoms etc.

///////////////////////////////     
//Rabcarl's auto Auto      //
//Al Kharid Warrior Killer  //
/////////////////////////////     
             ///
/////////////////////////
//Credits:              //
//Doomsday-         //
//helped a lot        //
//with some errors //
///////////////////////

{setup: be by the al-kharid warriors
and set the colors.
VERSION: 1.0
FUTURE PLANS: I plan to include an eat function and
possibly a walkback function(in the distant future)
}
program AlKharidWarriorOwner;
{.include SRL/SRL.scar}
{.include SRL/SRL/Skill/Fighting.scar}
var
warriorColor1,warriorColor2,warriorColor3:integer;
//setup what skill you want lamps to be used on: line 38
procedure setup;
begin
warriorColor1:=0;
warriorColor2:=0;
warriorColor3:=0;
end;

procedure FindRandoms;
begin
   setRun(true)
  FindNormalRandoms;
  ClickToContinue;
  CloseWindow;
  FindLamp('hitpoints');
  FindDead;
  FindMod;
  FindMime;
  FindMaze;
  FindQuiz;
  FindFrog;
  FindDemon;

end;

procedure loginIfneeded;
begin
if LoggedIn=false then LoginPlayer;
end;
procedure ownGuards;
begin
AttackMonster(warriorColor1,warriorColor2,warriorColor3,'Al-Kharid Warrior',2);
end;

begin
//main line
repeat
setupsrl;
setup;
FindRandoms;
LoginIfNeeded;
ownGuards;
until(false);
end.

edit: just so you know I haven’t finished a login procedure yet so… if it logs out it won’t login your current player
