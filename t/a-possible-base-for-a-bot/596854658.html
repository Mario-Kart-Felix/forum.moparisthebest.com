package bot.update.transformer;

import com.sun.org.apache.bcel.internal.classfile.Field;
import com.sun.org.apache.bcel.internal.classfile.JavaClass;
import com.sun.org.apache.bcel.internal.classfile.Method;
import com.sun.org.apache.bcel.internal.generic.*;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.PrintStream;
import java.util.Iterator;

/**
 * Created by IntelliJ IDEA.
 * User: Silab
 * Date: Feb 25, 2007
 * Time: 2:21:47 AM
 * Project: bot.update.transformer
 */
public class nonstatic extends Transformer {
    PrintStream ps;

    public nonstatic() {
        new File("client info").mkdir();
    }

    public boolean find(JavaClass clazz) {
        int i = 0;
        int j = 0;
        try {
            ps = new PrintStream(new File("client info//" + clazz.getClassName() + ".txt"));
        } catch (FileNotFoundException e) {
            e.printStackTrace();
        }
        ps.println("Class Name: " + clazz.getClassName());
        ps.println("Class isAbstract: " + clazz.isAbstract()+"\n\n");
        for (Field f : clazz.getFields()) {
            if (!f.isStatic()) {
                ps.println("Field Name: " + f.getName());
                ps.println("Field Type: " + f.getType()+"\n\n");
                i++;
            }

        }
        for (Method m : clazz.getMethods()) {
            if (!m.isStatic()) {
                ps.println("Method Name: " + m.getName());
                ps.println("Method Number of Args: " + m.getArgumentTypes().length);
                if (m.getArgumentTypes().length > 0) {
                    Type[] types = m.getArgumentTypes();
                    for (int x = 0; x < m.getArgumentTypes().length; x++)
                        ps.println("\t Arg[" + x + "]: " + types[x]);
                }
                ps.println("\n");
                j++;
            }

        ps.println("\n\n");
        ps.println("Total non-static fields: " + i);
        ps.println("Total non-static methods: " + j);
        ps.close();
        }
        return false;
    }

    public boolean transform(JavaClass clazz) {
        return false;
    }

    public void finish() {
        System.out.println("Finished Recording non static methods and fields");
    }
}

	public void makeGui(Object o) {
		if (o == null) {
			return;
		}
		Shell shell = new Shell(GuiThread.theDisplay);
		shell.setText("Explorer for " + o);
		shell.setLayout(new FillLayout());
		final Composite composite = new Composite(shell, SWT.NONE);
		composite.setLayout(new FillLayout());
		final Table table = new Table(composite, SWT.FULL_SELECTION
				| SWT.BORDER);
		table.setLinesVisible(true);
		table.setHeaderVisible(true);
		TableColumn newColumnTableColumn1 = new TableColumn(table, SWT.NONE);
		newColumnTableColumn1.setWidth(100);
		newColumnTableColumn1.setText("name");
		TableColumn newColumnTableColumn2 = new TableColumn(table, SWT.NONE);
		newColumnTableColumn2.setWidth(200);
		newColumnTableColumn2.setText("type");
		TableColumn newColumnTableColumn3 = new TableColumn(table, SWT.NONE);
		newColumnTableColumn3.setWidth(300);
		newColumnTableColumn3.setText("value");
		final Field fields[] = o.getClass().getDeclaredFields();
		final Object FieldObjects[]= new Object[fields.length];
		for(int i=0;i<fields.length;i++){
			fields[i].setAccessible(true);
			FieldObjects[i]=fieldObject(fields[i], o);
			fields[i].setAccessible(false);
		}
		for (Field field : fields) {
			
			field.setAccessible(true);
			
			String name = field.getName();
			String type = field.getType().toString();
			String value = fieldValue(field, o);
			TableItem newItemTableItem = new TableItem(table,
                    SWT.BORDER);
            newItemTableItem.setText(0, name);
            newItemTableItem.setText(1, type);
            newItemTableItem.setText(2, value);
			field.setAccessible(false);
		}
		table.addSelectionListener(new SelectionListener(){

			public void widgetDefaultSelected(SelectionEvent arg0) {}

			public void widgetSelected(SelectionEvent e) {
				try{
					makeGui(FieldObjects[table.getSelectionIndex()]);
				} catch (Exception k) {
					exception(k);
				}
			}
			
		});
		shell.pack();
		GuiThread.center(GuiThread.theShell.getBounds(), shell);
		shell.open();
	}
