public class LastManStanding {

	static Client c;
	public LastManStanding(final Client Client) {
		c = Client;
	}
	
	public static boolean playerAvailable = false;
	public static boolean startGame = false;
	
	public static int inArea() {
		int count = 0;
		for (final Player player : PlayerHandler.players) {
			if (player != null) {
				final Client c = (Client) player;
				if (c.inLMS()) {
				count++;
				}
			}
		}
		return count;
	}

	public static void startLMS() {
		if (inArea() >= 3) {
				countdownTimer();
				startGame = true;
		}
	}

	public static void countdownTimer () {
		if (startGame) {
			long startTimer = 60000;
			startTimer--;
			c.sendMessage("@blu@You have " + startTimer + " seconds to get ready!");
				if(startTimer < 0) {
					c.sendMessage("@red@THE MATCH HAS STARTED!");
					startTimer = 0;
				}
			} else {
				System.out.println("Game isn't on.");
			}
		}
	
	/**
	 * Now to check if there is 1 player standing.
	 */

	public void lastManLeft() {
		if(startGame) {
			if(inArea() == 1) {
				c.sendMessage("You are the last one left..");
			} else {
				System.out.println("Still people in there..");
			}
		} else 
			System.out.println("The game is not on");
	}
}
