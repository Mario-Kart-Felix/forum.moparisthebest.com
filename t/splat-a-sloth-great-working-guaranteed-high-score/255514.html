program SplatASloth_v2;
var x,y:integer;
begin
clickmouse(260,303,true);
wait(2000);
repeat
wait(100);
until(findcolor(x,y,74,150,40,160,50));
repeat
wait(1);
until(not(findcolor(x,y,74,150,40,155,45)));
keydown(32);
wait(30);
