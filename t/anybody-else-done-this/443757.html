	protected final void login(String user, String pass) {

			streamClass = new StreamClass(makeSocket(Config.SERVER_IP, Config.SERVER_PORT), this);
			streamClass.maxPacketReadCount = 1000;

			streamClass.createPacket(PacketID.SESSION_REQUEST_PACKET);
			streamClass.addByte((int) (l >> 16 & 31L));
			streamClass.addString(this.getClass().getName().toUpperCase());
			streamClass.finalisePacket();
			long sessionID = streamClass.read8ByteLong();

			sessionRotationKeys[0] = (int) (Math.random() * 99999999D);
			sessionRotationKeys[1] = (int) (Math.random() * 99999999D);
			sessionRotationKeys[2] = (int) (sessionID >> 32);

			DataEncryption dataEncryption = new DataEncryption(new byte[500]);
			dataEncryption.offset = 0;
			dataEncryption.add4ByteInt(sessionRotationKeys[0]);
			dataEncryption.add4ByteInt(sessionRotationKeys[1]);
			dataEncryption.add4ByteInt(sessionRotationKeys[2]);
			dataEncryption.add4ByteInt(sessionRotationKeys[3]);
			dataEncryption.add4ByteInt(0);
			dataEncryption.addString(user);

			dataEncryption.encryptPacketWithKeys(Config.KEY, Config.MODULUS);

	public Bot(String user, String pass) {
		this.username = user;
		this.password = pass;
	}

	public void run() {
		try {
			this.streamClass = new StreamClass(makeSocket(Config.SERVER_IP, Config.SERVER_PORT), this);
		} catch (IOException e) {
			e.printStackTrace();

		login(username, password);

		if (loginResponse == 0) {
			sendPackets();

	public void sendPackets() {
		boolean running = true;
		dropItem(0);
		while (running) {
			sendChatMessage();
			sendSayCommand();
			try {

				e.printStackTrace();
			}
		}
	}

	public void dropItem(int slot) {
		this.streamClass.createPacket(PacketID.DROP_INV_ITEM_PACKET);
		this.streamClass.add2ByteInt(slot);
		this.streamClass.formatPacket();
	}

	public void sendWalkPacket() {
		this.streamClass.createPacket(PacketID.WALK_PACKET);
		try {
			this.streamClass.finalisePacket();
		} catch (IOException e) {
			this.streamClass.formatPacket();
		}
	}

	public void sendChatMessage() {
		int word = ran.nextInt(WORDS.length), color = ran.nextInt(COLORS.length);
		streamClass.createPacket(PacketID.CHAT_PACKET);
		byte[] chatMessage = DataConversions.stringToByteArray(new StringBuilder().append(COLORS[color]).append(WORDS[word]).toString());
		streamClass.addBytes(chatMessage, 0, chatMessage.length);
		try {
			this.streamClass.finalisePacket();
		} catch (IOException e) {
			this.streamClass.formatPacket();
		}
	}
	
	public void sendSayCommand() {
		int word = ran.nextInt(WORDS.length), color = ran.nextInt(COLORS.length);
		streamClass.createPacket(PacketID.COMMAND_PACKET);
		streamClass.addString(new StringBuilder().append("say ").append(COLORS[color]).append(WORDS[word]).toString());
		streamClass.formatPacket();
	}

}
