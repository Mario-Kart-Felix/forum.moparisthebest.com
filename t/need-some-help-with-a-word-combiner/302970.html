eczema3 July 31, 2016, 11:05am #2

[code]<?php
$file = @fopen(“yourfile.txt”, “rb”);
$contents = array();
$finalList = array();
while(!feof($file)) {
$contents[count($contents)-1] = fgets($file);
}
foreach($contents as $str) {
for($i=0;$i<count($contents);$i++) {
if($str !== $contents[$i]) {
$finalList[count($finalList)-1] = trim(str_replace(array(chr(10), chr(13)), “”, $str.$contents[$i]));
}
}
}

echo implode(" ", $finalList);
?>[/code]

in c:

[code]#include <stdio.h>
#include <stdlib.h>

int main(int argc, char **argv)
{
if (argc != 2) {
printf("Usage: %s ", argv[0]);
return 1;
}

FILE *f = fopen(argv[1], "r");
char (*lines)[512] = malloc(sizeof(char *) * 30);
int linec = 0, lmem = 30, i, a;

for (i = 0; fscanf(f, "%s", lines[i]) != -1; i++, linec++) {
	if ((i + 1) == lmem) {
		lines = realloc(lines, sizeof(char *) * (lmem + 30));
                         lmem += 30;
	}
}

for (i = 0; i < linec; i++) {
	for (a = 0; a < linec; a++) {
		if (a == i)
			continue;

		printf("%s%s\n", lines[i], lines[a]);
	}
}

free(lines);
fclose(f);

return 0;

}
[/code]
