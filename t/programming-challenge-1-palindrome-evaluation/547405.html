#include <stdio.h>

#define FALSE 0
#define TRUE !FALSE
#define PALINDROME_OK "y"
#define PALINDROME_ER "n"
#define N_CHARS 26
#define GET_BIT(n, k) ((n & (1 << k)) >> k)
#define TOGGLE_BIT(n, k) (n ^= (1 << k))


inline static int is_palindrome(int memory)
{
    register int bit;
    register int odd_flag;
    for (bit = 0, odd_flag = 0; bit < N_CHARS; ++bit) {
        if (!GET_BIT(memory, bit)) {
            if (odd_flag) {
               return FALSE;
            } else {
                odd_flag = TRUE;
            }
        }
    }
    return TRUE;
}

int main(void)
{
    register int memory = 0xffffffff;
    register int ch;
    while ((ch = getc(stdin)) != EOF) {
        if (ch >= 'A' && ch <= 'Z') {
            ch = ch + ('A' - 'a');
        }
        if (ch < 'a' || ch > 'z') {
            continue;
        }
        TOGGLE_BIT(memory, ch % N_CHARS);
    }
    printf("%s\n", is_palindrome(memory) ? PALINDROME_OK : PALINDROME_ER);
}

public class PalindromeEvalulation {

    public boolean findPalindrome(String input) {
        int inputLength = input.length();
        char characters[] = input.toLowerCase().toCharArray();
        boolean oddFlag = false;
        for (int i = 0; i < inputLength; i++) {
            char a = characters[i];
            int count = 0;

            for (int x = 0; x < inputLength; x++) {
                char b = characters[x];
                if (a == b) {
                    count++;
                }
            }
            if (count % 2 == 0) {
                continue;
            }
            if (oddFlag) {
                return false;
            } else {
                oddFlag = true;
            }
        }
        return true;
    }

    public static void main(String[] args) {
        PalindromeEvalulation pe = new PalindromeEvalulation();
        Scanner s = new Scanner(System.in);
        while (true) {
            String input = s.next();
            System.out.println(pe.findPalindrome(input)? 'y' : 'n');
        }
    }

}

function is_palindrome(str) {
	if(!str) return 'y';
	var occ = [];
	str = str.toLowerCase().replace(/\s/g, '');
	occ = count(str, 0, occ);
	return Object.keys(occ).filter(function(o) {
		return occ[o] % 2 !== 0;
	}).length < 2 ? 'y' : 'n';
}

function count(str, index, map) {
	if(str.length === 1)
	    return map;
	if(str.length === index)
	    return map;
	var chk = str.charAt(index++); 
	if(/[a-zA-Z0-9]/.test(chk)) {
	    if(map[chk])
	        map[chk]++;
	    else 
	        map[chk] = 1;
	}
	return count(str, index, map);
}
