This will download a file from the internets! lel

saves the file as the last bit of text…
for example
java FileDownloader http://po3.runescape.com/runescape.jar would save as runescape.jar

adding http:// isn’t necessary, if you don’t add it, it will add it for you.

[code=java]import java.net.URLConnection;
import java.net.URL;
import java.io.IOException;
import java.io.FileOutputStream;
import java.io.File;
import java.io.InputStream;

public class FileDownloader {
public static void main(String[] args) {
if(args.length == 1) {
if(args[0].startsWith(“http://”))
new FileDownloader(args[0]);
else {
new FileDownloader(new StringBuilder().append(“http://”).append(args[0]).toString());
}
} else {
System.out.println("[USAGE]java FileDownloader fileToDownload");
}
}

public FileDownloader(String downloadFrom) {
    try {
        String toSave = downloadFrom.substring(downloadFrom.lastIndexOf("/") + 1);
        File file = new File(toSave);
        if(file.exists()) {
            System.out.println(new StringBuilder().append("Deleting ").append(toSave).append(".").toString());
            file.delete();
            System.out.println(new StringBuilder().append("Deleted ").append(file.toString()).append(".").toString());
        }
        URLConnection url = new URL(downloadFrom).openConnection();
        FileOutputStream output = new FileOutputStream(new File(toSave));
        InputStream input = url.getInputStream();
        byte[] bytes = new byte[url.getContentLength()];
        int x;
        int i = 0;
        
        while((x = input.read(bytes)) > -1) {
            output.write(bytes, 0, x);
            i += x;
        }
        System.out.println("Downloaded " + file.length() + " bytes to " + toSave);
        input.close();
        output.close();
    } catch(IOException e) {
        e.printStackTrace();
    }
}

}[/code]
