Obviously using Gasm to look up the bytecode and try to find the patterns does not work.
It's perfectly fine, just don't include the constant pool indexes. You have two choices:

[ul][li]Match surrounding instructions also[/li]
[li]Cast to INVOKESTATIC and check that it’s returning an Inetaddress.[/li][/ul]

The latter can be done by:

Instruction i = ....; if(i instanceof INVOKSTATIC) { INVOKSTATIC is = (INVOKESTATIC)i; }
You now have access to the methods provided by INVOKESTATIC, which you can view here: http://jakarta.apache.org/bcel/apidocs/org/apache/bcel/generic/INVOKESTATIC.html

For example, you could do, is.getReturnType().toString() …

public void update() {
	log("Downloading runescape.jar from " + server + "...");
	File jar = new File("in/runescape.jar");
	try {
		BufferedInputStream jarin = new BufferedInputStream(new URL("http://" + server + "/runescape.jar").openStream());
		FileOutputStream jarout = new FileOutputStream(jar);
		for(int read; (read = jarin.read()) != -1;) {
			jarout.write(read);
		}
		jarin.close();
		jarout.close();
		log("Patching and dumping runescape.jar...");
		JarFile jarfile = new JarFile(jar);
		if(new File(jarfile.getName()).exists()) {
			Enumeration entries = jarfile.entries();
			while(entries.hasMoreElements()) {
				JarEntry entry = (JarEntry)entries.nextElement();
				if(entry.getName().endsWith(".class")) {
					BufferedInputStream classin = new BufferedInputStream(jarfile.getInputStream(entry));
					JavaClass javaclass = new ClassParser(classin, entry.getName()).parse();
					patch(javaclass);
				}
			}
			log("Complete!");
		} else {
			log("error! could not find runescape.jar.");
		}
	} catch(IOException e) {
		e.printStackTrace();
	}
}

public void patch(JavaClass javaclass) {
	for(Method m : javaclass.getMethods()) {
		if(m.getName().equals("main")) {
			ConstantPoolGen poolgen = new ConstantPoolGen(javaclass.getConstantPool());
			MethodGen methodgen = new MethodGen(m, javaclass.getClassName(), poolgen);
			InstructionList list = methodgen.getInstructionList();
			InstructionFinder finder = new InstructionFinder(list);
			for(Iterator<InstructionHandle[]> i = finder.search("invokestatic"); i.hasNext();) {
				INVOKESTATIC invokestatic = (INVOKESTATIC)i.next()[0].getInstruction();
				if(invokestatic.getReturnType(poolgen).toString().equals("java.net.InetAddress")) {
					log("Connection method is " + javaclass.getClassName() + "." + invokestatic.getMethodName(poolgen) + ".");
					InstructionFactory factory = new InstructionFactory(poolgen);
					//list.append(factory.createPrintln("FHK, LOL"));
					//list.update();
					javaclass.setConstantPool(methodgen.getConstantPool().getConstantPool());
				}
			}
		}
	}
	try {
		javaclass.dump(new File("bot/" + javaclass.getFileName()));
	} catch(Exception e) {
		e.printStackTrace();
	}
}
