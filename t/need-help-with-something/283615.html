private void directFlushoutStream() throws java.io.IOException
{

private boolean fillinStream(int ct) throws IOException {
	inStream.currentOffset = 0;
	if(in.available() >= ct) {
		inStream.currentOffset = 0;
		in.read(inStream.buffer, 0, ct);
