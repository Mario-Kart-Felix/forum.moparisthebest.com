public void applyDead() {
		c.respawnTimer = 15;
		c.isDead = false;
		c.DC++;

		if (c.duelStatus != 6) {
			c.killerId = findKiller();
			final Client o = (Client) PlayerHandler.players[c.killerId];
			if (o != null) {

				if (c.killerId != c.playerId) {
					if (c.playerRights == 0 && c.isDonator == 0
							&& c.isExtreme == 0) {
						c.pkPoints = c.pkPoints + 4;
					} else if (c.playerRights == 0 && c.isDonator == 1
							&& c.isExtreme == 0) {
						c.pkPoints = c.pkPoints + 6;
					} else if (c.playerRights == 0 && c.isDonator == 0
							&& c.isExtreme == 1) {
						c.pkPoints = c.pkPoints + 12;
					}
				}
				c.playerKilled = c.playerId;

				if (o.duelStatus == 5) {
					o.duelStatus++;
					o.KC++;
				}
			}
		}
		c.faceUpdate(0);
		c.npcIndex = 0;
		c.playerIndex = 0;
		c.stopMovement();
		if (c.duelStatus <= 4) {
			c.sendMessage("Oh dear you are dead!");
		} else if (c.duelStatus != 6) {
			c.sendMessage("You have lost the duel!");
		}
		resetDamageDone();
		c.specAmount = 10;
		c.getItems().addSpecialBar(c.playerEquipment[c.playerWeapon]);
		c.lastVeng = 0;
		c.vengOn = false;

		resetFollowers();
		c.attackTimer = 10;
		removeAllWindows();
		c.tradeResetNeeded = true;
	}
