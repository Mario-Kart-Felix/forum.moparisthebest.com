// Autoclicker by The Rave.
// Get your X and Y co-ords using the color picker
// Setup lines 12-15

Program Autoclicker;
// You dont need to declare consts as variables
Var
  ClickMount : Integer;

// However, you cannot change the value of a const in a script, thus we
// need to make a variable and give it the same value as clickamount,
Const
  x = 100;          // Set x co-ordinate here
  y = 100;          // Set y co-ordinate here
  waittime = 1000;  // Set time between clicks
  clickamount = 10; // Set how many times you want mouse to click

Procedure Click;
Begin
  MoveMouseSmooth(x, y);
  HoldMouse(x, y, True);
  Wait(10);
  ReleaseMouse(x, y, True);
  Wait(WaitTime);
End;

Begin
  ClickMount := ClickAmount; //makes the variable clickmount, the same as the const clickamout
  Repeat  //tells script to repeat everything between repeat and until
    Click;
    ClickMount := ClickMount - 1; //dont ask me why you gotta do this, its annoying
  Until (ClickMount = 0); //condition to stop repeating
End.

[scar]Program AutoClicker;

{.include SRL\SRL.scar}

Var
Times: Integer;

const
Clickx = 100;
Clicky = 100;
ClickType = True; //True for left click, False for right click
ytol = 2; //Mouse(x, y, xtol, ytol, Type);
xtol = 2;
WaitT = 1000; //Time to wait between clicks(in milliseconds)
ClickAmount = 0; //0 for infinate

Procedure ClickTime;
begin
Mouse(Clickx, Clicky, xtol, ytol, ClickType);
end;

begin
SetUpSRL;
Times:= ClickAmount;
Repeat
ClickTime;
Wait(WaitT);
Times:= Times - 1;
Until(Times=0);
end.[/scar]

That way just uses SRL
