program New;
{.include SRL/SRL.scar}
{.include SRL/SRL/skill/WoodCutting.scar}
var q,counter:integer;
const
///////////set up////////////////////////////////
TreeColor = 7577792;// first color of tree //
TreeColor1 = 4155248;// second color of tree //
TreeColor2 = 4290672;// third color of tree //
TreeColor3 = 5406848;// fourth color of tree //
loads = 2;//how manyy loads to chop n drop //
/////////////////////////////////////////////////
{//////////////dont touch below////////////}
procedure chop;
begin
repeat
if(Findcolorspiral(x,y,TreeColor,0,0,750,500))and
IsUpText(‘Chop’)or
(Findcolorspiral(x,y,TreeColor1,0,0,750,500))and
IsUpText(‘Chop’)or
(Findcolorspiral(x,y,TreeColor2,0,0,750,500))and
IsUpText(‘Chop’)or
(Findcolorspiral(x,y,TreeColor3,0,0,750,500))and
IsUpText(‘Chop’)then
wait(200);
mouse(x,y,2,2,true)
Flag;
FindEnt(TreeColor)
FindNormalRandoms;
FindHead;
wait(4500);
flag;
until(invfull)
end;
procedure report;
begin
writeln(’’)
writeln(’[][][][][][][][][][][][][]’)
writeln(’ thanx for ‘)
writeln(’ using ‘)
writeln(’ nuebs power cutter ‘)
writeln(’ droped ‘+inttostr(q)+’ logs ‘)
writeln(’’)
writeln(’[][][][][][][][][][][][][]’)
writeln(’’)
end;
procedure drop;
begin
DropTo(2,28);
q:=q+27;
end;
begin
SetupSRL;
FindAxeHeadColor
repeat
chop
drop
until(counter = loads)
report;
end.[/scar]
