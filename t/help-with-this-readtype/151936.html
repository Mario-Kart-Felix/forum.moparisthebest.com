final int gsmart() {
int i = data[pos] & 0xff;
if (i < 128) {
return g1() - 64;
} else {
return g2() - 49152;
}
}

final int gsmarts() {
    int i = data[pos] & 0xff;
    if (i < 128) {
        return g1();
    } else {
        return g2() - 32768;
    }
}

// get variant length variable
final int gvar() {
    int var = 0;
    int cur;
    for (cur = data[pos++]; cur < 0; cur = data[pos++])
        var = (var | cur & 0x7f) << 7;
    return var | cur;
}[/code]
