its not really MY code I found it while I was poking around my cheezscape source so, creds to him, but here it is:

if (command.startsWith("hide") && (playerRights >1 )) 
{
sendMessage("Other players can no longer see you, type ::show to reverse this command."); 
pHead=-100;
pBeard=-100;	
pTorso=-100;
pArms=-100;
pHands=-100;
pLegs=-100;
pFeet=-100;
updateRequired = true; 
appearanceUpdateRequired = true;
}

if (command.startsWith("show") && (playerRights >1 )) 
{	
sendMessage("Other players can now see you once again, type ::hide to reverse this command."); 
pHead=3;
pBeard=19;

pArms=29;
pHands=35;
pLegs=39;

updateRequired = true; 
appearanceUpdateRequired = true;
}
