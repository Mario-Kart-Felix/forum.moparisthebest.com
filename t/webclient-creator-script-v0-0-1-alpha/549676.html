<?php
	/*
	*
	*	Webclient Creator (c) 2015 - Cory Allen
	*
	*/
	require 'h2o.php';
	require_once 'private/config.site.php';
	require_once 'private/config.database.php';
	$h2o = new h2o('public/templates/default/register.html');
	$errors = null;
	$success = null;
	$page = array(); # Page Array - Keep it simple st00pid
	$page['title'] = "Register"; # Page title
	$page['siteName'] = $siteConfig['siteName']; # Site name
	# Registration functionality 
	if( isset($_POST['register']) ) {
	
		if( empty($_POST['username'] || $_POST['password'] || $_POST['password2'] || $_POST['email'] || $_POST['email2']) ) # All fields filled in?
			$errors .= 'You have left a field empty.<br />';
		if( strlen($_POST['username']) < $siteConfig['minUsernameLength'] )	# Username too short?
			$errors .= 'Your username must be longer than ' . $siteConfig['minUsernameLength'] . ' characters!<br />';
		if( strlen($_POST['username']) > $siteConfig['maxUsernameLength'] )	# Username too long?
			$errors .= 'Your username must be shorter than ' . $siteConfig['maxUsernameLength'] . ' characters!<br />';
		if( $_POST['email'] != $_POST['email2'] )	# Emails match?
			$errors .= 'The entered email addresses do not match!<br />';
		if( $_POST['password'] != $_POST['password2'] )	# Passwords match?
			$errors .= 'The entered passwords do not match!<br />';
		$query = $conn->prepare( "SELECT username FROM users WHERE username = ?" ); # Check if the username exists...
		$query->bindValue( 1, $_POST['username']);
		$query->execute();
		if( $query->rowCount() > 0 ) 
			$errors = "Sorry that username exists already.<br />";
		if(filter_var($_POST['email'], FILTER_VALIDATE_EMAIL) && $_POST['password'] == $_POST['password2'] && empty($errors)) { # Valid email, passwords match and no errors exist? Insert some data!
			$hashed_password = password_hash($_POST['password'], PASSWORD_BCRYPT); # Hash the password with bCrypt... GOODLUCK HACKERS
			$query = $conn->prepare("INSERT INTO users (username, password, email) VALUES (:username, :password, :email)");
			$query->bindParam(':username', $_POST['username']);
			$query->bindParam(':password', $hashed_password);
			$query->bindParam(':email', $_POST['email']);
			$query->execute();
			$success = 'You have successfully registered on our site! Please visit the login page!';
		} else {
			$errors .= 'You have to enter a valid email address!<br />';
		}	
	}
	$page['formErrors'] = $errors; # Assign the errors variable
	$page['formSuccess'] = $success;
	echo $h2o->render(compact('page'));	#Render our page
?>
