Did anyone know you could make an array in scar using this?

MyArray := [‘value 1’, ‘value 2’, ‘value 3’, ‘value 4’];

Program QuickWayToMakeAnArray;

Var
  MyArray : Array of String;
  i : Integer;

Begin
  ClearDebug;
  MyArray := ['value 1', 'value 2', 'value 3', 'value 4'];
  WriteLn('A very quick way to make an array!');
  For i := 0 to GetArrayLength(MyArray)-1 do
  Begin
    WriteLn('  MyArray['+IntToStr(i)+'] := '''+MyArray[i]+''';');
  End;
End.
