{.Script Info: # ScriptName = Tree Smasher Advance # Author = Surajd # Description = Smashes aka Chops down any tree, anywhere =D # Version = v1.02 # Date = Date Published: 26/06/06 (v1.02) # Comments = Special Thanks to Spky for testing, WT-Fawki (SRL) and Kaitnieks (SCAR) and for everyone who is using! =D # Usage = Use Sunecrape, V-Bright, Setup Lines - 26 to 48, MultiUser Support will Auto-Run at start! # Usage = Axe in the FIRST INEVNTORY Slot! SCRIPT BEST USED NEAR A SINGLE TREE! /Script Info}

// V1.01 = Added some features and fixed a few bugs, Updated Dropping Procedure
// V1.02 = Added some FailSafes and lots of bug fixes and features! Also, New Chopping/Smashing Procedure: CREDITS TO Realrune for his RFindObj Function <3

Program TreeSmasherAdvance_by_Surajd;
{.include SRL\SRL.scar}
{.include SRL\SRL\Skill\Woodcutting.scar}
{.Include SRL\SRL\Misc\PlayerForm.scar}

Var
a, b : Integer;
STimes, FTimes, ETimes, DTimes, WTime, DCount : Integer;
A1, A2, A3, H : Integer;

Const
BenMouseBM = true; // Want to use BenLand100’s great groove? =D
TalkRandomly = true; // Talk Randomly or not?
TalkAboutSkill = true; // Types phrases about Woodcutting
AntiBanAB = true; // AntiBan functions or not?

RChat = 1; // Random-Talk time? EG: 2 for every 2 minutes
MouseSpeedMS = 11; // Change your Mouse-Speed, 11 works ok with BenMouse
WaitTime = 5000; // Waiting time between each Smash aka Chop
MaxWaitTime = 10000; // Max Waiting time between a Smash, aka Chop

TreeTypeTT = ‘Yew’; // Which Tree-Type are you Smashing? That is, the text which appears top-left of screeen (UPPER-CASE EG: Yew, Willow)
TTol = 10; // The Tolerance of finding the tree
HowManyDrops = 5; // Will Tree-Smash and after dropped the ammount of loads, will change player

Txt1 = ‘Yew’; // Enter any 3 letters on the text which appears top left! EG: “Yew” for Yew
Txt2 = ‘Cho’; // Enter any 3 letters on the text which appears top left! EG: “Cho” for Chop
Txt3 = ‘Dow’; // Enter any 3 letters on the text which appears top left! EG: “Dow” for Down

TColor1 = 4624512; // Tree Color 1
TColor2 = 3438696; // Tree Color 2
TColor3 = 870456; // Tree Color 3

LogColor = 6260645; // Color of the log you are Tree-Smashing in the inventory

//------------------------------------------------------
VersionNumber = ‘v1.02 by Surajd’; // Do not change!
//------------------------------------------------------

procedure ProgressReport;
begin
begin
WriteLn(’ ‘);
WriteLn(’<============== -’ + VersionNumber + ‘- Progress Report ===============>’);
WriteLn(’ Smashed: ’ + inttostr(STimes) + ’ Log(s)! ‘);
WriteLn(’ Dropped: ’ + inttostr(DTimes) + ’ Load(s)! ‘);
WriteLn(’ Fixed: ’ + inttostr(FTimes) + ’ Axe(s)! ‘);
WriteLn(’ Found: ’ + inttostr(ETimes) + ’ Ent(s)! ');
SRLRandomsReport;
end;
end;

Procedure SetupScript;
begin
Wait( 2122 + Random( 100 - 5 * 2 ));
PerfectNorth;
SetChat(‘hide’, 1);
SetChat(‘off’, 2);
SetChat(‘off’, 3);
SetRun (true);
GameTab(4);
end;

procedure AxeColors;
begin
GameTab(4);
Writeln(‘Getting Axe and Handle colors…’)
A1 := Getcolor(593, 220)
A2 := Getcolor(590, 219)
A3 := Getcolor(588, 220)
H := Getcolor(582, 220)
Writeln('Axe colors: ’ + inttostr(A1) + ’ ’ + inttostr(A2) + ’ ’ + inttostr(A3))
Writeln('Handle color: ’ + inttostr(H))
end;

// ------- Taken from Realrune’s: SRL Willow Killer + Banker ------ \
function RFindObj(var cx, cy: Integer; Text, Text2, Text3: String; color, color2, color3, tolerance: Integer): Boolean;
var
x, y, a, c, i, x1, y1, x2, y2 : Integer;
begin
if (FindMSColorTol(x, y, color, tolerance)) or (FindMSColorTol(x, y, color3, tolerance)) or (FindMSColorTol(x, y, color3, tolerance)) then
begin
x1 := 245;
y1 := 165;
x2 := 277;
y2 := 185;
repeat
if (not (Loggedin)) then
break;
a := a + 1;
if (a = 1) then
c := c + 1;
if (a = 3) then
c := c + 1;
for i := 1 to c do
begin
if (a = 1) then
begin
x1 := x1 + 30;
x2 := x2 + 30;
end;
if (a = 2) then
begin
y1 := y1 - 20;
y2 := y2 - 20;
end;
if (a = 3) then
begin
x1 := x1 - 30;
x2 := x2 - 30;
end;
if (a = 4) then
begin
y1 := y1 + 20;
y2 := y2 + 20;
end;
if (x1 = 485) and (x2 = 517) then
x2 := x2 - 2;
if (y1 = 325) and (y2 = 345) then
y2 := y2 - 7;
if (x2 > 515) then
Break;
if (FindColorTolerance(x, y, color, x1, y1, x2, y2, tolerance)) or (FindColorTolerance(x, y, color2, x1, y1, x2, y2, tolerance)) or (FindColorTolerance(x, y, color3, x1, y1, x2, y2, tolerance)) then
begin
MMouse(x, y, 0, 0)
cx:=x;cy:=y;
if (IsUpTextMulti(Text, Text2, Text3)) then
begin
Result := True;
Break;
end;
end;
end;
if (a = 4) then
a := 0;
until (x2 > 515) or (Result = True);
end;
end;
// ------- End of Realrune’s nice RFindObj =D ------ \

procedure AntiRandomsAR;
begin
Findnormalrandoms;
if ( AntiBanAB = True) then AntiBan;
if ( TalkRandomly = True) then RandomChatEvery( RChat );
if ( TalkAboutSkill = True) then GhostSpeak( ‘Woodcutting’ );
if ( AntiBanAB = True) then LeaveScreenEvery( 1 );
if ( FindFight ) then
begin
Writeln(‘Fight Found! Running…’);
RunAwayDirection(‘E’);
wait( 5000 + random( 1 + 99 / 2 ));
RunBack;
Writeln(‘Fight successfully dodged, continuing…’);
end;
FindEnt( TColor1 );
FindEnt( TColor2 );
FindEnt( TColor3 );
if (FindEnt( TColor1 )) or
FindEnt( TColor2 ) or
FindEnt( TColor3 ) then
ETimes := ETimes + 1
if (not (Findcolorspiral(x, y, A1, 554, 205, 743, 465))) then
begin
Writeln(‘Axe head missing! Looking very carefully to find it =/’);
FindHead;
FTimes := FTimes + 1
end;
end;

procedure ChopDownTree;
begin
repeat
if (RFindObj(a, b, Txt1, Txt2, Txt3, TColor1, TColor2, TColor3, TTol)) then
wait(523 + random(323 * 5 / 2));
repeat
wait(1023 + random(143 * 5 / 2));
WTime := WTime + 1;
until (WTime < MaxWaitTime);
Mouse(a, b, 1, 1, false);
ChooseOption(a, b, TreeTypeTT);
wait(WaitTime + Random(500 * 8 / 2 + 3 - 1));
AntiRandomsAR;
until (InvFull);
end;

Procedure DropLogsDL;
begin
if (InvFull) then
begin
DropTo(2, 28);
wait(1321 + random(224 * 4 / 1));
if (FindColorSpiralTolerance(a, b, LogColor, 565, 210, 734, 459, 4)) then
Mouse(a, b, 0, 0, false);
repeat
wait(421 + random(223 * 2 / 1))
DCount := DCount + 1
ChooseOption(a, b, ‘Drop’);
until (not( FindColorSpiralTolerance(a, b, LogColor, 565, 210, 734, 459, 4))) or ( DCount < 27 )
DTimes := DTimes + 1;
end;
end;

Procedure SetupPlayer;
begin
NextPlayer(true);
AxeColors;
Wait( 2122 + Random( 100 - 5 * 2 ));
PerfectNorth;
SetChat(‘hide’, 1);
SetChat(‘off’, 2);
SetChat(‘off’, 3);
SetRun (true);
GameTab(4);
end;

Procedure MainLoop;
begin
repeat
if (Not (LoggedIn)) then break;
SetupScript;
ChopDownTree;
DropLogsDL;
ProgressReport;
until (HowManyDrops = DTimes);
NextPlayer(true);
SetupPlayer;
if (Not (LoggedIn)) then NextPlayer(False);
MainLoop;
end;

Begin
DCount := 0
WTime := 0
STimes := 0
FTimes := 0
ETimes := 0
DTimes := 0
SetupSRL;
DisguiseScar(’ Notepad ');
SetupPlayers;
BenMouse := BenMouseBM;
MouseSpeed := MouseSpeedMS;
AxeColors;
MainLoop;
End.
