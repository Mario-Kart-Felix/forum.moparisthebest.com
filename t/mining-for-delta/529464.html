[code=java]package server.euthymia.jugador.habilidades;

import server.Server;
import server.euthymia.jugador.Client;
import server.euthymia.jugador.Config;
import server.euthymia.jugador.PlayerHandler;
import server.euthymia.jugador.Sounds;
import task.Task;
import util.clipping.region.ObjectDef;

public class Mining {

/*
 * The mining animation
 */
public static final int MINING_ANIMATION = 625;

/*
 * The mining soundz lolz
 */
public static final int MINING_SOUND = 432;
 
/*
 * the sound when there is no ore left in rock
 */
public static final int NO_ORE_SOUND = 431;


private enum pickaxeData {
	BRONZE(1265, 0),
	IRON(1267, 1),
	STEEL(1269, 2),
	MITHRIL(1273, 3),
	ADAMANT(1271, 4),
	RUNE(1275, 5);
	private int pickId, delayer;//Delayer means that that value will be removed from the ticks, so we can have some sort of pick efficiency
	
	private pickaxeData(int pickId, int delayer) {
		this.pickId = pickId;
		this.delayer = delayer;
	}
		
	public int getPick() {
		return pickId;
	}
		
	public int getDelay() {
		return delayer;
	}
}

private enum rockData {
	COPPER(2090, 436, 1, 17, 1),
	TIN(2094, 438, 1, 17, 1 ),
	IRON(2092, 440, 15, 35, 2),
	SILVER(2100, 442, 20, 40, 3),
	GOLD(2096, 444, 40, 65, 7),
	MITHRIL(2098, 447, 55, 80, 6),
	ADAMANT(2102, 449, 70, 95, 7),
	RUNE(2104, 451, 85, 125, 7);

	private int rockId, oreId, level, exp, delay;

	private rockData(int rockId, int oreId, int level, int exp, int delay) {
		this.rockId = rockId; 
		this.oreId = oreId;
		this.level = level;
		this.exp = exp;
		this.delay = delay;
	}
	
	public int getRock() {
		return rockId;
	}
	
	public int getOre() {
		return oreId;
	}
	
	public int getReqLevel() {
		return level;
	}
	
	public int getExp() {
		return exp * Config.xpRate;
	}
	 
	public int thicks() {
		return delay;
	}
}

public static void startPicking(Client player, int objectId) {
	for(rockData rock: rockData.values()) {
		for(pickaxeData pickaxe: pickaxeData.values()) {
			if(objectId == rock.getRock() && player.playerLevel[player.playerMining] >= rock.getReqLevel() && player.playerHasItem(pickaxe.pickId)) {
				String rockName = ObjectDef.getObjectDef(rock.getRock()).name;
				player.canMove = false;
				player.ore = rock.getOre();
				loopMine(player);
				getRock(player, rock.delay - pickaxe.delayer);
			}
		}
	}
}

public static void loopMine(Client player) {
	if(!player.canMove) {
		player.startAnimation(MINING_ANIMATION);
		Sounds.playerSoundEffect(player, MINING_SOUND);
	}
}

public static void getRock(final Client player, final int ticks) {
	Server.getTaskScheduler().schedule(new Task(2, true) {
		private int count = ticks;

		@Override
		protected void execute() {

			if (count > 0) {
				PlayerHandler.messageToAll = count + "...";
				count--;
			} else {
				Sounds.playerSoundEffect(player, NO_ORE_SOUND);
				player.canMove = true;
				player.startAnimation(0);
				player.addItem(player.ore, 1);
				stop();
			}
		}
	});
}

}
[/code]
