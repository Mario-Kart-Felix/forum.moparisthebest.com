I got bored and decided to use this as an excuse to practice using threads.

import java.awt.Color; import java.awt.Dimension; import java.awt.Graphics; import java.awt.Image; import java.awt.Rectangle; import java.awt.Toolkit; import javax.swing.JApplet; public class Ball extends JApplet implements Runnable { Graphics bufferGraphics; Image offscreen; Dimension dim; Thread animate; Rectangle paddle1,paddle2; Rectangle ball; static int x; static int y; static int stripe=20; static int hdirection=1; static int vdirection=-1; public void init() { setSize(520,520); x=250; y=250; dim=getSize(); offscreen = createImage(dim.width,dim.height); bufferGraphics = offscreen.getGraphics(); animate = new Thread(this); paddle1 = new Rectangle(10,200,15,50); paddle2 = new Rectangle(490,200,15,50); ball = new Rectangle(x,y,20,20); setFocusable(true); animate.start(); } public void paint(Graphics g) { bufferGraphics.clearRect(0,0,dim.width,dim.width); bufferGraphics.setColor(Color.black); bufferGraphics.fillRect(0,0,524,524); bufferGraphics.setColor(Color.white); bufferGraphics.fillOval(ball.x, ball.y, 20, 20); bufferGraphics.fillRect(paddle1.x,paddle1.y,paddle1.width,paddle1.height); bufferGraphics.fillRect(paddle2.x,paddle2.y,paddle2.width,paddle2.height); g.drawImage(offscreen,0,0,this); } public void update(Graphics g) { paint(g); } public void run() { while(true) { ball.x=x; ball.y=y; y+=vdirection*5; x+=hdirection*5; if(x > 500) hdirection=-1; else if( x < 0) hdirection=1; else if(y > 500) vdirection=-1; else if(y < 0) vdirection=1; else if(hdirection == -1) { if(paddle1.intersects(ball)) hdirection=1; else if(y > paddle1.y) paddle1.y+=5; else if(y < paddle1.y) paddle1.y-=5; } else if(hdirection == 1) { if(paddle2.intersects(ball)) hdirection=-1; else if(y > paddle2.y) paddle2.y+=5; else if(y < paddle2.y) paddle2.y-=5; } repaint(); try { animate.sleep(35); } catch(InterruptedException e) { } } } }
