Small app I wrote in C# to read in a directory of images, resize them each to a given size, then save them. I’m using it for the game I’m developing, as DirectX tends to complain when your textures aren’t a multiple of 2.

using System;
using System.Collections.Generic;
using System.Drawing;
using System.Drawing.Imaging;
using System.IO;
using System.Linq;

namespace TextureProcessor
{
    /// <summary>
    /// Stores an Image and Path
    /// </summary>
    struct Texture
    {
        public String Path;
        public Image Image;
        public Texture(String path)
        {
            this.Path = path;
            this.Image = Image.FromFile(path);
        }
        public Texture(String path, Image image)
        {
            this.Path = path;
            this.Image = image;
        }
    }
    class TextureProcessor
    {
        private string[] extentions = { ".jpg", ".png", ".gif", ".tif", ".tiff", ".jpeg" };
        static void Main(string[] args)
        {
            try
            {
                String path = args[0].TrimEnd('"').TrimStart('"');
                int sizeX = int.Parse(args[1]);
                int sizeY = int.Parse(args[2]);
                bool openExplorer = args[3] != null && args[3] == "-e";
                new TextureProcessor(path, new Size(sizeX, sizeY), openExplorer);
            }
            catch (Exception)
            {
                Console.WriteLine("Usage: textureprocessor \"folder_path\" size_x size_y [-e open explorer]");
            }
            Console.WriteLine("Finished processing.");
            Console.ReadKey();
        }
        public TextureProcessor(String path, Size size, bool openExplorer)
        {
            DirectoryInfo directory = new DirectoryInfo(path);
            List<Texture> input = new List<Texture>();
            foreach (FileInfo file in directory.GetFiles())
            {
                if (extentions.Contains(file.Extension))
                {
                    input.Add(new Texture(file.FullName));
                }
            }
            Console.WriteLine("Beginning resize to {0}x{1}", size.Width, size.Height);
            List<Texture> output = Resize(input, size);
            foreach (Texture tex in output)
            {
                tex.Image.Save(tex.Path, ImageFormat.Jpeg);
                Console.WriteLine("Processed {0}", Path.GetFileName(tex.Path));
            }
            if (openExplorer)
            {
                System.Diagnostics.Process.Start(path);
            }
        }
        /// <summary>
        /// Resizes a list of textures
        /// </summary>
        /// <param name="textures">The list of textures to resize</param>
        /// <param name="size">The size to which each texture should be resized</param>
        /// <returns>A list of resized textures, ready to be saved</returns>
        public List<Texture> Resize(List<Texture> textures, Size size)
        {
            List<Texture> output = new List<Texture>();
            foreach (Texture tex in textures)
            {
                Image texOuput = new Bitmap(size.Width, size.Height);
                using (Graphics g = Graphics.FromImage(texOuput))
                {
                    g.DrawImage(tex.Image, 0f, 0f, (float)size.Width, (float)size.Height);
                }
                output.Add(new Texture(tex.Path, texOuput));
                tex.Image.Dispose();
            }
            return output;
        }
    }
}

Usage: textureprocessor “folder_path” size_x size_y [-e open explorer]
folder_path is obviously the folder of images you wish to resize, the quotes are there because Windows sucks and allows spaces in paths.
size_x and size_y are self explanatory
If you provide the -e argument, the program will open the directory of images when it has finished processing them.
