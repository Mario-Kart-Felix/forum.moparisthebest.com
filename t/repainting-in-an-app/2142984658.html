Okay… I made something what might help you, it’s an applet, it repaints, via repaint() in run(), and it uses double buffering

import java.awt.*;
import java.applet.*;

public class Applet1 extends Applet implements Runnable
{
    Thread timer;           // The timer (Thread)
    Image offscreen;        // For double buffering
    Graphics b;             // For double buffering
    int X = 240;            // X Coordinate
    int Y = 140;            // Y Coordinate
    final int R = 20;       // Radius of the oval
    
    final int cycletime = 100;   // Sleep every 100 ms
    
	public void init() {

		setLayout(null);
		setSize(500,300);
		offscreen = createImage(getWidth(), getHeight());
		b = offscreen.getGraphics();
		timer = new Thread(this);
		timer.start();
	}
	
	public void paint(Graphics g) { 
	    b.clearRect(0,0,getWidth(),getHeight());
	    b.fillOval(X, Y, R, R);
	    g.drawImage(offscreen,0,0,this);
	    
	    X += (Math.random()*10 >= 5)?(Math.random()*30):(Math.random()*-30);
	    Y += (Math.random()*10 >= 5)?(Math.random()*30):(Math.random()*-30);
	    
	    if (X+R > getWidth())
	        X = getWidth()-R;
	    if (X < 0)
	        X = 0;
	    if (Y+R > getHeight())
	        Y = getHeight()-R;
	    if (Y < 0)
	        Y = 0;
	    
	}
	public void update(Graphics g) {
	    paint(g);
	}
	
	public void run() {
	    while (true) {
	        repaint();
	        try {
	            timer.sleep(cycletime);
	        }
	        catch (InterruptedException ie) {
	            System.out.println("Error in sleeping: "+ie); 
	            timer = null;
	        }
	    }
	}
}
