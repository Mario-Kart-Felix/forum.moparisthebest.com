package game.model.text;

import java.io.BufferedReader;
import java.io.IOException;
import java.nio.charset.Charset;
import java.nio.file.Files;
import java.nio.file.NoSuchFileException;
import java.nio.file.Paths;

/**
*@author Tyler (Tyb97)
*/
public class DialogueGrabber {

	static Charset charset = Charset.forName("US-ASCII");

	static String dir = new String("./data/speech/");
	
	static Dialogue speech;

	public static void grabDialogues(int index) {
		try (BufferedReader reader = Files.newBufferedReader(
				Paths.get(dir + index + ".txt"), charset)) {
			String line = null;
			int id = -1;
			String[] text = new String[3];
			while ((line = reader.readLine()) != null) {
				if (line.startsWith("ID:"))
					id = Integer.parseInt(line.substring(4));
				else if (line.startsWith("Text 1"))
					text[0] = line.substring(7);
				else if (line.startsWith("Text 2"))
					text[1] = line.substring(7);
				else if (line.startsWith("Text 3"))
					text[2] = line.substring(7);
			}
			speech = new Dialogue(id, text);
		} catch (NoSuchFileException e) {
			e.printStackTrace();
		} catch (IOException io) {
			io.printStackTrace();
		}
	}
	
	public static void playSpeech(String name, int id) {
		for (int i = 0; i < speech.getDialogues().length; i++) {
			GameMessage.sendMessage(name, speech.getDialogues()[i]);
		}
	}

}

If you do decide to go through with this, take a look at this, it’s taken from my text based game, bascially you can make multiple .txt files for each of the dialogues, instead of having to type out all the speech out in Java form, it’s much much quicker this way.

And of course, you can just increase the amount of handled text to 4 lines, since that’s the max for the Rs2 format.
