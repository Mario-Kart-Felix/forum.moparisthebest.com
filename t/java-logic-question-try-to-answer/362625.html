	}

		int highest = num[0];
		for(int i : num) {
			if(i > highest) {
			    highest = i;
			}
		}
		return highest;	
	}
	
	public int getAverage(int[] num) {
		int sum = 0;
		for(int i : num) {
			sum += i;
		}
		return sum / num.length;

EDIT: This one also works, and doesn’t use three loops. Just the one.

import java.util.Scanner;


public class GameQuestion {

	public static void main(String[] args) {
		GameQuestion game = new GameQuestion();
		game.scan = new Scanner(System.in);
		
		System.out.print("How Many Games?: ");
		game.logic(game.scan.nextInt());
		
	}
	
	public void logic(int len) {
		gameScores = new int[len];
		
		for(int i = 0; i < len; i++) {
			System.out.println("Game Score: ");
			gameScores[i] = scan.nextInt();
			
			if(i == 0) {
				lowest = gameScores[i];
			}
			
			if(gameScores[i] > highest) {
				highest = gameScores[i];
			} else if(gameScores[i] < gameScores[i-1]) {
				lowest = gameScores[i];
			}
			
			average += gameScores[i];
		}
		System.out.println("Higest: " + highest);
		System.out.println("Lowest: " + lowest);
		System.out.println("Average: " + average / gameScores.length);
	}
	
	Scanner scan;
	int[] gameScores;
	int highest = 0, lowest = 0;
	double average = 0.0;
}

public class Game {

public static void main(String[] args) {

			lowest = i;
		if ( i > highest)
			highest = i;
		average+=i;
		numbers[count] = i;
		count++;
