Procedure ClickM(mousex,mousey,ranx,rany:integer; left:boolean);
var tries,b,c:integer;
begin
GetMousePos(b,c);
HoldMouse(b,c,left);
repeat
wait(10+random(40));
tries:=tries+1;
until(tries>6);
GetMousePos(b,c);
wait(10+random(40));
ReleaseMouse(b,c,left);
end;
