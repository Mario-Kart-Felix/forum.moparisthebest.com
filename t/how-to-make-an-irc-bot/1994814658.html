	/*User For Making This Our Super Class*/
	public BaseCode(String userName,String userPass,String UserDescription,String Ip,int Port)
	{
		this.userName = userName;
		this.userPass = userPass;
		this.UserDescription = UserDescription;
		this.Ip = Ip;
		this.port = Port;
	}

	/*Connecting To A Irc Server*/
	public void connect()
	{
		try
		{
			connector = new Socket(Ip, port);
		}
		catch(Exception E)
		{
			System.out.println(E.toString());
			E.printStackTrace();
		}
	}
	/*user To Identifing,Joining Channels, and setting modes*/
	public void setup()
		{
			Send("USER " + userName + " 8 & " + UserDescription);
			Send("NICK " + userName);
			Send("Mode " + userName + " +BpT");
			if(userPass != null)
			{
				Send("NS IDENTIFY " + userPass);
			}
			for(int X = 0; X < channels.length; X++)
			{
				if(channels[X] != null)
				{
					Send("JOIN " + channels[X]);
				}
			}
			connected = true;
	}
	/*Used To Send Data To The Irc Server*/
	public void Send(String Sent)
	{
		try
		{
			output = new BufferedWriter(new OutputStreamWriter(connector.getOutputStream()));
			output.write(Sent);
			output.newLine();
			output.flush();
		}
		catch(Exception E)
		{
		System.out.println(E.toString());
		}
	}
