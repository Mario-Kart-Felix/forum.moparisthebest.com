function WaterWalk2(StartRadial, EndRadial: Integer; Radius,
FFlagBreakOut: Integer; Xoff, Yoff, Xmod, Ymod: Integer): Boolean;
var
i, X1, Y1: Integer;
begin
Result := False;
if(StartRadial = EndRadial)then
WriteLn(‘ERROR - WaterWalk can’‘t have the same values for both Start and End Radials!’)
else if (StartRadial < EndRadial) then
begin
repeat
for i := StartRadial to EndRadial do
begin
x1 := Round(Radius * Sine(i)) + 646;
y1 := Round(-Radius * Cose(i)) + 84;
if FindColor(x, y, WaterColor, X1, Y1, X1 + 1, Y1 + 1) then
begin
Mouse(x + Xoff, y + Yoff, Xmod, Ymod,True);
Result := True;
FFlag(FFlagBreakOut);
Exit;
end;
end;
Radius := Radius - 4;
until Radius <= 1;
end else
if (StartRadial > EndRadial) then
begin
repeat
for i := StartRadial downto EndRadial do
begin
x1 := Round(Radius * Sine(i)) + 646;
y1 := Round(-Radius * Cose(i)) + 84;
if (FindColor(x, y, WaterColor, X1, Y1, X1 + 1, Y1 + 1)) then
begin
Mouse(x + Xoff, y + Yoff, Xmod, Ymod,True);
Result := True;
FFlag(FFlagBreakOut);
Exit;
end;
end;
Radius := Radius - 4;
until Radius <= 1;
end;
end;
