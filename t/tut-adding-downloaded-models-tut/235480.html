Now you should have 2 new files in your godsword folder
Rename the one called 460012.dat to 460012
Rename the one called 460001.dat to 460001

The numbers are the model NUMBER so you cant have letters.
Now Hold shift and copy them both.

Now open the Modeling Client also found in my pack,

to empty it out lol, After the modeling folder is empty paste
the 2 Files you copied earlier in here.

class8.aByteArray178 = "A great weapon, used by the best warriors.".getBytes(); }

if(i == 134567){ class8.aStringArray189 = new String[5]; class8.aStringArray189[1] = "Wear";//Caption to click when you want to wield it. class8.anIntArray156 = new int[1];(item origional Colour) class8.anIntArray160 = new int[1];(item origional Colour) class8.anIntArray156[0] = 0;(item changed Colour) class8.anIntArray160[0] = 0;(item Changed Colour) class8.anInt174 = 140012;//item Drop Model id (140012) class8.anInt181 = 1957;//Zoom Number class8.anInt190 = 498; class8.anInt198 = 484; class8.anInt204 = 0; class8.anInt169 = -1; class8.anInt194 = -1; class8.anInt165 = 140001;//Male Model id (140001) class8.anInt200 = 140001;//Female Model ID This will be 140001 class8.anInt175 = -1; class8.anInt197 = -1; class8.aString170 = "Item Name ";//Item Name class8.aByteArray178 = "Examin Item Text Goes Here".getBytes();// Examin Text }
