[center][size=18pt]Calculator[/size][/center]
[center]@echo off title Calculator V1.0 by Anthony color 04 :top echo -------------------------------------------------------------- echo Calculator V1.0 by Anthony echo Type your Math Problem below echo -------------------------------------------------------------- set /p equation= set /a ans= %equation% cls echo %equation% = %ans% pause > Nul cls echo Previous Question: %equation% = %ans% goto top exit[/center]
[center][size=18pt]Guessing Game[/size][/center]
[center]@echo off color 0e title Guessing Game 1.0 by Anthony set /a guessnum=0 set /a answer=%RANDOM% set variable1=AnthonyRocks echo ------------------------------------------------- echo Welcome to the Guessing Game! echo. echo Try to guess the number! echo ------------------------------------------------- echo. :top echo. set /p guess= echo. if %guess% GTR %answer% ECHO Lower! if %guess% LSS %answer% ECHO Higher! if %guess%==%answer% GOTO EQUAL set /a guessnum=%guessnum% +1 if %guess%==%variable1% ECHO Hey, you found the secret number, the answer is: %answer% goto top :equal echo You Win! echo You Win! echo You Win! echo. echo It took you %guessnum% guesses. echo. pause [/center]
[center][size=18pt]Folder Protector[/size][/center]
[center]@echo off title Folder Protector 1.0 by Anthony color 0a set /a tries=3 set password=Password :top echo. echo ---------------------------------------------- echo. echo Folder Protector 1.0 by Anthony echo. echo ---------------------------------------------- echo You have %tries% attempts left. echo Enter password echo ---------------------------------------------- set /p pass= if %pass%==Password ( goto correct ) set /a tries=%tries -1 if %tries%==0 ( goto penalty ) cls goto top :penalty echo Sorry, too many incorrect passwords, initiating shutdown. start shutdown -s -f -t 35 -c "SHUTDOWN INITIATED" pause exit :correct cls echo ---------------------------------------------- echo Access Granted. echo. echo Folder Now Opening echo ---------------------------------------------- explorer Folder pause[/center]
