var
count : integer;
res : extended;
inv : boolean;
begin
if(abs(e) > 1)then
begin
e := pow(e,-1);
inv := true;

inv := false;

end;

procedure gravity(gravpointx,gravpointy,grav,pull:integer);
var
gravx,gravy:integer;
slope:extended;
begin
if(x - gravpointx = 0)or(y - gravpointy = 0)then
begin
if(x - gravpointx = 0)then
begin
gravy:= grav;
end else
begin
gravx:= grav;
end;
end else
begin
slope:= (x - gravx)/(y - gravy);
gravx:= round(grav/(1 + slope));
gravy:= round((grav/(1 + slope)) * slope);
end;
if(gravpointx > x)then
begin
x:= x - gravx;
end else
begin
x:= x + gravx;
end;
if(gravpointy > y)then
begin
y:= y - gravy;
end else
begin
y:= y + gravy;
end;
end;

In the code of the autoer, it will say something like //USER VARIABLES//, under here is where you make your settings. MonsterColor1-6 is where you input the colors of the monster you are fighting. UserPlayerForm is set to true if it is the first time you are running the script. RotateScreen simply rotates the screen, CheckUpText should probably be left true if you don’t know what you’re doing, and everything else is pretty easy to understand. What UsePlayerForm does is open up SRL’s player manager and let’s you enter your user/pass/nick combination. You can enter more than one character and the TimeToRun=25 means that it will run your character for 25 minutes, then switch characters, and keep doing this. If you only want one character just set TimeToRun to be really high (in the 500 minutes range) and it will never get a chance to switch.
There isn’t any specific place you have to train, it all just depends on what color settings you have.
