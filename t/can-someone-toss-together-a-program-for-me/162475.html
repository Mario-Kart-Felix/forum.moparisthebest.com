[code]import java.io.DataInputStream;
import java.io.DataOutputStream;
import java.net.Socket;

public class Client
{
private Socket soc = null;
private DataInputStream in = null;
private DataOutputStream out = null;

public Client()
{
	try
	{
		soc = new Socket("localhost", 7357);
		in = new DataInputStream(System.in);
		out = new DataOutputStream(soc.getOutputStream());
	}
	catch (Exception e)
	{
		e.printStackTrace();
	}
	start();
}

public void start()
{
	try
	{
		while(true)
		{
			System.out.print("Message : ");
			String line = in.readLine();

			out.writeInt(0);
			out.writeUTF(line);
			

			System.out.println("Sent packet : 0");
			System.out.println("Message : " + line);
		}
	}
	catch (Exception e)
	{
		e.printStackTrace();
	}
}

public static void main(String[] args)
{
	new Client();
}

}

[code]import java.io.DataInputStream;
import java.io.DataOutputStream;
import java.io.IOException;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.ArrayList;

public class mainClass {
private ServerSocket serv = null;
private DataInputStream in = null;
private DataOutputStream out = null;
private ArrayList ips = new ArrayList();
private String inputMessage = “”;
private int legnth = 0;
private Socket soc1 = null;

public mainClass()
{
	try
	{
		serv = new ServerSocket(7357);
	}
	catch (IOException e)
	{
		e.printStackTrace();
	}
	listen();
}

private void listen()
{
	while(true)
	{
		try
		{
			Socket soc = serv.accept();
			in = new DataInputStream(soc.getInputStream());
			System.out.println("Connected");
			
			try
			{
				int i = 0;
				while((i = in.readInt()) != -1)
				{
                                            
					inputMessage = in.readUTF();
					if (inputMessage.startsWith("ip:")) {
						ips.add(inputMessage.substring(4));

					} else {

						legnth = ips.size();
						for (int x = 0;  x <= legnth; x++) {
							try
							{
								soc1 = new Socket(ips.get(x), 7357);
								out = new DataOutputStream(soc1.getOutputStream());
								out.writeUTF(in.readUTF());
							}
							catch (Exception e)
							{
								e.printStackTrace();
							}
						}
					}
					
				}
			}
			catch(Exception e)
			{
				e.printStackTrace();
			}
		}
		catch(Exception e)
		{
			
		}
	}
}
public static void main(String[] args) {
	new mainClass();
}

}[/code]

if the string starts with ip: then it will add an ip to the ip list (this packet will be sent when someone signs on, this is the server side of a chat thing)

if it doesn’t start with ip: then it will go to the for loop and send the string to all of the ips
