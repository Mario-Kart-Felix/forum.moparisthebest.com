Make a new class and name is Interface. Fix the imports and packaging also…

package org.cleaned.rs2.model.quests.tutorial;

import org.cleaned.rs2.model.Client;

/**
 * Interfaces
 *
 * @author Arham 4
 */
public class Interface {

	/**
	 * Creates a flashing yellow arrow on certain coordinate
	 * 
	 * @param x
	 * @param y
	 * @param h
	 * @param pos
	 */
	public void createCoordIcon(int x, int y, int h, Client c) {
		createArrow(x, y, h, 2, c);
	}

	/**
	 * createArrow method for simpler arrow methods
	 * 
	 * @param type
	 * @param id
	 * @param c
	 */
	public static void createArrow(int type, int id, Client c) {
		Client.outStream.createFrame(254); // The packet ID
		Client.outStream.writeByte(type); // 1=NPC, 10=Player
		Client.outStream.writeWord(id); // NPC/Player ID
		Client.outStream.write3Byte(0); // Junk
	}

	/**
	 * createArrow method for simple arrow methods
	 * 
	 * @param x
	 * @param y
	 * @param height
	 * @param pos
	 * @param c
	 */
	public static void createArrow(int x, int y, int height, int pos, Client c) {
		Client.outStream.createFrame(254); // The packet ID
		Client.outStream.writeByte(pos); // Position on Square(2 = middle, 3 =
											// west, 4 = east, 5 = south, 6 =
											// north)
		Client.outStream.writeWord(x); // X-Coord of Object
		Client.outStream.writeWord(y); // Y-Coord of Object
		Client.outStream.writeByte(height); // Height off Ground
	}

	/**
	 * Adds side bar menu to the client
	 * 
	 * @param menuId
	 * @param form
	 * @param c
	 */
	public static void setSideBarMenu(int menuId, int form, Client c) {
		c.setSidebarInterface(menuId, form);
	}

	/**
	 * Creates a flashing yellow icon on a specific npc
	 * 
	 * @param c
	 * @param EntityIndex
	 *            defines which npc
	 * @param c
	 */
	public static void createEntityIcon(Client c, int EntityIndex) {
		createArrow(1, EntityIndex, c);
	}

	/**
	 * Removes a flashing yellow icon from a specific npc
	 * 
	 * @param c
	 * @param EntityIndex
	 *            defines the npc
	 * @param c
	 */
	public static void removeEntityIcon(Client c, int EntityIndex) {
		createArrow(0, EntityIndex, c);
	}
}
