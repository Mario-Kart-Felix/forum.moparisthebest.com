This is only a snippet and can be easily converted to any base.

You’re going to have to do all the coding of implanting this to work yourself such as making the banker issue cards, payment dates and making your stores accept the currency on the card (just convert some point shop system).

The idea behind this was pretty cool about “economizing” a server? Well the gift card could be used in many ways as it could be a donator option for those players wanting to give free cash out or whatever
Even add this to an Item for a more realistic feel?

To have an idea on how to use this system check out this paste: http://pastebin.com/Gs2GBZwB

[code=java]package com.kitty.money;

import com.kitty.Client;

/**

  • Superclass/parent class to handle money such as making payments and holding
  • your balance
  • @author Victoria Connelly

/
public class Money {
/*
* Defining Client ‘c’
*/
public Client c;

/**
 * balance i.e. money
 */
protected int balance;

/**
 * Make purchase method
 * 
 * @param amount
 */
public void makePurchase(int amount) {
	if (amount > balance) {
		c.sendMessage("Your balance is not high enough to make this purchase.");
                    return;
	}
	balance = balance - amount;
}

/**
 * Gets your available balance
 * 
 * @return balance
 */
public int getBalance() {
	return balance;
}

}
[/code]

[code=java]package com.kitty.money;

/**

  • Gift card allows you to spend a x amount stored on the gift card
  • @author Victoria Connelly

*/
public class GiftCard extends Money {
public GiftCard(int amount) {
balance = amount;
}
}
[/code]

[code=java]package com.kitty.money;

/**

  • Class to handle debit card purchase and balance and depositing more onto card
  • @author Victoria Connelly

*/
public class DebitCard extends Money {

/**
 * Constructor is used for instantiating the debit card allowing you to
 * create new objects to use the debit card appropriatly
 * 
 * @param amount
 */
public DebitCard(int amount) {
	balance = amount;
}

/**
 * Add money onto debit card
 * 
 * @param amount
 */
public void deposit(int amount) {
	balance = balance + amount;
}

}
[/code]

[code=java]package com.kitty.money;

/**

  • Class to handle credit card purchase and balance
  • @author Victoria Connelly

/
public class CreditCard extends Money {
/*
* The amount you can spend on the credit card
*/
private int credit;

/**
 * Credit card constructor allowing you to instantiate when you're using
 */
public CreditCard(int amount) {
	balance = 0;
	credit = amount;
}

/**
 * Handles paying of the credit card
 * 
 * @param amount
 */
public void makePayment(int amount) {
	balance = balance - amount;
}

/**
 * Gets available credit on credit card
 * 
 * @return
 */
public int getCredit() {
	return credit - balance;
}

/**
 * Make purchase method checks if you have enough credit/money to make a
 * purchase and handles the amount
 */
public void makePurchase(int amount) {
	if (amount > getCredit()) {
		c.sendMessage("You do not have enough credit to make this purchase.");
                    return;
	}
	balance = balance + amount;
}

}
[/code]
