int[] DragonImpRare = {/*add your rare reward items here*/}; /*create your list of items and be sure to seperate with commas*/ public int DragonImpRare() { return DragonImpRare[(int) (Math.random() * DragonImpRare.length)]; } int[] DragonImpCommon = {/*item ID's*/}; public int DragonImpCommon() { return DragonImpCommon[(int) (Math.random() * DragonImpCommon.length)]; } int[] DragonImpBones = {18830}; /*i used frost dragon bones but you can use any bones you want and is even set up to give a random bone if you choose*/ public int DragonImpBones() { return DragonImpBones[(int) (Math.random() * DragonImpBones.length)]; /*finds a random item from the list declared above*/ }

this is just a list of items telling the server what items to add, and gives it a random item from the list.

public void Lootimpjar(int itemId, int playerId) { if (itemId == 11256){ /*0.01% chance on the first try*/ if (Misc.random(1000) <= (1 + Misc.random(ImpLuck))) { /*picks a random number out of 1000 if it eaquals 1 + your luck number or less you get a random RARE item*/ getItems().addItem(DragonImpRare(), 1); c.getItems().deleteItem(11256, 1); sendMessage("You open the jar and find a <shad=15695415>RARE </col>item"); ImpLuck = 0;// sets the luck back to 0 when a rare item is earned }else{ if (Misc.random(250) <= (10 + Misc.random(ImpLuck))) { getItems().addItem(DragonImpCommon(), 1); c.getItems().deleteItem(11256, 1); sendMessage("You open the jar and find a <col=1532693>COMMON </col>item"); ImpLuck -= 2; //takes away 2 luck when a player gets a common item from the jar }else{ getItems().addItem(DragonImpBones(), 1); calls on the randomizer to pick an item id c.getItems().deleteItem(11256, 1); deletes the imp jar from the players inventory sendMessage("You open the jar and find some bones");//tells the player what item he got //sendMessage("impluck : "+ImpLuck+"."); for testing my luck method ImpLuck += 1;// adds luck after each fail } } } }
