Class: SerialHandler
Purpose: This will handle all File read’s and writes with Serializable objects.

[code=java5]
public class SerialHandler {

public static Object fetch(String file) {
    Object o = null;
    java.io.FileInputStream fis = null;
    ObjectInputStream dis = null;
    try {

        fis = new java.io.FileInputStream(file);
        dis = new ObjectInputStream(fis);

        o = dis.readObject();
    } catch (FileNotFoundException ex) {
    } catch (IOException ex) {
        System.err.println("IO exception fetching file:" + file);
    } catch (ClassNotFoundException ex) {
        System.err.println("Cant fix class not found?");
    } finally {
        try {
            if (fis != null) {
                fis.close();
            }
            if (dis != null) {
                dis.close();
            }
        } catch (IOException ex) {
            Logger.getLogger(Handler.class.getName()).log(Level.SEVERE, null, ex);
        }
    }
    return o;
}

public static void put(String file, Object obj) {
    if (obj instanceof Serializable == false) {
        return;
    }
    java.io.FileOutputStream fos = null;
    java.io.ObjectOutputStream oos = null;
    try {
        try {
            fos = new FileOutputStream(file, false);
            oos = new ObjectOutputStream(fos);
            oos.writeObject(obj);
        } catch (FileNotFoundException fnfe) {
        } catch (IOException ex) {
        } finally {
            if (fos != null) {
                fos.close();
            }
            if (oos != null) {
                oos.close();
            }
        }
    } catch (IOException ioe) {
    }
}

Class: Security
Purpose: Kind of a baseline example of how this would work. Yours will probably not run off of a securitymanager like mine, but i did document this to explain how it works.

[code=java5]public class Security extends SecurityManager implements Saveable {
/Keep these when saving/

/**
 * bannedIP - holds the list of IP's in numeric.
 */
private final List<Integer> bannedIP;
/**
 * bannedPlayer - holds the list of Players who are permanently banned from
 * this server.
 */
private final List<String> bannedPlayer;
/**
 * bannedMAC - holds the list of all banned NIC cards.
 */
private final List<Long> bannedMAC;
/**
 * connected - the list of users who are current connected to the server.
 * transient means it will not be included in the Serialized object.
 */
final transient LinkedList<String> connected;
/**
 * creates a new Copy of Security
 */
public Security() {
    connected = new LinkedList();
    bannedIP = new ArrayList();
    bannedPlayer = new ArrayList();
    bannedMAC = new ArrayList();
}
/**
 * Private method
 * @param numericIP - integer to format as an IP address using 
 * Dotted-decimal notation
 * @return Formatted view of the numericIP in "0.0.0.0" notation
 */
private String formatIP(int numericIP) {
    StringBuilder b = new StringBuilder();
    for (int i = 3; i >= 0; i--) {
        b.append((numericIP << i * 8) & 0xff);
        if (i != 0) {
            b.append('.');
        }
    }
    return b.toString();
}
/**
 * Private method
 * @param macAdd - macAddress to format
 * @return formatted Mac with standard "00:00:00:00:00:00" notation
 */
private String formatMAC(long macAdd) {
    StringBuilder b = new StringBuilder();
    for (int i = 5; i >= 0; i--) {
        b.append(Long.toHexString((macAdd << i * 8) & 0xff));
        if (i != 0) {
            b.append(':');
        }
    }
    return b.toString().toUpperCase();
}
/**
 * 
 * @param numericIP - numeric form of IP address to check against the 
 * bannedIP table.
 * @param port - port the client is attempting to connect too.
 * @throws SecurityException - if the port is not accepted or if numericIP 
 * is found in the banned IP table.
 */
public void checkIPAndPort(int numericIP, int port) {
    if (port != 43594) {
        throw new SecurityException(formatIP(numericIP) + " tried connecting on an invalid port:" + port);
    }
    if (bannedIP.contains(numericIP)) {
        throw new SecurityException(formatIP(numericIP) + " attempted to connect but is IP banned");
    }
}

/**
 *
 * @param MAC - MAC address to check against the banned MAC table
 * @throws SecurityException - if the address is found in the table.
 */
public void checkMAC(long MAC) {
    synchronized (bannedMAC) {
        if (bannedMAC.contains(MAC)) {
            throw new SecurityException(formatMAC(MAC) + " attemted to connect but is a banned Computer");
        }
    }
}

/**
 *
 * @param numericIP the IP address (IPv4) of the host to ban in numeric
 * form.
 */
public void banIP(int numericIP) {
    synchronized (bannedIP) {
        if (!bannedIP.contains(numericIP)) {
            bannedIP.add(numericIP);
        }
    }
}

/**
 * @param playerName - player to append to the banned Players table. Name is
 * only added if it does not already exist
 */
public void banPlayer(String playerName) {
    synchronized (bannedPlayer) {
        if (!bannedPlayer.contains(playerName)) {
            bannedPlayer.add(playerName);
        }
    }
}

/**
 * @param name Name to check against the list of banned Players
 * @throws SecurityException - if the name is found in the list
 */
public void checkPlayer(String name) {
    synchronized (bannedPlayer) {
        if (bannedPlayer.contains(name)) {
            throw new SecurityException(name + " is banned!");
        }
    }
}

/**
 * Synchronized removal of all the MAC address from the table of banned MACs
 * that match MAC.
 *
 * @param ip - the ip address in numeric form to remove from banned list.
 * @return If the ip address was removed from the list.
 */
public boolean removeMAC(long MAC) {
    synchronized (bannedMAC) {
        boolean ret = bannedMAC.remove(MAC);
        while (bannedMAC.remove(MAC));
        return ret;
    }
}

/**
 * Synchronized removal of all the IP address from the table of banned IPs
 * that match ip.
 *
 * @param ip - the ip address in numeric form to remove from banned list.
 * @return If the ip address was removed from the list.
 */
public boolean removeIP(int ip) {
    synchronized (bannedIP) {
        boolean ret = bannedIP.remove((Integer) ip);
        while (bannedIP.remove((Integer) ip));
        return ret;
    }
}

/**
 * Synchronized removal of all the occurrences from the table of banned
 * Players that match playerName.
 *
 * @param playerName - the Player's name to remove from the banned list.
 * @return If the name was removed from the list.
 */
public boolean removePlayer(String playerName) {
    synchronized (bannedPlayer) {
        boolean ret = bannedPlayer.remove(playerName);
        while (bannedPlayer.remove(playerName));
        return ret;
    }
}

@Override
public String getFilePath() {
    return "./SERIAL/Security.ser";
}

import java.io.FileOutputStream;
import java.io.IOException;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.io.Serializable;
import java.util.logging.Logger;

public final class SerialHandler
{

	private static final Logger logger = Logger.getLogger( SerialHandler.class.getName() );


	public static Object fetch( String file ) throws ClassNotFoundException, IOException
	{
		try( ObjectInputStream inputStream = new ObjectInputStream( new FileInputStream( file ) ) ) {
			return inputStream.readObject();
		}
	}


	public static void put( String file, Object obj ) throws IOException
	{
		if( obj.getClass().getSuperclass() != Serializable.class ) {
			logger.severe( obj + " is not an instance of serializable." );
			return;
		}

		try( ObjectOutputStream outputStream = new ObjectOutputStream( new FileOutputStream( file, false ) ) ) {

		}
	}

	private SerialHandler() { }
