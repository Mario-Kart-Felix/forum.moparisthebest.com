package org.arham4.model.player.quests;

import org.arham4.model.Client;

        public class DialogueHandler {

        /**
        *@Author: Arham Siddiqui
        *@Time: 9/29/12 - 11:54 PM - Saturday
        *@Purpose: Easier dialogue handling than normal delta servers.
        */

                Client c;
                public DialogueHandler (Client c) {
                        this.c = c; //Allows us to call Client
                }

                /**
                *Makes the dialogue go onwards.
                */

                public void n() {
                        c.NpcDialogue++; //Allows the dialogue to continue
                        c.NpcDialogueSend = true; //Don't know why but adding...
                }

                /**
                *Makes the dialogue end.
                */

                public void f() {
                        c.NpcDialogue = 0;
                        c.NpcDialogueSend = false; //Made to make the boolean false so it doesn't continue any dialogue.
                }

                /**
                *@npcId Gets the npc id so it can call the dialogue.
                *@NpcDialogue The integar for dialogues. NpcDialogue = 0;
                *All the chit chat is here...
                */

                public void Conversations() {
                        switch (c.npcId) { //Effective easier way for getting dialogue ids rather than a whole new void calling a continueDialogue void.
                        case 1: //If the npc you clicked is Man...
                                switch (c.NpcDialogue) { //made to ensure dialogue is going on here.
                                case 0: //send this first
                                        sendNpcChat("Yay! This worked!");
                                        n(); //increases NpcDialogue to continue dialogue
                                break;
                                case 1: //send this after case 0
                                        sendPlayerChat("I know awesome right?");
                                        n(); //increases NpcDialogue to continue dialogue
                                break;
                                case 2: //sends this after case 1
                                        sendNpcChat("Yeah bro!");
                                        n(); //increases NpcDialogue to continue dialogue
                                break;
                                case 3: //sends this after case 2
                                        f(); //closes all dialogue
                                break;
                                } //Close the dialogue chit chat for npc Man.
                        break; //Allows us to use a different npc now.
                        case 995: //If npc is RuneScape Guide...
                                if (c.tutorialProgress = 0) { //If the tutorialProgress integar is 0 (AKA He's new)
                                        switch (c.NpcDialogue) { //again, to ensure dialogue is here...
                                        case 0: //start by saying this...
                                                sendNpcChat("Welcome "+ c.playerName +"! Pleasure to meet you!");
                                                n(); //Go to next case
                                        break;
                                        case 1: //say this after case 0 due to n()
                                                sendPlayerChat("Hi. Who are you?");
                                                n(); //Go to next case
                                        break;
                                        case 2: // say this after case 1 due to n()
                                                sendNpcChat("Who am I you ask? Call me the Runescape Guide.");
                                                n(); //Go to next case
                                        break;
                                        case 3: // say this after case 2 due to n()
                                                sendNpcChat2("I would like to welcome you to Absillion. Here you will learn", "vast skills and go on glorious journeys...");
                                                n(); //Go to next case
                                        break;
                                        case 4: //say this after case 3 due to n()
                                                sendNpcChat("I hope you have a wonderful time in Absillion!");
                                                n(); //Go to next case
                                        break;
                                        case 5: //say this after case 4 due to n()
                                                sendPlayerChat("Thanks.");
                                                n(); //Go to next case
                                        break;
                                        case 6: //say this after case 5 due to n()
                                                sendNpcChat3("No problem sonny. Here, let me transport", "you to a place to start your adventure.", "That okay with ya?");
                                                n(); //Go to next case
                                        case 7: //say this after case 6 due to n()
                                                sendPlayerChat("Sure.");
                                                n(); //Go to next case
                                        break;
                                        case 8: //say this after case 7 due to n()
                                                sendNpcChat("Okay sonny.");
                                                n(); //Go to next case
                                        break;
                                        case 9: //say this after case 8 due to n()
                                                sendNpcChat("Ooga balooga boom!");
                                                c.posX = 3748;
                                                c.posY = 7262;
                                                n(); //Go to next case
                                        break;
                                        case 10: //say this after case 9 due to n()
                                                sendStatement("You arrive at the place to start your journey.");
                                                n(); //Go to next case
                                        break;
                                        case 11: //say this after case 10 due to n()
                                                sendNpcChat("Good luck sonny!");
                                                n(); //Go to next case
                                        break;
                                        case 12: //do this after case 11 due to n()
                                                f(); //finish the chit chat
                                        break;
                                } //Close the dialogue chit chat for npc RuneScape Guide if tutorialProgress is 0
                                } else { //But if tutorialProgress is not 0
                                        switch (c.NpcDialogue) {
                                                case 0: //Start by saying this
                                                        sendStatement("The RuneScape Guide is currently busy.");
                                                        n(); //Continue to next dialogue
                                                break;
                                                case 1: //Do this after case 0 due to n()
                                                        f(); //Finish the dialogue
                                                break;
                                        } //Close switch
                                } //Close else
                                break; //Allows us to use a different npc now                  
                        } //Close the dialogue choices for the npcId's.
                } //Close Conversations().

                //The talking stuff here
        }
