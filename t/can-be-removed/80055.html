public final class Universe {
    private static final ArrayList<Species> lifeforms;

    private Universe() {}

    Species[] getLifeforms() {
        Species[] a = new Species[lifeforms.size()];
        a = lifeforms.toArray(a);
        return a;
    }

    public static void addLifeform(Species s) {
        lifeforms.add(s);
    }

    public static void deleteLifeform(Species s) {
        lifeforms.remove(s);
    }
 
    public static void purge(Class<?> type) {
        for(Species s: getLifeforms())
            if(s.getClass() == type)
                deleteLifeform(s);
    }

    static {
        species = new ArrayList<Species>();
    }
}
