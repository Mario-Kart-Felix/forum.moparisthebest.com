WanneShop = 2;

} else if (NPCID == 553) { PutNPCCoords = true; WanneShop = 2;

shop = 2	Aubury's_Rune_Store				2	2	554	10000	555	10000	556	10000	557	10000	559	10000	558	10000	562	10000	560	10000	553	10000	565	10000
