Well i don’t know if this was created already but this is my first script.
it measures the width and height of something.

{.Script Info:
# ScriptName  = Find Width&Height
# Author      = Robbee
# Description = Useful and easy to use
# Version     = .
# Date        = N/A
# Comments    = its pretty cool
# Directions  = Drag the croshair over something you want to mesure
and hit run and it should tell you at the bottom Width and Height
/Script Info}
program Width&height;

var w,h: integer;

begin
  GetClientDimensions(w,h);
  Writeln('Width: '+IntToStr(w)+' Height: '+IntToStr(h));
end.
