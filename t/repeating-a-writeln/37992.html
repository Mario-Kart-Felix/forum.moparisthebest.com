Here is a quick script I made to send every key from 00000 to 99,999 which is 5 characters, and then if that doesn’t work, then it will try, 000000 to 999,999, if that doesn’t work, it will keep trying an extra digit until it cracks the password.

{ Program By Ron.

  Instructions:
    1. Bring up the rar password window and then drag the target on it.
    2. Run the script.
    3. If the script goes too quickly, then it will not be able to check
    every character so you may want to increase the WaitT in the const
    from 0 to something like... 20 or 40 or something.
    4. If all else fails, then just google rar password cracker or
    something and you'll probably find one. Good Luck.
}

program ProgramByRon;

const
  WaitT     = 0;  // How long each wait is before it tries again?
  MinDigits = 1;  // Check characters with this many digits minimum.
  MaxDigits = 25; // Check characters with this many digits maximum.
  
var
  a,i,Time : integer;

// ZeroDigits() will add 0's for digits not filled up by the Number.
// Made by Ron :)
function ZeroDigits(Number, Digits : Integer) : String;
var
  Str : String;
begin
  Str := IntToStr(Number);
  while (Length(Str) < Digits) do
  begin
    insert('0',Str,1);
  end;
  Result := Str;
end;

procedure CheckNumbers(Digits : Integer);
var
  Num : String;
begin
  for i := 1 to Digits do
  begin
    Num := Num + '9';
  end;
  for i := 0 to StrToInt(Num) do
  begin
    SendKeys(IntToStr(i)+Chr(13))
    Wait(WaitT);
    WriteLn('Tried: '+ZeroDigits(i, Digits));
  end;
end;

begin
  ClearDebug;
  a := MinDigits;
  Time := GetSystemTime;
  repeat
    WriteLn('Checking '+IntToStr(a)+' many digit(s) now.');
    CheckNumbers(a);
    a := a + 1;
  until(IsFKeyDown(12))or(a > MaxDigits)
  WriteLn('Script finished in: '+IntToStr((GetSystemTime-Time)/1000)+' seconds.');
end.
