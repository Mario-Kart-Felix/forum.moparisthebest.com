private static void downloadJar() {
	long time = 0;
	try {
		time = System.nanoTime();
		URL RS = new URL("jar:http://world37.runescape.com/runescape.jar!/");
		RS.openConnection();
		File thejar = new File("./runescape.jar");
		if (thejar.exists()) {
			thejar.delete();
		}
		InputStream in = new URL("http://world37.runescape.com/runescape.jar").openStream();
		
		FileOutputStream out = new FileOutputStream(thejar);
		
		byte[] buf = new byte[700];
		int i;
		while ((i = in.read(buf)) != -1) out.write(buf, 0, i);
		
		in.close();
		out.close();
	} catch (Exception e) {
		e.printStackTrace();
	}
	
	System.out.println("Jar has been downloaded");
	System.out.println("Download time : " + (((System.nanoTime() - time) / 1000000) / 1000)
	+ " seconds.");
}

private static void get() {
	try {
		File thejar = new File("./runescape.jar");
		JarFile jar = new JarFile(thejar);

		Enumeration<?> en = jar.entries();

		while (en.hasMoreElements()) {

			JarEntry entry = (JarEntry) en.nextElement();

			if (entry.getName().endsWith(".class")) {

				ClassParser p = new ClassParser(jar.getInputStream(entry),
						entry.getName());
				JavaClass jc = p.parse();

				ClassGen cg = new ClassGen(jc);

				String className = cg.getClassName();

				for (Field f : cg.getFields()) {

					if (f.isStatic())
						continue;

					if (f.getType() != null) {

						String fieldname = f.getName();

						System.out.println("" + f.getType() + " " + className + "." + fieldname);
					}
				}
			}
		}
	} catch (Exception e) {
		e.printStackTrace();
	}
