This my first script, an AutoDancer. Pretty simple, clicks the dance emote every 5 or so seconds.
Make sure you can see the dance emote before starting.

{-AutoDancer by SamTheSam-}
{-No setup required, make sure you can see the dance emote before starting-}

program Dance;
var
  Dance, x, y: Integer;

procedure Load;
begin
  Dance := BitmapFromString(32, 13, 'z78DA8D96510E83400844AFD468AD' +
    'E9273FBDFF916A16571F0C349584D815D96118D6EEEB6BDD967D7' +
    '8B727CD0E736FEBFB365F418C31D26DE43A63FC7EBB6D718F2CC1' +
    'F0D426AE9161F81D981BB401F97C3B613E7761E680DC636EE413C' +
    '7F52B65930C1E29394345057B25031183AE17FBE6EE30F37C4B3A' +
    '286C0496C86D83C424BE5AB999AC1066664256ED66E6F98A906C7' +
    '97D6270FF39AEC7835E99C9BA4A1D6F5491D16F1587CA864C9F46' +
    'B276C7CCEBAC224F16D06E3A9BA6531CBCE850F55FCC74AE28304' +
    'FFEFDA23ED18BC05BB05A7B8BA8A863BB9D02ED9A2A963881BFE8' +
    '0BF7A5D2741755A34EBA468A024D9F769E38D923C50F3C46A5B18' +
    'A4E8D44487D329219BC8A2E833FF56F81F2EF2675857AE919F3C7' +
    '4ADC116A51E6899C195A9E316BDDBECA4C8B56F45928339D4BD57' +
    '9C277F50CD47A1527A7B5E3BF51A3690CEBE229F11B55F8070215' +
    'B13B5DD5F9DCA9E239FBAABA305F409AF51FBF9B059E81FC0B0ED' +
    '91D3A');
end;

procedure ClickDance;
begin
  FindBitmapToleranceIn(Dance, x, y, 1, 1, 1024, 768, 10);
  MoveMouseSmooth(x, y);
  Wait(1000);
  ClearDebug;
  Status('Dancing...');
  Writeln('Starting Dancing....');
  repeat
    ClickMouse(x, y, True);
    Wait(5000 +Random(500));
  until (false);
end;

begin
  Load;
  ClickDance;
end.
