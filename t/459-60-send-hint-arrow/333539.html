opCode: 60
Payload Structure: byte,short,short,byte
Description:Sets the hint arrow for a player entity, mob entity, or absolute tile.

First byte is the arrow type.

-1 = turn off arrow
1 - arrow on NPC
2 - 6 = Arrow on location
2 - on tile
3 - slightly east
4 - slightly west
5 -slightly south
6 - slightly north
10 = arrow on Player

[code]public enum HintArrowType {
OFF((byte)-1),
NPC((byte)1),
TILE_CENTER((byte)2),
TILE_NORTH((byte)6),
TILE_SOUTH((byte)5),
TILE_WEST((byte)4),
TILE_EAST((byte)3),
PLAYER((byte)10);
private byte type;

private HintArrowType(byte i){
    type = i;
}
public byte getType(){
    return type;
}

}

public ActionSender sendHintArrow(HintArrowType type, Object o){
PacketBuilder pb = new PacketBuilder(60);
byte i = type.getType();
pb.put(i);
if(i >= 2 && i <= 6){
Location l = (Location)o;
pb.putShort(l.getX());
pb.putShort(l.getY());
pb.put((byte)l.getZ());
}
else{
pb.putShort(((Entity)o).getIndex());
pb.putShort(0);
