class Fibonacci {

	private:
		std::map<unsigned int, unsigned int> cache;
	
	public:
		unsigned int evaluate(unsigned int n){
			if(n < 2)
				return n;
			if(cache.count(n) == 0)
				cache.insert(std::pair<unsigned int, unsigned int>(n, evaluate(n - 1) + evaluate(n - 2)));
			return cache[n];

public class Fibonacci {	
	private static final double SQRT5 = Math.sqrt(5);
	private static final double A = (1 + SQRT5) / 2;
	private static final double B = (1 - SQRT5) / 2;
	
	public static int fibonacci(int n) {
		return (int) Math.rint((Math.pow(A, n) - Math.pow(B, n)) / SQRT5);
	}
}
