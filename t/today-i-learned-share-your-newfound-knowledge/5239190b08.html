Here is how I refactored that method:

[code=java] // Worth noting that here, widgetId and LoadWidget could be called interfaceId and loadInterface, but because of the way the 2d array is stored we can’t call the class Interface
static final boolean attemptLoadWidget(int widgetId) {
if (widgetsLoaded[widgetId]) {
return true;
}
if (!widgetFileSystem.hasFileBuffer(widgetId)) {
return false;
}
int childCount = widgetFileSystem.getChildCount(widgetId);
if (0 == childCount) {
widgetsLoaded[widgetId] = true;
return true;
}
if (null == loadedWidgets[widgetId]) {
loadedWidgets[widgetId] = new Widget[childCount];
}

    for (int childId = 0; childId < childCount; ++childId) {
        if (null == loadedWidgets[widgetId][childId]) {
            byte[] data = widgetFileSystem.getFile(widgetId, childId);
            if (data != null) {
                Widget child = loadedWidgets[widgetId][childId] = new Widget();
                child.widgetId = childId + (widgetId << 16);
                if (-1 == data[0]) {
                    child.decodeNewFormat(new Stream(data));
                } else {
                    child.decodeOldFormat(new Stream(data));
                }
            }
        }
    }
    widgetsLoaded[widgetId] = true;
    return true;
}[/code]
