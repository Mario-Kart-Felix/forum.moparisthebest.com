It is a how-far-would-an-object-be-to-look-like-a-certain-planet program. =/
ie., how far would you have to put a 12-foot wide sphere for it to be the same size as a planet in the sky.

import java.util.Scanner;
public class DistanceC {
  //Venus
    public static double venusS = 476403840.0;
    public static double venusDi = 1647360000000.0;
  // Mercury
    public static double mercuryS = 192107520.0;
    public static double mercuryDi = 3801600000000.0;
  //Mars
    public static double marsS = 265731840.0;
    public static double marsDi = 2217600000000.0;
  //Jupiter
    public static double jupiterS = 5622312960.0;
    public static double jupiterDi = 23443200000000.0;
  //Saturn
    public static double saturnS = 4750606080.0;
    public static double saturnDi = 47139840000000.0;
  //Uranus
    public static double uranusS = 2039748480.0;
    public static double uranusDi = 101819520000000.0;
  //Neptune
    public static double neptuneS = 1949904000.0;
    public static double neptuneDi = 169804800000000.0;
  //Pluto
    public static double plutoS = 90161280.0;
    public static double plutoDi = 190080000000000.0;
  //Sun
    public static double sunS = 54802471680.0;
    public static double sunDi = 5892480000000.0;

    public static double coinS;
    public static double ans;
    public static int planet;
    public static void main(String[] args) {
        System.out.print("Planet Numbers:\r\n 1: Venus \r\n 2: Mercury \r\n 3: Mars \r\n 4: Jupiter \r\n 5: Saturn \r\n 6: Uranus \r\n 7: Neptune \r\n 8: Pluto \r\n 9: Sun \r\n");
        Scanner Scan = new Scanner(System.in);
        System.out.print("Size of object: ");
        coinS = Scan.nextDouble();
        System.out.print("Which planet? ");
        planet = Scan.nextInt();
        switch (planet) {
            case 1: ans = venusDi / venusS * coinS / 12; break;
            case 2: ans = mercuryDi / mercuryS * coinS / 12; break;
            case 3: ans = marsDi / marsS * coinS / 12; break;
            case 4: ans = jupiterDi / jupiterS * coinS / 12; break;
            case 5: ans = saturnDi / saturnS * coinS / 12; break;
            case 6: ans = uranusDi / uranusS * coinS / 12; break;
            case 7: ans = neptuneDi / neptuneS * coinS / 12; break;
            case 8: ans = plutoDi / plutoS * coinS / 12; break;
            case 9: ans = sunDi / sunS * coinS / 12; break;
            default: System.out.println("Invalid Planet Number"); break;
        }
        if(ans < 5280) {
            System.out.println("Hold the object " + ans + "ft away from you");
            } else {
            ans = ans / 5280;
            System.out.println("Hold the object " + ans + "miles away from you");
        }
    }
}

For "Size of object: " you must put a number of INCHES. (eg. Size of object: 6).
For "Which planet? " you must put one of the planet (or sun) numbers printâ€™ed on startup (eg. Which planet? 3)

Have fun ;D

EDIT: Made the result a little more readable be converting the ft to miles if the ft > 5280.
