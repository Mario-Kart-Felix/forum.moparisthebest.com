program New;

   x,y,curangle:integer;

function abs(i : extended) : extended;
begin
     i:= sqrt(i*i);
     result := i;
end;
   
function fact(i : integer) : integer;
var
   holdi,res : integer;
begin
     holdi := i;
     res := 1;
     for i := 1 to holdi do
     begin
          res := i * res
     end;
     result := res;
end;
   
function arctan(e : extended) : extended;
var
   count : integer;
   res : extended;
   inv : boolean;
begin
     if(abs(e) > 1)then
     begin
          e := pow(e,-1);
          inv := true;
     end else
     begin
          inv := false;
     end;
     for count := 0 to 6 do
     begin
          res := res + ( ( pow(2,2 * count) * pow(fact(count),2) ) / fact(2 * count + 1) ) *  ( ( pow(e,2*count+1) ) / ( pow(1 + pow(e,2),count + 1) ) );
     end;
     if(inv)then
     begin
          res := 1.5707963267 - res;
     end;
     result := res;
end;

procedure findintercept(targx,targy,radius:integer);
var
   m:extended;
begin
     if(i = 0) or (j = 0)then
     begin
          if(i = 0)then
          begin
               y:= y + radius;
          end else
          begin
               x:= x + radius;
          end;
     end else
     begin
          m:= i / j;
          x:= x + round(sqrt((radius * radius)/(m*m+1)));
          y:= y + round(sqrt(m*m)*sqrt((radius * radius)/(m*m+1)));
     end;
end;

procedure adjangle(targx,targy,rate:integer);
var
   targangle,degrees:integer;
