function Cutting: Boolean;
var
Nx, Ny, N: Integer;
Colours: TIntegerArray;
TreeName: string;
begin
case LowerCase(TreeName) of
‘tree’ : Colours := [4156042, 2971727, 2439992];
‘oak’ : Colours := [3561845, 486492, 1064758];
‘willow’: Colours := [4150100, 2974794, 3297881];
end;
for N := 0 to 5 do
begin
if(‘tree’=TreeToChop)or(‘oak’=TreeToChop)or(‘willow’=TreeToChop)then
begin
QuickRandoms;
FindNormalRandoms;
if FindObjCustom(Nx, Ny, [TreeToChop, 'Chop down ’ + TreeToChop, Copy(TreeToChop, 2, Length(TreeToChop))], Colours, N)then
begin
if Random(5) = 4 then
begin
Mouse(Nx, Ny, 1, 1, False);
Wait(250 + Random(300));
ChooseOption(‘Chop down’);
end else
Mouse(Nx, Ny, 1, 1, True);
Wait(290 + Random(600));
Flag;
QuickRandoms;
FindNormalRandoms;
Result := True;
Exit;
end;
end;
end;
end;

procedure Banking;

begin

DropLogs;
end else
case LowerCase(BankColours) of
‘BankColour1’ : Colours := [4156042, 2971727, 2439992];
‘BankColour2’ : Colours := [3561845, 486492, 1064758];
‘BankColour3’ : Colours := [4150100, 2974794, 3297881];
end;
for N := 0 to 5 do
begin
QuickRandoms;
FindNormalRandoms;
if FindObjCustom(Ax, Ay, [BankColours, 'Use ’ + ‘bank.’, Copy(‘Use bank.’, 2, Length(Ay))], Colours, N)then
begin
Mouse(x,y,2,2,false);
ChooseOption(‘Use quickly.’);
end else
Mouse(x,y,2,2,true);
Wait(210+Random(70));
Mouse(720,400,2,2,false);
ChooseOption(‘Bank all.’);
Mouse(x,y,2,2,true);
Wait(300+Random(80));
CloseBank;

Wait(300+Random(100));

end;
end;
end;
