// Random Pass Generator by Aps

Program RandomPass;

Var a : Integer;
    StartTime : Integer;

Begin
  StartTime := GetTickCount;
  If(IsFKeydown(9))Then
  Begin
    Repeat
      a := 1 + Random(9999999);
      a := a * 1;
      a := a + 1 + Random(999999);
      Wait(1 + Random(100));
      WriteLn(IntToStr(a));
    Until(GetTickCount - StartTime > 5000);
    WriteLn('DING!');
  End;
End.
