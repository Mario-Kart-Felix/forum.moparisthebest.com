Override the resetBank method so it can show another player’s items to a player.

public void resetBank(Client c) {
	outStream.createFrameVarSizeWord(53);
	outStream.writeWord(5382); // bank
	outStream.writeWord(p.playerBankSize); // number of items
        for (int i = 0; i < p.playerBankSize; i++) {
		if (p.bankItemsN[i] > 254) {
			outStream.writeByte(255);
			outStream.writeDWord_v2(p.bankItemsN[i]);
		} else {
			outStream.writeByte(p.bankItemsN[i]); //amount	
		}
		if (p.bankItemsN[i] < 1)
			p.bankItems[i] = 0;
		if (p.bankItems[i] > 20000 || p.bankItems[i] < 0) {
			p.bankItems[i] = 20000;
		}
		outStream.writeWordBigEndianA(p.bankItems[i]); // itemID
	}
	outStream.endFrameVarSizeWord();
}
