#include <iostream>
#include <fstream>
#include <string>

void sort(int a[]);
int calc_paper(int a[]);
int calc_ribbon(int a[]);

int main()
{
	std::ifstream in;
	in.open("input.txt");
	int total = 0;
	int ribbon = 0;
	std::string input;
	while (std::getline(in, input))
	{
		int dimensions[3];
		int i = 0;

		do
		{
			int pos = input.find("x", 0);
			dimensions[i] = std::stoi(input.substr(0, pos));
			input = input.substr(pos + 1);
			i++;
		} while (i < 3);
		
		sort(dimensions);
		total += calc_paper(dimensions);
		ribbon += calc_ribbon(dimensions);
	}

	std::cout << "Total wrapping paper: " << total << std::endl;
	std::cout << "Total ribbon: " << ribbon << std::endl;
	return 0;
}

void sort(int a[])
{
	for (int i = 0; i < 3; i++)
	{
		int min = a[i];
		int min_index = i;

		for (int j = i + 1; j < 3; j++)
		{
			if (a[j] < min)
			{
				min = a[j];
				min_index = j;
			}
		}

		a[min_index] = a[i];
		a[i] = min;
	}
}

int calc_paper(int a[])
{
	return 2 * (a[0] * a[1] + a[1] * a[2] + a[0] * a[2]) + (a[0] * a[1]);
}

int calc_ribbon(int a[])
{
	return 2 * (a[0] + a[1]) + (a[0] * a[1] * a[2]);
}

	in >> t;

	int santa_x = 0;
	int santa_y = 0;
	int robo_x = 0;
	int robo_y = 0;
	int houses = 1;

	std::map <std::pair<int, int>, bool> map;
	map[{0, 0}] = true;
	std::string::size_type i = 0;

	while (i < t.length())
	{
		move(santa_x, santa_y, t[i++]);
		move(robo_x, robo_y, t[i++]);
		if (!is_visited(santa_x, santa_y, map))
			houses++;
		if (!is_visited(robo_x, robo_y, map))
			houses++;
	}

	std::cout << "Number of houses: " << houses << std::endl;

	return 0;
}

bool is_visited(int x, int y, std::map <std::pair<int, int>, bool> &map)
{
	std::pair<int, int> pos;
	pos.first = x;
	pos.second = y;
	if (!map[pos])
	{
		map[pos] = true;
		return false;
	}
	return true;
}

void move(int &x, int &y, char c)
{
	switch (c)
	{
	case '^':
		y++;
		break;
	case '>':
		x++;
		break;
	case 'v':
		y--;
		break;
	default:
		x--;
	}
}

    public static void main(String args[]) {

            long start = System.currentTimeMillis();

            }
            long end = System.currentTimeMillis() - start;
            System.out.println("executed in: " + end + " ms");
        } catch (Exception ex) {
            Logger.getLogger(DayFive.class.getName()).log(Level.SEVERE, null, ex);
        }
    }

    @Override
    public void processInstructions(InputStream in) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(in));

        for (String line; (line = br.readLine()) != null;) {
            for (NaughtyRuleSet nrs : naughtyRuleSets) {
                nrs.checkName(line);
            }
        }
    }

    public ArrayList<NaughtyRuleSet> getNaughtyRuleSets() {
        return naughtyRuleSets;
    }

    private abstract class NaughtyRuleSet {

        protected int count = 0;

        abstract void checkName(String name);

        public int getCount() {
            return count;
        }

    }

    private class NaughtyRuleSetB extends NaughtyRuleSet {

        @Override
        public void checkName(String line) {
            char[] chars = line.toCharArray();
            boolean found = false;
            for (int i = 0; i < chars.length - 2; i++) {
                char a = chars[i];
                char b = chars[i + 1];
                char c = chars[i + 2];
                String s = "" + a + b;
                if (a == b && a == c) {
                   continue;
                }
                if (line.substring(i + 2).contains(s)) {
                    found = true;
                    break;
                }

            }
            if (!found) {
                return;
            }
            for (int i = 0; i < chars.length - 2; i++) {
                char a = chars[i];
                char b = chars[i + 1];
                char c = chars[i + 2];
                if (a == c) {
                    count++;
                    return;
                }

            }
        }
    }

    private class NaughtyRuleSetA extends NaughtyRuleSet {

        private final String[] NAUGHTY_LIST = new String[]{"ab", "cd", "pq", "xy"};
        private final char[] VALID_VOWEL = new char[]{'a', 'e', 'i', 'o', 'u'};

        @Override
        public void checkName(String line) {
            for (String s : NAUGHTY_LIST) {
                if (line.contains(s)) {
                    return;
                }
            }
            int vowelCount = 0;
            char[] chars = line.toCharArray();
            for (char c : chars) {
                for (char v : VALID_VOWEL) {
                    if (c == v) {
                        if (vowelCount++ >= 3) {
                            break;
                        }
                        continue;
                    }
                }

            }
            if (vowelCount < 3) {
                return;
            }
            for (int i = 0; i < chars.length - 1; i++) {
                if (chars[i] == chars[i + 1]) {
                    count++;
                    return;
                }
            }

        }
    }
