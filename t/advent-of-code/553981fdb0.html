public class DaySix extends Day {

    private final Light[][] lights;
    private final int size;

    public DaySix(int size) {
        this.size = size;
        lights = new Light[size][size];
        for (int x = 0; x < size; x++) {
            for (int y = 0; y < size; y++) {
                lights[x][y] = new Light();
            }
        }
    }

    public static void main(String args[]) {
        DaySix day = new DaySix(1000);

        try {
            day.processInstructions(new FileInputStream("day_6_puzzle.txt"));
            System.out.println("Lit light count: " + day.getLitLightCount());
            System.out.println("Brightness Level: " + day.getBrightnessLevel());

        } catch (Exception ex) {
            Logger.getLogger(DaySix.class.getName()).log(Level.SEVERE, null, ex);
        }
    }

    @Override
    public void processInstructions(InputStream in) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(in));
        br.lines().forEach(this::processInstruction);
    }

    private void processInstruction(String line) {

        String[] data = line.split(" ");
        boolean isToggle = false;
        boolean on = false;
        if (data[0].equals("turn")) {
            if (data[1].equals("on")) {
                on = true;
            }
        } else {
            isToggle = true;
        }
        String[] startPosition = new String[2];
        String[] endPosition = new String[2];

        startPosition = data[isToggle ? 1 : 2].split(",");
        endPosition = data[isToggle ? 3 : 4].split(",");
        changeLights(isToggle, on, Integer.parseInt(startPosition[0]), Integer.parseInt(startPosition[1]), Integer.parseInt(endPosition[0]), Integer.parseInt(endPosition[1]));
    }

    private void changeLights(boolean toggle, boolean on, int sx, int sy, int ex, int ey) {

        for (int x = sx; x < ex + 1; x++) {
            for (int y = sy; y < ey + 1; y++) {
                lights[x][y].setOn(toggle ? !lights[x][y].isOn() : on);

                lights[x][y].setBrightness(toggle ? lights[x][y].getBrightness() + 2 : on ? lights[x][y].getBrightness() + 1 : lights[x][y].getBrightness() - 1);
            }
        }
    }

    public int getLitLightCount() {
        int i = 0;
        for (int x = 0; x < this.size; x++) {
            for (int y = 0; y < this.size; y++) {
                if (lights[x][y].isOn()) {
                    i++;
                }
            }
        }
        return i;
    }

    public int getBrightnessLevel() {
        int i = 0;
        for (int x = 0; x < this.size; x++) {
            for (int y = 0; y < this.size; y++) {
                i += lights[x][y].getBrightness();
            }
        }
        return i;
    }

    private class Light {

        private boolean on;
        private int brightness;

        private Light() {
            this.on = false;
            this.brightness = 0;
        }

        public boolean isOn() {
            return on;
        }

        public int getBrightness() {
            return brightness;
        }

        public void setOn(boolean on) {
            this.on = on;
        }

        public void setBrightness(int brightness) {

            this.brightness = brightness < 0 ? 0 : brightness;
        }

    }
}
