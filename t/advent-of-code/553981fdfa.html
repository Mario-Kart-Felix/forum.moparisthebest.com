public class DayTwentyTwo {

    private final Entity player;
    private final Entity enemy;

    public DayTwentyTwo() {
        this.player = new Player(50, 500, 0, 0);
        this.enemy = new Boss(71, 0, 10, 0); //puzzle input
    }

    public static void main(String args[]) {
        DayTwentyTwo day = new DayTwentyTwo();
        day.simulateBattle(false);

    }

    public void simulateBattle(boolean hardMode) {
        Battle b = new Battle(new Player(player.getHealth(), player.getMana(), player.getDamage(), player.getArmour()), new Boss(enemy.getHealth(), enemy.getMana(), enemy.getDamage(), enemy.getArmour()));
        while (b.getPlayer().health > 0 && enemy.getHealth() > 0) {
            for (Entity e : b.getEntities()) {
                if (hardMode) {
                    b.getPlayer().setHealth(b.getPlayer().getHealth() - 1);
                }
                b.getActiveSpells().forEach(s -> {
                    s.onTick(b);
                });
                b.getActiveSpells().removeIf(s -> s.getTick() == 0);

                e.attack(b);

            }
            System.out.println("Round: " + b.getPlayer().getHealth() + " " + b.getEnemy().getHealth() + " Mana used: " + player.getManaUsage());
        }
    }

    public class Battle {

        private final Entity player;
        private final Entity enemy;
        private final ArrayList<Spell> activeSpells;

        public Battle(Entity player, Entity enemy) {
            this.player = player;
            this.enemy = enemy;
            this.activeSpells = new ArrayList<Spell>();
        }

        public Entity getPlayer() {
            return player;
        }

        public Entity getEnemy() {
            return enemy;
        }

        public ArrayList<Spell> getActiveSpells() {
            return activeSpells;
        }

        private Entity[] getEntities() {
            return new Entity[]{player, enemy};
        }

    }

    public class Player extends Entity {

        private int manaUsage;

        public Player(int health, int mana, int damage, int armour) {
            super(health, mana, damage, armour);
            this.manaUsage = 0;
        }

        @Override
        public void attack(Battle b) {
            Spell s = null;
            //change attack methods to find best usage
            if (find(b.getActiveSpells(), Shield.class) == null) {
                s = new Shield();
            } else if (find(b.getActiveSpells(), Poison.class) == null) {
                s = new Poison();
            }
            if (s == null) {
                return;
            }
            s.cast(b);
            this.manaUsage += s.getManaCost();
        }

    }

    public static <T> T find(Collection<?> arrayList, Class<T> clazz) {
        for (Object o : arrayList) {
            if (o != null && o.getClass() == clazz) {
                return clazz.cast(o);
            }
        }

        return null;
    }

    public class Boss extends Entity {

        public Boss(int health, int mana, int damage, int armour) {
            super(health, mana, damage, armour);
        }

        @Override
        public void attack(Battle b) {
            b.getPlayer().setHealth(b.getPlayer().getHealth() - (this.getDamage() - b.getPlayer().getArmour()));
        }

    }

    public abstract class Entity {

        private int health;
        private int mana;
        private int damage;
        private int armour;

        public Entity(int health, int mana, int damage, int armour) {
            this.health = health;
            this.mana = mana;
            this.damage = damage;
            this.armour = armour;
        }

        public abstract void attack(Battle b);

        public int getHealth() {
            return health;
        }

        public void setHealth(int health) {
            this.health = health;
        }

        public int getMana() {
            return mana;
        }

        public void setMana(int mana) {
            this.mana = mana;
        }

        public int getDamage() {
            return damage;
        }

        public void setDamage(int damage) {
            this.damage = damage;
        }

        public int getArmour() {
            return armour;
        }

        public void setArmour(int armour) {
            this.armour = armour;
        }

    }

    public abstract class Spell {

        private int tick;

        public Spell(int tick) {
            this.tick = tick;
        }

        public void cast(Battle b) {
            b.getPlayer().setMana(b.getPlayer().getMana() - this.getManaCost());
            if (this.tick > 0) {
                b.getActiveSpells().add(this);
            }

        }

        public void onTick(Battle b) {
            this.tick--;
        }

        public int getTick() {
            return tick;
        }

        public abstract int getManaCost();
    }

    public class MagicMissle extends Spell {

        public MagicMissle() {
            super(0);
        }

        @Override
        public int getManaCost() {
            return 53;
        }

        @Override
        public void cast(Battle b) {
            b.getEnemy().setHealth(b.getEnemy().getHealth() - 4);
            super.cast(b);
        }

    }

    public class Drain extends Spell {

        public Drain() {
            super(0);
        }

        @Override
        public int getManaCost() {
            return 73;
        }

        @Override
        public void cast(Battle b) {
            b.getEnemy().setHealth(b.getEnemy().getHealth() - 2);
            b.getPlayer().setHealth(b.getPlayer().getHealth() + 2);
            super.cast(b);
        }
    }

    public class Shield extends Spell {

        public Shield() {
            super(6);
        }

        @Override
        public int getManaCost() {
            return 113;
        }

        @Override
        public void onTick(Battle b) {
            if (this.getTick() == 0) {
                b.getPlayer().setArmour(0);
            } else {
                b.getPlayer().setArmour(7);
            }
            super.onTick(b);
        }

    }

    public class Poison extends Spell {

        public Poison() {
            super(6);
        }

        @Override
        public int getManaCost() {
            return 173;
        }

        @Override
        public void onTick(Battle b) {
            b.getEnemy().setHealth(b.getEnemy().getHealth() - 3);
            super.onTick(b);
        }
    }

    public class Recharge extends Spell {

        public Recharge() {
            super(5);
        }

        @Override
        public int getManaCost() {
            return 229;
        }

        @Override
        public void onTick(Battle b) {
            b.getPlayer().setMana(b.getPlayer().getMana() + 101);
            super.onTick(b);
        }
    }
}
