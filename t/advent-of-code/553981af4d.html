[spoiler=Day 9][code=scala]import scala.io.{BufferedSource, Source}

object Day9 extends App {

var input9 = Source.fromFile(""“E:\advent\input9"”")
var map = Map(String,String),Int
var cities = SetString
def addCities(a:String,b:String,distance:Int) = {
map += (a,b) -> distance
map += (b,a) -> distance
}

def processInput(input:BufferedSource) = {
input.getLines().foreach{
_ split " " match {
case Array(a,,b,,distance) => addCities(a,b,distance.toInt); cities += a; cities += b
case _ =>
}
}
}

def distance(a:String,b:String) = map(a,b)
def measurePath(path:List[String]) = {var sum = 0; path.reduce{(a,b) => sum += distance(a,b); b}; sum}

processInput(input9)
val part1 = cities.toList.permutations map measurePath min
val part2 = cities.toList.permutations map measurePath max

println(s"Shortest route is $part1 miles long\nLongest route is $part2 miles long")
}

object Day10 extends App{
def processInput(input:String) = {
var charInRow = 0
var currentChar:Char = input.charAt(0)
var output = new StringBuilder()
input.foreach{ c =>
if(c != currentChar){
output.append(s"${charInRow}$currentChar")
charInRow = 0
currentChar = c
}
charInRow += 1
}
output.append(s"${charInRow}$currentChar")
output
}

def solve(input:String,n:Int) = {
var buffer:String = input
for (i <- 1 to n) {
buffer = processInput(buffer).toString()
}
buffer
}
var input10 = “1321131112"
println(s"Part 1: ${solve(input10,40).length}\nPart 2: ${solve(input10,50).length}”)
}

public class DayTen {

    public DayTen() {
    }
    public static void main(String args[]) {
        DayTen day = new DayTen();
        String input = "1113122113";
        for (int i = 0; i < 40; i++) {
      
            input = day.lookAndSay(input);

        }
        System.out.println("Part I: " + input.length());
        for (int i = 0; i < 10; i++) {
            input = day.lookAndSay(input);
        }
        System.out.println("Part II: " + input.length());
    }

    private String lookAndSay(String input) {
        char[] chars = input.toCharArray();
        char last = 0;
        int count = 1;
        String output = "";
        if (chars.length == 1) {
            output += 1;
            output += chars[0];
            return output;
        }
        for (int i = 0; i < chars.length; i++) {
            char c = chars[i];

            if (c == last) {
                count++;

            }
            if (i == 0) {
                last = c;

            }
            if (i == chars.length - 1 || c != chars[i + 1]) {
                output += count;
                output += c;
                count = 1;
            }
            last = c;
        }

        return output;
    }

}
