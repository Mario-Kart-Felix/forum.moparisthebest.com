package io.github.hadyn;

import java.util.Set;
import java.util.TreeSet;

/**
 * @author Hadyn Fitzgerald
 */
public class DrunkSanta {

  public static void main(String... args) {
    Set<Integer> codes = new TreeSet<>();
    int center = (1 << 16 - 1) >> 1;

    int currX = center;
    int currY = center;
    codes.add(mortonCode(currX, currY));

    String input = args[0];
    for(char c : input.toCharArray()) {
      switch (c) {
        case '^':
          currY += 1;
          break;
        case '<':
          currX -= 1;
          break;
        case '>':
          currX += 1;
          break;
        case 'v':
          currY -= 1;
          break;
        default:
          throw new IllegalArgumentException("Illegal character parsed.");
      }
      codes.add(mortonCode(currX, currY));
    }

    System.out.println(codes.size());
  }

  // Algorithm: https://fgiesen.wordpress.com/2009/12/13/decoding-morton-codes/

  private static int mortonCode(int x, int y) {
    return (part1by1(y) << 1) + part1by1(x);
  }

  private static int part1by1(int n) {
    n &= 0x0000ffff;
    n = (n ^ (n <<  8)) & 0x00ff00ff;
    n = (n ^ (n <<  4)) & 0x0f0f0f0f;
    n = (n ^ (n <<  2)) & 0x33333333;
    n = (n ^ (n <<  1)) & 0x55555555;
    return n;
  }
}

package io.github.hadyn;

import java.util.Set;
import java.util.TreeSet;

/**
 * @author Hadyn Fitzgerald
 */
public class DrunkSanta {

  public static void main(String... args) {
    Set<Integer> codes = new TreeSet<>();
    int center = (1 << 16) - 1 >> 1;

    int santaCurrX = center;
    int santaCurrY = center;
    int robotCurrX = center;
    int robotCurrY = center;
    codes.add(mortonCode(santaCurrX, santaCurrY));

    boolean flag = false;
    String input = args[0];
    for(char c : input.toCharArray()) {
      switch (c) {
        case '^':
          if(!flag) {
            santaCurrY += 1;
          } else {
            robotCurrY += 1;
          }
          break;
        case '<':
          if(!flag) {
            santaCurrX -= 1;
          } else {
            robotCurrX -= 1;
          }
          break;
        case '>':
          if(!flag) {
            santaCurrX += 1;
          } else {
            robotCurrX += 1;
          }
          break;
        case 'v':
          if(!flag) {
            santaCurrY -= 1;
          } else {
            robotCurrY -= 1;
          }
          break;
        default:
          throw new IllegalArgumentException("Illegal character parsed.");
      }
      codes.add(mortonCode(flag ? robotCurrX : santaCurrX, flag ? robotCurrY : santaCurrY));
      flag = !flag;
    }

    System.out.println(codes.size());
  }

  // Algorithm: https://fgiesen.wordpress.com/2009/12/13/decoding-morton-codes/

  private static int mortonCode(int x, int y) {

import com.adventofcode.modal.Santa;
import java.io.ByteArrayInputStream;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.util.Observable;
import java.util.Observer;
import java.util.logging.Level;
import java.util.logging.Logger;

/**
 *
 * @author Silabsoft
 */
public class DayOne implements Observer {

    private final Santa santa;
    private int processedInstructionCount;

    public DayOne() {
        this.santa = new Santa();
    }

    public static void main(String args[]) {
        try {
            DayOne day = new DayOne();
            day.processPuzzle(new FileInputStream(new File("day_1_puzzle.txt")));
            System.out.println("Santa is on floor: " + day.getSanta().getY());
        } catch (FileNotFoundException ex) {
            Logger.getLogger(DayOne.class.getName()).log(Level.SEVERE, null, ex);
        }
    }

    public void processPuzzle(InputStream input) {
        this.santa.addObserver(this);

        try {
            InputStreamReader isr = new InputStreamReader(input);
            int data = isr.read();
            processedInstructionCount++;
            while (data != -1) {
                processInstruction((char) data);

                processedInstructionCount++;
                data = isr.read();
            }
            isr.close();
        } catch (IOException ex) {
            ex.printStackTrace();
        }
    }

    public void processInstruction(char c) {
        switch (c) {

            case '(':

                santa.move(Santa.Direction.NORTH);
                break;
            case ')':
                santa.move(Santa.Direction.SOUTH);
                break;
            default:
                System.out.println("unknown puzzle input: " + c);
        }

    }

    public Santa getSanta() {
        return santa;
    }

    public int getProcessedInstructionCount() {
        return processedInstructionCount;
    }

    @Override
    public void update(Observable o, Object arg) {
        if (((int[]) arg)[1] == -1) {
            System.out.println("Santa first reached the basement on " + processedInstructionCount);
            this.santa.deleteObserver(this);
        }
    }
}



import java.util.Observable;

/**
 *
 * @author Silabsoft
 */
public class Santa extends Observable {

    public enum Direction {
        NORTH,
        SOUTH,
        EAST,
        WEST
    }
    private int x;
    private int y;

    public Santa() {
    }

    public void setPosition(int x, int y) {
        this.x = x;
        this.y = y;
        this.setChanged();
        this.notifyObservers(new int[]{x, y});
    }

    public int getX() {
        return x;
    }

    public int getY() {
        return y;
    }

    public void move(Direction d) {
        switch (d) {
            case NORTH:
                y++;
                break;
            case SOUTH:
                y--;
                break;
            case EAST:
                x++;
                break;
            case WEST:
                x--;
                break;
        }

        this.setChanged();
        this.notifyObservers(new int[]{x, y});
    }
}

import com.adventofcode.modal.Santa;
import com.adventofcode.modal.Santa.Direction;
import java.awt.Point;
import java.io.FileInputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.Observable;
import java.util.Observer;

/**
 *
 * @author Silabsoft
 */
public class DayThree extends Day implements Observer {

    private HashMap<Point, Integer> houses;
    private final ArrayList<Santa> santaList;

    public DayThree() {
        this.santaList = new ArrayList<Santa>();
        this.houses = new HashMap<Point, Integer>();
    }

    public static void main(String args[]) {
        DayThree day = new DayThree();
        try {
            day.santaList.add(new Santa());
            day.processInstructions(new FileInputStream("day_3_puzzle.txt"));
            System.out.println("Number of houses to receive at least 1 present: " + day.getHouses().size());
            day.santaList.clear();
            day.houses.clear();
            day.santaList.add(new Santa());
            day.santaList.add(new Santa()); //RoboSanta
            day.processInstructions(new FileInputStream("day_3_puzzle.txt"));
            System.out.println("Number of houses to receive at least 1 present from Santa and RoboSanta: " + day.getHouses().size());
        } catch (Exception ex) {
            ex.printStackTrace();
        }

    }

    @Override
    public void update(Observable o, Object arg) {
        int[] position = (int[]) arg;
        Point p = new Point(position[0], position[1]);
        int presents = houses.get(p) == null ? 1 : houses.get(p) + 1;
        houses.put(p, presents);
    }

    public HashMap<Point, Integer> getHouses() {
        return houses;
    }

    @Override
    public void processInstructions(InputStream in) throws Exception {
        for (Santa s : santaList) {
            s.addObserver(this);
            s.setPosition(0, 0);
        }
        try {
            InputStreamReader isr = new InputStreamReader(in);
            int data = isr.read();
            int santaListSize = santaList.size();
            int last = 0;
            while (data != -1) {

                processInstruction(santaList.get(last), data);
                if (santaListSize > 1) {
                    if (last++ >= santaListSize - 1) {
                        last = 0;
                    }

                }
                data = isr.read();
            }
            isr.close();
        } catch (IOException ex) {
            ex.printStackTrace();
        }
    }

    public void processInstruction(Santa s, int i) {

        Direction d = null;
        switch (i) {

            case '^':
                d = Direction.NORTH;
                break;
            case '<':
                d = Direction.WEST;
                break;
            case 'v':
                d = Direction.SOUTH;
                break;
            case '>':
                d = Direction.EAST;
                break;

            default:
                System.out.println("unknown puzzle input: " + i + " " + (char) i);
                return;
        }
        s.move(d);
    }
}
