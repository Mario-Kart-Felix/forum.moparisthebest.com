public static int calculateArea(int l, int w, int h) {
    
    int dim1 = l * w;
    int dim2 = w * h;
    int dim3 = h * l;
    
    int smallest = Integer.MAX_VALUE;
    
    int loop = 3;
    
    while (loop > 0) {
        
        switch (loop) {
            
            case 3:

    }

    public ArrayList<Reindeer> getReindeers() {

    public void processInstruction(String s) {
        String data[] = s.split(" ");
        reindeers.add(new Reindeer(data[0], Integer.parseInt(data[3]), Integer.parseInt(data[6]), Integer.parseInt(data[13])));

    public class Reindeer {

        private final String name;
        private final int speed;
        private final int flyDuration;
        private final int restDuration;
        private int points;

        public Reindeer(String name, int speed, int flyDuration, int restDuration) {
            this.name = name;
            this.speed = speed;
            this.flyDuration = flyDuration;
            this.restDuration = restDuration;

        }

        public int getDistanceTraveledByTotalTime(int travelDuration) {
            int travelDistance = 0;
            while (travelDuration > 0) {

                int secondsTraveled = travelDuration - flyDuration >= 0 ? flyDuration : travelDuration % flyDuration;
                travelDistance += (this.speed * secondsTraveled);
                travelDuration -= secondsTraveled;
                int restTime = travelDuration - restDuration >= 0 ? restDuration : travelDuration % restDuration;
                travelDuration -= restTime;
            }
            return travelDistance;
        }

        public String getName() {
            return name;
        }

        public int getSpeed() {
            return speed;
        }

        public void setPoints(int points) {
            this.points = points;
        }

        public int getPoints() {
            return points;
        }

        public int getFlyDuration() {
            return flyDuration;
        }

        public int getRestDuration() {
            return restDuration;
        }
