  • @author Arthur, Strucker, Mr Boogie…

  •     example with firemaking first, It's the easier skill so you'd better

public class Enums {

/**
 * well first of all we gotta do an enum, to do it we write the keyword
 * enum, after that, we type the name of the enum The conventions tell us to
 * do something different on the naming that on normal methods, the naming
 * on enums, is UpperCase that means that the first letter should be
 * capital.
 */
private enum LogData {
	
	/**
	 * then we gotta declare your constant/s, as I've said 
	 * a constant is final and static, and a constant is always typed
	 * in CAPITAL_LETTERS and to separate words we use an underscore
	 * 
	 * @param 1521
	 * 		it means the log id, the constructor will be defined below, as you keep reading
	 * 		you will understand what the 1521 means
	 */
	NORMAL(1521);
	
	/**
	 * now we need a variable to retrieve the value/s that will
	 * be stored in our constant, so I'll define it as logId
	 * 
	 * @param logId
	 * 		the id of the log that is going to be lit.
	 */
	private int logId;
	
	/**
	 * great, now we have to define wich parameters our constants have
	 * to take, for now it only takes a log, wich is our 1521, in our normal constant
	 * the 1521 number is the item index of the normal log
	 */
	private LogData(int logId) {
		
		/**
		 * the "this" keyword is used to access variables that are out
		 * of our method, and as we have the same name on the logId
		 * If i did logId = logId;, it would do nothing, so with the "this"
		 * keyword, i access the variable I've declared before the constructor
		 * 
		 * So what does it does?
		 * it makes the variable logId(the one we declared right before 
		 * making this constructor) the same value the constant has
		 * so now we can access our values in the constant
		 */
		this.logId = logId;
	}
	
	/**
	 * returns the id of the log
	 */
	private int getLog() {
		return logId;
	}
}

/**
 * great, we've completed an enum!, now to access the values in
 * it we need help from our loved for loop, we are gonna make a method
 * to check if the number put as a parameter matches the logid.
 */

/**
 * 
 * @param log
 *		this method receives a parameter to be comparised with the
 *		values in the enum
 */
private void comparison(int log) {
	/**
	 * Now to access the data in our enum we have to do a for loop
	 * to do it we type for( then the enum name, then the keyword to access it(You
	 * pick it, (i picked e.) then two dots ":", and then again the enum name but 
	 * right after the last letter we type .values(), then we close the for ) and open a bracket..
	 */
	for(LogData e: LogData.values()) {
		
		/**
		 * Now we are gonna check if the parameter matches a log id
		 */
		if(log == e.getLog()) {
			System.out.println("That number is a log!");
		} else if(log != e.getLog()) {
			System.out.println("This is embarrasing, seems like that isn't a log");
		}
	}
}

private Enums() {
	
	try {
		
	/**
	 * first it will check if the number 1 is an actual log
	 */
	comparison(1);
	
	/**
	 * after that will check if the numer 500 is an actual log
	 */
	comparison(500);
	
	/**
	 * Then we pause 2 seconds before the next check
	 */
	Thread.sleep(2000);
	
	/**
	 * After we waited 2 seconds, we check if 1521 is a log..
	 */
	comparison(1521);
	
	} catch (Exception e){
		
	}
}

/**
 * execute the program..
 */
public static void main(String args []) {
	Enums e = new Enums();
}

/**
 * And the output was...
 * This is embarrasing, seems like that isn't a log
 * This is embarrasing, seems like that isn't a log
 * That number is a log!
 */

}[/code]
