public static String titleCase(String name) {
    String[] words = name.split(" ");
    String result = "";
    for (String word : words) {
        for (int i = 0; i < word.length(); i++) {
            String next = word.substring(i, i + 1);
            if (i == 0) result += next.toUpperCase();
            else result += next.toLowerCase();
        }
        result += " ";
    }
    return result.substring(0, result.length() - 1);
}

public static String toTitleCase(String string) { Matcher m = Pattern.compile("\\s\\w").matcher(string); StringBuffer caseString = new StringBuffer(); while (m.find()) { m.appendReplacement(caseString, matcher.group().toUpperCase()); } matcher.appendTail(caseString); return caseString.toString; }

public class StringCapitaliser {

  public static void main(String[] args) {
    String testName = "p y r o 214 lol";
    System.out.println(capitaliseName(testName));
    long start = System.currentTimeMillis();
    for (int i = 0; i < 10000000; i++) {
      capitaliseName(testName);
    }
    long end = System.currentTimeMillis();
    System.out.println("Time taken: " + (end - start));
  }

  public static String capitaliseName(String name) {
    char[] input = name.toCharArray();

    int prevCharVal;
    int charVal = (int) input[0];
    if (charVal >= 97 && charVal <= 122) {
      charVal -= 32;
      input[0] = (char) charVal;
    }

    for (int i = 1; i < input.length; i++) {
      prevCharVal = (int) input[i - 1];
      charVal = (int) input[i];
      if (prevCharVal == 32) { //A space
        if (charVal >= 97 && charVal <= 122) {
          charVal -= 32;
          input[i] = (char) charVal;
        }
      }
      else {
        if (charVal >= 65 && charVal <= 90) {
          charVal += 32;
          input[i] = (char) charVal;
        }
      }
    }
    return new String(input);
  }
}
