public static String fixName(final String s) {

final char ac[] = s.toCharArray();
for(int j = 0; j < ac.length; j++)
if(ac[j] == ‘_’) {
ac[j] = ’ ';
if((j + 1 < ac.length) && (ac[j + 1] >= ‘a’)

ac[j + 1] = (char) ((ac[j + 1] + 65) - 97);

			ac[0] = (char) ((ac[0] + 65) - 97);
		}
		return new String(ac);
	} else {
		return s;
	}
