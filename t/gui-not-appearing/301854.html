import javax.swing.*;
import java.awt.*;


/**
 *
 * @author Lester Knome
 */
public class GUI extends JFrame {

    public Main main = new Main();

    public GUI() {
        super("Lester Knome's IRC Client");
        init(); // Start up the GUI
    }

    public void init() { //initialize all the GUI components
        //TODO: Add GUI
        this.setTitle("Lester Knome's IRC Client");
	this.setLocationRelativeTo(null);
        chatArea = new JTextArea();
        jScrollPane1 = new JScrollPane();

        chatArea.setColumns(20);
        chatArea.setRows(5);
        jScrollPane1.setViewportView(chatArea);

        this.add(chatArea);

        // And Finally,
        this.setResizable(false);
        this.setVisible(true);
	this.pack();
        notify();
    }
    
    public static void Print(String Text) {
        chatArea.append(Text + nline);
    }

    // GUI Components
    protected static JTextArea chatArea;
    private JScrollPane jScrollPane1;
    private static String nline = "\n";
}
