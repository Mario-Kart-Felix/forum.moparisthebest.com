  1. Here is the code for doing a shift/up arrow combo.
program ShiftUpArrow_Combo;
begin
  ClearDebug;  // Clears Debug window.
  KeyDown(16); // Holds down Shift key.
  KeyDown(38); // Holds down Up Arrow key.
  Wait(1000);  // Wait for 1 second.
  KeyUp(38);   // Releases the Up Arrow key.
  KeyUp(16);   // Releases the Shift key.
end.
