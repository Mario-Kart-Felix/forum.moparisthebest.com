[code=c]#include <stdio.h>

#define MAXSIZE 200
#define ARRSIZE 100

int main()
{
FILE *pointer;
char fn[MAXSIZE];
char line[MAXSIZE];
int line_c = 0,idx,data;
int arr[ARRSIZE];

for(idx = 0; idx < ARRSIZE; ++idx)
	arr[idx] = '\0';

printf("Enter filename: ");
scanf("%s",&fn);

if((pointer = fopen(fn,"r")) == NULL)
{
	printf("File cannot be read.\n");
	return 1;
}

while(fgets(line,sizeof(line),pointer) != NULL)
{
	sscanf(line,"%d +%d",&idx,&data);
	printf("[%d]: %d %d\n",line_c,idx,data);
	++line_c;
	arr[idx] = data;
}

fclose(pointer);

for(idx = 0; idx < ARRSIZE; ++idx)
{
	if(arr[idx] == '\0')
		break;
	printf("[%d]: %d\n",idx,arr[idx]);
}

}[/code]

Use fopen, fgets, and sscanf.

If you then wanted to split the 1099, you would use sscanf again:

... int exec, location; sscanf(arr[idx],"%2d%2d",&exec,&location); ...
%2d means to read only the first two integers from the given arr[idx].
