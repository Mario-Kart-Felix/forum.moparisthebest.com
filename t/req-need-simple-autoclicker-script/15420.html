Here is one too…

[scar]var
x, y: Integer;

const
WaitTime = 100;

procedure Click;
begin
if (IsFKeyDown(11)) then
begin
repeat
GetMousePos(x, y);
HoldMouse(x + 1, y + 1, True);
Wait(50 + Random(15));
ReleaseMouse(x + 1, y + 1, True);
Wait(WaitTime);
until (IsFKeyDown(12));
end;
end;

begin
repeat
Click;
until (False);
end.[/scar]

Edit: and here is a clicker with couple simple options…

[scar]var
x, y: Integer;

const
WaitTime = 100; // Time between clicks?
RWaitTime = 100; // Any random time? If NO, then change the number to 0.
Clicking = ‘Humanlike’; // “Humanlike” or UnHumanlike?

procedure HumanlikeClick;
begin
HoldMouse(x + 1, y + 1, True)
Wait(50 + Random(15))
ReleaseMouse(x + 1, y + 1, True);
end;

procedure UnHumanlikeClick;
begin
ClickMouse(x + 1, y + 1, True);
end;

procedure Casing;
begin
case Clicking of
‘Humanlike’: HumanlikeClick;
‘UnHumanlike’: UnHumanlikeClick;
end;
end;

procedure Click;
begin
if (IsFKeyDown(11)) then
begin
repeat
GetMousePos(x, y);
Casing;
Wait(WaitTime + Random(RWaitTime));
until (IsFKeyDown(12));
end;
end;

begin
repeat
Click;
until (False);
end.[/scar]
