package org.randqm.game.economy;

import org.randqm.game.mobs.players.Client;
import org.randqm.util.Constants;

/**

  • @Author: RandQm
  • @Project: Forgotten Warriors
  • @Date: 20th of April 2013
    */

public class Loan {

private Client c;

public Loan(final Client c) {
	this.c = c;
}

private int npcId;
private int value;

private int[] openLoans = new int[4]; //TODO: Save to character file


/*
 * Checks if the clicked id is a bank, if it is, it will start the conversation to get a loan.
 */
public void clickBank(int id, boolean isNpc) {
	npcId = 495;
	if (isNpc)
		npcId = id;
	if (isNpc) {
		for (int npc : Constants.BANKERS) {
			if (npc == id)
				c.getDialogueManager().dia(3);
		}
	} else {
		for (int object : Constants.BANK_OBJECTS) {
			if (object == id)
				c.getDialogueManager().dia(3);
		}
	}
}

/*
 * Handles opening a new loan.
 */
public void openLoan() {
	int loan = -1;
	for (int i = 1; i < openLoans.length; i ++) {
		if (openLoans[i] == 0) {
			loan = i;
			break;
		}
	}
	if (loan == -1) {
		System.out.println("[Loan] Player: "+c.playerName+" attemps to open more than 3 loans.");
		//TODO: write bug log
		return;
	}
	openLoans[loan] = value;
	c.getGoldManager().setGold(c.getGoldManager().getGold() + value);
	value = 0;
}

/*
 * Checks how many open loans the player has.
 */
public int getOpenLoans() {
	int count = 0;
	for (int i = 1; i < openLoans.length; i ++) {
		if (openLoans[i] > 0)
			count ++;
	}
	return count;
}


public int getNpcId() {
	return npcId;
}

public int getValue() {
	return value;
}

public void setValue(int value) {
	this.value = value;
}
