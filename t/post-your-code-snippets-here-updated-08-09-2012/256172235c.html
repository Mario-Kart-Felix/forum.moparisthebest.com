  • Utilitie used for file access.

    • Gets a file from a certain directory.

      */
      public static File getFile(String dir) {
      return new File(dir);
      }

    /**

    • Gets a file reader for a certain directory.
    • @param dir The directory of the file.
    • @return A file reader for a certain directory.
    • @throws FileNotFoundException If the file was not found.
      */
      public static BufferedReader getReader(String dir) throws FileNotFoundException {
      return new BufferedReader(new FileReader(dir));
      }

    /**

    • Builds an array of lines from a file.
    • @param dir The directory of the file.
    • @return null if we could not read correctly.
      */
      public static String[] buildArrayFromFile(String dir) {
      if(!doesFileExist(dir)) {
      return null;
      }
      ArrayList lineBuild = new ArrayList();
      try {
      BufferedInputStream stream = new BufferedInputStream(new FileInputStream(dir));
      while(stream.available() != 0) {
      lineBuild.add(getReader(dir).readLine());
      }
      } catch (Exception e) {
      return null;
      }
      return (String[]) lineBuild.toArray();
      }

    /**

    • Gets a file writer for a certain directory.
    • @param dir The directory of the file.
    • @return A file reader for a certain directory.
      */
      public static BufferedWriter getWriter(String dir) throws IOException {
      return new BufferedWriter(new FileWriter(dir));
      }

    /**

    • Writes an array of strings to a file.
    • @param dir The directory of the file.
    • @param write The array of strings to write.
    • @return true if the writing to file was successful, false if not.
      */
      public static boolean writeArrayToFile(String dir, String[] write) {
      try {
      BufferedWriter writer = getWriter(dir);
      for(String line : write) {
      writer.write(line); writer.newLine();
      }
      writer.close();
      } catch (Exception e) {
      return false;
      }
      return true;
      }

    /**

    • Checks whether or not a file exists.
    • @param dir The directory of the file.
    • @return true if the file exists, false if not.
      */
      public static boolean doesFileExist(String dir) {
      return getFile(dir).exists();
      }

[code]package org.lab;

/**

  • Container to other elements.

  • @author Thomas Nappo

  • @param Type of elements this container holds.
    */
    public class Container {

    /**

    • Constructs a new element container.
    • @param cap Container capacity (in slots).
      */
      public Container(int cap) {
      this.capacity = cap;
      }

    /**

    • The capacity of this container.
      */
      public int capacity;

    /**

    • Containment of elements of this container.
      */
      protected T[] containment;

    /**

    • Inserts an object into one of the slots.
    • @param object The object to insert.
    • @param slot The slot to insert it upon.
    • @returns true if we successfully inserted the object into the container’s slot,
    • false if we attempted to insert the object into a slot that does not exist.
      */
      public boolean insert(T object, int slot) {
      if(!validSlot(slot)) {
      return false;
      }
      containment[slot] = object;
      return true;
      }

    /**

    • Inserts an object into the next available slot of the container.
    • @param object The object to insert.
    • @returns true if we successfully inserted the object into the container,
    • false if our container reached it’s capacity.
      */
      public boolean insert(T object) {
      if(containment.length >= capacity) {
      return false;
      }
      containment[containment.length] = object;
      return true;
      }

    /**

    • Gets an object from the container.
    • @param slot The slot of the object.
    • @return The object of the container’s slots.
      */
      public T get(int slot) {
      return containment[slot];
      }

    /**

    • Removes an object from a specific slot of the container.
    • @param slot The slot to remove the object from.
    • @return true if we were successful in removing the object,
    • false if we attempted to remove from a slot that does not exist.
      */
      public boolean remove(int slot) {
      if(!validSlot(slot)) {
      return false;
      }
      containment[slot] = null;
      return true;
      }

    /**

    • Checks if a slot number is valid.
    • @param slot The slot number we will check.
    • @return true if the slot number is valid, false otherwise.
      */
      public boolean validSlot(int slot) {
      return (slot >= 0) && (slot < containment.length);
      }

    /**

    • Rearranges elements of certain slots in this container.
    • @param slot1 The first element slot that will be exchanged with the second.
    • @param slot2 The second element slot that will be exchanged with the first.
    • @return true if the slots successfully were rearranged, false if we were trying to rearrange invalid slots.
      */
      public boolean rearrange(int slot1, int slot2) {
      if(!validSlot(slot1) || !validSlot(slot2)) {
      return false;
      }
      T temp = containment[slot1]; // Temporary object of slot1.
      containment[slot1] = containment[slot2]; containment[slot2] = temp;
      return true;
      }
