package org.apollo.game.event.impl;

import org.apollo.game.event.Event;

/**
 * An event that sends the players weight w/ items
 * @author ares_
 *
 */
public class PlayerWeightEvent extends Event {

	/**
	 * The players weight
	 */
	private final int weight;
	
	/**
	 * The player weight event
	 * @param weight The players weight
	 */
	PlayerWeightEvent(int weight) {
		this.weight = weight;
	}
	
	/**
	 * The getter for the weight of the player
	 * @return the players weight
	 */
	public int getWeight() {
		return weight;
	}
	
}

The encoder

package org.apollo.net.release.r377;

import org.apollo.game.event.impl.PlayerWeightEvent;
import org.apollo.net.codec.game.DataType;
import org.apollo.net.codec.game.GamePacket;
import org.apollo.net.codec.game.GamePacketBuilder;
import org.apollo.net.release.EventEncoder;

/**
 * An {@link EventEncoder} for the {@link PlayerWeightEvent}
 * @author ares_
 * 
 */
public class PlayerWeightEventEncoder extends EventEncoder<PlayerWeightEvent> {

	@Override
	public GamePacket encode(PlayerWeightEvent event) {
		final GamePacketBuilder builder = new GamePacketBuilder(174); // should be the correct packet
		builder.put(DataType.SHORT, event.getWeight());
		return builder.toGamePacket();
	}

}

	/**
	 * Checks a player for a specific item. Also checks amount.
	 * 
	 * @param item
	 *            The item in question.
	 * @return Whether or not the player has the item in their bank, in their inventory, or on their person.
	 */
	public boolean checkForItem(Item item) {
		if (player.getInventory().getItemContainer().containsItem(item)) {
			return true;
		}
		
		if (player.getBank().containsItem(item)) {
			return true;
		}
		if (player.getEquipment().getItemContainer().containsItem(item)) {
			return true;
		}
		
		return false;
	}
	/**
	 * Checks if the container contains the specified item and amount.
	 * 
	 * @param item
	 *            The item to be checked.
	 * @return
	 */
	public boolean containsItem(Item item) {
		if (contains(item.getId())) {
			if (getCount(item.getId()) >= item.getCount()) {
				return true;
			}
		}
		return false;
	}
