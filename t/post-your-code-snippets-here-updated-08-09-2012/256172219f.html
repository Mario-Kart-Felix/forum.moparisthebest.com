  • FileContainer.java

    public final class FileContainer extends FileArchive implements Packable {

    /**

    • The compression flags for this {@link FileArchive}.
      */
      public static final int NONE = 0, BZIP2 = 1, GZIP = 2;

    /**

    • The compression type on this {@link FileArchive}, default is not compression.
      */
      private int compression = NONE;

    /**

    • The encryption keys for the {@link FileArchive}.
      */
      private int[] keys = null;

    /**

    • Option to set this container as encrypted or not.s
      */
      private boolean encrypted;

    /**
    *

    • @param src
    • @throws IOException
      */
      public void pack(byte[] src) throws IOException {
      pack(src, 0, src.length);
      }

    /**

    • Sets the bytes of this {@link FileContainer}.
    • @param src The byte array to pack into this {@link Packable}.
    • @param offset The offset in the source byte array.
    • @param length The length of the source byte array.
      */
      @Override
      public void pack(byte[] src, int offset, int length) throws IOException {
      byte[] array = null;
      if(compression != NONE)
      array = compression == GZIP ? CompressionUtils.gzip(src, offset, length) : compression == BZIP2 ? CompressionUtils.bzip2(src, offset, length) : null;
      else {
      array = new byte[length - offset];
      System.arraycopy(src, offset, array, 0, length);
      }
      ByteBuffer buffer = ByteBuffer.allocate(array.length + (compression == NONE ? 5 : 9));
      buffer.put((byte) compression);
      buffer.putInt(array.length);
      if(compression != NONE)
      buffer.putInt(src.length);
      buffer.put(array);
      if(encrypted && keys != null)
      encipherXTEA(buffer, 5, buffer.capacity(), keys);
      setArray(buffer.array());
      }

    /**

    • Unpacks the bytes parsed from the {@link FileIndex} to form the {@link FileContainer}.
    • @return The unpacked bytes of this container.
      */
      @Override
      public byte[] unpack() throws IOException {
      byte[] array = null;
      ByteBuffer buffer = ByteBuffer.wrap(getArray());
      if(encrypted && keys != null)
      decipherXTEA(buffer, 5, buffer.capacity(), keys);
      int ctype = buffer.get() & 0xFF;
      if(ctype < NONE || ctype > GZIP)
      throw new InvalidArchiveException("Invalid compression type "+ctype);
      int csize = buffer.getInt();
      if(ctype == NONE) {
      array = new byte[csize];
      buffer.get(array, 0, csize);
      } else {
      int usize = buffer.getInt();
      array = ctype == GZIP ? CompressionUtils.uGzip(buffer.array(), 9, csize) : CompressionUtils.uBzip2(buffer.array(), 9, csize);
      if(array.length != usize)
      throw new IOException(“Invalid decompression”);
      }
      compression = ctype;
      return array;
      }

      public void setCompression(int compression) {
      if(compression < NONE || compression > GZIP)

    • Enciphers an XTEA encryption into the byte buffer.
    • @param buffer The buffer to encipher.
    • @param offset The offset in the buffer to encipher to.
    • @param length The length of the byte array or amount of bytes to encipher.
    • @param keys The encryption keys.

      private static void encipherXTEA(ByteBuffer buffer, int offset, int length, int[] keys) {
      int position = buffer.position();
      buffer.position(offset);
      int rounds = (length - offset)/8;
      for(int r = 0; r < rounds; r++) {
      int v0 = buffer.getInt();
      int sum = 0;
      int delta = 0x9e3779b9;
      int v1 = buffer.getInt();
      int cycles = 32;
      while (cycles-- > 0) {
      v0 += ((v1 + (v1 << 4 ^ v1 >>> 5)) ^ (sum + keys[sum & 0x3]));
      sum += delta;
      v1 += (sum + keys[(0x1bbf & sum) >>> 11] ^ (v0 + (v0 << 4 ^ v0 >>> 5)));
      }
      buffer.position(buffer.position() - 8);
      buffer.putInt(v0);
      buffer.putInt(v1);
      }
      buffer.position(position);
      }

    /**

    • Deciphers an XTEA encryption from the byte buffer.
    • @param buffer The buffer to decipher.
    • @param offset The offset in the buffer to decipher from.
    • @param length The length of the byte array or amount of bytes to decipher.
    • @param keys The encryption keys.
      */
      private static void decipherXTEA(ByteBuffer buffer, int offset, int length, int[] keys) {
      int position = buffer.position();
      buffer.position(offset);
      int rounds = (length - offset)/8;
      for(int r = 0; r < rounds; r++) {
      int v0 = buffer.getInt();
      int sum = 0xC6EF3720;
      int delta = 0x9E3779B9;
      int v1 = buffer.getInt();
      int cycles = 32;
      while (cycles-- > 0) {
      v1 -= (sum + keys[(0x1bbf & sum) >>> 11] ^ (v0 + (v0 << 4 ^ v0 >>> 5)));
      sum -= delta;
      v0 -= (v1 + (v1 << 4 ^ v1 >>> 5) ^ (sum + keys[sum & 0x3]));
      }
      buffer.position(buffer.position() - 8);
      buffer.putInt(v0);
      buffer.putInt(v1);
      }
      buffer.position(position);
      }
      }

package org.evelus.util;

import java.io.ByteArrayInputStream;
import java.io.ByteArrayOutputStream;
import java.io.IOException;

import javax.crypto.Cipher;
import javax.crypto.spec.IvParameterSpec;
import javax.crypto.spec.SecretKeySpec;

/**
 * AESUtility.java
 * @version 1.0.0
 * @author Evelus Development (SiniSoul)
 */
public final class AESUtility {
    
    /**
     * The encoding table.
     */
    private static int[] t;
    
    /**
     * Encodes the characters from a string into a byte array.
     * @param s The key to encode.
     * @return The encoded array.
     */
    private static byte[] encode(String s) { 
        int len;
        byte[] array = null;
        if (s.length() > 0) {
            len = 3 * ((s.length() + 3) / 4) - 2;
            array = new byte[len];
            encode(array, 0, s);
        }
        return array;
    }

    /**
     * Encodes the characters from a string into a byte array.
     * @param dest The destination array.
     * @param off The offset in the destination array.
     * @param s The string to encode.
     * @return The difference in the amount of characters encoded and the offset.
     */
    private static int encode(byte[] dest, int off, String s) {
        int len = s.length();
        int oldOff = off;
        int pos = 0;
        while(len > pos) {
            int k0 = -1, k1 = -1, k2 = -1, k3 = -1;
            if(pos < len) {
                char c = s.charAt(pos);
                k0 = c >= 0 && c < t.length ? t[c] : -1;
            }
            if(pos + 1 < len) {
                char c = s.charAt(pos + 1);
                k1 = c >= 0 && c < t.length ? t[c] : -1;
            }
            if(pos + 2 < len) {
                char c = s.charAt(pos + 2);
                k2 = c >= 0 && c < t.length ? t[c] : -1;
            }
            if(pos + 3 < len) {
                char c = s.charAt(pos + 3);
                k3 = c >= 0 && c < t.length ? t[c] : -1;
            }
            dest[off++] = (byte) (k0 << 2 | k1 >>> 4);
            if (k2 != -1) {
                dest[off++] = (byte) (k2 >>> 2 | 240 & k1 << 4);
                if (k3 != -1) {
                    dest[off++] = (byte) (k3 | k2 << 6 & 192);
                    pos += 4;
                    continue;
                }
            }
            break;
        }
        return off - oldOff;
    }

    /**
     * Deciphers a pack file, inner.pack.gz
     * @param input The input stream to decrypt from.
     * @param key The encryption key.
     * @param iv The initialization vector.
     * @return The deciphered pack file.
     */
    public static byte[] decipherPack(InputStream input, String key, String iv) {
        byte[] encodedKey = encode(key);
        byte[] encodedIV = encode(iv);
        Cipher cipherObject = null;
        SecretKeySpec secretKeySpec = new SecretKeySpec(encodedKey, "AES");
        try {
            cipherObject = Cipher.getInstance("AES/CBC/PKCS5Padding");
            cipherObject.init(2, secretKeySpec, new IvParameterSpec(encodedIV));
        } catch (Exception ex) {
            throw new RuntimeException("Exception thrown : "+ex);
        }
        byte[] buffer = new byte[5242880];
        int offset = 0;
        try {
            int read = 0;
            while (read != -1) {
                read = input.read(buffer, offset, 5242880 - offset);
                if(read > 0)
                    offset += read;
            }
        } catch (IOException ioex) {
            throw new RuntimeException("IOException thrown : "+ioex);
        }
        byte[] dest = new byte[offset];
        System.arraycopy(buffer, 0, dest, 0, offset);
        byte[] decryptedData = null;
        try {
            decryptedData = cipherObject.doFinal(dest);
        } catch (Exception ex) {
            throw new RuntimeException("Exception thrown : "+ex);
        }
        Pack200.Unpacker unpacker = Pack200.newUnpacker();
        ByteArrayOutputStream bos = new ByteArrayOutputStream(5242880);
        try {
            JarOutputStream jarOutput = new JarOutputStream(bos);
            GZIPInputStream gzip = new GZIPInputStream(new ByteArrayInputStream(decryptedData));
            unpacker.unpack(gzip, jarOutput);
            jarOutput.close();
        } catch (IOException ioex) {
            throw new RuntimeException("IOException thrown : "+ioex);
        }
        return bos.toByteArray();
    }

public class SearchExpression {
    
   private static char DEFAULT_DELIMITER = '\1';
    
    private CursorNode evaluate(String str, CursorNode current, int height) {
        String[] strs = null;
        if(!str.equals("")) {
            System.out.println("Stack cursor : "+current.cursor);
            if(str.charAt(0) == '(') {
                strs = cut(str, '(', DEFAULT_DELIMITER);
                if(strs != null) {
                    CursorNode child = new CursorNode();
                    current.child = child;
                    child.parent = current;
                    child.position = current.cursor;
                    child.height = ++height;
                    return evaluate(join(strs), child, height);
                }
            }
            if(str.charAt(0) == ')') {
                strs = cut(str, ')', DEFAULT_DELIMITER);
                if(strs != null) {
                    CursorNode parent = (CursorNode) current.parent;
                    height = parent.height;
                    parent.cursor++;
                    parent.child = null;
                    return evaluate(join(strs), parent, height);
                }
            }
            if(str != null) {
                return evaluate(join(cut(str, str.charAt(0), DEFAULT_DELIMITER)), current, height);
            }    
        }
        current.cursor++;
        return current;
    }
    
    /**
     * 
     * @param node
     * @return 
     */
    private static String getTraceback(CursorNode node) {
        CursorNode origin = node;
        String str = "";
        while(node != null) {
            str =  (node == origin ? node.cursor - 1 : node.cursor) + " " + str ;
            node = (CursorNode) node.parent;
        }
        return str;
    }
    
    /**
     * 
     */
    private static class ListNode {
        ListNode parent;
        ListNode child;
    } 
    
    /**
     * Represents the cursor for when the expression is initializes evaluated.
     */
    private static class CursorNode extends ListNode {
        int position;
        int cursor;
        int height;
    }
    
    /**
     * 
     * @param str 
     */
    public SearchExpression(String str) {
        CursorNode rootCursor = new CursorNode();
        try {
            for(String s : split(str, ' '))
                rootCursor = evaluate(s, rootCursor, rootCursor.height);
            if(rootCursor.height != 0)
                throw new RuntimeException("Missing parenthesis in search expression...");
        } catch(RuntimeException exception) {
            System.err.println("RuntimeException thrown while parsing the search expression...");
            System.exit(0);
        }
    }  
}
