[code]package org.hyperion.rs2.model.bookreading;

import org.hyperion.rs2.model.Player;
import org.hyperion.rs2.model.bookreading.Page;

/**

  • A book object that holds all of its pages.
  • @author lurtz1230 (David P.)

*/
public class Book {

/**
 * The player that is reading the book.
 */
public Player player;

/**
 * The interface id for when a player reads a book.
 */
public static final int READ_BOOK_INTERFACE = 837;

/**
 * The books title.
 */
private String title;

/**
 * Returns the title of the book.
 * @return String
 */
public String getTitle() {
	return title;
}

/**
 * Sets the title of the book.
 * @param n The new name.
 */
public void setTitle(String n) {
	title = n;
}

/**
 * The current page the player is viewing.
 */
private int currentpage;

/**
 * Returns the current page.
 * @return Short
 */
public int getCurrentPage() {
	return currentpage;
}

public void setCurrentPage(int n) {
	currentpage = n;
}

/**
 * The current page that is being edited.
 */
private int currenteditpage;

/**
 * The array that holds all of the pages in the book.
 */
private Page[] Pages;

/**
 * Returns the pages list;
 * @return ArrayList
 */
public Page[] getPages() {
	return Pages;
}

/**
 * Add a page to the book.
 */
public Page addPage() {
	if (currenteditpage < Pages.length) {
		Pages[currenteditpage] = new Page();
		currenteditpage++;
		return Pages[currenteditpage-1];
	}
	return null;
}

/**
 * Sends the two pages to the player.
 * page # must be even since it sends two pages at once.
 * @param x The page number.
 */
public void SendPage(int x) {
	if (x > currenteditpage)
		return;
	if (x % 2 == 1)
		return;
	
	player.getActionSender().sendString(903, title);
	
	for (byte p=0; p < 11; p++) {
		if (Pages[x].getLines()[p] == null)
			player.getActionSender().sendString(843+p, " ");
		else
			player.getActionSender().sendString(843+p, Pages[x].getLines()[p]);
	}
	if (Pages[x+1] != null) {
		for (byte p=0; p < 11; p++) {
			if (Pages[x+1].getLines()[p] == null)
				player.getActionSender().sendString(854+p, " ");
			else
				player.getActionSender().sendString(854+p, Pages[x+1].getLines()[p]);
		}
	}
	else {
		for (byte p=0; p < 11; p++) {
			player.getActionSender().sendString(854+p, " ");
		}
	}
	
	player.getActionSender().sendString(14165, "Page " + (x+1));
	player.getActionSender().sendString(14166, "Page " + (x+2));
	player.getActionSender().sendInterface(READ_BOOK_INTERFACE);
	
	currentpage = x;
	
}


/**
 * The constructor.
 */
public Book(Player p, String name, int total) {
	this.player = p;
	this.title = name;
	if (total > 0)
		this.Pages = new Page[total];
}

}

 * Array holding all of the lines of the page.
 */
private String Lines[];

 * Returns the Lines array.
 * @return String []
 */
public String[] getLines() {
	return Lines;
}

/**
 * The current line that will be inserted next.
 */
private byte currentline;

/**
 * Returns the current line that will be added when addLine() is called.
 * @return byte
 */
public byte getCurrentLine() {
	return currentline;
}

/**
 * Adds a new line to the book.
 * @param newline
 */
public void addLine(String newline) {
	if (currentline == 11)
		return;
	Lines[currentline] = newline;
	currentline++;
}

/**
 * The constructor for the page.
 */
public Page() {
	Lines = new String[11];
}

}

	public static void checkPotential(Client c) {
		if (c.inWild()) {
			if (++c.epDelay == 750) {
			if(c.getItems().getCarriedWealth() > 300000 || c.getPA().getWearingAmount2() > 200000) {
				c.earningPotential += 13 + Misc.random(12);
				if (c.earningPotential > 100) 
					c.earningPotential = 100;
				c.sendMessage("Your EP increases to: " + c.earningPotential + ".");
				} else {
				c.sendMessage("You must risk more than 300K to increase your EP!");
				}
				c.epDelay = 0;
		}
	}
}
