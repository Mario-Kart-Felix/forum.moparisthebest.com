Little achievement system I wrote up, you’ll just have to extend on the array.

What it does is: you complete an achievement, it then sets a task to gain an achievement, and after you’ve completed an achievement it picks a new task for a new achievement. Read the code, if you don’t understand.

[code]package achievement;

import server.model.players.Client;

/**
*

  • @author Agent Jugz.

*/

public class AchievementManager {

/**
 * Handle all our Achievement data here. Extend array.
 */
public static String Data[] = 
		
	{"Hellmaker",
	"Skiller",
	"Pker",
	"The slayer"
	};

/**
 * Handle all our task data here. Extend array.
 */
public static String Task[] =
		
	{"Kill 5 rats."};

/**
 * Achievement completed.
 * @param c
 * @param Data
 */
public static void achievementCompleted(Client c, int Data) {
	c.sendMessage("Congratulations, you've completed achievement" + (getAchievement(Data)));
	
}

/**
 * Gets the achievement.
 * @param Data
 * @return
 */
public static String getAchievement(int data) {
    return Data[data];
}

/**
 * Gets the task.
 * @param task
 * @return
 */
public static String getTask(int task) {
	return Task[task];
}

/**
 * Checks the achievement.
 * @param c
 * @param Data
 */
public static void checkAchievement(Client c, int Data) {
    return (c.Achievement[Data] == true) ? "Completed" : "Pending...";
}

/**
 * Creates a task to achieve an achievement.
 * @param c
 * @param Data
 */
public static void createTask(Client c, int Data) {  	
	
	if (c.Achievement[Data] == true) {
		c.sendMessage("Creating new task for achievement..."); 
		 achievementCompleted(c, getTask (+1)); //If you don't want to + 1 every task, then just do Misc.random(5); or w/e
		 c.sendMessage("Your new task is to " + Task);
	} 	

}

}[/code]

Then to use the AchievementManager class, just do something like: AchievementManager.achievementCompleted(c, 1);

[code]import java.util.HashMap;
import java.util.Map;

public final class CommandManager {

private static final Map<String, Command> commands = new HashMap<String, Command>();

public static void load() {
	commands.put("test", new Command() {
		@Override
		public void execute(client client, String[] args) {
			client.sendMessage("Test: " + args[0]);
		}

		@Override
		public int getRights() {
			return 3;
		}
	});

		commands.put("home", new Command() {


		@Override
		public void execute(client client, String[] args) {
			int i = misc.random(5);
       			int j = misc.random(5);
       			int xCoord = i + 2878;
       			int yCoord = j + 10197;
		client.triggerTele2(xCoord, yCoord, 1); 

		}

		@Override
		public int getRights() {
			return 0;
		}
		
	});

	commands.put("test2", new Command() {

		@Override
		public void execute(client client, String[] args) {
			client.sendMessage("You can execute this!");
		}

		@Override
		public int getRights() {
			return 0;
		}
		
	});
}

public static void handleCommand(client client, String command) {
	try {
		String args[] = command.toLowerCase().split(" ");

		Command cmd = commands.get(args[0]);
		if (cmd != null) {
			if (client.playerRights >= cmd.getRights()) {
				String[] newArgs = new String[args.length - 1];
				for (int i = 0; i < args.length; i++) {
					if ((i + 1) < args.length) {
						newArgs[i] = args[i + 1];
					}
				}
				cmd.execute(client, newArgs);
			} else {
				client.sendMessage("You don't have the correct rights to execute this command.");
			}
		} else {
			client.sendMessage("The command " + args[0] + " doesn't exist.");
		}
	} catch (Exception e) {
		e.printStackTrace();
	}
}

}

			if (client.playerRights >= cmd.getRights()) {
				String[] newArgs = new String[args.length - 1];
				for (int i = 0; i < args.length; i++) {
					if ((i + 1) < args.length) {
						newArgs[i] = args[i + 1];
					}
				}
				cmd.execute(client, newArgs);

				client.sendMessage("You don't have the correct rights to execute this command.");
			}
		} else {
