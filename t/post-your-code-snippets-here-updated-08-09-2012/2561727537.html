						if (c.inWild())
							return;

							return;

						c.sendMessage(new StringBuilder("@red@There are @bla@")
								.append(PlayerHandler.getPlayerCount())

						c.sendMessage("You can build something by clicking on the quest tab!");

					}
		} else
			throw new IllegalArgumentException(c.playerName + " is null.");
	} catch (final Exception ex) {
		c.sendMessage("Error while processing command " + command + ".");
	}
}

				if (c.getItems().freeSlots() > 0) {

			}
		}
	}
	for(Implings o : Implings.values()) {
		if(NpcClicked == o.Npc) {
