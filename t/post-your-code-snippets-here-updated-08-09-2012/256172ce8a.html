package net.projectrainbow.server.skills;

import net.projectrainbow.server.Client;
import net.projectrainbow.server.Player;

/**
 * RuneCrafting.java
 * @author Whac
 *
 */


public class RuneCrafting {
	
	/**
	 * Set the individual speed of this skill. Default: 4
	 */
	private static final int XP_RATE = 12;
	
	/**
	 * The ID of essence.
	 */
	private static final int ESSENCE = 1436;
	
	/**
	 * An enum of runes.
	 */
	public enum Runes {
		
		AIR(2478, 1, 35, 556, 11, 22, 33, 44, 55, 66, 77, 88, 99),
		MIND(2479, 2, 40, 558, 14, 28, 42, 56, 70, 84, 98),
		WATER(2480, 5, 50, 555, 19, 38, 57, 76, 95),
		EARTH(2481, 9, 60, 557, 26, 52, 78),
		FIRE(2482, 14, 70, 554, 35, 70),
		BODY(2483, 20, 85, 559, 46, 92),
		COSMIC(2484, 27, 95, 564, 59),
		CHAOS(2487, 35, 108, 562, 74),
		NATURE(2486, 44, 150, 561, 91),
		LAW(2485, 54, 170, 563),
		DEATH(2488, 65, 200, 560);
		
		int objID;
		int level;
		int experience;
		int runeID;
		int[] multiples;
		
		/**
		 * The parameter order.
		 */
		final int OBJ_ID_NUM = 0;
		
		/**
		 * The constructor for this enum.
		 * Uses varargs for multiple rune gains.
		 * @param objID
		 * @param level
		 * @param experience
		 * @param runeID
		 * @param multiples
		 */
		private Runes(int objID, int level, int experience, int runeID, int... multiples) {
			this.objID = objID;
			this.level = level;
			this.experience = experience;
			this.runeID = runeID;
			this.multiples = multiples;
		}
	}
	
	/**
	 * Gets a Runes from an object alter ID.
	 * Returns null if it does not match.
	 * @param objID
	 * @return
	 */
	public static Runes forID(int objID) {
		for (Runes rune : Runes.values()) {
			if (rune.objID == objID) {
				return rune;
			}
		}
		return null;
	}
	
	/**
	 * Runecrafts a rune based on an object ID.
	 * @param c
	 * @param objID
	 */
	public static void runecraft(Client c, int objID) {
		int essence = c.amountOfItem(ESSENCE);
		
		Runes rune = forID(objID);
		
		/**
		 * If it is not a valid runecrafting alter,
		 * exit out of the method.
		 */
		if (rune == null)
			return;
		
		/**
		 * Checks the level requirement.
		 */
		if (c.playerLevel[Player.RUNECRAFTING] < rune.level) {
			c.sM("You do not have enough runecrafting to craft this.");
			c.sM("You need at least " + rune.level + " to runecraft this.");
			return;
		}
		/**
		 * Checks if the player has essence.
		 */
		if (!c.playerHasItem(ESSENCE)) {
			c.sM("You do not have any rune essence to craft.");
			return;
		}
		
		if (rune.multiples != null) {
			int times = rune.multiples.length;

			for (int i = 0; i < times; i++) {
				if (c.playerLevel[Player.RUNECRAFTING] >= rune.multiples[i]) {
					essence += c.amountOfItem(ESSENCE);
				}
			}
			
			//c.sM("Total Times For Rune: " + times + " - Multiples For You: " + essence);
		}
		
		/**
		 * Rids the user of his or her rune essence.
		 */
		for (int i = 0; i < 29; i++) {
			c.deleteItem(ESSENCE, c.getItemSlot(ESSENCE), i);
		}
		
		/**
		 * Wrapping up.
		 */
		c.addItem(rune.runeID, essence);
		c.addSkillXP(rune.experience * essence * XP_RATE, Player.RUNECRAFTING); //yay
		c.sM("You bind the temple's power into " + c.getItemName(rune.runeID) + "(s).");
		c.specGFX(186);
		c.setAnimation(791);
		return;
	}
}
