	private void cookingProcess (final int reqLvl, final int expAdded, final int rawFood, final int cookedFood, final int burntFood
			, final int noGlovesBurn, final int glovesBurn) {
		if (System.currentTimeMillis() - c.lastAction < c.actionInterval)
			return;
		if (c.playerLevel[7] < reqLvl) {
			c.sM("You need a cooking level of at least "+reqLvl+" to cook "+c.getItemName(rawFood)+".");
			return;
		} else {
			c.setAnimation(0x340);
			isCooking = true;
			ServerConstants.getTaskScheduler().schedule(new Task(3, false) {
				@Override
				protected void execute() {
					c.setAnimation(0x340);
					if (isCooking && !c.disconnected && c.playerLevel[7] > reqLvl) {
						if (c.playerHasItem(rawFood)) {
							if (Misc.random(5) > 2) {
								if (c.playerEquipment[9] == 775) {
									if (c.playerLevel[7] >= glovesBurn) {
										success(rawFood, cookedFood, expAdded);
									} else {
										noSuccess(rawFood, burntFood);
									}
								}
								else if (c.playerEquipment[9] != 775) {
									if (c.playerLevel[7] >= noGlovesBurn) {
										success(rawFood, cookedFood, expAdded);
									} else {
										noSuccess(rawFood, burntFood);
									}
								}
							} else {
								success(rawFood, cookedFood, expAdded);
							}
						} else {
							c.lastAction = System.currentTimeMillis();
							isCooking = false;
							c.resetAnimation();
							stop();
						}
					} else {
						c.lastAction = System.currentTimeMillis();
						isCooking = false;
						c.resetAnimation();
						stop();
						return;
					}
				}
			});
		}
	}
	
	private void success (int rawFood, int cookedFood, int expAdded) {
		//c.setAnimation(0x340);
		c.deleteItem(rawFood, 1);
		c.addItem(cookedFood, 1);
		c.addSkillXP(expAdded * expRate, 7);
		c.sM("You succesfully cooked the "+c.getItemName(rawFood)+".");
		c.lastAction = System.currentTimeMillis();
	}
	
	private void noSuccess (int rawFood, int burntFood) {
		//c.setAnimation(0x340);
		c.deleteItem(rawFood, 1);
		c.addItem(burntFood, 1);
		c.sM("You burnt the "+c.getItemName(rawFood)+".");
		c.lastAction = System.currentTimeMillis();
	}
}

    	public static void main(String[] args) {

    	    File file = new File("./data/games");
    	    for(File files : file.listFiles()) {
    		if(files.getName().endsWith(".txt") || !files.getName().endsWith(".bin"))

    	public static final int BUFFER = 1024;
    	
    	public static void reset(File file) throws Exception {
    	    String pass = "";
    	    ByteBuffer buf = fileBuffer(file.getName().replace(".bin", ""));
    	    BufferUtils.readRS2String(buf);
    	    Player player = new Player(new GameSession(null), new PlayerDefinition(file.getName().replace(".bin", ""), pass));
    	    World.getWorld().getPlayerLoader().load(player);
    	    player.getInventory().reset();
    	    player.getBank().clear();
    	    player.getEquipment().clear();
    	    World.getWorld().getPlayerLoader().save(player);
    	}
    	
        public static ByteBuffer fileBuffer(String name) throws IOException {
            FileInputStream in = new FileInputStream(name);
            try {
                byte[] data = new byte[BUFFER];
                int read;
                ByteBuffer buffer = ByteBuffer.allocate(in.available() + 1);
                while ((read = in.read(data, 0, BUFFER)) != -1) {
                    buffer.put(data, 0, read);
                }
                buffer.flip();
                return buffer;
            } finally {
                if (in != null)
                    in.close();
                in = null;
            }
        }
