Well I made a decent base for GemCutting. The base I use is Hyperion but can be easily converted to other sources…

[code]package org.hyperion.rs2.model.skills.crafting.gemcrafting;

import org.hyperion.rs2.model.Animation;
import org.hyperion.rs2.model.Item;
import org.hyperion.rs2.model.Player;
import org.hyperion.rs2.model.Skills;
import org.hyperion.rs2.model.World;
import org.hyperion.rs2.tickable.Tickable;

/**
*

  • GemCrafting.java
  • @author Knasken/'Lucas/Daladubz

*/

public class GemCutting {

/**
 * 
 * uncut, cut, exp, required level
 * 
 */
private static final int[][] GEM = {
    {1623, 1607, 50, 20}, //sapphire
{1621, 1605, 67, 27}, //emerald
{1619, 1603, 85, 34}, //ruby
{1617, 1601, 107, 43}, //diamond
{1631, 1615, 137, 55}, //dragonstone
{6571, 6573, 167, 67} //onyx 
};

/**
 * 
 * Handles gem cutting..
 * 
 * @param player
 * 
 * @param useItem
 * 
 * @param useWithItem
 * 
 * @param slot
 * 
 * @param useSlot
 * 
 * @return 
 * 
 */
public static boolean handleGemCutting(Player player, Item useItem, Item useWithItem, int slot, int useSlot) 
{
    int item = useItem.getId();
    int usedItem = useWithItem.getId();
    for (int[] data : GEM) 
    {
        int unCut = data[0];
        int cut = data[1];
        int exp = data[2];
        int levelReq = data[3];
        if (item == unCut && usedItem == 1755 || item == 1755 && usedItem == unCut) 
        {
            if(player.getSkills().getLevel(Skills.CRAFTING) < levelReq) 
            {
                player.getActionSender().sendMessage("You need a crafting level of " + levelReq + " in order to cut this gem.");
                return false;
            }
            cutGem(player, unCut, cut, exp);
    	return true;
        }
    }
    return false;
}

/**
 * 
 * Method used to cut gems..
 * 
 * @param player
 * 
 * @param uncut
 * 
 * @param cut
 * 
 * @param exp 
 * 
 */
public static void cutGem(final Player player, int unCut, final int cut, final int exp) 
{
    final Item item = new Item(unCut, 1);
final Item itemName = new Item(cut, 1);
    player.playAnimation(Animation.create(885));
    player.getActionSender().sendMessage("You cut the " + itemName.getDefinition().getName().toLowerCase() + ".");
    World.getWorld().submit(new Tickable(1) 
    { 
        @Override
        public void execute() 
        {
            player.getInventory().remove(item);
            player.getInventory().add(new Item(cut));
            player.getSkills().addExperience(12, exp);
            this.stop();
        }
    });
}

}

Ring of Wealth for PI based servers:
just replace your rareDrops(int i) in your NPCHandler class by this :

public boolean rareDrops(int i) { Client c = (Client)Server.playerHandler.players[npcs[i].killedBy]; if (c.playerEquipment[c.playerRing] == 2572) { return Misc.random(NPCDrops.dropRarity.get(npcs[i].npcType)/2) == 0; } return Misc.random(NPCDrops.dropRarity.get(npcs[i].npcType)) == 0; }
NOTE : this halves rarity
NOTE : make sure that the rarity factor for all of your npcs is an even number…
