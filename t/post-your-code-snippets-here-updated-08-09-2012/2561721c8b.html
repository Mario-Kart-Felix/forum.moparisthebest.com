Simple Skill Screen template (Note: You’ll have to add the packets for setting the number of pages yourself, since it changes between revisions)

import java.util.Arrays;

public final class SkillScreen {
    public static class SkillPage {
        public static final class Entry {
            private final int levelRequirement;
            private final int itemId;
            private final String description;

            public Entry(final int levelRequirement, final int itemId,
                final String description) throws IllegalArgumentException {
                if (levelRequirement < 1 || levelRequirement > 99) {
                    // switch 99 to MAX_LEVEL
                    throw new IllegalArgumentException(
                        "levelRequirement must be within [1..99]: "
                            + levelRequirement);
                }
                if (itemId < 0 || itemId >= 7956) {
                    // switch 7956 to MAX_ITEM_ID
                    throw new IllegalArgumentException(
                        "itemId must be within [0..7956]");
                }
                if (description == null) {
                    throw new IllegalArgumentException(
                        "description cannot be null");
                }
                if (description.length() == 0) {
                    throw new IllegalArgumentException(
                        "description cannot be empty string");
                }
                this.levelRequirement = levelRequirement;
                this.itemId = itemId;
                this.description = description;
            }

            public int getLevelRequirement() {
                return levelRequirement;
            }

            public int getItemId() {
                return itemId;
            }

            public String getDescription() {
                return description;
            }

            @Override
            public int hashCode() {
                final int prime = 31;
                int result = 1;
                result = prime * result + levelRequirement;
                result = prime * result + itemId;
                result = prime * result
                    + ((description == null) ? 0 : description.hashCode());
                return result;
            }

            @Override
            public boolean equals(final Object obj) {
                if (this == obj) {
                    return true;
                }
                if (obj == null) {
                    return false;
                }
                if (!(obj instanceof Entry)) {
                    return false;
                }
                final Entry other = (Entry) obj;
                if (levelRequirement != other.levelRequirement) {
                    return false;
                }
                if (itemId != other.itemId) {
                    return false;
                }
                if (description == null) {
                    if (other.description != null) {
                        return false;
                    }
                }
                else {
                    if (!description.equals(other.description)) {
                        return false;
                    }
                }
                return true;
            }

            @Override
            public String toString() {
                final StringBuilder builder = new StringBuilder();
                builder.append(Entry.class.getName());
                builder.append("[");
                builder.append("levelRequirement=").append(levelRequirement);
                builder.append(", ");
                builder.append("itemId=").append(itemId);
                builder.append(", ");
                builder.append("description=").append("\"").append(description)
                    .append("\"");
                builder.append("]");
                return builder.toString();
            }
        }

        private boolean isMembersOnly;
        private final Entry[] entries;

        public SkillPage(final boolean isMembersOnly, final Entry[] entries)
            throws IllegalArgumentException {
            if (entries == null) {
                throw new IllegalArgumentException("entries must not be null");
            }
            if (entries.length == 0) {
                throw new IllegalArgumentException(
                    "entries must contain at least one "
                        + Entry.class.getName());
            }
            this.isMembersOnly = isMembersOnly;
            this.entries = entries;// TODO: make defensive copy
        }
        
        public boolean isMembersOnly() {
            return isMembersOnly;
        }

        public Entry[] getEntries() {
            return entries;// TODO: make defensive copy
        }

        @Override
        public int hashCode() {
            final int prime = 31;
            int result = 1;
            result = prime * result + ((isMembersOnly) ? 1 : 0);
            result = prime * result
                + ((entries == null) ? 0 : Arrays.hashCode(entries));
            return result;
        }

        @Override
        public boolean equals(final Object obj) {
            if (this == obj) {
                return true;
            }
            if (obj == null) {
                return false;
            }
            if (!(obj instanceof SkillPage)) {
                return false;
            }
            final SkillPage other = (SkillPage) obj;
            if (isMembersOnly != other.isMembersOnly) {
                return false;
            }
            if (entries == null) {
                if (other.entries != null) {
                    return false;
                }
            }
            else {
                if (!Arrays.equals(entries, other.entries)) {
                    return false;
                }
            }
            return true;
        }

        @Override
        public String toString() {
            final StringBuilder builder = new StringBuilder();
            builder.append(SkillPage.class.getName());
            builder.append("[");
            builder.append("isMembersOnly=").append(isMembersOnly);
            builder.append(", ");
            builder.append("entries=").append(Arrays.toString(entries));
            builder.append("]");
            return builder.toString();
        }
    }

    private final String name;
    private final SkillPage[] pages;

    public SkillScreen(final String name, final SkillPage[] pages)
        throws IllegalArgumentException {
        if (name == null) {
            throw new IllegalArgumentException("name must not be null");
        }
        if (name.length() == 0) {
            throw new IllegalArgumentException("name cannot be empty string");
        }
        if (pages == null) {
            throw new IllegalArgumentException("pages must not be null");
        }
        if (pages.length == 0) {
            throw new IllegalArgumentException(
                "pages must contain at least one " + SkillPage.class.getName());
        }
        this.name = name;
        this.pages = pages;// TODO: make defensive copy
    }

    public String getName() {
        return name;
    }

    public SkillPage[] getPages() {
        return pages;// TODO: make defensive copy
    }

    @Override
    public int hashCode() {
        final int prime = 31;
        int result = 1;
        result = prime * result + ((name == null) ? 0 : name.hashCode());
        result = prime * result
            + ((pages == null) ? 0 : Arrays.hashCode(pages));
        return result;
    }

    @Override
    public boolean equals(final Object obj) {
        if (this == obj) {
            return true;
        }
        if (obj == null) {
            return false;
        }
        if (!(obj instanceof SkillScreen)) {
            return false;
        }
        final SkillScreen other = (SkillScreen) obj;
        if (name == null) {
            if (other.name != null) {
                return false;
            }
        }
        else {
            if (!name.equals(other.name)) {
                return false;
            }
        }
        if (pages == null) {
            if (other.pages != null) {
                return false;
            }
        }
        else {
            if (!Arrays.equals(pages, other.pages)) {
                return false;
            }
        }
        return true;
    }

    @Override
    public String toString() {
        final StringBuilder builder = new StringBuilder();
        builder.append(SkillScreen.class.getName());
        builder.append("[");
        builder.append("name=").append("\"").append(name).append("\"");
        builder.append(", ");
        builder.append("pages=").append(Arrays.toString(pages));
        builder.append("]");
        return builder.toString();
    }
}
