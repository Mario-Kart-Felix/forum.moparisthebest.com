Cutting gems for a Rune_Source base.

[code=java]package com.rs2.model.content.skills;

import java.util.HashMap;

import com.rs2.model.players.Item;
import com.rs2.model.players.Player;

public class CuttingGems {

/**
 * author: I Roll Deep (Diego)
 * @author Rodrigo(Imsuperman05) 
 * Gem Cutting system
 * @version 1.5
 */

/**
 * Constructor instead of annoying Player p/c in every method.
 */
static Player p;

public CuttingGems(final Player Player) {
	CuttingGems.p = Player;
}

/**
 * Format: Uncut, cut, Level, XP, Anim
 */

public static enum GEMS {
	SAPPHIRE(1623, 1607, 20, 70, 888), EMERALD(1621, 1605, 27, 150, 889), RUBY(
			1619, 1603, 34, 500, 887), DIAMOND(1617, 1601, 43, 500, 890), DRAGONSTONE(
			1631, 1615, 55, 700, 885), ONYX(6571, 6573, 67, 1500, 2727);

	int uncutId, cutId, LevelNeeded, XPGiven, AnimForGem;

	int getUncutID() {
		return this.uncutId;
	}

	int getCutID() {
		return this.cutId;
	}

	int getLevelNeed() {
		return this.LevelNeeded;
	}

	int getXPGiven() {
		return this.XPGiven;
	}

	int getAnim() {
		return this.AnimForGem;
	}

	GEMS(final int uncutId, final int cutId, final int LevelNeeded,
			final int XPGiven, final int AnimForGem) {
		this.uncutId = uncutId;
		this.cutId = cutId;
		this.LevelNeeded = LevelNeeded;
		this.XPGiven = XPGiven;
		this.AnimForGem = AnimForGem;
	}

	static HashMap<Integer, GEMS> CuttingGems = new HashMap<Integer, GEMS>();
	static {
		for (final GEMS f : GEMS.values()) {
			GEMS.CuttingGems.put(f.getUncutID(), f);
		}
	}

	static GEMS forId(final int id) {
		return GEMS.CuttingGems.get(id);
	}
};

public static void handleChisel(final Item id4, final Item id5) {
	final int id1 = id4.getId();
	final int id2 = id5.getId();
	if (id1 == 1755) {
		StartCutting(id1);
	} else {
		StartCutting(id2);
	}
}

public static boolean HasReqs(final int id1) {
	final GEMS g = GEMS.CuttingGems.get(id1);
	if (id1 == g.uncutId) {
		if (p.getSkill().getLevel()[12] >= g.getLevelNeed()) {
			return true;
		} else {
			p.getActionSender().sendMessage(
					"You need a crafting level of " + g.getLevelNeed()
							+ " to craft this.");
			return false;
		}
	} else {
		p.getActionSender().sendMessage("Wrong item..");
		System.out.println("Dumb kid tried using a chisel with a cut gem");
		return false;
	}
}

public static void StartCutting(final int id1) {
	final GEMS g = GEMS.CuttingGems.get(id1);
	final Item cut = new Item(g.getCutID());
	if (HasReqs(id1)) {
		p.getInventory().addItem(cut);
		p.getSkill().addExp(14, g.getXPGiven());
		p.getInventory().removeItem(new Item(id1));
		p.getUpdateFlags().sendAnimation(g.AnimForGem, 0);

	}
}

}[/code]
