import rs2.model.players.Client;
import rs2.util.Misc;

/**

  • A class handling armour set trading.
  • @author Relex lawl / Relex

*/

public class ArmourSets {
/**
* Enum containing armour set information.
/
private static enum Armour {
/*
* Bronze armour plateleg set.
/
BRONZE_ARMOUR_SET_LG(11814, 1155, 1117, 1075, 1189),
/*
* Bronze armour plateskirt set.
/
BRONZE_ARMOUR_SET_SK(11816, 1155, 1117, 1087, 1189),
/*
* Dwarf-multi cannon.
/
DWARF_CANNON_SET(11967, 6, 8, 10, 12);
/*
* The set’s box id.
/
private int armourSetId;
/*
* Items exchanged for box set.
/
private int[] armourPieceIds;
/*
* Constructor for armour enum.
*
* @param armourSetId The set’s box id.
* @param armourPieceIds Items exchanged for box set.
/
private Armour(int armourSetId, int… armourPieceIds) {
this.armourSetId = armourSetId;
this.armourPieceIds = armourPieceIds;
}
/*
* Gets the armour set’s item Id.
*
* @return armourSetId;
/
private final int getArmourSetId() {
return armourSetId;
}
/*
* Gets the armour piece’s item Id.
*
* @return armourPieceIds.
/
private final int[] getArmourPieceId() {
return armourPieceIds;
}
/*
* Gets the exchanged armor set’s name.
*
* @return Armour set name.
/
private final String getName() {
return Misc.optimizeText(toString().toLowerCase().replaceAll("_", " "));
}
}
/*
* Handles player exchanging armour sets.
*
* @param c Client (player) exchanging box set.
* @param armourSetId Item Id player is trying to exchange.
/
public static void exchangeArmourSet(Client c, final int armourSetId) {
Armour a = getArmourForId(armourSetId);
if (a == null) {
c.sendMessage(“You cannot exchange this item!”);
return;
}
if (c.getItems().freeSlots() < (a.getArmourPieceId().length - 1)) {
c.sendMessage(“Not enough inventory space to unassemble this set!”);
return;
}
c.getItems().deleteItem(a.getArmourSetId(), 1);
for (int i = 0; i < a.getArmourPieceId().length; i++) {
if (a.getArmourPieceId()[i] > 0)
c.getItems().addItem(a.getArmourPieceId()[i], 1);
}
c.sendMessage("You have succesfully exchanged your " + a.getName() + “.”);
}
/*
* Gets the Armour enum for said id.
*
* @param armourSetId Armour id to get Armour enum for.
* @return Armour enum.
*/
private static final Armour getArmourForId(final int armourSetId) {
for (Armour a : Armour.values()) {
if (a.getArmourSetId() == armourSetId)
return a;
}
return null;
}
