package org.zephyrion.framework.world.zone;

public class ZoneBuilder {
private byte flag;
private final int[] bounds;

public ZoneBuilder(int[] bounds) {
	if(bounds.length != 4) {
		System.out.println(this + ":\tFailed to initialize, bounds.length != 4");
		this.bounds = new int[4];
	} else {
		this.bounds = setBounds(bounds[0], bounds[1], bounds[2], bounds[3]);
	}
}

public ZoneBuilder setFlag(byte flag) {
	if (flag > 0x8) {// Replace with highest flag in Zone
		System.out.println(this + ":\tFailed to setFlag(" + flag + "); flag > 0x8");
		return this;
	}
	this.flag = flag;
	return this;
}

public ZoneBuilder setFlags(byte... flags) {
	for (byte flag : flags) {
		this.flag |= flag;
	}
	return this;
}

public byte getFlag() {
	return flag;
}

private int[] setBounds(int top, int left, int bottom, int right) {
	int[] bounds = new int[4];
	if (left > right) {
		System.out.println("Failed to setBounds() of " + this
				+ ", Left bound > Right bound");
	} else {
		bounds[0] = top;
	}
	if (right < left) {
		System.out.println("Failed to setBounds() of " + this
				+ ", Right bound < Left bound");
	} else {
		bounds[1] = left;
	}
	if (top < bottom) {
		System.out.println("Failed to setBounds() of " + this

		bounds[2] = bottom;
	}
	if (bottom > top) {
		System.out.println("Failed to setBounds() of " + this

	} else {
		bounds[3] = right;
	}
	return bounds;
}

public int[] getBounds() {
	return bounds;

public class Zone {
    private byte flag;
    private int[] bounds;

    }
   
    public Zone(ZoneBuilder builder) {
    	this.flag = builder.getFlag();
    	this.bounds = builder.getBounds();
    }

    public byte getFlag() {
    	return flag;
    }

    public int[] getBounds() {
           return bounds;
    }

    public boolean isSafe() {
    	return ((flag & Zone.SAFE) == Zone.SAFE) ? true : false;
    }
   
    public boolean isPVP() {
    	return ((flag & Zone.PVP) == Zone.PVP) ? true : false;
    }
   
    public boolean isFunPVP() {
    	return ((flag & Zone.FUN_PVP) == Zone.FUN_PVP) ? true : false;
    }
   
    public boolean isMultiCombat() {
    	return ((flag & Zone.MULTI) == Zone.MULTI) ? true : false;
    }
   
    public boolean isRestricted() {
    	return ((flag & Zone.RESTRICTED) == Zone.RESTRICTED) ? true : false;
    }
   
    public boolean isWilderness() {
    	return (isPVP() && isMultiCombat()) ? true : false;
    }

		return "[flag=" + flag + ", bounds => top=" + bounds[0] + ", left="
