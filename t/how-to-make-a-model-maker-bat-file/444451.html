[size=14pt]First of all open notepad. Type:

[code]@ECHO OFF
TITLE Model Maker
color 0B
:maker
cls

ECHO.
ECHO Welcome to Model Maker!
ECHO 1.) What Model Type are you making?
ECHO Types: mask, phat, helm, plate body, plate legs
ECHO.
SET /P ModelType=Model Type?

if %ModelType%==mask GOTO mask
if %ModelType%==phat GOTO phat
if %ModelType%==helm GOTO helm
if %ModelType%==plate body GOTO body
if %ModelType%==plate legs GOTO legs

:mask
cls
ECHO.

ECHO.

ECHO.

ECHO if(i == %Itemnumber%){ //~%Name%~
ECHO class8.aStringArray189 = new String[5];
