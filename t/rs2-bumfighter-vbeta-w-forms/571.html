Procedure Attack;
Begin

  If FindColorTolerance(x,y,StrToInt(MonColor1),20,20,510,330,StrToInt(MonTOL)) Or

   Begin
    MonMouse:=Random(4)
    Case MonMouse Of
     1: MoveMouseSmooth(x+Random(MonDistance),y+Random(MonDistance))
     2: MoveMouseSmooth(x-Random(MonDistance),y-Random(MonDistance))
     3: MoveMouseSmooth(x+Random(MonDistance),y-Random(MonDistance))
     4: MoveMouseSmooth(x-Random(MonDistance),y+Random(MonDistance))
    End;
    Wait(100+Random(100))

  End;
End;

Begin

  End;
End;
