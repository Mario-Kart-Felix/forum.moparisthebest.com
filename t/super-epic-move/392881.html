public void divideByZero(int radius) {
    stillgfx(gfxId, absY, absX);// Why the hell is it (y, x) ?
    // NOTE: This is for if you want the * pattern.
    for (int i = i; i <= radius; i++) {
        stillgfx(gfxId, absY - i, absX);
        stillgfx(gfxId, absY - i, absX - i);
        stillgfx(gfxId, absY - i, absX + i);
        stillgfx(gfxId, absY, absX + i);
        stillgfx(gfxId, absY, absX - i);
        stillgfx(gfxId, absY + i, absX);
        stillgfx(gfxId, absY + i, absX + i);
        stillgfx(gfxId, absY + i, absX - i);
    }

    // NOTE: This is if you want just a square.
    for (int i = 1; i <= radius; i++) {
        for (int j = 1; j <= radius; j++) {
            stillgfx(gfxId, absY + i, absX - j);
            stillgfx(gfxId, absY + i, absX + j);
            stillgfx(gfxId, absY - i, absX - j);
            stillgfx(gfxId, absY - i, absX + j);
        }
    }
}
