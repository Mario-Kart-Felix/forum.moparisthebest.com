package info.agathon.model.player;

import java.util.Iterator;

import info.agathon.model.Appearance;
import info.agathon.model.World;
import info.agathon.model.item.Equipment;
import info.agathon.model.item.Item;
import info.agathon.model.player.UpdateFlags.UpdateFlag;
import info.agathon.network.building.Packet;
import info.agathon.network.building.Packet.Type;
import info.agathon.network.building.PacketBuilder;
import info.agathon.utils.Utils;

/**
 * Updates the player.
 * 
 * @author Thomas LeGodais
 */
public class PlayerUpdating {

	/**
	 * The packet type for writing.
	 */
	private static Type VAR_SHORT = Packet.Type.VARIABLE_SHORT;

	/**
	 * The type of updates. The two depend what PoV you are going from, THIS is
	 * the player you currently are. And OTHER is a player that you can see
	 * (multiplayer).
	 * 
	 * @author Thomas LeGodais
	 */
	private static enum UpdateType {

		/*
		 * Updates this player.
		 */
		THIS,

		/*
		 * Updates other player.
		 */
		OTHER;

	}

	/**
	 * This can also be known as "PRE_UPDATE" because what we do here is prepare
	 * the player to get updated, then depending on what is flagged and so
	 * forth. We than update the player.
	 * 
	 * @param player
	 *            The player to update.
	 */
	public static void updatePlayer(Player player) {
		// Check to see if the map region needs to be updated, if so we send the
		// new player map region.
		if (player.isMapRegionChanging()) {
			player.getActionSender().sendLandscape(player);
		}

		// This main packet, which this is used to write in 'bits' instead or
		// writing in bytes. This will hold information about the local players.
		// This will also hold players to add/remove.
		PacketBuilder packet = new PacketBuilder(173, VAR_SHORT);
		packet.startBitAccess();

		// This will be the update block, this holds the block data to send
		// after the main packet.
		PacketBuilder block = new PacketBuilder();

		// Update the current player, also refered to "This" player. Then we are
		// going to update them based on their block.
		updateMovement(player, packet, UpdateType.THIS);
		updateBlock(player, block, false);

		// Now before we do anything else, we are going to add the currently
		// players in the list (size).
		packet.putBits(8, player.getRegionalPlayers().size());

		// This is just a simplified version for a 'for' loop using the
		// Iterator. This will iterate through the local player list.
		Iterator<Player> it$ = player.getRegionalPlayers().iterator();
		while (it$.hasNext()) {

			// This will be used to get the next element in the iteration. Which
			// in this enviroment we are getting the next local player.
			Player other = it$.next();

			// This will do some checks to see what we have to do with the
			// player, so here we are checking if the player is within distance
			// from us, and we also need to make sure the player isn't
			// teleporting. Than finally, we check to see if the player is still
			// in the world list.
			if (other.getLocation().withinDistance(player.getLocation()) && !other.isTeleporting() && World.getWorld().getPlayers().contains(other)) {

				// Update the other player, because we are already updated.
				updateMovement(other, packet, UpdateType.OTHER);

				// This will only be called if we are in a region with a player
				// and the players leaves the region and is not viewable, or the
				// player teleport. Or the player logs out from the world.
			} else {

				// We are going to remove the player that we were processing
				// from the loop (So we don't attempt to keep attempting to
				// update).
				it$.remove();

				// The order of bits that we write will tell the client to
				// remove the other player from are view(player).
				packet.putBits(1, 1);
				packet.putBits(2, 3);
			}
		}

		// This will be used to loop through every player in the server, so
		// everyone will get updated. Even if they are not within are view.
		for (Player others : World.getWorld().getPlayers()) {

			// Here we are going to check if there is enough space left in the
			// local player list for more players to be added.
			if (player.getRegionalPlayers().size() >= 255) {

				// This will add the players slowly, because if this is reached
				// there is not enough space for more players. They will be
				// added when another player gets removed from the list.
				break;
			}

			// This will be used to check to see if the other players will be
			// added, if not we will just ignore them.
			if (others == player || player.getRegionalPlayers().contains(others)) {
				continue;
			}

			// Now if everything is valid with adding the players, we are going
			// to add him to the local list.
			player.getRegionalPlayers().add(others);

			// Here we are going to add the new player in the packet.
			addPlayer(player, others, packet);

			// We are going to update the other players so we can see what they
			// look likes and communicate with them and so on. And we are going
			// to force send the upate.
			updateBlock(others, packet, true);
		}

		// Check to see if the block is empty. If it isn't we write one last
		// thing, then terminate. If it is we terminate normally.
		if (!block.isEmpty()) {

			// Write the magic id.
			packet.putBits(11, 2047);

			// Terminate, were done.
			packet.finishBitAccess();
			packet.put(block.toPacket().getPayload());

			// The block is empty, we can terminate normally.
		} else {

			// Terminate, were done.
			packet.finishBitAccess();
		}

		// Write the data to the players session.
		player.getSession().write(packet.toPacket());
	}

	/**
	 * This will add the new player to the packet, this is always known to be
	 * called "multiplayer". This will make it so we can see other players in
	 * are region.
	 * 
	 * @param player
	 *            The player that we are.
	 * @param others
	 *            The other players to add in our region.
	 * @param packet
	 *            The packet that we are adding them to.
	 */
	private static void addPlayer(Player player, Player others, PacketBuilder packet) {
		// We are going to write the other players index.
		packet.putBits(11, player.getIndex());

		// The xPos to add the player, The different of both of the players
		// locations(X-coord).
		int xPos = others.getLocation().getX() - player.getLocation().getX();

		// The yPos to add the player, The difference of both of players
		// location(Y-coords).
		int yPos = others.getLocation().getY() - player.getLocation().getY();

		// Write the coords, client reads Y, to X.
		packet.putBits(5, yPos).putBits(5, xPos);

		// Writes the rest, and direction.
		packet.putBits(1, 1).putBits(1, 1);
		packet.putBits(3, 1);
	}

	/**
	 * The update block of a player. This will be used to determine what we need
	 * to update on the player. For example if the Apperance mask is flagged,
	 * then we are going to update their apperance.
	 * 
	 * @param p
	 *            The player that we are updating.
	 * @param packet
	 *            The packet for building.
	 * @param fS
	 *            Are we forcing the update?
	 */
	private static void updateBlock(Player p, PacketBuilder packet, boolean fS) {

		// If there isn't an update required, and it isn't force send. Then we
		// do not need to do anything.
		if (!p.getUpdateFlags().isUpdateRequired() && !fS) {
			return;
		}

		// Generate are own block.
		PacketBuilder block = new PacketBuilder();

		// The defualt mask.
		int mask = 0x0;

		// Check to see if the other updates are flagged.
		if (p.getUpdateFlags().isFlagged(UpdateFlag.APPEARANCE) || fS) {
			mask |= 0x2;
		}

		// This will check to see if the mask would overflow a byte.
		if (mask >= 0x4) {
			mask |= 0x4;
			block.put((byte) (mask & 0xFF));
			block.put((byte) (mask >> 8));
		} else {
			block.put((byte) (mask & 0xFF));
		}

		// Apply the flagged updates.
		if (p.getUpdateFlags().isFlagged(UpdateFlag.APPEARANCE) || fS) {
			applyApperance(p, block);
		}

		// Finally convert to a packet.
		Packet buildPacket = block.toPacket();

		// Write the payload.
		packet.put(buildPacket.getPayload());
	}

	/**
	 * Applys the apperance to a player, meaning their items they are equiping
	 * and their looks, skin colour ect.
	 * 
	 * @param p
	 *            The player in which is being applied to.
	 * @param packet
	 *            The packet for writing.
	 */
	private static void applyApperance(Player p, PacketBuilder packet) {
		PacketBuilder playerProps = new PacketBuilder();
		Appearance app = p.getAppearance();
		playerProps.put((byte) app.getGender());
		if ((app.getGender() & 0x2) == 2) {
			playerProps.put((byte) 0);
			playerProps.put((byte) 0);
		}
		playerProps.put((byte) 3);
		playerProps.put((byte) 3);
		if (!app.isNpc()) {
			for (int i = 0; i < 4; i++) {
				if (p.getEquipment().get(i) != null) {
					playerProps.putShort(32768 + p.getEquipment().get(i).getDefinition().getEquipId());
				} else {
					playerProps.put((byte) 0);
				}
			}
			if (p.getEquipment().get(Equipment.SLOT_CHEST) != null) {
				playerProps.putShort(32768 + p.getEquipment().get(Equipment.SLOT_CHEST).getDefinition().getEquipId());
			} else {
				playerProps.putShort(256 + app.getLook(2));
			}
			if (p.getEquipment().get(Equipment.SLOT_SHIELD) != null) {
				playerProps.putShort(32768 + p.getEquipment().get(Equipment.SLOT_SHIELD).getDefinition().getEquipId());
			} else {
				playerProps.put((byte) 0);
			}
			Item chest = p.getEquipment().get(Equipment.SLOT_CHEST);
			if (chest != null) {
				if (!Equipment.isFullBody(chest.getDefinition())) {
					playerProps.putShort(32768 + p.getEquipment().get(Equipment.SLOT_CHEST).getDefinition().getEquipId());
				} else {
					playerProps.putShort((byte) 0);
				}
			} else {
				playerProps.putShort(256 + app.getLook(3));
			}
			if (p.getEquipment().get(Equipment.SLOT_LEGS) != null) {
				playerProps.putShort(32768 + p.getEquipment().get(Equipment.SLOT_LEGS).getDefinition().getEquipId());
			} else {
				playerProps.putShort(256 + app.getLook(5));
			}
			Item hat = p.getEquipment().get(Equipment.SLOT_HAT);
			if (hat != null) {
				if (!Equipment.isFullHat(hat.getDefinition()) && !Equipment.isFullMask(hat.getDefinition())) {
					playerProps.putShort(256 + app.getLook(0));
				} else {
					playerProps.put((byte) 0);
				}
			} else {
				playerProps.putShort(256 + app.getLook(0));
			}
			if (p.getEquipment().get(Equipment.SLOT_HANDS) != null) {
				playerProps.putShort(32768 + p.getEquipment().get(Equipment.SLOT_HANDS).getDefinition().getEquipId());
			} else {
				playerProps.putShort(256 + app.getLook(4));
			}
			if (p.getEquipment().get(Equipment.SLOT_FEET) != null) {
				playerProps.putShort(32768 + p.getEquipment().get(Equipment.SLOT_FEET).getDefinition().getEquipId());
			} else {
				playerProps.putShort(256 + app.getLook(6));
			}
			if (hat != null) {
				if (!Equipment.isFullMask(hat.getDefinition())) {
					playerProps.putShort(256 + app.getLook(1));
				} else {
					playerProps.put((byte) 0);
				}
			} else {
				playerProps.putShort(256 + app.getLook(1));
			}
		} else {
			playerProps.putShort(-1);
			playerProps.putShort(app.getNpcId());
		}
		for (int colour : app.getColours()) {
			playerProps.put((byte) colour);
		}
		playerProps.putShort(p.getEquipment().getStandAnim()); // standAnimIndex
		playerProps.putShort(0x337); // standTurnAnimIndex
		playerProps.putShort(p.getEquipment().getWalkAnim()); // walkAnimIndex
		playerProps.putShort(0x334); // turn180AnimIndex
		playerProps.putShort(0x335); // turn90CWAnimIndex
		playerProps.putShort(0x336); // turn90CCWAnimIndex
		playerProps.putShort(p.getEquipment().getRunAnim()); // runAnimIndex
		playerProps.putLong(Utils.stringToLong(p.getDetails().getName()));
		playerProps.put((byte) 33); // combat
		playerProps.putShort(0);// level
		Packet propsPacket = playerProps.toPacket();
		packet.putByteC(propsPacket.getLength());
		packet.put(propsPacket.getPayload());
	}

	/**
	 * Updates the movement of 2 different types of players, meaning this
	 * player(currently player) and other player (player in your region).
	 * 
	 * @param player
	 *            The player in which we are updating, this will vary from THIS,
	 *            to OTHER. Depends on what we are declaring.
	 * 
	 * @param packet
	 *            The packet used to write the bit data.
	 * 
	 * @param type
	 *            The type of movement that you are updating.
	 */
	private static void updateMovement(Player player, PacketBuilder packet, UpdateType type) {
		switch (type) {
		/*
		 * Update this player.
		 */
		case THIS:
			if (player.isTeleporting() || player.isMapRegionChanging()) {
				packet.putBits(1, 1);
				packet.putBits(2, 3);
				packet.putBits(1, player.isTeleporting() ? 1 : 0);
				packet.putBits(2, player.getLocation().getZ());
				packet.putBits(1, player.getUpdateFlags().isUpdateRequired() ? 1 : 0);
				packet.putBits(7, player.getLocation().getLocalX(player.getLastKnownRegion()));
				packet.putBits(7, player.getLocation().getLocalY(player.getLastKnownRegion()));
			} else {
				if (player.getSprites().getPrimarySprite() == -1) {
					if (player.getUpdateFlags().isUpdateRequired()) {
						packet.putBits(1, 1);
						packet.putBits(2, 0);
					} else {
						packet.putBits(1, 0);
					}
				} else {
					if (player.getSprites().getSecondarySprite() == -1) {
						packet.putBits(1, 1);
						packet.putBits(2, 1);
						packet.putBits(3, player.getSprites().getPrimarySprite());
						packet.putBits(1, player.getUpdateFlags().isUpdateRequired() ? 1 : 0);
					} else {
						packet.putBits(1, 1);
						packet.putBits(2, 2);
						packet.putBits(3, player.getSprites().getPrimarySprite());
						packet.putBits(3, player.getSprites().getSecondarySprite());
						packet.putBits(1, player.getUpdateFlags().isUpdateRequired() ? 1 : 0);
						// Note, running belongs here.
					}
				}
			}
			break;
		/*
		 * Update the other player.
		 */
		case OTHER:
			if (player.getSprites().getPrimarySprite() == -1) {
				if (player.getUpdateFlags().isUpdateRequired()) {
					packet.putBits(1, 1);
					packet.putBits(2, 0);
				} else {
					packet.putBits(1, 0);
				}
			} else if (player.getSprites().getSecondarySprite() == -1) {
				packet.putBits(1, 1);
				packet.putBits(2, 1);
				packet.putBits(3, player.getSprites().getPrimarySprite());
				packet.putBits(1, player.getUpdateFlags().isUpdateRequired() ? 1 : 0);
			} else {
				packet.putBits(1, 1);
				packet.putBits(2, 2);
				packet.putBits(3, player.getSprites().getPrimarySprite());
				packet.putBits(3, player.getSprites().getSecondarySprite());
				packet.putBits(1, player.getUpdateFlags().isUpdateRequired() ? 1 : 0);
			}
			break;
		}
	}
}
