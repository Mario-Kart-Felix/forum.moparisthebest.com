Here is a method broken down for you:

	public void yell(String message) {
		for (Player p : handler.players) { // loop through all players
			if ((p == null) || !p.isActive) // if they are nulled or inactive skip them
				continue; // prevents the instance from crashing your server
			client temp = (client) p; // making a temp client instance for the player
			if ((temp.absX > 0) && (temp.absY > 0)) // they must be on the map (dunno why lol)
				if ((temp != null) && !temp.disconnected && p.isActive) // final check
					temp.sendMessage(message); // FINALLY sends the player the message
		}
	}

to use:

yell("Hello Server");
