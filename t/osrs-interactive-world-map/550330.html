                    });
                }
            });

public void loadURL(final String url) {
    Platform.runLater(new Runnable() {
        @Override 
        public void run() {
            String tmp = toURL(url);

            if (tmp == null) {
                tmp = toURL("http://" + url);
            }

            engine.load(tmp);
        }
    });
}

private static String toURL(String str) {
    try {
        return new URL(str).toExternalForm();
    } catch (MalformedURLException exception) {
            return null;
    }
}
