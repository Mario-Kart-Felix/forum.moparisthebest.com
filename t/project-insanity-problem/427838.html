public void spawnNextWave(Client c) {
if (c != null) {
if (c.waveId >= WAVES.length) {
c.waveId = 0;
return;
}
if (c.waveId < 0){
return;
}
int npcAmount = WAVES[c.waveId].length;
for (int j = 0; j < npcAmount; j++) {
int npc = WAVES[c.waveId][j];
int X = coordinates[j][0];
int Y = coordinates[j][1];
int H = c.heightLevel;
int hp = getHp(npc);
int max = getMax(npc);
int atk = getAtk(npc);
int def = getDef(npc);

}

}
}
