{

static List<String> list = new LinkedList<String>();
    static{
        getList().add("bank");
    }

public void processPacket(Client c, int packetType, int packetSize) {
    if(getList().contains("bank")){
        new BankCommand();
    }
}

public static List<String> getList(){
    return list;
}

}[/code]
