Or what I would do is just do something like this:

public abstract class Room {
    protected Location location_;
    protected int size_;
    protected Room north_;
    protected Room east_;
    protected Room south_;
    protected Room west_;

    protected Room(Room north, Room east, Room south, Room west) {
        north_ = north;
        east_ = east;
        south_ = south;
        west_ = west;
    }

    public abstract Object[] getContents();

    public final Location getLocation() {
        return location_;
    }

    /**
     * Number of tiles in each direction for this room 
     * (consider changing to a static value to prevent odd shaped rooms)
     */
    public final int getSize() {
        return size_;
    }

    public final Room getNorthern() {
        return north_;
    }

    public final Room getEastern() {
        return east_;
    }

    public final Room getSouthern() {
        return south_;
    }

    public final Room getWestern() {
        return west_;
    }
}

public class MiningRoom extends Room {
    public MiningRoom(Room north, Room east, Room south, Room west) {
        super(north, east, south, west);
    }

    @Override
    public Object[] getContents() {
        // Here you would add the floor tile code, walls, doors, misc. objects, etc
    }
}
