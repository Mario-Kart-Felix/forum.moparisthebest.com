TeleNubby July 31, 2016, 5:51am #16

I created a simple XML generator myself for a bot I made a long time ago (nothing to do with RuneScape). It didn’t format the output (i.e. tabs/spaces) but it worked well. You can look at it or use it if you’d like, or even modify it.

package org.telenubby.holdem.xml;

import java.util.ArrayList;
import java.util.HashMap;

/**
* @author TeleNubby <a href="http://telenubby.info/">(Visit Site)</a>
*/
public class Property 
{
	
	/**
	 * The list of every Attribute in this Property object.
	 */
	private HashMap<String, String> attributeList;
	
	/**
	 * The list of keys for the HashMap attribute list.
	 */
	private ArrayList<String> keyList;
	
	/**
	 * The name of this Property.
	 */
	private String name;
	
	/**
	 * This Property's body message.
	 */
	private String body;
	
	/**
	 * Wether or not this Property has a body.
	 */
	private boolean hasBody;
	
	/**
	 * Wether or not this Property is closed.
	 */
	private boolean isClosed;

	/**
	 * The constructor for this class.
	 */
	public Property(String name)
	{
		this.name = name;
		this.hasBody = false;
		this.isClosed = false;
		keyList = new ArrayList<String>();
		attributeList = new HashMap<String, String>();
	}
	
	/**
	 * Adds an Attribute to this Property.
	 * 
	 * @param name the name of the attribute
	 * @param value the value of this attribute
	 * @return the Property after the attribute is added
	 */
	public Property addAttribute(String name, String value)
	{
		attributeList.put(name, value);
		keyList.add(name);
		return this;
	}
	
	/**
	 * Returns the name of this Property.
	 * 
	 * @return the name of this Property
	 */
	public String getName()
	{
		return name;
	}
	
	/**
	 * Returns this Property's body.
	 * 
	 * @return this Property's body
	 */
	public String getBody()
	{
		return body;
	}
	
	/**
	 * Returns true if this Property contains a body.
	 * 
	 * @return true if this Property contains a body
	 */
	public boolean hasBody()
	{
		return hasBody;
	}
	
	/**
	 * Returns true if this Property is closed.
	 * 
	 * @return true if this Property is closed
	 */
	public boolean isClosed()
	{
		return isClosed;
	}
	
	/**
	 * Closes the Property, disallowing it a following body.
	 * By default, there is always a body. Doing this closes the Property tag
	 * like so: <tag/>. Otherwise, it would be left as <tag> and a body
	 * would be created if a new Property was added.
	 * 
	 * @return the Property after it is "closed"
	 */
	public Property addEnd()
	{
		isClosed = true;
		return this;
	}
	
	/**
	 * Adds a manual body to the Property. E.g. <tag>0</tag> or <tag>test</tag>
	 * 
	 * @param body the body to add to this Property
	 * @return the Property after the body was added
	 */
	public Property addBody(String body)
	{
		this.body = body;
		this.hasBody = true;
		return this;
	}
	
	/**
	 * Returns a list of every attribute added to this Property.
	 * 
	 * @return a list of every attribute added to this Property
	 */
	public HashMap<String, String> getAttributes()
	{
		return attributeList;
	}
	
	/**
	 * Returns the set of attribute keys in order of creation.
	 * 
	 * @return the set of attribute keys in order of creation
	 */
	public ArrayList<String> getKeySet()
	{
		return keyList;
	}
	
}

package org.telenubby.holdem.xml;

import java.util.ArrayList;

/**
* @author TeleNubby <a href="http://telenubby.info/">(Visit Site)</a>
*/
public class Message 
{

	/**
	 * The list of every Property in this Message object (in order of creation).
	 */
	private ArrayList<Property> propList;
	
	/**
	 * The constructor for this class.
	 */
	public Message()
	{
		propList = new ArrayList<Property>();
	}
	
	/**
	 * Adds a Property object to this Message.
	 * 
	 * @param theProp the Property object to add to this Message
	 */
	public void addProperty(Property theProp)
	{
		propList.add(theProp);
	}
	
	/**
	 * Returns a String representation of this Message object.
	 * 
	 * @return a String representation of this Message object
	 */
	public String toString()
	{
		StringBuilder s = new StringBuilder();
		ArrayList<String> openProps = new ArrayList<String>();
		for(Property p : propList)
		{
			s.append('<').append(p.getName());
			for(String attName : p.getKeySet())
			{
				String attValue = p.getAttributes().get(attName);
				s.append(' ')
					.append(attName)
					.append("='")
					.append(attValue)
					.append("'");
			}
			if(p.isClosed())
				s.append(" /");
			if(!p.hasBody() && !p.isClosed())
				openProps.add(p.getName());
			s.append('>');
			if(p.hasBody())
			{
				s.append(p.getBody())
					.append("</")
					.append(p.getName())
					.append('>');
			}
		}
		Object[] openProp = openProps.toArray();
		for(int i = openProp.length-1; i > -1; i--)
			s.append("</").append(openProp[i]).append('>');
		return s.toString();
	}
	
}

			autoJoin.addProperty(new Property("body")
					.addAttribute("action", "autoJoin")
					.addAttribute("r", "-1"));
			System.out.println(autoJoin.toString());
