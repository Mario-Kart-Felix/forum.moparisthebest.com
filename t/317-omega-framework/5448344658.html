package omega.rs2.core;

import java.net.InetSocketAddress;
import java.util.concurrent.Executors;
import java.util.logging.Logger;

import org.jboss.netty.bootstrap.ServerBootstrap;
import org.jboss.netty.channel.socket.nio.NioServerSocketChannelFactory;

import omega.rs2.Constants;
import omega.rs2.core.network.netty.PipelineFactory;
import omega.rs2.utilities.Timer;

/**

  • @author Carlos Sanchez-Aviles

  • Class Name: Server.java

  • The main core of our Server.
    */
    public class Server implements Runnable {

    /* The logger class instance for this class. */
    private static final Logger log =
    Logger.getLogger(Server.class.getSimpleName());

    /* The bootstrap class instance for this class. */
    private static final ServerBootstrap bootStrap = new ServerBootstrap(
    new NioServerSocketChannelFactory(
    Executors.newCachedThreadPool(),
    Executors.newCachedThreadPool()));

    private static Server server;
    private String host;
    private int port;

    /**

    • Sets up a new Server.
    • @param port
    •  the port we're connecting to.

    */
    public Server(String host, int port) {
    this.host = host;
    this.port = port;
    }

    /**

    • Starts up the server.
    • @param args
    •  any runtime arguments.

    */
    public static void main(String args[]) {
    int port = Integer.parseInt(args[0]);
    String host = args[1];
    setInstance(new Server(host, port));
    (new Thread(getInstance())).start();
    }

    /**

    • Loads up all the resources.
    • @throws Exception
    •  if we get any errors whilst loading anything.

    */
    private void init() throws Exception {

    }

    @Override
    public void run() {
    try {
    System.out.println(“Starting up “+Constants.SERVER_NAME+””
    + “… [Host-”+host+" Port-"+port+"]");
    //Starts the timer.
    Timer timer = new Timer();

     	//Loads all resources.
 	getInstance().init();
 	
 	//Establishes a connection with the netty channel.
 	getInstance().startUp(port);
 	
 	//finished loading the server.
 	System.out.println(Constants.SERVER_NAME +" is now connected to port: "+port);
 	System.out.println("Loading the server [took "+timer.getElapsed()+" ms]...");
 	
 } catch (Exception ex) {
 	log.info("Error starting "+Constants.SERVER_NAME);
 }

    }

    /**

    • Starts up the new netty Server.
    • @param port
    •  the port to connect to.
    • @throws Exception
    •  if we get any errors whilst setting

    •  up the new netty channel.

    */
    private void startUp(int port) throws Exception {
    bootStrap.setOption(“keepAlive”, true);
    bootStrap.setPipelineFactory(new PipelineFactory());
    bootStrap.bind(new InetSocketAddress(port));
    }

    /**

    • Gets the Server class instance.
    • @return
    •  the server.

    */
    private static Server getInstance() {
    return server;
    }

    /**

    • Sets a new instance.
    • @param server
    •  the new server.

    */
    private static void setInstance(Server server) {
    if (Server.server != null) {
    log.info(“Server instance already in use.”);
    }
    Server.server = server;
    }

}
