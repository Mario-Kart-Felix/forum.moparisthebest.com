I figured it out. Here is the reverse of it. I posted this on villu, hopefully someone other than me will find it useful.

Function code:

function ConvertLvlToXp(Level : Integer) : integer;
var
  i, exp, diff : integer;
begin
  for i := 1 to Level do
  begin
    diff := Trunc(i + 300 * Pow(2, i / 7));
    exp := exp + diff;
  end;
  result := exp/4;
end;

Test code:

program New;

var
  i : integer;

{*******************************************************************************
function ConvertLvlToXp(Xp: Integer): Integer;
By: Ron
Description: Returns exp based on level
*******************************************************************************}
function ConvertLvlToXp(Level : Integer) : integer;
var
  i, exp, diff : integer;
begin
  for i := 1 to Level-1 do
  begin
    diff := Trunc((i) + 300 * Pow(2, (i) / 7));
    exp := exp + diff;
  end;
  result := exp/4;
end;

begin
  ClearDebug;
  for i := 1 to 100 do
  begin
    WriteLn('Level '+IntToStr(i)+': '+IntToStr(ConvertLvlToXp(i)));
  end;
end.
