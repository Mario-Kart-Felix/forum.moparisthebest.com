package server.model.objects.ClickObject.GlobalObjects;

import server.model.players.;
import server.event.;
import server.model.items.;
import server.world.;
import server.*;

public class PickableObjects {

public static void pickupFlax(final Client c, final int object, final int obX, final int obY) {

                    c.startAnimation(827);

                    EventManager.getSingleton().addEvent(new Event() {
                            public void execute(EventContainer container) {        
                                    Server.objectHandler.createAnObject(c, -1, obX, obY);
                                    c.getItems().addItem(1779, 1);
                                    container.stop();
                            }
                            public void stop() {

                            }
                    }, 750);
                    EventManager.getSingleton().addEvent(new Event() {
                            public void execute(EventContainer container) {
                                    Server.objectHandler.createAnObject(c, object, obX, obY);
                                    container.stop();
                            }
                            public void stop() {

                            }
                    }, 13000);
            }
