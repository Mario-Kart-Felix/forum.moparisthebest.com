program Skys_Shadow_Auto_Talker;



{.include SRL\SRL.scar}
{.include SRL\SRL\misc\Trade.scar}
var text, typehumanlike: string;

Procedure Setup;
begin;
     text:= 'text';//What you want it to say
     typehumanlike:= 'no';//Yes for human typing, no for super-spam!

end;

Procedure Talk;
begin
if typehumanlike = 'yes' then;
 begin
   TypeSend(text);
   Wait(Random(100)+Random(100));
 end
if typehumanlike  = 'no' then;
  begin
  SendKeys(text);
  end
end;

Procedure Tradehasbeenfound;
 Begin
   If(FindTrade = True)then
     begin
       accepttrade;
         Repeat
           Wait(1500)
         Until(TradeScreen=False)
     end
   else
     begin
     end
 End;

Begin
 Setup;
   Repeat
     Talk;
     Tradehasbeenfound;
   Until(false)
End.
