This is how I did the ranking if anyone want’s it. I find it quite useful for sorting values from high to low.

public static Map<Player, Integer> sortRank(Map<Player, Integer> map) {
        List<Entry<Player, Integer>> list = new LinkedList<Entry<Player, Integer>>(map.entrySet());
        Collections.sort(list, new Comparator<Entry<Player, Integer>>() {
            public int compare(Entry<Player, Integer> entry, Entry<Player, Integer> entry2) {
            	return entry.getValue().compareTo(entry2.getValue());
            }
        });
        Map<Player, Integer> sorted = new LinkedHashMap<Player, Integer>();
        int rank = map.size();
        for (Entry<Player, Integer> entry : list) {
        	sorted.put(entry.getKey(), rank--);
        }
        return sorted;
}

package org.dementhium.content;

import java.util.ArrayList;

import org.dementhium.content.areas.CoordinateEvent;
import org.dementhium.content.misc.Following;
import org.dementhium.model.Location;
import org.dementhium.model.Mob;
import org.dementhium.model.World;
import org.dementhium.model.map.path.DefaultPathFinder;
import org.dementhium.model.mask.Animation;
import org.dementhium.model.npc.NPC;
import org.dementhium.tickable.Tick;
import org.dementhium.util.Misc;

/**
 * @author Albert Beaupre
 */
public abstract class Family {

	/**
	 * The baby making emote.
	 */
	public static final Animation BANGING_EMOTE = Animation.create(3636);
	
	/**
	 * The first partner in the relationship.
	 */
	private Mob partner;

	/**
	 * The second partner in the relationship.
	 */
	private Mob partner2;
	
	/**
	 * They are a mobs because you never know...they might be npcs or gay.
	 */
	public Family(Mob partner, Mob partner2) {
		this.partner = partner;
		this.partner2 = partner2;
	}
	
	/**
	 * The couples will use different sex styles to make a baby.
	 * @note This is extremely unrealistic; babies cannot be made this way.
	 * @return an array of babies made by the couple.
	 */
	public abstract Child[] getBabies();
	
	/**
	 * @param babyId The index of the child.
	 * @return The built descriptions of the child.
	 */
	public String getDescriptions(int childId) {
		int length = 0;
		for (Object o : getBabies()[childId].getDiagnostics()) {
			if (o instanceof String)
				length++;
		}
		String[] strings = new String[length];
		for (int i = 0; i < length; ++i) {
			if (getBabies()[childId].getDiagnostics()[i] instanceof String)
				strings[i] = (String) getBabies()[childId].getDiagnostics()[i];
		}
		return Misc.formatString(strings);
	}
	
	/**
	 * The couple's baby making process.
	 */
	public void processBaby() {
		final Location l = partner2.getLocation();
		if (!World.getWorld().doPath(new DefaultPathFinder(), partner, 
			l.getX(), l.getY(), false, false).isRouteFound())
            return;
        else Following.combatFollow(partner, partner2);
		World.getWorld().submitAreaEvent(partner, new CoordinateEvent(partner2, 
			l.getX(), l.getY(), partner2.size()) {
			@Override
			public void execute() {
				partner.turnTo(partner2);
				partner2.turnTo(partner);
				partner.animate(BANGING_EMOTE);
				partner2.animate(BANGING_EMOTE);
				World.getWorld().submit(new Tick(1) {
					int count, index;
					@Override
					public void execute() {
						if (count >= 2) {
							World.getWorld().register(getBabies()[index].
								getChildEntity().getNPC().getId(), 
									Location.locate(l.getX()+1, l.getY(), l.getZ()));
							if (partner.isPlayer())
								partner.getPlayer().sendMessage("You have convieved a child! It is "+getDescriptions(index));
							if (partner2.isPlayer())
								partner2.getPlayer().sendMessage("You have convieved a child! It is "+getDescriptions(index));
							index++;
							if (index >= getBabies().length) {
								stop();
								partner.animate(-1);
								partner2.animate(-1);
								return;
							}
						}
						count++;
					}
				});
			}
		});
	}

	/**
	 * @return the first partner
	 */
	public Mob getPartner() {
		return partner;
	}

	/**
	 * @return the second partner.
	 */
	public Mob getPartner2() {
		return partner2;
	}
	
	/**
	 * Sets the first partner.
	 * @param partner The first partner.
	 */
	public void setPartner(Mob partner) {
		this.partner = partner;
	}
	
	/**
	 * Sets the second partner.
	 * @param partner2 The second partner.
	 */
	public void setPartner2(Mob partner2) {
		this.partner2 = partner2;
	}
	
	/**
	 * The interface for the couple's baby.
	 */
	private interface Child {
	
		/**
		 * @return The child entity.
		 */
		public Mob getChildEntity();

		/**
		 * @return The child's attributes.
		 */
		public abstract <T> Object[] getDiagnostics();
		
	}
	
	/**
	 * Holds all of the family repositories.
	 */
	public static class Repository {

		/**
		 * A list containing all of the families.
		 */
		private static ArrayList<Family> families = new ArrayList<Family>();
		
		/**
		 * Initalizes the repository for families.
		 */
		public static void init() {
			
		}
		
		/**
		 * @param family The family to add to the list.
		 */
		public static void add(Family family) {
			families.add(family);
		}
		
		/**
		 * @param index The specified index of the family.
		 * @return The specified family.
		 */
		public static Family get(int index) {
			return families.get(index);
		}
		
		/**
		 * @param family The specified family index.
		 * @return The specified family.
		 */
		public static Family get(Family family) {
			return families.get(families.indexOf(family));
		}
		
	}
	
	public static class DefaultChild extends Family {

		public DefaultChild(Mob partner, Mob partner2) {
			super(partner, partner2);
		}

		@Override
		public Child[] getBabies() {
			return new Child[] { new Child() {

				@Override
				public Mob getChildEntity() {
					return new NPC(13089);
				}
				
				public <T> Object[] getDiagnostics() {
					return new Object[] {
						"gooey", "fat", "brown",
						new Tick(55) {
							@Override
							public void execute() {
								System.out.println("Wart generated!");
							}
						}
					};
				}
			}};
		}
		
	}
}
