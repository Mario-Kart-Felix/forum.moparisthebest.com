public static void CatchImpling(Client c, int npcId) {
	Impling imp = Impling.forId(npcId);
	int equipped = c.playerEquipment[c.playerWeapon];
	if (System.currentTimeMillis() - c.foodDelay >= 1500) {
		if (c.playerLevel[21] >= imp.getReq()) {
			if (equipped == imp.getNet()) {
				if (c.getItems().playerHasItem(11260, 1)) {
					if (c.hasSpace()) {
						c.getItems().deleteItem(11260, 1);
						HandleCatch(c, imp.getItemId(), imp.getExp(), imp.getReq(), imp.getImpName());
					} else {
						c.sendMessage("Not enough inventory space.");
						return;
					}
				} else {
					c.sendMessage("you need a impling jar to catch this");
					return;
				}
			} else {
				c.sendMessage("You need to equip a butterfly net!");
				return;
			}
		} else {
			c.sendMessage("You need atleast " + imp.getReq()
					+ " Hunter To catch that Impling!");
			return;
		}
		c.foodDelay = System.currentTimeMillis();
	}
}

public static void HandleCatch(Client c, int ID, int XP, int Req, String Name) {
	if (c.playerLevel[21] + Misc.random(10) > Misc.random(20) + Req) {
		c.sendMessage("You Cought a " + Name + "!");
		c.getItems().addItem(ID, 1);
		c.startAnimation(6999);
		c.getPA().addSkillXP(XP * Config.HUNTER_EXPERIENCE, 21);
	} else {
		c.sendMessage("You Failed To Catch The " + Name + ".");
		c.startAnimation(6999);
	}
}

public static void giveRabbit(Client c) {
	if (Misc.random(10) < 3) {
		if (c.getItems().freeSlots() > 0) {
			c.getItems().addItem(3226, 1);
		} else {
			c.sendMessage("Not enough inventory space");
		}
	} else {
		c.sendMessage("you fail to catch the rabbit.");
	}
}
