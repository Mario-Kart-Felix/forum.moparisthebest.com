[code]package com.runescape.loader.util;

import java.io.DataInputStream;
import java.io.IOException;
import java.net.URL;

public class WorldList {

public static World[] worlds;
private static boolean loaded = false;

/**
 * Disallow constructor calls
 */
private WorldList() {

}

/**
 * similar to Object.newInstance();
 * @param worldListAddress the world list to load from.
 * @return instance of WorldList
 * @throws IOException if it fails to read the world data
 */
public static WorldList loadFromAddress(String worldListAddress) throws IOException {
    WorldList worldList = new WorldList();
    DataInputStream in = new DataInputStream(new URL(worldListAddress).openStream());
    in.skip(4);
    worlds = new World[in.readShort()];
    for (int i = 0; i < worlds.length; i++) {
        int j = in.readShort();
        StringBuffer prefix = new StringBuffer();
        byte b;
        while ((b = in.readByte()) != 0)
            prefix.append((char) b);
        worlds[i] = new World(prefix.toString(), j & 0x7FFF, in.readShort(), (j & 0x8000) != 0);
    }
    loaded = true;
    return worldList;
}

/**
 * @return returns true if the worldlist has been loaded.
 */
public static boolean hasLoaded() {
    return loaded;
}

/**
 * Inner Class World
 */
public static class World {
    public final String prefix;
    public final int id, players;
    public final boolean members;

    /**
     * @param prefix world prefix
     * @param id world number
     * @param players players on the world
     * @param members if the world is for members only
     */
    public World(String prefix, int id, int players, boolean members) {
        this.prefix = prefix;
        this.id = id;
        this.players = players;
        this.members = members;
    }

    /**
     * @return string format of the world.
     */
    public String toString() {
        return "World " + id + " (" + prefix + ")" + (members ? " - members" : " - free") + " - " + players + " players";
    }
}

}

My world class, probably different from anyone else.

[code]package bot.rsservers;

import java.net.URL;
import java.net.MalformedURLException;
import java.io.*;
import java.util.HashMap;

public class World {

private int location, players, world, serverStatus;
private boolean members;
private String activity, host;

private World(int world, int members, int serverStatus, String activity, String host, int players, int location) {
    this.world = world;
    this.members = members != 0;
    this.serverStatus = serverStatus;
    this.activity = activity;
    this.host = host;
    this.players = players;
    this.location = location;
}

public String getActivity() {
    return activity;
}

public int getLocation() {
    return location;
}

public int getServerStatus() {
    return serverStatus;
}

public boolean isAvailable() {
    return serverStatus == 0;
}

public boolean isFull() {
    return serverStatus == 2;
}

public boolean isOffline() {
    return serverStatus == 1;
}

public boolean isMembers() {
    return members;
}

public int getPlayers() {
    return players;
}

public URL getCodeBase() throws MalformedURLException {
    return new URL("http://" + host + ".runescape.com");
}

public int getWorld() {
    return world;
}

public static HashMap<Integer, World> getWorlds(DataInputStream stream) throws IOException {
    HashMap<Integer, World> worlds = new HashMap<Integer, World>();
    while(stream.available() != 0) {
        int world = stream.readInt();
        int members = stream.readInt();
        int serverStatus = stream.readInt();
        String activity = stream.readUTF();
        String serverCode = stream.readUTF();
        int players = stream.readInt();
        int location = stream.readInt();
        worlds.put(world, new World(world, members, serverStatus, activity, serverCode, players, location));
    }
    return worlds;
}
