<?php // You gotta connect through php in order to get connect with mysql so heres how, i explain in detail too: $dbhost = "localhost"; // this the host for your database, like www.localhost.com or your website give you a host $dbname = "whatsdatabasename"; // this going to be your database name where you store all your tables to operate your site $dbuser = "root"; // root is usually the localhost username so we use that instead, but it depends on your hoster, they will explain that too you though $dbpass = ""; // since username to access database is root on localhost we have no pass, thats usually empty if your using a php host on your computer (like wamp/xampp), but your host will tell you the pass once you subscribe to them $conn = mysql_connect($dbhost, $dbuser, $dbpass) or die("Error: " mysql_error()); // this is where we actually connect to mysql, we put variables into it to supply info for host, username, and password. if it dont connect it going to display mysql_error() $result = mysql_connect_db($dbname); // all this does is select the database 'whatsdatabasename' ?>
