there are tons of things that players and npcs have in common: update flags, position, slot, etc.

look at this https://github.com/lare96/asteria-2.0/blob/master/src/server/world/entity/Entity.java everything in that class is common to both npcs and players. so instead of declaring two identical fields in the player and npc class we can do it once in the entity class

and then while writing code you can do this

public void action(Entity e) { e.forceChat("Hi everyone!"); e.animation(new Animation(1)); e.move(new Position(3222, 3222)); }

and the method can take either a player or npc, because both player and npc extend the Entity class! if we didn’t have an entity class we’d have to make one method for players and the exact same one for npcs.

this is a small example, but this is so useful for things like combat. why? because you only have to write it once! and even better it supports player vs player, player vs npc, and npc vs npc since its entity based

[code=java]import server.euthymia.entity.Entity;
import server.euthymia.entity.player.Methods;
import server.euthymia.entity.player.PlayerHandler;
import server.task.Task;

public class Attack {

/*
 * The attacker
 */
public static Entity Attacker;

/*
 * The victim
 */
public static Entity Victim;

/*
 * Starts the process of attacking a victim
 */
public static void attack(Entity Attacker, Entity Victim) {
	if(Methods.GoodDistance(Victim.absX, Victim.absY, Attacker.absX , Attacker.absY, 2) && !Attacker.combating) {
		
		/********************Attacker*******************************/
		/*
		 * Sets the combating to true, so the attacker doesn't start another attacking process
		 */
		Attacker.combating = true;
		
		/*
		 * The hit
		 */
		int hitDiff = Hitting.getHit(Attacker);
		
		/*
		 * The strength bonus
		 */
		int strBonus = (int) (Attacker.playerBonus[2] * .25);
		
		/*
		 * Debug message for the attacker
		 */
		Attacker.sM("You attacked an entity");
		
		/*
		 * the attack animation
		 */
		Attacker.startAnimation(Animation
			.GetWepAnim(Attacker.playerEquipment[3]));
		
		/*
		 * update required for the attacker next anim.
		 */
		Attacker.updateRequired = true;
		
		/*
		 * Makes the sound for the attacker according to his weapon
		 */
		Attacker.playerSoundEffect(Attacker,Sound.GetWepSound(Attacker.playerEquipment[3]));
		
		/*
		 * Adds the exp 
		 * Exp = a hit * 2 added to strength
		 */
		Attacker.addSkillXP(2 *hitDiff, 2);
		
		/********************VICTIM*******************************/
		/*
		 * Deals the damage to the victim
		 */
		Victim.dealDamage(hitDiff+strBonus);
		
		/*
		 * Starts the block animation for the victim
		 */
		Victim.startAnimation(Animation.BLOCK_ANIM);
		
		/*
		 * tells the victim that he is under attack
		 */
		Victim.sM("You are under attack");
		
		/*
		 * Plays the sound for the victim of blocking according to the shield
		 */
		Victim.playerSoundEffect(Victim, Sound.GetShieldSound((Victim.playerEquipment[5])));
	
		
		/*
		 * The attacker re-attacks the victim
		 */
		reAttack(WeaponSpeed.getWeaponSpeed(Attacker.playerEquipment[3], Attacker), Attacker, Victim);
	}
}
public static void reAttack(final int weaponSpeed, final Entity Attacker, final Entity Victim) {
	Server.getTaskScheduler().schedule(new Task(2, true) {
		private int count = weaponSpeed;

		@Override
		protected void execute() {

			if (count > 0) {
				PlayerHandler.messageToAll = count + "...";
				count--;
			} else {
				if(Attacker.combating) {
					Attacker.combating = false;
					attack(Attacker, Victim);
				}
				stop();
			}
		}
	});
}
