Program SimpleTalker;
{.include SRL/SRL.scar}

const
Chat1 = ‘First sentence here.’; //Replace the sentence with your own.
Chat2 = ‘Second sentence here.’; //Replace the sentence with your own.
Chat3 = ‘Third sentence here.’; //Replace the sentence with your own.

procedure Const2Text;
begin
SetupSRL;
if(IsFKeyDown(1)) then
TypeSend(Chat1);
Sleep(71)
if(IsFKeyDown(2)) then
TypeSend(Chat2);
Sleep(69)
if(IsFKeyDown(3)) then
TypeSend(Chat3);
Sleep(62)
end;

begin
repeat
Const2Text;
Until(IsFKeyDown(9))
end.[/code]
