public static int getRandom(int lowest, int highest) {
	Random r = new Random();
	if (lowest > highest) {
		return -1;
	}
	long range = (long) highest - (long) lowest + 1;
	long fraction = (long) (range * r.nextDouble());
	int numberRolled = (int) (fraction + lowest);
	return numberRolled;
}
