program Myfirstpowerchopper;
{.include srl/srl.scar}

Const
treecolour1= 1517602;

var
x,y:integer;

Procedure Startcutting;
Var L:integer ;
begin
if(FindObj(x,y,‘ree’,TreeColour1,15))then
begin
Mouse(x,y,5,5,True);
ClearDebug;
writeln(‘Walkingtotree’);
Flag;
writeln(‘CuttingTree’);
L:= InvCount;
repeat
Wait(1000+random(1000));
if not (LoggedIn)then
exit;
Until(InvCount > L);
Writeln(‘TreeHasBeenCut’);
Wait(1000)
end;
end;

Procedure AntiBanActions;
var
DontBanMe: Integer;
begin
DontBanMe:= Random(2);
case DontBanMe of

0 : HoverSkill(‘random’, false);
1 : RandomRClick;
2 : GameTab(1 + Random(13));

end;
end;

procedure antirandoms;
begin
if FindFight then
begin
RunAwayDirection(‘W’);
Wait(12000+random(15000));
RunBack;
end;
FindLamp(‘woodcutting’);

end;

Procedure invcheck;
begin
if InvFull = true then
DropAll;

end;

Begin

ClearDebug;
SetupSRL;
ActivateClient;
MouseSpeed := 10 + Random(10);
SetRun(true);
MakeCompass(‘N’);
StartCutting;
AntiBanActions;
Antirandoms;
FindNormalRandoms;
invcheck;
Disguise(‘www.runescape.com - Mozilla Firefox’);
Repeat
Until(false);

end.[/code]
