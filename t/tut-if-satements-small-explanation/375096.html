	public static void main(String args[]){ // This is just an example for this being done in the main.
		int poosmells = 10; // variable, saying poosmells = 10 aka bad.(just for the explanation)
		
		
		if (poosmells == 0) { // if the poo smells nice(0), it will print the line into your cmd saying: yes.
		System.out.println("the poo smells nice!"); // print line yes.
	}else{ // otherwise
		System.out.println("the poo smells bad!"); // if the poo smells bad, it will say this.
	}
	}
}
