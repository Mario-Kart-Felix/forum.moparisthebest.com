public class GraphicsInterface extends JPanel {
	
	private static final long serialVersionUID = 1L;

	@Override
	public void paintComponent(Graphics graphic) {
		Graphics2D twoDimensional = (Graphics2D) graphic;
		paintGraphicsPlane(graphic);
		Stacker.drawBlocks(twoDimensional);
	}

	public void paintGraphicsPlane(Graphics graphic) {
		for (int i = 0; i < 500; i += 50) {
			graphic.drawLine(0, i, 500, i);
			graphic.drawLine(i, 0, i, 500);
		}
	}
}

package dylan;

import java.awt.Color;
import java.awt.Graphics2D;
import java.awt.Rectangle;
import java.util.ArrayList;
import java.util.Random;

import javax.swing.JFrame;

public class Stacker implements Runnable {

	private static long executionRate = 800;

	public static boolean failed = false;

	public static boolean succeeded = false;

	public static ArrayList<MobileBlock> previousBlocks = new ArrayList<MobileBlock>();

	public static MobileBlock mobileBlock = new MobileBlock(new Rectangle(100, 400, 50, 50));

	private static int currentLevel = 0;

	public static JFrame frame = new JFrame("Stacker Program");

	private static boolean running = true;

	private static Random random = new Random();

	public static boolean checkPosition() {
		for (MobileBlock prev : previousBlocks) {
			return mobileBlock.getBuild().x == prev.getBuild().x;
		}
		return false;
	}

	public static void drawBlocks(Graphics2D g2d) {
		g2d.setColor(Color.RED);
		g2d.draw(mobileBlock.getBuild());
		g2d.fill(mobileBlock.getBuild());
		for (MobileBlock prev : previousBlocks) {
			g2d.setColor(Color.GREEN);
			g2d.drawRect(prev.getBuild().x, prev.getBuild().y, prev.getBuild().width, prev.getBuild().height);
			g2d.fillRect(prev.getBuild().x, prev.getBuild().y, prev.getBuild().width, prev.getBuild().height);
			frame.repaint();
		}
		if (succeeded || failed) {   
			g2d.setColor(Color.BLUE);
			g2d.drawString("You Have " + (failed ? "Lost" : "Won") + "The Game!", 100, 100);
			running = false;
		}
	}

	public static void getSpawnPoint() {
		int modify = (random.nextInt(7) + 1) * 50;
		mobileBlock.getBuild().x = modify;
	}

	public static void main(String commandLine[]) {
		frame.setSize(500, 480);
		frame.setVisible(true);
		frame.setResizable(false);
		frame.add(new GraphicsInterface());
		frame.addKeyListener(new KeyEventListener());
		new Thread(new Stacker()).start();
	}

	public static void proceed() {
		mobileBlock.setState(MobileBlockState.NOT_MOVING);
		MobileBlock other = new MobileBlock(new Rectangle(mobileBlock.getBuild().x, mobileBlock.getBuild().y, mobileBlock.getBuild().width, mobileBlock.getBuild().height));
		previousBlocks.add(other);
		mobileBlock.getBuild().y -= 50;
		if (currentLevel == 8) {
			succeeded = true;
		} else {
			currentLevel++;
		}
		getSpawnPoint();
	}

	public static void updateLevel() {
		if (!checkPosition() && currentLevel != 0) {
			failed = true;
			return;
		}
		if (failed || succeeded) {
			return;
		}
		switch (currentLevel) {

		case 0:
		case 1:
		case 2:
		case 3:
		case 4:
			executionRate -= 130;
			break;

		case 5:
		case 6:
			executionRate -= 50;
			break;

		case 7:
			executionRate -= 10;
			break;
			
		}
		proceed();
	}

	@Override
	public void run() {
		while (running) {
			try {
				switch (mobileBlock.getState()) {

				case NOT_MOVING:
					mobileBlock.setState(MobileBlockState.MOVING_RIGHT);
					break;

				case MOVING_LEFT:
					if (mobileBlock.getBuild().x <= 0) {
						mobileBlock.setState(MobileBlockState.MOVING_RIGHT);
					} else {
						mobileBlock.getBuild().x -= 50;
					}
					break;

				case MOVING_RIGHT:
					if (mobileBlock.getBuild().x >= 450) {
						mobileBlock.setState(MobileBlockState.MOVING_LEFT);
					} else {
						mobileBlock.getBuild().x += 50;
					}
					break;

				}
				frame.repaint();
				Thread.sleep(executionRate);
			} catch (InterruptedException exception) {
				exception.printStackTrace();
			}
		}
	}
}
