package console;

import java.io.File;
import java.io.IOException;

public class FileHandlerTest {

public static void main (String[] args){
	String fileOriginal = 	"c:\\testFile1.text";
	String fileRenamed =	"c:\\testFile2.text";
	
	File f = new File(fileOriginal);
	if(!f.exists()){
		try {
			f.createNewFile();								
			Thread.sleep(5000);				
			f.renameTo(new File(fileRenamed));		
			Thread.sleep(5000);
			
			if(f.canExecute() && f.canRead() && f.canWrite() && f.exists()){
			f = new File(fileRenamed);				
			System.out.println("File deleted: " + f.delete());
			}
		} catch (IOException e) {
			e.printStackTrace();
		} catch (InterruptedException e) {
			e.printStackTrace();
		}					
	}			
}

}
