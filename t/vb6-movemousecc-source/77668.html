[code]import java.awt.*;

public class MouseSpline {
int botX = 0, botY = 0;

public MouseSpline() {
	moveMouseCC(100, 100, 5, 50, 5, 10);
}

public static void main(String... args) {
	new MouseSpline();
}

void moveMouse(int x, int y) {
	try{ Robot r = new Robot();
		r.mouseMove(x, y);
		Thread.sleep(150);
	}catch(Exception e) {}
}

public void moveMouseCC(int x, int y, int range, int segments, int pcount, int delay) {
	double u, nc1, nc2, nc3, nc4;
	Point[] p1 = new Point[pcount+2], p2 = new Point[0];
	Point tmp = new Point(0, 0);
	if(segments < 15)
		segments = 50;
	segments = segments + random(-12, 12);
	x = x + random(-range, range);
	y = y + random(-range, range);
	Polygon Poly = new Polygon();
	Poly.addPoint(botX, botY);
	if(pcount < 3)
		pcount = 3;
	if((Math.abs(botX - x) + Math.abs(botY - y)) < 150){
    		pcount = 3;
    		segments = 15;
		}
		for(int i = 0; i < pcount + 2; i++) {
    		if(i < 2){ 
    			p1[i] = new Point(botX, botY);
    		}else if(i >= pcount){
    			p1[i] = new Point(x, y);
    		}else{
    			p1[i] = new Point((int)(Math.min(botX, x) + Math.random() * Math.abs(botX - x)), 
    					(int)(Math.min(botY, y) + Math.random() * Math.abs(botY - y)));
    		}
		}
		for(int i = 1; i < pcount; i++){
			for(u = 0.0; u < 1; u -= (1 / segments)) {
			    nc1 = -(Math.pow(u,3)) / 6 + (u * u) / 2 - u / 2 + 1 / 6;
  		            nc2 = (Math.pow(u,3)) / 2 - (u * u) + 2 / 3;
  		            nc3 = (-(Math.pow(u,3)) + u * u + u) / 2 + 1 / 6;
  		            nc4 = (Math.pow(u,3)) / 6;
  		            tmp.x = (int)(nc1 * p1[i - 1].x + nc2 * p1[i].x + nc3 * p1[i + 1].x + nc4 * p1[i + 2].x);
  		            tmp.y = (int)(nc1 * p1[i - 1].y + nc2 * p1[i].y + nc3 * p1[i + 1].y + nc4 * p1[i + 2].y);
  		            Poly.addPoint(tmp.x, tmp.y);
			}
		}
		Poly.addPoint(x, y);
		int[] xPoints = Poly.xpoints;
		int[] yPoints = Poly.ypoints;
		for(int q = 0; q < xPoints.length - 1; q++)
			p2[q] = new Point(xPoints[q], yPoints[q]);
		for(int i = 0; i < p2.length; i++) {
			moveMouse(p2[i].x, p2[i].y);
			wait(random((int)(delay/2), (int)(delay*2)));
		}
}

public int random(int min, int max) {
    	return ((int) (Math.random() * (max - min))) + min;
	}
	
	public void wait(int ms) {
		try{ Thread.sleep(ms); }catch(Exception e){}
	}
