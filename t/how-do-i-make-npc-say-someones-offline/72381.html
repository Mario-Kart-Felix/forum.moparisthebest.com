Add this at your NPChandler.java

if (npcs[i].npcType == NPCIDHERE) { if (misc.random2(25) <= 3) { npcs[i].updateRequired = true; npcs[i].textUpdateRequired = true; if (PlayerHandler.isPlayerOn("YOURNAME")){ npcs[i].textUpdate = "YOURNAME(Owner)is ONLINE"; } else{ npcs[i].textUpdate = "YOURNAME (Owner) is OFFLINE"; } } }
