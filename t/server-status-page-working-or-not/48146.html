Program CheckForMopar;

Const
// How long to wait inbetween each check in seconds.
// Default is 10.
WaitInSeconds = 10;

// The script will check for this word, and if it’s found, then Mopar
// is online. This word is case sensitive. This word works for
// moparisthebest.com/smf and moparisthebest.com/smf
CheckWord = ‘inconvience’;

// What link to check to see if it’s online. Easy copy and paste is below.
// http://www.moparisthebest.com/smf
// http://www.moparisthebest.com/smf
ForumURL = ‘http://moparisthebest.com/serverstatus.php’;

Var
b, i : Integer;

// The Minimize Procedure will Minimize Scar by pressing Ctrl+M.
// Made by Ron.
Procedure MinimizeToTray;
Begin
KeyDown(17);
KeyDown(GetKeyCode(‘m’));
KeyUp(GetKeyCode(‘m’));
KeyUp(17);
End;

// Procedure to disguise scar using ThreadSafeCall by yakman!
Procedure DisguiseScar(Disguise: String);
Begin
GetApplication.Title := Disguise;
GetSelf.Caption := Disguise;
End;

// Procedure to disguise scar using ThreadSafeCall by yakman!
Procedure SafeDisguise(disguise : String);
Var
p : TVariantArray;
Begin
SetArrayLength(p,1);
p[0] := Disguise;
ThreadSafeCall(‘DisguiseScar’,p);
End;

Procedure WriteStatus(Status : String);
Begin
ClearDebug;
WriteLn(‘Mopar Status: ‘+Status+’.’);
WriteLn(’ Checked a total of ‘+IntToStr(b)+’ time(s).’);
WriteLn(’ Wait Time: ‘+IntToStr(WaitInSeconds)+’ seconds.’);
End;

Procedure Check_For_TOS;
Var
a : String;
Begin
// If it finds the word mopar in the ForumURL then do this.
If(Not(pos(‘mopar’, ForumURL) = 0))Then
Begin
a := GetPage(ForumURL);
If(Not(Pos(‘agree to everything stated below’, a) = 0))Then
Begin
WriteLn(‘Error: TOS Found!’);
WriteLn(’ Please get the TOS cookie before running this script for more’);
WriteLn(’ accurate results!’);
GetApplication.MESSAGEBOX(‘Please get the TOS cookie before running this script for more accurate results!’, ‘Error: TOS Found’, 0);
GetApplication.RESTORE;
GetApplication.BRINGTOFRONT;
TerminateScript;
End Else
Begin
WriteLn(‘TOS not detected.’);
End;
End;
End;

Procedure Check_Connection;
Var
c : String;
Begin
// Store text of moparisthebest.com/smf into variable c.
c := GetPage(ForumURL);
// If the length of c is greater than 5, then do this. Otherwise, mopar
// is not even online to display a message so why bother checking for
// a word.

Begin
// If the pos of the word is not 0 (aka if the word is found) then
// Website is online!
If(Not(Pos(CheckWord, c) = 0))Then
Begin
GetApplication.RESTORE;
GetApplication.BRINGTOFRONT;
WriteStatus(‘Connected’);
GetApplication.MESSAGEBOX(‘Website is online!’, ‘Yay!’, 0);
TerminateScript;

// If the pos of the word is 0 (aka if the word is not found) then
// Website is not online!
If(Pos(CheckWord, c) = 0)Then
Begin
b := b + 1;
WriteLn(‘Didn’‘t find the word.’);
WriteStatus(‘Disconnected’);

Begin
b := b + 1;
WriteStatus(‘Disconnected’);
End;
SafeDisguise(‘Checked a total of ‘+IntToStr(i)+’ time(s)’);

Procedure Start_The_Wait;
Begin
// Set i to equal WaitInSeconds.
i := WaitInSeconds;
// Keep looping until i is below 1.
Repeat
// Change status every second to show how many seconds remaining.
Status(‘Checking again for Mopar in ‘+IntToStr(i)+’ second(s).’);
