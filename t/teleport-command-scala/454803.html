def teleport(player : Player, args : Array[String], w : World) {
  try {
      val parsed = args.map(_.toInt)
      val p = Point.location(parsed(0), parsed(1))
      if (!w.withinWorld(p.getX, p.getY)) throw new IllegalArgumentException
      Logger.mod(player.getUsername + " teleported from " + player.getLocation + " to " + p)
      player.teleport(p.getX, p.getY, true)
    } catch {
      case _ : IllegalArgumentException => message(p, "Invalid Coordinates!")
      case _                            => message(p, "Invalid args. Syntax: TELEPORT x y")
    }
}
