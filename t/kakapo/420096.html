Also, since winterlove uses serialization for saving, you could easily switch to something like this:

package info.rs2asylum.rs2.util.loading;

import java.io.ByteArrayInputStream;
import java.io.ByteArrayOutputStream;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.Statement;

import info.rs2asylum.rs2.model.players.PlayerData;
import info.rs2asylum.rs2.net.Mysql;

public class MySqlPlayerLoader implements AbstractPlayerLoader {

	@Override
	public boolean save(PlayerData data) {
		try {
			PreparedStatement pStatement = null;
			ByteArrayOutputStream baos = new ByteArrayOutputStream();
			ObjectOutputStream oos = new ObjectOutputStream(baos);
			oos.writeObject(data);
			oos.flush();
			oos.close();
			baos.close();
			Statement statement = Mysql.getConnection().createStatement();
			statement.executeQuery ("SELECT null FROM characters WHERE name = '" + data.getUsername() + "' limit 1");
			ResultSet rs = statement.getResultSet();
			if (!rs.next()) {
				pStatement = Mysql.getConnection().prepareStatement("INSERT INTO characters (name, password, data) VALUES(?, ?, ?)");
				pStatement.setString(1, data.getUsername());
				pStatement.setString(2, data.getPassword());
				pStatement.setBytes(3, baos.toByteArray());
			} else {
				pStatement = Mysql.getConnection().prepareStatement("UPDATE characters SET data = ? WHERE name = ?");
				pStatement.setObject(1, baos.toByteArray());
				pStatement.setString(2, data.getUsername());
			}
			pStatement.executeUpdate();
			pStatement.close();
			return true;
		} catch (Exception e) {
			return false;
		} 
	}

	@Override
	public PlayerData load(String username) {
		PlayerData data = null;
		try {
			ByteArrayInputStream bais;
			ObjectInputStream ois;
			PreparedStatement pStatement;
			pStatement = Mysql.getConnection().prepareStatement("SELECT * FROM characters WHERE name = ? LIMIT 1");
			pStatement.setString(1, username);
			ResultSet rs = pStatement.executeQuery();
			if (rs.next()) {
				bais = new ByteArrayInputStream(rs.getBytes("data"));
				ois = new ObjectInputStream(bais);
				data = (PlayerData) ois.readObject();
				ois.close();
				bais.close();
			}
			pStatement.close();
		} catch (Exception e) {
			return null;
		}
		return data;
	}

}
