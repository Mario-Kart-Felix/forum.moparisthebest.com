import java.util.*;

public class Keyboard{
	public static void sendKeys(String s){
		Robot r = null;
		try{
			r = new Robot();
		}catch(AWTException e){}
		for(char c : s.toCharArray()){
			r.keyPress(getKeyCode(c));
			r.keyRelease(KeyEvent.VK_SHIFT);
		}
	}
