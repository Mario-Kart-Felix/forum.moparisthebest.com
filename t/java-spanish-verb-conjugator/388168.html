Alright so I decided to make a Spanish verb conjugator. Right now it doesn’t support irregular verbs such as tener, hacer, conocer, ect. It also only supports present, preterit, and imperfect indicative tense conjugations. I do plan on adding support for irregular verbs and other tenses, EX: Perhaps subjunctive tenses. It adds the endings to all of the verbs for the correct forms, however you may notice that some of the letters are screwed up. They are supposed to be accented letters but they won’t come up right. If anyone knows how to fix that I would love to know!

[code=java]abstract public class VerbTense {

userInput input = new userInput();

public String verb;
public String yo;
public String tu;
public String UdElElla;
public String nosotros;
public String vosotros;
public String UdsEllosEllas;

public void getVerb() {

	input.getUserInput("Enter the verb: ");

	verb = input.input;
	yo = input.input;
	tu = input.input;
	UdElElla = input.input;
	nosotros = input.input;
	vosotros = input.input;
	UdsEllosEllas = input.input;

}

public abstract void Conjugate();

[code=java]public class PresentTenseIndicative extends VerbTense{

userInput input = new userInput();
SpanishVerbConjugator conj = new SpanishVerbConjugator();

public void Conjugate() {

	getVerb();

	if (verb.endsWith("ar")) {
	
		yo = yo.replaceAll("ar", "o");
		tu = tu.replaceAll("ar", "as");
		UdElElla = UdElElla.replaceAll("ar", "a");
		nosotros = nosotros.replaceAll("ar", "amos");
		vosotros = vosotros.replaceAll("ar", "áis");
		UdsEllosEllas = UdsEllosEllas.replaceAll("ar", "an");
				
	}

	if (verb.endsWith("er")) {

		yo = yo.replaceAll("er", "o");
		tu = tu.replaceAll("er", "es");
		UdElElla = UdElElla.replaceAll("er", "e");
		nosotros = nosotros.replaceAll("er", "emos");
		vosotros = vosotros.replaceAll("er", "éis");
		UdsEllosEllas = UdsEllosEllas.replaceAll("er", "en");

	}

	if (verb.endsWith("ir")) {

		yo = yo.replaceAll("ir", "o");
		tu = tu.replaceAll("ir", "es");
		UdElElla = UdElElla.replaceAll("ir", "e");
		nosotros = nosotros.replaceAll("ir", "imos");
		vosotros = vosotros.replaceAll("ir", "ís");
		UdsEllosEllas = UdsEllosEllas.replaceAll("ir", "en");

		}
	
	System.out.println("");
	System.out.println("Conjugation of the verb: " + verb);
	System.out.println("");
	System.out.println("Yo: " + yo);
	System.out.println("Tú: " + tu);
	System.out.println("Usted, Él, Ella: " + UdElElla);

}

[code=java]import java.io.*;

public class userInput {

String input = null;

public String getUserInput(String prompt) {

	System.out.println(prompt + " ");
	
	try {
	
		BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
		input = reader.readLine();
		
		if (input.length() == 0) {

			return null;

		}

		return input;

	} catch (IOException e) {

		System.out.println("IOException: " + e);

	}

	return input;
}
