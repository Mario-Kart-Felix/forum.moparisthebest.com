  }

Simple example:

import java.util.Random;
public class Main {
    private static final Random r = new Random();
    private static final String[] personList = {"lol" ,"wat", "wtf", "omg"};
    public static void main(String[] args) {
            for(int i = 0; i <= 10; i++) {
                    int[] pair = randPair(personList.length);
                    System.out.println("Hi " + personList[pair[0]] + ", I'm " + personList[pair[1]] + ".");
            }
    }
    private static final int[] randPair(int maxLimit) {
            int speakerIdx = r.nextInt(maxLimit);
            int listenerIdx = r.nextInt(maxLimit);
            do { listenerIdx = r.nextInt(maxLimit); } while (speakerIdx == listenerIdx);
            return new int[]{speakerIdx, listenerIdx};
    }
}
