procedure Declareplayers;
begin
HowManyPlayers:=2;
NumberOfPlayers(HowManyPlayers);
CurrentPlayer:=0;

 Players[0].Name :='Player 1';
 Players[0].Pass :='';
 Players[0].Nick :='';
 Players[0].Loc  :='';
 Players[0].Active:=True;
 
 Players[1].Name :='Player 2';
 Players[1].Pass :='';
 Players[1].Nick :='';
 Players[1].Loc  :='';
 Players[1].Active:=True;

end;

procedure Next;
begin
NextPlayer(true)
end;

begin
SetUpSRL;
Declareplayers;
Next;
end.[/scar]
