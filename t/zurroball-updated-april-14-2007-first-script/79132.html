Script Just Copy and Paste:

[code]program OmgnomorenamesZurroBlaster;
//My Zurroball Script By: Omgnomorenames
//Took out the choice to do Snot, Lightning and Eyeball
//Wasn’t working properly
//Added choice to have auto restart
//Game Window should be 600 by 400!
//
//Start Zurroball
//Setup lines 19 and 22

var
Ball,Ball2,speed,runs: Integer;
x,y: Integer;
BallType,BallTime: Integer;
SendScore,RestartScript: String;

procedure Setup;
begin
//Setup
speed:= 20{Mouse Click speed 20-40 is good}
BallType:= 5{2=Metal,5=Normal}
//
SendScore:=‘Yes’{Yes or No}
RestartScript:=‘No’{Will repeat 3 times if selected}
//
//End of Setup Do not Touch anything else unless you
//know what you doing
end;
//Checks What ball is selected and Starts that ball
procedure Start;
begin
if Balltype = 1 then
begin
Balltime:= 100;
ClickMouse(326,271,true)
end else
if BallType = 2 then
begin
Balltime:=100
ClickMouse(334, 306,true)
end else
if BallType = 3 then
begin
Balltime:=200
ClickMouse(333,340,true)
end else
if BallType = 4 then
begin
Balltime:=100
ClickMouse(331,372,true)
end else
if BallType = 5 then
begin
BallTime:=100
ClickMouse(303,192,true)
end else
end;
//Self explained Finds and Clicks Ball
procedure ClickBall;
begin
if (FindColor(x,y,Ball2,1,1,600,319))
then begin
ClickMouse(x,y,true);
ball:=0
end else;
begin
ball:= ball+1;
end;
end;
//Main for each ball
procedure Runball;
begin
wait (2000)
Ball2:= GetColor(248, 88);
repeat
ClickBall;
wait(speed+random(5))
until (ball=BallTime);
end;
//Sendscore when complete if selected
procedure SendScores;
begin
if (SendScore = ‘Yes’) or (SendScore = ‘yes’) then
begin
ClickMouse (244, 344,true)
end else;
end;
//Main
begin
repeat
Setup;
Start;
Runball;
ball:=0;
Runball;
ball:=0;
Runball;
SendScores;
if (RestartScript = ‘No’) or (RestartScript = ‘no’) then
begin
runs:=3
end else;
if (RestartScript = ‘Yes’) or (RestartScript = ‘yes’) then
begin
ClickMouse(202, 305,true)
end;
runs:=runs+1
wait(1000)
until(runs >= 3)
end.[/code]
