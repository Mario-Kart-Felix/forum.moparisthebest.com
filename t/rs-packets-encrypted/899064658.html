    public static void packToStream(String s, Stream stream)
    {
        if(s.length() > 80)
            s = s.substring(0, 80);

        int i = -1;
        for(int j = 0; j < s.length(); j++)
        {
            char c = s.charAt(j);
            int k = 0;
            for(int l = 0; l < availChars.length; l++)
            {
                if(c != availChars[l])
                    continue;
                k = l;
                break;
            }

            if(k > 12)
                k += 195;
            if(i == -1)
            {
                if(k < 13)
                    i = k;
                else
                    stream.writeByte(k);
            } else
            if(k < 13)
            {
                stream.writeByte((i << 4) + k);
                i = -1;
            } else
            {
                stream.writeByte((i << 4) + (k >> 4));
                i = k & 0xf;
            }
        }

        if(i != -1)
            stream.writeByte(i << 4);
    }
