const
  FinalStep = 4; //Number of steps to work with in loading bar
  LoadingBarLength = 25; //How long the loading bar will be

Function LoadingBar(i: Integer): string;
var CurrentStatus, AmountToDraw: Extended;
var StatusDraw: Integer;
begin
  CurrentStatus := i / FinalStep;
  AmountToDraw := Round(CurrentStatus * LoadingBarLength);
  //next part can be optimized this is just quicker for me to type
  repeat
    Result := Result + '|';
    StatusDraw:= StatusDraw + 1;
  until(StatusDraw = AmountToDraw);
  repeat
    Result := Result + '_';
    StatusDraw:= StatusDraw + 1;
  until(StatusDraw = LoadingBarLength);
end;
