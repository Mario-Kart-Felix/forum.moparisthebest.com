There’s no need to do it twice like that, actually.

if(command.toLowerCase().startsWith("yell"))
{
    String text = command.substring(5).trim(); // The substring should be 5.
    text = Character.toUpperCase(text.charAt(0) + text.substring(1).toLowerCase());
    if(isDonor) // Or whatever variable you're using for that...
    {
        yell("[Donor] "+playerName+": "+text);
    }
    else
    {
        yell(playerName+": "+text);
    }
}
