Now when I close out of the script by clicking the X button it actually runs the script as if i clicked my Start button. How would I make it to only start the script when the Start button is clicked, and not run it if i close out of the box?

Weibs July 30, 2016, 1:57am #18

Yup, its possible. Simply declare a global variable as a boolean, and in the procedure that the start button is directed to just set that boolean to true, and check for it in the main loop, like this:

[code]program New;

var
f: TForm;
b: TButton;
ScriptStarted: Boolean;

procedure Start(Sender: TObject);
begin
f.Caption := f.Caption + ‘.’;
f.ModalResult := mrOk;
ScriptStarted := True;
end;

procedure InitForm;
begin
f := CreateForm;
f.Left := 250;
f.Top := 114;
f.Width := 123;
f.Height := 80;
f.Caption := ‘Test’;
f.Color := clBtnFace;
f.Font.Color := clWindowText;
f.Font.Height := -11;
f.Font.Name := ‘MS Sans Serif’;
f.Font.Style := [];
f.Visible := False;
f.PixelsPerInch := 96;
b := TButton.Create(f);
b.Parent := f;
b.OnClick := @Start;
b.Left := 20;
b.Top := 10;
b.Width := 75;
b.Height := 25;
b.Caption := ‘Start’;
b.TabOrder := 8;
end;

procedure SafeInitForm;
var
v: TVariantArray;

begin
SetArrayLength(v, 0);
ThreadSafeCall(‘InitForm’, v);
end;

procedure ShowFormModal;
begin
f.ShowModal;
end;

procedure SafeShowFormModal;
var
v: TVariantArray;

begin
SetArrayLength(v, 0);
ThreadSafeCall(‘ShowFormModal’, v);
end;

begin
SafeInitForm;
SafeShowFormModal;
WriteLn(’’);
if (not (ScriptStarted)) then
begin
WriteLn(‘Form exited.’);
WriteLn(’’);
TerminateScript
end else
WriteLn(‘Button Pressed!’);
WriteLn(’’);
end.[/code]
