Small slayer system I wrote up, kinda incomplete, you can take/give anyway you want. Posting in snippets because I don’t care to explain every piece of code.

This is the new slayer class:


package zerohour.model.players.skills;

import zerohour.model.players.Client;
import zerohour.util.Misc;

/**
 * The slayer class.
 * @author Tyler (Tyb97)
 *
 */
public class Slayer {

	public enum Difficulty {
		EASY, MEDIUM, HARD;
	}

	public static int[] belowThirty = { 1265, 1, 10, 123, 122, 282, 6217, 142,
			1712 };
	public static int[] belowSixty = { 193, 117, 941, 76, 1330, 145 };
	public static int[] belowNinety = { 111, 125, 49, 52 };
	public static int[] oneTwentySix = { 49, 21, 258, 50, 54, 53 };

	public static int[] amount = { 40, 60, 80 };

	public static int[] getLevel(final Client player) {
		if (player.combatLevel <= 30) {
			return belowThirty;
		} else if (player.combatLevel >= 31 && player.combatLevel <= 60) {
			return belowSixty;
		} else if (player.combatLevel >= 61 && player.combatLevel <= 90) {
			return belowNinety;
		} else if (player.combatLevel >= 91 && player.combatLevel <= 126) {
			return oneTwentySix;
		}
		return null;
	}

	public static Difficulty setDifficulty(final Client player) {
		if (player.taskAmount == amount[0]) {
			return Difficulty.EASY;
		} else if (player.taskAmount == amount[1]) {
			return Difficulty.MEDIUM;
		} else if (player.taskAmount == amount[2]) {
			return Difficulty.HARD;
		}
		return null;
	}

	public static void setTask(final Client player) {
		int random = 0;
		for (int i = 0; i < getLevel(c).length; i++) {
			random = i;
		}
		int variable = Misplayer.random(random);
		player.slayerTask = getLevel(c)[variable];
		player.taskAmount = amount[2];
		player.difficulty = setDifficulty(c);
	}
}

Replace your current appendSlayerExperience method located in the NPCHandler class with this.

	/**
	 * Slayer Experience
	 **/
	public void appendSlayerExperience(int i) {
		int npc = 0;
		Client c = (Client) Server.playerHandler.players[npcs[i].killedBy];
		if (c != null) {
			if (c.slayerTask == npcs[i].npcType) {
				c.taskAmount--;
				c.getPA().addSkillXP(npcs[i].MaxHP * Config.XP_RATE, 18);
				if (c.taskAmount <= 0) {
					c.sendMessage("You completed your easy slayer task!");
					c.getPA().addSkillXP((npcs[i].MaxHP * 8) * Config.XP_RATE, 18);
				}

			}
		}
	}
