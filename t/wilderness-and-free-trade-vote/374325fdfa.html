#!/usr/bin/env ruby
require 'net/http'
require 'open-uri'

SUBMIT_FORM_URI = URI.parse('http://services.runescape.com/m=poll/wildy.ws')
USER_REGEX = /hiscorepersonal\.ws\?user1=(.*?)"/

user = '' # if it gets stuck somewhere, change this value so it'll start there
found = 0

def submit_form(user)
  printf "Submitting form for #{user}... "
  res = Net::HTTP.post_form(SUBMIT_FORM_URI, { 'name' => user, 'anonymous' => 'true', 'submit' => 'vote' })
  puts "OK!"
  sleep 10
end

loop do
  f = open("http://services.runescape.com/m=hiscore/overall.ws?category_type=0&table=0&user=#{user.gsub(/ /, '%A0')}")
  begin
    while (line = f.gets)
      match = USER_REGEX.match line
      if match
        user = match[1].gsub /%A0/, ' '
        if found >= 0
          submit_form(user)
        end
        found += 1
      end
    end
  ensure
    f.close
  end
  found = -6 # because it includes 5 extra names
end
