var treeschopped : integer;
const
wieldingaxe=1;

tocut=500;//how many trees to cut script will logout when reached

Procedure DeclarePlayers;
begin
HowManyPlayers :=2;
NumberOfPlayers(HowManyPlayers);
CurrentPlayer:=0;
Players[0].Name :=’’
Players[0].Pass :=’’;
Players[0].Nick :=’’;
Players[0].Active:=True;

Players[1].Name :=’’;
Players[1].Pass :=’’;
Players[1].Nick :=’’;
Players[1].Active:=True;
end;

procedure report;
begin
writeln(’—chopped ‘+inttostr(treeschopped)+ ’ trees—’);
end;
procedure welcome;
begin
writeln(‘we will be chopping ‘+inttostr(tocut)+’ trees today’);
end;
procedure FindWcRandoms;
begin
FindNormalRandoms;
FindQuiz;
FindScapeRune;
FindMime;
FindHead;

FindDemon;
FindName;

end;

procedure towillows;
begin;
SetRun(true)
mouse(706,110,1,2,true)
flag;
mouse(718,72,1,2,true);
flag;
wait(30000)
end;

procedure cutwillows;
var x,y,prevcount,a:integer;
begin
prevcount:=invcount;
if FindObj(X,y,‘Chop down’,2633768,10) then
repeat
Mouse(x,y,10,5,true);
until(invfull)
end;

procedure tobank;
begin
findwcrandoms;
SetRun(true);
mouse(575,79,1,2,true)
flag;
wait(30000)
mouse(599,80,1,2,true)
flag;
wait(5000)
if(findcolorspiral(x,y,4440801,566,20,752,166))Then
wait(500+random(200));
clickmouse(x,y,true)
end;
procedure bankit;
begin
repeat
OpenBank3;
until(BankScreen)
wait(700+random(600))
Deposit(2,28,2);
end;

procedure banktrip;
begin
tobank;

towillows;

begin
welcome;
