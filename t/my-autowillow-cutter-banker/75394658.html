//////MARTIJNTJUHS first script!!!/////
//////its an willow cutter banker…/////
//////very special thanx to Cheeshunk and whiteshadow!!!/////
//////map facing ‘S’ and start in bank/////
///importenst thing!!!have fun!!!with doing nothing!!!////
///send errors to marta_sparta_martin_air@hotmail.com////
{.include srl\srl.scar}
{.include srl\srl\skill\woodcutting.scar}
Procedure DeclarePlayers;
begin
HowManyPlayers :=2;
NumberOfPlayers(HowManyPlayers);
CurrentPlayer:=0;
Players[0].Name :=’’
Players[0].Pass :=’’;
Players[0].Nick :=’’;
Players[0].Active:=True;

Players[1].Name :=’’;
Players[1].Pass :=’’;
Players[1].Nick :=’’;
Players[1].Active:=True;
end;
procedure FindWcRandoms;
begin
FindNormalRandoms;
FindQuiz;
if(FindScapeRune)then Logout;
FindMime;
FindHead;
FindFight;
FindDemon;
FindName;
TalkToRand;
FindEnt(2633768);
FindWcRandoms;
WriteLn(‘random check’);
if not loggedin then loginplayer;
end;

procedure towillows;
begin;
SetRun(true)
mouse(706,110,1,2,true)
flag;
mouse(718,72,1,2,true);
flag;
wait(30000)
end;

procedure cutwillows;
var x,y,prevcount,a:integer;
begin
prevcount:=invcount;
if FindObj(X,y,‘Chop down W’,2633768,10) then
begin
Mouse(x,y,10,5,true);
a:=0;
repeat
wait(1000);
a:=a+1;
if a=3 then break;
until(invcount>prevcount)
end;
end;

procedure tobank;
begin
SetRun(true);
mouse(575,79,1,2,true)
flag;
wait(30000)
mouse(599,80,1,2,true)
flag;
wait(5000)
if(findcolorspiral(x,y,4507110,566,20,752,166))Then
wait(500+random(200));
clickmouse(x,y,true)
end;
procedure bankit;
begin
repeat
OpenBank3;
until(BankScreen)
wait(700+random(600))
Deposit(2,28,2);
end;

procedure banktrip;
begin
tobank;
bankit;
towillows;
end;

begin
Setupsrl;
declareplayers;
FindAxeHeadColor;
ChatsOff;
SetRun(true);
towillows;
repeat
cutwillows;
if(invFull)then banktrip;
until (false)
end.
