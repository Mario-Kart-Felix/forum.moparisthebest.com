public static void pack() throws IOException {
	Logger.getInstance().info("Packing new packed item definitions...");
	RandomAccessFile raf = new RandomAccessFile("data/itemDefinitions.bin", "rw");
	raf.writeShort(SIZE);
	for(int i = 0; i < SIZE; i++) {
		ItemDefinition def = definitions[i];
		if(def == null) {
			raf.writeShort(-1);
			continue;
		}
		raf.writeShort(def.id);
		raf.writeByte(def.isStackable() ? 1 : 0);
		raf.writeByte(def.isNoted() ? 1 : 0);
		raf.writeDouble(def.iventoryWeight);
		raf.writeDouble(def.wearingWeight);
		raf.writeByte(def.bonus.length);
		for(int j = 0; j < def.bonus.length; j++) {
			raf.writeShort(def.bonus[j]);
		}
		raf.writeBytes(def.name);
		raf.writeByte(0);
		raf.writeBytes(def.examine);
		raf.writeByte(0);
	}
}

private static void loadFromBin(File f) throws IOException {
	FileChannel channel = new RandomAccessFile(f, "r").getChannel();
	ByteBuffer buffer = channel.map(MapMode.READ_ONLY, 0, channel.size());
	int size = buffer.getShort();
	int mySize = 0;
	for(int i = 0; i < size; i++) {
		int id = buffer.getShort();
		if(id == -1) {
			continue;
		}
		ItemDefinition def = new ItemDefinition();
		def.id = id;
		def.stackable = buffer.get() == 1;
		def.noted = buffer.get() == 1;
		def.iventoryWeight = buffer.getDouble();
		def.wearingWeight = buffer.getDouble();
		int length = buffer.get();
		def.bonus = new int[length];
		for(int j = 0; j < length; j++) {
			def.bonus[j] = buffer.getShort();
		}
		def.name = readString(buffer);
		def.examine = readString(buffer);
		definitions[id] = def;
		mySize++;
	}
	Logger.getInstance().info("Loaded " + mySize + " item definitions.");
	channel.close();
	channel = null;
	buffer = null;
}

public static String readString(ByteBuffer buf) {
	StringBuilder sb = new StringBuilder();
	byte b;
	while((b = buf.get()) != 0 && buf.remaining() > 0) {
		sb.append((char) b);
	}
	return sb.toString();
}
