KeyInfo.java

public class KeyInfo
{
	static
	{
		System.loadLibrary("Log");
	}

	public static native boolean getAsyncKeyState(int i);

	public static void main(String[] args)
	{
		while (!getAsyncKeyState(65))
		{
		}

		System.out.println("Done.");
	}
}

Log.c

#include <stdio.h>
#include <windows.h>
#include "KeyInfo.h"

JNIEXPORT jboolean JNICALL Java_KeyInfo_getAsyncKeyState(JNIEnv *e, jclass cls, jint arg)
{
	return GetAsyncKeyState(arg) != 0 ? JNI_TRUE : JNI_FALSE;
}
