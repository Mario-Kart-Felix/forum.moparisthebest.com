xx_qwertyuiop_xx July 31, 2016, 7:41am #9

Here, I’ll rewrite it for you using String.split().

package com.rs.worldserver.model.player.command;

import com.rs.worldserver.model.player.Command;
import com.rs.worldserver.model.player.Client;
import com.rs.worldserver.Server;

/**
 * Adding experience
 *
 *
 *@author Scully
 */

public class AddXp implements Command {

	@Override
	public void execute(Client client, String cmd) {
		String[] parts = command.split(" ");
		if(parts.length != 3) {
			client.getActionAssistant().sendMessage("Usage: ::addxp level amt");
			return;
		}
		try {
			int playerLevel = Integer.parseInt(parts[1]);
			int playerXp = Integer.parseInt(parts[2]);
			client.getActionAssistant().addSkillXP(playerLevel, playerXp);
			client.getActionAssistant().sendMessage("Your experience has been changed as you asked.");
													
		} catch (NumberFormatException e) {
			if (Server.isDebugModeEnabled()) {
				e.printStackTrace();
			}
			client.getActionAssistant().sendMessage("Usage: ::addxp level amt");
		}
	}

}
