Function Modulation(range:Integer):integer;
Var
A, B, E, F, Modulated:Integer;
C, D:Extended;
Modded:Boolean;
Begin
Result := 0;
repeat
  A:=RanDom(Range)
  B:=RanDom(Range)
  C:=Sqrt(A)+(Random(Range))
  D:=Sqrt(b)+(Random(Range))
  E:=Random(Trunc(C))
  F:=Random(Trunc(D))
If(not SameValue(E,F))then
Begin
If (E+F<=Range) then
  Modulated:=(E+F)
  Modded:=true
End;
until(Modded=True)
  Result := modulated;
End;

I found this kind of interesting, too. I decided to test out the chances of the returned number from Random() being odd or even. Here it is. Strangely enough, I haven’t got one result where there were more odd numbers than even. In fact, the lowest average of even numbers I’ve gotten was 67.32.

Use this to test it out, just change the number within OddOrEven() in the main loop.

[SCAR]Var
Numbers: Array [1…100] of Integer;
i: Integer;

Procedure GetRandomNumbers;
Begin
For i:=1 To 100 Do
Begin
Numbers[i]:=Random(100);
End;
End;

Procedure OddOrEven(Times: Integer);
Var
Odd, Even, TimesDone: Integer;
AvgOdd, AvgEven: Extended;

Begin
If Times<=0 Then Times:=1;
Repeat
GetRandomNumbers;
For i:=1 To 100 Do
Begin
If Numbers[i]/2 = Round(Numbers[i]/2) Then
Begin
Even:=Even+1;
End Else Odd:=Odd+1;
End;
TimesDone:=TimesDone+1;
Until(TimesDone=Times);
AvgEven := Even/Times;
AvgOdd := Odd/Times;
If Even > Odd Then
Begin
WriteLn(‘Majority of numbers are: Even.’);
End Else If Odd > Even Then
Begin
WriteLn(‘Majority of numbers are: Odd.’);
End Else If Odd = Even Then WriteLn(‘Half of numbers were Even and Odd.’);
WriteLn(’’);
WriteLn(‘There was a total of: ‘);
WriteLn(’ ‘+IntToStr(Even)+’ even numbers.’);
WriteLn(’ ‘+IntToStr(Odd)+’ odd numbers.’);
WriteLn(’’);
WriteLn(‘There was an average of: ‘);
WriteLn(’ ‘+FloatToStr(AvgEven)+’ even numbers.’);
WriteLn(’ ‘+FloatToStr(AvgOdd)+’ odd numbers.’);
End;

Begin
OddOrEven(100);
End.[/SCAR]
