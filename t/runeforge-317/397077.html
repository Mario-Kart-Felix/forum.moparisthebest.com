package net.forge.entity.player;

/**
 */
public class PlayerSettings {
  
    /**
     * Public  - First 4 bits
     * Private - Second 4 bits
     * Requests - Third 4 bits
     */
    public enum CHAT_TYPES {
        PUBLIC, PRIVATE, REQUESTS
    }
    
    /**
     * 
     */
    public enum CHAT_STATES {
        ON, OFF, FRIENDS, HIDE
    }
    
    /**
     * The local chat settings which include what messages the player would
     * like to receive and not receive. Default is always [true, true, true]
     */
    short chatsettings = 0;
    
    /**
     * 
     */
    static final int[] REGEX_MASKS = {
        (1 << 5) - 1, ((1 << 9) - 1) - (1 << 5) - 1, ((1 << 13) - 1) - ((1 << 9) - 1)
    };
    
    /**
     * 
     * @param type
     * @param state 
     */
    public void setChatSetting(CHAT_TYPES type, CHAT_STATES state) {
        int regex = type == CHAT_TYPES.PUBLIC ? 0 : (type == CHAT_TYPES.PRIVATE ? 1 : 2);
        int mod = state == CHAT_STATES.ON ? 0 : (state == CHAT_STATES.OFF ? 2 : (state == CHAT_STATES.FRIENDS ? 1 : 3));
        /* Remove regex settings */
        chatsettings &= ~REGEX_MASKS[regex];
        chatsettings |= 1 << ((regex * 4) + mod);
    }
    
    /**
     * 
     * @param type
     * @return 
     */
    public int getChatSetting(CHAT_TYPES type) {
         short setting = chatsettings;
         int regex = type == CHAT_TYPES.PUBLIC ? 0 : (type == CHAT_TYPES.PRIVATE ? 1 : 2);
         setting &= REGEX_MASKS[regex];
         byte b = (byte) (setting >> (regex * 4));
         int bits = 0;
         for(; bits < 4; bits++) {
             b >>= 1;
             if(b == 0)
                 break;
         }
         return bits;
    }
}
