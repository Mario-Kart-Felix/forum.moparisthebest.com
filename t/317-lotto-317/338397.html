// - LottoFixed 1.0 by `ZeroScape - a fixed version of djdoyle1’s Lotto-system. -
// - Remember to give the credits to djdoyle1 for his ideas. Have fun =)! -

public int lottoPrice = 100000;
public int timeSinceLotto;
public int played;
public int serverOnlineTime;
public int millisecondsToSeconds;
public boolean printBatch = false;
public int[] possibleRewards = { 5698, 4151, 1042, 1375, 995, 995, 995, 995 };

public void lotto() {
	if (playerHasItemAmount(995, lottoPrice) == true) {
		if (timeSinceLotto == 0) {
			lottoPlayed = true;
			deleteItem(995, getItemSlot(995), lottoPrice);
			int number1 = misc.random(2);
			int number2 = misc.random(2);
			sM("Your number is: " + number1 + ", the computers number is: " + number2 + "!");
			if (number1 == number2) {
				if (printBatch == true) {
					System.out.println(playerName + " played lotto, and won!");
				}
				sM("Congratulations! You won the lotto! Your reward is..");
				int rewardID = possibleRewards[(int) (Math.random()*possibleRewards.length)];
				int rewardAmount = 1;
				if (rewardID == 995) {
					int GpRandomize = misc.random(4);
					if (GpRandomize == 0) {
						rewardAmount = 1000000;
					}
					if (GpRandomize == 1) {
						rewardAmount = 2000000;
					}		
					if (GpRandomize == 2) {
						rewardAmount = 3000000;
					}
					if (GpRandomize == 3) {
						rewardAmount = 4000000;
					}
					if (GpRandomize == 4) {
						rewardAmount = 5000000;
					}		
				}
				addItem(rewardID, rewardAmount);
				sM("" + rewardAmount + "x " + getItemName(rewardID) + "!");
			} else {
				sM("Awwww! You lost the game. Better luck next time!");
			}
		} else {
			sM("You've already played lotto in the last 12 hours!");
		}
	} else {
		sM("Sorry! You must have at least 100,000gp to play lotto!");
	}
