Or you could use this:

[SCAR]{----------Inscructions--------
| First, setup the 2 Consts |
| on lines 25 & 26. Then |
| follow these steps: |
| |
| 1. Only have 1 pick. |
| |
| 2. Start by the ess to mine. |
| |
| 3. Position your mouse over |
| a high part of the ess, |
| so nothing will move in |
| the way of it. |
| |
| 4. Press F8 To start. |
|==============================|
| This script is a free script |
| by Weibs. |
------------------------------}
Program New;
{.include SRL/SRL.scar}
{.include SRl/SRL/Skill/Mining.scar}

Const
LoadsToDo = 100; //# of Loads To Do.
PickEquipped = True; //Is the pickaxe equipped?

Var
EssX, EssY, WX, WY, Loads: Integer;

Procedure WaitToStart;
Begin
Repeat
Wait(100);
Until(IsFKeyDown(8));
End;

Procedure StartScript;
Begin
If (Not (IsUpText(‘ine’))) Then
WriteLn(‘Mouse was not over the ess.’);

GetMousePos(EssX, EssY);
WX := EssX;
WY := EssY;
If (Not (FindPickHeadColors)) Then
TerminateScript;

GameTab(4);

If InvFull Then
Begin
If PickEquipped Then
Begin
DropTo(1, 28);
End Else
DropTo(2, 28);
End;
End;

Procedure DropEss;
Begin
If PickEquipped Then
Begin
DropTo(1, 28);
End Else
DropTo(2, 28);

Loads := Loads + 1;
End;

Procedure PowerMine;
Var
i: Integer;

Begin
GameTab(4);
MMouse(EssX, EssY, 3, 3);
If IsUpText(‘ine’) Then
Begin
GetMousePos(WX, WY);
Mouse(WX, WY, 0, 0, true);
End Else
Begin
MMouse(1 + Random(515), 1 + Random(338), 5, 5);
End;

Repeat
Wait(1000+Random(1000));
i := i + 1;
If i = 15 Then
Begin
MMouse(1 + Random(515), 1 + Random(338), 5, 5);
i := 0;
End;
Until(InvFull);
End;

Procedure SetupScript;
Begin
SetupSRL;
SetupSRLMining;
End;

Begin
SetupScript;
WaitToStart;
StartScript;

Repeat
PowerMine;
DropEss;

If Loads = LoadsToDo Then
Begin
  LogOut;
  Break;
End;

If (Not (Loggedin)) Then
  Break;

Until(False);
End.[/SCAR]
