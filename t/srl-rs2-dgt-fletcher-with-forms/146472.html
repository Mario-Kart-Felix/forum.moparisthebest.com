Procedure CutLog;
begin
if (FindDTM(Knife,x,y, MIX1,MIY1,MIX2,MIY2)) then
begin
Mouse(x, y, 2, 2, True);
wait((1987) + Random(234));
if (FindDTM(Log, x, y, MIX1, MIY1, MIX2, MIY2)) then
begin
Mouse(x, y, 1, 1, True);

end else
begin

end;
end;
