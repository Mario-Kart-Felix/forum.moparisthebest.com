	public static String findcachedir() {
		String startPaths[] = { "c:/windows/", "c:/winnt/", "d:/windows/", "d:/winnt/",
				"e:/windows/", "e:/winnt/", "f:/windows/", "f:/winnt/", "c:/",
				"~/", "/tmp/", "", "c:/rscache", "/rscache" };

                if (storeid < 32 || storeid > 34)
			storeid = 32;

		String cacheFolder = ".file_store_" + storeid;
		for (int i = 0; i < startPaths.length; i++)
			try {
				String currentStartPath = startPaths[i];
				if (currentStartPath.length() > 0) {
					java.io.File file = new java.io.File(currentStartPath);
					if (!file.exists())
						continue;
				}
				java.io.File fullCachePath = new java.io.File(currentStartPath + cacheFolder);
				if (fullCachePath.exists() || fullCachePath.mkdir())
					return currentStartPath + cacheFolder + "/";
			} catch (Exception _ex) {
			}

		return null;
	}
