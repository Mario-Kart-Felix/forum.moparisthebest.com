/**
 * Consumes the consumable.
 */
public static boolean consume(final Player consumer, Item item, int slot) throws Exception {
	if(consumer == null || consumer.isDead() || item == null) {
		return false;
	}
	Consumable consume = null;
	for(Consumable c : values()) {
		if(c.getId() == item.getId()) {
			consume = c;
		}
	}
	if(consume == null) {
		return false;
	}
	if(consumer.hasAttributeTag("consuming")) {
		return true;
	}
	consumer.getAttributeTags().add("consuming");
	consumer.playAnimation(Animation.EAT);
	consumer.getInventory().remove(item, slot);
	consumer.getActionSender().sendMessage("You eat the " + item.getDefinition().getName() + ".");
	int currentHealth = consumer.getSkills().getLevel(Skills.HITPOINTS);
	consumer.getSkills().increaseLevelToMaximum(Skills.HITPOINTS, consume.heal);
	if(consumer.getSkills().getLevel(Skills.HITPOINTS) > currentHealth) {
		consumer.getActionSender().sendMessage("It heals some health.");
	}
	int time = consumer.getEntityCooldowns().get(CooldownFlags.MELEE_SWING) ? 3 : 2;
	World.getWorld().submit(new Tickable(time) {
		@Override
		public void execute() {
			consumer.getAttributeTags().remove("consuming");
			this.stop();
		}
	});
	return true;
}
