Clawz_fury August 2, 2016, 3:02pm #25

Would this List, not accept nulls?

[code=java]import java.util.concurrent.atomic.AtomicInteger;
import java.util.concurrent.atomic.AtomicReferenceArray;

public class RS2List {

private AtomicReferenceArray<T> list;
private int maxIndex;
private AtomicInteger pointerIndex;

public RS2List(int size) {
	maxIndex = size + 1;
	pointerIndex = new AtomicInteger(1);
	list = new AtomicReferenceArray<T>(maxIndex);
}

public T get(int index) {
	return list.get(index);
}

public int length() {
	return list.length();
}

public void set(int index, T object) {
	list.set(index, object);
}

public void remove(int index) {
	list.set(index, null);
	pointerIndex.set(index);
}

public int returnIndex() {
	int pointedIndex = pointerIndex.get();
	if (get(pointedIndex) == null)
		return pointedIndex;
}

}[/code]
