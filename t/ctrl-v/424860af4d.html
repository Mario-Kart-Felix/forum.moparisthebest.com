/**
 * Rotates a vertex around the z axis.
 * @param vertex The vertex to preform the rotation on.
 * @param trigindex The trigonometry angle table index. 
 * @param ox The origin x coordinate.
 * @param oy The origin y coordinate.
 */
private static void zAxisRotation(Vertex vertex, int trigindex, int ox, int oy) {
    int sine = TriangleRasterizer.SINETABLE[trigindex];
    int cosine = TriangleRasterizer.COSINETABLE[trigindex];
    int x = vertex.x - ox;
    int y = vertex.y - oy;
    vertex.x = (x * cosine - y * sine >> 16) + ox;
    vertex.y = (x * sine + y * cosine >> 16) + oy;
}
