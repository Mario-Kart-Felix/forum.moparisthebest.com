} else

			if(misc.random(3) == 2) {
			currentHealth += 2;
			} else

			if(misc.random(3) == 3) {
			currentHealth += 1;
			}

			if (currentHealth > playerLevel[playerHitpoints])
				currentHealth = playerLevel[playerHitpoints];
			sendMessage("You eat the shrimp.");
		break;[/code]
