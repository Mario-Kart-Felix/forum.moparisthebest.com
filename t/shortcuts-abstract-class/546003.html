package com.rs.game.player.content.agility;

import com.rs.game.Animation;
import com.rs.game.WorldTile;
import com.rs.game.item.Item;
import com.rs.game.player.Player;
import com.rs.game.player.Skills;
import com.rs.game.player.quests.Quest;
import com.rs.game.tasks.WorldTask;
import com.rs.game.tasks.WorldTasksManager;

/**

  • Handles a shortcut.

  • @author Arham Siddiqui
    */
    public abstract class Shortcut {

    /**

    • The type of shortcut.
      */
      public enum ShortcutType {
      ;

      private Animation animation;

      ShortcutType(Animation animation) {
      this.animation = animation;
      }

      public Animation getAnimation() {
      return animation;
      }
      }

    /**

    • Does the shortcut addressed to do.
    • @param shortcut The shortcut addressed to do.
    • @param player The player who is doing the shortcut.
      */
      public void doShortcut(final Shortcut shortcut, final Player player) {
      if (player.getSkills().getLevel(Skills.AGILITY) < shortcut.getLevelRequired()) {
      player.getPackets().sendGameMessage(“You need an Agility level of at least " + shortcut.getLevelRequired() + " to use this obstacle.”);
      return;
      }
      WorldTasksManager.schedule(new WorldTask() {
      @Override
      public void run() {
      player.setNextAnimation(shortcut.getShortcutType().getAnimation());
      this.stop();
      }
      }, 1);
      }

    public abstract ShortcutType getShortcutType();

    /**

    • Gets the destination.
    • @return The destination.
      */
      public abstract WorldTile getDestination();

    /**

    • Gets the level required.
    • @return The level required.
      */
      public abstract int getLevelRequired();

    /**

    • Gets the Items required to do the shortcut.
    • @return The required items.
      */
      public Item[] getItemRequirements() {
      return null;
      }

    /**

    • Gets the Skill requirements to do the shortcut.
    • @return The skill requirements.
      */
      public int[] getSkillRequirements() {
      return null;
      }

    /**

    • Gets the Quests required to do the shortcut.
    • @return The required quests.
      */
      public Quest[] getQuestRequirements() {
      return null;
      }
