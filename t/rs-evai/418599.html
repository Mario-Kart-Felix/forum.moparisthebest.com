Scripting in Java?

Example (Welcome.java):

import com.rsevai.game.event.Event;
import com.rsevai.game.event.player.PlayerJoinEvent;
import com.rsevai.game.event.player.PlayerListener;
import com.rsevai.game.event.player.PlayerQuitEvent;
import com.rsevai.game.script.Script;

public class Welcome extends Script implements PlayerListener {
	
	@Override
	public void onEnable() {
		this.getServer().getEventManager().register(this, Event.Type.PLAYER_JOIN);
		this.getServer().getEventManager().register(this, Event.Type.PLAYER_QUIT);
		System.out.println("[Welcome] started.");
	}

	@Override
	public void onDisable() {
		System.out.println("[Welcome] stopped.");
	}

	public void onPlayerJoin(PlayerJoinEvent event) {
		System.out.println(event.getPlayer().getUsername() + " has joined!");
		event.getPlayer().getActionSender().sendMessage("Welcome!");
	}

	public void onPlayerQuit(PlayerQuitEvent event) {
		System.out.println(event.getPlayer().getUsername() + " has quit! What a shame!");	
	}

}

I’ll implement other script loaders later, but at the moment, java is the simplest.
