the bytecode is pretty straight forward

char[] cs = str.toCharArray();
int n = cs.length;
int i = 0;
while (i < n) {
    char c = cs[i];
    switch (i % n) {
        case 0:
            c = c ^ KEY0;
        case 1:
            c = c ^ KEY1;
        case 2:
            c = c ^ KEY2;
        case 3:
            c = c ^ KEY3;
        default:
            c = c ^ KEY4;
    }
    c[i] = c;
    ++i;
}
