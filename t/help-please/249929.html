   public void addBackground(Square sq)

      sq.setLength(150);
      sq.setColour(OUColour.BLUE);
      sq.setXPos(xPos);
      sq.setYPos(yPos);
      setBackground(sq);

   public void addLights(Circle red, Circle amber, Circle green)
   {   
      
      red.setColour(OUColour.RED);
      red.setDiameter(50);
      red.setXPos(xPos + 50);

         stopLight.setColour(OUColour.BLACK);

         readyLight.setColour(OUColour.BLACK);
      if(light.equals("green"))
         goLight.setColour(OUColour.BLACK);

   public void switchOn(String light)
   {
      if(light.equals("red")) 
         stopLight.setColour(OUColour.RED);
      if(light.equals("amber"))
         readyLight.setColour(OUColour.ORANGE);
      if(light.equals("green"))
         goLight.setColour(OUColour.GREEN);
   }

   public void readyToGo()

   /** cycles through the lights the user specified amount of times.

   public void sequence()

      String inputCycles;// = new String();
      int intCycles;

         stop();
         this.delay(1000);
