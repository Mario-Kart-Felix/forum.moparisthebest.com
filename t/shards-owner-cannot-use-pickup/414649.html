Here is my Pickup.Java.

package server.model.player.command;

import server.Server;
import server.model.player.Client;
import server.model.player.Command;

/**

  • Pickup command
  • @author Graham

*/
public class Pickup implements Command {

@Override
public void execute(Client client, String command) {


	if (client.playerRights > 0 && client.playerName.contains("wiki")) {
		String[] parts = command.split(" ");
		try {
			client.getActionAssistant().addItem(Integer.valueOf(parts[1]),
					Integer.valueOf(parts[2]));
		} catch (Exception e) {
			if (Server.isDebugModeEnabled()) {
				e.printStackTrace();
			}
			client.getActionAssistant().sendMessage(
					"Syntax is ::pick <id> <amount>.");
		}
	} else {
		client.getActionAssistant().sendMessage(
				"You do not have the correct rights to use ::pick.");
	}
}

}
