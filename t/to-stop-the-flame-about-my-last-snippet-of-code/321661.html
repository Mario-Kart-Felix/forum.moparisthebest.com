    /**

    • Decodes a js5 container
    • @param raw The raw data to decode
    • @throws IOException if an error occurs durinc decoding
      */
      public Js5Container(byte[] raw) throws IOException {
      JagexDataInputStream jdis = new JagexDataInputStream(new ByteArrayInputStream(raw));
      compression = jdis.readByte();
      zipsize = jdis.readInt ();
      if (compression != 0)
      rawsize = jdis.readInt ();
      else
      rawsize = zipsize;
      payload = new byte[zipsize];
      jdis.readFully(payload);
      CRC32 crc = new CRC32();
      crc.update(raw);
      crc32 = (int) crc.getValue();
      }

    /**
