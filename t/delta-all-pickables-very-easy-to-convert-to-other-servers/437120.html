In a new class called Pickables:

public class Pickables {
	
	Client c;
	
	public Pickables (Client c) {
		this.c = c;
	}
	
	private boolean isPicking = false;
	private boolean finishedPicking = true;
	private int getPickable;
	private int pickableX;
	private int pickableY;
	
	public int[] pickableIds = {1161, 313, 5583, 5584, 5585, 312, 3366, 2646};
	
	
	public void handlePickables (int pickableId, int coordX, int coordY) {
	
		if ((c.GoodDistance(coordX, coordY, c.absX, c.absY, 1) == true)) {
			getPickable = pickableId;
			pickableX = coordX;
			pickableY = coordY;
			
			switch(pickableId) {
			case 1161: //Cabbage
				startPicking(1965);
				break;
			case 313: //wheat
			case 5583:
			case 5584:
			case 5585:
				startPicking(1947);
				break;
			case 312: //Potato
				startPicking(1942);
				break;
			case 3366: //Onion
				startPicking(1957);
				break;
			case 2646: //Flax
				startPicking(1779);
				break;
			}
		} else {
			return;
		}
	}
	
	private void startPicking(int newItem) {
		if (finishedPicking == true) {
			isPicking = true;
			finishedPicking = false;
			c.startAnimation(827);
			pickingProcess(newItem);
		}
	}
	
	private void pickingProcess (final int newItem) {
		if (isPicking) {
			server.getTaskScheduler().schedule(new Task(1, false) {
				@Override
				protected void execute() {
					c.addItem(newItem, 1);
					c.ReplaceObject(pickableX, pickableY, -1, 0, 10);
					finishPicking(pickableX, pickableY, getPickable);
					finishedPicking = true;
					isPicking = false;
					c.resetAnimation();
					stop();
				}
			});
		}
	}
	
	private void finishPicking(final int xSpot, final int ySpot, final int pickableId) {
		server.getTaskScheduler().schedule(new Task(20, false) {
			@Override
			protected void execute() {
				c.ReplaceObject(xSpot, ySpot, pickableId, 0, 10);
				stop();
			}
		});
	}	
}
