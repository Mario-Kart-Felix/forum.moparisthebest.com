I created a music system a while back, it was decent but wasn’t that great, therefore, I decided to create a new one; quite similar. This is unfinished, but it’s near finished and contains like more then half the songs, just add the rest (it’s easy and user-friendly).
This system loads song(s) externally via a text document, yes not an XML file. Thought I’d go with something different here. I’m going to load regions via XML.

LoadMusicData:

[code]package neitiznot.model.content.music;

import java.io.BufferedReader;
import java.io.FileReader;

/**
*

  • @author Rhubarb/Int Bauk

*/
public class LoadMusicData {

/**
 * Gets the song.
 * @param location
 * @return -1
 */
public int getSong(String location) {
	String fileName = "c:\\musicData.txt";
	try {
		 FileReader inputFile = new FileReader(fileName);
		 BufferedReader bufferReader = new BufferedReader(inputFile);
		 while ((location = bufferReader.readLine()) != null) {
			 System.out.println(location);
	}
		 bufferReader.close();
} catch(Exception e) {
	System.out.println("Error while reading music data file: " + e.getMessage());

}
return -1;
}
}

MusicHandler.java:

[code]package neitiznot.model.content.music;

import neitiznot.model.content.music.WorldManager.Locations;

/**
*

  • @author Rhubarb/Int Bauk

*/
public class MusicHandler {

/**
 * The current song id.
 */
private int currentSong = 0;
/**
 * Current areas available.
 */
private String currentAreas = "";

/**
 * LoadMusicData class: reference(s) MusicHandler class.
 */
LoadMusicData loadMusicData = new LoadMusicData();

/**
 * Gets the current song from the enum Locations locations.
 */
final WorldManager.Locations locations = Locations.forId(currentSong);

/**
 * Sets the song id.
 */
private void setSong() {
	currentSong = loadMusicData.getSong(currentAreas);
}

/**
 * Sets the song id to a specific area.
 */
private void setSongArea() {
	locations.getSongId();
}

/**
 * Handles the music.
 */
public void handleMusic() {
	setSong();
	setSongArea();
}

}

	TUT,
	CW,
	BARROWS_UNDER,
	BARROWS_TOP,
	WILDERNESS,
	ARDY,
	EDGEVILLE,
	WIZARD_TOWER,

	FALADOR,
	FIGHT_CAVES,
	PIRATES_HOUSE,
	BARBARIAN_VILLAGE,
	THE_ABYSS,
	APE_ATOLL,
	BANDIT_CAMP,
	BARBARIAN_AGILITY,
	BEDABIN_CAMP,
	CRASH_ISLAND,
	CATHERBY,
	WHITE_WOLF_MOUNTAIN,
	BURTHROPE,
	CAMELOT,
	CANFIS;

	int songId;
	
	/**
	 * Gets the song id.
	 * @return songId;
	 */
	public int getSongId() {
		return songId;
	}
	
	/**
	 * Sets the location of the songs in order (ordinal()).
	 * @param songId
	 * @return null;
	 */
	public static Locations forId(int songId) {
		for(Locations locations : Locations.values()) {
			if(locations.ordinal() == songId) {
				return locations;
			}
		}
		return null;
		
	}
}

}
