Here’s a simple example

import java.awt.Dimension;
import java.awt.FlowLayout;

import java.util.Random;

import javax.swing.JFrame;
import javax.swing.JProgressBar;

public final class Progress extends Thread {

  private JFrame frame;

  private JProgressBar bar;

  public Progress() {

    frame = new JFrame("Please wait...");
    frame.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
    frame.setResizable(false);

    frame.setLayout(new FlowLayout(FlowLayout.CENTER, 25, 25));

    bar = new JProgressBar(JProgressBar.HORIZONTAL, 0, 100);
    bar.setPreferredSize(new Dimension(640, 200));
    bar.setStringPainted(true);
    frame.add(bar);

    frame.pack();
    frame.setVisible(true);

    Thread loading = this;
    loading.start();

  }

  public final void run() {

    try {

      StringBuilder str = new StringBuilder(4);
      Random rand = new Random();

      for(int value = 0; value <= 100; value++) {

        bar.setValue(value);
        str.append(value);
        str.append('%');
        bar.setString(str.toString());
        str.setLength(0);

        sleep(rand.nextInt(75));
      }

      frame.setTitle("All done!");
      bar.setString("Finished, exiting...");
      sleep(1000);

    } catch(InterruptedException ex) {
    }

    frame.dispose();

  }

}
