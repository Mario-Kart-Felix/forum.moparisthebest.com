[code]package org.row;

public class Row {

private static int[] row = { 10, 45, 3, 4, 6, 5, 6, 10, 49, 493, 492, 59, 10};
public static void main(String[] args) {
	
	int n = row.length;
	for(int a = 0; a < n-1; a++) {
		for(int b = a+1; b < n; b++) {
			if(row[a] < row[b]) {
				int i = row[a];
				row[a] = row[b];
				row[b] = i;
			}
		}
	}
	
	for(int i = 0; i < n; i++) {
		System.out.print( row[i] + "; ");
	}
}

}
