But try not to use SendKeys as it will give you a decent ban. Instead, try using keydowns and keyups or have those in a separate procedure to save space, like so:[code=geshi]// --> TrainerScape/Easter guy1’s AutoTalker!
// --> Version 1.0, Please don’t flame. its my first script.
// --> Join Trainerscape.smfforfree4.com To See More Of My Scripts
// --> Release date: Saturday, March 15, 2008
// --> READ INSTRUCTIONS.
//–> this SCAR Script is an auto talker
// --> HOW TO USE SCRIPT: The point is so that it makes mistakes. Where
//–>it says “Good Spelling.” Put how you actually want it spelt. if it says
//–>“Mistake” make a mistake in the typing. that way, people know your
//–>actually typing. (But your not :] ) Example, i was buying full sara, so
//–> I made mistakes.

program script;

const
Message1=‘Buying full sara 1.2M’; //Good Spelling.
Message2=‘Buying full sara 1.2M’; //Good spelling.
Message3=‘Buyign full sara 1.2M’; //Mistake.
Message4=‘Buying full sara 1.2M’; // Good spelling.
Message5=‘Buyin full saar 1.2M’; // Mistake
Message6=‘Buyign full sara 1.2M’; // Mistake
Message7=‘Buying full sara 1.2M’; // Good spelling.
Message8=‘Buying full sara 1.2M’; // Good spelling.

procedure letter(let: integer);
begin
KeyDown(Let);
Wait(20+Random(21));
KeyUp(Let);
Wait(80+Random(91));
end;

procedure Typing(Text: String);
var
i: Integer;
begin
for i:=1 to length(Text) do
letter(GetKeyCode(Text[i]));
KeyDown(13) //will press enter/return.
Wait(100+Random(51));
KeyUp(13);
end;

procedure Talker;
begin
wait(1000) //1000 = 1 second, 2000 = 2 second, ect.
Typing(Message1); // Tells scar to type the message.
Typing(Message2);
Typing(Message3);
Typing(Message4);
Typing(Message5);
Typing(Message6);
Typing(Message7);
Typing(Message8);
end;

begin
repeat
Talker;
until(false) //Delete the (false) if you want it to repeat it once.
end.[/code]
