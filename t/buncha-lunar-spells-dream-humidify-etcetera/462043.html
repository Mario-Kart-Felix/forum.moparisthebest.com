[code]package server.model.players.content.lunarspells;

import server.Config;

public class LunarSpells {

/**
 * Loads the spell information - use as:
 * 
 * <LunarSpells.loadInformation(c, actionButtonid);>
 * 
 * @param c
 * @param actionButtonId
 */
public static void loadInformation(Client c, int actionButtonId) {
	castSuperglassMake(c);
	castDream(c);
	castStringJewellery(c);
	castSpellbookSwap(c);
	castPlankMake(c, actionButtonId);
	castHunterKit(c);
	castHumidify(c);	
}

private static final int GLASS_REQUIRED_LEVEL = 77;
private static final int STRING_REQUIRED_LEVEL = 80;
private static final int SPELLBOOK_REQUIRED_LEVEL = 96;
private static final int PLANK_REQUIRED_LEVEL = 86;
private static final int HUMIDIFY_REQUIRED_LEVEL = 68;
private static final int REQUIRED_COMBAT = 85;
private static final int DREAM_REQUIRED_LEVEL = 79;
private static final int HUNTERKIT_REQUIRED_LEVEL = 71;


private static final int HUNTER_KIT = 11159;
public static final int HEAL = 10;
public static final int MAGIC_XP = Config.MAGIC_EXP_RATE;

public static final int ASTRAL = 9075, 
		COSMIC = 564, BODY = 559, 
		FIRE = 554, WATER = 555, 
		NATURE = 561, EARTH = 557, 
		LAW = 563, AIR = 556;

private static final int BUCKET_OF_SAND = 1783, 
		SODA_ASH = 1781,
		SEAWEED = 401,
		MOLTEN_GLASS = 1775;

private static final int[] UNSTRUNG = { 
	1673, 1675, 1677, 1679, 1681, 1683 
	};
private static final int[] STRUNG = { 
	1692, 1694, 1696, 1698, 1700, 1702 
	};
private static final int[] EMPTY = { 
	229, 1831, 1829, 1827, 1825, 1925,
	1923, 1935, 5331, 5332, 5333, 5334, 5335, 5337, 5338, 5339, 6667,
	7688, 731, 1980, 434
	};
private static final int[] FULL = { 
	227, 1823, 1823, 1823, 1823, 1929,
	1921, 1937, 5340, 5340, 5340, 5340, 5340, 5340, 5340, 5340, 6668,
	7690, 732, 4458, 1761 
	};
private static final int[] HUNTER_ITEMS = { 10150, 10010, 10006, 10031,
	10029, 596, 10008 
	};
private static final int[][] CONVERT = { 
	{ 1511, 960 },
	{ 1521, 8778 }, 
	{ 6333, 8770 },
	{ 6332, 8782 }, 
	};

/**
 * Checks the log.
 * @param item
 * @return
 */
private static boolean correctLog(int item) {
	return item == 1511 || item == 1521|| item == 6333
			|| item == 6332;
	}

/**
 * Deletes runes.
 * @param c
 * @param runes
 * @param amount
 */
public static void deleteRunes(Client c, int[] runes, int[] amount) {
	if(c.playerRights == 0) {
		for(int i = 0; i < runes.length; i++) {
			c.getItems().deleteItem(runes[i], c.getItems().getItemSlot(runes[i]), amount[i]);
		}
	}
}

/**
 * Opens the hunter kit.
 * @param c
 */
public static void openHunterKit(Client c) {
	if (c.getItems().playerHasItem(HUNTER_KIT, 1)) {
		if(c.getItems().freeSlots() > 6) {
			for (int element : HUNTER_ITEMS) {
				c.getItems().addItem(element, 1);
				c.getItems().deleteItem(HUNTER_KIT, c.getItems().getItemSlot(HUNTER_KIT), 1);
			}
		} else {
			c.sendMessage("You need at least seven inventory spaces to open this.");
		}
	}
}

/**
 * Casts the dream spell.
 * @param c
 */
public static void castDream(Client c) {
	if (c.getLastSpell() < 1800) {
		return;
	}
	if(c.playerLevel[6] < 79) {
		c.sendMessage("You don't have the required magic level to cast this spell!");
		c.sendMessage("You need a magic level of " + DREAM_REQUIRED_LEVEL + " to cast this spell.");
		return;
	}
	if(c.getItems().freeSlots() > 0) {
	if(c.getItems().playerHasItem(ASTRAL, 2) && 
			c.getItems().playerHasItem(COSMIC, 1) && 
			c.getItems().playerHasItem(BODY, 5)) {
		
		c.getItems().deleteItem(ASTRAL, c.getItems().getItemSlot(ASTRAL), 2);
		c.getItems().deleteItem(COSMIC, c.getItems().getItemSlot(COSMIC), 1);
		c.getItems().deleteItem(BODY, c.getItems().getItemSlot(BODY), 15);
		c.forcedText = "Zzzzz";
		c.startAnimation(6295);
		if (c.playerLevel[3] < c.getLevelForXP(c.playerXP[3])) {
			c.playerLevel[3] += HEAL;
			if (c.playerLevel[3] > c.getLevelForXP(c.playerXP[3]))
				c.playerLevel[3] = c.getLevelForXP(c.playerXP[3]);
		c.getPA().addSkillXP(1 * MAGIC_XP, c.playerMagic);
	} else
		c.sendMessage("You don't have the required runes to cast this spell.");
	} else {
		c.sendMessage("You don't have enough inventory space to cast this spell.");
	}
	c.setSpellDelay();
	}
}

/**
 * Casts the humidify spell.
 * @param c
 */
public static void castHumidify(Client c) {
	if (c.getLastSpell() < 3500) {
		return;
	}
	if(c.playerLevel[6] < HUMIDIFY_REQUIRED_LEVEL) {
		c.sendMessage("You don't have the required magic level to cast this spell!");
		c.sendMessage("You need a magic level of " + HUMIDIFY_REQUIRED_LEVEL + " to cast this spell.");
		return;
	}
	if (c.combatLevel < REQUIRED_COMBAT) {
		c.sendMessage("You don't have the required combat level to cast this spell!");
		c.sendMessage("You need a combat level of " + REQUIRED_COMBAT + " to cast this spell.");
		return;
	}
	boolean can = false;
	for (int element : EMPTY) {
		if (c.getItems().playerHasItem(element)) {
			can = true;
		}
	}
	if (can) {
			if (!c.getItems().playerHasItem(ASTRAL, 1)
					|| !c.getItems().playerHasItem(WATER, 3)
					|| !c.getItems().playerHasItem(FIRE, 1)) {
			return;
		}
		c.getPA().addSkillXP(1 * MAGIC_XP, c.playerMagic);
		c.gfx0(1061);
		c.startAnimation(6294);
		
		try {
			for (int i = 0; i < 28; i++) {
				if (c.getItems().playerHasItem(EMPTY[i])) {
					c.getItems().deleteItem(EMPTY[i], c.getItems().getItemSlot(EMPTY[i]),1);
					c.getItems().addItem((FULL[i]), i);
				}
			}
		} catch (Exception e) {

		}
	} else {
		c.sendMessage("You don't have any empty vessesls to fill.");
	}
	c.setSpellDelay();
}

/**
 * Casts the hunter kit spell.
 * @param c
 */
public static void castHunterKit(Client c) {
	if (c.getLastSpell() < 1800) {
		return;
	}
	if(c.playerLevel[6] < 71) {
		c.sendMessage("You don't have the required magic level to cast this spell!");
		c.sendMessage("You need a magic level of " + HUNTERKIT_REQUIRED_LEVEL + " to cast this spell.");
		return;
	}
	if(c.getItems().freeSlots() > 0) {
	if(c.getItems().playerHasItem(ASTRAL, 2) &&
				c.getItems().playerHasItem(NATURE, 1) && 
				c.getItems().playerHasItem(EARTH, 15)) {
		
		c.getItems().deleteItem(ASTRAL, c.getItems().getItemSlot(ASTRAL), 2);
		c.getItems().deleteItem(NATURE, c.getItems().getItemSlot(NATURE), 1);
		c.getItems().deleteItem(EARTH, c.getItems().getItemSlot(EARTH), 15);
		
		c.startAnimation(6303);
		c.gfx0(1074);
		c.getPA().addSkillXP(1 * MAGIC_XP, c.playerMagic);
		c.getItems().addItem((HUNTER_KIT), 1);
	} else
		c.sendMessage("You don't have the required runes to cast this spell.");
	} else {
		c.sendMessage("You don't have enough inventory space to cast this spell.");
	}
	c.setSpellDelay();
}

/**
 * Casts the plank make spell.
 * @param c
 * @param itemId
 */
public static void castPlankMake(final Client c, int itemId) {
	if (c.getLastSpell() < 3000) {
		return;
	}
	if(c.playerLevel[6] < 86) {
		c.sendMessage("You don't have the required magic level to cast this spell!");
		c.sendMessage("You need a magic level of " + PLANK_REQUIRED_LEVEL + " to cast this spell.");
		return;
	}
	if (!correctLog(itemId)) {
		c.sendMessage("You can't turn that into a plank!");
		return;
	}
	if (c.getItems().playerHasItem(ASTRAL, 2)
			&& c.getItems().playerHasItem(NATURE, 1)
			&& c.getItems().playerHasItem(EARTH, 15)) {
	c.startAnimation(6298);
	c.gfx100(1063);
	for (int[] element : CONVERT) {
		if (itemId == element[0]) {
			c.getItems().deleteItem(ASTRAL, c.getItems().getItemSlot(ASTRAL), 2);
			c.getItems().deleteItem(NATURE, c.getItems().getItemSlot(NATURE), 1);
			c.getItems().deleteItem(EARTH, c.getItems().getItemSlot(EARTH), 15);
			c.getItems().deleteItem(element[0], c.getItems().getItemSlot(element[0]), 1);
			c.getItems().addItem((element[1]),1);	
			c.sendMessage("You make the log into a plank!");
			c.getPA().addSkillXP(1000 * MAGIC_XP, c.playerMagic);
			c.getPA().sendFrame106(6);
		}
	}
} else {
	c.sendMessage("You don't have the required runes to cast this spell.");
	}
	c.setSpellDelay();
}

/**
 * Casts the spell book swap spell.
 * @param c
 */
public static void castSpellbookSwap(Client c) {
	if (c.getLastSpell() < 1800) {
		return;
	}
	if(c.playerLevel[6] < SPELLBOOK_REQUIRED_LEVEL) {
			c.sendMessage("You don't have the required magic level to cast this spell!");
			c.sendMessage("You need a magic level of " + SPELLBOOK_REQUIRED_LEVEL + " to cast this spell.");
		return;
	}
	if (!c.getItems().playerHasItem(ASTRAL, 3)
			|| !c.getItems().playerHasItem(COSMIC, 2)
			|| !c.getItems().playerHasItem(LAW, 1)) {
	return;
	}
	c.getPA().addSkillXP(1 * MAGIC_XP, c.playerMagic);
	c.setSpellSwap(true);
	c.startAnimation(6299);
	c.gfx0(1062);
	c.setSpellDelay();
}

/**
 * Casts the string jewellery spell.
 * @param c
 */
public static void castStringJewellery(Client c) {
	if (c.getLastSpell() < 1800) {
		return;
	}
	if(c.playerLevel[6] < 80) {
		c.sendMessage("You don't have the required magic level to cast this spell!");
		c.sendMessage("You need a magic level of " + STRING_REQUIRED_LEVEL + " to cast this spell.");
		return;
	}
	
	boolean can = false;
	for (int element : UNSTRUNG) {
		if (c.getItems().playerHasItem(element)) {
			can = true;
		}
	}
	if (!can) {
			c.sendMessage("You don't have any jewelry to string.");
		return;
	}
		
		if(c.getItems().playerHasItem(ASTRAL, 2) &&
				c.getItems().playerHasItem(WATER, 5) && 
				c.getItems().playerHasItem(EARTH, 10)) {
		
		c.getPA().addSkillXP(1 * MAGIC_XP, c.playerMagic);
		c.startAnimation(4412);
		c.gfx100(728);
		try {
			for (int i = 0; i < 28; i++) {
			c.getItems().deleteItem(ASTRAL, c.getItems().getItemSlot(ASTRAL), 2);
			c.getItems().deleteItem(WATER, c.getItems().getItemSlot(WATER), 5);
			c.getItems().deleteItem(EARTH, c.getItems().getItemSlot(EARTH), 10);
			
				if (c.getItems().playerHasItem(UNSTRUNG[i])) {
					c.getItems().deleteItem(UNSTRUNG[i], c.getItems().getItemSlot(UNSTRUNG[i]), 1);
					c.getItems().addItem((STRUNG[i]),1);
					c.getPA().addSkillXP(1 * MAGIC_XP, c.playerCrafting);
				}
			}
		} catch (Exception e) {

		}
	} else {
		c.sendMessage("You dont have enough runes to cast this spell.");
	}
		c.setSpellDelay();
}

/**
 * Casts the super glass make spell.
 * @param c
 */
public static void castSuperglassMake(Client c) {
	if (c.getLastSpell() < 1800) {
		return;
	}

	if (c.playerLevel[6] < 77) {
		c.sendMessage("You don't have the required magic level to cast this spell!");
		c.sendMessage("You need a magic level of " + GLASS_REQUIRED_LEVEL
				+ " to cast this spell.");
		return;
	}
	if (c.getItems().playerHasItem(BUCKET_OF_SAND)
			&& c.getItems().playerHasItem(SODA_ASH)
			&& c.getItems().playerHasItem(SEAWEED)) {
		if (c.getItems().playerHasItem(ASTRAL, 2)
				&& c.getItems().playerHasItem(AIR, 10)
				&& c.getItems().playerHasItem(FIRE, 6)) {

			c.getItems().deleteItem(ASTRAL,
					c.getItems().getItemSlot(ASTRAL), 2);
			c.getItems().deleteItem(AIR, c.getItems().getItemSlot(AIR), 10);
			c.getItems()
					.deleteItem(FIRE, c.getItems().getItemSlot(FIRE), 6);

			c.getPA().addSkillXP(1 * MAGIC_XP, c.playerMagic);
			c.startAnimation(4413);
			c.gfx100(729);
			for (int i = 0; i < 28; i++) {
				if (c.getItems().playerHasItem(BUCKET_OF_SAND)
						&& c.getItems().playerHasItem(SODA_ASH)
						&& c.getItems().playerHasItem(SEAWEED)) {
					c.getItems().deleteItem(BUCKET_OF_SAND,
							c.getItems().getItemSlot(BUCKET_OF_SAND), 1);
					c.getItems().deleteItem(SODA_ASH,
							c.getItems().getItemSlot(SODA_ASH), 1);
					c.getItems().deleteItem(SEAWEED,
							c.getItems().getItemSlot(SEAWEED), 1);
					c.getItems().addItem((MOLTEN_GLASS), 1);
					c.getPA().addSkillXP(1 * MAGIC_XP, c.playerCrafting);
				}
			}
		} else
			c.sendMessage("You don't have the required runes to cast this spell.");
	} else {
		c.sendMessage("You dont have the correct items to cast this spell.");
		c.sendMessage("You need 1 bucket of sand, soda ash and seaweed to cast this spell.");
	}
	c.setSpellDelay();
}

}
