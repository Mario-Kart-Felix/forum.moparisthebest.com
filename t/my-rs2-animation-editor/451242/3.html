[code=java]package jagex.graphics.media3d.animation;

import jagex.io.Buffer;
import jagex.resource.Archive;

public class Animation {

public static void load(Archive archive) {
Buffer data = new Buffer(archive.extract(“seq.dat”, null));
count = data.read_u16();
if (animations == null)
animations = new Animation[count];
for (int j = 0; j < count; j++) {
if (animations[j] == null)
animations[j] = new Animation();
animations[j].decode(data);
}

}

public int duration(int frame_ptr) {
int k = frame_durations[frame_ptr];
if (k == 0) {
Frame frame = Frame.lookup(primary_frames[frame_ptr]);
if (frame != null)
k = frame_durations[frame_ptr] = frame.duration;
}
if (k == 0)
k = 1;
return k;
}

public void decode(Buffer data) {
do {
int i = data.read_u8();
if (i == 0)
break;
if (i == 1) {
num_frames = data.read_u8();
primary_frames = new int[num_frames];
secondary_frames = new int[num_frames];
frame_durations = new int[num_frames];
for (int j = 0; j < num_frames; j++) {
primary_frames[j] = data.read_u16();
secondary_frames[j] = data.read_u16();
if (secondary_frames[j] == 65535)
secondary_frames[j] = -1;
frame_durations[j] = data.read_u16();
}

  } else if (i == 2)
    loop_delta = data.read_u16();
  else if (i == 3) {
    int k = data.read_u8();
    interleave_order = new int[k + 1];
    for (int l = 0; l < k; l++)
      interleave_order[l] = data.read_u8();

    interleave_order[k] = 0x98967f;
  } else if (i == 4)
    stretches = true;
  else if (i == 5)
    priority = data.read_u8();
  else if (i == 6)
    player_replacement_shield = data.read_u16();
  else if (i == 7)
    player_replacement_weapon = data.read_u16();
  else if (i == 8)
    max_loops = data.read_u8();
  else if (i == 9)
    /* when animating, 0 -> block walking, 1 -> yield to walking, 2 -> interleave with walking */
    precedence_animating = data.read_u8();
  else if (i == 10)
    /* when walking, 0 -> block walking, 1 -> yield to walking, 2 -> never used... interleave with walking? */
    precedence_walking = data.read_u8();
  else if (i == 11)
    replay_mode = data.read_u8();
  else if (i == 12)
    /* never used */
    anInt308 = data.read_32();
  else
    System.out.println("Error unrecognised seq config code: " + i);
} while (true);
if (num_frames == 0) {
  num_frames = 1;
  primary_frames = new int[1];
  primary_frames[0] = -1;
  secondary_frames = new int[1];
  secondary_frames[0] = -1;
  frame_durations = new int[1];
  frame_durations[0] = -1;
}
if (precedence_animating == -1)
  if (interleave_order != null)
    precedence_animating = 2;
  else
    precedence_animating = 0;
if (precedence_walking == -1) {
  if (interleave_order != null) {
    precedence_walking = 2;
    return;
  }
  precedence_walking = 0;
}

}

public Animation() {
loop_delta = -1;
stretches = false;
priority = 5;
player_replacement_shield = -1;
player_replacement_weapon = -1;
max_loops = 99;
precedence_animating = -1;
precedence_walking = -1;
replay_mode = 2;
}

public static int count;
public static Animation animations[];
public int num_frames;
public int primary_frames[];
public int secondary_frames[];
public int frame_durations[];
public int loop_delta;
public int interleave_order[];
public boolean stretches;
public int priority;
public int player_replacement_shield;
public int player_replacement_weapon;
public int max_loops;
public int precedence_animating;
public int precedence_walking;
public int replay_mode;
public int anInt308;
}[/code]

mod_taharok August 3, 2016, 10:36pm #10

Okay well since no one else mentioned it, you probably should read up a bit more on how digital animations work. Your “skins” are probably more appropriately called “joints” or “bones.” In skeletal animation, a bone or cluster is a group of vertices (optionally weighted) which follow some sequence of transforms directed by an animation. In your videos, you rotate groups of vertices after selecting a “skin,” which is why I’m pretty sure that’s supposed to be a bone/cluster. A group of clusters is called a skin/a group of bones is called a skeleton, whatever naming you prefer.

Lord_Harry August 3, 2016, 10:39pm #11

The ‘skins’ do not represent bones themselves. A ‘skin’ contains groups of bones, not groups of vertices. So in other words, a skin contains groups of groups of vertices.

For example, there is a ‘skin’ which controls the entire body. This skin contains every bone (vertex group) on the character.

I admit ‘skin’ probably isn’t the best term to use, but for now it will do until I find the correct term. The only other appropriate term I can think of is ‘bone group’. If you have any other name ideas feel free to suggest them.

mod_taharok August 3, 2016, 11:36pm #14

Okay turns out I was being dumb before. You skin name is wrong. A cluster would link a bone to a series of vertices (such “L lower arm”). Your “left arm” isn’t a skin, it’s just another cluster. In skeletal animations, bones are hierarchical. The left arm is an entire bone, containing other child bones. A skin is a total collection of clusters for an entire mesh (so “entire body” is the only skin in that list).
