I was wanting to know how to add anti-randoms to my script. I tried to see what others have done in the way of anti-randoms, but the only ones I’ve found are from people using SRL, and I’m using squig2. My code is posted below and is almost fully working (well, it does work, it just can’t find the color of the bank symbol as it changes constantly, but that can be solved by manually selecting colors). Anyway, if anyone would have any ideas as to how I can incorporat anti-randoms, it would be greatly appreciated.

program WillowMuncher;
{.include squig2.txt}

const
  bankSymbolColor01  = 11777473; //ID color for the bank symbol
  bankSymbolColor02  = 4376554;  //ID color for the bank symbol
  bankSymbolColor03  = 3379890;  //ID color for the bank symbol
  intervalBase       = 400;      //The base wait time between actions
  timesToRepeat      = 1;        //The number of times that the script runs through
  willowColor01      = 4014144;  //ID color for willow trees
  willowColor02      = 4483168;  //ID color for willow trees
  willowColor03      = 6062224;  //ID color for willow trees
  willowColor04      = 1523246;  //ID color for willow trees on the minimap
  
var
  itemsInInventory, itemsInInventoryCheck, itemTimer   : integer;
  mousePositionX, mousePositionY                       : integer;
  reportTimer, reportTimesBanked, reportWillowsBanked  : integer;
  timesToRepeatCounter                                 : integer;

procedure bankItems;
begin
  clickHuman(mousePositionX, mousePositionY, true);
  mapFlagging;
  openBank
  mapFlagging;
  wait(200 + random(50));
  if(isBank)then
  begin
    deposit(1, 1, 28);
    reportWillowsBanked := reportWillowsBanked + 28;
    reportTimesBanked := reportTimesBanked + 1;
  end else drop(1, 29);
end;

function colorSearch(searchColor01, searchColor02, searchColor03, x1, y1, x2, y2, tolerance : integer) : boolean;
var colorFound : boolean;
begin
  colorFound := false;
  if(findColorTolerance(mousePositionX, mousePositionY, searchColor01, x1, y1, x2, y2, tolerance) or
     findColorTolerance(mousePositionX, mousePositionY, searchColor02, x1, y1, x2, y2, tolerance) or
     findColorTolerance(mousePositionX, mousePositionY, searchColor03, x1, y1, x2, y2, tolerance)) then
  begin
    colorFound := true
    moveMouseSmooth(mousePositionX, mousePositionY);
    wait(50 + random(50));
  end;
  result := colorFound;
end;

procedure cutWillow;
begin
  clickHuman(mousePositionX, mousePositionY, true);
  mapFlagging;
  wait(200 + random(50));
  colorSearch(willowColor01, willowColor02, willowColor03, 4, 4, 515, 338, 0);
end;

procedure reportProgress;
begin
  writeLn('');
  writeLn('=============== Progress Report ================');
  writeln('Worked for ' + intToStr(getSystemTime / 1000 - reportTimer));
  writeLn('Banked ' + intToStr(reportTimesBanked) + ' loads(s).');
  writeLn('Banked ' + intToStr(reportWillowsBanked) + ' log(s).');
  writeLn('================================================');
  writeLn('');
end;

procedure walkToWillow;
begin
  colorSearch(mousePositionX, mousePositionY, willowColor04, 600, 100, 630, 145, 0);
  wait(intervalBase + random(100));
  clickHuman(mousePositionX, mousePositionY, true);
  mapFlagging
end;

begin
  reportTimer := getSystemTime / 1000
  setupSquig('Document1 - Microsoft Word')
  itemsInInventory := itemCount;
  itemTimer := getSystemTime / 1000;
  reportWillowsBanked := 0;
  timesToRepeatCounter := timesToRepeat;
  if(colorSearch(willowColor01, willowColor02, willowColor03, 4, 5, 515, 338, 0))then cutWillow;
  repeat
  begin
    repeat
    begin
      if(not(isTextUp('Chop')))then
      begin
        if(colorSearch(willowColor01, willowColor02, willowColor03, 4, 5, 515, 338, 0))then cutWillow;
        itemsInInventoryCheck := itemCount;
        itemTimer := getSystemTime / 1000;
      end;
      if((getSystemTime / 1000) - itemTimer > 5) then
      begin
        itemsInInventoryCheck := itemCount;
        if(itemsInInventory = itemsInInventoryCheck)then
        begin
          if(colorSearch(willowColor01, willowColor02, willowColor03, 4, 5, 515, 338, 0))then cutWillow;
        end;
        itemsInInventory := itemsInInventoryCheck;
        itemTimer := getSystemTime / 1000;
      end;
    end;
    until(fullBag)
    if(colorSearch(bankSymbolColor01, bankSymbolColor01, bankSymbolColor01, 570, 0, 730, 160, 15))then
    begin
      bankItems;
    end else
    begin
      drop(1, 29);
    end;
    reportProgress;
  end;
  until(timesToRepeatCounter = 0)
end.
