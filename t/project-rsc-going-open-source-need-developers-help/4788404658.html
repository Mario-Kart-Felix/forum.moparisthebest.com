import org.moparscape.msc.gs.model.dialog.NpcDialog

class Klarense extends NpcDialog {

override def init {
	if (qstage >= 2 && boughtShip != 1) {
		this + option1
		this + option2
		this + option3
		this + option4
	}
	else {
		this + option1
		this + option2
		this + option3
	}
}

override def begin {
	if (qstage == 0 || qstage == 4) {
		if (fixedShip2 != 1) {
			this > "Oh it's you, I thought you were dead"; breath
			this > "It wasn't easy but I got the lady back"; breath
			this > "if you want to sail you'll have to fix it all over again";
		}
		else {
			this < "Hello there"; breath
			this > "Good to see you made your trip back safe!"
		}
		exit
	}
	else if (boughtShip == 1) {
		this < "Hey how can I sail in this ship?"; breath
		this > "first you should fix the ship damage if you haven't already"; breath
		this > "to fix her up you will need some wooden planks"; breath
		this > "and hammer them over with steel nails"; breath
		this < "then could you take me to crandor isle when it's ready?"; breath
		this > "don't count on it, but you can try asking somebody else"; breath
		this > "I've heard there is on old sailor living in draynor village"; breath
		this > "you might have better luck with him though"; breath
		exit
	}
	else {
		this > "You're interested in a trip on the lumbridge lady are you?"; breath
		this > "I admit she looks fine, but she isn't seaworthy right now"; breath
		end
	}
}

		Array("Well even if I know how to get there",
		"I wouldn't like to risk it",

lazy val option4 = new GenericEnd("Can I buy this ship?", npc, player) {
