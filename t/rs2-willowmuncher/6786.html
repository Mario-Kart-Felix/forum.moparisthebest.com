This isn’t really my first script, its a revamp of my second. I haven’t figured out yet how to incorporate randoms, so any help there would be great. The rest of the script should work fine. Start amongst the willows in draynor, any detail, facing north. It’ll chop down some willows, then bank them when its done. As I don’t have anti-randoms yet, I didn’t include a loop for it to head back to the willows and restart, simply because I don’t want anyone to set it up, walk off and come back later to find that they died…


program WillowMuncher;
{.include squig2.txt}

const
  bankSymbolColor01  = 11777473; //ID color for bank symbol
  bankSymbolColor02  = 4376554;  //ID color for bank symbol
  bankSymbolColor03  = 3379890;  //ID color for bank symbol
  intervalBase       = 400;      //The base wait time between actions
  willowColor01      = 4014144;  //ID color for willow tree
  willowColor02      = 4483168;  //ID color for willow tree
  willowColor03      = 6062224;  //ID color for willow tree
  
var itemsInInventory, itemsInInventoryCheck, itemTimer, mousePositionX, mousePositionY : integer;

function colorSearch(searchColor01, searchColor02, searchColor03, x1, y1, x2, y2 : integer) : boolean;
var colorFound : boolean;
begin
  colorFound := false;
  if(findColorTolerance(mousePositionX, mousePositionY, searchColor01, x1, y1, x2, y2, 0) or
     findColorTolerance(mousePositionX, mousePositionY, searchColor02, x1, y1, x2, y2, 0) or
     findColorTolerance(mousePositionX, mousePositionY, searchColor03, x1, y1, x2, y2, 0)) then
  begin
    colorFound := true
    moveMouseSmooth(mousePositionX, mousePositionY);
    wait(50 + random(50));
  end;
  result := colorFound;
end;

procedure cutWillow;
begin
  if(isTextUp('Chop'))then
  begin
    clickHuman(mousePositionX, mousePositionY, true)
    mapFlagging;
    wait(200 + random(50));
    colorSearch(willowColor01, willowColor02, willowColor03, 4, 4, 515, 338);
  end;
end;

procedure depositItems;
begin
  openBank
  wait(1000 + random(200));
  deposit(1, 1, 28)
end;

procedure walkToBank;
begin
  clickHuman(mousePositionX, mousePositionY, true);
  mapFlagging;
end;

begin
  setupSquig('Document1 - Microsoft Word')
  itemsInInventory := itemCount;
  itemTimer := getSystemTime / 1000;
  if(colorSearch(willowColor01, willowColor02, willowColor03, 4, 5, 515, 338))then cutWillow;
  repeat
  begin
    if(not(isTextUp('Chop')))then
    begin
      if(colorSearch(willowColor01, willowColor02, willowColor03, 4, 5, 515, 338))then cutWillow
      itemsInInventoryCheck := itemCount;
      itemTimer := getSystemTime / 1000;
    end;
    if((getSystemTime / 1000) - itemTimer > 5) then
    begin
      itemsInInventoryCheck := itemCount;
      if(itemsInInventory = itemsInInventoryCheck)then
      begin
        if(colorSearch(willowColor01, willowColor02, willowColor03, 4, 5, 515, 338))then cutWillow;
      end;
      itemsInInventory := itemsInInventoryCheck;
      itemTimer := getSystemTime / 1000;
    end;
  end;
  until(fullBag)
  if(colorSearch(bankSymbolColor01, bankSymbolColor01, bankSymbolColor01, 570, 0, 730, 160))then
  begin
    walkToBank;
    depositItems;
  end else
  begin
    drop(1,29)
  end;
end.
