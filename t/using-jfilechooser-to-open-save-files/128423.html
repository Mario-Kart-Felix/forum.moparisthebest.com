Try the following:

import java.awt.event.ActionListener;
import java.awt.event.ActionEvent;
import javax.swing.JFrame;
import javax.swing.JTextArea;
import javax.swing.JMenuBar;
import javax.swing.JMenu;
import javax.swing.JMenuItem;
import javax.swing.JFileChooser;
import javax.swing.JOptionPane;
import javax.swing.filechooser.FileNameExtensionFilter;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;

/**
* This is a simple class to show the user how to use
* the JFileChooser class with a JTextPane.
*
* @author Avail
*/
public class FileChooser extends JFrame implements ActionListener {

	private JTextArea textArea;
	private JMenuItem open;
	private JMenuItem save;
	private JMenuItem exit;

	/**
	* The constructor for this class.
	*/
	public FileChooser() {
		super("A Simple JFileChooser Class");
		setSize(300, 300);
		textArea = new JTextArea();
		add(textArea, "Center");
		JMenuBar menuBar = new JMenuBar();
		JMenu file = new JMenu("File");
		open = new JMenuItem("Open");
		save = new JMenuItem("Save");
		exit = new JMenuItem("Exit");
		file.add(open);
		file.add(save);
		file.addSeparator();
		file.add(exit);
		menuBar.add(file);
		add(menuBar, "North");
		open.addActionListener(this);
		save.addActionListener(this);
		exit.addActionListener(this);
	}

	/**
	* Is called when a component with an ActionListener
	* is used.
	* The JFileChooser methods used show an example of
	* both a FileSaver and a FileOpener. It also shows
	* an example of a FileFilter being used with both.
	*
	* @param e The Event of the called Listener
	*/
	public void actionPerformed(ActionEvent e) {
		Object comp = e.getSource();
		JFileChooser fileChooser = new JFileChooser(new File("./"));
		FileNameExtensionFilter filter =
		 new FileNameExtensionFilter("Text Documents", "txt");
		fileChooser.addChoosableFileFilter(filter);
		int value = 0;
		int result = JFileChooser.APPROVE_OPTION;
		File f = null;
		if (comp.equals(open)) {
			value = fileChooser.showOpenDialog(this);
			if (value == result) {
				try {
					f = fileChooser.getSelectedFile();
					FileInputStream fis = new FileInputStream(f);
					byte[] buffer = new byte[(int)f.length()];
					fis.read(buffer);
					fis.close();
					textArea.setText(new String(buffer));
				} catch (IOException ioe) { ioe.printStackTrace(); }
			}
		}
		else if (comp.equals(save)) {
			boolean makeFile;
			value = fileChooser.showSaveDialog(this);
			if (value == result) {
				f = fileChooser.getSelectedFile();
				if (f.exists()) {
					if (JOptionPane.showConfirmDialog(null, "Overwrite existing document?") == 0)
						makeFile = true;
					else
						makeFile = false;
				}
				else
					makeFile = true;
				if (makeFile) {
					try {
						FileOutputStream fos = new FileOutputStream(f);
						fos.write(textArea.getText().getBytes());
						fos.close();
					} catch (IOException ioe) { ioe.printStackTrace(); }
				}
			}
		}
		else if (comp.equals(exit))
			System.exit(0);
	}

	/**
	* The entry point for the jvm.
	*
	* @param s Inputed Arguments
	*/
	public static void main(String[] s) {
		new FileChooser().setVisible(true);
	}
}
