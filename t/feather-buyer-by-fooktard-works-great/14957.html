{.include SRL/SRL.scar}

var
tenwhite: integer;

procedure loadbmps;
begin
tenwhite := BitmapFromString(15, 7, ‘z78DA7373C30E0C30002E594’ +
‘AF4D2DF644CF66036197F980C1E93F18709A62E00B8888FE9’);

end;

procedure buyingtenfeathers;
begin
mouse(378, 81,10,10,false);
wait(100+random(20));
findbitmap(tenwhite,x,y);
wait(200+random(50));
mouse(x,y,10,10,true)
end;

begin
setupsrl;
loadbmps;
repeat
buyingtenfeathers;
until(false)
end.[/scar]
