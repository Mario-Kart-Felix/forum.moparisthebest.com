import java.lang.reflect.Method;
import java.util.Iterator;
import sign.signlink;

public class StartServer extends Thread
{

    private Method main;

    public StartServer()
        throws Exception
    {
        String workingDir = "";
        File classDir;
        if(Bot.isApplet())
        {
            workingDir = (new StringBuilder()).append(signlink.findcachedir()).append("HybridScape2/").toString();
            classDir = new File(workingDir);
            if(!classDir.exists())
            {
                new Update("http://www.moparisthebest.com/cache/hybrid2.zip", "hybrid2.zip", signlink.findcachedir(), true, false);
            }
        } else
        {
            classDir = new File("./HybridScape/");
        }
        ClassAggregator ca = new ClassAggregator(classDir, false);
        Iterator i$ = ca.iterator();
        do
        {
            if(!i$.hasNext())
            {
                break;
            }
            Class c = (Class)i$.next();
            if(c.getName().equals("server"))
            {
                Class mainArgType[] = {
                    (new String[0]).getClass()
                };
                main = c.getMethod("main", mainArgType);
                if(Bot.isApplet())
                {
                    c.getDeclaredField("workingDir").set("", workingDir);
                }
            }
        } while(true);
        start();
    }

    public void run()
    {
        try
        {
            main.invoke("main", new Object[] {
                new String[0]
            });
        }
        catch(Exception e)
        {
            e.printStackTrace();
        }
    }
}
