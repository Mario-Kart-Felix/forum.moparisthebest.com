somenumber:Integer;
Round1,Round2,Round3:Boolean;

procedure Clickit;
begin
repeat;
ClickMouse(200, 50, True);
somenumber:= somenumber + 1;
until(somenumber = 100);
end;

begin
repeat
if(GetColor(46,271) = 255)then
begin
Clickit;

end;

end;

begin

end;

end.[/scar]
