[code]
//Start near Yew trees in edgeville
//Setup Lines 11-16
//Drag crosshairs onto runescape
//Version 1.2!

program YewCutter;
{.include SRL/SRL.Scar}
{.include SRL/SRL/Skill/WoodCutting.Scar}
{.include SRL/SRL/Extended/xMapWalk.Scar}

const
LoadsToDo = 10; // The number of loads to do
WaitTime = 10000; //The time to click.
YewColor = 2903391; //Yew tree color here.
YewColor2 = 2771008; //A second tree color here.
Bank = 3324887; //bank color

var
Loads, fs, fs2, sx, sy, ex, ey:Integer;

procedure SetUp;
begin
HighestAngle;
MakeCompass(‘N’);
GameTab(4);
end;

function FindFastRandoms: Boolean; //Function by WT-Fakawi
var
i: Integer;
begin
for i := 1 to 8 do
begin
case I of
1: if FindDead then
Result := True;
2: if FindMod then
Result := True;
3: if FindMime then
Result := True;
4: if FindMaze then
Result := True;
5: if FindQuiz then
Result := True;
6: if FindDemon then
Result := True;
7:
begin
if NoGameTab then
begin
Result := True;
Logout;
Exit;
end;
end;
7:
begin
if InBlack then
begin
Result := True;
Logout;
Exit;
end;
end;
8: RC;
end;
wait(1);
end;
end;

procedure antiz;
begin
case Random(6) of

0: AlmostLogout;

1: begin
HoverSkill(‘Woodcutting’, True);
Wait(564 + random(365));
GameTab(4);
end;

2: begin
HoverSkill(‘Prayer’, false);
Wait(733 + random(421));
GameTab(4);
end;

3: begin
GameTab(1)
Wait(126 + random(895));
GameTab(4)
end;

4: SayCurrentLevels(‘Woodcutting’);

5: PickUpMouse;
end;
end;

Procedure FindThoseRandoms;
begin
if(FindFight)then
begin
RunAwayDirection(‘N’);
Wait(9500+Random(5500));
RunBack;
end;
if(FindDead)then
begin
Writeln(‘You are Dead!’);
Writeln(‘Next player.’);
end;
antiz;
DwarfItem;
SolveFrog;
FindCerter;
FindDemon;
FindQuiz;
SolveBox;
SolveSandWich;
FindNormalRandoms;
FindFastRandoms;
FindLamp(‘woodcutting’);
end;

Procedure CutYews;
begin
SetRun(false);
repeat
if FindMSColorTol(x,y,YewColor,3) or
FindMSColorTol(x,y,YewColor2,3) then
begin
MMouse(x,y,3,3)
If IsUpTextMulti(‘Chop’,‘down’,‘Yew’) then
begin
FindThoseRandoms;
Mouse(x,y,2,2,True)
wait(375)
FindThoseRandoms;
RandomRClick;
Wait(WaitTime+Random(1312))
end else
begin
y := y + 10;
MMouse(x,y,3,3)
FindThoseRandoms;
If IsUpTextMulti(‘Chop’,‘down’,‘Yew’) then
begin
Mouse(x,y,2,2,True)
Wait(WaitTime+Random(1312));
end else
begin
Writeln(‘Door shut’);
LowestAngle;
MakeCompass(‘N’);
MMouse(244,111,3,3)
wait(1125)
Mouse(244,111,2,2,True)
end;
end;
end;
until(InventoryFull)
end;

Procedure BankYews;
begin
FindThoseRandoms;
if not (BankScreen) then
repeat
if (FindObj3(x, y, ‘booth’, 2842230, 10)) then
begin
Mouse(x, y, 4, 4, False)
end else
if (FindObj3(x, y, ‘booth’, 12611, 10)) then
begin
Mouse(x, y, 4, 4, False);
Wait(250);
end else
if (ClickOption(‘quickly’, 1)) then
begin
FFlag(5);
repeat
Wait(100);
fs := fs + 1;
until (BankScreen) or (fs >= 50) or ((GetColor(136, 40) = 255) and (GetColor(202, 39) = 255));
end;
fs2 := fs2 + 1;
until (BankScreen) or (fs2 >= 5) or ((GetColor(136, 40) = 255) and (GetColor(202, 39) = 255));
if (BankScreen) then
begin
while (FindColor(x, y, 65536, sx, sy, ex, ey)) do
begin
Mouse(x + 4, y + 4, 3, 3, False);
Wait(800);
ClickOption(‘All’, 2);
ClickOption(‘Cancel’, 2);
Loads:=Loads+1;
end;
end;
end;

procedure getb;
Begin
SetRun(true);
FindThoseRandoms;
If FindSymbol(x, y, ‘Bank’)then
begin
MMouse(x,y,2,2);
FindThoseRandoms;
wait(1009)
mouse(x,y,2,2,True);
FFlag(5);
FindThoseRandoms;
end else
if FindMSColorTol(x,y,Bank,3)then
begin
MMouse(x,y,2,2);
FindThoseRandoms;
wait(1009)
mouse(x,y,2,2,True);
FFlag(5);
FindThoseRandoms;
end else
writeln(‘Bank not found’)
end;

procedure BackToYews;
begin
if FindSymbol(x, y, ‘tree’)then
begin
MMouse(x,y,3,3);
wait(340);
FindThoseRandoms;
Mouse(x,y,2,2,True);
Wait(30);
FFlag(3);
FindThoseRandoms;
Wait(2654+Random(1232))
end;
end;

procedure ProgressReport;
begin
WriteLn(’-----------------Reeces Yew Chopper--------------’)
WriteLn(’-------------------------------------------------’)
WriteLn(‘You have cut and banked’ +inttostr(Loads)+ ‘loads’)
WriteLn(’-------------------------------------------------’)
WriteLn(’-----------------Reeces Yew Chopper--------------’)
end;

begin
SetUpSRL;
activateclient;
SetUp;
repeat
CutYews;
FindThoseRandoms;
ProgressReport;
getb;
FindThoseRandoms;
BankYews;
FindThoseRandoms;
ProgressReport;
BackToYews;
ProgressReport;
until(Loads >= LoadsToDo)
ProgressReport;
Writeln(‘Thank you for useing Reeces Yew Chopper’);
end.[/code]
