/**

  • All data for Lodestones.
  • @author Arham 4

public class Lodestone {

/**

  • Data for clicked buttons.
    */
    public static int[][] buttons = { { 102111, 3093, 3493 },
    { 102108, 3210, 3424 }, { 102117, 3023, 3208 },
    { 102129, 2606, 3093 }, { 102120, 3093, 3244 },
    { 102126, 2813, 3447 }, { 102132, 2724, 3484 },
    { 102096, 2926, 3559 }, { 102105, 3303, 3123 },
    { 102093, 3222, 3218 }, { 102099, 3176, 2987 },
    { 102114, 2965, 3380 }, { 102123, 2663, 3310 },
    { 102090, 2135, 3942 }, };

/**

  • Get the button ID.
  • @return
    */
    @SuppressWarnings(“unused”)
    public static int getButtonID() {
    for (int i = 0; i < buttons.length; i++)
    return buttons[i][0];
    return 0;
    }

/**

  • Get the X coordinate to teleport to.
  • @return The X coordinate.
    */
    @SuppressWarnings(“unused”)
    public static int getX() {
    for (int i = 0; i < buttons.length; i++)
    return buttons[i][1];
    return 0;
    }

/**

  • Get the Y coordinate to teleport to.
  • @return The Y coordinate.
    */
    @SuppressWarnings(“unused”)
    public static int getY() {
    for (int i = 0; i < buttons.length; i++)
    return buttons[i][2];
    return 0;
    }
    } [/code]
