this isn’t worth being uploaded. for 3 reasons,
There is already a password field for such things.
your password is hard coded
[php]
stop ();
var myPassword = “theblackfortress”;
submit_btn.onRelease = function ()
{
if (password_txt.text == myPassword)
{
gotoAndStop(3);
}
else
{
gotoAndStop(2);

};

again_btn.onRelease = function ()
