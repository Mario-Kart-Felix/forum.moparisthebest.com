For the run code, try this instead.

			case 74192:
				if (c.isRunning2 && c.isRunning) {
					c.isRunning2 = false;
					c.isRunning = false;
					return;
				}
				else if (!c.isRunning2 && !c.isRunning && c.playerEnergy > 0) {
					c.isRunning = true;
					c.isRunning2 = true;
					return;
				}
				break;
