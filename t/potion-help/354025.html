If youre talking about something like all-in-one or 3-in-one pot then:
*Only works if your current code works for strength potion.

case 2438:       
                                root = getLevelForXP(2);
                                toAdd = 5 + (int)Math.round(root * 0.1414141414);
                                if ((skillLvl[0] + toAdd) > (root + toAdd)) {//attack
                                        skillLvl[0] = root + toAdd;
                                } else {
                                        skillLvl[0] += toAdd;
                                }
                                if ((skillLvl[2] + toAdd) > (root + toAdd)) {//strength
                                        skillLvl[2] = root + toAdd;
                                } else {
                                        skillLvl[2] += toAdd;
                                }
                                if ((skillLvl[1] + toAdd) > (root + toAdd)) {//defence
                                        skillLvl[1] = root + toAdd;
                                } else {
                                        skillLvl[1] += toAdd;
                                }
                                getActionSender().setSkillLvl(this, 0);//attack
                                getActionSender().setSkillLvl(this, 2);//strength
                                getActionSender().setSkillLvl(this, 1);//defence
                        break;//<-- you were missing a semicolon ( ; ) without it you would recieve an error.
