public static final int worldID = 2;
public static Class<?> clientClass;
public static Object client;

public Loader() throws Exception {
	parse();
	ClassLoader classLoader = new URLClassLoader(new URL[] { new URL(loaderURL) });
	clientClass = classLoader.loadClass("client");
	client = clientClass.newInstance();

	Applet applet = (Applet) (client);
	applet.setStub(this);
	applet.init();
	applet.start();

	JPanel panel = new JPanel(new BorderLayout());
	panel.setPreferredSize(new Dimension(512, 345));
	panel.add(applet);

	setTitle("RuneScape Classic");
	setDefaultCloseOperation(EXIT_ON_CLOSE);
	setResizable(false);
	add(panel, BorderLayout.CENTER);
	pack();
}

private void parse() {
	try {
		URL url = new URL("http://classic" + worldID + ".runescape.com/plugin.js?param=o0,a1,s0");
		BufferedReader br = new BufferedReader(new InputStreamReader(url.openStream()));
		String raw;
		String param_name = null;
		String value = null;
		while ((raw = br.readLine()) != null) {
			if (!raw.contains("param name") || !raw.contains("value") || raw.contains("haveie6"))
				continue;
			raw = raw.replace("(", "");
			raw = raw.replace(")", "");
			raw = raw.replaceAll("document.write", "").replaceAll("\"", "").replaceAll("'", "").replaceAll("<", "").replaceAll(">", "").replaceAll(";", "");
			if (raw.indexOf("param name") != -1 && raw.indexOf("value") != -1) {
				if (raw.contains("error_loader_crash") || raw.contains("http://www.runescape.com/classicapplet/"))
					param_name = raw.trim().substring("param_name=".length(), raw.indexOf("value") - 3);
				else
					param_name = raw.trim().substring("param_name=".length(), raw.indexOf("value") - 2);
				raw = raw.substring(raw.indexOf(param_name) + (param_name.length() + 1), raw.length());
				value = raw.trim().substring("value=".length(), raw.length());
				parameters.put(param_name, value);
			}
		}
		parameters.put("haveie6", "0");
	} catch (IOException ex) {
		ex.printStackTrace();
	}
}

It was last updated in September. The current client revision is 233. It was 204 before the reopening. Private servers use 202.

IIRC, clientside changes since 2009:

  • Login responses now displayed on a transparent background
  • Friends list fixed
  • You can now use more than one colour modifier in a chat message
  • Report abuse screen updated
  • Right click report abuse on players
  • Buy/sell/trade X items
  • Changes to the structure of the client (GameWindowMiddleMan was removed, quite a few new classes have been added)
  • Changes to obfuscation (saev’s deobber no longer works, it’s pretty much the same obfuscation as RS2 pre-bot nuke)
  • Cache format changed, there’s now a CRC check too
  • Support for display names and email address logins (email support was what the latest update added)
  • Moderator crowns
