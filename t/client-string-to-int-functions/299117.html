static String decode(int i) {
    int len = 0;
    for (int j = i; j != 0; j /= RANGE, len++);
    byte[] b = new byte[len];
    while (len-- > 0) {
        b[len] = (byte) (i % RANGE + OFFSET);
        i /= RANGE;
    }
    return new String(b);
