import java.io.*;

/*

  • @author Sk8rdude
  • This was made in an attempt to ‘simplify’ making compilers.
    */

/**

  • MOVE ALL FILES THAT ARE BACKUPS INTO A FOLDER NAMED “backup” - Caps does NOT matter.
    **/
    public class CompilerMaker {

    static String pathToFiles = “src\”;//The files
    static String pathToLibs = “deps\”;//Where all the .jar files are (such as mysql.jar and so on)
    static String compilerName = “generatedCompiler”;//The name of your new compiler (DO NOT ADD .bat)

    ///IGNORE THESE
    static boolean hasPassed = false;
    static boolean hasPassed2 = false;
    static String computerOS = System.getProperty(“os.name”);
    static String str = “”;

    public static void main(String[] args) throws IOException {
    File path = new File(".\"+pathToFiles);
    File libspath = new File(".\"+pathToLibs);
    if(computerOS.toLowerCase().contains(“win”)){
    scan(libspath, 1);
    System.out.println(“Successfully wrote jars, finding java files and writing…”);
    scan(path, 0);
    System.out.println(“Successfully wrote java files, writing end code”);
    writeFinalCompiler();
    System.out.println(“Successfully wrote end code, program complete.”);
    }else
    System.out.println(“You use an unsupported operating system. Sorry about that.”);

    }
    public static void scan(File file, int type) throws IOException {
    BufferedWriter bw = null;
    try{
    bw = new BufferedWriter(new FileWriter(compilerName+".bat", true));
    if(!hasPassed){
    String jToWrite = System.getProperty(“java.home”);
    jToWrite = jToWrite.replace(“jre”, “bin\javac.exe”);
    bw.write("@echo off");
    bw.newLine();
    bw.write(“TITLE Compiler - By Sk8r”);
    bw.newLine();
    bw.write("""+jToWrite+"""+" -classpath “);
    hasPassed = true;
    }
    switch(type){
    case 0://Scanning for .java’s
    for (File f : file.listFiles()){
    if (f.isDirectory()){
    str = “”+f;
    try {
    if(!str.toLowerCase().contains(“backup”)){
    if(hasPassed2){
    bw.write(” -d bin “);
    hasPassed2 = false;
    }
    if(!str.endsWith(“model”)){
    bw.write(str.substring(2)+”\*.java “);
    }
    }
    bw.flush();
    }
    catch (IOException ioe2) {}
    scan(f, 0);
    }
    }
    break;
    case 1://Scanning for .jars
    for (File f : file.listFiles()){
    if (f.isDirectory())
    scan(f, 1);
    else{
    str = “”+f;
    if(str.contains(“jar”)){
    try {
    bw.write(str.substring(2).replace(”\", “/”) +";");
    bw.flush();
    }catch (IOException ioe2) {}
    }
    }
    }
    hasPassed2 = true;
    break;
    }
    bw.close();
    }catch (IOException io){}
    }
    public static void writeFinalCompiler() {
    BufferedWriter bw = null;
    try{
    bw = new BufferedWriter(new FileWriter(compilerName+".bat", true));
    bw.newLine();
    bw.write(“TITLE Finished compiling!”);
    bw.newLine();
    bw.write(“pause”);
    bw.flush();
    bw.close();
    }catch (IOException ioe2) {}
    }
    }[/code]

[code]package org.tools;

import java.io.BufferedWriter;
import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class CompilerMaker {

public static void main(String[] args) throws IOException {
	StringBuilder builder = new StringBuilder();
	for (File file : listFiles("./src")) {
		if (file.isDirectory()
				&& file.listFiles().length > 0) {
			builder.append(file.getPath().substring(2) + "\\*.java ");
			System.out.println("appending new directory to builder...");
		}
	}
	System.out.println("Compiler information has been written!");
	BufferedWriter writer = new BufferedWriter(new FileWriter("./compiler.txt"));
	writer.write(builder.toString());
	writer.close();
}

public static List<File> listFiles(String directory) {
	final File folder = new File(directory);
	final List<File> files = new ArrayList<File>();
	final File[] folderArray = folder.listFiles();
	
	files.addAll(Arrays.asList(folderArray));
	for (File file : folderArray) {
		if (file.isDirectory()) {
			files.addAll(listFiles(file.getPath()));
		}
	}
	return files;
}

}[/code]

My code does the same.
First, it builds the beginning of the batch file:

@echo off TITLE Compiler - By Sk8r "C:\Program Files\Java\jdk1.7.0_25\bin\javac.exe" -classpath

Then it loops through the deps and get’s all the jars (the newest one allows you to select where the deps is, or if it’s named libs, etc you can pick that):

Next, it searches all the files for the src (Source):

Finally, it makes the end of the batch:

TITLE Finished compiling! pause

If you’re wondering why I have an instance of TITLE before and after the actual code that executes the compiler, it’s because while you’re compiling, it will say the first title, but as soon as it finishes it reads the next line, which is the new title, sets the new title, and then pauses.

Basically I did it because I can hover the icon on my pc and see what the title is to know when it’s done.
