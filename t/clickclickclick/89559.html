program SimpleAutoClicker;

var
x, y: Integer;
const
itemcolor= 50687;    //Place color here.
speed= 20;         //Speed in Miliseconds between clicks

Procedure ClickColor;
  begin
  FindColor(x, y, itemcolor,0,0,549,399);
  wait(0);
  MoveMouse(x, y);
  Clickmouse(x, y, true);
  end;

Begin
Cleardebug
repeat
ClickColor;
Wait(speed);
until(false);
End.
