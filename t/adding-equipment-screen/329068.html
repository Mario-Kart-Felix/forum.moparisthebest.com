c.showInterface(15106);
c.sendFrame126(“Stab: +”+c.playerBonus[0]+"",15113);
c.sendFrame126(“Slash: +”+c.playerBonus[1]+"",15114);
c.sendFrame126(“Crush: +”+c.playerBonus[2]+"",15115);

break;

case 59004: //Close Equipment stats
c.closeInterface();
break;
