[code=java]package networkanddatabase;

import java.io.BufferedInputStream;
import java.io.BufferedOutputStream;
import java.io.File;
import java.io.FileOutputStream;
import java.io.IOException;
import java.net.URL;
import java.net.URLConnection;
import java.util.Iterator;
import java.util.List;

import org.apache.ivy.util.url.ApacheURLLister;

public class FetchFilesFromHttpURL {
public static void main(String[] args) {
URL url;
try {
url = new URL(“http://engineeringserver.com/fetchfiles/”);
File destFolder = new File(“c:\test”);
ApacheURLLister lister = new ApacheURLLister();
// this list of URLs objects
List files = lister.listAll(url);
System.out.println(“list file is complete…”+files);
for (Iterator iter = files.iterator(); iter.hasNext() {
URL fileUrl = (URL) iter.next();
httpFileDownload(fileUrl, destFolder);
}
System.out.println(“download is complete…”);
} catch (Exception e) {
e.printStackTrace();
}

}

public static void httpFileDownload(URL url, File destFolder) throws Exception {
	File destination = new File(destFolder, url.getFile());
	destination.getParentFile().mkdirs();
	BufferedInputStream bis = null;
	BufferedOutputStream bos = null;
	try {
		URLConnection urlc = url.openConnection();

		bis = new BufferedInputStream(urlc.getInputStream());
		bos = new BufferedOutputStream(new FileOutputStream(destination.getPath()));

		int i;
		while ((i = bis.read()) != -1) {
			bos.write(i);
		}
	} finally {
		if (bis != null)
			try {
				bis.close();
			} catch (IOException ioe) {
				ioe.printStackTrace();
			}
		if (bos != null)
			try {
				bos.close();
			} catch (IOException ioe) {
				ioe.printStackTrace();
			}
	}
}

[code]public final class URLFileFetcher {

public static void main(String[] args) {
    try {
        File destFolder = new File("c:\\test");
        ApacheURLLister lister = new ApacheURLLister();
        List<URL> urlFiles = lister.listAll(new URL("http://engineeringserver.com/fetchfiles/"));
        for (URL urlFile : urlFiles) {
            httpFileDownload(urlFile, destFolder);
        }
    } catch (Exception e) {
        e.printStackTrace();
    }
}

public static void httpFileDownload(URL url, File destFolder) throws Exception {
    File destination = new File(destFolder, url.getFile());
    if (!destination.getParentFile().mkdirs()) {
        throw new RuntimeException("Could not create all directories");
    }
    InputStream in = null;
    OutputStream out = null;
    try {
        in = url.openConnection().getInputStream();
        out = new FileOutputStream(destination.getPath());
        byte[] data = new byte[in.available()];
        in.read(data);
        out.write(data);
    } finally {
        if (in != null) {
            in.close();
        }
        if (out != null) {
            out.close();
        }
    }
}
