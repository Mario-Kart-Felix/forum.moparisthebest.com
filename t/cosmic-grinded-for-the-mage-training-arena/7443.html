Well it took me some time to perfect this script but here it is. I got to level 85 magic using it. You might look at my script and say “There are no randoms!”, but because its all takes place in a mini game, runescape didn’t put any randoms in it:).

P.S.
It is set for Lvl-5 Jewelry so its leech proof.

[scar]//All credit goes to rjop455 and squig :D.
//This script will shut off if it cant find the deposit box but not if you run out of runes.
//Start just above the deposit box facing north.
//Enjoy :D.

program CosmicGrinder;
{.include Squig.txt}

VAR
a, b, clicks, checksbad, checks, firsttime, go1, color1, TotalDone, TotalXP: Integer;

Procedure DisguiseScar;
var application:tapplication;
Self: TForm;
disguised:string;
begin
Application:= GetApplication;
Application.Title:= disguised;
Self:= GetSelf;
GetSelf.Caption:= ‘Winamp’;
end;

Procedure ChangeReport(Reason : String);
Begin
Status(Reason);
ClearReport;
AddToReport(‘Rjop455’‘s Cosmic grinder’);
AddToReport(’*******’);
AddToReport(‘Currently: ’ + Reason);
AddToReport(WorkedFor);
AddToReport(‘Enchanted: ’ + inttostr(TotalDone) + ’ times’ );
AddToReport(’’ + inttostr(TotalXP) + ’ Magic XP’ );
End;

Procedure Go;
begin
ChangeReport(‘Running/Walking to Cube Pile…’);
wait(random(10)+20)
MouseC(595 + random(1), 41 + random(2), 1);
wait(random(1000)+7000)
end;

Procedure PickUp;
begin
repeat
if(firsttime = 0) then
begin
ChangeReport(‘Getting color…’);
firsttime:= 1;
FindTextArcSpiral(a, b, 240 + random(5), 140 + random(5), 14, 208, ‘Take-from’)
Getmousepos(a, b);
color1:= getcolor(a,b);
end;
if(firsttime = 1)then
begin
FindColor(a, b, color1, M1, M2, M3, M4);
wait(random(20) + 20);
Getmousepos(a, b);
MouseC(a, b, 0);
if(IsTextUp(‘Take-from’) = true)then
begin
ChangeReport(‘Picking up cubes…’);
MouseC(a, b, 1);
end;
if(IsTextUp(‘Take-from’) = false)then
begin
ChangeReport(‘Can’‘t find cube searching…’);
FindTextArcSpiral(a, b, 240 + random(5), 140 + random(5), 14, 208, ‘Take-from’)
Getmousepos(a, b);
color1:= getcolor(a,b);
end;
end;
wait(random(500) + 00);
until(FULLBAG = true);
end;

Procedure Enchant1;
begin
OpenMagic;
go1:= 0;
repeat
MouseC(701 + random(12), 363 + random(13), 1);
wait(random(50) + 50);
if(IsTextUp(‘Cast’) = true)then
begin
ChangeReport(‘Enchanting…’);
MouseC(701 + random(15), 434 + random(15), 0);
if(IsTextUp(‘Cast Enchant Lvl-5 Jewelry on Orb’) = true)then
begin
go1:= 1;
end;
if(go1 = 0)then
begin
wait(random(50)+20)
TotalDone:= TotalDone + 1;
TotalXP:= TotalXP + 58;
MouseC(701 + random(15), 434 + random(15), 1);
wait(random(300) + 1200);
end;
end;
if(IsTextUp(‘Cast’) = false)then
begin
OpenMagic;
end;
until(go1 = 1);
end;

Procedure Deposit1;
begin
ChangeReport(‘Running/Walking to deposit box…’);
clicks:= 0;
checks:= 0;
MouseC(700 + random(2), 118 + random(2), 1);
Getmousepos(a, b);
MouseC(a, b + random(5), 1);
wait(random(1000) + 6000);
ChangeReport(‘Searching for deposit box…’);
if(FindTextArcSpiral(a, b, 388 + random(5), 217 + random(5), 10, 300, ‘Deposit’) = true)then
begin
ChangeReport(‘Depositing…’);
Getmousepos(a, b);
MouseC(a, b, 1);
wait(random(1000) + 2000);
clicks:= 1;
checksbad:= 0;
end;
if(clicks = 0)then
begin
ChangeReport(‘Couldn’‘t find deposit spot moving and researching for it…’);
MouseC(657 + random(1), 95 + random(2), 1);
wait(random(1000) + 2000);
if(FindTextArcSpiral(a, b, 388 + random(15), 217 + random(15), 10, 300, ‘Deposit’) = true)then
begin
Getmousepos(a, b);
MouseC(a, b, 1);
wait(random(1000) + 2000);
checks:= 1;
checksbad:= 0;
end;
if(checks = 0)then
begin
checksbad:= checksbad + 1;
writeln(‘Couldn’‘t find deposit box ’ + inttostr(checksbad) + ’ times in a row’);
end;
if(checksbad >= 3)then
begin
writeln(‘Couldn’‘t find deposit box 3 times in a row shutting down.’);
KillScript;
end;
end;
end;

begin
SetupSquig;
DisguiseScar;
activateclient;
wait(1000)
movemouse(400,200)
firsttime:= 0;
repeat
MapCompass(‘N’);
Go;
PickUp;
Enchant1;
Deposit1;
ChangeReport(‘Finished enchanting.’);
until(false);
end.[/scar][br][size=1]Posted at: August 01, 2006, 11:27:39 AM[/size][hr]Can anyone rate this script? 1 - 10? No flaming please.
