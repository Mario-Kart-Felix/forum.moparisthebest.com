program PeanutPlayer;

var BPlayGame {Play Game Button}, BNextRound {Next Round Button},Log {The Logs}, Clouds {Clouds When Running},x,y: Integer;
procedure StartGame;
begin
If(FindDTM(BPlayGame,x,y,190,190,470,270))Then
ActivateClient;
ClearDebug;
end;
procedure PlayGame;
begin
repeat
SendArrowSilent(3); //Hit left and right to charge up the bar
SendArrowSilent(1);
until(FindColor(x,y,255,160,350,160,353)); //Wait for the energy bar to turn red

SendArrowSilent(0); //Hit Up (Release Peanut)

repeat
  Wait(1000); //Wait for the intro to stop
until(FindDTM(Clouds,x,y,340,30,430,90)); //Look for the clouds in the main part


if(not(FindColorTolerance(x,y,16764622,330,200,360,250,20)))Then
  begin
    repeat
      SendArrowSilentWait(1,10); //Move right until the dog is under the peanut
    until(FindColorTolerance(x,y,16764622,330,200,360,250,20));
  end;

repeat
if(FindDTM(Log,x,y,400,250,490,310))Then
  begin
    SendArrowSilent(0); //Jump over logs
    SendArrowSilent(1);
  end;
until(FindColorTolerance(x,y,6518656,300,130,390,300,3));
    SendArrowSilent(1); //Catch the peanut in a backflip
    SendArrowSilent(3);
    SendArrowSilent(1);
    SendArrowSilent(3);
    SendArrowSilent(0);

end;
procedure NextRound;
begin
repeat
wait(1000+random(17));
until(FindDTM(BNextRound,x,y,60,80,250,140));
end;
begin
BPlayGame := DTMFromString(‘78DA63CC63646060076224F0FFFF7F300D136’ +
