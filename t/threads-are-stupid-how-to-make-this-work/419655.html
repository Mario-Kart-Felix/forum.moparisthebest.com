package com.mud.client;

import java.net.*;
import java.io.*;

public class ConnectionHandler extends Client implements Runnable{
    
    static int pid;
    static Socket socket;
    static DataInputStream in;
    static DataOutputStream out;
    static String host = "localhost";
    static int PORT = 7777;
    public ConnectionHandler(){
        try {
            socket = new Socket(host,PORT);
            in = new DataInputStream(socket.getInputStream());
            out = new DataOutputStream(socket.getOutputStream());
            pid = in.readInt();
            player[pid] = new Player();
            player[pid].setPos(x,y);
        } catch (UnknownHostException ex) {} catch (IOException ex) {}
    }

    public void run() {
        Output test = new Output(socket,out);

                Thread.sleep(200);
