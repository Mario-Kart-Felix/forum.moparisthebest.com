                                    tempOut.write(tempAdd);
                                    tempOut.newLine();
                            }
                    }
                    fileStream.close();
                    tempOut.close();
                    charFile.delete();
                    tempCharFile.renameTo(charFile);
            } catch (Exception e) {
                    e.printStackTrace();
            }
    }
