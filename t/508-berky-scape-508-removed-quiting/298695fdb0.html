public void appendHit(int damage, int posion) {
    if (damage > currentHP) {
        damage = currentHP;
    }
    currentHP -= damage;
    if (currentHP <= 0) {
        currentHP = 0;
        attackingPlayer = false;
        isDead = true;
    }
    if (!hit1UpdateReq) {
        hitDiff1 = damage;
        posionHit1 = posion;
        hit1UpdateReq = true;
    } else {
        hitDiff2 = damage;
        posionHit2 = posion;
        hit2UpdateReq = true;
    }
    updateReq = true;
}
