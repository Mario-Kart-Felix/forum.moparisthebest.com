I’m officially back working on this full time now David and I have been collaborating and while he gets the open beta ready for you all I will be creating the backbone so that we can write content efficiently!

To do that we will be deprecating the Ruby scripting system and moving to a Lua scripting system instead! The perks of this include the availability of inline functions bound to bindings for easy access to blocking, meta data, and other helpful information.

eg:

-- Routine for burying bones, uses the last clicked item
-- as the bone item to remove and give experience for.
prayer.bury = function()

	-- Start the animation and wait a few ticks 
	msg('You dig a hole in the ground')
	animate(burying)
	delay(2)

	-- Remove the bone from the inventory and give the experience
	msg('You bury the bones.')
	addxp(skills.prayer, get('burying.exp'))
	inv_remove(inv, last_slot)
end

To couple with this there will also be a custom file format for declaring meta information for different entities.

eg:

npc.
    man.
        pickpocket.
            reward_item :   %i(@item.coins, 5)
            level       :   1
        combat.
            att_lvl     :   1
            def_lvl     :   1
            str_lvl     :   1
            aggressive  :   false

We want you guys to be able to make the Moparscape server yours, so I hope you’re excited! Guides and more updates will follow as time passes.

public void playerEnvironment(LuaTable env, Player player) {
	entityEnvironment(env, player);
	env.set("msg", new MessageCommand(player));
	env.set("msgbox", new MessageBoxCommand(player, env, globals));
	env.set("skillbox", new SkillBoxCommand(player, env, globals));
	env.set("objbox", new ObjectBox(player, globals));
	env.set("intclose", new CloseInterfacesCommand(player));

	env.set("skill_lvl", new SkillLevelCommand(player));
	env.set("skill_cur", new SkillCurrentCommand(player));

	env.set("showint", new ShowInterfaceCommand(player));
	env.set("showchat", new ShowChatInterfaceCommand(player));

	env.set("addxp", new AddExperienceCommand(player));

	env.set("runon", new RunOnCommand(player));
	env.set("runoff", new RunOffCommand(player));
	env.set("setrun", new SetRunCommand(player));
	env.set("getrun", new GetRunCommand(player));
	
	env.set("getwildlvl", new GetWildernessLevelCommand(player));
	
	env.set("teleport", new TeleportCommand(player));
	
	env.set("flashtab", new FlashTabCommand(player));

	// env.set("inv_size", new InventorySizeCommand(player));
	// env.set("inv_remove", new InventoryRemoveCommand(player));

	// env.set("give", new GiveItemCommand(player));
}

public void entityEnvironment(LuaTable env, Entity entity) {
	basicEnvironment(env);
	// attribute system
	env.set("getstr", new GetStringCommand(entity));
	env.set("getnum", new GetNumberCommand(entity));
	env.set("putstr", new StoreStringCommand(entity));
	env.set("putnum", new StoreNumberCommand(entity));
	env.set("hasvar", new HasVarCommand(entity));
	env.set("delvar", new DeleteVarCommand(entity));
	
	env.set("gfx", new GraphicCommand(entity));
	env.set("anim", new AnimateCommand(entity));
	env.set("say", new SayCommand(entity));
	env.set("teleport", new TeleportCommand(entity));
	
	// entity specific
	env.set("uid", entity.getId());
}

public void basicEnvironment(LuaTable env) {
	env.set("delay", new DelayCommand(globals));
	env.set("fork", new ForkCommand(contextFactory));
	env.set("exec", new ExecuteCommand(env));
	env.set("rand", new RandomCommand());
}
