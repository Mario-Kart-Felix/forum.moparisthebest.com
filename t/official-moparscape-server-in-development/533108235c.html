java_import ‘net.scapeemulator.game.model.player.ComponentListenerAdapter’

module Fletching

’

def bind_handlers
RECIPES.each do |hash, recipes|
bind :item_on_item, :first => ItemOnItemDispatcher::reverse_hash(hash)[0], :second => ItemOnItemDispatcher::reverse_hash(hash)[1] do
Fletching::show_fletching_interface(player, recipes)

def show_fletching_interface(player, recipes)
  id = recipes.size == 1 ? 309 : (301 + recipes.size)
  for i in (0...recipes.size)
    player.send InterfaceItemMessage.new(id, i + 2, 200, recipes.values[i].product_id)
    string_index = 7 + (4 * i) + recipes.size - 2
    item_name = Item.new(recipes.values[i].product_id).definition.name
    player.set_interface_text(id, string_index, "<br><br><br><br><br>#{item_name}");
  end
  player.get_interface_set.open_chatbox id
  listener = FletchingInterfaceListener.new
  player.get_interface_set.get_chatbox.set_listener(listener)
  listener.set_context(player, recipes)
end

end

class FletchingTask < Action

def initialize(player, recipe, amt)
  super(player, recipe.delay, true)
  @recipe = recipe
  @amt = amt
  @count = 0
end

def execute
  if(!@cleared)
    mob.get_walking_queue.reset
    @cleared = true
  end
  if @recipe.check_reqs(mob)
    @recipe.fletch(mob)
    @count += 1
    if @count >= @amt
      mob.cancel_animation
      stop
    end
  else
    mob.cancel_animation
    stop
  end
end

end

class FletchingInterfaceListener < ComponentListenerAdapter
def set_context(player, recipes)
@player = player
@recipes = recipes
end

def inputPressed(component, component_id, dynamic_id)
  offset = component.get_current_id == 309 ? 3 : (4 + (component.get_current_id - 303))
  component_id -= offset
  recipe = @recipes.values[component_id / 4]
  option = 3 - (component_id % 4)
  amt = component.get_current_id == 309 ? AMOUNTS_309[option] : AMOUNTS[option]
  @player.get_interface_set.get_chatbox.reset
  @player.start_action(FletchingTask.new(@player, recipe, amt))
end

def componentClosed(unused)
  component = player.get_interface_set.get_chatbox

  component.reset

class FletchingInterfaceListener < ComponentListenerAdapter
def set_context(player, recipes)
@player = player
@recipes = recipes
end

def inputPressed(component, component_id, dynamic_id)
  parent_id = component.get_current_id
  offset = parent_id == 309 ? 3 : (4 + (parent_id - 303))
  component_id -= offset
  recipe = @recipes.values[component_id / 4]
  option = 3 - (component_id % 4)
  case option
  when 0..1
    amt = 5 ** option
  when 2
    amt = parent_id == 309 ? 'X' : 10
  when 3
    amt = parent_id == 309 ? recipe.calculate_max(@player) : 'X'
  end
  if amt == 'X'
    close
    input_listener = FletchingInputListener.new
    input_listener.set_context(@player, recipe)
    @player.get_script_input.set_integer_input_listener input_listener
    @player.get_script_input.show_integer_script_input
  else
    @player.get_interface_set.get_chatbox.reset
    @player.start_action FletchingTask.new(@player, recipe, amt)
  end
end

def componentClosed(unused)
  component = player.get_interface_set.get_chatbox
  component.remove_listener
  component.reset
end

def close
  component = @player.get_interface_set.get_chatbox
  component.remove_listener
  component.reset
end

end

class FletchingInputListener < IntegerScriptInputListener
def set_context(player, recipe)
@player = player
@recipe = recipe
end

def inputReceived(value)
  @player.start_action FletchingTask.new(@player, @recipe, value) if value > 0
  @player.get_script_input.reset
end

end
end

require 'java'

java_import 'org.apollo.game.model.Animation'

CHISEL_ID = 1755
GEM_CUTTING_ANIM = Animation.new(885);

class Gems

uncut = [1623, 1621, 1619, 1617, 1631, 6571, 1625, 1627, 1629]
cut = [1607, 1605, 1603, 1601, 1615, 6573, 1609, 1611, 1613]
level_needed = [1, 13, 16, 20, 27, 34, 43, 55, 67]
exp_given = [15, 20, 25, 50, 67, 85, 107, 137, 167]

def canCutGem(id)
  for canCut in 0..uncut.length
    if(uncut[canCut] == id)
      return canCut;
    end
   end
   return nil
end

def makeGem(character, item, slot)
  skills = character.skill_set
  get_crafting = skills.get_skill(Skill::CRAFTING).maximum_level
  remove(in item) = character.inventory.remove(item)
  add(in item) = character.inventory.add(item)
  anim(in id) = character.player_animation id
       
  if (get_crafting < level_req[slot])
    character.send_message "You need a crafting level of #{level_needed[slot]} to cut this gem."
    return
  end
       
  remove(item)
  anim(GEM_CUTTING_ANIM)
  if add(cut[slot])
    character.send_message "You carefully cut the gem."
    skills.add_experience Skill::CRAFTING, exp_given[slot]
  end
end

on :event, :ItemOnItem do |ctx, player, event|

  itemId = player.inventory.get(event.getItemSlot).id;
  usedOnItem = player.inventory.get(event.getUsedId).id;
       
  if canCutGem(itemId) != nil && usedOnItem == CHISEL
    makeGem(player, itemId, canCutGem(itemId))
  end
   
  if itemId == CHISEL && canCutGem(usedOnItem) != nil
    makeGem(player, usedOnItem, canCutGem(usedOnItem))
  end
end
