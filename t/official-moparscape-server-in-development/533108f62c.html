How will this new plugin system affect development?

Well that’s simple, it’ll make it much easier to create plugins! You simply now extend the base plugin class like in the example provided below:

class TestPlugin < Plugin::Base
  def load
    p 'Hello world!'
  end
end

Binding hooks is now easier as ever as well! You can now provide the name of the method when binding a hook thats within the plugin and away you go. This helps people better understand and organize their inputs.

def load
  bind :button, :id => 5, :action => 'say_hello'
end

def say_hello
  p 'Hello!'
end

You can also now turn individual plugins off and reload single plugins seamlessly!
Patented ‘Aw-Yeah-Its-Ruby’ technology.

The region system is currently in development but I thought I’d give a sneak peak into some of the power you will have at your hands.

You will have the ability to declare unique regions for the server that will allow you to modified characters who enter these areas over time.

To add, you will also have the ability to declare triggers which are only activated when a character enters a specific area. Great for quests and other goodies! The clipping and traversal system will be ported to this new region system and also to add the amount of memory the server uses will be cut drastically with this change.

All regions are saved when out of use and all changes to the world are marked as permanent so if you spawn an object in a specific region it will stay there until you wipe the temporary object data space!

package us.evelus.world.model;

import io.netty.buffer.ByteBuf;

import java.util.LinkedHashMap;
import java.util.Map;

public final class Region {

    private Map<Position, SceneObject> loadedObjects = new LinkedHashMap<>();

    private byte[] areaFlags = new byte[4096];
    private int[] areaIds = new int[8];

    private byte[] triggerFlags = new byte[4096];
    private int[] triggerIds = new int[8];

    private byte[] tileFlags = new byte[4096];

    public Region() {}

    public void decodeObjectData(ByteBuf buf) {
        while(true) {
            Position position = Position.fromHash14(buf.readShort());
            int type = buf.readShort();

            SceneObject object = new SceneObject();
            object.setPosition(position);

            loadedObjects.put(position, object);
        }
    }

    public void decodeTileData(ByteBuf buf) {
        while(true) {

            // Read the opcode from the buffer
            int opcode = buf.readByte();
            if(opcode == 0) {
                break;
            }

            // Reserved for setting a tile flag
            if(opcode == 1) {
                tileFlags[buf.readShort()] = buf.readByte();
            }
        }
    }

    public void decodeAreaData(ByteBuf buf) {
        while(true) {

            // Read the opcode from the buffer
            int opcode = buf.readByte();
            if(opcode == 0) {
                break;
            }

            // Reserved for setting an area id
            if(opcode == 1) {
                areaIds[buf.readByte()] = buf.readShort();
            }

            // Reserved for updating an area flag
            if(opcode == 2) {
                areaFlags[buf.readShort()] = buf.readByte();
            }
        }
    }

    public void decodeTriggerData(ByteBuf buf) {
        while(true) {

            // Read the opcode from the buffer
            int opcode = buf.readByte();
            if(opcode == 0) {
                break;
            }

            // Reserved for setting a trigger id
            if(opcode == 1) {
                triggerIds[buf.readByte()] = buf.readShort();
            }

            // Reserved for updating a trigger flag
            if(opcode == 2) {
                triggerFlags[buf.readShort()] = buf.readByte();
            }
        }
    }
}
package us.evelus.world.model;

import io.netty.buffer.ByteBuf;
import io.netty.buffer.Unpooled;
import io.netty.handler.codec.base64.Base64;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.sql.*;

public final class RegionLoader {

    private static final Logger logger = LoggerFactory.getLogger(RegionLoader.class);
    private static Connection connection;

    private static PreparedStatement queryStatement;

    public static void init(String database) throws ClassNotFoundException {

        Class.forName("org.sqlite.JDBC");

        try {
            // Initialize the connection to the database
            connection = DriverManager.getConnection("jdbc:sqlite:" + database);

            // Prepare the query statement
            queryStatement = connection.prepareStatement("SELECT obj_data, tile_data, area_data, trigger_data FROM regions WHERE loc = ? LIMIT 1");
        } catch(SQLException ex) {
            logger.error("Failed to open a connection to the database", ex);
            return;
        }

        logger.info("Region loader successfully initialized!");
    }

    public static Region load(Position position) {

        try {

            // Execute the query to grab the data for the region
            queryStatement.setInt(1, position.hash18());
            ResultSet results = queryStatement.executeQuery();

            // If there are no results for the query return null
            if(!results.next()) {
                return null;
            }

            // Create and decode the data for the specified region
            Region region = new Region();

            region.decodeObjectData(unpackData(results.getBytes("obj_data")));
            region.decodeTileData(unpackData(results.getBytes("tile_data")));
            region.decodeAreaData(unpackData(results.getBytes("area_data")));
            region.decodeTriggerData(unpackData(results.getBytes("trigger_data")));

            return region;
        } catch(SQLException | RuntimeException ex) {
            logger.error("Failed to load region " + position.hash18(), ex);
            return null;
        }
    }

    public boolean save(Region region) {
        return false;
    }

    private static ByteBuf packData(ByteBuf buf) {
        return null;
    }

    private static ByteBuf unpackData(byte[] bytes) {
        return Base64.decode(Unpooled.wrappedBuffer(bytes));
    }
}
