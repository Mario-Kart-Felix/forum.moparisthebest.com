It works properly. Take a look at the canSeeMee method and the status updating methods.

[code=java]public class Friends {

public enum PrivacySetting {
	ON, FRIENDS, OFF, HIDE
}

public enum FriendsListStatus {
	LOADING, CONNECTING, CONNECTED
}

private final Player player;

private List<Long> friendsList;
private List<Long> ignoreList;

private final Set<Long> onlineFriends = new HashSet<Long>();

private PrivacySetting privateChat = PrivacySetting.ON;
private PrivacySetting publicChat = PrivacySetting.ON;
private PrivacySetting trade = PrivacySetting.ON;

private int messageTicket;


public Friends(Player player) {
	this.player = player;
}

public void init() {
	player.send(new PrivacySettingsUpdateMessage(publicChat.ordinal(), privateChat.ordinal(), trade.ordinal()));
	player.send(new FriendsListStatusMessage((byte) FriendsListStatus.CONNECTED.ordinal()));

	List<Long> toUpdate = new ArrayList<Long>(friendsList);
	for (Player other : World.getWorld().getPlayers()) {
		if (toUpdate.isEmpty()) {
			break;
		}
		if (toUpdate.contains(other.getLongUsername())) {
			toUpdate.remove(other.getLongUsername());
			if (other.getFriends().canSeeMe(player)) {
				player.send(new FriendStatusMessage(other.getLongUsername(), 1));
				onlineFriends.add(other.getLongUsername());
			} else {
				player.send(new FriendStatusMessage(other.getLongUsername(), 0));
			}
		}
	}

	for (Long friend : toUpdate) {
		player.send(new FriendStatusMessage(friend, 0));
	}
	if (ignoreList.size() > 0) {
		player.send(new SendIgnoresMessage(ignoreList));
	}
	login();
}

public void privateChatSent(long name, byte[] packed) {
	if(!friendsList.contains(name)) {
		return;
	}
	// Repack the text to ensure its all valid
	String text = ChatUtils.unpack(packed);
	packed = new byte[256];
	int len = ChatUtils.pack(text, packed);
	packed = Arrays.copyOf(packed, len);
	Player other = World.getWorld().getPlayerByLongName(name);
	if (other != null) {
		if(other.getFriends().canSeeMe(player)) {
			other.send(new PrivateChatReceivedMessage(player.getLongUsername(), (byte) player.getRights(), packed, other.getFriends().getNextMessageTicket()));
			player.send(new PrivateChatSentMessage(name, packed));
		}
	}
	
}

public void addFriend(long name) {
	if (friendsList.contains(name) || ignoreList.contains(name) || name == player.getLongUsername()) {
		return;
	}
	if (friendsList.size() >= 200) {
		player.sendMessage("Your friends list is full!");
		return;
	}
	friendsList.add(name);
	Player other = World.getWorld().getPlayerByLongName(name);
	if (other != null) {
		if (other.getFriends().canSeeMe(player)) {
			loggedIn(other);
		}
		if (privateChat == PrivacySetting.FRIENDS) {
			other.getFriends().loggedIn(player);
		}
	}
}

public void addIgnore(long name) {
	if (friendsList.contains(name) || ignoreList.contains(name) || name == player.getLongUsername()) {
		return;
	}
	if (ignoreList.size() >= 100) {
		player.sendMessage("Your ignore list is full!");
		return;
	}
	ignoreList.add(name);
	Player other = World.getWorld().getPlayerByLongName(name);
	if (other != null) {
		other.getFriends().loggedOut(player);
	}
}

public void removeFriend(long name) {
	if (!friendsList.contains(name) || name == player.getLongUsername()) {
		return;
	}
	friendsList.remove(name);
	Player other = World.getWorld().getPlayerByLongName(name);
	if (other != null) {
		if (privateChat == PrivacySetting.FRIENDS) {
			other.getFriends().loggedOut(player);
		}
	}
}

public void removeIgnore(long name) {
	if (!ignoreList.contains(name) || name == player.getLongUsername()) {
		return;
	}
	ignoreList.remove(name);
	Player other = World.getWorld().getPlayerByLongName(name);
	if (other != null) {
		if (privateChat == PrivacySetting.ON) {
			other.getFriends().loggedIn(player);
		}
	}
}

public void loggedIn(Player other) {
	if (friendsList.contains(other.getLongUsername())) {
		if (!onlineFriends.contains(other.getLongUsername())) {
			player.send(new FriendStatusMessage(other.getLongUsername(), 1));
			onlineFriends.add(other.getLongUsername());
		}
	}
}

public void loggedOut(Player other) {
	if (onlineFriends.contains(other.getLongUsername())) {
		player.send(new FriendStatusMessage(other.getLongUsername(), 0));
		onlineFriends.remove(other.getLongUsername());
	}
}

public boolean canSeeMe(Player other) {
	switch (privateChat) {
	case FRIENDS:
		return friendsList.contains(other.getLongUsername());
	case ON:
		return !ignoreList.contains(other.getLongUsername());
	default:
		return false;
	}
}

public void login() {
	for (Player other : World.getWorld().getPlayers()) {
		if (canSeeMe(other)) {
			other.getFriends().loggedIn(player);
		}
	}
}

public void refresh() {
	for (Player other : World.getWorld().getPlayers()) {
		if (canSeeMe(other)) {
			other.getFriends().loggedIn(player);
		} else {
			other.getFriends().loggedOut(player);
		}
	}
}

public void changePrivateChat(int status) {
	if (status > 2 || status < 0 || status == privateChat.ordinal()) {
		return;
	}
	privateChat = PrivacySetting.values()[status];
	refresh();
}

public void changePublicChat(int status) {
	if (status > 3 || status < 0 || status == publicChat.ordinal()) {
		return;
	}
	publicChat = PrivacySetting.values()[status];
}

public void changeTrade(int status) {
	if (status > 2 || status < 0 || status == trade.ordinal()) {
		return;
	}
	trade = PrivacySetting.values()[status];
}

public void loadedLists(List<Long> friendsList, List<Long> ignoreList) {
	this.friendsList = friendsList;
	this.ignoreList = ignoreList;
}

public void loadedPrivateChat(int privateChat) {
	this.privateChat = PrivacySetting.values()[privateChat];
}

public void loadedPublicChat(int publicChat) {
	this.publicChat = PrivacySetting.values()[publicChat];
}

public void loadedTrade(int trade) {
	this.trade = PrivacySetting.values()[trade];
}

public List<Long> getFriendsList() {
	return friendsList;
}

public List<Long> getIgnoreList() {
	return ignoreList;
}

public PrivacySetting getPrivateChat() {
	return privateChat;
}

public PrivacySetting getPublicChat() {
	return publicChat;
}

public PrivacySetting getTrade() {
	return trade;
}

public int getNextMessageTicket() {
	return ++messageTicket;
}
