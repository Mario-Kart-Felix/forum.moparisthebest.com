[code=java]package grandexchange;

import java.util.Date;

public class GEOffer implements Comparable {

public enum OfferType { BUY, SELL }

private final int posterId;
private final OfferType type;
private final int itemId;
private final int price;
private final Date timePosted;
private final int offerAmount;
private boolean cancelled;
private int amountComplete;
private int unclaimedCoins;
private int unclaimedItems;

public GEOffer(int posterId, OfferType type, int itemId, int price, int offerAmount, Date timePosted) {
	this.posterId = posterId;
	this.type = type;
	this.itemId = itemId;
	this.offerAmount = offerAmount;
	this.price = price;
	this.timePosted = timePosted;
}

@Override
public int compareTo(GEOffer compare) {
	if(price == compare.getPrice()) {
		//Oldest offers first
		return timePosted.compareTo(compare.getTimePosted());
	}
	//Buy offers highest price first
	return (price - compare.getPrice()) * (type == OfferType.BUY ? -1 : 1);		
}

public void update(int amount, int tPrice) {
	if(type == OfferType.BUY) {
		int priceDifference = price - tPrice; //If we're buying and they sold for cheaper, give us money back
		unclaimedCoins += priceDifference * amount;
		unclaimedItems += amount;
	} else {
		unclaimedCoins += tPrice * amount;
	}
	amountComplete += amount;
}

public boolean active() {
	return !cancelled && getAmountRemaining() > 0;
}

@Override
public String toString() {
	return "GEOffer: " + type.toString() + " " + amountComplete + "/" + offerAmount + " of " + itemId + " for " + price + " each. Posted: " + (timePosted.getTime() - 1388440000000L)/100 + 
			"; coins: " + unclaimedCoins + "; items: " + unclaimedItems;
}

public int getPosterId() {
	return posterId;
}

public OfferType getType() {
	return type;
}

public int getItemId() {
	return itemId;
}

public int getPrice() {
	return price;
}

public int getOfferAmount() {
	return offerAmount;
}

public Date getTimePosted() {
	return timePosted;
}

public int getAmountComplete() {
	return amountComplete;
}

public int getAmountRemaining() {
	return offerAmount - amountComplete;
}

public void setAmountComplete(int amountComplete) {
	this.amountComplete = amountComplete;
}

	return unclaimedCoins;

package grandexchange;

import grandexchange.GEOffer.OfferType;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Date;
import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;
import java.util.TreeSet;

public class GrandExchange {

	private static Map<Integer, TreeSet<GEOffer>> sellOffers = new HashMap<Integer, TreeSet<GEOffer>>();
	private static Map<Integer, TreeSet<GEOffer>> buyOffers = new HashMap<Integer, TreeSet<GEOffer>>();

	public static void main(String... args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		while (true) {
			System.out.print("Input (buy/sell, id, price, amt): ");
			String input = br.readLine();
			String[] data = input.split(" ");

			placeOffer(OfferType.values()[Integer.parseInt(data[0])], 0, Integer.parseInt(data[1]), Integer.parseInt(data[2]), Integer.parseInt(data[3]), new Date());
			System.out.println("Current buy offers:");
			for (TreeSet<GEOffer> offerSet : buyOffers.values()) {
				Iterator<GEOffer> offers = offerSet.iterator();
				while (offers.hasNext()) {
					System.out.println(offers.next());
				}
			}
			System.out.println("Current sell offers:");
			for (TreeSet<GEOffer> offerSet : sellOffers.values()) {
				Iterator<GEOffer> offers = offerSet.iterator();
				while (offers.hasNext()) {
					System.out.println(offers.next());
				}
			}
		}
	}
	
	public static void placeOffer(OfferType type, int playerId, int itemId, int price, int amt, Date d) {
		GEOffer offer = new GEOffer(playerId, type, itemId, price, amt, d);
		 Map<Integer, TreeSet<GEOffer>> typeOffers = (type == OfferType.BUY ? sellOffers : buyOffers);
		if (typeOffers.containsKey(itemId)) {
			Iterator<GEOffer> offers = typeOffers.get(itemId).iterator();
			while (offers.hasNext() && offer.active()) {
				GEOffer compareOffer = offers.next();
				if ((compareOffer.getPrice() > price && type == OfferType.BUY) || (compareOffer.getPrice() < price && type == OfferType.SELL)) {
					break;
				}
				if (!compareOffer.active()) {
					continue;
				}
				if (compareOffer.getAmountRemaining() >= amt) {
					offer.update(amt, compareOffer.getPrice());
					compareOffer.update(amt, compareOffer.getPrice());
				} else {
					offer.update(compareOffer.getAmountRemaining(), compareOffer.getPrice());
					compareOffer.update(compareOffer.getAmountRemaining(), compareOffer.getPrice());
				}
			}
		}
		addOffer(offer);
	}
	

	public static void addOffer(GEOffer offer) {
		Map<Integer, TreeSet<GEOffer>> offerMap = (offer.getType() == OfferType.BUY ? buyOffers : sellOffers);
		TreeSet<GEOffer> offerList = offerMap.get(offer.getItemId());
		if (offerList == null) {
			offerList = new TreeSet<GEOffer>();
			offerMap.put(offer.getItemId(), offerList);
		}
		offerList.add(offer);
	}

}
