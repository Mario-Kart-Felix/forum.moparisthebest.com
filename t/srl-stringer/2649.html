Const
AmountToDo = 2; // Amount of bows to make.

var
amount,amountleft,amountdone:integer;


Procedure BankingWithdraw;
begin
     if(not (inventorycount=0))then
 begin
  OpenBank3;
  DepositAll;
  BankingWithdraw;
  end else
     begin
     if (not (Bankscreen))
     then
     begin
     OpenBank3;
     Fixbank;
       end else
      Fixbank;
      if (amountleft > 14)then
      begin
      Amount := 14;
      Withdraw(1,1,14);
      Withdraw(2,1,0);
      CloseWindow;
      Wait(random(350)+150);
      DragItem(1,28);
        end  else
         begin
          Amount := AmountLeft
          Withdraw(2,1,1);
          Withdraw(1,1,1);
          wait(100+random(100));
          Withdraw(2,1,(amount-1));
          Withdraw(1,1,(amount-1));
          Closewindow;
          Wait(random(350)+100);
          wait(100 + random(50));
          DragItem(1,28)
          wait(500+random(100));
          DragItem(2,27)
    end;
   end;
 end;


Procedure Stringin;
var Strung : integer;
begin
 MouseSpeed := 12
 Strung := 0
 repeat
 begin
   UseItem(27);
   Wait(300+random(150));
   UseItem(28);
   Wait(300+random(150));
   Strung := Strung + 1
   UseItem(28);
   Wait(300+random(150));
   UseItem(27);
   Strung := Strung + 1
   Wait(300+random(150));
   end;
 until(Strung = Amount);
 amountleft := amountleft -(Amount)
 amountdone := amountdone +(Amount)

end;

FindSandwich;
