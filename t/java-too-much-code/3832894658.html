    public enum Stone {
    COAL(0, 50),
    IRON(1, 100),
    GOLD(2, 200),
    DIAMOND(3, 500);

    private int id;
    private int price;

    private Stone(int id, int price) {
    this.id = id;
    this.price = price;
    }

    public int getId() {
    return id;
    }

    public int getPrice() {
    return price;
    }
