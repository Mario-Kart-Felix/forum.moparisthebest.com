begin
if(TalkingMode = ‘Spam’)or(TalkingMode = ‘spam’)then
Wait(100);

SendKeysVb(’{ENTER}’, false);
if(TalkingMode=‘Human’)or(TalkingMode=‘human’)then

TypeSend(WhatYourGonnaSay);

Wait(100);
