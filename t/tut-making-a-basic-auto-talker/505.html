copy and paste 

'    ///////////////////////////////////////
'   //         Basic Auto-Talker         //
'  //          Coding By Chezy          //
' // Contact thechezman14@hotmail.com  //
'///////////////////////////////////////
Private Sub Command1_Click()
'Sets the time on the auto-talker
Timer1.Interval = Text2.Text * 1000
'Turns on the timer
Timer1.Enabled = True
End Sub
Private Sub Command2_Click()
'Turns off the timer
Timer1.Enabled = False
End Sub
Private Sub Form_Load()
'Changes the tital to "Basic Auto-Talker
Me.Caption = "Basic Auto-Talker"
'Makes so the auto-talker doesnt strat automaticly
Timer1.Enabled = False
'Changes the text in the first textbox to "Your Message Here"
Text1.Text = "You Message Here"
'Changes the text in the second textbox to "7"
Text2.Text = "7"
'Changes the name on the first command button to "Start"
Command1.Caption = "&Start"
'Changes the name on the second command button to "Stop"
Command2.Caption = "S&top"
End Sub
Private Sub Timer1_Timer()
'Type's whatever's in the first textbox
SendKeys Text1.Text
'Works like pressing enter on the keyboard
SendKeys "{enter}"
End Sub
