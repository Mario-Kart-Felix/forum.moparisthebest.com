procedure FindRandoms;
begin
if FindPick=false then
ItemGrab;

FindNormalRandoms;

if(FindFight)then
begin
writeln(‘Fight Fought’)
setrun(true)
RunAwayDirection(‘w’)
wait(10000)
runback;
end;
end;

procedure MineOre;

begin

repeat

MouseSpeed:= 13+random(7);
FindRandoms;
FindPick;

if(not(FindObj(x, y, ‘Mine’, RockColor, 10)))then
begin
Writeln(‘Ore not Found’);
TerminateScript;
end

else
begin
Mouse(x, y, 1, 1, true);
Wait(OreClick * 1000 + random(1000));
end;

until(InvFull)

end;

procedure DropOre(Wield : Boolean);

begin

case Wield of

True: Begin
MouseSpeed:= 13+random(7);
DropTo(1, 14);
FindRandoms;
DropTo(15, 28);
numrocksmined:=numrocksmined+28
loads:=loads+1;
End;

False: Begin
MouseSpeed:= 13+random(7);
DropTo(2, 14);
FindRandoms;
DropTo(15, 28);
numrocksmined:=numrocksmined+27
loads:=loads+1;
End;

end;

end;

Procedure Report;

begin

writeln(‘This is Wall"s Power Miner v1.0’)
writeln('You have mined: ’ + inttostr(numrocksmined))
writeln(‘End Report’)

end;

procedure PowerMine;

begin

MineOre;
DropOre(PickWielded);
Report;

end;

begin
ClearDebug;
ClearReport;
SetupSrl;
SetupSrlMining;
DeclarePlayers;
ActivateClient;

repeat
if(not(LoggedIn))then
LoginPlayer;
repeat
z:=z+1
SendArrowSilentWait(0,1000);
until(z = 5)

if(LoggedIn)then
NickNameBMP := CreateBitmapMaskFromText(Players[CurrentPlayer].Nick, UpChars);

FindPickHeadColor;
chatsoff;

repeat
powermine;
until(loads>=LoadsPerUser)

begin
repeat
Wait(1000);
FindRandoms;
until(not(LoggedIn));

wait(2000)
NextPlayer(True);
end;

if (not(Loggedin)) then NextPlayer(False);
FindPickHeadColor;

until(false)
end.[/scar]
