First, add the Snowflake class to your client:

[code=Java]package com.jagex.runescape.display;

import java.util.Arrays;
import java.util.Collections;
import java.util.List;
import java.util.Random;

import com.jagex.runescape.DrawingArea;

public class Snowflake {

static final int BOTTOM_OF_SCREEN = 329;

public enum FallSpeed {
	SLOW(1),
	MEDIUM(2),
	FAST(3);
	
	private int speed;
	
	FallSpeed(int speed) {
		this.speed = speed;
	}
	
	public int getSpeed() {
		return this.speed;
	}
	
	private static final List<FallSpeed> VALUES 
					= Collections.unmodifiableList(Arrays.asList(values()));
	private static final int SIZE = VALUES.size();
	private static final Random RANDOM = new Random();

	public static FallSpeed randomSpeed()  {
	    return VALUES.get(RANDOM.nextInt(SIZE));
	}
}

int y;
int x;
int size;
int lastTick;
int colour;
int alpha;
FallSpeed speed;
boolean awaitingRemoval;

public Snowflake(int x, int y, int size, int startTick, int colour, int alpha, FallSpeed speed) {
	this.y = x;
	this.x = y;
	this.size = size;
	this.lastTick = startTick;
	this.colour = colour;
	this.alpha = alpha;
	this.speed = speed;
}

public void move() {
	if (this.atBottom())
		return;
	
	Random r = new Random();
	int changeDirection = r.nextInt(9);
	
	if (changeDirection == 1)
		this.x--;
	
	if (changeDirection == 2)
		this.x++;
}

public void fall(int delta) {
	if (this.atBottom())
		return;
	
	y += (delta * speed.getSpeed());
}

public void melt(int delta) {
	this.setAlpha(this.alpha - (delta * 5));
	if (this.getAlpha() == 0)
		this.awaitingRemoval = true;
}

public void draw(int currentTick) {		
	int delta = (currentTick - this.lastTick);
	this.lastTick = currentTick;
	
	this.move();
	this.fall(delta);
	
	if (this.atBottom())
		this.melt(delta);
	
	DrawingArea.drawFilledCircleAlpha(
			this.y, 
			this.x, 
			this.size, 
			this.colour, 
			this.alpha);
}

public int getY() {
	return this.y;
}

public void setY(int y) {
	this.y = y;
}

public int getAlpha() {
	return this.alpha;
}

public void setAlpha(int alpha) {
	if (alpha < 0)
		alpha = 0;
		
	this.alpha = alpha;
}

public boolean atBottom() {
	if (this.getY() >= BOTTOM_OF_SCREEN + this.size) {
		this.setY(BOTTOM_OF_SCREEN + this.size);
		return true;
	}
	return false;
}

public boolean awaitingRemoval() {
	return this.awaitingRemoval;
}

}[/code]

Now head to the renderGameView method in your Client class, and add this above it:

ArrayList<Snowflake> snowflakes; int SNOWFLAKE_FALL_DENSITY = 30;

You can change SNOWFLAKE_FALL_DENSITY to change the density of the snow fall, the lower the number the denser the snow.

Inbetween animateTexture(textureId); and draw3dScreen();, add the following:

[code=Java]if (snowflakes == null)
