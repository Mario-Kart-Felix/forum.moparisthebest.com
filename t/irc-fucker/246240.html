/*
 * Usage:
 * Connect
 * send PASS, USER, NICK, MODE, JOIN

import java.net.Socket;
import java.io.*;

public class ripIRC {
	
	BufferedWriter out = null;
	BufferedReader in = null;
	Socket ircSocket = null;
	
/*
 * The send method sends a message to the Writer
 */
	public void send(String... toWrite) throws IOException{
		for(String write : toWrite){
			out.write(write);
			out.flush();
		}
	}

/*
 * The read method reads the input from the BufferedReader, and prints it to the console
 */
	public String read() throws IOException{
		String input = in.readLine();
		return input;
	}

/*
 * disconnect disconnects from the server
 */	
	public void disconnect() throws IOException{
		ircSocket.close();
	}

/*
 * The connect method connects to the given server, via the given port
 * It also initializes the BufferedWriter and BufferedReader
 */	
	public void connect(String ircServer, int ircPort){
		try{
			ircSocket = new Socket(ircServer, ircPort);
			out = new BufferedWriter(new OutputStreamWriter(ircSocket.getOutputStream()));
			in = new BufferedReader(new InputStreamReader(ircSocket.getInputStream()));
			
		} catch(IOException l){
			System.out.println("Error " + l);
		}
		
	}

/*
 * This method prints the normal messages from IRC
 */	
	public void normalMessageIRC() throws IOException{
		String input = read();
		
		if (input.contains("PRIVMSG")){
			int x = input.indexOf("PRIVMSG") + 7;
			int y = input.indexOf("!");
			String output = input.substring(x, input.length());
			String msgSender = input.substring(1, y);
			System.out.println(msgSender + output);
		}
	}

/*
 * This method prints the raw input from IRC
 */	
	public void rawMessage() throws IOException{
		System.out.println(read());
	}
}
