	int a = p.skillLvl[2];
	int b = p.equipmentBonus[10];
	double c = (double)a;
	double d = (double)b;
	double f = 0;
	double h = 0;
	f = (d*0.00175)+0.1;
	h = Math.floor(c*f+2.05);
	if(p.prayOn[p.strLow]) {
	h *= 1.05;
	}
	if(p.prayOn[p.strMid]) {
	h *= 1.10;
	}
	if(p.prayOn[p.strHigh]) {
	h *= 1.15;
	}
	if(p.prayOn[p.chivalry]) {
	h *= 1.18;
	}
	if(p.prayOn[p.piety]) {
	h *= 1.23;
	}
	return (int) h;
}
