Want a power miner? Here’s a basic one of what you described:[code]
program New;
{.include SRL/SRL.scar}

const
OreToMine = ‘Iron’; //Ore to mine = Tin, Copper, Iron, Coal, Silver, Gold,
//Mithril, Adamantite, Runite (mmust be exact).
Loads = 10; //How many loads to do?

function FindOre: TPoint;
var
x, y: Integer;
begin
MoveMouseSplineEx(MSX1, MSY1, 3, 3, 16, 5, 5);
repeat
GetMousePos(x, y);
MoveMouse(x+1, y);
if(x>MSX2-1) then
MoveMouse(MSX1, y+1);
Until(IsUpTextMultiCustom([‘ine’, OreToMine]));
GetMousePos(x, y);
Result:=IntToPoint(x, y);
end;

procedure HCMBox(x1, y1, x2, y2: Integer; Left: Boolean);
var
i, xr, yr: Integer;
begin
xr:=RandomRange(x1, x2);
yr:=RandomRange(y1, y2);
repeat
HoldMouse(xr-1, yr-1, Left);
Wait(50+Random(51)+Random(51)+Random(21));
ReleaseMouse(xr, yr, Left);
Inc(i);
Until(i>3);
end;

procedure MineOres;
var
tp: TPoint;
begin
repeat
tp:=FindOre;
if((tp.x>MSX1) and (tp.y>MSY1)) then
begin
HCMBox(tp.x-Random(3), tp.y-Random(3), tp.x+Random(3), tp.y+Random(3), True);
tp.x:=MSX1-1;
tp.y:=MSY1-1;
end;
Until(InvCount>29);
end;

var
LC: Integer;

begin
repeat
MineOres;
DropAll;
LC:=LC+1;
Until(LC>Loads);
end.[/code]
