public void process() {
	for (int i = 0; i < MaxObjects; i++) {
		if (ObjectID[i] > -1) {
			if (ObjectDelay[i] > 0) {
				ObjectDelay[i]--;
			}
			if (ObjectDelay[i] == 0) {
				if (ObjectOpen[i] != ObjectOriOpen[i]) {
					for (int j = 0; j < PlayerHandler.maxPlayers; j++) {
						if (PlayerHandler.players[j] != null) {
							PlayerHandler.players[j].ChangeDoor[i] = true;
						}
					}
					ObjectOpen[i] = ObjectOriOpen[i];
				}
			}
		}
	}
}
