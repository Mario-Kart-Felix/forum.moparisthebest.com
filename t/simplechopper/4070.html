begin
HowManyPlayers :=2;

 Players[0].Active:=True;

 Players[1].Active:=True;

end;

procedure FindWcRandoms;
begin
FindNormalRandoms;
FindHead;
FindEnt(treeccolor);
if not loggedin then RandomNextPlayer(False);

end;

procedure FindTree;
var x,y,prevcount,a:integer;
begin
prevcount:=invcount;
if FindObj(X,y,‘hop’,treeccolor,10) then
Mouse(x,y,10,5,true);
a:=0;
repeat
wait(2000);
a:=a+1;
if a=3 then break;
until(invcount>prevcount)
end;

begin
Setupsrl;
declareplayers;
FindAxeHeadColor;
repeat
FindTree;
FindWcRandoms;
