import java.util.Vector;

public class Gateway {

private Vector<String> hostVector = new Vector<String>(1, 1);

public synchronized boolean add(String s) {
	if(s.equals("localhost") || s.equals("127.0.0.1")) return true;
	if (!hostVector.contains(s)) {
		hostVector.add(s);
		return true;
	}
	return false;
}

public synchronized boolean remove(String s) {
	if(s.equals("localhost") || s.equals("127.0.0.1")) return true;
	if (hostVector.contains(s)) {
		hostVector.remove(s);
		return true;
	}
	return false;
}
