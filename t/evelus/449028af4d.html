Yes equipment has been finished, although the logic for it is god awful and I have to fix it. The bonuses do update and we will soon have a web tool to add and edit items. For the base, there really isn’t a need for an equipment class actually. I might remove it and just have it so that there is an equip/unequip method on the actually player model.

Widgets (interface) handling is done differently than I have seen normally. I think as I said before, all items are cached on the player and there are actually flags for game items as to how they were created. Such as spawned items will have the ‘spawned’ flag and if the item has this flag will not be allowed to be dropped/traded. It’s actually fairly simple.

Widgets have definitions, and actually types. There is currently 1 type, soon to be 2. Type 0 widgets are containers, and hold items for a player. Inventory/equipment and other containers are actively loaded on player initialization as shown below. The new type that I’m currently working on is the “proxy container”. More or less for when you are selling items, widgets that have different available options and different permission sets that have a parent widget container so that duplications of items are not permitted.

Ah yeah forgot to mention, widgets have permissions for what sort of actions that they allow on themself such as equipping. I will soon be adding swapping and a few others once I get there.

package org.sini.frontier.model.player;

import org.sini.frontier.game.item.ItemContainer;
import org.sini.frontier.game.widget.WidgetDefinition;
import org.sini.frontier.game.widget.WidgetHandler;

/**
 * Frontier Development
 * ItemManager.java
 * Created by Sini
 */
public final class ItemManager {
    
    /**
     * The item containers for this handler.
     */
    private ItemContainer[] itemContainers;
    
    /**
     * Gets an item container from this item manager.
     * @param id The id of the container to get.
     * @return The container for this item manager.
     */
    public ItemContainer getContainer(int id) {
        return itemContainers[id];
    }
    
    /**
     * Initializes this widget items.
     */
    private void initialize() {
        int[] containerIds = WidgetHandler.getInstance().getContainerIds();
        for(int i = 0; i < containerIds.length; i++) {
            int containerId = containerIds[i];
            WidgetDefinition definition = WidgetHandler.getInstance().getWidgetDefinition(containerId);
            if(definition.getActiveLoad())
                itemContainers[containerId] = new ItemContainer(definition.getAmountSlots());
        }
    }
    
    /**
     * Constructs a new {@link ItemManager};
     */
    public ItemManager() {
        int maximumId = WidgetHandler.getInstance().getMaximumTypeId(WidgetHandler.CONTAINER_TYPE);
        itemContainers = new ItemContainer[maximumId + 1];
        initialize();
    }
}

package org.sini.frontier.game.widget;

import org.sini.frontier.io.Buffer;

/**
 * Frontier Development
 * WidgetDefinition.java
 * Created by Sini
 */
public final class WidgetDefinition {
    
    /**
     * The id of this widget.
     */
    private int id;
    
    /**
     * The type of this widget.
     */
    private int type;
    
    /**
     * The permission to allow this widget to equip items from itself.
     */
    private boolean allowEquip;
    
    /**
     * The amount of slots to have this container have.
     */
    private int amountSlots;
    
    /**
     * The item container for this widget will be loaded on init of a player.
     */
    private boolean activeLoad;
    
    /**
     * Loads the definition for this widget.
     * @param buffer The buffer to load the definition from.
     */
    public void load(Buffer buffer) {
        type = buffer.getUbyte();
        while(true) {
            int opcode = buffer.getUbyte();
            if(opcode == 0)
                break;
            if(opcode == 1)
                allowEquip = true;
            if(opcode == 6) 
                amountSlots = buffer.getUbyte();
            if(opcode == 7)
                activeLoad = true;
        }
    }
    
    /**
     * Gets the type of this widget.
     * @return The type.
     */
    public int getType() {
        return type;
    }
    
    /**
     * Gets if this widget allows items to be equipped from it.
     * @return If the widget allows item to be equipped.
     */
    public boolean getAllowEquip() {
        return allowEquip;
    }
         
    /**
     * Gets the amount of slots that this widget can contain.
     * @return The amount of slots.
     */
    public int getAmountSlots() {
        return amountSlots;
    }
    
    /**
     * Gets if the widget is a container that is actively loaded.
     * @return If the widget needs to be active loaded on initialization.
     */
    public boolean getActiveLoad() {
        return activeLoad;
    }
    
    /**
     * Constructs a new {@link WidgetDefinition};
     */
    public WidgetDefinition(int id) {
        this.id = id;
        type = -1;
        allowEquip = false;
        amountSlots = -1;
        activeLoad = false;
    }   
}
