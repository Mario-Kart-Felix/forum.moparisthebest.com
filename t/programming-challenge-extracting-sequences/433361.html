        if (input.Equals("")) {
            Console.WriteLine("USAGE: num1,num2,...,num3");

        }
        string[] numbers = input.Split(',');

        List<int> sequence = new List<int>();
        List<List<int>> sequences = new List<List<int>>();

        foreach (int n in numbers.Select(Int32.Parse)) {
            if (n % 2 == 0) {
                sequence.Add(n);
            } else if (sequence.Count > 0 && !sequences.Contains(sequence)) {
                sequences.Add(sequence);
                sequence = new List<int>();
            }
        }

        if (sequence.Count > 0) {
            sequences.Add(sequence);
        }

        sequences.Sort((seq1, seq2) => seq1.Sum() > seq2.Sum() ? -1 : (seq1.Sum() == seq2.Sum() ? 0 : 1));

        foreach (string res in sequences.Select(s => s.Aggregate("", (current, number) => current + (" " + number)))) {
            Console.WriteLine(res.Substring(1));
        }
        Console.ReadKey();
    }
}

[code=haskell]import Data.List

{-

  • Write an algorithm that takes an integer array as input and outputs an array of arrays.
  • Each array in the output array should represent a sequence of even integers in the input array.
  • The output array has to be sorted (sorted sequences) in descending order, the value of a sequence is the sum of it’s elements.
    -}

build :: Integral a => [a] -> [[a]]
build l = let res = (takeWhile even . dropWhile odd) l in
if null (tail res) then [res]
else res : build (tail l \ res)

sortlist :: Integral a => [[a]] -> [[a]]
sortlist = sortBy $ \a b -> compare (sum b) (sum a)

extract :: Integral a => [a] -> [[a]]
extract = sortlist . build

main :: IO ()
main = mapM_ print $ extract [1, 3, 2, 4, 10, 8, 9, 9, 6, 2, 5, 10, 200, 4, 8, 3, 0, 2000, 1, 10000][/code]

Hackish (fast) Java solution:

[code=java]import java.util.Collection;
import java.util.Deque;
import java.util.ArrayDeque;
import java.util.Map;
import java.util.IdentityHashMap;
import java.util.Set;
import java.util.TreeSet;
import java.util.Comparator;
import java.util.Random;

class SequenceExtractor {

static Collection<Deque<Integer>> extract(int[] input) {
    final int n = input.length;

    Set<Deque<Integer>> seqs = new TreeSet<Deque<Integer>>(new Comparator<Deque<Integer>>() {
        private final Map<Deque<Integer>, Integer> tab = new IdentityHashMap<Deque<Integer>, Integer>(n);

        private int sum(Deque<Integer> seq) {
            Integer lookup = tab.get(seq);
            if (lookup == null) {
                int sum = 0;
                for (Integer val : seq) {
                    sum += val.intValue();
                }

                tab.put(seq, sum);
                return sum;
            } else {
                return lookup.intValue();
            }
        }

        public int compare(Deque<Integer> u, Deque<Integer> v) {
            return sum(v) - sum(u);
        }
    });

    int cur = 0;
    boolean state = false;
    Deque<Integer> seq = null;
    while (cur < n) {
        int in = input[cur++];

        if ((in & 1) == 0) {
            if (!state) {
                state = true;
                seq = new ArrayDeque<Integer>(n);
            }
            seq.add(in);
        } else if (state) {
            state = false;
            seqs.add(seq);
        }
    }

    if (state) {
        seqs.add(seq);
    }

    return seqs;
}

static void print(Collection<Deque<Integer>> results) {
    for (Deque<Integer> seq : results) {
        StringBuilder builder = new StringBuilder();
        for (Integer val : seq) { 
            builder.append(val + " ");
        }
        System.out.print(builder.append("\n").toString());
    }
}

public static void main(String[] argv) {
    final int[] input = new int[] { 1, 3, 2, 4, 10, 8, 9, 9, 6, 2, 5, 10, 200, 4, 8, 3, 0, 2000, 1, 10000 };
    final int n = input.length;

    final Random rand = new Random();
    long dur = 0;

    Collection<Deque<Integer>> rsl;

    for (int i = 0; i < 10000; ++i) {
        input[rand.nextInt(n)]++;

        long start = System.nanoTime();

        rsl = extract(input);

        dur += (System.nanoTime() - start);

        print(rsl);
    }

    System.out.printf("avg period per extraction: %dns%n", dur / 10000);
}
