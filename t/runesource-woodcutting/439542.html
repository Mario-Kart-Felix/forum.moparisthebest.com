[shadow=cyan,left][size=24pt]Woodcutting for RuneSource[/size][/shadow]

Contains:

  • All regular trees
  • Checking axe for required level
  • Cutting timer influenced by type of tree, woodcutting level and used axe.

Read before criticism:

  • Yesterday was the first time ever I looked into runesource.
    I come from delta, so it would be highly unlikely is everything was already perfectly done to “runesource norms”
  • This has no stumps, since this is just for a private server and not for a remake. It should be that way.
  • Criticism is highly appreciated since I want to learn to do it better.[/center]

[size=18pt]Step 1:[/size]
[b]Open up your ObjectPacketHandler class and locate to you 1-click-object method.

Here we will add a check, to see if the clicked object is a tree.
If it is it will check for the object’s(tree) data, and start cutting.

So, add this to it:[/b]

for (int tree : player.getWoodcutting().allTrees) { if (player.getClickId() == tree) { player.getWoodcutting().getTreeData(player.getClickId()); } }

Your method should now look like this:

	private void handleFirstClick(Player player, Packet packet) {
		player.setClickX(packet.getIn().readShort(true, StreamBuffer.ValueType.A, StreamBuffer.ByteOrder.LITTLE));
		player.setClickId(packet.getIn().readShort());
		player.setClickY(packet.getIn().readShort(StreamBuffer.ValueType.A));
		WalkToActions.setActions(Actions.OBJECT_FIRST_CLICK);
		WalkToActions.doActions(player);
		System.out.println(player.getClickId());
		for (int tree : player.getWoodcutting().allTrees) {
			if (player.getClickId() == tree) {
				player.getWoodcutting().getTreeData(player.getClickId());
			}
		}
		
	}

[size=18pt]Step 2:[/size]
Open up your player class and add this instance.

private Woodcutting woodcutting = new Woodcutting(this);

and add this return

public Woodcutting getWoodcutting() {
        return woodcutting;
	}

[size=18pt]Step 3:[/size]
[b]Now we will make, that when our player walks away, the woodcutting process doesn’t continue.

So open up your MovementHandler class and locate to your “resetOnWalkPacket” method,
and add this to it:[/b]

player.getWoodcutting().isCutting = false;
player.getWoodcutting().receivedData = false;

Your method should now look like this:

	public void resetOnWalkPacket() {
		player.getWoodcutting().isCutting = false;
		player.getWoodcutting().receivedData = false;
		player.getAttributes().put("isBanking", Boolean.FALSE);
		player.getAttributes().put("isShopping", Boolean.FALSE);
		player.getActionSender().removeInterfaces();
		player.getUpdateFlags().faceEntity(65535);
		TradeManager.handleWalkAway(player);
		player.getAttackHandler().resetAttackOnMove();
		player.setFollowingCombat(null);
	}

[size=18pt]Step 4:[/size]
[b]Now we can add our woodcutting base.

Create a new class preferably called “Woodcutting.java” to wherever your skills are located.
(You might have to change the package declaration to yours.)
and add this to it: [size=7pt]explanation in code[/size][/b]

package com.rs2.model.content.skills;

import com.rs2.model.World;
import com.rs2.model.players.Item;
import com.rs2.model.players.Player;
import com.rs2.model.players.Skill;
import com.rs2.model.players.container.Container;
import com.rs2.model.tick.Tick;

public class Woodcutting {
	
	/**
	 * @Author: RandQm
	 * @Date: 12 April 2012
	 * @Info: Woodcutting base
	 */
	
	final Player player;
	
	public Woodcutting (Player player) {
		this.player = player;
	}

	/*
	 * This boolean will check if the player is currently cutting.
	 */
	public boolean isCutting = false;
	
	/*
	 * Checking if the player has received the data of a tree before cutting.
	 */
	public boolean receivedData = false;
	
	/*
	 * Controlling how much woodcutting experience will be added when cut 1 log.
	 */
	private double expAdded;
	
	/*
	 * Handling the time to cut a log according to the tree type.
	 */
	private double defaultTimer;
	
	/*
	 * Handling the influence by the used axe on the woodcutting timer.
	 */
	private double axeInfluence;
	
	/*
	 * The type of log gained from the tree. (the log id)
	 */
	private int logType;
	
	/*
	 * An array of all cutable trees.
	 */
	public int[] allTrees = {1276, 1277, 1278, 1280, 1330, 1332, 3033, 3034, 3035, 3036, 3879, 3881, 3882, 3883
		, 1315, 1316, 1318, 1319, 1282, 1283, 1284, 1285, 1286, 1287, 1289, 1290, 1291, 1365, 1383, 1384, 5902, 5903, 5904
		, 3037, 1281, 1308, 5551, 5552, 5553, 1307, 4674, 1309, 1306};
	
	
	/*
	 * Controlling the data for our trees.
	 * Checking the array for the type of tree and if the player has
	 * the required level to cut it. If he does, setting the data for the type of tree.
	 */
	public void getTreeData(int treeId) {	
		if (receivedData == false) {
			
			int[] normalTrees = {
					1276, 1277, 1278, 1280, 1330, 1332, 3033, 3034, 3035, 3036, 3879, 3881, 3882, 3883 //Normal tree
					, 1315, 1316, 1318, 1319 //Evergreen
					, 1282, 1283, 1284, 1285, 1286, 1287, 1289, 1290, 1291, 1365, 1383, 1384, 5902, 5903, 5904 //Dead tree
					};
			for (int normal : normalTrees) {
				if (treeId == normal) {
					if (player.getSkill().getLevelForXP(player.getSkill().getExp()[Skill.WOODCUTTING]) < 1)
						player.getActionSender().sendMessage("You need a woodcutting level of at least 1 to cut this tree.");
					else
						expAdded = 25;
					logType = 1511;
					defaultTimer = 6;
				}
			}	
			
			int[] oakTrees = {3037, 1281};
			for (int oak : oakTrees) {
				if (treeId == oak) {
					if (player.getSkill().getLevelForXP(player.getSkill().getExp()[Skill.WOODCUTTING]) < 15)
						player.getActionSender().sendMessage("You need a woodcutting level of at least 15 to cut oak.");
					else
						expAdded = 37.5;
					logType = 1521;
					defaultTimer = 7;
				}
			}
			
			int[] willowTrees = {1308, 5551, 5552, 5553};
			for (int willow : willowTrees) {
				if (treeId == willow) {
					if (player.getSkill().getLevelForXP(player.getSkill().getExp()[Skill.WOODCUTTING]) < 30)
						player.getActionSender().sendMessage("You need a woodcutting level of at least 30 to cut this willow.");
					else
						expAdded = 67.5;
					logType = 1519;
					defaultTimer = 8;
				}
			}
			
			int[] mapleTrees = {1307, 4674};
			for (int maple : mapleTrees) {
				if (treeId == maple) {
					if (player.getSkill().getLevelForXP(player.getSkill().getExp()[Skill.WOODCUTTING]) < 45)
						player.getActionSender().sendMessage("You need a woodcutting level of at least 45 to cut this maple.");
					else
						expAdded = 100;
					logType = 1517;
					defaultTimer = 9;
				}
			}
			
			if (treeId == 1309) { //Yew
				if (player.getSkill().getLevelForXP(player.getSkill().getExp()[Skill.WOODCUTTING]) < 60)
					player.getActionSender().sendMessage("You need a woodcutting level of at least 60 to cut yew");
				else
					expAdded = 175;
				logType = 1515;
				defaultTimer = 10;
			}

			if (treeId == 1306) { //Magic
				if (player.getSkill().getLevelForXP(player.getSkill().getExp()[Skill.WOODCUTTING]) < 75)
					player.getActionSender().sendMessage("You need a woodcutting level of at least 75 to cut magic.");
				else
					expAdded = 250;
				logType = 1513;
				defaultTimer = 11.5;
			}
			receivedData = true;
		}
		startWoodcutting();
	}
	
	
	/*
	 * Checking if we got the data for the tree, if we do, start cutting the tree.
	 * Every random amount of ticks logs and woodcutting experience get added, as long the player is cutting.
	 */
	private void startWoodcutting() {
		if (receivedData == true) {	
			if (!hasAxeEquipped()) {
				receivedData = false;
				return;
			} else {
				final double totalTimer = defaultTimer - levelInfluence() - axeInfluence;
				isCutting = true;
				player.getActionSender().sendMessage("You swing your axe at the tree...");
				player.getUpdateFlags().sendAnimation(875, 0);
				World.submit(new Tick( (int)totalTimer ) {
					@Override
					public void execute() {
						if (isCutting == true) {
							if (player.getInventory().getItemContainer().freeSlot() == -1) {
								player.getActionSender().sendMessage("Not enough space in your inventory.");
								receivedData = false;
								isCutting = false;
								this.stop();
							} else {
								player.getUpdateFlags().sendAnimation(875, 0);
								player.getActionSender().sendMessage("You get some logs.");
								player.getSkill().addExp(Skill.WOODCUTTING, expAdded);
								player.getInventory().addItem(new Item(logType, 1));
							}
						} else {
							player.getUpdateFlags().reset();
							receivedData = false;
							isCutting = false;
							this.stop();
						}
					}
				});
			}
		}
	}
	
	
	/*
	 * Checking for the used axe, required level and
	 * setting the influence on the timer for the equipped axe.
	 */
	private boolean hasAxeEquipped() {
		int wcLvl = player.getSkill().getLevelForXP(player.getSkill().getExp()[Skill.WOODCUTTING]);
		Container equip = player.getEquipment().getItemContainer();

		if (equip.contains(6739)) {
			if (wcLvl < 61) {
				player.getActionSender().sendMessage("You need a woodcutting level of at least 61 to use a dragon axe.");
				return false;
			} else
				axeInfluence = 3.05;
			return true;
		}
		else if (equip.contains(1359)) {
			if (wcLvl < 41) {
				player.getActionSender().sendMessage("You need a woodcutting level of at least 41 to use a rune axe.");
				return false;
			} else
				axeInfluence = 2.2;
			return true;
		}
		else if (equip.contains(1357)) {
			if (wcLvl < 31) {
				player.getActionSender().sendMessage("You need a woodcutting level of at least 31 to use an adamant axe.");
				return false;
			} else
				axeInfluence = 1.65;
			return true;
		}
		else if (equip.contains(1355)) {
			if (wcLvl < 21) {
				player.getActionSender().sendMessage("You need a woodcutting level of at least 21 to use a mithril axe.");
				return false;
			} else
				axeInfluence = 1.2;
			return true;
		}
		else if (equip.contains(1361)) {
			if (wcLvl < 11) {
				player.getActionSender().sendMessage("You need a woodcutting level of at least 11 to use a black axe.");
				return false;
			} else
				axeInfluence = 0.75;
			return true;
		}
		else if (equip.contains(1353)) {
			if (wcLvl < 6) {
				player.getActionSender().sendMessage("You need a woodcutting level of at least 6 to use a steel axe.");
				return false;
			} else
				axeInfluence = 0.4;
			return true;
		}
		else if (equip.contains(1351)) {
			axeInfluence = 0.15;
			return true;
		}
		else if (equip.contains(1349)) {
			axeInfluence = 0;
			return true;
		} else {
			player.getActionSender().sendMessage("You need to equip an axe to cut trees.");
			return false;
		}
	}
	
	
	/*
	 * Calculating the influence on the woodcutting timer according to the players woodcutting level.
	 */
	double levelInfluence() {
		int wcLvl = player.getSkill().getLevelForXP(player.getSkill().getExp()[Skill.WOODCUTTING]);
		if (wcLvl < 5)
			return 0;
		else if (wcLvl < 10 && wcLvl >= 5)
			return 0.04;
		else if (wcLvl < 15 && wcLvl >= 10)
			return 0.09;
		else if (wcLvl < 20 && wcLvl >= 15)
			return 0.15;
		else if (wcLvl < 25 && wcLvl >= 20)
			return 0.22;
		else if (wcLvl < 30 && wcLvl >= 25)
			return 0.3;
		else if (wcLvl < 35 && wcLvl >= 30)
			return 0.39;
		else if (wcLvl < 40 && wcLvl >= 35)
			return 0.49;
		else if (wcLvl < 45 && wcLvl >= 40)
			return 0.6;
		else if (wcLvl < 50 && wcLvl >= 45)
			return 0.72;
		else if (wcLvl < 55 && wcLvl >= 50)
			return 0.85;
		else if (wcLvl < 60 && wcLvl >= 55)
			return 0.99;
		else if (wcLvl < 65 && wcLvl >= 60)
			return 1.14;
		else if (wcLvl < 70 && wcLvl >= 65)
			return 1.3;
		else if (wcLvl < 75 && wcLvl >= 70)
			return 1.47;
		else if (wcLvl < 80 && wcLvl >= 75)
			return 1.65;
		else if (wcLvl < 85 && wcLvl >= 80)
			return 1.84;
		else if (wcLvl < 90 && wcLvl >= 85)
			return 2.04;
		else if (wcLvl < 95 && wcLvl >= 90)
			return 2.25;
		else if (wcLvl < 99 && wcLvl >= 95)
			return 2.47;
		else if (wcLvl >= 99)
			return 2.72;
		else
			return 0;
	}

}
