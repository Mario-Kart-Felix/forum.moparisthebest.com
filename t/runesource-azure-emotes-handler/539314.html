[code=java]package net.pure2.poseidon.player;

import net.pure2.poseidon.Client;

/**
*

  • @author Pure_
    */
    public final class Emotes {

    private static final int EMOTE_WAIT_CYCLE = 1000;

    /**

    • Storing all of our emotes.
      */
      public enum Emote {
      YES(855, 168),
      NO(856, 169),
      THINK(857, 162),
      BOW(858, 164),
      ANGRY(859, 165),
      CRY(860, 161),
      LAUGH(861, 170),
      CHEER(862, 171),
      WAVE(863, 163),
      BECKON(864, 167),
      CLAP(865, 172),
      DANCE(866, 166),
      PANIC(2105, 52050),
      JIG(2106, 52051),
      SPIN(2107, 52052),
      HEAD_BANG(2108, 52053),
      JOY_JUMP(2109, 52054),
      RASPBERRY(2110, 52055),
      YAWN(2111, 52056),
      SALUTE(2112, 52057),
      SHRUG(2113, 52058),
      BLOW_KISS(1368, 43092),
      GLASS_BOX(1131, 2155),
      CLIMB_ROPE(1130, 52103),
      LEAN(1129, 52106),
      GLASS_WALL(1128, 2154),
      GOBLIN_BOW(2127, 52071),
      GOBLIN_DANCE(2128, 52072),
      SCARED(2836, 59062),
      ZOMBIE_WALK(3544, 72032),
      ZOMBIE_DANCE(3543, 72033),
      RABBIT_HOP(3866, 72254);

      private int animationId;
      private int buttonId;

      private Emote(int animationId, int buttonId) {
      this.animationId = animationId;
      this.buttonId = buttonId;
      }

      /**

      • Fetches an Emote.
        */
        public Emote getEmote(int buttonId) {
        for (Emote emote : Emote.values()) {
        if (emote.buttonId == buttonId) {
        return emote;
        }
        }
        return null;
        }

      public int getAnimationId() {
      return animationId;
      }
      }

    /**

    • Attempts to perform an emote.

    • @param c the Client to update

    • @param buttonId the button ID

    • @return if the animation was successfully started
      */
      public static boolean performEmote(Client c, int buttonId) {
      Emote emote = Emote.getEmote(buttonId);

      if (emote != null) {
      c.startAnimation(emote.getAnimationId(), EMOTE_WAIT_CYCLE);
      return true;
      }
      return false;
      }
      }[/code]
