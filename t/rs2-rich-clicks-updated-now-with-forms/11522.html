Click8Y:=334 //y position for eighth click
Click9X:=845 //x position for ninth click
Click9Y:=334 //y position for ninth click
Click10X:=845 //x position for tenth click
Click10Y:=334 //y position for tenth click
Waittime:=25 //Time in MS between clicks
Waittimeextra:=50 //Random extra wait time in MS
Clickrepeat:=1 //How many times it will repeat your click sequence.
End;

Procedure Clicking;
Begin
Stop:=0
Repeat
Stop:=Stop+1
Mouse(Click1X,Click1Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click2X,Click2Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click3X,Click3Y,Clicktol,Clicktol,true)
