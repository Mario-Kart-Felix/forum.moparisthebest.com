I hope you learn better by example:

import java.io.IOException;
import java.io.InputStream;
import java.util.List;
import java.util.ArrayList;

import javax.xml.parsers.DocumentBuilder;
import javax.xml.parsers.DocumentBuilderFactory;
import javax.xml.parsers.ParserConfigurationException;

import org.w3c.dom.Document;
import org.w3c.dom.Element;
import org.w3c.dom.Node;
import org.w3c.dom.NodeList;
import org.xml.sax.SAXException;

/**
 * Provides static tools for reading and parsing
 * an XML file to a Book object.
 */
public class BookReader {

	/**
	 * Since this class is intended to be used
	 * in a static context, allow no instances
	 * of this class.
	 */
	private BookReader() {
	}

	/**
	 * Parse a new Book object based on the following
	 * XML format:
	 * 
	 * <book>
	 * 		<person>
	 * 			<first>Melvin</first>
	 * 			<last>Purvis</last>
	 * 			<age>56</age>
	 * 		</person>
	 * 		<person>
	 * 			<first>Evelyn</first>
	 * 			<last>Frechette</last>
	 * 			<age>62</age>
	 * 		</person>
	 * 		<person>
	 * 			<first>John</first>
	 * 			<last>Dillinger</last>
	 * 			<age>31</age>
	 * 		</person>
	 * </book>
	 * 
	 * @param in The source of input for XML that should be parsed into Book format.
	 * @return A Book object based on the supplied data stream.
	 * @throws SAXException When an XML parsing or operation error occurs.
	 * @throws ParserConfigurationException As specified by its documentation, it will
	 * only be thrown when a serious configuration error occurs.
	 * @throws IOException When an IO error occurs while reading the supplied input source.
	 */
	public static final Book parseBook(InputStream in) throws SAXException, ParserConfigurationException, IOException {
		DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
		DocumentBuilder builder = factory.newDocumentBuilder();
		Document doc = builder.parse(in);
		Element root = doc.getDocumentElement();
		root.normalize();
		if (!root.getNodeName().equals("book"))
			throw new IllegalStateException("Expected root node value of \"book\"");
		NodeList persons = root.getElementsByTagName("person");
		List<Person> people = new ArrayList<Person>();
		for (int i = 0; i < persons.getLength(); i++) {
			Node item = persons.item(i);
			if (item.getNodeType() == Node.ELEMENT_NODE) {
				NodeList children = item.getChildNodes();
				String firstName = null, lastName = null, age = null;
				for (int j = 0; j < children.getLength(); j++) {
					Node child = children.item(j);
					String name = child.getNodeName();
					if (name.equals("first"))
						firstName = child.getChildNodes().item(0).getNodeValue();
					else if (name.equals("last"))
						lastName = child.getChildNodes().item(0).getNodeValue();
					else if (name.equals("age"))
						age = child.getChildNodes().item(0).getNodeValue();
					child = null;
					name = null;
				}
				people.add(new Person(firstName, lastName, age));
				children = null;
				firstName = lastName = age = null;
			}
			item = null;
		}
		Book book = new Book(people.toArray(new Person[0]));
		people.clear();
		return book;
	}
}

/**
 * Represents a Book object as parsed
 * from the XML file.
 */
class Book {

	private final Person[] people;

	/**
	 * Construct a new Book object with the
	 * specified Person objects.
	 */
	Book(Person... persons) {
		this.people = persons;
	}

	/**
	 * Get the Person objects stored within
	 * this book instance.
	 */
	public Person[] getPeople() {
		return people;
	}
}

/**
 * Represents a person inside of a Book
 * object.
 */
class Person {

	private final String firstName, lastName, age;

	/**
	 * Construct a new Person object with arguments
	 * firstName, lastName, and age.
	 */
	Person(String firstName, String lastName, String age) {
		this.firstName = firstName;
		this.lastName = lastName;
		this.age = age;
	}

	/**
	 * Get this person's first name.
	 * 
	 * Note: this value can be null.
	 */
	public String getFirstName() {
		return firstName;
	}

	/**
	 * Get this person's last name.
	 * 
	 * Note: this value can be null.
	 */
	public String getLastName() {
		return lastName;
	}

	/**
	 * Get this person's age.
	 * 
	 * Note: this value can be null.
	 */
	public String getAge() {
		return age;
	}
}
