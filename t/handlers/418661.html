This is more or less what you want to do. You do need to make the variable static though because its value needs to be the same throughout all the classes. This will cut out all that crap with making static methods though.

[code=java5]import java.util.Scanner;

public class Test {

static boolean right;
String test;

public static void main(String[] args) {

    Test t = new Test();
    t.start();

}

public void start() {

    Scanner scan = new Scanner(System.in);
    test = scan.nextLine();

    right = scan.nextBoolean();

    Test2 t2 = new Test2();
    t2.test2();

}

public boolean getRight() {

    return right;

}

public void setRight(boolean value) {

    right = value;

}

}[/code]

[code=java5]public class Test2 {

Test t = new Test();
boolean test;

public void test2() {

    test = t.getRight();

    System.out.println(test);

    test = false;

    t.setRight(test);

    System.out.println(t.getRight());

}

}[/code]
