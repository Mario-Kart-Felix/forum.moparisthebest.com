Installation Instructions

  1. Copy [ code ] into a text file.
  2. Save the file into a location which is memorable. call the file MMusicDownloader.java
  3. locate the file using file browser, terminal or cmd.
  4. depending on your OS, for windows create a batch file such that it contains (Right Click -> New -> Text File, Then save as run.bat)
@echo off
javac -cp . MMusicDownloader.java
java -cp . MMusicDownloader
pause

for linux, use line 2 and 3 in terminal!

  1. Run by double clicking run.bat
  2. Enter the friendID of the artist (obtained from myspace)
  3. Wait for the download :), downloaded will be placed in ./downloaded-myspace/artist/

import java.io.;
import java.net.URL;
import java.net.URLConnection;

/**

  • @author Oliver

    public class MMusicDownloader {

    public static void main(String… args) {
    try {
    new MMusicDownloader();
    } catch (IOException e) {
    e.printStackTrace();
    } catch (InterruptedException e) {
    e.printStackTrace();
    }
    }

    MMusicDownloader() throws IOException, InterruptedException {
    String friendid = JOptionPane.showInputDialog(“Enter FriendID of Music Artist”);
    try {
    download(“http://mediaservices.myspace.com/services/media/musicplayerxml.ashx?b=” + friendid, “music”);
    } catch (Exception e) {
    System.out.println(“Invalid FriendID”);
    return;
    }
    BufferedReader br = new BufferedReader(new FileReader(“music”));
    String ashxFile = “”;
    for (String line = br.readLine(); line != null; line = br.readLine())
    ashxFile += line;
    String artistName = ashxFile.substring(ashxFile.indexOf("")).replaceAll("\*", “”).replaceAll("\/", “”).replaceAll(""", “”).replaceAll("\?", “”).replaceAll("\<", “”).replaceAll("\>", “”);
    System.out.println(“Grabbing " + artistName + “'s Songs”);
    File artistDirectory = new File(“downloaded-myspace/” + artistName);
    artistDirectory.mkdirs();
    ashxFile = ashxFile.substring(ashxFile.indexOf(”") + 10, ashxFile.indexOf(""));
    String[] songs = ashxFile.split("<song “);
    for (String song : songs) {
    if (song.equals(”"))
    continue;
    String songURL = song.substring(song.indexOf(“durl=”") + 6, song.indexOf("" token"));
    String songName = song.substring(song.indexOf(“title=”") + 7, song.indexOf("" songid")).replaceAll(“â€™”, “’”).replaceAll("’", “”);
    download(songURL, artistDirectory.getAbsolutePath() + “/” + artistName + " - " + songName + (songName.endsWith(".mp3") ? “” : “.mp3”));
    }
    JOptionPane.showMessageDialog(null, "Finished Downloading " + artistName);
    }

    public static void download(String url, String fileName) {
    try {
    URLConnection connection = new URL(url).openConnection();
    File file = new File(fileName);
    int length = connection.getContentLength();
    System.out.println(“Saving " + file + " (which is " + connection.getContentType() + “:” + length + " bytes)…”);
    InputStream instream = connection.getInputStream();
    FileOutputStream outstream = new FileOutputStream(file);
    int size = 0, copy = 0;
    while ((copy = instream.read()) != -1) {
    outstream.write(copy);
    size++;
    }
    if (length != size) {
    instream.close();
    outstream.close();
    } else {
    System.out.println(“Successfully loaded " + file + “:” + size + " bytes.”);
    instream.close();
    outstream.close();
    }
    } catch (Exception e) {
    e.printStackTrace();
    }
    }
