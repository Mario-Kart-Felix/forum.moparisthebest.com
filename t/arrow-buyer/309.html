Ok, to run this script you need SRL, it will buy arrows from Lowe in varrock and stop when there are 0 in stock or if you log out.

[code]{.include SRL\SRL.scar}

var
DTM: Integer;

begin
DTM := DTMFromString(‘78DA63BCCAC0C0D0CA8002FEFF87D08C503EE’ +
‘33520D14840CD6520514740CD5522D45C02120D44A869C1AF0600’ +
‘D06113FE’);
repeat
if(FindColor(x, y, 2070783, 298, 87, 481, 104))then
repeat
Mouse(226, 139, 5, 5, false)
Mouse(193, 207, 8, 8, true)
until(FindDtmIn(x, y, DTM, 200, 119, 245, 159))
if(not LoggedIn)then Exit;
until(false)
end.[/code]

const
AN = 0; // 0 = bronze, 1 = iron, 2 = steel, 3 = mith, 4 = addy

var
X, Y: Integer;

procedure GetArrows;
begin
if(FindColor(x, y, 2070783, 0, 0, 765, 505))then
repeat
if(AN=0)then //Bronze
repeat
MouseC(94+random(5), 84+random(5), 2)
