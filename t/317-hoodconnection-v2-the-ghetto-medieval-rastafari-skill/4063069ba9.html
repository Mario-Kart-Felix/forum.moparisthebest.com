package server.model.npc;

import server.*;
import server.events.Event;
import server.model.player.Client;
import server.model.player.Player;
import server.model.player.Packet;
import server.events.EventContainer;
import server.events.EventManager;
import server.util.Misc;
import server.world.AnimationManager;
import server.model.player.PlayerConstants;

public class NPCvsNPC {

	public static boolean NPCAttackNPC(final NPC thisNPC, final NPC targetNPC){

		EventManager.getSingleton().addEvent(thisNPC, new Event() {

			@Override
			public void execute(EventContainer c) {
				if(targetNPC == null || targetNPC.isDead || thisNPC.isDead){
					c.stop();
					return;
				}

				tick(c, targetNPC, thisNPC);
			}

			@Override
			public void stop() {
				thisNPC.isBusyFollowNpc = false;
				thisNPC.freezeTimer = 0;
			}

		}, 2500);
	return false;
	}

		private static void tick(EventContainer c, final NPC targetNPC, final NPC thisNPC) {
			thisNPC.faceToNpc(targetNPC.getNpcId());
				thisNPC.setAnimNumber(AnimationManager.getDefendAnimation(targetNPC));
				int hitx = Misc.random(thisNPC.getDefinition().getMaximumHit());
		if(Misc.random(thisNPC.getDefinition().getCombat()) < 40+Misc.random(targetNPC.getDefinition().getCombat())) {
				hitx = 0;
		}

		if(thisNPC.getDefinition().MagicNpc() || thisNPC.getDefinition().RangeNpc()){
		if(thisNPC.getDefinition().getMagicProjectile() != 0){
		int pX = thisNPC.getAbsX();
		int pY = thisNPC.getAbsY();
		int nX = targetNPC.getAbsX();
		int nY = targetNPC.getAbsY();
		int offX = (pY - nY)* -1;
		int offY = (pX - nX)* -1;	
		for(Player p : Server.getPlayerManager().getClientRegion(10002)){
			if(p != null) {
				((Client) p).getActionAssistant().createPlayersProjectile(pY, pX, offY, offX, thisNPC.getHeightLevel(), 50, 30, thisNPC.getDefinition().getMagicProjectile(), 43, 31, targetNPC.npcId+1, 15);
			}
		}
		}
		}
				targetNPC.hitNPC(hitx, 0);
				thisNPC.setAnimNumber(AnimationManager.getAttackAnimation(thisNPC));
				thisNPC.setAnimUpdateRequired(true);
				thisNPC.setUpdateRequired(true);
		}

	public static int getAttackDistance(NPC npc){
			if(npc.getDefinition().RangeNpc() || npc.getDefinition().MagicNpc()){
				return 7;
			}
			return 1;
	}
	public static int getDistanceToStop(NPC npc){
			if(npc.getDefinition().RangeNpc() || npc.getDefinition().MagicNpc()){
				return 7;
			}
			return 15;
	}

	public static boolean SPINNERS(final NPC thisNPC, final NPC targetNPC){
		EventManager.getSingleton().addEvent(thisNPC, new Event() {

			@Override
			public void execute(EventContainer c) {
				if(targetNPC == null || targetNPC.isDead || thisNPC.isDead){
					c.stop();
					return;
				}

				tickSPINNERS(c, targetNPC, thisNPC);	

			}

			@Override
			public void stop() {
				thisNPC.abcdfe = false;
				thisNPC.isBusyFollowNpc = false;
				thisNPC.freezeTimer = 0;
			}

		}, 5000);
	return false;
	}
		private static void tickSPINNERS(EventContainer c, final NPC targetNPC, final NPC thisNPC) {
			if(targetNPC.hp < 200){
				targetNPC.hp += Misc.random(25);
				thisNPC.setAnimNumber(AnimationManager.getAttackAnimation(thisNPC));
				thisNPC.setAnimUpdateRequired(true);
				thisNPC.setUpdateRequired(true);
			}
		}

}
