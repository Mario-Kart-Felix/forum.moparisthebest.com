public boolean ApplyDead() { if(IsDeadTimer == false){ actionAmount++; actionTimer = 20; ResetAttack(); ResetAttackNPC(); setAnimation(0x900); sendMessage("You have died."); removeAllItems(); IsDeadTimer = true; } if(actionTimer == 0 && IsDeadTimer == true){ teleportToX = 3254; teleportToY = 3420; IsDeadTeleporting = true; playerLevel[playerHitpoints] = getLevelForXP(playerXP[playerHitpoints]); setSkillLevel(playerHitpoints, playerLevel[playerHitpoints], playerXP[playerHitpoints]); resetAnimation(); updateRequired = true; appearanceUpdateRequired = true; } return true; }
