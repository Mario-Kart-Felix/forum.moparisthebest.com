            World.getWorld().register(npc);
            npc.getCombatState().startAttacking(player, false);

        }

        World.getWorld().register(TZ_KEK_SPAWN2);
    }

                spawnNextWave(false);
            }
        } else if (victim.isPlayer()) {
            exit((Player) victim);
        }
