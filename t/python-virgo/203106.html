import Pythirc
import sys
import re
from time import sleep

print "Pythirc - Welcome \n"
connection = Pythirc.IRCon('irc.deltaanime.net', 6667)
IRCFunc = Pythirc.IRCFun()
IRCFunc.identify("Virgo", "Pass", "#bots")

while 1:
    read = connection.read()
    ugly = read
    removeUgly = re.compile(":(.+?)!(.+?)@(.+?) (.+?) (.+?) :(.+)") 
    match = removeUgly.search(ugly)
        
    if match:
        cleaned = "[" + match.group(1) + "/" + match.group(4) + "/" + match.group(5) + "]: " + match.group(6)
        print cleaned
    
    var = None

    if read.startswith('PING'):
        space = read.find(' ') + 1
        IRCFunc.sendRaw('PONG ' + read[space:])
        print 'Replied to a PONG - ' + read[space:]

    if read.startswith(':'):
        var = connection.parseText(read)

    if var is not None:
        messageSent = var['messageSent']
        messageSender = var['messageSender']
        currChannel = var['currChannel']
        theHost = var['theHost']
        chanMode = var['chanMode']
        
        if messageSent.startswith("!say"):
            toSay = messageSent[messageSent.find(" ") + 1:]
            IRCFunc.chanSend(currChannel, "[" + var['messageSender'] + "]: " + toSay)
            
        if messageSent.startswith("!quit") and connection.isController(theHost):
            sys.exit(1)
        
        if messageSent.startswith("!exec") and connection.isController(theHost):
            code = messageSent[messageSent.find(" ") + 1:]
            try:
                exec(str(code))
            except Exception, e:
                IRCFunc.chanSend(currChannel, ' Error: ' + str(e))

The test bot I have provided only has !say, !quit, and !exec as functions, it’s up to you to build on that. Also, the bot requires a file called “controllers.txt” with your IRC Hostname in it to execute functions that have “and connection.isController(theHost):” on the end.
