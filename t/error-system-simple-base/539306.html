public class Constants {
/**
* Enumeration of Errors that may occur.
* This enumeration is defined in stages.
* The stage can be true or false.
* false is initialized first as the error has not occurred.
* once error has occured during the process,
* the error stage becomes true and the client declares the error.
/
public enum ERROR {
/*
* Stage -> Internal error occurred.
/
CLIENT_INTERNAL_ERROR(false),
/*
* Stage -> Loading error occurred.
/
CACHE_LOADING_ERROR(false),
/*
* Stage -> Cache has not been found.
*/
CACHE_NOT_FOUND(false);

	//Messages that will be initialized upon error.
	public static String firstErrorMessage = null;
	public static String secondErrorMessage = null;
	public static String thirdErrorMessage = null;
	public static String fourthErrorMessage = null;
	public static String fifthErrorMessage = null;
	public static boolean isMessageGenerated = false;

	/**
	 * Error stage (current stage).
	 */
	private boolean errorStage;

	/**
	 * Constructor.
	 *
	 * @param errorStage
	 */
	private ERROR(boolean errorStage) {
		this.errorStage = errorStage;
	}

	/**
	 * Creates the message to be drawn on the error screen.
	 */
	public static void createMessages() {
		if(isMessageGenerated) {
			return;
		}
		if(CLIENT_INTERNAL_ERROR.getProblem()) {
			firstErrorMessage = "An internal error occurred in the Avarrocka client.";
			secondErrorMessage = "Please download our leatest client on our website:";
			thirdErrorMessage = "www.avarrocka.org";
			fourthErrorMessage = "If you're playing in our webclient, please reload this page.";
		} else if(CACHE_LOADING_ERROR.getProblem()) {
			firstErrorMessage = "A cache loading error occurred in the Avarrocka client.";
			secondErrorMessage = "Please check the cache location and it's size.";
			thirdErrorMessage = "Refer to our forums for more help.";
		} else if(CACHE_NOT_FOUND.getProblem()) {
			firstErrorMessage = "The client was unable to find the created cache.";
			secondErrorMessage = "Please reload the client, so the client could download the new cache.";
			thirdErrorMessage = "If this issue still happens, please refer to our forums.";
		}
		isMessageGenerated = true;
	}

	/**
	 * Loading process.
	 *
	 * @return true for fully loaded/false if not fully loaded.
	 */
	public static boolean problemOccured() {
		if(CLIENT_INTERNAL_ERROR.getProblem() || CACHE_LOADING_ERROR.getProblem() || CACHE_NOT_FOUND.getProblem()) {
			return true;
		} else {
			return false;
		}
	}

	/**
	 * Get the current stage.
	 *
	 * @return stage
	 */
	public boolean getProblem() {
		return errorStage;
	}

	/**
	 * Sets the stage.
	 *
	 * @param errorStage
	 */
	public void setProcess(boolean errorStage) {
		this.errorStage = errorStage;
	}
}

}

Now for the usage. You will just have to call the error to true once it occurs. and the screen will display the report you’ve selected.

So here is an example of “Cache not found”.

In signlink before:

public void run() { active = true; final String cacheDir = findCacheDir(); try { cache_dat = new RandomAccessFile(cacheDir + "main_file_cache.dat", "rw"); for(int j = 0; j < 5; j++) { cache_idx[j] = new RandomAccessFile(cacheDir + "main_file_cache.idx" + j, "rw"); } } catch(final Exception exception) { }

will become:

public void run() { active = true; final String cacheDir = findCacheDir(); try { cache_dat = new RandomAccessFile(cacheDir + "main_file_cache.dat", "rw"); for(int j = 0; j < 5; j++) { cache_idx[j] = new RandomAccessFile(cacheDir + "main_file_cache.idx" + j, "rw"); } } catch(final Exception exception) { Constants.ERROR.CACHE_NOT_FOUND.setProcess(true); }
