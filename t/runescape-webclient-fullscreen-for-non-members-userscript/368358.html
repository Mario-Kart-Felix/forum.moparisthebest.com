// ==UserScript==
// @name           free RS fullscreen(script 1/2)
// @namespace      http://userscripts.org/users/
// @description    gives you fullscreen as a RS non-member. u must also have the script "free RS fullscreen(script 2/2)" for this to work.

// ==/UserScript==

script 2:

// ==UserScript==
// @name           free RS fullscreen(script 2/2)
// @namespace      http://userscripts.org/users/
// @description    second part of RS fullscreen script. u need free RS fullscreen(script 1/2) also. this also removes the ads for your.
// @include        *world*.runescape.com*
// ==/UserScript==

var s = document.getElementsByTagName('script');//sets var s as an array of the script elements in the webpage
for(var i=0;i<s.length;i++){//loops through the array to remove the script tags
s[i].parentNode.removeChild(s[i]);//gets the script tag's parent then removes the tag from the parent
i--;//decrements i since the array is one less now
}//removing the scripts from the page will keep the page from recorrecting itself it seems jagex had functions to detect tampering with the page and it trys to fix them sometimes.
document.getElementById('gametable').childNodes[1].childNodes[0].setAttribute('style','visibility:hidden');//sets the cell's, that contains the ad box(iframe) and menubar, visibility to hidden. so it doesnt overlap the game
document.getElementById('tbi').src="";//sets the ad's iframe src to nothing
document.getElementById('tbi').width="";//sets the ad's iframe width to nothing
document.getElementById('tbi').height="";//sets the ad's iframe height to nothing
//by doing the three above, the width and height dont really matter they are kinda extra, it should not reload the ad and make the game flicker.
document.getElementById("gameframe").setAttribute('style','position:absolute;top:0px;left:0px');//sets the div tag's, that contains the applet, position to 0,0. filling the window with the applet.
