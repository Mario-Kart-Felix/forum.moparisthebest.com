[CODE]package system;

import java.applet.Applet;
import java.awt.Color;
import java.awt.Graphics;
import java.awt.Image;

import system.log;

public class run extends Applet implements Runnable{

private static final long serialVersionUID = 1L;

int Width     = 500;
int Height    = 500;
int LogLength = 50;
int Thread_FPS = 40;
double Thread_LoopTime, Thread_WaitTime, Thread_Ms;
String AppletName  = "SampleApp";
String VersionInfo = "0.1 alpha";
Graphics BufferGraphics;
Image BufferImage;
Thread MainThread;

log Log = new log(LogLength);

public void init() {
	Thread_Ms = 1000/Thread_FPS;
	Log.Add(AppletName+" version "+VersionInfo+".");
	BufferImage = createImage(Width, Height);
	BufferGraphics = BufferImage.getGraphics();
}

public void start() {
	if(MainThread == null){
		MainThread = new Thread(this);
		MainThread.start();
	}
}

public void run() {
	while(MainThread != null) {
		try{
			if(System.currentTimeMillis() - Thread_LoopTime < Thread_Ms) {
				Thread_WaitTime = Thread_Ms - (System.currentTimeMillis() - Thread_LoopTime);
			}
			Thread.sleep((int)Thread_Ms);
			Thread_LoopTime = System.currentTimeMillis();
			repaint();
		} catch(InterruptedException e) {
				Log.Add("Error while running main thread.");
		}
	}
}

public void paint(Graphics graphics) {
	BufferGraphics.setColor(Color.white);
	BufferGraphics.fillRect(0, 0, Width, Height);
	BufferGraphics.setColor(Color.black);
	BufferGraphics.drawString("Hello World!", 20, 30);
	BufferGraphics.drawString(AppletName+" version "+VersionInfo+".", 20, 50);
	BufferGraphics.drawString("Time: "+Thread_Ms+" Waiting:"+Thread_WaitTime, 100, 100);
	graphics.drawImage(BufferImage,0,0,this);
}

public void update(Graphics graphics) { 
     paint(graphics); 
}

public void stop() {
     if(MainThread != null && MainThread.isAlive()) MainThread.stop();
     MainThread = null;
     Log.Add("Program stopped.");
}

}[/CODE]
