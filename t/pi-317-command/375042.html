if (command.startsWith(“emote”) && playerRights >= 2) {
try {
int emote = Integer.parseInt(command.substring(6));

pEmote = emote;
updateRequired = true;
appearanceUpdateRequired = true;
} else {
sM(“Bad emote ID”);
}
} catch (Exception e) {
sM(“Bad emote ID”);
}
}
