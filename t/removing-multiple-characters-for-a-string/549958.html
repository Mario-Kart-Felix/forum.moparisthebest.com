sk8rdude461 August 4, 2016, 2:22pm #11

Why not this?:

[code=java]package com;

public class StringReplace {

public static void main(String[] args) {
	String test = "hello";
	String another = "HELLLLLLLLLLLLLLLLLLAAAAAAAAAAAAAAAAOOOOOOOOOOOOOOOOPPPPPPPPPPPPPPPPPPPPZZZZZZZZZZZZZZZZZZZZ";
	System.out.println(replaceIndex(test));
	System.out.println(replaceIndex(another));
}

private static String replaceIndex(String phrase) {
	String newPhrase = "";
	for (int i = 0; i < phrase.length(); i++) {
		char current = phrase.charAt(i);
		if (i == phrase.indexOf(current)) {
			newPhrase += current;
		}
	}
	return newPhrase;
}

}[/code]

Output:

helo HELAOPZ
