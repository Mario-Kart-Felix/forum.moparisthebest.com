[code=Java]final int DAY = 86400000; //the amount of milliseconds in a day
final int DAYS_BEFORE_LOGIC = 60; //the amount of days before logic is done

private int timelapse(Date then, Date now) {
return (int)((now.getTime() - then.getTime()) / DAY);
}[/code]

[code=Java]ArrayList abandoned = new ArrayList();

Calendar now = Calendar.getInstance();
now.setTimeInMillis(System.currentTimeMillis());

final File dir = new File("./accounts/");

for (final File char : dir.listFiles()) {
if (char.isDirectory()) {
return;
} else {
Calendar then = Calendar.getInstance();
then.setTimeInMillis(char.lastModified());

    int days = timelapse(then.getTime(), now.getTime());

    if (days >= DAYS_BEFORE_LOGIC) {
        abandoned.add(char.getName());
    }
}

}[/code]this could probably be done a lot more efficiently. this is gonna take a great bit of time, as it will check every file’s last modified date and create calendar instances for it.
