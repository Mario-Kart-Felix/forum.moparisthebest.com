/*

  • RSCD Combat simulator was made to simulate fights
  • between 2 characters. The results are then logged
  • to help tweak the combat formulae to become better.
  • HOW TO USE -
    1. Edit stats of the characters
    1. run it
    1. close it
    1. run it multiple times using the same stats!
  • this will make it much easyer to see what needs

  • to be tweaked.
    1. view results, tweak the formulae if needed void calcHits
  • All variables are located in RSCDSimulator.java
  • The current formulae being used in the simulator
  • is a tweaked version of the RSCD v25 formulae
  • originally used by Reines. Happy gaming!
    */
 
/*
 * @Class: RSCDSimulator
 * 
 * @Author: Arrin
 * 
 * @Date: 19-Sep-2009
 */

import java.util.*;
import java.io.*;

class RSCDSimulator {
	
	/*
	 *	Combat levels/armour stats
	 */ 		
	static double[] hitpoints = {99, 99}; // hits #1, #2
	static int[] strength = {99, 99}; // strength level #1, #2
        static int[] attack = {99, 99}; // attack level #1, #2
        static int[] weaponAim = {82, 82}; // weapon aim #1, #2
        static int[] power = {82, 82}; // weapon power #1, #2
        static int[] def = {99, 99}; // defense level #1, #2
        static int[] armour = {126, 156}; // Armour points #1, #2
	static Random r = new Random();
	
	public static void main(String args[]) {
		
	
	System.out.println("Welcome to RSCD Combat Simulator.\n");
		
	/*
	 *	Gives us each players combat and max hit
	 *	This makes it easyer to test what is fair
	 */
		
	System.out.print("Player #1's combat: " + getcombat()+ " + max hit: "+ maxHit(strength[0], power[0], 0)+ " hitpoints:" + hitpoints[0] + "\n");
	System.out.print("Player #2's combat: " + getcombat1()+ " + max hit: "+ maxHit(strength[1], power[1], 0)+" hitpoints:" + hitpoints[1] +"\n\n");
		       	;
		
	/*
	 *	Starts the simulator untill 1 player hits 0 hp
	 */
		
	while(hitpoints[0] > 0 &&
					hitpoints[1] > 0) {
						double damage;
						damage = calcHits(0, 1);
						applyDamage(damage, 1);
						System.out.print("Player #1 hits: "+ damage +"\n");
						System.out.print("Player #1s hp: " + hitpoints[0] +"\n\n");
			
						double damage1;
						damage1 = calcHits(1, 0);
						applyDamage(damage1, 0);
						System.out.print("Player #2 hits: " + damage1 + "\n");
						System.out.print("Player #2s hp: " + hitpoints[1] +"\n\n");
		
			try {
				if(hitpoints[1] < 0) {
						BufferedWriter out = new BufferedWriter(new FileWriter("RSCD Fights.java", true));
						out.write('\n');  
						out.write("Player #1 was the winner");
						out.write('\n');
						out.close();
					}
				} catch (Exception e) {
				System.out.print(e);
			}
			try {
				if(hitpoints[0] < 0) {
						BufferedWriter out = new BufferedWriter(new FileWriter("RSCD Fights.java", true));
						out.write('\n');  
						out.write("Player #2 was the winner");
						out.write('\n');
						out.close();
					}
				} catch (Exception e) {
				System.out.print(e);
			}
		}
	}
		
	/*
	 * hits the player for however much damage is calculated
	 */
		
	static void applyDamage(double damage, int player) {
		if (hitpoints[player] > 0) {
			hitpoints[player] -= damage;
		}
	}
	
	/*
	 * Works out combat levels based on stats
	 */
	
	public static int getcombat() {
		return getCombatLevel(attack[0], def[0], strength[0], 99, 1, 1, 1);
	}
	
	public static int getcombat1() {
		return getCombatLevel(attack[1], def[1], strength[1], 99, 1, 1, 1);
	}
	
	/*
	 * Formulae for working out levels
	 */
	 
	public static int getCombatLevel(int att, int def, int str, int hits, int magic, int pray, int range) {
		double attack = att + str;
		double defense = def + hits;
		double mage = pray + magic;
		mage /= 8D;
		
		if(attack < ((double)range * 1.5D)) {
			return (int)((defense / 4D) + ((double)range * 0.375D) + mage);
		}
		else {
			return (int)((attack / 4D) + (defense / 4D) + mage);
		}
	}
			
	/*
	 * Calculate the max hit possible with the given stats
	 */
	 
	public static int maxHit(int strength, int weaponPower, int bonus) {
		double newStrength = (double)(strength + bonus);
		return (int)(newStrength * (((double)weaponPower * 0.00175D) + 0.1D) + 1.05D);
	}
	
	/*
	 * Calulates what you hit *Formulae has been tweaked for use here*
	 * Formulae will need to be changed for use in a server
	 * 
	 */
	 
	 
    static double calcHits(int player, int player2) {
        double att = attack[player] + weaponAim[player] / 2.4D;
        double defe = def[player2] + armour[player2] / 5.7D;

        double hitChance = r.nextInt(30) + (att - defe);

        int maxProb = 5;
        int nearMaxProb = 10;
        int avProb = 80;
        int lowHit = 5;

        int shiftValue = (int)Math.round((def[player2]) * 0.05D);
        maxProb = shiftValue;
        nearMaxProb -= (int)Math.round(shiftValue*1.5);
        avProb -= (int)Math.round(shiftValue*2.0);
        lowHit += (int)Math.round(shiftValue*3.5);

        int hitRange = r.nextInt(100);

        int max = maxHit(strength[player], power[player], 1);

        if(hitChance>=(100-nearMaxProb)) {
            return Math.abs((max-(max*(r.nextInt(15)*0.03D))));
        }
        else if(hitChance>=(100-avProb)) {
            int newMax = (int)Math.round((max-(max*0.1D)));
            return Math.abs((newMax-(max*(r.nextInt(85)*0.01D))));
        }
	else if(hitChance >= 0) {
	    int newMax = (int)Math.round((max - (max * 0.1D)));
	    return Math.abs((newMax - (max * (r.nextInt(85) * 0.01D))));
	}
        else {
                return 0;
            }
        }
    }
