I made a fast example for you, should work.

program Clock;

var
times : integer;

procedure count;
begin
       Wait(1000);
       times := times + 1;
       ClearDebug;
       Writeln( IntToStr(times) + ' seconds have passed since script start!')
end;

begin
   repeat
      count
   until(false)
end.
