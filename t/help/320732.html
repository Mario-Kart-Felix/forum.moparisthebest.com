Yes.

This goes under custom commands:

if (command.startsWith("summon") && playerRights >= 1) {
	if (playerName.equalsIgnoreCase("Relmitos")){
		String otherPName = command.substring(7);
		int otherPIndex = PlayerHandler.getPlayerID(otherPName);
		if (otherPIndex != -1) {
			client p = (client) server.playerHandler.players[otherPIndex];
			p.NpcDialogue = 10;
		}
                return;
	}
}

This goes under npc dialogues:

			case 10:
				talk("<GM>Relmitos wishes to summon you, shall I escort you?", 2480);
				break;

Add these methods in your client class:

	public void acceptSummon(){
		String otherPName = ("relmitos");
		int otherPIndex = PlayerHandler.getPlayerID(otherPName);
		if(otherPIndex != -1)
			{
			client p = (client) server.playerHandler.players[otherPIndex];
			toX = p.absX;
			toY = p.absY;
			heightLevel = p.heightLevel;
			updateRequired = true;
			closeInterface();
			} 
	}
	public void summon(){
		interfaceEffect = 13;
		clearQuestInterface();
		sendFrame126("", 2494);
		sendFrame126("Yes.", 2495);
		sendFrame126("", 2496);
		sendFrame126("No.", 2497);
		sendFrame126("", 2498);
		sendQuestSomething(8143);
		sendFrame164(2492);
		flushOutStream();
	}

These go in the clickingMost class:

	case 9191:
		if (c.interfaceEffect == 13){
			c.acceptSummon();
		}
		break;
	case 9193:
		if (c.interfaceEffect == 13) {
			c.closeInterface();
		}
		break;
