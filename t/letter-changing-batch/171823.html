You could write a small commandline program like the following delphi one, but I’d suggest using basic file writing functions rather than the TStringList to make ths ize of the program smaller, just didn’t feel like spending much time on it Also added a compiled version

[code=delphi]program ReplaceChar;

{$APPTYPE CONSOLE}

uses
Classes, SysUtils;

var
cOld, cNew: Char;
s: TStringList;
f: string;

begin
if ParamCount = 3 then
begin
s := TStringList.Create;
cOld := ParamStr(2)[1];
cNew := ParamStr(3)[1];
f := ExpandFileName(ParamStr(1));
s.LoadFromFile(f);
s.Text := StringReplace(s.Text, cOld, cNew, [rfReplaceAll]);
s.SaveToFile(f);
s.Free;
end;
end.[/code]

(Example) In ur batch file: ReplaceChar urfile.txt e z

Since Freddy1990 did one in Delphi, I got bored and decided to write one in Java.

[code]import java.io.BufferedWriter;
import java.io.Console;
import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.io.RandomAccessFile;

/**

  • Author: Yz85Racer

  • Date: 3/02/2008
    */
    public class LetterReplace {

    public static void main(String[] args) {
    new LetterReplace().replace();
    }

    private void replace() {
    Console con = System.console();
    File rFile = new File(con.readLine("Please specify the path to the file: "));
    if(rFile.exists()) {
    try {
    long start = System.currentTimeMillis();
    RandomAccessFile raf = new RandomAccessFile(rFile, “r”);
    byte[] b = new byte[(int) rFile.length()];
    raf.readFully(b);
    String buffer = new String(b).replaceAll(con.readLine("Letter to be replaced? "), con.readLine("Letter to replace with? "));
    BufferedWriter out = new BufferedWriter(new FileWriter(rFile));
    out.write(buffer);
    out.flush();
    con.format(“Application completed in " + (System.currentTimeMillis() - start/1000) + " seconds.”);
    } catch(IOException e) {
    e.printStackTrace();
    }
    } else {
    con.format(“Specified file (” + rFile.getPath() + “) does not exist.\n”);
    }
    }
    }[/code]
