Change your GWD interface code to this:

[code=java]
public static void godWars(TextDrawingArea[] tda) {
RSInterface rsinterface = addTabInterface(16210);
addText(16211, “God Wars”, tda, 0, 0xff9040, true, true);
addText(16212, “Total Killcount:”, tda, 0, 0xff9040, true, true);
addText(16216, “0”, tda, 0, 0x66FFFF, true, true);//armadyl

	rsinterface.children = new int[3];
	rsinterface.childX = new int[3];
	rsinterface.childY = new int[3];
	
	rsinterface.children[0] = 16211;
	rsinterface.childX[0] = -52+375+30;
	rsinterface.childY[0] = 7;
	rsinterface.children[1] = 16212;
	rsinterface.childX[1] = -52+375+30;
	rsinterface.childY[1] = 30;
	rsinterface.children[2] = 16216;
	rsinterface.childX[2] = -52+460+30;
	rsinterface.childY[2] = 31;

}[/code]
