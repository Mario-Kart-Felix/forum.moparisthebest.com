/**
 * GateManager
 * 
 * March 1, 2014
 * 
 * Dan0194
 */
package org.hyperion.rs2.content.world.gates;

import java.util.HashMap;
import java.util.Map;

import org.hyperion.rs2.content.world.gates.impl.*;
import org.hyperion.rs2.model.Location;
import org.hyperion.rs2.model.Player;

public class GateManager {

	/**
	 * Direction of the gate
	 */
	public enum GateOrientation {
		LEFT, RIGHT;
	}

	/**
	 * Where the gate is positioned
	 */
	public enum GateDirection {
		/**
		 * The South gate handler
		 */
		SOUTH(new SouthGate()), 
		/**
		 * The east gate handler
		 */
		EAST(new EastGate()), 
		/**
		 * The north gate handler
		 */
		NORTH(new NorthGate()), 
		/**
		 * The west gate handler
		 */
		WEST(new WestGate());

		private GateHandler gateHandler;

		GateDirection(GateHandler gateHandler) {
			this.gateHandler = gateHandler;
		}

		public GateHandler getGateHandler() {
			return gateHandler;
		}
	}

	/**
	 * the singleton
	 */
	public static final GateManager gatesManager = new GateManager();

	/**
	 * The map
	 */
	private static Map<Location, Gate> gates = new HashMap<Location, Gate>();

	/**
	 * @param player
	 * @param id
	 * @param x
	 * @param y
	 */
	public GateManager openGate(final Player player, final int x,
			final int y) {
		final Gate gate = gates.get(Location.create(x, y, 0));
			if(gate.isOpen()) {
			gate.getDirection().getGateHandler()
					.closeGate(player, gate, x, y);
			} else {
				gate.getDirection().getGateHandler()
				.openGate(player, gate, x, y);
			}
		return this;
	}
	
	/**
	 * Gets the gates partner
	 * @param gate
	 */
	public GateManager getGatePartner(Gate gate) {
		final int x = gate.getDirection() == GateDirection.NORTH ? +(gate
				.getOrientation() == GateOrientation.LEFT ? 1 : -1) : gate
				.getDirection() == GateDirection.SOUTH ? +(gate
				.getOrientation() == GateOrientation.LEFT ? -1 : 1) : 0;
		final int y = gate.getDirection() == GateDirection.EAST ? +(gate
				.getOrientation() == GateOrientation.LEFT ? -1 : 1)
				: gate.getDirection() == GateDirection.WEST ? +(gate
						.getOrientation() == GateOrientation.LEFT ? 1 : -1) : 0;
		final GateOrientation gateOrientation = gate.getOrientation() == GateOrientation.LEFT ? GateOrientation.RIGHT
				: GateOrientation.LEFT;
		Gate partnerGate = new Gate(gate.getLocation().getX() + x, gate
				.getLocation().getY() + y, gate.getDirection(), gateOrientation);
		gates.put(partnerGate.getLocation(), partnerGate);
		return this;
	}

	/**
	 * @param gate
	 */
	public void bind(Gate gate) {
		gates.put(gate.getLocation(), gate);
		this.getGatePartner(gate);
	}

	/**
	 * @return gates
	 */
	public Map<Location, Gate> getGates() {
		return gates;
	}

	/**
	 * @return gatesManager
	 */
	public static GateManager getGateManager() {
		return gatesManager;
	}
}

    package org.hyperion.rs2.content.world.gates.impl;

import org.hyperion.rs2.content.world.gates.Gate;
import org.hyperion.rs2.content.world.gates.GateHandler;
import org.hyperion.rs2.content.world.gates.GateManager;
import org.hyperion.rs2.model.Location;
import org.hyperion.rs2.model.Player;

public class EastGate implements GateHandler {

/**
 * (non-Javadoc)
 * 
 * @see org.hyperion.rs2.content.world.gates.GateHandler#openGate(org.hyperion.rs2.model.Player,
 *      org.hyperion.rs2.content.world.gates.Gate, int, int)
 */
@Override
public void openGate(Player player, Gate gate, int x, int y) {
	switch (gate.getOrientation()) {
	case LEFT:
		player.getActionSender().removeObject(x, y);
		player.getActionSender().removeObject(x, y - 1);
		player.getActionSender().sendObject(1551, x, y, 1, 0);
		player.getActionSender().sendObject(1554, x + 1, y, 1, 0);
		GateManager
				.getGateManager()
				.getGates()
				.put(Location.create(x, y, 0),
						new Gate((true), gate.getDirection()));
		break;
	case RIGHT:
		player.getActionSender().removeObject(x, y);
		player.getActionSender().removeObject(x, y + 1);
		player.getActionSender().sendObject(1554, x + 1, y + 1, 1, 0);
		player.getActionSender().sendObject(1551, x, y + 1, 1, 0);
		GateManager
				.getGateManager()
				.getGates()
				.put(Location.create(x, y + 1, 0),
						new Gate((true), gate.getDirection()));
		break;
	}
}

/**
 * (non-Javadoc)
 * 
 * @see org.hyperion.rs2.content.world.gates.GateHandler#closeGate(org.hyperion.rs2.model.Player,
 *      org.hyperion.rs2.content.world.gates.Gate, int, int)
 */
@Override
public void closeGate(Player player, Gate gate, int x, int y) {
	player.getActionSender().removeObject(x, y);
	player.getActionSender().removeObject(x + 1, y);
	player.getActionSender().sendObject(1553, x, y, 0, 0);
	player.getActionSender().sendObject(1551, x, y - 1, 0, 0);

}

}

First: I personally would’ve used an enum for storing the fence data. Other than that (and those are really just personal preferences) this looks pretty good.
