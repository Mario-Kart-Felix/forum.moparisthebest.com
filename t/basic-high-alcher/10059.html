Ahh! Very detectable! Lemme fix this up for you a little. I made a few tweaks. I’ll take it off here if you want too. [code]//Basic High Aclher by Do0mzy/Do0mz//
//Place the alching item in the 15th inv slot//
//select the RS screen and start :)//
//Fixed a little by Bullzeye95//
//No longer have to start in the magic screen//
//less detectible//
//You can now tell it how many items to alch!//

program Alcher;

{.include srl/srl.scar}

var alched : integer;

//////////////////SETUP//////////////////

const AlchNumber = 100; //How many items do you want to alch?

procedure alch;

begin
if(not(gametab(7))) then
gametab(7);
wait(1500+random(600));
ClickMouse(676,330,true);
wait(1000+random(500));
ClickMouse(676,330,true);
alched:=(alched+1);
end;

begin
setupsrl;
ActivateClient;
Status(‘Alching ’ + inttostr(alchnumber) + ’ Items’);
begin
if(not(gametab(7))) then
gametab(7);
HoverEvery(1+random(3),‘magic’);
RotateEvery(1+random(3));
repeat
alch;
until(alchnumber=alched)
end;
end.[/code]
