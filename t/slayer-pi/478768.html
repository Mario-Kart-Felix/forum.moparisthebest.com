	if (taskLevel == 1) {
		return c.playerLevel[c.playerSlayer] >= lowReqs[random];		
	} else if (taskLevel == 2) {
		return c.playerLevel[c.playerSlayer] >= medReqs[random];		
	} else if (taskLevel == 3) {
		return c.playerLevel[c.playerSlayer] >= highReqs[random];
