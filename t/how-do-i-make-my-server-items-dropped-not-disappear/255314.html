/**
 * Registers an item to be removed after 3 minutes
 */
public void registerItem(final Item i) {
  		if(i.getLoc() == null) {
  			delayedEventHandler.add(new DelayedEvent(null, 180000) {
  				public void run() {
  					ActiveTile tile = getTile(i.getLocation());
				if(tile.hasItem(i)) {
					unregisterItem(i);
				}
  					running = false;
  				}
  			});
  		}
}
