JaydenD12 July 31, 2016, 2:28pm #2

public int[] plateBody = {platebody ids..}; public int[] shields = {shield ids..}; public int[] plateLegs = {leg ids..};

^ Make sure you seperate the ids with a comma.

public int randomPlateBodys() 
{
	return plateBody[(int) (Math.random() * plateBody.length)];
}
public int randomShields() 
{
	return shields[(int) (Math.random() * shields.length)];
}
public int randomPlateLegs() 
{
	return plateLegs[(int) (Math.random() * plateLegs.length)];
}
if (command.startsWith("armour")) 
{
	addItem(randomPlateBodys(), 1);
	addItem(randomShields(), 1);
	addItem(randomPlateLegs(), 1);
}

Should work quite nicely.
