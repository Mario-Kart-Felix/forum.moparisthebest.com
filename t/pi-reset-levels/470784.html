I found this command you could probably make this work.

if (playerCommand.startsWith("reset")) { if (c.inWild()) { c.sendMessage("You are not aloud to reset stats while in Wild!"); return; } for (int j = 0; j < c.playerEquipment.length; j++) { if (c.playerEquipment[j] > 0) { c.sendMessage("Please take all your armour and weapons off before using this command."); return; } } String skillName = playerCommand.substring(6); Object[][] SKILL_INFO = { {"Attack", 0}, {"Defence", 1}, {"Strength", 2}, {"Hitpoints", 3}, {"Range", 4}, {"Prayer", 5}, {"Mage", 6} }; for (int i = 0; i < SKILL_INFO.length; i++) { Object[] data = SKILL_INFO[i]; if (skillName.equalsIgnoreCase((String) data[0])) { c.playerXP[(Integer) data[1]] = c.getPA().getXPForLevel(1) + 5; c.playerLevel[(Integer) data[1]] = c.getPA().getLevelForXP(c.playerXP[(Integer) data[1]]); c.getPA().refreshSkill((Integer) data[1]); } } }
