	/*
	 * It's thief, not theif. Also the following should be fixed for conventions:
	 * TheifStall(), XPamount, and an else if statement is unnecessary.
	 */
	public int theifTimer = 0;// Sets the timer equal to 0, default.
	
	/**
	 * 
	 * @param stallName
	 * @param message
	 * @param lvlReq
	 * @param XPamount
	 * @param item
	 * @param itemAmount
	 * @param delay
	 * @param emote
	 */

	public void TheifStall(String stallName, String message, int lvlReq, int XPamount, int item, int itemAmount, int delay, int emote) {
		if (theifTimer == 0) {// If the timer is 0 do the following
			if (playerLevel[17] >= lvlReq) {// If the players skill Id 17 (thieving) is equal to or greater than the required level do the following
				setAnimation(emote);// Play an animation equal to emote, which could be any int
				sendMessage("You steal from the " + stallName);// Prints a message to the player including the stall's name
				sendMessage(message);// Prints another message to the player with whatever message is equal to
				addItem(item, itemAmount);// Gives the player their item and the amount set
				addSkillXP(XPamount, 17);// Gives the player the set amount of Xp, to thieving (skill Id 17)
				theifTimer = delay;// Sets the theifTimer back to the amount delay is set to. (8 seconds you said)
			} else if (playerLevel[17] < lvlReq) {// If the players skill Id 17 (thieving) is less than the required message do the following
				sendMessage("You need a theiving level of " + lvlReq+ " to theif from this stall.");// Sends a message to the player with the requirements needed
			}// Closes the above else if statement
		}// Closes theifTimer
	}// Closes the method
