assuming you have java 6 with jdic, and some kind of icon named sss.gif, it would sit in the task bar, and when you click it will save screen shot to working dir or specified dir

[code]import java.awt.;
import java.awt.event.;
import java.awt.image.BufferedImage;
import java.io.File;
import java.util.*;

import javax.imageio.ImageIO;
import javax.swing.*;

import org.jdesktop.jdic.tray.*;

public class ScreenShotSaver implements ActionListener {
static Toolkit tool = Toolkit.getDefaultToolkit();
static Dimension size = tool.getScreenSize();
static Rectangle screen = new Rectangle(0, 0, size.width, size.height);
static Robot r;
static {
try {
r = new Robot();
} catch (AWTException ex) {
// wow r we fked lol
}
}

SystemTray tray = SystemTray.getDefaultSystemTray();
TrayIcon trayicon;
JPopupMenu menu = new JPopupMenu();

File path = new File(System.getProperty("user.dir", "C:/"));
String name = "screen#.png";
int count = 1;

public ScreenShotSaver() {
    JMenuItem path = new JMenuItem("Set Path");
    path.addActionListener(this);
    menu.add(path);
    
    JMenuItem name = new JMenuItem("Set Name");
    name.addActionListener(this);
    menu.add(name);
    
    JMenuItem capture = new JMenuItem("Capture");
    capture.addActionListener(this);
    menu.add(capture);
    
    JMenuItem capturec = new JMenuItem("Capture Coords");
    capturec.addActionListener(this);
    menu.add(capturec);
    
    ImageIcon icon = new ImageIcon("sss.gif");
    trayicon = new TrayIcon(icon, "Screen Shot Saver", menu);
    tray.addTrayIcon(trayicon);
}

java.text.SimpleDateFormat sdf = new java.text.SimpleDateFormat("MMdd-hhmmss");

public void actionPerformed(ActionEvent e) {
    String cmd = e.getActionCommand();
    try {
        if (cmd.equals("Set Path")) {
            String input = JOptionPane.showInputDialog("Enter path:",
                    path.getAbsolutePath());
            if (input != null)
                path = new File(input);
        } else if (cmd.equals("Set Name")) {
            String input = JOptionPane.showInputDialog("Enter name:", name);
            if (input != null) {
                name = input;
            }
        } else if (cmd.equals("Capture")) {
            String n = name.replaceAll("%", sdf.format(new Date()));
            File file = new File(path, n.replaceAll("#", "" + count));
            while (file.exists() && name.indexOf("#") > -1) {
                count++;
                file = new File(path, name.replaceAll("#", "" + count));
            }
            count = 1;
            ImageIO.write(screenCapture(), "png", file);
        } else if (cmd.equals("Capture Coords")) {
            String n = name.replaceAll("%", sdf.format(new Date()));
            File file = new File(path, n.replaceAll("#", "" + count));
            while (file.exists() && name.indexOf("#") > -1) {
                count++;
                file = new File(path, name.replaceAll("#", "" + count));
            }
            count = 1;
            String input = JOptionPane.showInputDialog("Enter x, y, w, h:", screen.toString());
            int x = 0, y = 0, w = size.width, h = size.height;
            if (input != null) {
                StringTokenizer tok = new StringTokenizer(input, ", ");
                x = Integer.parseInt(tok.nextToken());
                y = Integer.parseInt(tok.nextToken());
                w = Integer.parseInt(tok.nextToken());
                h = Integer.parseInt(tok.nextToken());
            }
            BufferedImage img = screenCapture(x, y, w, h);
            ImageIO.write(img, "png", file);
        }
    } catch (Exception ex) {
        ex.printStackTrace();
    }
}

public static BufferedImage screenCapture() {
    return r.createScreenCapture(screen);
}    

public static BufferedImage screenCapture(int x, int y, int width, int height) {
    return r.createScreenCapture(new Rectangle(x, y, width, height));
}

public static void main(String[] args) {
    ScreenShotSaver sss = new ScreenShotSaver();
}

}[/code]
