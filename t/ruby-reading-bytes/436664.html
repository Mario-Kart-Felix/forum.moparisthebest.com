Edit: made this method. this works.

def read(file, offset, length, buffer)

  file.seek(offset)
    
  num_bytes = 0
  while num_bytes != length and not file.eof?
    buffer[num_bytes] = file.readbyte
    num_bytes+=1
  end
    
  return num_bytes
end
