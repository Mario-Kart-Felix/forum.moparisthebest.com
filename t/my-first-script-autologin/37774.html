[code]{---------------->}
program autologin;
{<----------------}
{.include srl/srl.scar}

const
username =‘USERNAME’; //add your username
password =‘peepee’; //add your password

var
xrandom,yrandom:Integer;

procedure login;
begin
x:=468;
y:=291;
MMouse(x,y,xrandom,yrandom)
wait(1000);
Mouse(0, 0, 1, 1, true)
wait(1000);
Typesend(username)
KeyDown(40);
Typesend(password)
wait(1000);
x:=331;
y:=320;
Mouse(0, 0, 1, 1, true);
wait(1000);
Writeln(‘now logged in, hoped you enjoyed it!’)
Terminatescript;
end;

begin
ActivateClient;
SetUpSRL;
Cleardebug;
writeln(‘Logging in’)
login;
end.[/code]
