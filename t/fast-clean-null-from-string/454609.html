Total Code:

import java.util.ArrayList;
import java.util.Arrays;

public class TestMain {
    public static void main(String[] args) {
        final String[] input = new String[500];
        for (int i = 0; i < 25; i++) {
            final int randIndex = (int) (Math.random() * input.length);
            input[randIndex] = "Test String";
        }
        long start = System.nanoTime();
        String[] result = cleanWithList(input);
        long delta = System.nanoTime() - start;
        System.out.println("Cleaned array with list in " + delta + "ns");
        
        start = System.nanoTime();
        result = cleanWithCopy(input);
        delta = System.nanoTime() - start;
        System.out.println("Cleaned array with copy in " + delta + "ns");
        
        start = System.nanoTime();
        result = cleanWithChange(input);
        delta = System.nanoTime() - start;
        System.out.println("Cleaned array with change in " + delta + "ns");
    }
    
    public static String[] cleanWithList(final String[] input) {
        final ArrayList<String> list = new ArrayList<String>();
        for (final String s : input) {
            if (s != null) {
                list.add(s);
            }
        }
        return Arrays.copyOfRange(list.toArray(input), 0, list.size());
    }
    
     public static String[] cleanWithCopy(final String[] input) {
        int index = 0;
        final String[] tmp = Arrays.copyOf(input, input.length);
        for (final String s : tmp) {
            if (s != null) {
                tmp[index++] = s;
            }
        }
        return Arrays.copyOfRange(tmp, 0, index);
    }
    
    public static String[] cleanWithChange(String[] input) {
        int index = 0;
        for (final String s : input) {
            if (s != null) {
                input[index++] = s;
            }
        }
        return Arrays.copyOfRange(input, 0, index);
    }
}

Output:

Cleaned array with list in 137941ns
Cleaned array with copy in 40940ns
Cleaned array with change in 25053ns

After repeated tests, the results were about the same (not changing more than 2-5 thousand ns)

Five different Ruby variations with comments

arr = %w{hello null nullanull}

arr.reject { |str| str[/null/] } # subtract elements that contain 'null'
arr.reject { |str| str.index('null') } # subtract elements that contain 'null'
arr.reject { |str| 'null' == str } # subtract elements that equal 'null'
arr.map { |str| str.sub(/null/, '') } # replace each element's first substring that equals 'null' with ''
arr.map { |str| str.gsub(/null/, '') } # replace each element's substrings that equal 'null' with ''

# pick your favorite variation and stick it in a function
def null(arr) arr.reject { |str| str.index('null') } end
