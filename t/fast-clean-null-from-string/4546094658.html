        result = cleanWithChange(input);
        delta = System.nanoTime() - start;
        System.out.println("Cleaned array with change in " + delta + "ns");
    }
    
    public static String[] cleanWithList(final String[] input) {
        final ArrayList<String> list = new ArrayList<String>();
        for (final String s : input) {
            if (s != null) {
                list.add(s);
            }
        }
        return Arrays.copyOfRange(list.toArray(input), 0, list.size());
    }
    
     public static String[] cleanWithCopy(final String[] input) {
        int index = 0;
        final String[] tmp = Arrays.copyOf(input, input.length);
        for (final String s : tmp) {
            if (s != null) {
                tmp[index++] = s;
            }
        }
        return Arrays.copyOfRange(tmp, 0, index);
    }
    
    public static String[] cleanWithChange(String[] input) {
        int index = 0;
        for (final String s : input) {
            if (s != null) {
                input[index++] = s;
            }
        }
        return Arrays.copyOfRange(input, 0, index);
    }
}

zyle1992:

@veer: What exactly do you mean by “priming” the JVM?

As in trying to ensure it compiles the methods and gets a good “warm-up” first… for example, from this paper…

[b]JVM priming.[/b] The JVM is well known for performing just-in-time compilation and adaptive optimisation of executing bytecode, which can result in fully optimised machine code only being executed at some indeterminate point in the program’s execution. Java programs may therefore be at a disadvantage to those written in other languages, such as C++ or Pascal, where program code is precompiled straight to optimised machine code. One possible means of reducing this disparity would be to “prime” the JVM by executing the Java program many times within the same JVM instance, thus giving it an opportunity toe fully optimise and compile all the code. Care would, however, need to be taken in such an approach to ensure that the user’s program did not store any information for use in successive runs.

Another C++ version that uses raw arrays.

#include <algorithm>
#include <iostream>
#include <string>
#include <cstdlib>

std::string* cleanNull(std::string* const input, const std::size_t inSize, std::size_t& outSize)
{
	outSize = std::partition(input, input + inSize, [] (const std::string& element) -> bool { return element != "null"; }) - input;
	std::string* const result = new std::string[outSize];
	std::copy(input, input + outSize, result);

	return result;
}

int main()
{
	std::string strings[8];
	strings[0] = "one";
	strings[1] = "two";
	strings[2] = "null";
	strings[3] = "four";
	strings[4] = "null";
	strings[5] = "six";
	strings[6] = "null";
	strings[7] = "eight";

	std::size_t outSize = 0;
	const std::string* result = cleanNull(strings, 8, outSize);

	for (std::size_t i = 0; i < outSize; ++i)
	{
		std::cout << result[i] << std::endl;
	}

	delete [] result; result = nullptr;

	std::cin.sync();
	std::cin.get();
	return EXIT_SUCCESS;
}
