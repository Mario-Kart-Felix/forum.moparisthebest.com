[code]package org.rscangel.client;

import java.io.BufferedInputStream;
import java.io.FileInputStream;

import javazoom.jl.player.Player;

public class Mp3 {
/**
* Mp3 to be played (include path)
*/
public String filename;

/**
 * Instance of the player.
 */
private Player player;

/**
 * The thread that plays the song.
 */
private Thread thread;

/**
 * Whether or not the player is playing a mp3.
 */
public boolean playing;

public Mp3(String filename) {
	this.filename = filename;
}

public Mp3() {
}

/**
 * Stops the player, and thread.
 */
public void close() {
	if (player != null)
		player.close();
	thread.interrupt();
	playing = false;
}

/**
 * Stops player, and thread if running, then plays the mp3.
 */
public void play() {
	if (playing) {
		close();
	}
	playing = true;
	
	try {
		FileInputStream fis = new FileInputStream(filename);
		BufferedInputStream bis = new BufferedInputStream(fis);
		player = new Player(bis);
	} catch (Exception e) {
		System.out.println("Problem playing file " + filename);
		System.out.println(e);
	}
	
	thread = new Thread(new Runnable() {
		public void run() {
			try {
				player.play();
			} catch (Exception e) {
				System.out.println(e);
			}
		}
	});
	thread.start();

}

}[/code]
