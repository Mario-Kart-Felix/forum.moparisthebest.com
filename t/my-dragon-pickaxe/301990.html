ItemDef.java FOR MY BASE

case 15259: itemDef.actionData(1, "Wield"); itemDef.models(48471, 48470, 48470, -1, -1); itemDef.modelData(1180, 700, 1100, 0, 0); itemDef.itemData("Dragon pickaxe", "Used for mining."); break;

or if you don’t use a switch statement:

if(i == 15259) { itemDef.actionData(1, "Wield"); itemDef.models(48471, 48470, 48470, -1, -1); itemDef.modelData(1180, 700, 1100, 0, 0); itemDef.itemData("Dragon pickaxe", "Used for mining."); }

Methods for renamed(you don’t need if you use my base)

public void actionData(int a, String b) { actions = new String[5]; actions[a] = b; } public void totalColors(int total) { originalModelColors = new int[total]; modifiedModelColors = new int[total]; } public void colors(int id, int original, int modified) { originalModelColors[id] = original; modifiedModelColors[id] = modified; } public void itemData(String n, String d) { name = n; description = d.getBytes(); } public void models(int mID, int mE, int fE, int mE2, int fE2) { modelID = mID; maleEquip1 = mE; femaleEquip1 = fE; maleEquip2 = mE2; femaleEquip2 = fE2; } public void modelData(int mZ, int mR1, int mR2, int mO1, int mO2) { modelZoom = mZ; modelRotation1 = mR1; modelRotation2 = mR2; modelOffset1 = mO1; modelOffset2 = mO2; }
