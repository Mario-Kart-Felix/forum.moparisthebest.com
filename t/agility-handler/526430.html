[code=java]import server.model.players.Client;

/**

  • Handles some of agility courses, obstacles
    */
    public class Agility {

    Client c;

    /*

    • The constructor for the client
      */
      public Agility(Client c) {
      this.c = c;
      }

    public enum agilityObstacles {

     WILDERNESS_OBSTACLE_PIPE(2203, 1, 1, 1, 1, 1, 1);

 private int objectID, experience, reqLevel, emote, time, x, y;

 private agilityObstacles(int objectID, int experience, int reqLevel,
 		int emote, int time, int x, int y) {
 	this.objectID = objectID;
 	this.experience = experience;
 	this.reqLevel = reqLevel;
 	this.emote = emote;
 	this.time = time;
 	this.x = x;
 	this.y = y;
 }

    }

    public void handleAgility(int objectID) {
    for (agilityObstacles e : agilityObstacles.values()) {
    if (objectID == e.objectID) {
    if (c.playerAgility == e.reqLevel) {
    c.startAnimation(e.emote, e.time);
    c.absX = e.x;
    c.absY = e.y;
    c.getPA().addSkillXP(e.experience, c.playerAgility);
    } else {
    c.sendMessage(“You need a level of " + e.reqLevel
    + " to cross this”);
    }
    }
    }
    }

}[/code]
