@echo off
@title Server

:menu
cls
echo Welcome to SpyderScape
echo Please choose an option:
echo.
echo 1. Start server
echo 2. Compile server
echo 3. Compile and start server
echo 4. Start www.runescape.com
echo 5. Exit
echo.
set /p mode=Please choose an option: 
if %mode%==1 goto run
if %mode%==2 goto compile
if %mode%==3 goto compilerun
if %mode%==4 goto runescape
if %mode%==5 goto exit
goto wrong

:wrong
cls
echo You chose a wrong option!
echo Please choose either option 1, 2, 3, 4, or 5.
echo.
echo Press any key if you wish to go back to the menu . . .
pause > nul
goto menu

:run
cls
java -cp .;./jython.jar server
pause
goto menu

:compile
cls
javac *.java
pause
goto menu

:compilerun
cls
echo Compiling . . .
javac *.java
echo Compiled - finished.
java -cp .;./jython.jar server
echo Server started!
pause

:runescape
cls
echo Now launching Runescape through Mozilla Firefox . . .
start firefox http://www.runescape.com
pause
goto menu

:exit
exit
