Ya that works, but instead of switching twice, just do this:

[code]public class MessageTimer extends Thread implements Runnable {

public void run() {
    int timeStopped = System.currentTimeMillis(); //get current ms
    while (true) { //infinite loop
        if ((System.currentTimeMillis() - timeStopped) >= 120000) { //check if 120000 ms have passed (2 minute)
            if (firstMsg) {
                PlayerHandler.messageToAll = "Message number 1"; //send message to all players online
            } else {
                PlayerHandler.messageToAll = "Message number 2";
            }
            firstMsg = !firstMsg;
            timeStopped = System.currentTimeMillis();
        }
        try { //try clause
            Thread.sleep(50); //sleep thread
        } catch(InterruptedException ie){} //catch a java.lang.InterruptedException
    }
}
boolean firstMsg = true;

}[/code]
