/* File: ChatServer.java */

// all these imports import the shit to have the server run (DONT FUCK WITH THEM)
import java.io.BufferedOutputStream;
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.ArrayList;

/**
*

  • @author AMERIKA

  • @about Basic chat server

  • @about Use Telnet to connect, and chat
    */
    public class ChatServer implements Runnable {// declares the class name ChatServer and makes it Runnable, or in retard terms, capable of starting new Threads

    public static ArrayList clients = new ArrayList();//this ArrayList will contain a list of all connected Sockets. Used to send messages
    public Socket client;//since this is not static, it allows it to be dynamic for each thread. Within each connection, this socket is set to them
    public BufferedReader in;//this is set so we can read the input from the connected socket in a thread (since it isnt static)

    public ChatServer(Socket sock) throws IOException {//when we start a new thread, this is executed first
    client = sock;//we set the Socket client to equal the Socket sock (sent from the Server listener)
    in = new BufferedReader(new InputStreamReader(client.getInputStream()));//we just setup the BufferedReader in
    }

    public void run() {//this function is when the Constructor is done, this will read all input
    try {//incase some shit goes wrong
    while (client.isConnected()) {//as long as the client is connected, everything in here will loop
    String x = null;//sets String x to equal nothing, or null
    if ((x = in.readLine()) != null) {//this basically says while x is Socket input, and does not equal null, do the following:
    System.out.println(x);//prints the text we get so the console can see everything
    sendMsg(x);//sends the message to all other Sockets (excludes the sender)
    }
    }
    clients.remove(client);//since this is outside of the while statement, it means when the Socket disconnects, it will be removed from the ArrayList of connected Sockets
    } catch (Exception ex) {
    ex.printStackTrace();//incase something goes wrong, we can see all of the errors
    }
    }

    public void sendMsg(String msg) throws IOException {//the method to send the message
    for (Socket s : clients) {//for all the Sockets
    if (s != client) {//if the Socket is not the sender
    OutputStreamWriter out = new OutputStreamWriter(new BufferedOutputStream(s.getOutputStream()));//sets up the output stream, so we can send the sockets messages
    out.write(msg + “\r\n”);//this will input the message, but not send it. \r\n is needed so it will read a new line.
    out.flush();//this takes all the printed lines in the writer, and sends them to the Socket specified in out
    }
    }
    }

    public static void main(String[] args) {//our main method
    try {
    ServerSocket serv = new ServerSocket(9030);//starts the Server, on port 9030
    while (true) {//an infinite loop
    Socket s = serv.accept();//listens for a connection. once made, it will fill this in, and continue to be processed.
    clients.add(s);//adds the Socket to the ArrayList of Sockets currently connected.
    (new Thread(new ChatServer(s))).start();//starts a new thread, so this one can go back to listening for connections
    }
    } catch (Exception ex) {
    ex.printStackTrace();//incase something goes wrong, we can see all of the errors
    }
    }
    }[/code]
