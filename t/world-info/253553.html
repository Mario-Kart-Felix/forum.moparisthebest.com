Here’s my deob 519 version of it:

public static boolean readWorldList(int dummy, byte data[]) {
	mc_readWorldList++;
	
	StreamParent stream = new StreamParent(data);
	int j = stream.readUnsignedByte((byte) 107);
	if (j != 1) 
		return false;

	boolean hasInfo = stream.readUnsignedByte((byte) 101) == 1;
	if (hasInfo) 
		Class12_Sub13.readWorldList_WorldInfo((byte) -38, stream);

	Class12_Sub22_Sub3.readWorldList_PlayersOnline(76, stream);
	return true;
}
public static void readWorldList_WorldInfo(byte dummy, StreamParent stream) {
	mc_readWorldList_WorldInfo++;
	int i = stream.readInt(false);
	Class85_Sub3.worldListCountries = new WorldListCountry[i];
	for (int j = 0; i > j; j++) {
		Class85_Sub3.worldListCountries[j] = new WorldListCountry();
		Class85_Sub3.worldListCountries[j].flagID = stream.readInt(false);
		Class85_Sub3.worldListCountries[j].location = stream.readString(30634);
		System.out.println("[" + j + "] (" + Class85_Sub3.worldListCountries[j].flagID + ") --> " + 
				new String(Class85_Sub3.worldListCountries[j].location.aByteArray2479));
	}

	SocketIO.worldOffset = stream.readInt(false);
	Class80.maxWorldID = stream.readInt(false);
	Class12_Sub26_Sub5.entryCount = stream.readInt(false);
	System.out.println("[" + SocketIO.worldOffset + "][" + Class80.maxWorldID + "][" + Class12_Sub26_Sub5.entryCount + "]");
	Node.worldListWorlds = new WorldListWorld[1 + Class80.maxWorldID - SocketIO.worldOffset];
		
	for (int k = 0; k < Class12_Sub26_Sub5.entryCount; k++) {
		int l = stream.readInt(false);
		WorldListWorld world = Node.worldListWorlds[l] = new WorldListWorld();
		world.locationID = stream.readUnsignedByte((byte) 115);
		/* The settings (from right to left)
		 * Bit 1: memberstate --> 0=free, 1=member
		 * Bit 2: quickchat --> 0=no, 1=yes
		 * Bit 3: pvp --> 0=no, 1=yes
		 * Bit 4: lootshare --> 0=no, 1=yes
		 * Bit 5: yellow activity --> 0=no, 1=yes
		 */
		world.settings = stream.readUnsignedDWord(65280);
		world.worldID = l + SocketIO.worldOffset;
		world.activity = stream.readString(30634);
		world.url = stream.readString(30634);
		System.out.println("[" + k + "] (" + world.locationID + ", " + world.settings + ", " + world.worldID + ") -->" + 
				new String(world.activity.aByteArray2479) + ":" + new String(world.url.aByteArray2479));
	}

	Class133.anInt2279 = stream.readUnsignedDWord(65280);
	System.out.println(Class133.anInt2279);
	Class24.aBoolean680 = true;
}
public static void readWorldList_PlayersOnline(int dummy, StreamParent stream) {
	mc_readWorldList_PlayersOnline++;
	for (int j = 0; Class12_Sub26_Sub5.entryCount > j; j++) {
		int k = stream.readInt(false);
		int l = stream.readUnsignedShort((byte) 109);
		if (l == 0xffff)
			l = -1;

		if (Node.worldListWorlds[k] != null) 
			Node.worldListWorlds[k].players = l;

		System.out.println("World[" + k + "] " + l);
	}
}
