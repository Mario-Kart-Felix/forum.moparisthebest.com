/**
* This method gets a bitmap from the BitMapManager that is younger than the
* specified maxAgeInMillisecs
*
* @author Eric
* @param maxAgeInMillisecs
* @return int[][][]
/
protected int[][][] getPixelArray(int maxAgeInMillisecs) {
return theBot.theBitMapManger.getPixels(maxAgeInMillisecs);
}
/*
* This method returns all the Valid Pixels from a Pixel Array That fall
* with in the tolerances all colors as specified
*
* @author Eric
* @param int[][][]
* @param red
* @param green
* @param blue
* @param redTolerance
* @param greenTolerance
* @param blueTolerance
* @param Tolerance
* @return Pixel[]
*/
protected Pixel[] findPixels(int[][][] pixelArray, int red, int green,
int blue, int redTolerance, int greenTolerance, int blueTolerance,
Rectangle rectangle) {
ArrayList list = new ArrayList(0);
LogManager.log(theBot.theApplet.uid, pixelArray[10][10][0] + " "
+ pixelArray[10][10][1] + " " + pixelArray[10][10][2]);
for (int y = rectangle.y; y < rectangle.height; y++) {
for (int x = rectangle.x; x < rectangle.width; x++) {
int redChange = red - pixelArray[y][x][0];
if (redChange < 0) {
redChange *= -1;
}
if (redChange > redTolerance) {
break;
}
int greenChange = green - pixelArray[y][x][1];
if (greenChange < 0) {
greenChange *= -1;
}
if (greenChange > greenTolerance) {
break;
}
int blueChange = blue - pixelArray[y][x][2];
if (blueChange < 0) {
blueChange *= -1;
}
if (blueChange > blueTolerance) {
break;
}
list.add(new Pixel(pixelArray[y][x][0], pixelArray[y][x][1],
pixelArray[y][x][2], x, y));
}
}
Pixel pixelAnswer[] = new Pixel[list.size()];
for (int i = 0; i < pixelAnswer.length; i++) {
pixelAnswer[i] = (Pixel) list.get(i);
}
return pixelAnswer;
}
