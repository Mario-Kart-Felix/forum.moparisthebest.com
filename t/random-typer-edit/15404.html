Edit: Made a better version. Stops by hitting space bar, and hits enter after so many characters.

//Random ass typer by\\
//------M0ng00se-----\\
///////////////////////
//Hit SPACEBAR to stop\\
////////////////////////

program RandomTyper;
var
t : integer;
Const
waittime=0; // Time in between each typed letter. In milliseconds.
times=5; //How many characters to type before hitting enter(put 0 for never)

Procedure Termination;
begin
if(iskeydown(chr(32)))then
begin
TerminateScript;
end else
end;

Procedure TypeRandom;
begin
  Case Random(26) of
  0: sendkeys('a');
  1: sendkeys('b');
  2: sendkeys('c');
  3: sendkeys('d');
  4: sendkeys('e');
  5: sendkeys('f');
  6: sendkeys('g');
  7: sendkeys('h');
  8: sendkeys('i');
  9: sendkeys('j');
  10: sendkeys('k');
  11: sendkeys('l');
  12: sendkeys('m');
  13: sendkeys('n');
  14: sendkeys('o');
  15: sendkeys('p');
  16: sendkeys('q');
  17: sendkeys('r');
  18: sendkeys('s');
  19: sendkeys('t');
  20: sendkeys('u');
  21: sendkeys('v');
  22: sendkeys('w');
  23: sendkeys('x');
  24: sendkeys('y');
  25: sendkeys('z');
end;
t:=t+1;
Termination;
end;


Procedure Hitenter;
begin
Termination;
If(t >= times)then
begin
SendKeys(chr(13));
t:=0;
end else
end;

Begin
wait(500);
repeat
termination;
TypeRandom;
wait(10);
termination;
hitenter;
wait(waittime);
until(false)
End.

Instead of making the termination part a procedure, you could just do it in the Until(). Like so.

Begin Wait(500); Repeat TypeRandom; Wait(10); hitenter; Wait(waittime); Until(IsKeyDown(Chr(32))); End.

Also, in stead of using case of for the random words, you could do something like this…

[code]
Procedure TypeRandom;
var
Chars: String;

begin
Chars:=‘abcdefghijklmnopqrstuvwxyz1234567890`~!@#$%^&*()-_=+[]{}|/?.,<>";:’’ ';
if random(Length(Chars)+1)=0 then
begin
SendKeys(Chr(13));
Exit;
end;
SendKeys(StrGet(Chars, Random(Length(Chars))+1));
end;[/code]
