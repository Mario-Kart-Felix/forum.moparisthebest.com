It’s just a simple JTextArea within a JScrollPane.

Here’s something I just wrote up:

[code=java]import java.awt.BorderLayout;
import java.awt.Color;
import java.awt.Dimension;
import java.io.Console;

import javax.swing.JFrame;
import javax.swing.JPanel;
import javax.swing.JPopupMenu;
import javax.swing.JScrollPane;
import javax.swing.JTextArea;

/**

  • The Main class.

  • @author PrimaDude.
    */
    public class Main extends Client {

    private JFrame frame;
    private JPanel gamePanel = new JPanel();
    private JTextArea textArea;

    public static void main(String args[]) {
    new Main();
    }

    /**

    • The Constructor.
    • Creates and displayes the frame and it’s components.
      */
      public Main() {
      JPopupMenu.setDefaultLightWeightPopupEnabled(false);
      JFrame.setDefaultLookAndFeelDecorated(true);
      frame = new JFrame(“Client”);
      frame.setLayout(new BorderLayout());
      frame.setResizable(false);
      gamePanel.setLayout(new BorderLayout());
      gamePanel.add(this);
      gamePanel.setPreferredSize(new Dimension(765, 503));
      textArea = new JTextArea();
      JScrollPane scroll = new JScrollPane(textArea, 22, 31);
      scroll.setPreferredSize(new Dimension(765, 150));
      textArea.setForeground(Color.white);
      textArea.setBackground(Color.black);
      frame.add(gamePanel, “Center”);
      frame.add(scroll, “South”);
      frame.pack();
      frame.setVisible(true);
      }

    /**

    • Prints a message then creates a new line.
    • @param message The message to print.
      */
      private void printMessage(String message) {
      textArea.append(message +"\n");
      }

}[/code]
