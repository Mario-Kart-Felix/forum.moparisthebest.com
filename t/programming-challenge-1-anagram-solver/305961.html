Ok, this challenge will require you to write an anagram solver. Here’s how it will go:

The input to the program will look like the following:

e.g.

4
lol
little
bitch
niglet
nlgiet hbcti tltiel oll

The program will read the information in, and attempt to unscramble the anagram by using the dictionary. If an anagram can be unscrambled to no words or more than one word, then output a question mark for it.

e.g.

2 hi ih ih
Would output:
?

Similarly,

1 lol nder
Would output:
?
.

[center]Any specifics?[/center]
The number of words in a dictionary is (0…100].
The number of anagrams in a list is (0…20].
The words in a dictionary and the anagrams in the list are all lowercase letters.
The number of letters in a word or anagram is (0…20].
The number of letters in anagram list is (0…100].

typedef struct _dict_entry {
char *word;
char *sorted;
int len;
struct _dict_entry *next;
} dict_entry;

dict_entry *dict_read(void) {
int dict_size;
scanf("%d\n", &dict_size);
dict_entry *dict = (dict_entry *) calloc(sizeof(dict_entry), dict_size);

int i;
for (i = 0; i < dict_size; ++i) {
    dict[i].next = dict + i + 1;
    /* 20 (max word size) + 1 (terminator) */
    dict[i].word = (char *) malloc(21);
    fgets(dict[i].word, 32, stdin);
    dict[i].word = strtok(dict[i].word, "\n");
    dict[i].len = strlen(dict[i].word);
}

dict[dict_size - 1].next = NULL;

return dict;

}

void dict_free(dict_entry *head) {
dict_entry *cur;
for (cur = head; cur; cur = cur->next) {
free(cur->word);
free(cur->sorted);
}
free(head);
}

void dict_print(dict_entry *head, FILE *stream) {
dict_entry *cur;
for (cur = head; cur; cur = cur->next) {
fprintf(stream, “word: %s\n”, cur->word);
fprintf(stream, “sorted: %s\n”, cur->sorted);
}
}

int char_comp(const void *elem1, const void *elem2) {
char *c1 = (char *) elem1;
char *c2 = (char *) elem2;
return *c1 - *c2;
}

void dict_sort_all(dict_entry *head) {
dict_entry *cur;
for (cur = head; cur; cur = cur->next) {
cur->sorted = strdup(cur->word);
qsort(cur->sorted, cur->len, 1, char_comp);
}
}

main() {
dict_entry *dict = dict_read();
dict_sort_all(dict);

/* 20 (max word size) * 100 (max sentence length) + 19 (number of spaces in sentence) + 1 (terminator) */

char *scrambled;
for (scrambled = strtok(line, " "); scrambled;
         scrambled = strtok(NULL, " ")) {

    char *sorted = strdup(scrambled);

import java.math.BigInteger;
import java.util.Map;
import java.util.HashMap;
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.text.CharacterIterator;
import java.text.StringCharacterIterator;
import java.lang.String;
import java.lang.Integer;
import java.math.BigInteger;

public class AnagramSolver
{
private static final int MAX_LETTERS = 10;
private static final int MAX_ALPHABET = 26;
private Map<BigInteger, Word> dict;

public AnagramSolver()
{
	dict = new HashMap<BigInteger, Word>();
}

public void addWord(String word)
{
	BigInteger bi = calculateValue(word);
	
	Word existingWord = dict.get(bi);
	if(existingWord == null)
	{
		Word wordObj = new Word(word);
		dict.put(bi, wordObj);
	}
	else
	{
		existingWord.setDuplicate();
	}
}

protected BigInteger calculateValue(String word)
{
	BigInteger bi = new BigInteger(Integer.toString(MAX_LETTERS));
	bi = bi.pow(MAX_ALPHABET);
	
	int[] letterCount = new int[MAX_ALPHABET];
	
	CharacterIterator iter = new StringCharacterIterator(word);
	for(char ch = iter.first(); ch != CharacterIterator.DONE; ch = iter.next())
	{
		int letterIdx = (int)ch - (int)'a';
		BigInteger currBigInt = new BigInteger(Integer.toString(MAX_LETTERS));
		currBigInt = currBigInt.pow(letterIdx);
		currBigInt = currBigInt.add(new BigInteger(Integer.toString(letterCount[letterIdx]++)));
		bi = bi.add(currBigInt);
	}
	return bi;
}

public void solveAnagram(String anagram)
{
	String[] words = anagram.split("\\s+");
	for(int i = 0; i < words.length; ++i)
	{
		String output = "?";
		for(Map.Entry<BigInteger, Word> entry : dict.entrySet())
		{
			if(calculateValue(words[i]).equals(entry.getKey()))
			{
				if(!entry.getValue().hasMultipleCopies())
					output = entry.getValue().getWord();
				break;
			}
		}
		System.out.println(output);
	}
}

public static void main(String[] args)
{
	try
	{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		
		int numWords = Integer.parseInt(br.readLine());
		AnagramSolver as = new AnagramSolver();
		for(int i = 0; i < numWords; ++i)
