import operator, string
lines = file("an.txt", "r").read().splitlines()
anagrams = set( map(lambda x: tuple(sorted(x)), lines.pop().split()) )
dict_lst = map(lambda x: tuple(sorted(x)), lines[1:])
dict_set, dict_dic = set( dict_lst ), {}
d, un = map(operator.setitem, [dict_dic]*len(lines[1:]), dict_lst, lines[1:]), dict_set.intersection(anagrams)
if not un: print '?'
for anag in un: print "(%s -> %s)  " % ( string.joinfields( anag ), ('?', dict_dic[anag])[int(dict_lst.count( anag ) == 1)]  )
