Did it because i was bored lol.
I know i could use the run exe function but id rather use the mouth functions.
Script can be cleaned easilly but im lazy lol.
Requires resolution set to 1024x768.
Made on Vista transformation skin but should work with xp as it uses co-ords not DTMS/Bitmaps.

[scar]Program OpenFireFox;
{/-------------------}
{ FireFox Starter }
{ Author: Nix }
{ Date: 05/07/06 }
{ Desc: Opens Firefox }
{-------------------/}
const WaitTime = 500; //ms

Procedure ClickStart;
Begin
MoveMouseSmooth(30, 752);
wait(WaitTime);
ClickMouse(30, 752, True);
wait(WaitTime);
End;

Procedure SelectFox;
Begin
wait(WaitTime);
MoveMouseSmooth(73,315);
wait(WaitTime);
ClickMouse(73,315,true);
wait(WaitTime);
End;

Procedure MaximizeFF;
Begin
MoveMouseSmooth(938, 12);
wait(WaitTime);
ClickMouse(938, 12, true);
wait(WaitTime);
End;

Procedure GotoUrl;
Begin
MoveMouseSmooth(713,63);
wait(WaitTime);
ClickMouse(713,63,true);
wait(WaitTime);
SendKeys(‘www.moparisthebest.com/smf/index.php’);
wait(WaitTime);
MoveMouseSmooth(844,63);
wait(WaitTime);
ClickMouse(844,63,true);
wait(WaitTime);
End;

Begin
wait(WaitTime);
ClickStart;
wait(WaitTime);
SelectFox;
wait(WaitTime);
MaximizeFF;
wait(WaitTime);
GotoUrl;
wait(WaitTime);
end.

[/scar]
