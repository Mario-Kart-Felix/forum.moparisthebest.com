[code]package server.model.players.content.music;

import java.util.HashMap;

/**
*

  • @author Rhubarb/Int Bauk

*/
public class WorldManager {

/**
 * Our enum.
 *
 *
 */
public enum LocationHandler {
	TUTORIAL(1, 2625, 2687, 4670, 4735),
	YANILLE(48, 2532, 2621, 3071, 3122);
	
	int songId; int locationX; int locationX1;
	int locationY; int locationY1;
	
	/**
	 * Our constructor for the enum LocationHandler.
	 * @param songId
	 * @param locationX
	 * @param locationX1
	 * @param locationY
	 * @param locationY1
	 */
	LocationHandler(int songId, int locationX, int locationX1, 
			int locationY, int locationY1) {
		this.songId = songId;
		this.locationX = locationX;
		this.locationX1 = locationX1;
		this.locationY = locationY;
		this.locationY1 = locationY1;		
	}
	
	/**
	 * Creates a new hash map for the enum LocationHandler.
	 */
	public static HashMap<Integer, LocationHandler> location = 
			new HashMap<Integer, LocationHandler>();

	/**
	 * Used for inheriting the song id.
	 * @param songId
	 * @return songId
	 */
	public static LocationHandler forSongId(int songId) {
		return location.get(songId);
	}
	
	/**
	 * Gets the song id.
	 * @return songId.
	 */
	public int getSongId() {
		return songId;
	}
}

}

[code]package server.model.players.content.music;

import server.model.players.Player;
import server.model.players.content.music.WorldManager.LocationHandler;

/**
*

  • @author Rhubarb/Int Bauk

*/
public class MusicHandler {

/**
 * Player constructor.
 */
private Player player;

/**
 * Create a new instance for the class MusicData musicData.
 */
MusicData musicData = new MusicData();

/**
 * Create a new instance for the class WorldManager worldManager.
 */
WorldManager worldManager = new WorldManager();


/**
 * Constructor for player class.
 * @param player
 */
public MusicHandler(Player player) {
	this.player = player;
}

/**
 * Declare the current song and current area id.
 */
private int currentArea, currentSong = 0;

/**
 * Gets the songId(currentArea);
 */
final WorldManager.LocationHandler location = LocationHandler.forSongId(currentArea);

/**
 * Sets a song.
 */
private void setSong() {
	currentSong = musicData.getSong(currentArea);
}

/**
 * Updates song.
 */
private void updateSong() {
	setSong();
}

/**
 * Set the area(s) identification.
 */
private void setSongArea() {
	location.getSongId();
}

/**
 * Updates the song via location.
 */
private void updateAreaSong() {
	setSongArea();
}

/**
 * Handles the music sent to the client.
 */
public void handleMusic() {
	updateSong();
	updateAreaSong();
	/**
	 * player.sendSong(currentSong);
	 */
}

}
