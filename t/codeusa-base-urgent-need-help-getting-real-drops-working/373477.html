public boolean arrayContains(int[] array, int value) {
    for (int i : array) {
        if (value == i) {
            return true;
        }
    }
    return false;
}

public int[] removeValueFromArray(int[] array, int value) {
    int index = -1;
    for (int i = 0; i < array.length; i++) {
        if (value == array[i]) {
            index = i;
        }
    }
    if (index != -1) {
        array[index] = 0;
    }
    return array;
}

public int[][] removeValueFromArray(int[][] array, int value) {
    int indexI = -1;
    int indexJ = -1;
    for (int i = 0; i < array.length; i++) {
        for (int j = 0; j < array[i].length; j++) {
            if (value == array[i][j]) {
                indexI = i;
                indexJ = j;
            }
        }
    }
    if (indexI != -1 && indexJ != -1) {
        array[indexI][indexJ] = -1;
    }
    return array;
}
