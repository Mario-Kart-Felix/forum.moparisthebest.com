package oremtius.beer

class BottlesOfBeer {

	def static void main(String[] args) {
		println(start(99));
	}

	def static start(int amount) '''
		«FOR i : amount .. 1»
			«i.bottles» of beer on the wall, «amount.bottles» of beer.
			Take one down and pass it around, «(i - 1).bottles» of beer on the wall.
		«ENDFOR»
		
		No more bottles of beer on the wall, no more bottles of beer.
		Go to the store and buy some more, «amount.bottles» of beer on the wall.
	'''

	def static bottles(int amount) {
		switch amount {
			case 0:
				'no more bottles'
			case 1:
				'one bottle'
			default: '''
				«amount» bottles
			'''
		}.toString.toFirstUpper
	}

}

public class BottlesOfBeer {

    /**
     * The main method.
     * @param args The command-line arguments.
     */
    public static void main(String[] args) throws Exception {
        startSong(99);
    }

    /**
     * Starts the '99 Bottles of Beer' song.
     * @param bottles The amount of beer on the wall and how many bottles of beer.
     */
    private static void startSong(int bottles) throws Exception {
        int originalAmount = 0;
        boolean first = true;
        originalAmount = bottles;
        while (bottles != 0) {
            if (!first) {
                Thread.sleep(3500);
            } else {
                first = false;
            }
            System.out.println(bottles + " bottles of beer on the wall,");
            Thread.sleep(3500);
            System.out.println(bottles + " bottles of beer.");
            Thread.sleep(3350);
            String word = (bottles - 1) > 0 ? Integer.toString(bottles) + " bottles" : "no more bottles";
            if (bottles - 1 == 1) {
                word = "1 bottle";
            }
            System.out.println("Take one down and pass it around,");
            Thread.sleep(3500);
            System.out.println(word + " of beer on the wall.");
            System.out.println();
            bottles--;
        }
        System.out.println("No more bottles of beer on the wall,");
        Thread.sleep(3500);
        System.out.println("no more bottles of beer.");
        Thread.sleep(3350);
        System.out.println("Go to the store and buy some more,");
        Thread.sleep(3500);
        System.out.println(originalAmount + " bottles of beer on the wall.");
    }
}
