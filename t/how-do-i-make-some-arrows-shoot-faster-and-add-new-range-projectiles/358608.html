public static final int SPRITE_PROJECTILE_START = 3160;

     Projectile projectile = new Projectile(owner, n, 5); // The '5' is what you will change for your new projectile. 5 is the projectile for the cannon ball
     // Method to get the player's who you want to inform of the projectile
    for(Player p : playersToInform) {
           p.informOfProjectile(projectile);
           p.informOfModifiedHits(n);
    }
}
