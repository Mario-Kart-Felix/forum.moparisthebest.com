import java.math.BigDecimal;
import java.math.BigInteger;

import MathCalculators.util.BigIntegerUtils;

public class RamanujanPICalculator {
    // Math.sqrt(8) / 9801;
    public static final BigDecimal RAMANUJAN_BASE = calculateRamanujanBase();
    private int precisionValue = 10000;
    private BigInteger summationIterator = ZERO;
    private BigDecimal summationValue = new BigDecimal(ZERO);


    public static BigDecimal calculateRamanjuanBase() {
        BigDecimal ramanjuanBase = BigIntegerUtils.sqrt(BigDecimal.valueOf(8), precisionValue);
        ramanjuanBase = ramanjuanBase.divide(BigDecimal.valueOf(9801), precisionValue, BigDecimal.ROUND_HALF_UP);
        return ramanjuanBase;
    }


    public void run() {
        BigDecimal currentSummation = null;

        /*
         * Executes: (4n)!(1103 + 26390n) / ((n!)^4)(396^4n)
         *           for n = 0 -> infinity
         */
        while(isSearching) {
            // Calculates (4n)
            BigInteger i = BigInteger.valueOf(4 * summationIterator)

            // Initializes the current summation value to the factorial of (4n)
            currentSummation = new BigDecimal(BigIntegerUtils.factorial(i));

            // Calculates (1103 + 26390n)
            i = BigInteger.valueOf(1103).add(BigInteger.valueOf(26390).multiply(summationInteger));

            // Calculates (4n)!(1103 + 26390n)
            currentSummation = currentSummation.multiply(new BigDecimal(i));

            // Calculates ((n!)^4) and divides currentSummation by the result
            i = BigIntegerUtils.factorial(summationInteger).pow(4);
            currentSummation = currentSummation.divide(new BigDecimal(i), precisionValue, BigDecimal.ROUND_HALF_UP);

            // Calculates (396^4n) and divides currentSummation by the result
            i = BigInteger.valueOf(396);
            currentSummation = currentSummation.divide(new BigDecimal(i), precisionValue, BigDecimal.ROUND_HALF_UP);

            summationValue = summationValue.add(currentSummation);
            if(summationValue.subtract(summationValue).equals(new BigDecimal("0E-"+ precisionValue))) {
                isSearching = false;
            } else {
                summationIterator.add(ONE);
            }
        }

        summationValue = summationValue.multiply(RAMANUJAN_BASE);
        summationValue = BigDecimal.valueOf(1).divide(summationValue, precisionValue, BigDecimal.ROUND_HALF_UP);

    }

Ahh, guess I forgot to include some methods. That was another one of my class, the BigIntegerUtils. Here is a version that can be run as it, without any modifications.

import java.io.FileOutputStream;

import java.math.BigDecimal;
import java.math.BigInteger;
import java.util.Scanner;
import java.util.zip.GZIPOutputStream;

/**
 *
 * @author S
 */
public class RamanujanPiCalculator implements Runnable {
    public static final int precisionValue = 5000;
    public static final BigDecimal RAMANUJAN_BASE;  // Math.sqrt(8) / 9801;
    public static final BigInteger TWO = BigInteger.valueOf(2),
                                   FOUR = BigInteger.valueOf(4),
                                   FIVE = BigInteger.valueOf(5),
                                   TEN = BigInteger.valueOf(10);
    
    public static final String OUPUT_FILENAME = "piDump.gz";
    
    private boolean isSearching = false;
    
    

    static {
        BigDecimal ramanujanBase = sqrt(BigDecimal.valueOf(8), precisionValue);
        ramanujanBase = ramanujanBase.divide(BigDecimal.valueOf(9801), precisionValue, BigDecimal.ROUND_HALF_UP);
        RAMANUJAN_BASE = ramanujanBase;
    }

    public static void main(String[] args) {
        new RamanujanPiCalculator();
    }
    
    
    public RamanujanPiCalculator() {
        System.out.println("Beginning calculation of pi using one of Ramanujan's formulas");
        System.out.println("Press [Enter] to cancel execution and output calculation");
        
        isSearching = true;
        new Thread(this).start();
        new Scanner(System.in).nextLine();
        isSearching = false;
    }


    public void run() {
        BigDecimal currentSummation = null;
        BigInteger i;
        
        BigInteger summationIterator = BigInteger.ZERO;
        BigDecimal summationValue = BigDecimal.ZERO;

        /*
         * Executes: (4n)!(1103 + 26390n) / ((n!)^4)(396^4n)
         *           for n = 0 -> infinity
         */
        while(isSearching) {
            // Calculates (4n)!
            i = factorial(FOUR.multiply(summationIterator));
            currentSummation = new BigDecimal(i);

            // Calculates (1103 + 26390n)
            i = BigInteger.valueOf(1103).add(BigInteger.valueOf(26390).multiply(summationIterator));
            currentSummation = currentSummation.multiply(new BigDecimal(i));

            // Calculates ((n!)^4) and divides currentSummation by the result
            i = factorial(summationIterator).pow(4);
            currentSummation = currentSummation.divide(new BigDecimal(i), precisionValue, BigDecimal.ROUND_HALF_UP);

            // Calculates (396^4n) and divides currentSummation by the result
            i = BigInteger.valueOf(396).pow(4 * summationIterator.intValue());
            currentSummation = currentSummation.divide(new BigDecimal(i), precisionValue, BigDecimal.ROUND_HALF_UP);

            summationValue = summationValue.add(currentSummation);
            if(summationValue.subtract(summationValue).equals(new BigDecimal("0E-"+ precisionValue))) {
                isSearching = false;
            } else {
                summationIterator.add(BigInteger.ONE);
            }
        }

        summationValue = summationValue.multiply(RAMANUJAN_BASE);
        summationValue = BigDecimal.ONE.divide(summationValue, precisionValue, BigDecimal.ROUND_HALF_UP);
        
        // Dump calculation
        byte[] outputtedCalculation = summationValue.toPlainString().getBytes();
        
        try {
            GZIPOutputStream outputStream = new GZIPOutputStream(new FileOutputStream(OUPUT_FILENAME));
            outputStream.write(outputtedCalculation);
            outputStream.close();
        } catch(Exception e) {
            System.err.println("Unable to output calculation: ");
            e.printStackTrace();
        }
    }


    public static final BigInteger factorial(final BigInteger startInteger) {
        if(startInteger.equals(BigInteger.ZERO))
            return BigInteger.ONE;

        BigInteger factorialValue = BigInteger.ONE,
                   iterationInteger = startInteger;

        while(iterationInteger.compareTo(BigInteger.ONE) > 0) {
            factorialValue = factorialValue.multiply(startInteger);
            iterationInteger = iterationInteger.subtract(BigInteger.ONE);
        }
        return factorialValue;
    }

    public static final BigDecimal sqrt(final BigDecimal originalValue, int divisionScale) {
        // Must be >= 1, otherwise doesn't run square root algorithm
        int maxIterations = 50;

        // Make sure originalValue is a non-negitive number
        if(originalValue.compareTo(BigDecimal.ZERO) == -1) {
            throw new ArithmeticException("Square root of a negative number.");
        }

        if(maxIterations < 1)
            return originalValue;

        BigDecimal lastValue = BigDecimal.ZERO,
                   sqrtValue = getInitialApproximation(originalValue),
                   errorValue = null;

        int currentIteration = 0;
        boolean isCalculating = true;
        while(isCalculating) {
            lastValue = sqrtValue;
            sqrtValue = originalValue.divide(sqrtValue, divisionScale, BigDecimal.ROUND_HALF_UP);
            sqrtValue = sqrtValue.add(lastValue);
            sqrtValue = sqrtValue.divide(BigDecimal.valueOf(2), divisionScale, BigDecimal.ROUND_HALF_UP);
            errorValue = originalValue.subtract(sqrtValue.multiply(sqrtValue));

            if(++currentIteration == maxIterations) {
                isCalculating = false;
            } else if(lastValue.equals(sqrtValue)) {
                isCalculating = errorValue.abs().compareTo(BigDecimal.ONE) > -1;
            }
        }
        return sqrtValue;
    }

    private static final BigDecimal getInitialApproximation(final BigDecimal originalValue) {
        int totalIntegerDigits = originalValue.toBigInteger().toString().length();
        if ((totalIntegerDigits % 2) == 0) {
            totalIntegerDigits--;
        }
        totalIntegerDigits /= 2;

        return BigDecimal.ONE.movePointRight(totalIntegerDigits);
    }
}

This will calculate the first 5000 values of pi. Change the precisionValue integer to increase the precision.
