[code=java5]import java.awt.Rectangle;
import java.awt.event.KeyEvent;

public class Player {

//Rectangle to represent the player
Rectangle player;

//Variables to hold position and change in position
static int x;
static int y;
int dx;
int dy;

//Player consructor
//sets the xy and y of the player
//makes the rectangle at the the players x and y and sets a width and height
public Player(int x, int y) {

    Player.x = x;
    Player.y = y;

    player = new Rectangle(getX(),getY(),25,25);

}

public Player() {

    //do nothing

}

//returns the x coordinate of the player
public static int getX() {

    return x;

}

//returns the y coordinate of the player
public static int getY() {

    return y;

}

//returns the rectangle representing the player
public Rectangle getPlayer() {

    return player;

}

//moves the player
public void move() {

    if(x < 0) {

        dx = 0;
        x = 0;

    } else if(y < 0) {

        dy = 0;
        y = 0;

    } else if(x > 460) {

        dx = 0;
        x = 460;

    } else if(y > 436) {

        dy = 0;
        y = 436;

    } else {

        x += dx;
        y += dy;

    }

    player.setLocation(x, y);

}

//keyPressed method
//Handles what happens when the arrow keys are pressed
public void keyPressed(KeyEvent e) {

    int key = e.getKeyCode();

    switch(key) {

        case KeyEvent.VK_UP: dy = -2; break;
        case KeyEvent.VK_DOWN: dy = 2; break;
        case KeyEvent.VK_RIGHT: dx = 2; break;
        case KeyEvent.VK_LEFT: dx = -2; break;

    }

}

//KeyReleased method
//Handles what happens when the arro keys are released
public void keyReleased(KeyEvent e) {

    int key = e.getKeyCode();

    switch(key) {

        case (KeyEvent.VK_UP): dy = 0;
        case (KeyEvent.VK_DOWN): dy = 0;
        case (KeyEvent.VK_RIGHT): dx = 0;
        case (KeyEvent.VK_LEFT): dx = 0;

    }
}

[code=java5]import java.awt.Rectangle;

public class Enemy {

//Rectangle representing the square
Rectangle EnemyPlayer;

//object of the player class
Player player;

//ints for position and change in position
static int x;
static int y;
int dx;
int dy;

//ints for getting the players coordinates
int px;
int py;

//Enemy constuctor
//sets the x and y coordinates and makes the rectangle representing the enemy
//makes an object out of the player
public Enemy(int x, int y) {

    Enemy.x = x;
    Enemy.y = y;
    
    EnemyPlayer = new Rectangle(getX(),getY(),50,50);

    player = new Player();

}

//blank constuctor
public Enemy() {

    //do nothing

}

//returns the x coordinate of the enemy
public static int getX() {

    return x;

}

//returns the y coordinate of the enemy
public static int getY() {

    return y;

}

//returns the rectangle representing the enemy
public Rectangle getEnemy() {

    return EnemyPlayer;

}

//follow method
public void follow() {

    //get the players coordinates
    px = Player.getX();
    py = Player.getY();

    //x coordinates
    if(px > x) {

        dx = 1;
        x += dx;

    } else if (px < x) {

        dx = -1;
        x += dx;

    } else {

        dx = 0;
        //do nothing

    }

    //y coordinates
    if(py > y) {

        dy = 1;
        y += dy;

    } else if (py < y) {

        dy = -1;
        y += dy;

    } else {

        dy = 0;
        //do nothing

    }

    EnemyPlayer.setLocation(x, y);

}
