        public byte[] buffer = new byte[5000]; //maximum possible data sent to server by 317 (if i remember correctly)

        public bool interpreteIncommingPackets()
        {
            try
            {
                if(inStreamDecryption == null) return false;
                if (!fin.CanRead) return false;
                int size = fin.Read(buffer, 0, 5000);

                if (size == 0)
                    return false;
                 else
                    ms_in = new System.IO.MemoryStream(buffer, 0, size); //load it up into memorystream for easy reading

                while (ms_in.Position < ms_in.Length) //keep looping until no pending packets remaining
                {
                    packetType = ms_in.ReadByte();
                    packetType = packetType & 0xff; //must be unsigned (& 0xff) or will be outofbounds error in packetSizes
                    packetType = packetType - inStreamDecryption.getNextKey() & 0xff;

                    packetSize = packetSizes[packetType];

                    if (packetSize == -1) // this is a variable size packet, the next byte containing the length of data
                    {
                        if (ms_in.Length > 0) //check could be zero then someone is packet hacking!
                        {
                            packetSize = ms_in.ReadByte();
                        }
                        else
                        {
                            println("We got a packet hacker/spammer who doesn't know what he is doing!");
                            clientConnected = false;
                            return false;
                        }
                    }
                    if (packetSize > 0) //extra data
                    {
                        inStream.currentOffset = 0;
                        ms_in.Read(inStream.buffer, 0, packetSize);
                        /*
                         * Notice how similar the above is to this method
                         *  public void fillInStream(int forceRead)
                         *   {
                         *           inStream.currentOffset = 0;
                         *           fin.Read(inStream.buffer, 0, forceRead);
                         *   }
                         *   unfortunately we cannot use fin as we already 
                         *   completely read all the data out nothing left in it.
                         */
                    }
                    timeOutCounter = 0;
                    println("Current commandID: " + packetType + " size: " + packetSize);
                }
                ms_in.Dispose();
            }
            catch(System.IO.IOException ioe)
            {
                println("BlakeScape Server: Exception!");
                println(ioe.Message);

                //__ex.printStackTrace(); 
                clientConnected = false;
                return false; //added this to get out of endless loop if player disconnects -sspoke
            }
            return true;
        }
