In math class we received an extra credit assignment in which we were supposed to find all prime numbers from 1 - 200. I figured it would be much funner (and likely faster) to make a simple C++ app to do it for me, and I decided to post it. It’s self-explanatory.

#include <iostream>

using namespace std;

int intWhole, divs, maxNum;
double Number, dblWhole, num;

bool whole(double numTest)
{
    intWhole = Number / num;
    dblWhole = Number / num;
    return (intWhole == dblWhole);
}

int main()
{
    cout << "Please enter a maximum number to test for: ";
    cin >> maxNum;
    cout << "Prime numbers: " << endl;
    for (Number = 1; Number <= maxNum; Number++)
    {
        divs = 0;
        for (num = 1; num <= Number; num++)
        {
            if (whole(Number / num))
            {
                divs++;
                if (divs <= 2 && num >= Number)
                {
                    cout << Number << ",    ";
                    break;
                }
            }
        }
    }
    cout << endl << endl;
    system("pause");
    
    return 0;
}
