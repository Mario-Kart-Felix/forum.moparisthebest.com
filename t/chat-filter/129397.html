  1. Add these to player.java

[code]public String GenerateC(int length) {
String temp = “”;
for (int i = 0; i < length; i++) {
String te2 = temp;
temp = te2 + “*”;
}
return temp;
}

    public String censorStr(String str)
{
    String temp = str;
    try
    {
        BufferedReader in = new BufferedReader(new FileReader("data/censor.txt"));
        String data = null;
        while ((data = in.readLine()) != null)
        {
                    String temp2 = data;
                    temp = temp.replaceAll(temp2, GenerateC(temp2.length()));
        }
    }
    catch (IOException e)
    {
        System.out.println("Critical error while checking is a censored word");
        e.printStackTrace();
    }
    return temp;

}[/code]2. And change your appendPlayerChatText to this

protected void appendPlayerChatText(stream str) { str.writeWordBigEndian(((chatTextColor&0xFF) << 8) + (chatTextEffects&0xFF)); str.writeByte(playerRights); str.writeByteC(chatTextSize); // no more than 256 bytes!!! String temp = misc.textUnpack(chatText, chatTextSize); String temp2 = censorStr(temp); misc.textPack(chatText, temp2); str.writeBytes_reverse(chatText, chatTextSize, 0); }
