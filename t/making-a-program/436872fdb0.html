I just started learning Python yesterday. Did another quadratic formula calculator, this time in Python.

Give suggestions, comments, etc.

# ########################### #
# Quadratic Formula Calculator
# by Whac
#
#
# Now in Python!
# March 24, 2012
# ########################### #

import string;
import random;
import sys;
import math;

print """
============================
Quadratic Formula Calculator
============================

Enter a, b, and c for the equation:
ax^2 + bx + c = 0

""";

try:
	# parts make it simpler to understand vs
	# it being done in one line, etc.
	a = float(raw_input("A: "));
	b = float(raw_input("B: "));
	c = float(raw_input("C: "));
	
	part1 = (-1 * b);
	part2 = math.sqrt(math.pow(b, 2) - (4 * a * c));
	part3 = 2 * a;
	
	x1 = (part1 + part2) / part3;
	x2 = (part1 - part2) / part3;
	
	print "X can be " + str(x1) + " or " + str(x2) + "\n";
except NameError:
	print "Invalid input. Please try again.\n";
except ValueError:
	print "Invalid input. Please try again.\n";
