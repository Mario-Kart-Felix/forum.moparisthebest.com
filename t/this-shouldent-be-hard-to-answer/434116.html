@echo off
Title File Renamer
cls
:Start1
cls
set /p FN=Type the name of the foler you would like to to rename (enter MF to make a foler):

IF '%FN%' == 'MF' GOTO MF
IF '%FN%' == 'mf' GOTO MF


IF NOT EXIST %FN% Goto error1.



:Start 2
cls
set /p NN=Type the new file name:

IF EXIST %NN% Goto error2

ren %FN% %NN%
echo %FN% renamed to %NN% successfuly
pause
Exit

:MF
cls
set /p NFN=Type the folder name of the folder you would like to make:
MD %NFN%
echo You will be directe to the beginning.
pause
Goto Start1

:error1
cls
echo Sorry! That file does not exixt please try again. (Remember to put the file extensions in! eg .txt,.exe,.bat,.doc!).
Pause
Goto Start1

:error2
cls

exit
