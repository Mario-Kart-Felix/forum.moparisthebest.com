Here is one i wrote.

import java.util.ArrayList;

import org.apache.bcel.classfile.Method;
import org.apache.bcel.generic.ClassGen;
import org.apache.bcel.generic.Instruction;
import org.apache.bcel.generic.MethodGen;
import org.apache.bcel.generic.SIPUSH;

public class Version {
   public static final int lastKnownClientVersion = 500;

   public static int getVersion(ClassGenList cgs) {
      ClassGen client = cgs.findClass("client");
      for (Method m : client.getMethods()) {
         if (m.getName().equals("main")) {
            ArrayList<Short> shorts = new ArrayList<Short>();
            for (Instruction i : (new MethodGen(m, client.getClassName(), client.getConstantPool())).getInstructionList().getInstructions()) {
               if (i instanceof SIPUSH) {
                  shorts.add(((SIPUSH) i).getValue().shortValue());
               }
            }
            for (int i = 0; i < shorts.size();) {
               if (shorts.get(i) < lastKnownClientVersion)
                  shorts.remove(i);
               else
                  i++;
            }
            for (int i = 0; i < shorts.size(); i++) {
               if (shorts.get(i) == 503) {
                  shorts.remove(i);
                  break;
               }
            }
            for (int i = 0; i < shorts.size(); i++) {
               if (shorts.get(i) == 765) {
                  shorts.remove(i);
                  break;
               }
            }
            java.util.Collections.sort(shorts);
            return shorts.get(0);
         }
      }
      return -1;
   }
}
