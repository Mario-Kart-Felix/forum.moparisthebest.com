skull = 1 melee = 64

If you use delta, you might need to replace your client.java with these methods to get prayer fully working! [SERVER SIDE]

==============================================================================================================

public void checkHead(){ if(!isSkulled){ headIcon = 0; } if(isSkulled){ headIcon = 1; } } public void getHead(){ if(ProtMage && !isSkulled) headIcon = 4; if(ProtRange && !isSkulled) headIcon = 2; if(ProtMelee && !isSkulled) headIcon = 64; if(Retribution && !isSkulled) headIcon = 8; if(Redemption && !isSkulled) headIcon = 32; if(Smite && !isSkulled) headIcon = 16; if(ProtMage && isSkulled) headIcon = 4; if(ProtRange && isSkulled) headIcon = 2; if(ProtMelee && isSkulled) headIcon = 64; if(Retribution && isSkulled) headIcon = 8; if(Redemption && isSkulled) headIcon = 32; if(Smite && isSkulled) headIcon = 16; else if(!ProtMage && !ProtRange && !ProtMelee && !Retribution && !Redemption && !Smite) headIcon = 1; } public void turnOffHead(){ if(ProtMage) headIcon = 4; if(ProtRange) headIcon = 2; if(ProtMelee) headIcon = 64; if(Retribution) headIcon = 8; if(Redemption) headIcon = 32; if(Smite) headIcon = 16; else if(!isSkulled && !ProtMage && !ProtRange && !ProtMelee && !Retribution && !Redemption && !Smite) headIcon = 0; }
