if(!is_dir($hashes_location))
mkdir($hashes_location) or die(‘Failed to create hashes location.’);

$characters = array();

$characters[] = $number;
foreach(range(‘a’,‘z’) as $letter)
$characters[] = $letter;
foreach(range(‘A’,‘Z’) as $capital_letter)
$characters[] = $capital_letter;
foreach(array(’!’,’@’,’#’,’$’,’%’,’^’,’&’,’*’,’(’,’)’,’_’,’+’,’{’,’}’,’|’,’:’,’"’,’<’,’>’,’?’,’-’,’=’,’[’,’]’,’\’,’;’,’’’,’,’,’.’,’/’) as $character)
$characters[] = $character;

$md5_nodupes = false;
$sha1_nodupes = false;
$md5_nodupes_count = 0;
$sha1_nodupes_count = 0;
$wrote_count = false;

function hasher($length,$current_position,$current_string)
{
global $rainbow_table_root,$hashes_location,$characters,$md5_nodupes,$sha1_nodupes,$md5_nodupes_count,$sha1_nodupes_count,$wrote_count;
if($length == $current_position)
{
$skip_md5 = $skip_sha1 = $written = false;

	$passwd = $current_string;

	$md5 = md5($passwd);
	$sha1 = sha1($passwd);

	if(!is_dir($hashes_location.DS.$length))
		mkdir($hashes_location.DS.$length) or die('Failed to create '.$hashes_location.DS.$length);
	if(!is_dir($hashes_location.DS.$length.DS.'md5'))
		mkdir($hashes_location.DS.$length.DS.'md5') or die('Failed to create '.$hashes_location.DS.$length.DS.'md5');
	if(!is_dir($hashes_location.DS.$length.DS.'sha1'))
		mkdir($hashes_location.DS.$length.DS.'sha1') or die('Failed to create '.$hashes_location.DS.$length.DS.'sha1');

	$filename = $hashes_location.DS.$length.DS.'md5'.DS.substr($md5,0,4).'.txt';
	$pointer = fopen($filename,'a+');
	if($md5_nodupes === false)
	{
		$contents = fread($pointer,filesize($filename));
		if(!strpos($contents,$md5)) // to prevent dupes
		{
			if($md5_nodupes_count > 5)
				$md5_nodupes = true;
			$md5_nodupes_count++;
			$written = fwrite($pointer,$passwd.' '.$md5."\n");
		}
		else
			$skip_md5 = true;
	}
	else
		$written = fwrite($pointer,$passwd.' '.$md5."\n");
	fclose($pointer);

	$filename = $hashes_location.DS.$length.DS.'sha1'.DS.substr($sha1,0,4).'.txt';
	$pointer = fopen($filename,'a+');
	if($sha1_nodupes === false)
	{
		$contents = fread($pointer,filesize($filename));
		if(!strpos($contents,$sha1)) // to prevent dupes
		{
			if($sha1_nodupes_count > 5)
				$sha1_nodupes = true;
			$sha1_nodupes_count++;
			$written = fwrite($pointer,$passwd.' '.$sha1."\n");
		}
		else
			$skip_sha1 = true;
	}
	else
		$written = fwrite($pointer,$passwd.' '.$sha1."\n");
	fclose($pointer);

	if(time() % 10 == 0 && $wrote_count === false) // update this less often
	{
		$pointer = fopen($rainbow_table_root.'count.txt','w');
		$wrote_count = fwrite($pointer,$count);
		fclose($pointer);
	}
	elseif(time() % 10 == 0)
		$wrote_count = true;
	else
		$wrote_count = false;

	if($written !== false)
		echo $passwd."\n"; // reduce printing to a very minimal otherwise the hashing goes slower...
	else
		echo $passwd." skip\n";
	ob_flush();
}
else
	for($index = 0; $index < count($characters); $index++)
		hasher($length,$current_position + 1,$current_string.$characters[$index]);

}

for($length = 0; $length < 10; $length++)
hasher($length,0,’’);
