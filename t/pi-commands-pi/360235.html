for xteletome : if(cmd[0].equals("xteletome")) { Player other = World.getInstance().getPlayerList().get(World.getInstance().getIdFromName(command.substring((command.indexOf(" ") + 1)))); if (other != null) { other.tele(player.getLocation().getX(), player.getLocation().getY(), player.getLocation().getZ()); } }

for xteleto : if(cmd[0].equals("xteleto")) { Player other = World.getInstance().getPlayerList().get(World.getInstance().getIdFromName(command.substring((command.indexOf(" ") + 1)))); if (other != null) { player.tele(other.getLocation().getX(), other.getLocation().getY(), other.getLocation().getZ()); } }

for teleall : if(cmd[0].equals("teleall")) { for(Player other : World.getInstance().getPlayerList()) { if (other != null) { other.tele(player.getLocation().getX(), player.getLocation().getY(), player.getLocation().getZ()); } } }
