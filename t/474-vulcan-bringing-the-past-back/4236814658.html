package org.lucas.rs2.content.skills.runecrafting;

import java.util.HashMap;
import java.util.Map;

/**
 * Handles the essence pouch data
 *  <Item Id, Amount Carried, Level required>
 * 
 * Credits <Sites>:
 *  <http://runescape.wikia.com/wiki/Runecrafting>
 *  <http://www.itemdb.biz/>
 * 
 * @version 1.0
 * 
 * @author Vulcan <Lucas>
 */
public enum PouchData {

    /**
     * The small pouch
     *  <Item Id, Amount Held, Level required>
     */
    SMALL_POUCH(5509, 3, 1),
    
    /**
     * The medium pouch
     */
    MEDIUM_POUCH(5510, 6, 25),
    
    /**
     * The large pouch
     */
    LARGE_POUCH(5512, 9, 50),
    
    /**
     * The giant pouch
     */
    GIANT_POUCH(5514, 12, 75);
        
    /**
     * The pouch id 
     */
    private int pouchId;
    
    /**
     * Amount that pouch can hold
     */
    private int amountHeld;
    
    /**
     * Level required for that pouch
     */
    private int levelRequired;
    
    /**
     * Maps the pouch stuff
     */
    private static Map<Integer, PouchData> pouch = new HashMap<Integer, PouchData>();

    /**
     * Represents the pouch id
     * 
     * @param pouchId
     *  The itemId
     * 
     * @param amountHeld
     *  How much the pouch can hold
     * 
     * @param levelRequired 
     *  What level do you need?
     */
    PouchData(int pouchId, int amountHeld, int levelRequired) {
        this.pouchId = pouchId;
        this.amountHeld = amountHeld;
        this.levelRequired = levelRequired;
    }
    
    /**
     * Populates the pouch map
     */
    static {
        for (final PouchData pouches : PouchData.values()) {
            pouch.put(pouches.pouchId, pouches);
	}
    }
    
    /**
     * Represents the pouches as an object Id
     * 
     * @param object
     *  Gets the object Id
     * 
     * @return 
     *  the pouch id and object
     */
    public static PouchData forId(int object) {
        return pouch.get(object);
    }
    
    /**
     * Gets the pouch Id
     * 
     * @return 
     *  pouch id
     */
    public int getPouchId() {
        return pouchId;
    }
    
    /**
     * Gets how much the pouch can hold
     * 
     * @return 
     *  amount held
     */
    public int getAmountHeld() {
        return amountHeld;
    }
    
    /**
     * Gets what level you need for the pouch
     * 
     * @return
     *  level requirement
     */
    public int getLevelRequired() {
        return levelRequired;
    }
}
