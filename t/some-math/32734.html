Just 3 easily made methods for 3 different math things.

public static double getRange(double[] d){ double high = 0; double low = 32564; for(int x = 0; x < d.length; x++){ if(high < d[x]){ high = d[x]; } if(low > d[x]){ low = d[x]; } } return high - low; }

public static double getMode(double[] d){
		int highestoccurence = 0;
		for(int x = 0; x < d.length; x++){
			int a = getOccurences(d[x], d);
			if(a > highestoccurence){
				highestoccurence = x;
			}
		}
		return d[highestoccurence];
	}

public static double getMean(double[] d){ double total = 0; for(int x = 0; x < d.length; x++){ total += d[x]; } return total / d.length; }
Oh, and getOccurences so you can use getMode…

public static int getOccurences(double d, double[] src){ int returnednum = 0; for(int x = 0; x < src.length; x++){ if(src[x] == d){ returnednum++; } } return returnednum; }
