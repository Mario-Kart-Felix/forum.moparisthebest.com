[code]public class AnExampleClass {
// This is in the global scope.
public static int LSMini[] = {#, #, #}; // public modifier allows it to be accessed by any class. static modifier means it stays the same throughout all instances of this class and can be accessed statically, like AnExampleClass.LSMini as a pose to anExampleClassInstance.LSMini if it wasn’t static.

public void anExampleMethod {
// this is the local scope of this method.
