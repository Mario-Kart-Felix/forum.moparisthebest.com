Name: Lure_Fisher

Purpose: To Own Fishing!

How To Use: [i]Start near a fishing spot with lure at the top… Preferably barb village, Enter color of the fishing spot and change the 1337 to the amount of milliseconds it takes you fish by fish, then run the script and watch your levels rise, must have fly fishing rod and feather in first 2 slots!

Note:

Please post progress reports if you use… THANKS!

Hope to have forms out soon! and auto color :D[/i]

[code]program Lure_Fisher;

{.include SRL\SRL.scar}

var

drops:integer;

const
fish1 = 0; // Color of fish spot 1
fishwait = 1337; // Time in milliseconds to click again!

procedure startfish;
begin
repeat
findcolorspiral(x,y,fish1,3,3,515,336)
mmouse(x,y,3,3)
istextat2(9,9,‘Lure’,20)
mouse(x,y,2,2,true)
wait(fishwait+random(300))
until(invfull)
if (invfull = true) then
Dropto(3, 28);
drops := drops + 1;
end;

procedure jellytot;
begin
RotateEvery(3 + random(5));
RandomChatEvery(5 + random(4));
end;

procedure ProgressReport;
begin
WriteLn(’<===================================================================>’);
WriteLn(’ ‘);
WriteLn(‘Dropped ’ + inttostr(drops) + ’ Fish.’);
WriteLn(’ ');
WriteLn(‘Made by Rolling On Java’);
end;

procedure findrandoms;
begin
FindNormalRandoms;
end;

begin
SetupSRL;
activateClient;
cleardebug;
disguiseScar(‘Starting in 3’);
wait(1000);
disguiseScar(‘Starting in 2’);
wait(1000);
disguiseScar(‘Starting in 1’);
wait(1000);
disguiseScar(‘Started!!!’);
wait(1000);
disguiseScar(‘Page 404 - URL Not Found’);
repeat
jellytot;
startfish;
findrandoms;
until(false)
end.[/code]

Enjoy
