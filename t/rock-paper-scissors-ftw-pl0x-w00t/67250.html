public boolean checktype()throws IOException
{
    KeyboardReader type = new KeyboardReader();
    String checkit = type.readString("GUI or Regular?");
    if(checkit.charAt(0) == 'r' || checkit.charAt(0) == 'R')

        return false;
        return false;
}
