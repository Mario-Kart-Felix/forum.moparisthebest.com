private long lastReport;

private int showAbuseBox = 0;

private String offender = "";

private boolean isOkToReport() {
	return (System.currentTimeMillis() - lastReport) >= 60000;
}

public void setLastReport() {
	lastReport = System.currentTimeMillis();
}

private void handleAbuseClick(int type) {
	showAbuseBox = 0;
	if(!isOkToReport()) {
		displayMessage("You already sent a snapshot under 60 secs ago! Do not abuse this system", 3, 0);
		return;
	}
	setLastReport();
	super.streamClass.createPacket(230);
	super.streamClass.addTwo4ByteInts(DataConversions.usernameToHash(offender));
	super.streamClass.addByte(type);
	super.streamClass.formatPacket();
	offender = "";
}

public static String capitalizeString(String string) {
	  char[] chars = string.toLowerCase().toCharArray();
	  boolean found = false;
	  for (int i = 0; i < chars.length; i++) {
	    if (!found && Character.isLetter(chars[i])) {
	      chars[i] = Character.toUpperCase(chars[i]);
	      found = true;
	    } else if (Character.isWhitespace(chars[i]) || chars[i]=='.' || chars[i]=='\'') { // You can add other chars here
	      found = false;
	    }
	  }
	  return String.valueOf(chars);
	}
