Here is a script. I dont know if it works because I have no idea on how to select the client if its a fullscreen game. But here it is:

program New;
var s, d, f, g : integer;

procedure forcecarbon;
begin
repeat
sendkeys('forcecarbon' + chr(13));
wait(500);
s := s + 1;
until(s = 21);
end;

procedure forcefood;
begin
repeat
sendkeys('forcefood' + chr(13));
wait(500);
d := d + 1;
until(d = 21);
end;

procedure forcenova;
begin
repeat
sendkeys('forcenova' + chr(13));
wait(500);
f := f + 1;
until(f = 21);
end;

procedure forceore;
begin
repeat
sendkeys('forceore' + chr(13));
wait(500);
g := g + 1;
until(g = 21);
end;

begin
activateclient;
forcecarbon;
forcefood;
forcenova;
forceore;
end.
