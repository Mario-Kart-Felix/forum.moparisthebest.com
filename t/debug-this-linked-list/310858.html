#include “list.h”

void init_list(list *l)
{

l->a = malloc(sizeof(void *) * 10);
l->max = 10;
l->index = 0;

}

int size(list *l)
{
return l->index;
}

void add(list *l, void *obj)
{
if (l->index >= l->max) {
realloc((l->a), sizeof(void *) * (l->max * 2));
l->max *= 2;
}

l->a[l->index++] = obj;

}

void delete(list *l, int index)
{
l->a[index] = NULL;

int i = index;
int a;
for (a = 0; a < (l->index - index)-1; a++, i++)
    l->a[i] = ((i + 1) == l->index) ? NULL : l->a[i+1];

l->index--;

}

void *get(list *l, int index)
{
return l->a[index];
}

void clear(list *l)
{
int i;
for (i = 0; i < l->index; i++) {
free(l->a[i]);
l->a[i] = NULL;
}
}

void dispose(list *l)
{
int i = 0;
for (; i < l->index; i++) {
