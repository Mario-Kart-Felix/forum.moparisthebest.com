It’s because they noticed it was more-or-less causing subtle bugs because people would forget to put a break, when they really meant to have on there. Although, you can do this to have the same effect (to make it explicit so it’s what you’re actually meaning to do)

using System;

namespace Test
{
	class MainClass
	{
		public static void Main (string[] args)
		{
			int x = 3;
			
			switch(x) {
			case 1: // No code, it doesn't require a 'goto'
			case 2: Console.WriteLine("2"); goto case 3;
			case 3: Console.WriteLine("3"); break;
			default:
				Console.WriteLine("DEFAULT"); break;
			};
		}
	}
}
