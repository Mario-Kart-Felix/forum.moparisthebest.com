/**
 * 		
 * @author tateiscoool
 * 
 * A class handling the start of an attack
 *
 */
public class AttackHandler {

	/**
	 * The entity that is attacking
	 */
	private final Entity ATTACKER;

	/**
	 * The entity which is under attack
	 */
	private Entity victim;

	/**
	 * The timer of when the player can next hit
	 */
	private int attackTimer = -1;
	
	/**
	 * An integer holding the spell id for if the player is performing a single spell
	 */
	private int spellId = -1;
	
	/**
	 * An integer holding the spell id when a player is autocasting
	 */
	private int autocastId = -1;

	/**
	 * A list containing the entity's current
	 * pending hits
	 */
	public LinkedList<Hit> Hits = new LinkedList<Hit>(); 
	
	/**
	 * Handles the special attack class for each player
	 */
	private SpecialAttack specialAttack;
	
	/**
	 * Contains whether the special button has been clicked or not
	 */
	private boolean specialClicked = false;

	/**
	 * Gets the attack class for the attack style
	 * @return the attack class
	 */
	private Attack attack() {
		ATTACKER.setAttackType(getAttackType());
		if (ATTACKER.getAttackType() == AttackTypes.MELEE) {
			return new MeleeAttack(ATTACKER, victim);
		} else if (ATTACKER.getAttackType() == AttackTypes.RANGE) {
			return new RangeAttack(ATTACKER, victim);
		} else if (ATTACKER.getAttackType() == AttackTypes.MAGIC) {
			return new MagicAttack(ATTACKER, victim);
		} else {
			return null;
		}
	}

	/**
	 * Sets the attack type the entity is using
	 */
	private AttackTypes getAttackType() {
		if (getSpellId() != -1 || getAutocastId() != -1)
			return AttackTypes.MAGIC;
		else if (ATTACKER instanceof Player) {
			Player player = (Player) ATTACKER;
			if (player.getEquipment().getItemContainer().get(Constants.EQUIPMENT_SLOT_WEAPON) != null) {
				if (Equipment.isRanged(player.getEquipment().getItemContainer().get(Constants.EQUIPMENT_SLOT_WEAPON).getId()))
					return AttackTypes.RANGE;
			}
		} 
		return AttackTypes.MELEE;
	}

	/**
	 * Initiates the class
	 * @param entity the attacker
	 */
	public AttackHandler(Entity entity) {
		ATTACKER = entity;
		if (entity instanceof Player)
			specialAttack = new SpecialAttack((Player)entity);
	}


	/**
	 * This method starts the attack between the attacker
	 * and the victim.
	 * It is used to run all the checks that must be run before
	 * any type of combat can be initiated. E.g. that the attack
	 * timer has run down.
	 * 
	 * @param attackType The way in which the attacker called this method
	 * @param victim The /possible/ victim of the attack (Checks if possible first)
	 */
	public void startAttack(AttackType attackType, Entity victim, int spellId) {
		if (getAutocastId() != -1)
			setSpellId(getAutocastId());
		setSpellId(spellId);
		if (attackTimer > 0 || !canAttack(victim)) {
			return;
		}
		ATTACKER.turnTo(victim);
		this.victim = victim;
		if (specialClicked && getSpellId() == -1) {
			getSpecialAttack().doSpecialAttack(victim);
			return;
		}
		attack().startAttack();
	}

	public void attackProcess() {
		attackTimer--;
		if (victim != null)
			if (victim.isDead())
				victim = null;
		if (victim instanceof Player)
			if (((Player) victim).getLoginStage() == LoginStages.LOGGED_OUT)
				victim = null;
		if (attackTimer <= 0 && victim != null) {
			startAttack(AttackType.CONTINUED_ATTACK, victim, -1);
		}
		for (int i = 0; i < Hits.size(); i++) {
			Hits.get(i).decreaseHitTimer();
			if (Hits.get(i).getNeedsRemoval()) {
				if (ATTACKER instanceof Player)
					attack().addExp(((Player) ATTACKER), Hits.get(i).getDamage());
				Hits.get(i).getVictim().getAttackHandler().setVictim(ATTACKER);
				Hits.remove(i);
			}
		}
	}

	/**
	 * A method which resets the attack
	 * of an entity. To be used when the
	 * attacker dies.
	 */
	public void resetAttackOnDeath() {
		Hits.clear();
		victim = null;
		attackTimer = -1;
	}

	/**
	 * Resets attack for when the attacker moves
	 */
	public void resetAttackOnMove() {
		victim = null;
	}

	/**
	 * Sets the attack timer
	 * @param time the attacktimer
	 */
	public void setAttackTimer(int time) {
		attackTimer = time;
	}
	
	/**
	 * @return the attackTimer
	 */
	
	public int getAttackTimer() {
		return attackTimer;
	}


	/**
	 * @return the spellId
	 */
	public int getSpellId() {
		return spellId;
	}
	

	/**
	 * @param spellId the spellId to set
	 */
	public void setSpellId(int spellId) {
		this.spellId = spellId;
	}
	
	/**
	 * @return the autocastid
	 */
	public int getAutocastId() {
		return autocastId;
	}
	
	/**
	 * @param autocastId the autocastid to set
	 */
	public void setAutocastId(int autocastId) {
		this.autocastId = autocastId;
	}

	/**
	 * Changes the target a player is attacking
	 * @param entity the new victim
	 */
	public void setVictim(Entity entity) {
		this.victim = entity;
	}
	
	/**
	 * Sets a spell to autocast
	 * @param buttonId The button pressed
	 */
	public boolean setAutocast(int buttonId) {
		if (ATTACKER instanceof Npc)
			return false;
		Player player = (Player) ATTACKER;
		for (int i = 0; i < player.getCombatUtils().autocastButtons.length; i++) {
			if (player.getCombatUtils().autocastButtons[i][0] == buttonId) {
				this.autocastId = (player.getCombatUtils().autocastButtons[i][1]);
				return true;
			}
		}
		return false;
	}

	/**
	 * An enum containing all the possible ways the main
	 * method of this class can be used from.
	 */
	public static enum AttackType {
		MAGIC_ON_NPC,
		MAGIC_ON_PLAYER,
		ATTACK_PLAYER,
		ATTACK_NPC,
		CONTINUED_ATTACK,
		RETALIATION;
	}

	/**
	 * Adds a hit.
	 * @param attackTypes What attack type is being used
	 * and therefore which max hit needs to be fetched.
	 */
	public void addHit(Entity victim, AttackTypes attackType, int endGfx, int movementDelay, int specialDamage) {
		int nextDamage = 0;
		boolean attackHits = true;
		if (ATTACKER instanceof Player) {
			Player attackingPlayer = (Player) ATTACKER;
			if (victim instanceof Player) {
				Player victimPlayer = (Player) victim;
				int attackerChance = attackingPlayer.getCombatUtils().getHitChance();
				int victimDefence = victimPlayer.getCombatUtils().getDefence(attackingPlayer.getAttackStyle());
				if (attackerChance <= victimDefence) {
					if (new SecureRandom().nextInt(100) <= (double)((double) (attackerChance - 1) / (2 * victimDefence)) * 100) {
						attackHits = true;
					} else {
						attackHits = false;
					}
				} else if (attackerChance > victimDefence) {
					if (new SecureRandom().nextInt(100) <= (double) (1 - ( (double) (victimDefence + 1) / (2 * attackerChance))) * 100) {
						attackHits = true;
					} else {
						attackHits = false;
					}
				}
			}
			if (attackHits && attack().getMaxHit() != -1) {
				nextDamage = new SecureRandom().nextInt(attack().getMaxHit() + 1);
			} else
				nextDamage = 0;
		} else if (ATTACKER instanceof Npc && attack().getMaxHit() != -1) {
			nextDamage = new SecureRandom().nextInt(attack().getMaxHit());
		}
		if (attack().getMaxHit() == -1)
			nextDamage = -1;
		if (attackType != AttackTypes.MAGIC && attackType != AttackTypes.RANGE && attackType != AttackTypes.SPECIAL)
			Hits.add(new Hit(victim, nextDamage, ATTACKER.grabHitTimer(), attackType, endGfx, movementDelay));
		else if (attackType == AttackTypes.SPECIAL)
			Hits.add(new Hit(victim, specialDamage, ATTACKER.grabHitTimer(), attackType, endGfx, movementDelay));
		else
			Hits.add(new Hit(victim, nextDamage, ATTACKER.grabProjectileHitTimer(ATTACKER.getPosition(), victim.getPosition()), attackType, endGfx, movementDelay));
		if (attackType != AttackTypes.MAGIC) {
			if (new SecureRandom().nextInt(100) <= 20) {
				if (ATTACKER instanceof Player) {
					if (WeaponInfo.checkWeaponForPoison(((Player)ATTACKER).getEquipment().getItemContainer().get(Constants.EQUIPMENT_SLOT_WEAPON)))
						victim.applyPoison(WeaponInfo.getPoisonDamageForWeapon(((Player)ATTACKER).getEquipment().getItemContainer().get(Constants.EQUIPMENT_SLOT_WEAPON)));
				}
			}
		}
	}

	/**
	 * Gets the victim
	 */
	public Entity getVictimEntity() {
		return victim;
	}

	/**
	 * Checks if the attacker can attack the victim
	 * @param victim the victim
	 */
	public boolean canAttack(Entity victim) {
		if ((ATTACKER == null) || (victim == null)) {
			return false;
		}
		if ((ATTACKER.isDead()) || (victim.isDead())) {
			return false;
		}
		if ((ATTACKER instanceof Player) && (victim instanceof Player)) {
			if (!Areas.isInWildernessArea(victim.getPosition())) {
				Player player = (Player) ATTACKER;
				player.getActionSender().sendMessage("That player is not in the wilderness.");
				return false;
			}
			if (!Areas.isInWildernessArea(ATTACKER.getPosition())) {
				Player player = (Player) ATTACKER;
				player.getActionSender().sendMessage("You are not in the wilderness.");
				return false;
			}
		}
		if (victim.getAttackHandler().getVictimEntity() != ATTACKER && victim.getAttackHandler().getVictimEntity() != null && !Areas.isInMultiArea(victim.getPosition())) {
			if (ATTACKER instanceof Player) {
				Player player = (Player) ATTACKER;
				player.getActionSender().sendMessage("Your target is already in combat.");
			}
			return false;
		}
		if (Misc.getDistance(ATTACKER.getPosition(), victim.getPosition()) == 0) {
			//TODO: Make this do something
			return false;
		}
		if (!distanceCheck(victim)) {
			ATTACKER.setFollowingCombat(victim);
			return false;
		}
		return true;
	}

	/**
	 * Checks the distance between the entity and attacker
	 * @return
	 */
	private boolean distanceCheck(Entity victim) {
		if (getAttackType() == AttackTypes.RANGE) {
			if (Misc.getDistance(ATTACKER.getPosition(), victim.getPosition()) <= 7) {
				if (ATTACKER instanceof Player)
					((Player) ATTACKER).getMovementHandler().reset();
				else if (ATTACKER instanceof Npc)
					((Npc) ATTACKER).getNpcMovement().reset();
				return true;
			}
		} else if (getAttackType() == AttackTypes.MAGIC) {
			if (Misc.getDistance(ATTACKER.getPosition(), victim.getPosition()) <= 7) {
				if (ATTACKER instanceof Player)
					((Player) ATTACKER).getMovementHandler().reset();
				else if (ATTACKER instanceof Npc)
					((Npc) ATTACKER).getNpcMovement().reset();
				return true;
			}
		} else if (getAttackType() == AttackTypes.MELEE) {
			if (Misc.getDistance(ATTACKER.getPosition(), victim.getPosition()) == 1) {
				return true;
			}
		}
		return false;
	}
