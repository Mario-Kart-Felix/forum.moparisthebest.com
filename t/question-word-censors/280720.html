Here’s something i made quick:

[code]
class Test2 {

public static void main(String[] args) {
	String a = "i am not a fuckface";
	String[] b = { "fuck" }; 
	for(int i = 0; i < b.length; i++) {
		if(a.contains(b[i])) {
			System.out.println("Word '"+a+"' contained '"+b[i]+"'");
			a = a.replaceAll(b[i], replaceWithChars(b[i].length()));
		}
	}
	System.out.println(a);
}

public static String replaceWithChars(int len) {
	String newString = "";
	char c = '*';
	for(int i = 0; i < len; i++)
		newString += c;
	System.out.println(newString);
	return newString;
}

}[/code]
