Heres an example of an applet I made…use it as a guide (I can’t explain lol):

[code]mport java.awt.BorderLayout;
import java.awt.Container;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.*;
import java.awt.BorderLayout;
import java.awt.Container;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.*;
import java.awt.BorderLayout;
import java.awt.Container;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.*;

public class TipCalc extends JApplet implements ActionListener {

/**
 * @param args
 */
private JTextField bill;	
private JTextField percent;
private JButton button;
private String string1;
private String string2;
private int billint;
private double percentint;
private JTextField answergui;
public static void main(String[] args) {
	// TODO Auto-generated method stub
	TipCalc program = new TipCalc();
	program.setSize(300, 110);
	program.setVisible(true);
	
}
public void init() {
	Container c = getContentPane();
	c.setLayout(new BorderLayout());
	
	JLabel billabel = new JLabel("Bill?");
	bill = new JTextField(5);
	bill.addActionListener(this);
	JLabel perlabel = new JLabel("Percent");
	percent = new JTextField(5);
	percent.addActionListener(this);
	button = new JButton("Add");
	button.addActionListener(this);
	answergui = new JTextField();
	answergui.addActionListener(this);
	
	JPanel top = new JPanel();
	top.add(billabel);
	top.add(bill);			
	top.add(perlabel);	
	top.add(percent);
	top.add(button);
	
	c.add(top, BorderLayout.NORTH);
	c.add(answergui, BorderLayout.SOUTH);
	
}
@Override
public void actionPerformed(ActionEvent e) {
	// TODO Auto-generated method stub
	Object sender = e.getSource();
	if(sender == button){
		string1 = bill.getText();
		billint = Integer.parseInt(string1);
		//System.out.println(billint);
		string2 = percent.getText();
		percentint = Integer.parseInt(string2);
		//System.out.println(percentint);
		double tip = billint * (percentint / 100);
		double total = tip + billint;
		System.out.println("You should pay: $" + total);
		answergui.setText("The tip is: $" + tip + "\n" + "You should pay a total of: $" + total);
		
	}
}

}
[/code]
