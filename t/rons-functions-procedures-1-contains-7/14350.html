// The Replace() function will replace a string with another string.
// Made by Ron. :)
function Replace(TheString, Replacee, Replacer : String) : String;
var
  a : LongInt;
begin
  a := Pos(Replacee, TheString);
  if(a = 0)then
  begin
    // Do nothing...
  end else
  begin
    Delete(TheString, a, Length(Replacee));
    Insert(Replacer, TheString, a);
  end;
  Result := TheString;
end;

Function: FindAllOcc();
Syntax: FindAllOcc(LookFor, TheString : String) : Integer;
What it does: Finds all the occurrences of what you’re looking for in a string and returns how many times it was found.
Note: This function requires the Replace() function before it.

FindAllOcc() function code only.

// The FindAllOcc() function finds all the occurrences of something in a string.
// Made by Ron. :)
Function FindAllOcc(LookFor, TheString : String) : Integer;
Var
  a : Integer;
  b : LongInt;
  FunctStr : String;
Begin
  FunctStr := TheString;
  Repeat
    b := Pos(LookFor, FunctStr);
    If(Not(b = 0))Then
    Begin
      a := a + 1;
    End;
    FunctStr := Replace(FunctStr, LookFor, '');
  Until(b = 0)
  Result := a;
End;

Function: Explode();
Syntax: Explode(Separator, TheString : String): Array of String;
What it does: Looks through a string and separates it with whatever you supplied as a separator, and takes each value and puts them into an array.
Note: This function requires the Replace() and FindAllOcc() function before it.

Explode() function code only:

// The Explode procedure looks through a string and separates it with whatever
// you suplied as a separator, and takes each value and puts them into an array.
// Example: Explode('*', 'a*b*c*d*e*');
// Made by Ron. Fixed by Weibs. :)
function Explode(Separator, TheString : String) : Array of String;
var
  ArrayName : Array of String;
  FunctStr, c : String;
  a, i : Integer;
begin
  FunctStr := TheString;
  a := FindAllOcc(Separator, FunctStr);
  SetArrayLength(ArrayName, a+1);
  for i := 0 to a do
  begin
    if(i = a)then
    begin
      ArrayName[i] := FunctStr;
    end else
    begin
      c := Copy(FunctStr, 1, Pos(Separator, FunctStr)-1);
      ArrayName[i] := c;
      FunctStr := Replace(FunctStr, c+Separator, '');
    end;
  end;
  Result := ArrayName;
end;

Function: Shift();
Syntax: Shift(TheString : String; HTML : Boolean) : String;
What it does: It basically acts as a shift button and converts numbers, letters, and chars to its Shift form.
Note: This function requires the Replace() function before it.

Shift() function code only:

// The Shift() function will type in TheString as if shift was down.
// Made by Ron. :)
Function Shift(TheString : String; HTML : Boolean) : String;
Var
  i, c : Integer;
  Shifts, ShiftsA, Betw : String;
Begin
  Shifts  := '1234567890`-=[]\;''/,.';
  ShiftsA := '!@#$%^&*()~_+{}|:"?<>';
  For c := 1 to Length(TheString) do
  Begin
    For i := 1 to Length(Shifts) do
    Begin
      If(StrGet(TheString, c) = StrGet(Shifts, i))Then
      Begin
        TheString := Replace(TheString, StrGet(Shifts, i), StrGet(ShiftsA, i));
      End;
    End;
  End;
  TheString := Uppercase(TheString);
  If(HTML)Then
  Begin
    TheString := Replace(TheString, '<', '&lt;');
    TheString := Replace(TheString, '>', '&gt;');
  End;
  Result := TheString;
End;

Function: ShiftB();
Syntax: ShiftB(TheString, Between1, Between2 : String; HTML : Boolean) : String;
What it does: Acts just like Shift() except you can set what to convert to the shift form by setting up Between1 and Between2.
Note: This function requires the Replace() function before it if you are going to use it.

ShiftB() function code only:

// The ShiftB() function will type in between Between1 and Between2
// as if shift was down. Made by Ron. :)
Function ShiftB(TheString, Between1, Between2 : String; HTML : Boolean) : String;
Var
  i, c : Integer;
  Shifts, ShiftsA, Betw : String;
  Pos1 : LongInt;
Begin
  Shifts  := '1234567890`-=[]\;''/,.';
  ShiftsA := '!@#$%^&*()~_+{}|:"?<>';
  Betw := Uppercase(Between(Between1, Between2, TheString));
  Pos1 := Pos(Between1, TheString);
  Delete(TheString, Pos1+Length(Between1), Length(Betw));
  For c := 1 to Length(Betw) do
  Begin
    For i := 1 to Length(Shifts) do
    Begin
      If(StrGet(Betw, c) = StrGet(Shifts, i))Then
      Begin
        Betw := Replace(Betw, StrGet(Shifts, i), StrGet(ShiftsA, i));
      End;
    End;
  End;
  If(HTML)Then
  Begin
    Betw := Replace(Betw, '<', '&lt;');
    Betw := Replace(Betw, '>', '&gt;');
  End;
  Insert(Betw, TheString, Pos1+Length(Between1));
  Result := TheString;
End;

Procedure: CreateFile();
Syntax: CreateFile(Information, Location, Title : String);
What it does: Creates a file by putting in the Information into the Location of the file, and labels it with the Title.

CreateFile() function code only:

// CreateFile will put in the information into the location of the file
// and label it with the Title. Made by Ron. :)
Procedure CreateFile(Information, Location, Title : String);
Var
  OutFile : Integer;
Begin
  OutFile := ReWriteFile(Location, False);
  // Check to see if the file was permitted.
  If(OutFile < 0)Then
  Begin
    // File not permitted, terminating.
    TerminateScript;
  End;
  WriteFileString(OutFile, Information);
  CloseFile(OutFile);
  WriteLn('Saved the file "' + Title + '" to:');
  WriteLn('  ' + Location);
End;
