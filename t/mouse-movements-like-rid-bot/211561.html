[code=java]public class HumanMouseMovement {

public HumanMouseMovement() { }

public static long fittsLaw(double targetDist, double targetWidth, double reactTime, double timePerBit) {
	return (long)(reactTime + timePerBit * (Math.log10(targetDist / targetWidth + 1.0D) / Math.log10(2D)));
}

public static long hicksLaw(double options, double timePerBit) {
	return (long)(timePerBit * (Math.log10(options + 1.0D) / Math.log10(2D)));
}

public static long reactionExpected(int options, double focus, double activity) {
	double reaction = hicksLaw(options, 125D + 50D * Math.random());
	return fittsLaw(activity, focus, reaction, 125D + 50D * Math.random());
}

public static long reactionRandom(double focus, double work) {
	double reaction = hicksLaw(100D * work, 125D + 50D * Math.random());
	return fittsLaw(work, 1.0D - focus, reaction, 125D + 50D * Math.random());
}

public static long hesitance(double focus, double activity) {
	return fittsLaw(activity, focus, 25D + 25D * Math.random(), 50D + 50D * Math.random());
}
