Something like this should work.

program New;

var
  x, y, x2, y2 : integer;

begin
  ClearDebug;
  GetClientDimensions(x2, y2);
  ActivateClient;
  if(FindColor(x, y, 16777215, 0, 0, x2, y2))then
  begin
    HoldMouse(x, y, True);
    MoveMouse(0, 0);
    ReleaseMouse(0, 0);
  end;
end.
