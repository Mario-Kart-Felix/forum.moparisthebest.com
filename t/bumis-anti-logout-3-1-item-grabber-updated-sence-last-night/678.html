program BumAntiLogout;
{.include SSI2.txt}
{.include BoxBreaker2.txt}
var
 ItemColor,TextColor: Integer;
 ItemGrabbed: Integer;
 InvenCount,RandomInven,DoInvenCount,CheckFighting: Integer;
 DoingInven,DoingMain,CheckStats,CheckingStats: Integer;
 IAmFighting,DoingSpecial: Boolean;
Procedure Variables;
Begin
 //Set Item Name On Line-
 ItemColor:=1346321
 TextColor:=3437773
 UpChars:= LoadChars2(AppPath + 'CharsRS22\');
 ChatChars:= LoadChars2(AppPath + 'CharsChat2\');
 InvenCount:=0
End;

Procedure Review;
Begin
 Writeln('')
 Writeln('')
 Writeln('::::::::::::::::::::::::::::::::::::::::::::::::')
 Writeln(': Anti Logout v3/Bumis')
 Writeln(': Picked up items ' + IntToStr(ItemGrabbed) + ' times.')
 If DoingInven=1 Then
  Begin
   Writeln(': Current Process: Doing Inventory')
  End;
 If DoingMain=1 Then
  Begin
   Writeln(': Current Process: Doing Main Procedure')
  End;
 If IAmFighting=True Then
  Begin
   Writeln(': Current Process: Doing Special')
  End;
 If CheckingStats=1 Then
  Begin
   Writeln(': Current Process: Checking Stats')
  End;
 Writeln('::::::::::::::::::::::::::::::::::::::::::::::::')
Writeln('')
Writeln('')
End;

Procedure DoInven;
Begin
Wait(400+Random(300))
DoInvenCount:=0
DoingInven:=1
FindStrangeBox;
 Repeat
  Review;
  Status('Doing Inventory')
  DoInvenCount:=DoInvenCount+1
  RandomInven:=Random(5)
   Case RandomInven Of
    1:
     Begin
      Wait(500+Random(200))
      MMouse(581,225,7,7)
      Wait(1000+Random(2000))
     End;
    2:
     Begin
      Wait(500+Random(200))
      MMouse(611,333,20,20)
      Wait(1000+Random(2000))
     End;
    3:
     Begin
      Wait(5000+Random(5000))
     End;
    4:
     Begin
      MMouse(599,323,20,20)
      Wait(100+Random(100))
      MMouse(394,271,20,20)
      Wait(100+Random(100))
      MMouse(293,108,20,20)
      Wait(500+Random(500))
      MoveMouseSmooth(0+random(510),0+Random(330))
     End;
    5:
     Begin
      MMouse(599,323,20,20)
      Wait(100+Random(100))
      MMouse(394,271,20,20)
      Wait(100+Random(100))
      MMouse(293,108,20,20)
      Wait(500+Random(500))
      MoveMouseSmooth(0+random(510),0+Random(330))
     End;
   End;
Until(DoInvenCount=5+random(6))
DoingInven:=0
End;

Procedure CheckSpecial;
Begin
Review;
IAmFighting:=True
Wait(1300+Random(400))
MMouse(200,200,75,75)
Wait(500+Random(100))
MMouse(559,186,5,5)
Wait(400+Random(200))
GetMousePos(x,y)
ClickMouse(x,y,true)
 If FindColorTolerance(x,y,1203866,615,440,641,452,10) Then
  Begin
   DoingSpecial:=True
   Review;
   MoveMouseSmooth(577+Random(110),442+Random(11))
   Wait(1000+Random(400))
   GetMousePos(x,y)
   ClickMouse(x,y,true)
   Wait(5000+random(1400))
   CheckFighting:=0
  End;
IAmFighting:=False
DoingSpecial:=False
Review;
End;

Procedure AmIFighting;
Begin
 If FindColorTolerance(x,y,65280,187,30,419,243,10) Then
  Begin
   IAmFighting:=True
   Review;
   CheckSpecial;
  End;
IAmFighting:=False
Review;
End;

Procedure CheckMyStats;
Begin
 Wait(500+Random(500))
 MMouse(583,186,7,7)
 Wait(200+Random(100))
 GetMousePos(x,y)
 MMouse(x,y,2,2)
 Wait(200+Random(200))
 MoveMouseSmooth(559+Random(50),212+Random(80))
 Wait(500+Random(300))
 GetMousePos(x,y)
 MMouse(x,y,2,2)
End;

Procedure AntiLogOut;
Begin
DoingMain:=1
Status('Main Procedure')
Review;
Wait(500+random(2000))
GetMousePos(x,y)
MMouse(x,y,2,2)
Wait(1000+Random(3000))
 Begin
  MMouse(361,233,77,77)
  Wait(100+Random(400))
  GetMousePos(x,y)
  MMouse(x,y,5,5)
  Wait(100+Random(400))
  GetMousePos(x,y)
  MMouse(x,y,3,3)
  Wait(100+Random(400))
  MMouse(250,150,250,150)
  If CheckStats=7+Random(2) Then
   Begin
    CheckingStats:=1
    Review;
    CheckMyStats;
    CheckStats:=0
   End;
  InvenCount:=InvenCount+1
  CheckFighting:=CheckFighting+1
  If CheckFighting=1+Random(4) Then
   Begin
    CheckFighting:=0
    DoingMain:=0
    Review;
    AmIFighting;
   End;
  If InvenCount=6+Random(4) Then
   Begin
    MMouse(650,183,4,4)
    DoingMain:=0
    DoInven;
   End;
 Review;
 End;
DoingMain:=0
End;

Procedure LookForItem;
Begin
DoingMain:=1
Review;
Status('Searching for item.')
 If FindColorTolerance(x,y,ItemColor,0,0,466,299,5) Then
  Begin
   MMouse(x,y,5,5)
   Wait(50+Random(50))
   GetMousePos(x,y)
   MMouse(x,y,2,2)
   Wait(150+Random(200))
   If IsTextInAreaEx(36,5,124,25,x,y,'Nature rune',100,UpChars,True,false,0,1,TextColor) then
    Begin
     GetMousePos(x,y)
     ClickMouse(x,y,True)
     MMouse(647,184,5,5)
     Wait(300+Random(300))
     GetMousePos(x,y)
     ClickMouse(x,y,true)
     Wait(500+random(500))
     FindStrangeBox;
     ItemGrabbed:=ItemGrabbed+1
     Review;
     AntiLogout;
    End;
  End;
 DoingMain:=0
End;

begin
Variables;
BoxOpenerSetup;
DisguiseScar('Bearshare Pro 4.5.1');
Writeln('')
Writeln('')
Writeln('')
Writeln('::::::::::::::::::::::::::::::::::::::::::::::::::::')
Writeln(': Anti-LogOut 3.1 / Item Grabber                   :')
Writeln(':  Written by Bumis, Based off of BumTokel Coding  :')
Writeln('::::::::::::::::::::::::::::::::::::::::::::::::::::')
Writeln('')
Writeln('')
Writeln('')
Wait(5000)
IAmFighting:=False
 Repeat
 Review;
 LookForItem;
 Wait(1000+Random(500))
 AntiLogout;
 Until(False)
end.
