public static void openGate(Player player, final GameObject gameObject) {
        GameObject otherGate = null;
        for (GameObject object : player.getRegion().getGameObjects()) {
            if (object.getLocation().getX() == gameObject.getLocation().getX() - 1 || object.getLocation().getX() == gameObject.getLocation().getX() + 1 || object.getLocation().getY() == gameObject.getLocation().getY() - 1 || object.getLocation().getY() == gameObject.getLocation().getY() + 1) {
                for (int closedGateId : closedGateIds) {
                    if (object.getDefinition().getId() == closedGateId) {
                        otherGate = object;
                    }
                }
            }
        }
        if (otherGate == null)
            return;
        GameObject referredOpeningGate = null;
        GameObject openingGate = null;
        if (otherGate.getDefinition().getId() < gameObject.getDefinition().getId())
            referredOpeningGate = otherGate;
        else if (otherGate.getDefinition().getId() > gameObject.getDefinition().getId())
            referredOpeningGate = gameObject;
        if (referredOpeningGate != null) {
            int openingIndex = 0;
            int otherIndex = 0;
            for (int i = 0; i < closedGateIds.length; i++) {
                if (closedGateIds[i] == referredOpeningGate.getDefinition().getId()) {
                    openingIndex = i;
                }
            }
            if (referredOpeningGate.getDefinition().getId() == otherGate.getDefinition().getId()) {
                for (int i = 0; i < closedGateIds.length; i++) {
                    if (closedGateIds[i] == gameObject.getDefinition().getId()) {
                        otherIndex = i;
                    }
                }
            } else {
                for (int i = 0; i < closedGateIds.length; i++) {
                    if (closedGateIds[i] == otherGate.getDefinition().getId()) {
                        otherIndex = i;
                    }
                }
            }
            GameObject otherOpenedGate = null;
            Location otherOpenedGateLocation = null;
            Location originalOpenedGateLocation = null;
            ObjectManager.removeGlobalObject(gameObject);
            ObjectManager.removeGlobalObject(otherGate);
            int rotation = 0;
            if (referredOpeningGate.getRotation() == 0)
                rotation = 4;
            else if (referredOpeningGate.getRotation() == 2)
                rotation = 1;
                originalOpenedGateLocation = Location.create(referredOpeningGate.getLocation().getX() + 1, referredOpeningGate.getLocation().getY(), referredOpeningGate.getLocation().getZ());
                otherOpenedGateLocation = Location.create(originalOpenedGateLocation.getX() + 1, originalOpenedGateLocation.getY(), originalOpenedGateLocation.getZ());
            else if (referredOpeningGate.getRotation() == 3)
                rotation = 2;
                originalOpenedGateLocation = Location.create(referredOpeningGate.getLocation().getX(), referredOpeningGate.getLocation().getY() - 1, referredOpeningGate.getLocation().getZ());
                otherOpenedGateLocation = Location.create(originalOpenedGateLocation.getX(), originalOpenedGateLocation.getY() - 1, originalOpenedGateLocation.getZ());
            else if (referredOpeningGate.getRotation() == 4)
                rotation = 3;
                originalOpenedGateLocation = Location.create(referredOpeningGate.getLocation().getX() - 1, referredOpeningGate.getLocation().getY(), referredOpeningGate.getLocation().getZ());
                otherOpenedGateLocation = Location.create(originalOpenedGateLocation.getX() - 1, originalOpenedGateLocation.getY(), originalOpenedGateLocation.getZ());
            if (rotation == 0)
                otherOpenedGateLocation = Location.create(originalOpenedGateLocation.getX(), originalOpenedGateLocation.getY() + 1, originalOpenedGateLocation.getZ());
                originalOpenedGateLocation = Location.create(referredOpeningGate.getLocation().getX(), referredOpeningGate.getLocation().getY() + 1, referredOpeningGate.getLocation().getZ());
            openingGate = new GameObject(GameObjectDefinition.forId(openGateIds[openingIndex]), originalOpenedGateLocation, referredOpeningGate.getType(), rotation);
            otherOpenedGate = new GameObject(GameObjectDefinition.forId(openGateIds[otherIndex]), otherOpenedGateLocation, referredOpeningGate.getType(), rotation);
            ObjectManager.createGlobalObject(openingGate);
            ObjectManager.createGlobalObject(otherOpenedGate);
            player.getRegion().getGameObjects().add(openingGate);
            player.getRegion().getGameObjects().add(otherOpenedGate);
            gatesPending.add(new GameObject[]{gameObject, otherGate, openingGate, otherOpenedGate});
            World.getWorld().submit(new Event(60000) {
                @Override
                public void execute() {
                    closeGate(gameObject);
                    this.stop();
                }
            });
            // otherOpenedGate = new GameObject(openGateIds[index2], Location.create(player.getLocation().getX() - 1, player.getLocation().getY()))
        }
    }
