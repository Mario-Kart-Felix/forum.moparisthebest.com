	for (people p : people.values()) {
	if(NPC == p.getId() && c.playerLevel[17] <= p.getLevel()) {
		c.sM("You need a thieving level of "+p.getLevel()+" to thieve a "+c.GetNpcName(NPC)+"");
		return;
	}
	if(System.currentTimeMillis() - lastThieve < 2000) {
		return;
	}
	if(c.absX == server.npcHandler.npcs[NPCSlot].absX && c.absY == server.npcHandler.npcs[NPCSlot].absY) {
		c.fmwalkto(0, -1);
		c.TurnPlayerTo(server.npcHandler.npcs[NPCSlot].absX, server.npcHandler.npcs[NPCSlot].absY);			
	}

switch(misc.random(1)) {
case 0:
pickpocket(c, NPC);
lastThieve = System.currentTimeMillis();
break;
case 1:
fail(c, NPC, NPCSlot);
lastThieve = System.currentTimeMillis();
break;
}
}
}

public static void pickpocket(client c, int NPC) {
	for (people p : people.values()) {
	if(NPC == p.getId() && c.playerLevel[17] >= p.getLevel()) {
		c.startAnimation(881);
		c.sM("You pickpocket the "+c.GetNpcName(NPC)+".");
		c.addItem(995, 3 + misc.random(p.getCash()));
		c.addSkillXP(p.getExp(), 17);
	}
}
}

public static void fail(final client player, int NPC, final int NPCSlot) {
	for (final people p : people.values()) {
	if(NPC == p.getId() && player.playerLevel[17] >= p.getLevel()) {
	player.walkDelay = 2000;
	player.lastWalk = System.currentTimeMillis();
	server.npcHandler.failText(NPCSlot, ""+player.playerName+" is stealing!");
	server.npcHandler.npcs[NPCSlot].TurnNpcTo(player.absX, player.absY);
