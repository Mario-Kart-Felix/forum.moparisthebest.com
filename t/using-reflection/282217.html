ReflectionTest.java

import java.net.*;
import java.lang.reflect.*;
import java.io.*;

public class ReflectionTest {
	public static void main(String[] args) throws Exception {
		ClassLoader loader = new URLClassLoader(new URL[] { new File("Test.java").toURI().toURL()});
		Class clazz = loader.loadClass("Test");
		Object client    =  clazz.newInstance(); //Maps to an instance of the client class in your example.
		Field foo   = clazz.getDeclaredField("t");
		foo.setAccessible(true); //Since the field is private we MUST do this.
		Object bar = foo.get(client);
		
		Class clazz2 = bar.getClass();
		Field foo2   = clazz2.getDeclaredField("value");
		foo2.setAccessible(true); //Since the field is private we MUST do this.
		int value    = foo2.getInt(bar);
		System.out.println(value);
	}
}
