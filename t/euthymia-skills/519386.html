 * Basic firemaking class

public class Firemaking {

	

	Client c;

	/*
	 * The constrctor
	 */
	public Firemaking(Client c) {
		this.c = c;
	}

	
	/*
	 * Have we lit the fire yet?
	 */
	public boolean litFire;

        /*
	 *checks if player has level, basically does the checks and	
	 *spawns an unlit log, also it makes us not move.
	 */
	public void preLit(int lvl, int xp, int log) {
		if (c.playerLevel[11] >= lvl) {
			c.toX = c.absX;
			c.toY = c.absY;
			c.addSkillXP(xp, 11);
			c.deleteItem(log, 1);
			// c.setAnimation(733);
			c.GlobalObjects.makeGlobalObject(c.absX, c.absY, 309, 0, 10);
			c.canMove = false;
			lit();
		} else {
		}
	}
		
	/*
	 *So here appears he fire, and allow us to move
	 */
	public void lit() {
		Server.getTaskScheduler().schedule(new Task(2, true) {
			private int count = 1;

			@Override
			protected void execute() {

				if (count > 0) {
					PlayerHandler.messageToAll = count + "...";
					count--;
				} else {
					c.GlobalObjects.deletethatobject(c.absX, c.absY);
					c.GlobalObjects.makeGlobalObject(c.absX, c.absY, 2732, 0, 10);
					c.canMove = true;
					litFire = true;
					stop();

				}
			}
		});
	}

	/*
 	 *This method is just called in the needed packet.
	 */
	public void litAll(int itemUsed, int useWith){
		  if ((itemUsed == 590) && (useWith == 1511)) {
           preLit(1, 200, 1511);

public void smelt(int ore, int ore2, int reqlvl, int xp, int bar) {
	c.closeInterface();
	if (c.playerHasItem(ore, 1) && c.playerHasItem(ore2, 1)
			&& c.playerLevel[13] >= reqlvl) {
		c.deleteItem(ore, 1);
		c.deleteItem(ore2, 1);
		c.addItem(bar, 1);
		c.sM("You make a bar.");
		c.setAnimation(899);
		c.addSkillXP(xp, 13);
	} else {
		if (c.playerLevel[13] < reqlvl) {
			c.sM("You dont have the required level to smelt this bar");
		}
	}
}

public void fletchBow() {
	if (c.fletchAmount < 1) {
		c.resetAction();
		return;
	}
	c.fletchAmount--;
	c.closeInterface();
	c.IsBanking = false;
	c.pEmote = 885;
	c.updateRequired = true;
	c.appearanceUpdateRequired = true;
	if (c.playerHasItem(c.logs[c.fletchLog])) {
		c.deleteItem(c.logs[c.fletchLog], 1);
		c.addItem(c.fletchId, 1);
		c.addSkillXP(c.fletchExp, c.playerFletching);
	} else {
		c.resetAction();
	}
}

public void fletchBow(boolean shortBow, int amount) {
	c.closeInterface();
	if (shortBow) {
		if (c.playerLevel[c.playerFletching] >= c.shortreq[c.fletchLog]) {
			c.fletchId = c.shortbows[c.fletchLog];
			c.fletchExp = c.shortexp[c.fletchLog];
		} else {
			c.sM("You need " + c.shortreq[c.fletchLog] + " to fletch this.");
			c.resetAction();
		}
	} else {
		if (c.playerLevel[c.playerFletching] >= c.longreq[c.fletchLog]) {
			c.fletchId = c.longbows[c.fletchLog];
			c.fletchExp = c.longexp[c.fletchLog];

			c.sM("You need " + c.longreq[c.fletchLog] + " to fletch this.");
			c.resetAction();
		}
	}
	c.fletching = true;
	c.fletchAmount = amount;
}
