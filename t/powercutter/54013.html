{=======================================]
[----------Power Cutter Basic-----------]
[ -----------------------------------   ]
[  Credits: Made by Chris (usernamelol)  ]
[---------------------------------------]
[            How to use:                 ]
[---------------------------------------]
[       1. You must have SRL installed. ]
[       2. Open RuneScape.              ]
[       3. Go to a tree to chop         ]
[       4. Use the dible to get color   ]
[       5. Change treecolor1=15123344;  ]
[       6. Target the RuneScape client  ]
[       7. Compile and run the script!  ]
[       8. Thanks for running Basic PC  ]
[---------------------------------------]
[=======================================}

program powercutter;
{.include srl/srl.scar}
{.include srl/srl/skill/woodcutting.scar}

const
treecolor1=15123344;  //--=MUST CHANGE=--//
timetochop=15000; //--Can change, its telling how long it will chop for--


procedure chop;
begin
if(findcolorspiraltolerance(x,y,treecolor1,msx1,msy1,msx2,msy2,25))then
mmouse(x,y,3,3)
if(IsUpTextMulti('chop','op do','down'))then
begin
mouse(x,y,3,3,true)
end;
end;

procedure treewait;
begin
wait(timetochop)
end;

procedure findrandoms;
begin
FindNormalRandoms;
end;

begin
cleardebug;
setupsrl;
activateClient;
repeat
treewait;
chop;
until(false)
end.
