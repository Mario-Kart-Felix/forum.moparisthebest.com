package org.arsenal.tools;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.IOException;
import java.util.jar.Attributes;
import java.util.jar.JarEntry;
import java.util.jar.JarOutputStream;
import java.util.jar.Manifest;

import javax.swing.JProgressBar;

/**
 * Created under the @see Arsenal Software Licensing 
 * @author Arsenal Development Team
 *
 */
public class AutoJar {
	/**
	 * File Directory of the listing of files to compress
	 */
	public static String dirName = "";
	/**
	 * Output Jar File name
	 */
	public static String outputName = "";
	/**
	 * Filter Word Result
	 */
	private static String filterWord = ".class";
	/**
	 * Main executable class
	 */
	public static String mainFile = "";
	/**
	 * Manifest Version
	 */
	private static String manifestVersion = "1.0";
	/**
	 * Create Manifest File
	 */
	public static boolean isManifest;
	/**
	 * Filter by File Type
	 */
	public static boolean isFiltered;
	/**
	 * Manifest File
	 */
	private static Manifest manifest;
	/**
	 * Progress Bar
	 */
	public static JProgressBar progressBar;
	
	public AutoJar() {
		try {
			if(isManifest) {
				manifest = new Manifest();
				System.out.println("Creating manifest...");
				appendAttributes(manifest);
			} else {
				manifest = new Manifest();
			}
			JarOutputStream out = new JarOutputStream(new FileOutputStream(dirName+"/"+outputName), manifest);
			System.out.println("Getting all files in directory: "+dirName);
			getAllFiles(out);
			out.close();
			System.out.println("Jar Successfully Created.");
		} catch (FileNotFoundException e) {
			e.printStackTrace();
		} catch (IOException e) {
			e.printStackTrace();
		}
	}
	/**
	 * Gets all Files in directory listing
	 * @param out JarOutputStream
	 * @throws IOException
	 */
	public static void getAllFiles(JarOutputStream out) throws IOException {
		File directory = new File(dirName);
		File[] files = directory.listFiles();
		for(int i = 0; i < files.length; i++) {
			/**
			 * If file does not contain filtered word result we skip it from compression
			 */
			if(isFiltered) {
				if(!files[i].getName().endsWith(filterWord))
					return;
			}
			progressBar.setValue(i);
			JarEntry filesToAdd = new JarEntry(files[i].getName());
			filesToAdd.setTime(files[i].lastModified());
			out.putNextEntry(filesToAdd);
			System.out.println("[Compressing] "+files[i].getName());
			compressFile(out, files[i]);
		}
		progressBar.setValue(100);
	}
	/**
	 * Compresses File into output Jar file
	 * @param out JarOutputStream
	 * @param file File to Compress
	 * @throws IOException
	 */
	public static void compressFile(JarOutputStream out, File file) throws IOException {
		byte[] buffer = new byte[10240];
		FileInputStream fileIn = new FileInputStream(file);
		while(true) {
			int length = fileIn.read(buffer, 0, buffer.length);
			if(length <= 0)
				break;
			out.write(buffer, 0, length);
		}
		fileIn.close();
	}
	/**
	 * Creates Manifest with given attributes
	 * @param manifest Manifest File
	 */
	public static void appendAttributes(Manifest manifest) {
		manifest.getMainAttributes().put(Attributes.Name.MANIFEST_VERSION, manifestVersion);
		manifest.getMainAttributes().put(Attributes.Name.MAIN_CLASS, mainFile.replace(".class", ""));
	}

}

				AutoJar.outputName = fileName.getText()+".jar";
