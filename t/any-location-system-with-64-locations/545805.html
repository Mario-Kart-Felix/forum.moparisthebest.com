Also included commands (for PI, but can be converted very easily):

Player list (shows all the players online with their location) (probably wasn’t done the most efficient way):

			if (playerCommand.equalsIgnoreCase("playerlist") || (playerCommand.equalsIgnoreCase("plist")) || (playerCommand.equalsIgnoreCase("playerson"))) {
				c.clearPlayersInterface();
				c.sendMessage("There are currently [@blu@"+PlayerHandler.getPlayerCount()+ "@bla@] players online.");
				c.getPA().sendFrame126("Server name - Online Players", 8144);
				c.getPA().sendFrame126("@dbl@Amount of players online: [" + PlayerHandler.getPlayerCount()+ "]", 8145);

				int line = 8147;
				for (int i = 1; i < Config.MAX_PLAYERS; i++) {
					Client p = c.getClient(i);
					if (!c.validClient(i))
						continue;
					if (p.playerName != null) {
						String title = "";
						if (p.playerRights == 1) {
							title = "@red@Mod@dbl@, ";
						} else if (p.playerRights == 2) {
							title = "@red@Admin@dbl@, ";
						} else if (p.playerRights == 3) {
							title = "@red@Owner@dbl@, ";
						} else if (p.playerRights == 4) {
							title = "@gre@Donator@dbl@, ";
						}
						title += "level-" + p.combatLevel;
						String extra = "";
						if (c.playerRights > 0) {
							extra = "(" + p.playerId + ") ";
						}
						c.getPA().sendFrame126("@dre@" + extra + p.playerName + "@dbl@ ("+ title + ") is " + p.getRegion().getLocation() + ".", line);
						line++;
					}
				}
				c.getPA().showInterface(8134);
				c.flushOutStream();
			}
