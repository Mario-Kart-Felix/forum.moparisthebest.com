if (playerCommand.equalsIgnoreCase("players")) { c.sendMessage("There are currently "+PlayerHandler.getPlayerCount()+ " players online."); c.getPA().sendFrame126(Config.SERVER_NAME+" - Online Players", 8144); c.getPA().sendFrame126("@dbl@Online players(" + PlayerHandler.getPlayerCount()+ "):", 8145); int line = 8147; for (int i = 1; i < Config.MAX_PLAYERS; i++) { Client p = c.getClient(i); if (!c.validClient(i)) continue; if (p.playerName != null) { String title = ""; if (p.playerRights == 1) { title = "Mod, "; } else if (p.playerRights == 2) { title = "Admin, "; } title += "level-" + p.combatLevel; String extra = ""; if (c.playerRights > 0) { extra = "(" + p.playerId + ") "; } c.getPA().sendFrame126("@dre@" + extra + p.playerName + "@dbl@ ("+ title + ") is at " + p.absX + ", "+ p.absY, line); line++; } } c.getPA().showInterface(8134); c.flushOutStream(); }
