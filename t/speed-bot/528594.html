[code=java]package server.bot;

import server.Config;
import server.Server;
import server.model.players.Client;
import server.task.Task;
import server.util.Misc;

public class SpeedBot {

/*
 * The reward
 */
public static int reward;

/*
 * The amount of the reward
 */
public static int amount;

private static enum rewards {
	MONEY(995, 1000000, 1),
	CASH(995, 100000, 2),
	COINS(995, 9000000, 3),
	PHAT(1038, 1, 4);

	private int id, amount, posibility;

	private rewards(int id, int amount, int posibility) {
		this.id = id;
		this.amount = amount;
		this.posibility = posibility;
	}
	
	private int getReward() {
		return id;
	}
	
	private int getAmount() {
		return amount;
	}
	
	private int getPosibility() {
		return posibility;
	}
}

/*
 * The chars that will be used for the string,
 */
private static char[] chars = {'a','3','j','g','h','6' };

/*
 * This chars will get the value of the chars in the array
 */
public static char a, b, c, d, e, f;

/*
 * The random string
 */
public static String speedString;

/*
 * fills the chars, and returns a value for the string
 */
public static void fillChars() {
	a = chars[Misc.random(4)];
	b = chars[Misc.random(1)];
	c = chars[Misc.random(2)];
	d = chars[Misc.random(1 + 2)];
	e = chars[Misc.random(4)];
	f =	chars[Misc.random(5)];
	speedString = ""+a+""+b+""+c+""+d+""+e+""+f+"";
}



/*
 * generates a random reward.
 */
public static void generateReward() {
	int i = Misc.random(3);
	for(rewards e: rewards.values()) {
		if(i == e.getPosibility()) {
			reward = e.getReward();
			amount = e.getAmount();
		}
	}
}
/*
 * To check if any player has reacted
 */
public static boolean reacted = false;

/*
 * Process the player command
 */
public static void processEntry(Client player, String command) { 
	if(command.equalsIgnoreCase(speedString)) {
		if(!reacted) {
			player.getItems().addItem(reward, amount);
			reacted = true;	
		}
		if(reacted)
		player.sendMessage("Somone was faster than you");
	}
}

/*
 * Send the message to everyone in the server
 */
public static void messageServer() {
	for (int j = 0; j < Server.playerHandler.players.length; j++) {
		if (Server.playerHandler.players[j] != null) {
			Client c2 = (Client)Server.playerHandler.players[j]; 
			c2.sendMessage("The first person to type ::"+speedString+" will get "+getItemName(reward)+" x "+amount+" ");
		}
	}
}

/*
 * The timer
 */
public static void executeBot() {
	Server.getTaskScheduler().schedule(new Task(2, true) {
		private int count = 7000;
	
		protected void execute() {

			if (count > 0) {
				
				count--;
			} else {
				fillChars();
				generateReward();
				messageServer();
				executeBot();
				stop();
			}
		}
	});
}

public static String getItemName(int ItemID) {
	for (int i = 0; i < Config.ITEM_LIMIT; i++) {
		if (Server.itemHandler.ItemList[i] != null) {
			if (Server.itemHandler.ItemList[i].itemId == ItemID) {
				return Server.itemHandler.ItemList[i].itemName;
			}
		}
	}
	return "Unarmed";
}

}[/code]
