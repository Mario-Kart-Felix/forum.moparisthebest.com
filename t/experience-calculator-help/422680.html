Revised with a little help from this tutorial:

http://www.dreamincode.net/forums/topic/17705-basic-gui-concepts/

[code]import java.awt.;
import javax.swing.;
import java.awt.event.*;

public class Experience extends JFrame {
private static final int WIDTH = 400;
private static final int HEIGHT = 300;

private JLabel monsterL, experienceL, totalL;
private JTextField monsterTF, experienceTF, totalTF;
private JButton calculateB, exitB;

//Button Handlers
private CalculateButtonHandler cbHandler;
private ExitButtonHandler ebHandler;

public Experience() {

	monsterL = new JLabel("Exp Per Monster: ", SwingConstants.RIGHT);
	experienceL = new JLabel("Total Monsters: ", SwingConstants.RIGHT);
	totalL = new JLabel("Total Experience: ", SwingConstants.RIGHT);

	monsterTF = new JTextField(10);
	experienceTF = new JTextField(10);
	totalTF = new JTextField(10);

	//Specify handlers for each button and add (register) Action Listeners to each button
	calculateB = new JButton("Calculate");
	cbHandler = new CalculateButtonHandler();
	calculateB.addActionListener(cbHandler);
	exitB = new JButton("Exit");
	ebHandler = new ExitButtonHandler();
	exitB.addActionListener(ebHandler);

	setTitle("Zach's Monster Calculator");
	Container pane = getContentPane();
	pane.setLayout(new GridLayout(4, 2));

	//Add things to the pane in the order you want them to appear (left to right, top to bottom)
	pane.add(monsterL);
	pane.add(monsterTF);
	pane.add(experienceL);
	pane.add(experienceTF);
	pane.add(totalL);
	pane.add(totalTF);
	pane.add(calculateB);
	pane.add(exitB);

	setSize(WIDTH, HEIGHT);
	setVisible(true);
	setDefaultCloseOperation(EXIT_ON_CLOSE);
}

private class CalculateButtonHandler implements ActionListener {
	public void actionPerformed(ActionEvent e) {
		double monster, experience, total;

		monster = Double.parseDouble(monsterTF.getText()); //We use the getText and setText methods to manipulate the data entered into those fields.
		experience = Double.parseDouble(experienceTF.getText());
		total = monster * experience;

		totalTF.setText("" + total);
	}
}

public class ExitButtonHandler implements ActionListener {
	public void actionPerformed(ActionEvent e) {
		System.exit(0);
	}
}

public static void main(String[] args) {
	Experience experObj = new Experience();
}

}[/code]
