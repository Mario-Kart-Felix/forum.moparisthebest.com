package mti;

/**
* @author Danny Barrett
*/

public class Main {

	String[] naturals = {
		"C", "D", "E", "F", "G", "A", "B"
	};
	int[] major = {
		1, 2, 2, 1, 2, 2, 2
	};

	public static void main(String args[]) {
		Main main = new Main();
		main.printMajor("A");
		System.out.println();
	}
	
	void printMajor(String root) {
		String[] scale = new String[7];
		int rootNumber = 0;
		int offset = 0;
		for(int i = 0; i < 7; i++) {
			if(naturals[i].equalsIgnoreCase(root)) {
				rootNumber = i;
				break;
			}
		}
		for(int j = 0; j < 7; j++) {
			if((rootNumber + j) > 6) {
				offset = (rootNumber + j - 7);
				if(major[j] == 2 && major[offset] == 1)
					scale[j] = naturals[offset] + "#";
				else if(major[j] == 1 && major[offset] == 2)
					scale[j] = naturals[offset] + "b";
				else
					scale[j] = naturals[offset];
			} else {
				offset = (rootNumber + j);
				if(major[j] == 2 && major[offset] == 1)
					scale[j] = naturals[offset] + "#";
				/*else if(major[j] == 1 && major[offset] == 2)
					scale[j] = naturals[offset] + "b";*/
				else
					scale[j] = naturals[offset];
			}
		}
		for(int k = 0; k < 7; k++) {
			System.out.print(scale[k] + " ");
		}
	}

}

Also for anyone whos wondering for clarification, the desired output(in C for simplicity) would be:

C Major: C D E F G A B
D Dorian: D E F G A B C
E Phrygian: E F G A B C D
F Lydian: F G A B C D E
G Mixolydian: G A B C D E F
A Minor: A B C D E F G
B Locrian(or Diminished): B C D E F G A

Dm (D F A)
Em (E G B)
