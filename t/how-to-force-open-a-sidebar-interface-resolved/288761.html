[code] public void setSidebar(Player p, int barId)
{
/*
* forces inventory to certain sidebar
*/
p.outStream.createFrame(106);
p.outStream.writeByteC(barId);
}

public void setSidebar(Player p, int barId, int interfaceId)
{
	/*
	 * sets an interface to a sidebar
	 */
	p.outStream.createFrame(71);
	p.outStream.writeWord(interfaceId);
	p.outStream.writeByteA(barId);
}[/code]
