A library to access the Grand Exchange from within Java. This could be used to display amount of money made, or could be used to find greatest price differences for grand exchange rwt’ing. For the entire IDEA Project file use this link: http://uppit.com/H3J1M2

The source java file is below for anyone who would wants to review

[code=java]/* Package: gedb
Description: Runescape Grand Exchange Library

solarwind, code ‘pretifi-cation’ and regex assistance

package gedb;

import java.io.;
import java.net.;
import java.util.regex.*;

public class GEDB_Interface {
public String OBJECTID;
/* Example usage:
public static void main(String args[]) throws Exception {
String objectid = GEDB_Interface.LookupObjId(“blue partyhat”);
GEDB_Interface gi = new GEDB_Interface(objectid);
gi.getMarketPrice();
gi.getMinPrice();
gi.getMaxPrice();
gi.getPriceMargin();
}
*/

/* Constructor
 * Parameters: String ObjectID: the object ID of the item to get the price for.
 */
public GEDB_Interface(String ObjectID) {
	this.OBJECTID = ObjectID;
}

/* Method Name: getMarketPrice
 * Returns: double with Market Price for item
 * Caveats: Uses regex defined in function to determine Market Price.
 */
public double getMarketPrice() {
	Pattern ptnWebTblTitle = Pattern.compile("(?i)Market price:");
	double dblMktPrice;
	try {
		dblMktPrice = LookupPrice(OBJECTID, ptnWebTblTitle);
		return dblMktPrice;
	} catch(Exception ex) {
		log("RuntimeException: " + ex);
		return -1;
	}
}

/* Method Name: getMinPrice
 * Returns:	double with Min Price for item
 * Caveats: Uses regex defined in function to determine Min Price.
 */
public double getMinPrice() {

	Pattern ptnWebTblTitle = Pattern.compile("(?i)Minimum price:");
	double dblMktPrice;
	try {
		dblMktPrice = LookupPrice(OBJECTID, ptnWebTblTitle);
		return dblMktPrice;
	} catch(Exception ex) {
		log("RuntimeException: " + ex);
		return -1;
	}
}

/* Method Name: getMaxPrice
 * Returns: double with Max Price for item
 * Caveats: Uses regex defined in function to determine Max Price.
 */
public double getMaxPrice() {

	final Pattern ptnWebTblTitle = Pattern.compile("(?i)Maximum price:");
	try {
		return LookupPrice(OBJECTID, ptnWebTblTitle);
	} catch(Exception ex) {
		log("RuntimeException: " + ex);
		return -1;
	}
}

/* Method Name: getPriceMargin	 
 * Returns: double with difference between Max price and Min price.
 * Caveats: None.
 */
public double getPriceMargin() {
	double dblMaxPrice = getMaxPrice();
	double dblMinPrice = getMinPrice();
	return dblMaxPrice - dblMinPrice;
}

/* Method Name: RSifyValue
 * Parameters: dblValue - double to convert to RS-like value
 * Returns: string with print-ready value ie: 12M gp, 100K gp, 999gp
 * Caveats: None
 */
public String RSifyValue(double dblValue) {
	if(dblValue >= 1000000) { //handle millions
		return Double.toString(dblValue / 1000000) + "M gp";
	} else if(dblValue >= 1000) { //handle thousands
		return Double.toString(dblValue / 1000) + "K gp";
	} else {
		return Integer.toString((int) dblValue) + " gp"; //get rid of decimal crap
	}
}

/* Method Name: LookupObjId
 * Parameters: strObjectName - String with exact name of object to lookup ID for.
 * Returns:	string with object id.
 * Caveats:	Must be an exact match (not case sensitive).
 * Description:	Uses three regexes to locate object ID.  First looks for
 *              links matching current pattern with strLinkRegex. Then
 *              uses strObjRegex to find the exact item we want (sometimes
 *   			the search returns multiple results). Then uses pNumbersOnly
 * 	    		to find the actual object id inside the link.
 */
public static String LookupObjId(String strObjectName) throws Exception {
	try {
        // matches any positive int (no decimals)
        Pattern pNumbersOnly = Pattern.compile("[0-9][0-9]*");
        // matches link to items
        String strLinkRegex = "<a href=\"\\./v[^>]*>[^<]*</a>";
		String strObjRegex = "[0-9]*(?=\" />\\s(?i)" + strObjectName + ")";
		Pattern pSearchString = Pattern.compile(strLinkRegex);
		Pattern pViewItem = Pattern.compile(strObjRegex);
		String strUrlObject = strObjectName.replace(" ", "+"); //replace spaces with %22
		strUrlObject = "%22" + strUrlObject + "%22"; //put quotes on string for exact match
		URL u = new URL("http://itemdb-rs.runescape.com/results.ws?query=" + strUrlObject + "&price=all&members=");
		URLConnection uc = u.openConnection();
		BufferedReader brInputReader = new BufferedReader(new InputStreamReader(uc.getInputStream()));
		String strInputLine;
		Matcher mObjLine;
		Matcher mSearchString;
		String strFiltered = "";
		String strObjSubset = "";
		boolean bSearchStringFound;
		boolean bObjLineFound = false;
		while((strInputLine = brInputReader.readLine()) != null && !bObjLineFound) {
			// do nothing
			mSearchString = pSearchString.matcher(strInputLine);
			bSearchStringFound = mSearchString.find();
			if(bSearchStringFound) {
				strFiltered = mSearchString.group();
			}
			if(bSearchStringFound) {
				mObjLine = pViewItem.matcher(strFiltered);
				bObjLineFound = mObjLine.find();
				if(bObjLineFound) {
					strObjSubset = mObjLine.group();
				}
			}
		}
		brInputReader.close();
		if(strInputLine == null) {
			throw new RuntimeException("No Match Found");
		}
		try {
			Matcher mNumbersOnly = pNumbersOnly.matcher(strObjSubset);
			if(mNumbersOnly.find()) {
				return mNumbersOnly.group();					 
			} else {
				log(pNumbersOnly.matcher(strInputLine).find());
				log(strInputLine);
				throw new RuntimeException("No Match Found");
			}
		} catch(NumberFormatException nfe) {
			log("NumberFormatException: " + nfe);
			return "";
		}
	} catch(MalformedURLException me) {
		log("MalformedURLException: " + me);
		return "";
	} catch(IOException ioe) {
		log("IOException: " + ioe);
		return "";
	}
}
/* Method Name: LookupPrice
 * Parameters: strObjectName - String with exact name of object to lookup ID for.
 *             ptnWebTblTitle - Regex Pattern for finding Column title
 *                              in price list.
 * Returns:	double with object price matching regex.
 * Caveats:	none
 * Description:	Uses three regexes to locate object ID.  First looks for
 *              links matching current pattern with strLinkRegex. Then
 *              uses strObjRegex to find the exact item we want (sometimes
 *   			the search returns multiple results). Then uses pNumbersOnly
 * 	    		to find the actual object id inside the link.
 */
private double LookupPrice(String strObjId, Pattern ptnWebTblTitle) throws Exception {
	try {
		Pattern pNumbersOnly = Pattern.compile("[0-9,.]*?[mMkK]?$");
		URL url = new URL("http://itemdb-rs.runescape.com/viewitem.ws?obj=" + strObjId);
		URLConnection uc = url.openConnection();
		BufferedReader brInputReader = new BufferedReader(new InputStreamReader(uc.getInputStream()));
		String strstrInputLine;
		while((strstrInputLine = brInputReader.readLine()) != null && !ptnWebTblTitle.matcher(strstrInputLine).find()) {
		}
		    brInputReader.close();
		    if(strstrInputLine == null) {
			    throw new RuntimeException("No Match Found");
		    } else {
		    }
		    try {
			    Matcher mNumbersOnly = pNumbersOnly.matcher(strstrInputLine);
			    if(mNumbersOnly.find()) {
				    String strRtnPrice = mNumbersOnly.group().replace(",", "");
				    double rtnPrice;
				    if(strRtnPrice.endsWith("m")) {
					    strRtnPrice = strRtnPrice.replace("m", "");
					    rtnPrice = Double.parseDouble(strRtnPrice) * 1000000;
				    } else if(strRtnPrice.endsWith("k")) {
					    strRtnPrice = strRtnPrice.replace("k", "");
					    rtnPrice = Double.parseDouble(strRtnPrice) * 100000;
				    } else {
					    rtnPrice = Double.parseDouble(strRtnPrice);
				    }
				return rtnPrice;
	        } else {
				log(pNumbersOnly.matcher(strstrInputLine).find());
				throw new RuntimeException("No Match Found");
			}
		} catch(NumberFormatException nfe) {
			log("NumberFormatException: " + nfe);
			return -1;
		}
	} catch(MalformedURLException me) {
		log("MalformedURLException: " + me);
		return -1;
	} catch(IOException ioe) {
		log("IOException: " + ioe);
		return -1;
	}
}

/* Internal Functions: */
private static void log(Object o) {
	System.out.println(o.toString());
}
