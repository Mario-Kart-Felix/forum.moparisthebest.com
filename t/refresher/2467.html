# ScriptName  = Refresher
# Author      = Ron
# Description = Refreshes lol
# Version     = 1.1
# Date        = May 6, 2006
# Comments    = I had this on Kaitnieks.com
# How To      = Edit line 34 and match the coordinates to where your refresh button is.
# Notes       = Press F10 to stop!
/Script Info}

Program Refresher;

Const
     WaitT = 1000; // How long do you want to wait inbetween each click?

Var
   Refresh : Integer;

Procedure ProgressReport;
Begin
     ClearDebug;
     WriteLn('/===== AutoRefresher by Ron =====\');
     WriteLn('');
     WriteLn('  Refreshed the page '+IntToStr(Refresh)+' time(s).');
     WriteLn('');
     WriteLn('\================================/');
End;

begin
     ActivateClient;
     
     // Clicks
     ClickMouse(234, 123, true);
     
     // Repeats this section...
     Repeat
           // Presses F6
           KeyDown(117);
           
           // Releases F6
           KeyUp(117);

           // Adds one to the variable Click
           Refresh := Refresh + 1;
           
           // Waits a little randomly incase the website can tell if you're
           // using a macro.
           Wait(WaitT+Random(500));
           
           // Prints progress report.
           ProgressReport;
           
     // Until F10 is pressed.
     Until(IsFKeyDown(10));
End.
