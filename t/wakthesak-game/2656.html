Program WAKtheSAK;
const
a=0;

Procedure clickstart;
Begin
movemouse(609,364); //if you want to play on a diffrent site the window will be a diffrent size so you will need to change these coordinates
ClickMouse(609,364,true); //and here
end;

procedure Wak;
Begin;
MoveMouseSmooth(632,237); //and here
HoldMouse(632,237,true); //and here
MoveMouseSmooth(57,183); //and here
ReleaseMouse(57,183,true); //you can guess what im gonna say
wait(6000);
end;

procedure playagain;
Begin;
movemouse(358,298) //and here
clickmouse(358,298,true) //and here
end;

begin
clickstart;
wak;
repeat
until a<4;
playagain;
wak;
repeat
until a<4;
end.[/code]
