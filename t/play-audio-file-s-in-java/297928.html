Here is how you would play an audio file using the sun.audio package,this was created using Eclipse and Java 1.6, so it’s pretty old.

[code]import java.awt.Container;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.FileInputStream;
import java.io.IOException;

import javax.swing.JButton;
import javax.swing.JFrame;

import sun.audio.AudioPlayer;
import sun.audio.AudioStream;

public class PlayAudio extends JFrame implements ActionListener{
JButton button;
boolean stop = true;

public PlayAudio(){
	Container c = getContentPane();
	button = new JButton("Play");
	button.addActionListener(this);
	c.add(button);

	setTitle("Play audio");
	setSize(200,100);
	setVisible(true);
}

public void actionPerformed(ActionEvent e){
	try{			
		AudioPlayer p=AudioPlayer.player;
		AudioStream as =	
			new AudioStream(new FileInputStream("your file here")); 	
		p.start(as);		
	}
	catch(IOException IOE){}		
}

public static void main(String[] args){
	PlayAudio PA = new PlayAudio();
}

}

[code]package com.light.test;

import javax.sound.sampled.AudioInputStream;
import javax.sound.sampled.AudioSystem;
import javax.sound.sampled.DataLine;
import javax.sound.sampled.LineUnavailableException;
import javax.sound.sampled.SourceDataLine;
import javax.sound.sampled.UnsupportedAudioFileException;
import javax.swing.JButton;
import javax.swing.JFrame;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.File;
import java.io.IOException;

/**

  • Date: Aug 26, 2009

  • @author Light232(light232@hotmail.com)
    */
    public final class GUIAudioPlayer {

    private final JFrame playerFrame;
    private final String audioFile;

    public GUIAudioPlayer(String audioFile) {
    this.playerFrame = new JFrame("Playing: " + audioFile);
    this.audioFile = audioFile;
    }

    private void init() throws UnsupportedAudioFileException, IOException, LineUnavailableException {
    this.playerFrame.setSize(200, 55);
    this.playerFrame.setVisible(true);
    this.playerFrame.setLayout(null);
    ActionListener buttonListener = new Player();
    JButton button = new JButton(“Play”);
    button.addActionListener(buttonListener);
    button.setSize(60, 20);
    button.setLocation(0, 0);
    this.playerFrame.add(button);
    button = new JButton(“Stop”);
    button.addActionListener(buttonListener);
    button.setLocation(124, 0);
    button.setSize(60, 20);
    this.playerFrame.add(button);
    }

    private final class Player implements ActionListener {

     private final AudioInputStream audioInput;
 private final SourceDataLine sourceDataLine;

 private Player() throws UnsupportedAudioFileException, IOException, LineUnavailableException {
     this.audioInput = AudioSystem.getAudioInputStream(new File(GUIAudioPlayer.this.audioFile));
     this.sourceDataLine = (SourceDataLine) AudioSystem.getLine(new DataLine.Info(SourceDataLine.class
             , this.audioInput.getFormat()));
 }

 public void actionPerformed(ActionEvent e) {
     if (e.getActionCommand().equals("Play")) {
         this.play();
         //maybe disable the play button & enable stop button?
     } else {//has to be stop :P
         this.stop();
     }
 }

 private void play() {
     Thread playerThread = new Thread(new Runnable() {
         public void run() {
             try {
                 Player.this.sourceDataLine.open();
                 Player.this.sourceDataLine.start();
                 final byte[] audioBuffer = new byte[Player.this.audioInput.available()];
                 Player.this.audioInput.read(audioBuffer, 0, audioBuffer.length);
                 Player.this.sourceDataLine.write(audioBuffer, 0, audioBuffer.length);
                 Player.this.sourceDataLine.drain();
                 Player.this.sourceDataLine.close();
             } catch (Exception ioe) {
                 Player.this.stop();
             }
         }
     }, "AudioPlayer");
     playerThread.setDaemon(true);
     playerThread.start();
 }

 private void stop() {
     //handling
 }

    }

    public static void main(String[] args) {
    GUIAudioPlayer audioPlayer = new GUIAudioPlayer(args[0]);
    try {
    audioPlayer.init();
    } catch (UnsupportedAudioFileException e) {
    System.err.println(“Unsupported audio file [” + args[0] + “].”);
    } catch (IOException e) {
    e.printStackTrace();
    } catch (LineUnavailableException e) {
    System.err.println(“Unavailable line!”);
    }
    }
    }[/code]
