private void createFile(String usr){
      //This will create a .txt file for every username.
      private Formatter f;
      private File path;
      //This is the path that you want to save the names of the users at.
      path = new File("C:\\EnterPathYouWantToSaveToHere\\" +usr);
      //This checks to see if the file exists already (this method is built in)
      if(path.exists())
            //This says if the file already exists execute this piece of code, and don't create the file.
            System.out.printf("The username %s already exists, please try again.", usr);
      else
            //This says if the file doesn't already exist then execute this piece of code.
            System.out.printf("Thank you for registering %s", usr);
            //This creates the new file.
            f.format = new Formatter(path);
}
