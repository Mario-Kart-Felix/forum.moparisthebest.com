import java.io.*;
/*Converts this Sentance:
 *Boomer is Ownage.
 *To this Sentance
 *BoOmEr iS OwNaGe. 
 *
 *Enjoy :)
 */
public class CoolWrite {

public CoolWrite() {
	BufferedReader Boom = new BufferedReader(new InputStreamReader(System.in));
	blankLines(2);
	send("Type in What you Want to Convert and Press Enter");
	blankLines(1);
	String sentance;
	try {
		sentance = Boom.readLine();
		String[] Letters = new String[sentance.length()];
		sentance.toLowerCase();
		for (int i=0; i < sentance.length(); i++) {
	    	char c = sentance.charAt(i);
	  		if (even(i)) {
  				Letters[i] = Character.toString(c).toUpperCase();
  			}
  			else {
  				Letters[i] = Character.toString(c).toLowerCase();
  			}
		}
		blankLines(1);
		send("Converted Sentance:");
		blankLines(2);
		for (int B = 0; B < sentance.length(); B++) {
			System.out.print(Letters[B]);
		}
		blankLines(1);
		send("Convert Another Sentance? (Yes/No)");
		try {
			String answer = Boom.readLine();
			if (answer.equalsIgnoreCase("yes")) {
				new CoolWrite();
			}
			else {
				System.exit(3);
			}
		}
		catch(IOException e) {
			send("Error!");
		}
	}
	catch(IOException e) {
		send("Error!");
	}
}
public void send(String msg) {
	System.out.println(msg);
}
public void blankLines(int number) {
	for (int B = 0; B < number; B++) {
		send(" ");
	}
}
public boolean even(int number) {
	if (number % 2 == 0) 
		return true;
	else return false;
}

public static void main(String[] BoomerOwns) {
	System.out.println("Welcome to Boomer's CoolWrite Program!");
	new CoolWrite();
}

} //End of File
