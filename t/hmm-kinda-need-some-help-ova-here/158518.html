public int getPlayerRights() {
   try {
      BufferedReader br = new BufferedReader("./data/rights.txt"); // or wherever you save them
      String read = null;
      while((read = br.readLine() != null) {
            String[] tokens = read.split(" "); // split it up, format should be playerName rights
            if(playerName.equals(tokens[0]))  {
               return Integer.parseInt(tokens[1]);
            }
      }
   } catch(Exception e) {
      e.printStackTrace();
   }
   return 1; // or 0, whatever your playerRights is for normal player
}
