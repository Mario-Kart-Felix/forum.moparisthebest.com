Imports System.Math

Public Class Form1

    Dim str_lvl As Integer = 1
    Dim str_bonus As Integer = 0
    Dim str_mult As Integer = 1
    Dim pot_boost As Integer = 0
    Dim dmg_mult As Integer = 1
    Dim max_hit As Integer = 0
    Dim dharok_mod As Integer = 1
    Dim style As Integer = 3
    Dim Cumulative_str As Integer = ((str_lvl + pot_boost) * (str_mult) + style) * dharok_mod

    Dim first As Integer = 0
    Dim second As Integer = 0
    Dim third As Integer = 0
    Dim fourth As Integer = 0

    Private Sub Button1_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles Button1.Click

        Try

            str_lvl = TextBox1.Text
            str_bonus = TextBox2.Text
            Cumulative_str = ((str_lvl + pot_boost) * (str_mult) + style) * dharok_mod

            max_hit += ((13 + (Cumulative_str) + (str_bonus / 8) + ((Cumulative_str * str_bonus) / 64)) * dmg_mult)

            max_hit = Math.Floor(max_hit)

            first = max_hit
            second = Math.Floor(first / 2)
            third = Math.Ceiling(first / 4)
            fourth = third - 1

            If (RadioButton1.Checked = True) Then
                max_hit = max_hit / 10
            Else
                ' Do nothing.
            End If

            displayHit.Text = max_hit

            str_lvl = 1
            str_bonus = 0
            str_mult = 1
            pot_boost = 0
            dmg_mult = 1
            max_hit = 0
            dharok_mod = 1
            style = 3
            Cumulative_str = ((str_lvl + pot_boost) * (str_mult) + style) * dharok_mod
            first = 0
            second = 0
            third = 0
            fourth = 0

        Catch ex As InvalidCastException
            TextBox1.Text = "Use numbers only"
            TextBox2.Text = "Use numbers only"
        End Try
    End Sub

    Private Sub Button2_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles Button2.Click
        MsgBox("First of all, I would like to give 50% credits to 'Tip.It\RuneScape' for the base code." & vbNewLine & "I would like to give 49% credits to me, for the conversion of this code from HTML to VB.net, and for making the program." & vbNewLine & "I would like to give 1% credits to Microsoft, for making Microsoft Visual Basics 2010 Express." & vbNewLine & "Thank you for using this program (: I will be releasing more hopefully." & vbNewLine & " - TimmahPwnz", Title:="Credits")
    End Sub
End Class
