{.include Osi.txt}

const
OakLog1=4419213; //try to pick you own color of the logs if it dosen’t drop the logs
OakLog2=6326438; //The same as above
Username=’’;
Password=’’;

procedure CutOak;
begin
if
findcolor(x,y,4486295,50,42,455,281)
or
findcolor(x,y,5075056,50,42,455,281)
then
wait(random(50)+500)
MoveMouseSmooth(x,y)
wait(random(100)+200)
clickmouse(x,y,false)
wait(random(200)+500)
if findcolor(x,y,16776960,40,70,427,275)
then movemousesmooth(x,y)
clickmouse(x,y,true)
wait(random(2000)+25000)
end;

procedure DropOak;
begin
if findcolor(x,y,OakLog1,574,214,730,45
then movemousesmooth(x,y)
wait(random(50)+100)
Dropcolor(OakLog1);
dropcolor(OakLog2);
wait(random(200)+1000)
end;

begin
repeat
CutOak;
DropOak;
until(false)
end.
