public class ItemOnObject {

/*
 * item constants
 */
public static final int EMPTY_VIAL = 229;
public static final int VIAL_OF_WATER = 227;
public static final int EMPTY_BUCKET = 1925;
public static final int BUCKET_OF_WATER = 1929;
public static final int BUCKET_OF_MILK = 1927;

/*
 * Water sources object type constants
 */
public static final int WELL = 884;
public static final int FOUNTAIN = 879;
public static final int DAIRY_COW = 8689;

public static final int FILLABLES[][] = { { EMPTY_VIAL, EMPTY_BUCKET },
		{ VIAL_OF_WATER, BUCKET_OF_WATER }, };

public static void ItemOnObject(Client thePlayer, int itemId, int objectId) {

	/*
	 * fillables
	 */
	for (int i = 0; i < FILLABLES.length; i++) {
		if (objectId == WELL || objectId == FOUNTAIN) {
			if (itemId == FILLABLES[0][i]) {
				thePlayer.deleteItem(FILLABLES[0][i], 1);
				thePlayer.addItem(FILLABLES[1][i], 1);
			}
		}
	}

	/*
	 * cow miliking
	 */
	if (objectId == DAIRY_COW) {
		if (itemId == EMPTY_BUCKET) {
			thePlayer.deleteItem(EMPTY_BUCKET, 1);
			thePlayer.addItem(BUCKET_OF_MILK, 1);
			thePlayer.sM("You milk the cow.");
		}
	}
}
