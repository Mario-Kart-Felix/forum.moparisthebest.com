Console application; requires DirectX 8 or above.

[code]using System;
using System.Collections.Generic;
using System.Text;
using Microsoft.DirectX;
using Microsoft.DirectX.Direct3D;

namespace uberhaxs
{
class ScreenHax
{
static void Main(string[] args)
{
foreach (AdapterInformation adapter in Manager.Adapters)
{
// print out some info about it
Console.Write(adapter.Information.ToString() + "\nDisplay Modes: ");

            // and its display modes
            foreach (DisplayMode mode in adapter.SupportedDisplayModes)
            {
                Console.Write("{0}x{1} : {2} @ {3} Hz, ",
                    mode.Width,
                    mode.Height,
                    mode.Format,
                    mode.RefreshRate);
            }

            // and our current display mode?
            Console.WriteLine("\n Current Display Mode: {0}x{1} : {2} @ {3} Hz",
                adapter.CurrentDisplayMode.Width,
                adapter.CurrentDisplayMode.Height,
                adapter.CurrentDisplayMode.Format,
                adapter.CurrentDisplayMode.RefreshRate);
        }

        Console.WriteLine("\n\nPress any key to exit...");
        // wait for user input to close console
        Console.ReadChar();
    }
}

}
[/code]
