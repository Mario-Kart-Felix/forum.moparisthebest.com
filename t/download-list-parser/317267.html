import java.io.BufferedWriter;
import java.io.File;
import java.io.FileWriter;
import java.io.InputStream;
import java.net.HttpURLConnection;
import java.net.URI;
import java.net.URLConnection;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class Downloads {

public void parse(File f, int[] pages) throws Exception {
	BufferedWriter bw = new BufferedWriter(new FileWriter(f));
	for (int page : pages) {
		if (page < 0) {
			bw.close();
			throw new RuntimeException("Invalid page: " + page);
		}
		System.out.println("- Parsing page: " + page);
		HttpURLConnection http = (HttpURLConnection) new URI(URL_BASE + ((page - 1) * 20) + ".html").toURL().openConnection();
		http.setRequestProperty("Cookie", "mpr_agree2=yes");
		http.connect();
		parseTo(bw, readURL(http));
	}
	System.out.println("\nSuccesfully parsed all pages and wrote to: " + f.toString());
	bw.close();
}

public void parseTo(BufferedWriter bw, String buff) throws Exception {	
	Matcher m = Pattern.compile("(<b>)?<span title=\"[^\"]+\"><span id=\"[^\"]+\"><a href=\"([^\"]+)\">([^<]+)</a></span></span>(</b>)?[^<]+<small id=\"[^\"]+\">[^<]{0,}(<a href=\"[^\"]+\">[^<]+</a>[^<]+){0,}</small>[^<]+</td>[^<]+<td[^>]+>[^<]+<a href=\"[^\"]+\" title=\"View the profile of ([^\"]+)\"").matcher(buff);
	while(m.find()) {
		if (m.group(1) != null) {
			continue;
		}
		bw.write("Title: " + m.group(3));
		bw.newLine();
		
		bw.write("By: " + m.group(6));
		bw.newLine();
		
		bw.write("Page: " + m.group(2));
		bw.newLine();
		bw.newLine();
		
		bw.flush();
	}
}

private String readURL(URLConnection urlc) throws Exception {
	InputStream in = urlc.getInputStream();
	StringBuffer sb = new StringBuffer();
	byte[] buff = new byte[256];
	int readc = 0;
	
	while((readc = in.read(buff)) != -1) {
		sb.append(new String(buff, 0, readc));
	}
	
	return sb.toString().replaceAll("&nbsp;", "").replaceAll("&ndash;", "-").replaceAll("&mdash;", "-").replaceAll("&amp;", "&").replaceAll("&ldquo;", "\"").replaceAll("&rdquo;", "").replaceAll("&rsquo;", "'");
}

public static void main(String[] args) {
	if (args.length < 2) {
		System.out.println("Usage: java Downloads <file_out> <page1> <page2> <page3>..");
		System.out.println("For example, to output the parsing of pages 1, 7 and 8 in the downloads section to \"out.txt\", you would run this application as follows: java Downloads out.txt 1 7 8");
		return;
	}
	
	try {
		int[] a = new int[args.length - 1];
		for (int i = 0; i < (args.length - 1); i++) {
			a[i] = Integer.parseInt(args[i + 1]);
		}
		new Downloads().parse(new File(args[0]), a);
	} catch(Exception e) {
		e.printStackTrace();
	}
}

private final String URL_BASE = "http://www.moparisthebest.com/smf/index.php/board,49.";

}[/code]
