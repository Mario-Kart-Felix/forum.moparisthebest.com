rova1 August 1, 2016, 1:39am #5

Or you could add in clickingmost.java where the actual teleport is added, example:

case 4140: //home tele
case 73100:
case 50235:
    if(c.playerHasItem(563, 1) && c.playerHasItem(554, 1) && c.playerHasItem(556, 3)){
            if (c.playerLevel[6] >= 25){
                c.triggerTele(3087, 3491, 0);
                c.deleteItem(563, 1);
                c.deleteItem(554, 1);
                c.deleteItem(556, 3);
            }  else
                c.sM("You need a magic level of 25 or better to cast this spell.");
    } else 
       c.sM("You don't have enough runes to cast this spell.");
break;
