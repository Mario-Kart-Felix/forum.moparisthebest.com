So most people use languages such as Ruby for their plugin systems; however I don’t know any of those, so I decided to write a simple Java plugin system. This system uses interfaces, allowing multiple plugin implements in one class, which is always good for stuff such as minigames and skills. I’ve added one plugin with this ([B]ButtonPlugin.java[/B]) for an example. Feel free to add more.

NOTE: I wrote this on a Hyperion; it can be converted to [B]ANY[/B] base with a few simple path changes.

[SPOILER=PluginManager.java]

package org.hyperion.rs2.util.plugin;

import java.io.File;
import java.util.LinkedList;
import java.util.List;
import java.util.logging.Logger;

import org.hyperion.rs2.model.Player;

/**
 * Manages all of the plug-in's.
 * 
 * TODO Add more plug-in's.
 * 
 * @author Michael P
 *
 */
public class PluginManager {
	
	/**
	 * The logger.
	 */
	private static final Logger logger = Logger.getLogger(PluginManager.class.getName());

	/**
	 * A listing of all the <code>ButtonPlugin</code>s.
	 */
	public static final List<ButtonPlugin> buttonPlugin = new LinkedList<ButtonPlugin>();
	
	/**
	 * Loads all of the plug-in's.
	 * 
	 * @throws ClassNotFoundException
	 *            If the class is not found.
	 * @throws IllegalAccessException 
	 *            If an illegal access happens.
	 * @throws InstantiationException 
	 *            If an instantiation happens.
	 */
	public static void init() throws InstantiationException, IllegalAccessException, ClassNotFoundException {
		
		/*	Log the loading of the plug-in's.	*/
		logger.info("Loading plugins...");
		
		/*	We load all of the files and packages in the rs2 package.	*/
		File[] file = new File("./src/org/hyperion/rs2/").listFiles();
		
		/*	Our list of packages.	*/
		LinkedList<String> packages = new LinkedList<String>();
		
		/*	Our list of files.	*/
		LinkedList<String> files = new LinkedList<String>();
		
		/*
		 * We loop through our loaded files and packages and add them to our listing.
		 */
		for (File f : file) {
			
			/*
			 * If the file/package is null we skip it and continue looping.
			 */
			if (f == null)
				continue;
			
			/*
			 * If the file is a file we add it to our files listing.
			 */
			if (f.isFile())
				
				/*	We add the file to the files listing and replace the needed Strings.	*/
				files.add(f.getPath().replace(f.getName(), "").replace("\\", "/"));
			
			/*
			 * If the file is a package/directory we add it to our packages listing.
			 */
			else if (f.isDirectory())
				
				/*	We add the package/directory to the packages listing and replace any \ with a / for proper file reading.	*/
				packages.add(f.getPath().replace("//", "/"));
			
		}
		
		/*
		 * We loop through all the listed packages.
		 */
		while (!packages.isEmpty()) {
			
			/*	We set our directory.	*/
			String dir = packages.poll();
			
			/*
			 * We loop through all the files in the directory.
			 */
			for (File f : new File(dir).listFiles()) {
				
				/*
				 * If the file/package is null we skip it and continue looping.
				 */
				if (f == null)
					continue;
				
				/*
				 * If the file is a file we add it to our files listing.
				 */
				if (f.isFile())
					
					/*	We add the file to the files listing and remove the file name along with
					 * replacing any \ with a / for proper file reading.	*/
					files.add(f.getPath().replace(f.getName(), "").replace("\\", "/"));
				
				/*
				 * If the file is a package/directory we add it to our packages listing.
				 */
				else if (f.isDirectory())
				
					/*	We add the package/directory to the packages listing and replace 
					 * any \ with a / for proper file reading.	*/
					packages.add(f.getPath().replace("//", "/"));
			}
		}
		
		/*
		 * We loop through all the listed files.
		 */
		while (!files.isEmpty()) {
			
			/*	We set our path.	*/
			String path = files.poll();
			
			/*
			 * We loop through all the files.
			 */
			for (File f : new File(path).listFiles()) {
				
				/*
				 * If the file/package is null we skip it and continue looping.
				 */
				if (f == null)
					continue;
				
				/*
				 * If the file is an actual file, we finally handle the file and check to 
				 * see if it's an implement of the <code>InterfacePlugin</code> class.
				 */
				if (f.isFile()) {

					/*	We set the new file path.	*/
					String newPath = f.getPath().replace(".\\", "").replace("src\\", "").replace("\\", ".").replace(".java", "");
					
					/*	We create the object class along with a new instance.	*/
					Object object = Class.forName(newPath).newInstance();
					
					/*
					 * If the object class is an implement of the <code>InterfacePlugin</code>
					 * we add it to the listing.
					 */
					if (object instanceof ButtonPlugin)
						buttonPlugin.add((ButtonPlugin) object);
					
				}
				
			}
		}
		
		/*	Log the finish of the plug-in loading.	*/
		logger.info("Loaded plugins!");
		
	}
	
	/**
	 * Handles the button plug-in's.
	 * 
	 * @param player
	 *            The player to handle the button plug-in's for.
	 * @param button
	 *           The button to handle.
	 * @return If the plug-in should be handled <code>true</code>,
	 * if the plug-in shouldn't be handled <code>false</code>.
	 */
	public static boolean handleButton(Player player, int button) {
		for (ButtonPlugin plugin : buttonPlugin) {
			if (plugin.handleButton(player, button))
				return true;
		}
		return false;
	}
	
}

package org.hyperion.rs2.util.plugin;

import org.hyperion.rs2.model.Player;

/**
 * Represents a button plug-in.
 * 
 * @author Michael P
 *
 */
public interface ButtonPlugin {

	/**
	 * Handles a button click.
	 * 
	 * @param player
	 *            The player to handle the button for.
	 * @param button
	 *            The button to handle.
	 * @return If the button should be handled <code>true</code>,
	 * if the button shouldn't be handled <code>false</code>.
	 */
	public boolean handleButton(Player player, int button);
	
}
