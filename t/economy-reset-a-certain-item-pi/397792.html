public static void main(String[] args) {
    try {
        final File charDir = new File("path/to/characters");
        final File[] characters = charDir.listFiles();
        FileReader fileReader = null;
        BufferedReader reader = null;
        FileWriter writer = null;
        final ArrayList<String> lines = new ArrayList<String>();
        for (final File f : characters) {
            if (f == null || f.isDirectory()) {
                continue;
            }
            fileReader = new FileReader(f);
            reader = new BufferedReader(fileReader);
            writer = new FileWriter(f);
            String line = reader.readLine();
            while (line != null) {
                if (line.startsWith("beginning of inventory/bank/equipment line") && line.contains("dragon claws id")) {
                    continue;
                }
                lines.add(line);
                line = reader.readLine();
            }
            for (String s : lines) {
                writer.write(s.getBytes());
            }
            lines.clear();
            writer.close();
            reader.close();
        }
    }
    catch (final Exception e) {
        e.printStackTrace();
    }
}
