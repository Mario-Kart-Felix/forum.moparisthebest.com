package mscp.chess;

import java.awt.Color;
import java.awt.Font;
import java.awt.FontMetrics;
import java.awt.Graphics;
import java.awt.Graphics2D;
import java.awt.RenderingHints;
import java.awt.image.BufferedImage;
import java.io.FileOutputStream;
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;

import javax.imageio.ImageIO;

public class BracketGenerator {

	private static final int tierSize = 50;

	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
		System.out.print("How many users are in the tournament? ");
		int numUsers = scanner.nextInt();
		if (numUsers < 2) {
			System.out.println("You need at least 2 or "
					+ "more users for a bracket!");
			return;
		}
		int rounds = getRoundCount(numUsers);
		List<String> names = readUsers(scanner, numUsers);
		// randomly shuffle the user names
		Collections.shuffle(names);
		System.out.print("Destination file (PNG format): ");
		String fname = scanner.nextLine();
		Graph graph = generateGraph(names, rounds);
		dumpGraph(graph, fname);
		System.out.println("Bracket generated!");
	}

	private static List<String> readUsers(Scanner input, int count) {
		System.out.println("Start listing all " + count
				+ " users, one line per name: ");
		List<String> users = new ArrayList<String>();
		input.nextLine(); // skip println
		for (int i = 0; i < count; i++)
			users.add(input.nextLine());
		return users;
	}

	private static void dumpGraph(Graph graph, String filename) {
		try {
			ImageIO.write(graph.image, "png", new FileOutputStream(filename));
		} catch (Throwable t) {
			t.printStackTrace();
		}
	}

	private static Graph generateGraph(List<String> users, int tiers) {
		List<Node> nodes = new ArrayList<Node>();
		Node root = createTree(nodes, tiers);
		// fill the tree linearly with users
		for (int i = 0; i < users.size(); i++)
			nodes.get(i).value = users.get(i);
		// correct the tree's deformities where a bye would exist
		verifyNode(root, tiers);
		// draw the tree
		Graph graph = new Graph(tiers, users);
		drawRootNode(graph, root, tiers);
		return graph;
	}

	private static Node createTree(List<Node> nodes, int tiers) {
		// every x is a single tier in size; the y is based on how many rounds
		// are left to generate
		if (tiers <= 0)
			return null;
		Node root = new Node();
		// construct the tree
		visitNode(root, nodes, tiers);
		sort(nodes);
		return root;
	}

	private static void sort(List<Node> nodes) {
		// by default, the users will be sorted from top to bottom from the
		// right to the left; we want it to be top to bottom alternating left to
		// right
		int[] indexes = new int[nodes.size()];
		int half = indexes.length / 2;
		for (int i = 0, idx = 0; i < half; i++) {
			// switch from right->left to left->right:
			indexes[idx++] = half + i;
			indexes[idx++] = i;
		}
		// if odd number of elements, the last index is the last index in value
		if ((indexes.length % 2) != 0)
			indexes[indexes.length - 1] = indexes.length - 1;
		List<Node> clone = new ArrayList<Node>();
		for (int i = 0; i < indexes.length; i++)
			clone.add(nodes.get(indexes[i]));
		Collections.copy(nodes, clone);
		clone.clear();
	}

	private static void visitNode(Node parent, List<Node> nodes, int tiers) {
		Node upper = new Node(), lower = new Node();
		(parent.upper = upper).parent = (parent.lower = lower).parent = parent;
		if (--tiers > 0) {
			// recurse to the next tier
			visitNode(upper, nodes, tiers);
			visitNode(lower, nodes, tiers);
		} else {
			nodes.add(upper);
			nodes.add(lower);
		}
	}

	private static void verifyNode(Node node, int tiers) {
		// remove nodes with no values
		verify0(node);
		// now, for each tier, remove nodes where both upper and lower values
		// don't exist
		for (int i = 0; i < tiers; i++)
			verify1(node);
	}

	// for a node with no upper and lower nodes (a user node), if it has no
	// value, it can be removed
	private static void verify0(Node node) {
		if (node.upper != null || node.lower != null) {
			verify0(node.upper);
			verify0(node.lower);
		} else if (node.value == null) {
			if (node.parent.upper == node)
				node.parent.upper = null;
			else
				node.parent.lower = null;
		}
	}

	private static void verify1(Node node) {
		if (node.upper != null || node.lower != null) {
			if (node.upper != null)
				verify1(node.upper);
			if (node.lower != null)
				verify1(node.lower);
		} else if (node.value == null) { // has to be a non-user node
			// both are null...remove the node from its parent
			if (node.parent.upper == node)
				node.parent.upper = null;
			else
				node.parent.lower = null;
		}
	}

	private static void drawRootNode(Graph graph, Node root, int tiers) {
		// center graph
		root.position = graph.getCenter();
		// draw right side of the graph
		graph.push(root.position);
		root.upper.position = root.position;
		drawNode(graph, root.upper, 1, tiers);
		// draw left side of the graph (flip x)
		root.lower.position = root.position;
		graph.jump(root.position);
		drawNode(graph, root.lower, -1, tiers);
		// draw the winner line
		graph.jump(root.position);
		graph.push(new Point(root.position.x, root.position.y
				- seriesMultiply(tierSize, tiers - 1)));
	}

	private static void drawNode(Graph graph, Node parent, int scaleX, int tiers) {
		if (--tiers > 0) {
			float x = parent.position.x, y = parent.position.y, height = seriesMultiply(
					tierSize, tiers - 1);
			x += seriesMultiply(tierSize, tiers - 1) * scaleX;
			if (parent.upper != null && parent.lower != null) {
				// draw horizontal
				Point fork;
				graph.push(fork = new Point(x, y));
				parent.upper.position = new Point(x, y - height);
				parent.lower.position = new Point(x, y + height);
				// draw vertical for upper
				graph.push(parent.upper.position);
				drawNode(graph, parent.upper, scaleX, tiers);
				// draw vertical for lower
				graph.jump(fork);
				graph.push(parent.lower.position);
				drawNode(graph, parent.lower, scaleX, tiers);
			} else {
				// bye condition
				Point fork;
				graph.push(fork = new Point(x, y));
				Node remainder = parent.upper == null ? parent.lower
						: parent.upper;
				remainder.position = fork;
				drawNode(graph, remainder, scaleX, tiers);
			}
		} else {
			graph.push(new Point(parent.position.x + tierSize * scaleX,
					parent.position.y));
			graph.push(parent.value, scaleX == -1);
		}
	}

	private static final int getRoundCount(int users) {
		// since tournaments are binary trees, every next power of two (or
		// increased bit length) is another round in the tournament; non-power
		// of two numbers round up to the next power of two, because someone has
		// to wait at least one week (in a 4 player tournament, there are 2
		// rounds; but, in a 5 player tournament, there must be 3 rounds)
		return bitcount(nexthigher(users));
	}

	// http://en.wikipedia.org/wiki/Power_of_two
	private static final int nexthigher(int n) {
		n = n - 1;
		n |= (n >> 1);
		n |= (n >> 2);
		n |= (n >> 4);
		n |= (n >> 8);
		n |= (n >> 16);
		return n + 1;
	}

	private static final int bitcount(int n) {
		int count = 0;
		while (n != 0) {
			count++;
			n >>= 1;
		}
		return count - 1;
	}

	private static final float seriesMultiply(float factor, int count) {
		for (int i = 0; i < count; i++)
			factor *= 2;
		return factor;
	}

	private static final float seriesSum(float factor, int count) {
		float sum = 0;
		while (count-- > 0)
			sum += seriesMultiply(factor, count);
		return sum;
	}

	private static final class Point {

		private float x, y;

		private Point() {
		}

		private Point(float x, float y) {
			this.x = x;
			this.y = y;
		}
	}

	private static final class Node {

		private Node parent, upper, lower;
		private Point position;
		private String value;
	}

	private static final class Graph {

		private int width, height;
		private Point last;
		private BufferedImage image;
		private Font userNameFont;
		private Graphics graphics;

		private Graph(int treeTiers, List<String> users) {
			// one x horizontal for every round = half of the width
			this.width = (int) (seriesSum(tierSize, treeTiers - 1) + tierSize) * 2;
			// x ys based on tiers - 1, decrementing each time = half the height
			this.height = (int) (seriesSum(tierSize, treeTiers - 1) + tierSize) * 2;
			// now increase the height for "good measure"
			this.width = (width * 3) / 2;
			this.height = (height * 3) / 2;
			// height must be something
			this.height = Math.max(50, height);
			this.userNameFont = new Font("Times New Roman", Font.PLAIN, 12);
			init(users);
		}

		private void init(List<String> users) {
			this.image = new BufferedImage(width, height,
					BufferedImage.TYPE_INT_ARGB);
			this.graphics = image.getGraphics();
			graphics.setFont(userNameFont);
			FontMetrics fm = graphics.getFontMetrics();
			int maxWidth = 0;
			for (String user : users)
				maxWidth = Math.max(maxWidth, fm.stringWidth(user));
			this.image = new BufferedImage(width += (maxWidth * 2), height
					+ userNameFont.getSize(), BufferedImage.TYPE_INT_ARGB);
			this.graphics = image.createGraphics();
			((Graphics2D) graphics).setRenderingHint(
					RenderingHints.KEY_ANTIALIASING,
					RenderingHints.VALUE_ANTIALIAS_ON);
			graphics.setFont(userNameFont);
			graphics.setColor(Color.white);
			graphics.fillRect(0, 0, width, height);
			graphics.setColor(Color.black);
		}

		private Point getCenter() {
			return new Point(width / 2.f, height / 2.f);
		}

		private void push(Point p) {
			if (last != null)
				graphics.drawLine((int) last.x, (int) last.y, (int) p.x,
						(int) p.y);
			this.last = p;
		}

		private void push(String value, boolean pushLeft) {
			if (last != null) {
				int x = (int) last.x, y = (int) last.y;
				if (pushLeft)
					x -= graphics.getFontMetrics().stringWidth(value) + 10;
				else
					x += 10;
				y += userNameFont.getSize() / 2;
				graphics.drawString(value, x, y);
			}
		}

		private void jump(Point p) {
			this.last = p;
		}
	}
}
