Add this public void (changing whatever you want for your server…if you want to shorten it then remember to change the number on the right):



    public void commandMenu() {

	sendQuest("@blu@          ~ZezimaLand~ Commands", 8144);  //Title
	clearQuestInterface();
        sendQuest("@red@~Information~", 8145);
        sendQuest("@whi@::rules - The rules of ZezimaLand", 8147);
        sendQuest("@whi@::modinfo - Information on becoming a Moderator", 8148);
        sendQuest("@whi@::updates - Recent updates to the server", 8149);
        sendQuest("@whi@(usually not updated)", 8150);
        sendQuest("@red@~Online Players~", 8152);
        sendQuest("@whi@::players - Displays players online & their combat level", 8153);
        sendQuest("@red@~Areas~", 8155);
        sendQuest("@whi@::home - Teleports you to home", 8156);
        sendQuest("@whi@::zezima - Teleports you to Zezima's house", 8157);
        sendQuest("@whi@::zezimashouse - Teleports you to Zezima's other house", 8158);
        sendQuest("@whi@[Visit the emote tabs for more areas]", 8159);
        sendQuest("@red@~Training~", 8161);
        sendQuest("@whi@::train - The main training area", 8162);
        sendQuest("@whi@::slayer - Train slayer here", 8163);
        sendQuest("@whi@::void - Enemies that drop void robes (rare drops)", 8164);
        sendQuest("@whi@::barrows - Enemies that drop barrows", 8165);
        sendQuest("@whi@::jad - Where you fight Jad (drops firecape & hard to defeat)", 8166);
        sendQuest("@whi@::kq - Where you fight Kalphite Queen (random drops)", 8167);
        sendQuest("@whi@::skills - Teleports you to skills", 8168);
        sendQuest("@red@~Pking~", 8170);
        sendQuest("@whi@::funpk - Teleports you to FunPk (Items not lost on death)", 8171);
        sendQuest("@whi@::strpk - Pking depending on your strength level", 8172);
        sendQuest("@whi@::teampk - Teleports you to TeamPk (Main Pking area)", 8173);
        sendQuest("@red@~Useful Needs~", 8175);
        sendQuest("@whi@::bank - Accesses your bank", 8176);
        sendQuest("@whi@::char - Changes your character's clothes & looks", 8177);
        sendQuestSomething(8143);
        showInterface(8134);
        flushOutStream();

    }

Then add this command where you put your other commands (Search “god”):

if(command.equalsIgnoreCase("commands")) { commandMenu(); }
