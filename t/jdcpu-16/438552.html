SUB A, [0x1000]          ; 7803 1000
IFN A, 0x10              ; c00d 
   SET PC, crash         ; 7dc1 001a [*]
                      
; Do a loopy thing
SET I, 10                ; a861
SET A, 0x2000            ; 7c01 2000
:loop

   SET PC, loop          ; 7dc1 000d [*]
