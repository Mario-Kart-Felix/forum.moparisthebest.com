[center][size=20pt]RuneCrafting Tutorial for Delta[/size]

In this tutorial we are going to add full runecrafting, which means with doubling, tripling, … runes at a higher level.[/center]

[size=14pt]1.[/size]
Lets start with the essentials,

First make sure to remove your current runecrafting from your source,
You should find it under your object clicking packet. Where you will see the name of the void
which you have to remove to.

Next,

Open up your server class and search for something like:

public static TextHandler TextHandler = null;

add this:

public static RuneCrafting RuneCrafting = null;

now search for something like:

TextHandler = new TextHandler();

and add this under it:

RuneCrafting = new RuneCrafting();

[size=14pt]2.[/size]

Make a new textfile and name it “RuneCrafting”
Save and close,
now open up again and paste this code in it:
[size=6pt]Code-explanation added in the code itself.[/size]

public class RuneCrafting {
	
	/**
	 * @author RandQm
	 * @based off unknown
	 * 8 February 2012
	 */
	
	private int expRate = 4;
	
	/* The runecrafting method */
	public void runecraft(final int playerId, int reqLvl, int expAdded, int rune, int a2, int a3, int a4, int a5, int a6, int a7, int a8, int a9, int a10, String Type) {
		final client c = (client) PlayerHandler.players[playerId];
		/*
		 *  This is so we can type "rcLvl" instead of having to type
		 * "c.playerLevel[c.playerRunecrafting]" everytime.
		 */
		int rcLvl = c.playerLevel[c.playerRunecrafting];
		/* Checking for rune essence */
		int runeEssence = c.amountOfItem(1436);
		/* If the players runecrafting level is lower than the
		 * required level to craft the runes, it will send the player
		 * a message with the level he needs.
		 */
		if (rcLvl < reqLvl) {
			c.sM("You need a runecrafting level of at least "+reqLvl+" to craft "+Type+" runes.");
			return;
		}
		/*
		 * If the player does NOT (not is the "!") have the item 1436
		 * which is rune essence, then it will send the player a message
		 * to tell him/her.
		 */
		if (!c.playerHasItem(1436)) {
			c.sM("You do not have any rune essence to craft.");
			return;
		}
		/* If the players runecrafting level is equal or higher than the needed
		 * level to craft the runes it will go check if the players level is high
		 * enough to multiply the amount of given runes and how many times.
		 */
		if (rcLvl >= reqLvl) {
			if (rcLvl >= a2 && rcLvl < a3)
				runeEssence = c.amountOfItem(1436) * 2;
			if (rcLvl >= a3 && rcLvl < a4)
				runeEssence = c.amountOfItem(1436) * 3;
			if (rcLvl >= a4 && rcLvl < a5)
				runeEssence = c.amountOfItem(1436) * 4;
			if (rcLvl >= a5 && rcLvl < a6)
				runeEssence = c.amountOfItem(1436) * 5;
			if (rcLvl >= a6 && rcLvl < a7)
				runeEssence = c.amountOfItem(1436) * 6;
			if (rcLvl >= a7 && rcLvl < a8)
				runeEssence = c.amountOfItem(1436) * 7;
			if (rcLvl >= a8 && rcLvl < a9)
				runeEssence = c.amountOfItem(1436) * 8;
			if (rcLvl >= a9 && rcLvl < a10)
				runeEssence = c.amountOfItem(1436) * 9;
			if (rcLvl >= a10)
				runeEssence = c.amountOfItem(1436) * 10;
		}
		/* If above is checked, the amount of rune essence in the players inventory
		 * gets removed (deleteItem) and gets replaced by the new runes made (addItem)
		 * then the player recieves experience ((exp multiplied by amount of rune essence) multiplied by the exprate)
		 * Then a message gets send to the player
		 * An animation and gfx starts.
		 */
		for (int i = 0; i < 29; i++)
			c.deleteItem(1436, c.getItemSlot(1436), i);
			c.addItem(rune, runeEssence);
			c.addSkillXP((expAdded * runeEssence) * expRate, c.playerRunecrafting);
			c.sM("You bind the temple's power into "+c.getItemName(rune)+".");
			c.specGFX(186);
			c.setAnimation(791);
			return;
	}
	
	/* This is mainly added here to not dump everything in the client class,
	 * You could also add every of this method to the object clicking packet if you wanted.
	 * Now this checks the object id so with other words, it checks on what altar the player wants to make runes.
	 */
	public void runeCraft(final int playerId, int object) {
		/* E.G. the object id is 2478 which is an air altar */
		if (object == 2478) {
			/* The runecrafting void of above gets executed with between () the values of our constructor.
			 *  The "air" is a string for the message send to the player */
			runecraft(playerId, 1, 10, 556, 11, 22, 33, 44, 55, 66, 77, 88, 99, "air");
		}
		if (object == 2479) {
			runecraft(playerId, 2, 11, 558, 14, 28, 42, 56, 70, 84, 98, 150, 150, "mind");
		}
		if (object == 2480) {
			runecraft(playerId, 5, 12, 555, 19, 38, 57, 76, 95, 150, 150, 150, 150, "water");
		}
		if (object == 2481) {
			runecraft(playerId, 9, 13, 557, 26, 52, 78, 150, 150, 150, 150, 150, 150, "earth");
		}
		if (object == 2482) {
			runecraft(playerId, 14, 14, 554, 35, 70, 150, 150, 150, 150, 150, 150, 150, "fire");
		}
		if (object == 2483) {
			runecraft(playerId, 20, 15, 559, 46, 92, 150, 150, 150, 150, 150, 150, 150, "body");
		}
		if (object == 2484) {
			runecraft(playerId, 27, 16, 564, 59, 150, 150, 150, 150, 150, 150, 150, 150, "cosmic");
		}
		if (object == 2487) {
			runecraft(playerId, 35, 17, 562, 74, 150, 150, 150, 150, 150, 150, 150, 150, "chaos");
		}
		if (object == 50000) {
			runecraft(playerId, 44, 18, 561, 91, 150, 150, 150, 150, 150, 150, 150, 150, "nature");
		}
		if (object == 2485) {
			runecraft(playerId, 54, 19, 563, 150, 150, 150, 150, 150, 150, 150, 150, 150, "law");
		}
		if (object == 2488) {
			runecraft(playerId, 65, 20, 560, 150, 150, 150, 150, 150, 150, 150, 150, 150, "death");
		}
		if (object == 2490) {
			runecraft(playerId, 65, 21, 565, 150, 150, 150, 150, 150, 150, 150, 150, 150, "blood");
		}
	}
}
