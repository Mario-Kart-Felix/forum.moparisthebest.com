		$namefile = fopen('name.txt', 'r');
		do {
			$line = fgets($namefile);
			if (!eregi('//', $line)) {
				$temp_array = split('#_', $line);
				if (isset($temp_array[1])) { $item_job[$temp_array[0]] = $temp_array[1]; }
				$item_job[$temp_array[0]] = str_replace('#', '', $item_job[$temp_array[0]]);
				$item_job[$temp_array[0]] = str_replace('_', ' ', $item_job[$temp_array[0]]);
				$item_name[$temp_array[0]] = ucfirst(strtolower($item_job[$temp_array[0]]));
			}
		} while (!feof($namefile));
		fclose($namefile);
		ksort($item_job);
		ksort($item_name);
		
		$descfile = fopen('description.txt', 'r');
		do {
			if (($line = fgets($descfile)) && eregi("^[0-9]+#", $line)) {
				$array_key = str_replace("#\n", '', $line);
				while (($line = fgets($descfile)) && !eregi("^#", $line)) {
					if (eregi("\+", $line)) {
						if (isset($summary[$array_key])) {
							$summary[$array_key] .= ', ' . $line;
						} else {
							$summary[$array_key] = str_replace("\n", '', $line);
						}
					} elseif (eregi('class', $line)) {
						$item_class[$array_key] = str_replace(array('Class :^777777 ', '^000000', "\n"), '', $line);
					} elseif (eregi('defense', $line)) {
						$defense_rate[$array_key] = str_replace(array('Defense :^777777 ', '^000000', "\n"), '', $line);
					} elseif (eregi('equipped', $line)) {
						$equipped_on[$array_key] = str_replace(array('Equipped On :^777777 ', '^000000', "\n"), '', $line);
					} elseif (eregi('weight', $line)) {
						$weight[$array_key] = str_replace(array('Weight :^777777 ', '^000000', "\n"), '', $line);
					} elseif (eregi('applicable', $line)) {
						$applicable_job[$array_key] = str_replace(array('Applicable Job :^777777 ', '^000000', "\n"), '', $line) + ' ';
					}
				}
			}
		} while (!feof($descfile));
		fclose($descfile);
		ksort($summary);
		ksort($item_class);
		ksort($defense_rate);
		ksort($equipped_on);
		ksort($weight);
		ksort($applicable_job);

' <br />	<b>Defense Rate:</b>  ' . $defense_rate[$key] . ' <br />	<b>Equipped On:</b>  ' . $equipped_on[$key] . ' <br />	<b>Weight:</b>  ' . $weight[$key] .
