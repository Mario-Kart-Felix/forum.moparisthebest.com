public static boolean[] itemStackable = new boolean[7000]; public static boolean[] itemIsNote = new boolean[7000];

[code] static {
int counter = 0;
int c;

    try {
        FileInputStream dataIn = new FileInputStream(new File("data/stackable.dat"));
        while ((c = dataIn.read()) != -1) {
            if (c == 0)
                itemStackable[counter] = false;
            else
                itemStackable[counter] = true;
            counter++;
        }
        dataIn.close();
    } catch (IOException e) {
        System.out.println("Critical error while loading stackabledata! Trace:");
        e.printStackTrace();
    }

    counter = 0;

    try {
        FileInputStream dataIn = new FileInputStream(new File("data/notes.dat"));
        while ((c = dataIn.read()) != -1) {
            if (c == 0)
                itemIsNote[counter] = true;
            else
                itemIsNote[counter] = false;
            counter++;
        }
        dataIn.close();
    } catch (IOException e) {
        System.out.println("Critical error while loading notedata! Trace:");
        e.printStackTrace();
    }

}[/code]

This should work, although I’m not sure about the methods you use so things may need to be changed.
