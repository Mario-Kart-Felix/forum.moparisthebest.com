[scar]{=========================================================================]
[ Rich Clicks ]
[ Version: 1.0 ]
[=========================================================================]
| Description: |
| A Macro that clicks to set co-ordinates |
[=========================================================================]
[ Credits: Bumis for the idea of this script.(its based on his BumClicker)]
[ Plz give me feedback to improve the script ]
[=========================================================================]
[Version: ]
[1.0- Script released ]
[=========================================================================]
[Future Updates: Adding a form for user friendly setup ]
[=========================================================================]
[Notes: All clicks are by default set to the buckle of the backpack. ]
[In other words the inventory button. ]
[=========================================================================}

program RichClicks;
{.include SSI2.txt}
var
Click1x,Click1y,waittime,waittimeextra,clicktol,
Click2x,Click2y,waittime,waittimeextra,clicktol,
Click3x,Click3y,waittime,waittimeextra,clicktol,
Click4x,Click4y,waittime,waittimeextra,clicktol,
Click5x,Click5y,waittime,waittimeextra,clicktol,
Click6x,Click6y,waittime,waittimeextra,clicktol,
Click7x,Click7y,waittime,waittimeextra,clicktol,
Click8x,Click8y,waittime,waittimeextra,clicktol,
Click9x,Click9y,waittime,waittimeextra,clicktol,
Click10x,Click10y,stop,Clickrepeat: Integer;

Procedure Setup;
Begin
Clicktol:=5 //This is the tolerance it randomly adds in cordinants to the x and y of your clicks.
Click1X:=845 //x position for first click
Click1Y:=334 //y position for first click
Click2X:=845 //x position for second click
Click2Y:=334 //y position for second click
Click3X:=845 //x position for third click
Click3Y:=334 //y position for third click
Click4X:=845 //x position for fourth click
Click4Y:=334 //y position for fourth click
Click5X:=845 //x position for fith click
Click5Y:=334 //y position for fith click
Click6X:=845 //x position for sixth click
Click6Y:=334 //y position for sixth click
Click7X:=845 //x position for seventh click
Click7Y:=334 //y position for seventh click
Click8X:=845 //x position for eighth click
Click8Y:=334 //y position for eighth click
Click9X:=845 //x position for ninth click
Click9Y:=334 //y position for ninth click
Click10X:=845 //x position for tenth click
Click10Y:=334 //y position for tenth click
Waittime:=25 //Time in MS between clicks
Waittimeextra:=50 //Random extra wait time in MS
Clickrepeat:=1 //How many times it will repeat your click sequence.
End;

Procedure Clicking;
Begin
Stop:=0
Repeat
Stop:=Stop+1
Mouse(Click1X,Click1Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click2X,Click2Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click3X,Click3Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click4X,Click4Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click5X,Click5Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click6X,Click6Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click7X,Click7Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click8X,Click8Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click9X,Click9Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Mouse(Click10X,Click10Y,Clicktol,Clicktol,true)
Wait((Waittime)+random(Waittimeextra))
Until(stop=Clickrepeat)
End;

begin
Setup;
Clicking;
end.

[/scar]
