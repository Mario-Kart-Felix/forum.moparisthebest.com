procedure GotoSpot1;
var a1,b1 :Integer;
begin
if(FindColor(a1,b1,fishsymb,600,25,710,140)= true)Then
wait(100);

end;

var a1,b1 :Integer;
begin
if(FindColor(a1,b1,fishsymb,600,110,710,140)= true) then
