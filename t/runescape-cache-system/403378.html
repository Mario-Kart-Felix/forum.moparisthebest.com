Just fixed a major bug with the cache archive class and hit a bit of understanding, cache index 1 - 4 are GZIPed without an archive format.

int i = 0;
            synchronized(gzip_buffer) {              
                try
                {
                    GZIPInputStream gzipinputstream = new GZIPInputStream(new ByteArrayInputStream(bytes));
                    do
                    {
                        if(i == gzip_buffer.length)
                            throw new RuntimeException("buffer overflow!");
                        int k = gzipinputstream.read(gzip_buffer, i, gzip_buffer.length - i);
                        if(k == -1)
                            break;
                        i += k;
                    } while(true);
                }
                catch(IOException _ex)
                {
                    throw new RuntimeException("error unzipping");
                }
            }
            bytes = new byte[i];
            System.arraycopy(gzip_buffer, 0, bytes, 0, i);
