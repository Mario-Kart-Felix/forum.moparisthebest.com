I’ve done this… I’m not sure if it’s the best way though.

/*
 * To change this template, choose Tools | Templates
 * and open the template in the editor.
 */

/**
 *
 * @author Steven
 */
public class Monster {

    /**
     * The name of the monster.
     */
    private String name;
    /**
     * What type of attacks the monster does.
     */
    private ATTACK_TYPES[] attackTypes;
    /**
     * The maximum damage the monster can do.
     * This should be calculated using the monsters levels.
     */
    private int maxDamage;
    /**
     * The maximum damage the monster resists.
     */
    private int damageResistance;
    /**
     * What level is needed to kill it.
     * This is not used yet.
     */
    private int levelReq;
    private int[] levels;

    /**
     * Contains the list of item types.
     *
     * EG:
     * MELEE
     * RANGE
     * MAGIC
     */
    public static enum ATTACK_TYPES {

        MELEE,
        RANGE,
        MAGIC
    };

    /**
     * Creates the new monster
     *
     * @param name The name of the monster
     * @param attackTypes The type of attacks the monster does
     * @param levelReq What level is needed to kill it.
     */
    public Monster(String name, ATTACK_TYPES[] attackTypes, int[] levels, int levelReq) {
        this.name = name;
        this.attackTypes = attackTypes;
        this.levelReq = levelReq;
        this.levels = levels;
        this.maxDamage = getLevel(Levels.STRENGTH.getLevelID())+1;
        this.damageResistance = 0;
    }

    /**
     * Gets the monsters name.
     *
     * @return name
     */
    public String getName() {
        return name;
    }

    /**
     * Gets the attack types of the monster.
     *
     * @return attackTypes
     */
    public ATTACK_TYPES[] getItemType() {
        return attackTypes;
    }

    /**
     * Gets the maximum damage the monster can do.
     *
     * @return maxDamage
     */
    public int getMaxDamage() {
        return maxDamage;
    }

    /**
     * Gets the damage resistance.
     *
     * @return damageResistance.
     */
    public int getDamageResistance() {
        return damageResistance;
    }

    /**
     * Gets the required level to kill this monster.
     *
     * @return levelReq
     */
    public int getLevelRequired() {
        return levelReq;
    }

    public int getLevel(int skillID) {
        return levels[skillID];
    }

    public static int[] setLevels(int[] skillIDs, int[] levels) {
        int[] lev = new int[Levels.values().length];
        for (int i = 0; i < lev.length; i++) {
            lev[i] = 1;
        }

        int counter = 0;
        for(int s : skillIDs) {
            lev[s] = levels[counter];
            counter++;
        }
        return lev;

    }

    public static ATTACK_TYPES getAttackTypeForString(String s) {
        if(s.equals("MELEE")) {
            return ATTACK_TYPES.MELEE;
        } else if(s.equals("RANGE")) {
            return ATTACK_TYPES.RANGE;
        } else if(s.equals("MAGIC")) {
            return ATTACK_TYPES.MAGIC;
        } else {
            return null;
        }
    }
}

I then read every file in the “Monster” folder and loaded them. (This was only temporary. I was using this way to check my combat works fine.).

EDIT: Just thought I’d add that this was for a text-based game, therefore it doesn’t contain any xy coords or render methods.
