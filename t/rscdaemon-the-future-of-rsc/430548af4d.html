Zilent August 2, 2016, 4:10pm #83

The client GUI system is coming along well. User interfaces can now be loaded from XML, JSON, or INI files. Below is a simple test GUI. So far the only built in components are labels and buttons, but the overall system is in place meaning that adding new component types is a breeze. It’s not meant to be a one stop shop for all things GUI, but only to facilitate building interfaces in native code…because let’s face it, who wants to recompile the source just to adjust the position of a label?

<?xml version="1.0" encoding="UTF-8" ?>

<!-- A simple example of the RSCDaemon native client development tool -->
<!-- This tool is still in development and is considered to be unstable -->

<!-- the root component -->
<rscd_gui>

<!-- initial (x, y) position of the gui -->	
	<x> 0 </x>
	<y> 0 </y>
	
<!-- initial width and height of the gui -->
	<width>512</width>
	<height>512</height>

<!-- initial focus settings of the gui -->
	<focusable>true</focusable>
	<focused>true</focused>

<!-- initial visibility settings of the gui
     defaults to true if not specified -->
	<visible>true</visible>
	
<!-- events that should enter the gui's delegation chain
     a blank field indicates that the event should be delegated
	 to any child components that are marked to receive them -->
	<events>
		<event>
			<type>MOUSE_ENTER</type>
			<func>onMouseEnter</func>
		</event>
		<event>
			<type>MOUSE_EXIT</type>
			<func>onMouseExit</func>
		</event>
		<event>
			<type>MOUSE_CLICK</type>
			<func></func>
		</event>
	</events>

<!-- components that populate the gui; child components may be
     nested to the maximum depth allowed by the boost::property_tree
	 implementation -->	
	<children>
		<child>
			<type>CLabel</type>
			<x>256</x>
			<y>256</y>
			<width>150</width>
			<height>20</height>
			<text>This is a label.</text>
			<color>65535</color>
			<font>4</font>
			<align>center</align>
			<events>
				<event>
					<type>MOUSE_ENTER</type>
					<func>changeLabelToRandomColor</func>
				</event>
				<event>
					<type>MOUSE_EXIT</type>
					<func>changeLabelToRandomColor</func>
				</event>
			</event>
		</child>
		<child>
			<type>CButton</type>
			<x>200</x>
			<y>200</y>
			<width>50</width>
			<height>50</height>
			<caption>Click Me</caption>
			<font>4</font>
			<align>center</align>
			<background_color>0</background_color>
			<caption_color>255</caption_color>
			<events>
				<event>
					<type>MOUSE_CLICK</type>
					<func>onCButtonClicked</func>
				</event>
			</events>
		</child>
	</children>
	
</rscd_gui>
