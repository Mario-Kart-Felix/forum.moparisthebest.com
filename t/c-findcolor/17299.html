#include <windows.h> bool FindColor(long color, int XS, int YS, int XF, int YF) { HDC dc1; int x,y; bool found = false; dc1 = GetDC(NULL); for(y = YS; y < YF; y++){ for(x = XS; x < XF; x++){ if(GetPixel(dc1,x,y)==color){ SetCursorPos(x,y); found = true; } } } return found; }
Returns true if the color is found, false if it isn’t.
