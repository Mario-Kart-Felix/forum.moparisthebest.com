Hey, well I’ve made my first code out for RSPS and I’m pretty proud of myself.

What this does, is simply count from 1-100, each second it counts a number. When it reaches, 10, 20, 30, 40, 50 it imprints a little message to tell you how many seconds it’s gone for and has a 5 second break between each 10 seconds. Then when it reaches 100 it will stop, but if it goes pass 100 I’ve made it imprint an error and to stop.

public class WhileCounter {

	/**
	 * @name WhileCounter.java
	 * @author Skyro
	 * 
	 * @param args
	 * @throws InterruptedException
	 * 
	 */

	public static void main(String[] args) throws InterruptedException {

		String name = "WhileCounter ";

		for (int i = 1; i < 101; i++) {
			System.out.println("Number: " + i + ".");
			Rest(1);

			if (i == 10) {
				System.out.println(name
						+ "has looped 10 times and is taking a break...");
				Rest(5);
			} else if (i == 20) {
				System.out.println(name
						+ "has looped 20 times and is taking a break...");
				Rest(5);
			} else if (i == 30) {
				System.out.println(name
						+ "has looped 30 times and is taking a break...");
				Rest(5);
			} else if (i == 40) {
				System.out.println(name
						+ "has looped 40 times and is taking a break...");
				Rest(5);
			} else if (i == 50) {
				System.out.println(name
						+ "has looped 50 times and is taking a break...");
				Rest(5);
			} else if (i == 100) {
				System.out.println(name
						+ "has looped 100 times and has stopped.");
				Thread.interrupted();
			} else if (i >= 101) {
				System.out.println(name
						+ "has encountered and error and has been stopped.");
				Thread.interrupted();
			}
		}
	}

	public static void Rest(int seconds) {
		try {
			Thread.sleep(seconds * 1000);
		} catch (Exception e) {
			e.printStackTrace();
			System.out.println("Counter has slept for " +

			seconds + " seconds.");
		}
	}
}
