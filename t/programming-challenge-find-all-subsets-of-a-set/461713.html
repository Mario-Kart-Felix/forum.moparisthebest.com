Write an algorithm that takes a set of integers and finds the set of all subsets of that set. Each subset and the set of all subsets have to be sorted. You can assume that the initial set is sorted.

For example if A = { 1 2 3 4 }, then all subsets of set A would be the following.

{ }
{ 1 }
{ 2 }
{ 3 }
{ 4 }
{ 1 2 }
{ 1 3 }
{ 1 4 }
{ 2 3 }
{ 2 4 }
{ 3 4 }
{ 1 2 3 }
{ 1 2 4 }
{ 1 3 4 }
{ 2 3 4 }
{ 1 2 3 4 }
#include <algorithm>
#include <iostream>
#include <vector>
#include <cstdlib>

std::vector<std::vector<unsigned int>> subsets(const std::vector<unsigned int>& set)
{
	const std::size_t setSize = set.size();
	const std::size_t subsetsSize = 1 << setSize;
	std::vector<std::vector<unsigned int>> subsets(subsetsSize);

	for (std::size_t i = 0; i < subsetsSize; ++i)
	{
		for (std::size_t j = 0; j < setSize; ++j)
		{
			if ((i & 1 << j) != 0)
			{
				subsets[i].push_back(set[j]);
			}
		}
	}

	std::sort(subsets.begin(), subsets.end(), 
		[] (const std::vector<unsigned int>& a, const std::vector<unsigned int>& b) -> bool 
	{ 
		return a.size() < b.size() || a.size() == b.size() && std::lexicographical_compare(a.begin(), a.end(), b.begin(), b.end()); 
	});

	return subsets;
}

int main()
{
	const unsigned int elements[] = { 1, 2, 3, 4 };
	const std::vector<unsigned int> set(elements, elements + 4);

	for (const std::vector<unsigned int>& subset : subsets(set))
	{
		std::cout << "{ ";

		for (const unsigned int i : subset)
		{
			std::cout << i << " ";
		}

		std::cout << "}" << std::endl;
	}

	std::cin.sync();
	std::cin.get();
	return EXIT_SUCCESS;
}

from itertools import chain, combinations

def powerset(_set):
        '''
                Simple function that uses existing python functionality to yield the powerset
        '''
        return chain.from_iterable(combinations(_set, r) for r in range(len(_set) + 1))

def main():
        '''
                Main method, demonstrates the powerset function
        '''
        _set = [ 1, 2, 3, 4]
        for _subset in powerset(_set):
                print(_subset)

#Call main
if __name__ == "__main__":
        main()
