def static subset(T… vals) {
val size = vals.length

	val powerset = <Set<T>>newHashSet

	for (var bit = 0; bit < (1 << size); bit++) {
		val set = <T>newHashSet
		for (var index = 0; index < size; index++) {
			if ((bit >> index) % 2 == 1) {
				set += vals.get(index)
			}
		}
		powerset += set
	}

	return powerset.sortWith [ one, two |
		Integer.valueOf(one.size).compareTo(two.size)
	]
}[/code]

[code=Java] public static Set<Set> subsets(Set superset) {
Set<Set> subsets = new HashSet<Set>();
subsets.add(new HashSet());

    for (T e : superset) {
        Set<Set<T>> tmp = new HashSet<Set<T>>(subsets);
        for (Set<T> subset : tmp) {
            Set<T> ext = new HashSet<T>(subset);
            ext.add(e);
            subsets.add(ext);
        }
    }
    return subsets;
}[/code]
