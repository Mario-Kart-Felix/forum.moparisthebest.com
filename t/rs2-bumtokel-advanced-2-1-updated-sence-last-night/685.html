[CHANGES FROM 2.0: SPECIAL ATTACK, BOX BREAKER FIXED, SCAR DISGUISE]

Someone posted that they didn’t like how the mouse moved, and that they didn’t like the fact that they woke up with a bunch of boxes…

Guess what? Problem solved. The new version now does random special attacks when it can, does the box random event, moves the mouse much more human-like, checks experiance and inventory once in a while to seem more human.

Just stand outside the tokel mini game, High/Low Detail (Best used on the darkest setting), your character will enter the cave, over and over, giving you tokuls and experiance.

Enjoy. First script I put a slight bit of effort into. This one took me over an hour

[Code]
{
#BumTokel Advanced 2.1

Simulates human like behavior.

Uses BoxBreaker2 Include to avoid Boxes!

Does special attack randomly.

  • Stand outside the Tokel Minigame Cave, with
  • map facing south/up, so you can see the cave
  • then run the script. Your character will fight
  • to the death over and over, collecting you tokuls
  • giving you fighting experiance
    }

program BumTokuler;
{.include SSI2.txt}
{.include BoxBreaker2.txt}
var
cave: Integer;
CaveEntered: Integer;
InvenCount,RandomInven,DoInvenCount,CheckFighting: Integer;
DoingInven,DoingMain,CheckStats,CheckingStats: Integer;
IAmFighting,DoingSpecial: Boolean;
Procedure Variables;
Begin
Cave:=9277080
UpChars:= LoadChars2(AppPath + ‘CharsRS22’);
ChatChars:= LoadChars2(AppPath + ‘CharsChat2’);
CaveEntered:=0
InvenCount:=0
End;

Procedure Review;
Begin
Writeln(’’)
Writeln(’’)
Writeln(’::::::::::::::::::::::::::::::::::::::::::::::::’)
Writeln(’: BumTokul v2.1’)
Writeln(’: You have entered the cave ’ + IntToStr(CaveEntered) + ’ times.’)
If DoingInven=1 Then
Begin
Writeln(’: Current Process: Doing Inventory’)
End;
If DoingMain=1 Then
Begin
Writeln(’: Current Process: Doing Main Procedure’)
End;
If IAmFighting=True Then
Begin
Writeln(’: Current Process: Doing Special’)
End;
If CheckingStats=1 Then
Begin
Writeln(’: Current Process: Checking Stats’)
End;
Writeln(’::::::::::::::::::::::::::::::::::::::::::::::::’)
Writeln(’’)
Writeln(’’)
End;

Procedure DoInven;
Begin
Wait(400+Random(300))
DoInvenCount:=0
DoingInven:=1
FindStrangeBox;
Repeat
Review;
Status(‘Doing Inventory’)
DoInvenCount:=DoInvenCount+1
RandomInven:=Random(5)
Case RandomInven Of
1:
Begin
Wait(500+Random(200))
MMouse(581,225,7,7)
Wait(1000+Random(2000))
End;
2:
Begin
Wait(500+Random(200))
MMouse(611,333,20,20)
Wait(1000+Random(2000))
End;
3:
Begin
Wait(5000+Random(5000))
End;
4:
Begin
MMouse(599,323,20,20)
Wait(100+Random(100))
MMouse(394,271,20,20)
Wait(100+Random(100))
MMouse(293,108,20,20)
Wait(500+Random(500))
MoveMouseSmooth(0+random(510),0+Random(330))
End;
5:
Begin
MMouse(599,323,20,20)
Wait(100+Random(100))
MMouse(394,271,20,20)
Wait(100+Random(100))
MMouse(293,108,20,20)
Wait(500+Random(500))
MoveMouseSmooth(0+random(510),0+Random(330))
End;
End;
Until(DoInvenCount=5+random(6))
DoingInven:=0
End;

Procedure CheckSpecial;
Begin
Review;
IAmFighting:=True
Wait(1300+Random(400))
MMouse(200,200,75,75)
Wait(500+Random(100))
MMouse(559,186,5,5)
Wait(400+Random(200))
GetMousePos(x,y)
ClickMouse(x,y,true)
If FindColorTolerance(x,y,1203866,615,440,641,452,10) Then
Begin
DoingSpecial:=True
Review;
MoveMouseSmooth(577+Random(110),442+Random(11))
Wait(1000+Random(400))
GetMousePos(x,y)
ClickMouse(x,y,true)
Wait(5000+random(1400))
CheckFighting:=0
End;
IAmFighting:=False
DoingSpecial:=False
Review;
End;

Procedure AmIFighting;
Begin
If FindColorTolerance(x,y,65280,187,30,419,243,10) Then
Begin
IAmFighting:=True
Review;
CheckSpecial;
End;
IAmFighting:=False
Review;
End;

Procedure CheckMyStats;
Begin
Wait(500+Random(500))
MMouse(583,186,7,7)
Wait(200+Random(100))
GetMousePos(x,y)
MMouse(x,y,2,2)
Wait(200+Random(200))
MoveMouseSmooth(559+Random(50),212+Random(80))
Wait(500+Random(300))
GetMousePos(x,y)
MMouse(x,y,2,2)
End;

Procedure InTheCave;
Begin
DoingMain:=1
Status(‘Main Procedure’)
Review;
Wait(500+random(2000))
GetMousePos(x,y)
MMouse(x,y,2,2)
Wait(1000+Random(3000))
Begin
MMouse(361,233,77,77)
Wait(100+Random(400))
GetMousePos(x,y)
MMouse(x,y,5,5)
Wait(100+Random(400))
GetMousePos(x,y)
MMouse(x,y,3,3)
Wait(100+Random(400))
MMouse(250,150,250,150)
If CheckStats=7+Random(2) Then
Begin
CheckingStats:=1
Review;
CheckMyStats;
CheckStats:=0
End;
InvenCount:=InvenCount+1
CheckFighting:=CheckFighting+1
If CheckFighting=1+Random(4) Then
Begin
CheckFighting:=0
DoingMain:=0
Review;
AmIFighting;
End;
If InvenCount=6+Random(4) Then
Begin
MMouse(650,183,4,4)
DoingMain:=0
DoInven;
End;

Review;
End;
DoingMain:=0
End;

Procedure OutSideCave;
Begin
DoingMain:=1
Review;
Status(‘Finding Cave’)
If FindColorTolerance(x,y,cave,0,0,466,299,5) Then
Begin
MMouse(x,y,5,5)
Wait(50+Random(50))
GetMousePos(x,y)
MMouse(x,y,2,2)
Wait(150+Random(200))
If IsTextInAreaEx(4,4,25,25,x,y,‘Enter’,100,UpChars,True,false,0,1,13224393) then
Begin
GetMousePos(x,y)
ClickMouse(x,y,True)
MMouse(647,184,5,5)
Wait(300+Random(300))
GetMousePos(x,y)
ClickMouse(x,y,true)
Wait(500+random(500))
FindStrangeBox;
CaveEntered:=CaveEntered+1
Review;
InTheCave;
End;
End;
DoingMain:=0
End;

begin
Variables;
IAmFighting:=False
DisguiseSCAR(‘Windows Media Player’)
BoxOpenerSetup;
Repeat
Review;
OutSideCave;
Wait(1000+Random(500))
InTheCave;
Until(False)
end.[/Code]
