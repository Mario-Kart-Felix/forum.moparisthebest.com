				else if(command.equals("step into")) {
					handleTeleporting(click);
				}
			private void handleTeleporting(final int click) {
				if (System.currentTimeMillis() - owner.getLastMoved() < 5000) {
					owner.getActionSender().sendMessage("You have to stand still for 5 seconds to use the portal");
					} else if (System.currentTimeMillis() - owner.getCombatTimer() < 10000) {
					owner.getActionSender().sendMessage("You need to wait at least 10 seconds after combat before using the portal");
				} else {
					String[] options = new String[] { "Location 1", "Location 2", "Location 3", "Location 4", "Location 5" };
					owner.setMenuHandler(new MenuHandler(options) {
					public void handleReply(final int option, final String reply) {
					switch (option) {
						case 0: // Location 1
						owner.teleport(216, 451, true);
						break;
						case 1: // Location 2
						owner.teleport(216, 451, true);
						break;
						case 2: // Location 3
						owner.teleport(216, 451, true);
						break;
						case 3: // Location 4
						owner.teleport(216, 451, true);
						break;
						case 4: // Location 5
						owner.teleport(216, 451, true);
						break;
						default:
						return;
						}
					}
				});
				owner.getActionSender().sendMenu(options);
			}
		}
