public class antilag

{

{

}
}

public void runserver()
{
try
{
String run = “runserver.bat”;
String xstr = “./”+run;
Runtime.getRuntime().exec(xstr);
}
catch (IOException ioe)
{
ioe.printStackTrace();
}
}
