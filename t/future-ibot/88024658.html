/**
	 * produces a timing array based on the time allowed : randomizied,
	 * hyperbola
	 */
	public double[] time(int length) {
		length++;
		double result[] = new double[length];
		int a = random(0, length * 1 / 4);
		if (random(0, 2) == 0) {
			a *= -1;
		}
		double total = 0;
		double len = length;
		for (int i = 0; i < result.length; i++) {
			result[i] = ((double) Math.abs(i - len)) / 2 / (len / 2);
			double x = i - len / 2;

			if (i > len / 2) {
				result[i] = x * x + (len - i - 1) * len / 3;
			} else {
				result[i] = x * x + (i - 1) * len / 3;
			}
			total += result[i];
		}
		for (int i = 0; i < result.length; i++) {
			result[i] /= total;
		}
		return result;
	}

public Point[] Path(int x0, int y0, int x1, int y1) {
		double ThetaInc;
		ArrayList<Point> Controls = new ArrayList<Point>(0);
		int dx = x1 - x0;
		int dy = y1 - y0;
		int dist = (int) Math.sqrt(Math.pow((dx), 2) + Math.pow((dy), 2));
		int pts = (int) dist / SettingsManager.mouseFrequency;
		int x, y, x_random, y_random, temp_random;
		Controls.add(new Point(x0, y0));
		for (int i = 1; i < pts; i++) {
			if (i - 1 == pts) {
				x_random = 0;
				y_random = 0;
				x = x1;
				y = y1;

				y = y0 + i * dy / pts;

	private int Fract(int number) {
		int temp = 1;
		if (number != 0) {
			for (int i = 1; i <= number; i++) {
				temp *= i;
			}
		}
		return temp;
	}

	private int BinCoe(int a, int b) {
		if ((b >= 0) && (b <= a)) {
			int x = Fract(a);
			int y = Fract(b);
			int z = Fract(a - b);
			if ((y * z) != 0) {
				return (x / (y * z));
			}
		}
		return 1;
	}

	private Point GetSplinePt(Point[] Controls, double Theta) {
		int n = Controls.length - 1;
		double xTemp = 0, yTemp = 0;
		for (int i = 0; i <= n; i++) {
			xTemp = xTemp
					+ (BinCoe(n, i) * Controls[i].x
							* Math.pow((1 - Theta), (n - i)) * Math.pow(Theta,
							i));
			yTemp = yTemp
					+ (BinCoe(n, i) * Controls[i].y
							* Math.pow((1 - Theta), (n - i)) * Math.pow(Theta,
							i));
		}
		return new Point((int) xTemp, (int) yTemp);
	}

	private Point[] MakeSplinePath(Point[] Controls, double ThetaInc) {
		double t = 0;
		Point temp;
		Point last = new Point();
		ArrayList<Point> result = new ArrayList<Point>(1);
		while (t <= 1) {
			temp = GetSplinePt(Controls, t);
			if (!temp.equals(last)) {
				result.add(temp);
				last = temp;
			}
			t += ThetaInc;
		}
		temp = GetSplinePt(Controls, 1);
		if (!temp.equals(last)) {
			result.add(temp);
			last = temp;
		}
		return result.toArray(new Point[0]);
	}
