//STDCALL - SCAR/Simba
long GetFunctionCount() {
    return NumExports;
}
//STDCALL - SCAR/Simba
long GetFunctionInfo(int index, void*& address, char*& def) {
    if (index < NumExports) {
        address = (void*)GetProcAddress(dllinst, exports[index*2]);
        strcpy(def, exports[index*2+1]);
        return index;
    }
    return -1;
}

as for the openGL wrapper, you can use the one from silentwolf to get started I have attached it to this post.

When: glEnable GL_DEPTH_TEST (2929) - the viewport of the gameworld is drawing.
when glDisable GL_DEPTH_TEST - the viewport of the gameworld has finished drawing and the gl is now rendering the 2d graphics.

that should get you started.
