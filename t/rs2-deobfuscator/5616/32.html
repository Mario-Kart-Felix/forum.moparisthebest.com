Hmm :p, here’s my copy of a word finder. When I have more time I’ll put in some regex.

import java.io.*;
import java.util.*;

public class Finder {

    public String path = "";
    public String[] wordsToFind = new String[100];
    public HashMap<Integer, String> fileList = new HashMap<Integer, String>();

    public static void main(String[] args) {
        Finder find = new Finder();
    }
    
    public Finder() {
        loadProperties();
        loadFileList();
        try {
            for (int x = 0; x < fileList.size(); x++) {
                int count = 0;
                String line = "";
                BufferedReader br = new BufferedReader(new FileReader(path + fileList.get(x)));
                while ((line = br.readLine()) != null) {
                    count++;
                    for (int y = 0; y < wordsToFind.length; y++) {
                        if (wordsToFind[y] != null) {
                            if (line.indexOf(wordsToFind[y]) != -1) {
                                System.out.println(wordsToFind[y] + " found on line " + count + " in class " + fileList.get(x));
                            }
                        }
                    }
                }
                count = 0;
            }
        } catch (IOException ioe) {
            ioe.printStackTrace();
        }
    }
    
    public void loadProperties() {
        try {
            Properties p = new Properties();
            InputStream is = new FileInputStream("Settings.ini");
            p.load(is);
            path = p.getProperty("Path");
            wordsToFind = p.getProperty("Words").split(", ");
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
    
    public void loadFileList() {
        File folder = new File(path);
        if (folder.exists()) {
            String[] files = folder.list();
            for (int x = 0; x < files.length; x++) {
                if (files[x].endsWith(".java")) {
                    fileList.put(fileList.size(), files[x]);
                }
            }
        } else {
            System.out.println("The path in the settings does not exist!");
            System.exit(0);
        }
    }
}

And it loads the path and words to find from an ini file.
