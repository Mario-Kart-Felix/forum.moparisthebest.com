        layout.add(layoutD, "South");
        layoutD.add(scroll, "South");
    }

        messages.setText("");
	searchFiles(searchText.getText());
    }

    boolean contains(String original, String checkFor) {
	original = original.toLowerCase();
	checkFor = checkFor.toLowerCase();
        return !original.contains(checkFor);
    }

    void searchFiles(String textToSearch) {
	File file = new File("./");

        for (String a : file.list()) {
            File temp = new File(a);
            if(!temp.isDirectory()) {
                searchFile(a, textToSearch);
            }
        }
    }

    void searchFile(String file, String textToSearch) {
        try {
            String[] skippedExtensions = {".zip", ".rar", ".gz", ".jar", ".7z"};

            for(String ext : skippedExtensions) {
                if(file.endsWith(ext)) {
                    return;
                }
            }

            int lineNum = 0;
	    BufferedReader br = new BufferedReader(new FileReader(file));
            do {

	        String currLine;
		do {
		    if ((currLine = br.readLine()) == null) {
		        return;
		    }
		    lineNum++;
		} while(contains(currLine, textToSearch));
  		
		addMessage("        " + file + " (Line " + lineNum + ")     ");
		addMessage("            " + currLine + "     ");
		addMessage("");
	    } while (true);
        } catch (Exception e) {
            e.printStackTrace();
	}
    }

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.BufferedReader;
import java.io.File;
import java.io.FileReader;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class MainFrame extends JFrame implements ActionListener {

    public MainFrame(String title) {
        super(title);

        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setMinimumSize(new Dimension(765, 500));

        JLabel searchForText = new JLabel("Search For: ", SwingConstants.RIGHT);
        searchBox = new JTextField(30);
        searchBox.addActionListener(this);
        JButton searchButton = new JButton("Search!");
        searchButton.addActionListener(this);

        caseSensitiveOption = new JCheckBox("Case Sensitive");
        regexOption = new JCheckBox("Use Regex");
        resultArea = new JTextArea(10, 10);
        resultArea.setForeground(Color.GREEN);
        resultArea.setBackground(Color.BLACK);
        resultArea.setEditable(false);
        resultArea.setFocusable(false);
        resultScrollPane = new JScrollPane(resultArea);

        JPanel searchPanel = new JPanel(new FlowLayout());
        JPanel resultPanel = new JPanel(new BorderLayout());

        searchPanel.add(searchForText);
        searchPanel.add(searchBox);
        searchPanel.add(searchButton);
        searchPanel.add(caseSensitiveOption);
        searchPanel.add(regexOption);

        resultPanel.add(resultScrollPane);

        add(searchPanel, BorderLayout.PAGE_START);
        add(resultPanel, BorderLayout.CENTER);

        pack();

        setVisible(true);
    }

    public void actionPerformed(ActionEvent actionEvent) {
        resultArea.setText("");
        searchFiles(searchBox.getText());
    }

    boolean contains(String line, String toFind) {
        if(!regexOption.isSelected()) {
            if(!caseSensitiveOption.isSelected()) {
                toFind = toFind.toLowerCase();
                line = line.toLowerCase();
            }
            return line.contains(toFind);
        }
        Matcher regexMatcher  = Pattern.compile(toFind).matcher(line);

       return regexMatcher.find();
    }


    void addMessage(String s) {
        resultArea.append(s + "\n");
        resultScrollPane.getVerticalScrollBar().setValue(resultScrollPane.getVerticalScrollBar().getMaximum());
    }

    void searchFiles(String textToSearch) {
        File file = new File("./");

        for (String a : file.list()) {
            File temp = new File(a);
            if(!temp.isDirectory()) {
                searchFile(a, textToSearch);
            }
        }
    }

    void searchFile(String file, String textToSearch) {
        try {
            String[] skippedExtensions = {".zip", ".rar", ".gz", ".jar", ".7z"};

            for(String ext : skippedExtensions) {
                if(file.endsWith(ext)) {
                    return;
                }
            }

            int lineNum = 0;
            BufferedReader br = new BufferedReader(new FileReader(file));

            do {
                String currLine;
                do {
                    if ((currLine = br.readLine()) == null) {
                        return;
                    }
                    lineNum++;
                } while(!contains(currLine, textToSearch));
                    addMessage("        " + file + " (Line " + lineNum + ")     ");
                    addMessage("            " + currLine + "     ");
                    addMessage("");
            } while (true);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    public static void main(String[] args) {
        new MainFrame("[Synchronized_ Updated]: File Searcher");
    }

    private JTextField searchBox;
    private JCheckBox caseSensitiveOption;
    private JCheckBox regexOption;
    private JTextArea resultArea;
    private JScrollPane resultScrollPane;
}

No CoonHunter, you need to use regex ‘patterns’ in order for it to recognize it.

For example, try searching for: [tt]x.men[/tt]
‘.’ is a special character in regex patterns that basically is a wild card, and if you search for it, anything like xtmen, x-men, x6men etc would show up.

There’s also \d, which means digit. [tt]x\dmen[/tt] would match: x6men, x1men, x9men etc

Tons of other possibilities too
