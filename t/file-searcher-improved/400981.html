Surely you could write your own? Even one like this could do the trick. (C# lolololoololol)

using System;
using System.Text.RegularExpressions;
using System.IO;

namespace RegexSearchC
{
    class MainClass
    {
        private int matchCount = 0;
	
	private void search(string fileName, int lineNum, string text, string pat) {
	    Regex r = new Regex(pat, RegexOptions.IgnoreCase);
	    Match m = r.Match(text);
	
	    while (m.Success) {
	        matchCount++;
   		for (int i = 1; i <= 2; i++) {
      		    Group g = m.Groups[i];
      		    CaptureCollection cc = g.Captures;
      		    
    		    for (int j = 0; j < cc.Count; j++) {
                        System.Console.WriteLine("[" + fileName + "]: FOUND: \""+text+ "\" AT LINE: " + lineNum);
      		    }
                }
		m = m.NextMatch();
	    }	
 	}
		
	public static void Main (string[] args) {
            MainClass main = new MainClass();
	    string[] aFiles = Directory.GetFiles(".");
			
	    Console.WriteLine("Please enter the regex: ");
	    string reg = Console.ReadLine();

	    foreach(string s in aFiles) {
	        int lineNum = 0;
		foreach (string line in File.ReadAllLines(s)) {
		    lineNum ++;
	            main.search(s, lineNum, line, reg);
        	}
		Console.WriteLine("[" + s + "]: Matches: " + main.matchCount);
		main.matchCount = 0;
	    }
	}
    }
}
