[code]program myfirstscript;
{.include srl/srl.scar}
{.include srl/srl/skill/woodcutting.scar}

const
treecolor1=5075056;
treecolor2=2897976;
treecolor3=3825752;
timetowait1=5000;

Procedure findtree;
begin
if(findcolorspiraltolerance(x,y,treecolor1,msx1,msy1,msx2,msy2,25))or
(findcolorspiraltolerance(x,y,treecolor2,msx1,msy1,msx2,msy2,25))or
(findcolorspiraltolerance(x,y,treecolor3,msx1,msy1,msx2,msy2,25))then
begin
MMouse(x, y, 3, 3);
if(IsUpTextMulti(‘chop’,‘op do’,‘down’))then
begin
writeln(“Found a tree.”)
Mouse(x, y, 2, 2, True);
Wait(13);
Exit;
end else
begin
writeln(“No trees were found, did you pick good colors?”)
end;
end;
end;

Procedure antirandoms;
begin
if FindFight then
begin
RunAwayDirection(‘n’);
Wait(10000+random(2000));
RunBack;
end;
FindLamp(‘mining’);
TypeSend(‘this is so boring’);
end;

procedure invcheck;
begin
if InvCount = 28 then
begin
DropTo(2, 28);
end;
end;

begin
SetUpSRL;
SetRun(true);
repeat
MakeCompass(‘n’);
FindNormalRandoms;
findtree;
invcheck;
timetowait1;
until(false);
end.[/code]
