[code=python]#Solves Quadratic equation

ax**2(ax’squared’) + bx + c = 0

#import real numbers
import math

a = float(input("Enter a: "))
b = float(input("Enter b: "))
c = float(input("Enter c: "))

#Finds the Discriminant
d = (b**2) - (4ac)

#finds solutions
solt1 = (-b-math.sqrt(d))/(2a)
solt2 = (-b+math.sqrt(d))/(2a)
print “The Discriminant is”, d
print (“The solutions are {0} and {1}”.format(solt1, solt2))
[/code]
