public static void GodWars(TextDrawingArea[] wid) {
	RSInterface Tab = addTab(16210);
	addText(16211, "NPC killcount", wid, 1, 0xff9040);
	addText(16212, "Armadyl kills", wid, 1, 0xff9040);
	addText(16213, "Bandos kills", wid, 1, 0xff9040);
	addText(16214, "Saradomin kills", wid, 1, 0xff9040);
	addText(16215, "Zamorak kills", wid, 1, 0xff9040);
	addText(16216, "0", wid, 1, 0x66FFFF);//armadyl
	addText(16217, "0", wid, 1, 0x66FFFF);//bandos
	addText(16218, "0", wid, 1, 0x66FFFF);//saradomin
	addText(16219, "0", wid, 1, 0x66FFFF);//zamorak
	Tab.isMouseoverTriggered = false;
	Tab.totalChildren(9, 9, 9);
	Tab.child(0, 16211, -52+375, 7);
	Tab.child(1, 16212, -52+374, 30);
	Tab.child(2, 16213, -52+372, 44);
	Tab.child(3, 16214, -52+381, 58);
	Tab.child(4, 16215, -52+375, 73);
	Tab.child(5, 16216, -52+455, 31);
	Tab.child(6, 16217, -52+455, 45);
	Tab.child(7, 16218, -52+455, 59);
	Tab.child(8, 16219, -52+455, 74);
}
	public void totalChildren(int t) {
		children = new int[t];
		childX = new int[t];
		childY = new int[t];
	}
	public void child(int id, int interID, int x, int y) {
		children[id] = interID;
		childX[id] = x;
		childY[id] = y;
	}
