Ron July 30, 2016, 8:34am #33
XxKanexX:

I don’t think there is a function that saves to the clipboard, so no, you can’t.

Yeah there is. I’ve used it in a number of my scripts. Here is some test code for it. 

program ClipboardTestByRon;

var
  Form1 : TForm;
  Memo1 : TMemo;

procedure CopyToClip(Sender : TObject);
begin
  Memo1.SelectAll;
  Memo1.CopyToClipboard;
end;

procedure PasteFromClip(Sender : TObject);
begin
  Memo1.PasteFromClipboard;
end;

procedure InitForm;
var
  Button1, Button2 : TButton;
begin
  Form1 := CreateForm;
  Form1.Caption := 'Clipboard Test by Ron';
  Form1.Position := poScreenCenter;
  Form1.BorderIcons := [biSystemMenu, biMinimize];
  Form1.BorderStyle := bsSingle;
  Memo1 := TMemo.Create(Form1);
  Memo1.Parent := Form1;
  Memo1.Top := 10;
  Memo1.Left := 10;
  Memo1.Width := Form1.ClientWidth - (Memo1.Left * 2);
  Memo1.Height := Form1.ClientHeight - 55;
  Button1 := TButton.Create(Form1);
  Button1.Parent := Form1;
  Button1.Caption := 'Copy To Clip';
  Button1.Width := 100;
  Button1.Top := Memo1.Height + Memo1.Top + 10;
  Button1.Left := 10;
  Button1.OnClick := @CopyToClip;
  Button2 := TButton.Create(Form1);
  Button2.Parent := Form1;
  Button2.Caption := 'Paste From Clip';
  Button2.Width := Button1.Width;
  Button2.Top := Button1.Top;
  Button2.Left := Form1.ClientWidth - Button2.Width - Button1.Left;
  Button2.OnClick := @PasteFromClip;
end;

procedure SafeInitForm;
var
  v : TVariantArray;
begin
  SetArrayLength(v, 0);
  ThreadSafeCall('InitForm', v);
end;

procedure ShowFormDesign;
begin
  Form1.ShowModal;
end;

procedure SafeShowFormDesign;
var
  v : TVariantArray;
begin
  SetArrayLength(v, 0);
  ThreadSafeCall('ShowFormDesign', v);
end;

procedure MainInitForm;
begin
  try
    SafeInitForm;
    SafeShowFormDesign;
  finally
    FreeForm(Form1);
  except
    WriteLn('An error seems to have occurred in: InitForm');
  end;
end;

begin
  ClearDebug;
  MainInitForm;
end.

Also, thanks for all the great comments. I appreciate them very much.
