[CODE]{.Script Info:

ScriptName = PowerMinerSD

Author = Surajd

Description = Powermines any rock, anywhere! =D

Version = v1.0

Date = Published 3/07/06

Comments = Must have Pickaxe WIELDED! Setup Lines: 36-52 and Setup your players in the PlayerForm which will AutoStart!

/Script Info}

program PowerMiner_by_Surajd;
{.include SRL\SRL.scar}
{.include SRL\SRL\Skill\Mining.scar}
{.include SRL\SRL\Misc\PlayerForm.scar}

var
a, b: Integer;
MTime: LongInt;
MiTimes, DTimes, ETimes: Integer;

const

// ********** Colors taken from Pups Powerminer **************************
Copper = 4549781;
Tin = 9277080;
Clay = 4553880;
Iron = 2371664;
Silver = 12434885;
Coal = 2834753;
Gold = 5161431;
Mithril = 7359564;
Adamant = 6322272;
Runite = 0; // If your Powermining Runite Ore, Set the Color here!
// ************************************************************************

RockType = Tin; // Which Rock would you like to PowerMine on?
OreTol = 5; // Tolerance of Finding the Rock

AntiBanAB = True; // Do you want Anti-Ban functions for less detectability?
BenMouseBM = False; // Do you want to use BenMouse’s Mouse Procedure? (Works best withought BenMouse!)
MouseSpeedMS = 10; // Choose how fast you want the Mouse to move at!

TalkRandomly = True; // Talk Randomly or not?
RChat = 1; // Random-Talk time? EG: 1 for every 1 minute
TalkAboutSkill = True; // Talks about the Skill: Mining

WaitTime = 5000; // Waiting time between each swing
MaxWaitTime = 10000; // The MAX Waiting time before each swing

RunDi = ‘N’; // What direction to run if a Fight is found

HowManyD = 5; // How many Loads to Drop before Switching Players!

//------------------------------------------------------
VersionNumber = ‘v1.0 by Surajd’; // Do not change!
//------------------------------------------------------

procedure ProgressReport;
begin
begin
WriteLn(’ ‘);
WriteLn(’<============== -’ + VersionNumber + ‘- Progress Report ===============>’);
WriteLn(’ Worked for ’ + TimeRunning);
WriteLn(’ ‘);
WriteLn(’ Mined: ’ + inttostr(MiTimes) + ’ Ore(s)! ‘);
WriteLn(’ Dropped: ’ + inttostr(DTimes) + ’ Load(s)! ‘);
WriteLn(’ Encountered: ’ + inttostr(ETimes) + ’ Fight(s)! ');
SRLRandomsReport;
end;
end;

procedure SetupScript;
begin
PerfectNorth;
SetChat(‘Hide’, 1);
SetChat(‘Off’, 2);
SetChat(‘Off’, 3);
SetRun(True);
FindPickHeadColor;
GameTab(4);
end;

// ************** Taken from Freddy1990’s FeatherBuyer! **************

procedure MouseHuman(X, Y, XR, YR: integer; click, left: Boolean);
var
xx, yy: Integer;
begin
xx := X + Random(XR);
yy := Y + Random(YR);
MoveMouseSmoothEx(xx, yy, 3, 8, 20, 20, 10);
if click = True then
begin
Wait(200 + Random(200));
HoldMouse(xx, yy, left);
Wait(20 + Random(15));
ReleaseMouse(xx, yy, left);
end;
end;
// *****************************************************************

procedure AntiRandomsAR;
begin
FindPick;
GasColors(a, b);
if (FindGas(a, b - 15)) then
begin
FindSymbol(a, b, ‘Mining Site’);
MouseHuman(a, b, 1, 2, true, true);
WriteLn(’ Found Gas, Gas-Protection Activated! ‘);
FindNormalRandoms;
if (FindFight = True) then
begin
WriteLn(’ We are in a fight, Running… ‘);
ETimes := ETimes + 1
SetRun(True);
RunAwayDirection(RunDi);
Wait(5000 + Random(191));
RunBack;
WriteLn(’ Successfully Dodged… Continuing… ');
if (AntiBanAB = True) then AntiBan;
if (TalkRandomly = True) then RandomChatEvery(RChat);
if (TalkAboutSkill = True) then GhostSpeak(‘Mining’);
if (AntiBanAB = True) then LeaveScreenEvery(1);
end;
end;
end;

procedure StartMining;
begin
Writeln(’ Starting to PowerMiner… ');
MarkTime(MTime);
repeat
AntiRandomsAR;
if (FindColorSpiralTolerance(a, b, RockType, 0, 0, 515, 338, OreTol)) then
MouseHuman(a, b, 1, 1, false, false);
wait(1000 + Random(500 - 100 * 1 - 2));
MouseHuman(a, b, 1, 1, true, true);
MiTimes := MiTimes + 1
wait(WaitTime + Random(1500));
repeat
AntiRandomsAR;
Wait(500 + Random(100));
MouseHuman(a, b, 392, 241, false, false);
until (TimeFromMark(MTime) >= MaxWaitTime);
until (InvFull);
end;

procedure DropAllOres;
begin
WriteLn(’ Dropping Ores… ');
DropAll;
wait(1001 + Random(250));
DwarfItem;
wait(1002 + Random(350));
DropAll;
wait(1003 + Random(450));
DTimes := DTimes + 1
ProgressReport;
end;

procedure MainLoop;
begin
SetupPlayers;
wait(5000 + Random(500));
SetupScript;
repeat
if (not (LoggedIn)) then NextPlayer(false);
SetRun(True);
StartMining;
DropAllOres;
until (HowManyD = DTimes);
ProgressReport;
NextPlayer(true);
end;

begin
MiTimes := 0
DTimes := 0
ETimes := 0
SetupSRL;
SetupSRLMining;
DisguiseScar(‘Notepad’);
BenMouse := BenMouseBM;
MouseSpeed := MouseSpeedMS;
MainLoop;
end.[/CODE]
