public void createGroundItem(int itemID, int itemX, int itemY, int itemAmount) {// Phate: Omg fucking sexy! creates item at absolute X and Y
	player.getIoSession().write(new StaticPacketBuilder().setId(85)					// Phate: Spawn ground item
		.addByteC(itemY - 8 * mapRegionY(itemY))
		.addByteC(itemX - 8 * mapRegionX(itemX))
		.toPacket());
	player.getIoSession().write(new StaticPacketBuilder().setId(44)
		.addEndianA(itemID)
		.addShort(itemAmount)
		.addB(0)								// x(4 MSB) y(LSB) coords
		.toPacket());
		System.out.println("CreateGroundItem "+itemID+" "+(itemY - 8 * mapRegionX(itemX))+","+(itemY - 8 * mapRegionY(itemY))+" "+itemAmount);
}

public int mapRegionX(int x){return ((x>>3)-6);}
public int mapRegionY(int y){return ((y>>3)-6);}

	public void AddItem(int item, int amount, int slot) {
		if (amount > 254) {
			player.getIoSession().write(new StaticPacketBuilder().setId(34).setSize(Packet.Size.VariableByte)
			.addShort(3214)
			.addByte((byte)slot)
			.addShort(item+1)
			.addByte((byte)255)
		.addByte((byte)(amount >> 24))
		.addByte((byte)(amount >> 16))
		.addByte((byte)(amount >> 8))
		.addByte((byte)(amount))
			.toPacket());
		} else {
			player.getIoSession().write(new StaticPacketBuilder().setId(34).setSize(Packet.Size.VariableByte)
			.addShort(3214)
			.addByte((byte)slot)
			.addShort(item+1)
			.addByte((byte)amount)
			.toPacket());
		}
	}
