Yes, this isn’t complex in any way, nor does it handle anything advanced. Just a quick way to check your answers, and I want to contribute it.

import java.util.Scanner;


public class Math {

	
	static Scanner scanner = new Scanner(System.in);

	public static void main(String[] args) {
		calculate();
	}
	
	private static void calculate() {
		double answerX, answerY, x1, x2, y1, y2;
		
		System.out.print("X1:");
		x1 = scanner.nextDouble();
		System.out.print("\nX2:");
		x2 = scanner.nextDouble();
		System.out.print("\nY1:");
		y1 = scanner.nextDouble();
		System.out.print("\nY2:");
		y2 = scanner.nextDouble();
		
		answerX = (x1 + x2) / 2;
		answerY = (y1 + y2) / 2;
		
		System.out.println("(" + answerX + ", " + answerY + ")");
	}

}

Enjoy.
