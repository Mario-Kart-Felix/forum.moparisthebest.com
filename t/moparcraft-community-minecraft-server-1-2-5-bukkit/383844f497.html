Well, that was painful, but I accomplished it:

$HTTP["url"] =~ "/worldmap/tiles/faces/8x8/" {
  server.error-handler-404 = "/steve8x8.png"
}
$HTTP["url"] =~ "/worldmap/tiles/faces/16x16/" {
  server.error-handler-404 = "/steve16x16.png"
}
$HTTP["url"] =~ "/worldmap/tiles/faces/32x32/" {
  server.error-handler-404 = "/steve32x32.png"
}
$HTTP["url"] =~ "/worldmap/tiles/faces/body/" {
  server.error-handler-404 = "/stevebody.png"
}
