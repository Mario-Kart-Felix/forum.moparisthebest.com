Here is a method I wrote years ago to dump files edited with BCEL:

	/**
	 * Dumps all the classes to a jar file of your choise.
	 * @param outputFile The file to dump to
	 * @param classes The classes to dump
	 */
	public static void dumpClassesToJar(File outputFile, Map<String, ClassGen> classes) throws IOException {
		if(outputFile.exists())
			outputFile.delete();

		JarOutputStream out = new JarOutputStream(new FileOutputStream(outputFile));
		JarEntry manifestEntry = new JarEntry("META-INF/MANIFEST.MF");
		out.putNextEntry(manifestEntry);
		//TODO we're not sun microsystems, change that attribute to something else
		//talk to other devs, it will probably end up saying cherokee or something
		out.write("Manifest-Version: 1.0\nCreated-By: 1.6.0 (Sun Microsystems Inc.)\n\n\n".getBytes());
		out.flush();
		out.closeEntry();
		for(ClassGen cg : classes.values()) {
			
			JarEntry entry = new JarEntry(cg.getClassName()+".class");
			out.putNextEntry(entry);
			cg.getJavaClass().dump(out);
			out.flush();
			out.closeEntry();
		}
		out.finish();
		out.close();
	}
