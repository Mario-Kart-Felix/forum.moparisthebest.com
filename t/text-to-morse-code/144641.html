public class Morse {

public static String them[] = new String[255];
/** Creates a new instance of Morse */
public Morse() {

}
public static void setThem() {
        for(int i = 0; i<255; i++){
        them[i] = "non";
    }
    them[97] = "•-";
    them[98] = "-•••";
    them[99] = "-•-•";
    them[100] = "-••";
    them[101] = "•";
    them[102] = "••-•";
    them[103] = "--•";
    them[104] = "••••";
    them[105] = "••";
    them[106] = "•---";
    them[107] = "-•-";
    them[108] = "•-••";
    them[109] = "--";
    them[110] = "-•";
    them[111] = "---";
    them[112] = "•--•";
    them[113] = "--•-";
    them[114] = "•-•";
    them[115] = "•••";
    them[116] = "-";
    them[117] = "••-";
    them[118] = "•••-";
    them[119] = "•--";
    them[120] = "-••-";
    them[121] = "-•--";
    them[122] = "--••";
    them[48] = "•----";
    them[49] = "••---";
    them[50] = "•••--";
    them[51] = "••••-";
    them[52] = "•••••";
    them[53] = "-••••";
    them[54] = "--•••";
    them[55] = "---••";
    them[56] = "----•";
    them[57] = "-----";
    them[32] = "   ";
    them[46] = "       ";
}

public static String toMorse(String b) {
    setThem();
    if (!b.equals(null)) {
    b = b.toLowerCase();
    StringBuffer s = new StringBuffer();
    for(char c : b.toCharArray()){
        int em = Integer.parseInt(Integer.toString(c));
        if(them[em] != "non"){
        s.append(them[Integer.parseInt(Integer.toString(c))] + " ");
        } else {
            return "Invalid Characters";                
        }
    }
    return s.toString();
} else {
        return "Error";
}
}

}
