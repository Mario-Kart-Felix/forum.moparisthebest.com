program New;

{.include srl/srl.scar}

procedure SiMMouse(x, y, rx, ry: Integer); // Spline Mouse From SI, slightly modded by RealRune
var
  cx, cy, mx, my, ox, oy, ms: Integer;
  mu, mum1, mum12, mu2, incr: Extended;
begin
  GetMousePos(cx, cy);
  if(x=cx)and(y=cy)then
    Exit;
  x := x + ((Random(rx)* 2 - Random(rx)* 2) div 2);
  y := y + ((Random(ry)* 2 - Random(ry)* 2) div 2);
  if(Random(2) + 1 = 1)then
  begin
    ox:= x + Random(Round(Abs(cx - x))) div 2;
    oy:= cy + Random(Round(Abs(cy - y))) div 2;
  end else
  begin
    ox:= cx + Random(Round(Abs(cx - x))) div 2;
    oy:= y + Random(Round(Abs(cy - y))) div 2;
  end;
  repeat
    incr := ((Random(2) + 1) / 1000) + Random(11) / 10000;
    mu:= mu + incr;
    mu2 := mu * mu;
    mum1 := 1 - mu;
    mum12 := mum1 * mum1;
    mx := Trunc(cx * mum12 + 2 * ox * mum1 * mu + x * mu2);
    my := Trunc(cy * mum12 + 2 * oy * mum1 * mu + y * mu2);
    MoveMouse(mx, my);
    ms := ms + 1;
    if(ms mod 45 = 0)then
      //Wait(3+Random(2));
      Wait((Round(Random(2) + MouseSpeed) / 3));
  until mu >= 1;
  MoveMouseSmooth(x, y);
end;

procedure RRMMouse(X, Y, Rx, Ry: Integer); // BenMouse, Modded by RealRune
Var
 A : TPointArray;
 B, C, D, Sx, Sy : Integer;
begin
  GetMousePos(Sx, Sy);
  C := 0;
  D := 0;
  if(X > Sx)then
    C := -10 - Random(21)
     Else
    C := 10 + Random(21);
  if(Y > Sy)then
    D := -10 - Random(21)
     Else
    D := 10 + Random(21);
  A := MakeMouseSplinePath(Sx, Sy, X + C, Y + D, 7, 50, 70);
   For B := 0 to GetArrayLength(A) - 1 Do
    begin
    MoveMouse(A[B].X, A[B].Y);
    Wait((Round(Random(2) + MouseSpeed) / 10));
    if(Distance(Sx, Sy, x, y) < 50)then
     Wait(1);
    if(Random(250) = 1)then
     begin
      Case Random(6) Of
      0 : IdleTime(100, 100, 1.0);
      1 : IdleTime(100, 100, 1.0);
      2 : IdleTime(100, 100, 1.0);
      3 : IdleTime(150, 150, 1.0);
      4 : IdleTime(200, 200, 1.0);
      5 : IdleTime(500, 500, 1.0);
      end;
     Break;
     end;
    end;
  Wait((Round(Random(2) + MouseSpeed) / 10));
  SiMMouse(x + random(rx), y + random(ry), 0, 0);
  if(Random(20) = 1)then
   begin
   Wait((Round(Random(2) + MouseSpeed) / 10));
   IdleTime(100, 100, 1.0);
   SiMMouse(x + random(rx), y + random(ry), 0, 0);
   end;
  Wait(30 + Random(30));
end;

Var
 CX, CY : Integer;

begin
  MouseSpeed := 40;
  GetClientDimensions(CX, CY);
   repeat
   RRMMouse(1 + Random(CX), 1 + Random(CY), 0, 0);
   until(False);
end.

 A : TPointArray;
 B, C, D, Sx, Sy, Cx, Cy : Integer;
begin
  if Not UseRRMouse then
  begin
    MMouse(x, y, rx, ry);
    exit;
  end;
  GetMousePos(Sx, Sy);
  C := 0;
  D := 0;
  if(X > Sx)then
    C := -10 - Random(21)

   end;

end;

procedure RRMouse(x, y, rx, ry : integer; left : boolean);  // Moves then Clicks.
var
  a, b, c : Integer;
begin
  RRMMouse(x, y, rx, ry);
  GetMousePos(b, c);
  HoldMouse(b + 1, c, left);
  repeat
    Wait(10 + Random(40));
    a := a + 1;
  until (a > 4);
  GetMousePos(b, c);
  ReleaseMouse(b, c, left);
  Wait(20 + Random(20));
end;

Var
 CX, CY : Integer;

begin
  MouseSpeed := 10;
  BenMouse := True;
  GetClientDimensions(CX, CY);
  repeat
    RRMMouse(1 + Random(CX), 1 + Random(CY), 0, 0);
  until(False);
end.
