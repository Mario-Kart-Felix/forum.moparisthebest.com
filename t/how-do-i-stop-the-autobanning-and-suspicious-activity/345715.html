	// check ipbans -bakatool
	try {
		BufferedReader in = new BufferedReader(new FileReader("config//"
				+ file + ".txt"));
		String data = null;
		while ((data = in.readLine()) != null) {
			if (playerName.equalsIgnoreCase(data)) {
				return true;
			}
		}
	} catch (IOException e) {
		System.out.println("Critical error while checking for data!");
