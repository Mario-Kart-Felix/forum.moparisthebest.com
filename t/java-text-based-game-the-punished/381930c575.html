//save method
public void save() {

	//Make objects out of the nessesary classes
	mainMenu menu = new mainMenu();

	//Try to
	try {

		//Declare and initialize a new print writer with the file set to save.txt
		PrintWriter save = new PrintWriter(new FileWriter("save.txt"));
		
		//Loop through every string in the player inventory array list
		for(String str : Player.inventory) {
			
			//Print the string and a comma to the save.txt file
			save.print(str + ",");
			
		}
		
		//Print a new line in the file
		save.println("");
		
		//Print the gold coins to the file
		save.println(Player.goldCoins);

		//Print the gold coins to the file
		save.println(Player.hasContract);

		//Print the current company to the file
		save.println(Player.currentCompany);
		
		//Print the current ore to the file
		save.println(Player.currentOre);
		
		//Print the current ore count to the file
		save.println(Player.currentOreCount);
		
		//Print the current price to the file
		save.println(Player.currentPrice);
		
		//Print the reputation of the player to the file
		save.println(Player.reputation);
		
		//Print the durabilities for all pickaxes to the file
		save.println(Pickaxe.normalPickaxeDurability);
		save.println(Pickaxe.bronzePickaxeDurability);
		save.println(Pickaxe.steelPickaxeDurability);
		save.println(Pickaxe.diamondPickaxeDurability);
		
		//Close the print writer
		save.close();
		
		//Print out a new line
		System.out.println("");
		
		//Start the main menu
		menu.start();

	}

	//Catch the exception if the file can't be saved
	catch (Exception e) {

		//Print out that the file couldn't be saved
		System.out.println("");
		System.out.println("Error: The file couldn't be saved.");
		System.out.println("");
		
		//Start the main menu
		menu.start();


	}

import java.io.;

//Load class
public class Load {

//Make objects out of the classes needed
Player player = new Player();

//Declare variables
String[] elements;
int goldCoins;
boolean hasContract;
String currentCompany;

int currentOreCount;
int currentPrice;
int reputation;
int pD;
int bPD;
int sPD;
int dPD;

//Declare the boolean is loaded
boolean isLoaded() {
	
	//Loop through the strings in the element array
    for (int i=0; i < elements.length; i++) {

		//Loop through the items in the players inventory
        for (int j=0; j < Player.inventory.size(); j++) {
	
			//If the the number that i refers to in elements is the same as the item in the players inventory j
            if(elements[i].equals(Player.inventory.get(j))){
				
				//return true
                return true;
				
			}

        }

    }

	//If it doesn't return false
    return false;
		
}

//load method
public void load() {

	//Try to
	try {
		
		//Declare a new file (save.txt) and a scanner (save.txt)
		File file = new File("save.txt");
		Scanner savedInv = new Scanner(file);
		
		//Elements = the strings in the file separated by a ","
		elements = savedInv.nextLine().split(",");
	
		//goldCoins = integer on the seond line
		goldCoins = Integer.parseInt(savedInv.nextLine());
		
		//hasContract = boolean on the third line
		hasContract = Boolean.parseBoolean(savedInv.nextLine());
		
		//currentCompany = string on the fourth line
		currentCompany = savedInv.nextLine();
		
		//currentOre = string on the fifth line
		currentOre = savedInv.nextLine();
		
		//currentOreCount = integer on the sixth line
		currentOreCount = Integer.parseInt(savedInv.nextLine());
		
		//currentPrice = integer on the seventh line
		currentPrice = Integer.parseInt(savedInv.nextLine());

			menu.start();
	   
	    //If is loaded returns false
		} else {
	
			//Loop through the strings in the array elements
			for (String str : elements) {
		
				//Add the item to the inventory
				Player.inventory.add(str);
				
			}
			
			//Set the values of the variables to the variables in the player class for use
			Player.goldCoins = goldCoins;

			Player.currentOre = currentOre;

			Pickaxe.diamondPickaxeDurability = dPD;
		
		}
		
		//Print out everything loaded
