})

public static void transform(String fileName, Hashtable<String, byte[]> cache, boolean verbose) throws IOException {
	List<Entry<String, byte[]>> entries = new LinkedList<Entry<String,byte[]>>();
	entries.addAll(cache.entrySet());
	ClassGen[] classGens = new ClassGen[entries.size()];
	for(int i = 0; i < classGens.length; i++) {
		Entry<String, byte[]> entry = entries.get(i);
		classGens[i] = new ClassGen(new ClassParser(new ByteArrayInputStream(entry.getValue()), entry.getKey().replace('.', '/') + ".class").parse());
	}
	TransformerConfiguration[] transformerConfigurations = new TransformerConfiguration[0];
	try {
		transformerConfigurations = TransformerConfiguration.loadFromXML(fileName);
	} catch(Throwable t) {
		t.printStackTrace();
	}
	int errorCount = 0;
	for(TransformerConfiguration transformerConfiguration : transformerConfigurations) {
		try {
			transformerConfiguration.transform(classGens);
		} catch(Throwable t) {
			t.printStackTrace();
			errorCount++;
		}
	}
	if(verbose)
		Logger.log(Priority.NORMAL, "finished " + transformerConfigurations.length + " transformations (with " + (errorCount == 0 ? "no" : errorCount) + " error" + (errorCount == 1 ? "" : "s") + ")");
}
