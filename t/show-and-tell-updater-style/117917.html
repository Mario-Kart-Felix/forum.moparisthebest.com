Well, judging by the discussion in the client hacking forums, it looks as if many people are working on updaters. So, I (kinda…) thought, why not make a post where we can show off our updater’s design! Rather than post the whole updater, just post your StringWrapper transformer, as it is probably the easiest to do.

The aim of this post is for people to share their updater’s design and to gain ideas in how to better it. Remember, speed isn’t everything.

I’ll start it off:

[code]package franks.bot.util.transformer;

import com.sun.org.apache.bcel.internal.generic.ClassGen;
import com.sun.org.apache.bcel.internal.generic.Type;
import com.sun.org.apache.bcel.internal.generic.ArrayType;
import com.sun.org.apache.bcel.internal.classfile.Field;
import franks.bot.hooks.MutableString;

public final class MutableStringT extends Transformer {
public MutableStringT() {
super(MutableString.class);
}

public boolean identify(ClassGen cg) {
    return cg.getInterfaces().length == 1
            && cg.containsMethod("toString", "()Ljava/lang/String;") != null
            && cg.containsMethod("hashCode", "()I") != null;
}

protected void addAccessors() {
    Type byteArray = new ArrayType(Type.BYTE, 1);
    ClassGen cg = getClassGen();
    for (Field f : cg.getFields()) {
        if (f.isStatic()) {
            continue;
        }
        if (f.getType().equals(Type.BOOLEAN)) {
            addField("isMutable", f);
        } else if (f.getType().equals(Type.INT) && !f.isPrivate()) {
            addField("getLength", f);
        } else if (f.getType().equals(byteArray)) {
            addField("getBytes", f);
        } else if (f.getType().equals(Type.INT) && f.isPrivate()) {
            addField("getHash", f);
        }
    }
}

[code]package update;

import org.apache.bcel.generic.;
import java.util.;
import org.apache.bcel.classfile.*;
import org.apache.bcel.Constants;

public class StringWrapper extends Hook {

public StringWrapper(HashMap<String, ClassGen> m, Properties props) {
    super(m, props);
}

public boolean run() throws Exception {
    for (ClassGen cg : getMap().values()) {
        for (Method m : cg.getMethods()) {
            if (m.getName().equals("toString") 
                    && cg.getInterfaceNames().length > 0) {
                set("StringWrapper", cg.getClassName());
                addInterface(cg.getClassName(),
                        "info.scanningyour.ares.hooks.StringWrapper");
                break;
            }
        }
        if (get("StringWrapper") != null
                && get("StringWrapper").equals(cg.getClassName())) {
            for (Field f : cg.getFields()) {
                if (!f.isStatic()
                        && f.getType().equals(new ArrayType(Type.BYTE, 1))) {
                    set("StringWrapperBytes",
                            (cg.getClassName() + "." + f.getName()));
                    addMethodToReturnField(cg, "getBytes", f.getType(),
                            cg.getClassName(), f.getName(), f.getType(),
                            Constants.GETFIELD);
                    return true;
                }
            }
        }
    }
    return false;
}

}
