I’ve fixed it up to pretty much what you got but so it works(as far as I know, haven’t tested), however you should download SRL and take a look at that. ClickMouse is so detectable in RS2.

procedure RightClicking; var xs, ys, xe, ye : Integer; begin xs := 3; ys := 60; xe := 508; ye := 351; if (FindColor(x, y, orecolor, xs, ys, xe, ye)) then begin WriteLn('Found orecolor'); Wait(delay + Random(50)); ClickMouse(x, y, False); end else if (FindColor(x, y, orecolor2, xs, ys, xe, ye)) then begin WriteLn('Found orecolor2'); Wait(delay + Random(50)); ClickMouse(x, y, False); end else if (FindColor(x, y, orecolor3, xs, ys, xe, ye)) then begin WriteLn('Found orecolor3'); Wait(delay + Random(50)); ClickMouse(x, y, False); end; end;
