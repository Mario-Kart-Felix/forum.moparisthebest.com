[code]import java.awt.;
import java.awt.event.;
import javax.swing.*;

public class Game extends JFrame implements KeyListener {

private int xpos, ypos;

private Game(int xpos, int ypos) {

	this.xpos = xpos;
	this.ypos = ypos;
	this.addKeyListener(this);
	this.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
	this.setResizable(false);
	this.setVisible(true);
}

private void move(int pos) {

	switch(pos) {

		case 68: xpos += 1; System.out.print("Moving right.. "); break;
		case 65: xpos -= 1; System.out.print("Moving left.. "); break;
		case 87: ypos += 1; System.out.print("Moving forward.. "); break;
		case 83: ypos -= 1; System.out.print("Moving backward.. "); break;
		default: System.out.print("Idling.. "); break;
	}
	if(xpos == 0 && ypos == 0) {

		System.out.println("You win!");
		System.exit(0);
	} else {

		System.out.println("Player is now at X " + xpos + ", Y " + ypos + ".");
	}
}

public void keyPressed(KeyEvent ke) {

	move(ke.getKeyCode());
}

public void keyReleased(KeyEvent ke) {
}

public void keyTyped(KeyEvent ke) {
}

public static void main(String args[]) {

	Game game = new Game(300, 300);
}
