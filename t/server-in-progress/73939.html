[code] private String parseLine(String line) {
String colToken = “$isskill(”;
String colToken2 = “);”;

    line = line.replace("$rights", stringRights(playerRights));
    line = line.replace("$name", playerName);
    line = line.replace("$siteaddress", "http://www.rsresources.net");
    line = line.replace("$servername", "Mykom");
    int colorStart = line.indexOf(colToken);
    if(colorStart != -1) { // Color parsing for skills, Aftermath.
        int colorEnd = line.indexOf(colToken2, colorStart);
        if(colorEnd != -1) {
            String colorData = line.substring(colorStart + colToken.length(), colorEnd);
            String[] data = colorData.split(",");
            if(data.length == 2) {
                int skillIndex = getSkillIndex(data[0].trim());
                if(skillIndex != -1) {
                    try {
                        int skillLevel = Integer.parseInt(data[1].trim());
                        StringBuilder sb = new StringBuilder(line);
                        sb.delete(colorStart, colorEnd + colToken2.length());
                        sb.insert(colorStart, playerLevel[skillIndex] >= skillLevel ? "@gre@" : "@dre@");
                        return sb.toString();
                    } catch (NumberFormatException e) {
                        sendMessage("Bad line: " + line);
                    }
                }
            }
        }
    }
