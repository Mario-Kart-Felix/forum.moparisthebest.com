program Age;
var s,age,xtra,xq,q: Integer;
var bleap: Boolean;
begin
age := 14; //Change to your age in years.
s := 60 * 60 * 24 * 365 * age
xtra := age DIV 4
xq := xtra * 86400
q := s + xq
bleap := false; // If this is true, it’ll do leap years. If false, it’ll tell your age in seconds without leap years.
If (age >= 4) and (bleap = true) Then
writeln(‘You are ’ + inttostr(q) + ’ seconds old counting leap year(s)!’)
else
writeln(‘You are ’ + inttostr(s) + ’ seconds old, not counting leap years!’);
end.
