OK you just have to set the 2 colors u want and it should click them

[scar]program clicker;
{.include Osi.txt}
{.include par.txt}

const
ballspot=0; // set colors here
ballcolor=10862030; // set colors here

procedure clickball;
var
x2,y2: integer;
begin
WAIT(35+RANDOM(29))
if(findcolorspiral2(x2,y2,ballspot,1,1,511,413))or
(findcolorspiral2(x2,y2,ballcolor,1,1,511,413))then
mouse(x2,y2,3,3, true);
end;

begin
activateClient;
wait(2000)
repeat
clickball;

until(false)
end.[/scar]
