this is the code i use to hook the camera fields.
incase your interested:

package com.genesis.updater.transformer;

import com.genesis.updater.util.InstructionSearcher;
import org.apache.bcel.Constants;
import org.apache.bcel.classfile.Method;
import org.apache.bcel.generic.*;

/**
 * com.genesis.updater.transformer
 *
 * @author Oliver
 */
public class CameraT extends Transformer {

    public CameraT(ClassGen[] classes) {
        super(classes);
    }

    public void transform(String argument, boolean verbose) throws Throwable {
        for (final ClassGen classGen : classes) {
            final ConstantPoolGen constantPool = classGen.getConstantPool();
            if (constantPool.lookupMethodref("java.lang.Math", "atan2", "(DD)D") > 0 && (constantPool.lookupDouble(325.949) > 0 || constantPool.lookupDouble(-325.949) > 0)) {
                for (final Method method : getStaticMethods(classGen)) {
                    int found = 0;
                    if (method.isFinal() && (method.getReturnType() == Type.BOOLEAN)) {
                        InstructionSearcher.Constraint constraint = new InstructionSearcher.Constraint() {
                            public boolean matches(InstructionHandle i) {
                                return (i.getInstruction() instanceof InvokeInstruction)
                                        && ((InvokeInstruction) i.getInstruction()).getMethodName(constantPool).equals("atan2");
                            }
                        };
                        InstructionSearcher searcher = new InstructionSearcher(method, InstructionSearcher.MODE_START);
                        if (searcher.nextInstruction(InvokeInstruction.class, constraint)) {
                            String names[] = {"getCameraAngleY", "getCameraAngleX"};
                            for (String name : names) {
                                searcher.nextInstruction(PUTSTATIC.class);
                                addGetMutator("client", (FieldInstruction) searcher.getNext(), constantPool, name, "I", verbose);
                                found++;
                                if (found == 1)
                                    searcher.nextInstruction(InvokeInstruction.class, constraint);
                            }
                            break;
                        }
                    }
                }
            }
            for (final Method method : getStaticMethods(classGen)) {
                if ((method.getModifiers() == (Constants.ACC_STATIC | Constants.ACC_FINAL)) && (method.getReturnType() == Type.INT) && method.getArgumentTypes().length == 1) {
                    InstructionSearcher.Constraint constraint = new InstructionSearcher.Constraint() {
                        public boolean matches(InstructionHandle i) {
                            return (i.getInstruction() instanceof FieldInstruction)
                                    && ((FieldInstruction) i.getInstruction()).getFieldType(constantPool).equals(Type.getType(byte[][][].class));
                        }
                    };
                    InstructionSearcher searcher = new InstructionSearcher(method, InstructionSearcher.MODE_START);
                    if (new InstructionList(method.getCode().getCode()).getInstructions().length > 100)
                        continue;
                    if (searcher.nextInstruction(GETSTATIC.class, constraint)) {
                        searcher.getPrev();
                        searcher.prevInstruction(GETSTATIC.class);
                        addGetMutator("client", (FieldInstruction) searcher.getPrev(), constantPool, "getCameraZ", "I", verbose);
                        searcher.nextInstruction(GETSTATIC.class, constraint);
                        addGetMutator("client", (FieldInstruction) searcher.getNext(), constantPool, "getTerrainSpecial", "[[[B", verbose);
                        searcher.nextInstruction(GETSTATIC.class);
                        FieldInstruction groundLevel = (FieldInstruction) searcher.getNext();
                        putToMnemonic("getGroundLevel", (groundLevel).getClassName(constantPool) + "." + (groundLevel).getFieldName(constantPool));
                        addGetMutator("client", groundLevel, constantPool, "getGroundLevel", "I", verbose);
                        searcher.nextInstruction(GETSTATIC.class);
                        addGetMutator("client", (FieldInstruction) searcher.getNext(), constantPool, "getCameraX", "I", verbose);
                        searcher.nextInstruction(GETSTATIC.class);
                        addGetMutator("client", (FieldInstruction) searcher.getNext(), constantPool, "getCameraY", "I", verbose);
                        break;

                    }
                }
            }
        }
    }
}
