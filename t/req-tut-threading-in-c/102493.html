#define WIN32_LEAN_AND_MEAN
#define STRICT
#include <stdio.h>
#include <windows.h>

DWORD WINAPI ThreadProc(PVOID param){
printf(“This message was printed from a different thread”);
return 0xA43D; /* Made up gibberish return value. */
}

DWORD WINAPI WinMain(HINSTANCE curProc, HINSTANCE nulProc, LPSTR cmdText, int iForgetWhatThisIs){
DWORD dwTid;
HANDLE hThread = CreateThread(0, NULL, ThreadProc, 0, 0, &dwTid);
WaitForSingleObject(hThread, INFINITE); /* To keep from exiting before it finishes */
CloseHandle(hThread);
return 0;
