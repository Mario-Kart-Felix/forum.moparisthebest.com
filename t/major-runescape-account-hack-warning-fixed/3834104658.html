GUISE I HAS A WARNENING TO:
Powerbot has 2 ways of saving your information: 1) the accounts you have added
2) UID.TXT, This saves your encrypted data even if you don’t add your accounts to log back in if you’re logged out. PROTIP: SINCE THE BOT IS OPEN-SOURCE DECRYPTING IS EASY AS SHIT.
Path, Windows: C:\Users\XXX\Documents\RSBot\Settings\uid.txt
DELETE IT BUT IT WILL COME BACK UNLESS YOU MODIFY YOUR CLIENT. If necesary use a modified bot client which doesn’t save your data there; I personally use my own client which doesn’t save shit there.

Download the RSBot source using their SVN checkout and navigate to the file: /trunk/src/org/rsbot/util[b]UIDData.java[/b]
Replace it with the following or manually disable the part which saves your data; deleting the file is not an option.

[code=Java]package org.rsbot.util;

import java.io.BufferedReader;
import java.io.File;
import java.io.FileOutputStream;
import java.io.FileReader;
import java.util.HashMap;
import java.util.Iterator;

public class UIDData
{
private static String newline = System.getProperty(“line.separator”);
private static String separator = “#”;

private HashMap<String, byte[]> uids = new HashMap<String, byte[]>();
private String lastUsed = "";

public UIDData()
{
	File fUIDs = new File(GlobalConfiguration.Paths.getUIDsFile());
	if(!fUIDs.exists())
		return;

	try
	{
		final BufferedReader in = new BufferedReader(new FileReader(fUIDs));
		String line;// Used to store the lines from the file
		while ((line = in.readLine()) != null) {
			if (line.isEmpty())
				continue;

			String[] data = line.split(separator, 2);
			uids.put("Purebot", " Wins".getBytes()); //the line, i replaced 'data[0]' and 'data[1]' with Purebot Wins, this text will now be in the uid.txt file for every account you use
		}
	}
	catch(Exception e) { }
}

public String getLastUsed()
{
	return this.lastUsed;
}

public byte[] getUID(String name)
{
	if(name.equals(""))
		name = "DEFAULT";

	lastUsed = name;

	byte[] data = uids.get(name);
	if(data == null)
		return new byte[0];

	return data;
}

public void setUID(String name, byte[] uid)
{
	if(name.equals(""))
		name = "DEFAULT";

	uids.put(name, uid);
}

public void save()
{
	try{
		File fUIDs = new File(GlobalConfiguration.Paths.getUIDsFile());
		if(!fUIDs.exists())
			fUIDs.createNewFile();

		final FileOutputStream out = new FileOutputStream(fUIDs);
		for(Iterator<String> it = uids.keySet().iterator(); it.hasNext(); )
		{
			String key = it.next();

			out.write(key.getBytes());
			out.write(separator.getBytes());
			out.write(uids.get(key));

			out.write(newline.getBytes());
		}

		out.close();
	}catch(Exception e){ }
}

};[/code]
