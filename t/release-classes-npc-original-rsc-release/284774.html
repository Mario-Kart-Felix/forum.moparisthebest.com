import org.rscdaemon.server.model.*;

public class ProfessorOddenstein implements NpcHandler {
/**
* World instance
*/
public static final World world = World.getWorld();

public void handleNpc(final Npc npc, Player player) throws Exception {
  		player.informOfNpcMessage(new ChatMessage(npc, "WRITE YOU CHAT MESSAGE HERE", player));
  		player.informOfNpcMessage(new ChatMessage(npc, "YTHIS IS WHAT YOU WANT THE NPC TO SAY IN THE BEGINNING", player));
  		player.informOfNpcMessage(new ChatMessage(npc, "YOU ARE EDITING THIS TEXT FOR THE BEGINNING CHAT MESSAGE", player));
  		player.informOfNpcMessage(new ChatMessage(npc, "YOUR WELCOME FOR THE RELEASE", player));
  		player.setBusy(true);
		world.getDelayedEventHandler().add(new ShortEvent(player) {
		public void action() {
		owner.setBusy(false);
			String[] options = new String[]{"I wish to become a mighty warrior", "I would like to become a powerful magician", "I want to be an Adventurer", "Make me a Ranger"};//NPC Chat options, you can make more if you want
			owner.setMenuHandler(new MenuHandler(options) {
			public void handleReply(final int option, final String reply) {
			if(owner.isBusy()) {
						return;
					}
					owner.informOfChatMessage(new ChatMessage(owner, reply, npc));
					owner.setBusy(true);
					world.getDelayedEventHandler().add(new ShortEvent(owner) {
						public void action() {
							owner.setBusy(false);
							if(option == 0) {
									owner.informOfNpcMessage(new ChatMessage(npc, "Ahh wise choice! You are now a Mighty Warrior!", owner));
									owner.getInventory().add(new InvItem(10, 1000)); //Add Items Here
									owner.getInventory().add(new InvItem(1263, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(67, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(17, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(16, 1)); //Add Items Here
									owner.getActionSender().sendInventory();	
									owner.setCurStat(0, 7);//(Stat Number,Stat Level)
									owner.setMaxStat(0, 7);//(Stat Number,Stat Level)
									owner.setExp(0, Formulae.lvlToXp(7));//(Stat Number,Stat Level)
									owner.setCurStat(2, 9);//(Stat Number,Stat Level)
									owner.setMaxStat(2, 9);//(Stat Number,Stat Level)
									owner.setExp(2, Formulae.lvlToXp(9));//(Stat Number,Stat Level)
									owner.setCurStat(3, 12);//(Stat Number,Stat Level)
									owner.setMaxStat(3, 12);//(Stat Number,Stat Level)
									owner.setExp(3, Formulae.lvlToXp(12));//(Stat Number,Stat Level)
									owner.getActionSender().sendMessage("Good luck on your journey!");
									owner.teleport(123, 647, false);
									npc.unblock();						
							}
							if(option == 1) {
									owner.informOfNpcMessage(new ChatMessage(npc, "Ahh wise choice! You are now a Powerful Magician!", owner));
									owner.getInventory().add(new InvItem(10, 1000)); //Add Items Here
									owner.getInventory().add(new InvItem(1263, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(31, 25)); //Add Items Here
									owner.getInventory().add(new InvItem(32, 25)); //Add Items Here
									owner.getInventory().add(new InvItem(33, 25)); //Add Items Here
									owner.getInventory().add(new InvItem(35, 50)); //Add Items Here
									owner.getInventory().add(new InvItem(100, 1)); //Add Items Here
									owner.getActionSender().sendInventory();	
									owner.setCurStat(6, 11);//(Stat Number,Stat Level)
									owner.setMaxStat(6, 11);//(Stat Number,Stat Level)
									owner.setExp(6, Formulae.lvlToXp(11));//(Stat Number,Stat Level)
									owner.getActionSender().sendMessage("Good luck on your journey!");
									owner.teleport(123, 647, false);
									npc.unblock();						
							}
							if(option == 2) {
									owner.informOfNpcMessage(new ChatMessage(npc, "Ahh wise choice! You are now a Skilled Adventurer!", owner));
									owner.getInventory().add(new InvItem(10, 1000)); //Add Items Here
									owner.getInventory().add(new InvItem(1263, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(87, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(166, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(17, 1)); //Add Items Here
									owner.getActionSender().sendInventory();	
									owner.setCurStat(8, 7);//(Stat Number,Stat Level)
									owner.setMaxStat(8, 7);//(Stat Number,Stat Level)
									owner.setExp(8, Formulae.lvlToXp(7));//(Stat Number,Stat Level)
									owner.getActionSender().sendMessage("Good luck on your journey!");
									owner.teleport(123, 647, false);
									npc.unblock();						
							}
							if(option == 3) {
									owner.informOfNpcMessage(new ChatMessage(npc, "Ahh wise choice! You are now a Sharp-Shooting Archer!", owner));
									owner.getInventory().add(new InvItem(10, 1000)); //Add Items Here
									owner.getInventory().add(new InvItem(1263, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(11, 50)); //Add Items Here
									owner.getInventory().add(new InvItem(15, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(16, 1)); //Add Items Here
									owner.getInventory().add(new InvItem(189, 50)); //Add Items Here
									owner.getActionSender().sendInventory();	
									owner.setCurStat(4, 15);//(Stat Number,Stat Level)
									owner.setMaxStat(4, 15);//(Stat Number,Stat Level)
									owner.setExp(4, Formulae.lvlToXp(15));//(Stat Number,Stat Level)

									npc.unblock();						
													
							}
						}
					});
				}
			});
			owner.getActionSender().sendMenu(options);
			}
	});
		npc.blockedBy(player);
