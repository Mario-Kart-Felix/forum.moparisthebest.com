// Smack The Rabbit v.1 by Ron
// Game http://forums.ngemu.com/arcade.php?do=play&gameid=434
// Put the target on the screen. Start the game.
// The script will automatically start the game for you.
// F12 to quit the script.

Program SmackTheRabbitByRon;

Const
  // Five different colors of the Rabbits.
  Color1 = 16777215;
  Color2 = 16119285;
  Color3 = 12373180;
  Color4 = 13619151;
  Color5 = 14013909;

Var x, y : Integer;

// Find the color procedure. Checks for 5 different color at a tolerance of
// 5 and clicks on the spot it finds the color on. This also finds colors
// of the dead bodies. Which is hard to avoid.
Procedure FindTheColor(x1, y1, x2, y2 : Integer);
Begin
  If(FindColorTolerance(x, y, Color1, x1, y1, x2, y2, 5))Then
  Begin
    ClickMouse(x, y, True);
  End Else
  If(FindColorTolerance(x, y, Color2, x1, y1, x2, y2, 5))Then
  Begin
    ClickMouse(x, y, True);
  End Else
  If(FindColorTolerance(x, y, Color3, x1, y1, x2, y2, 5))Then
  Begin
    ClickMouse(x, y, True);
  End
  If(FindColorTolerance(x, y, Color4, x1, y1, x2, y2, 5))Then
  Begin
    ClickMouse(x, y, True);
  End Else
  If(FindColorTolerance(x, y, Color5, x1, y1, x2, y2, 5))Then
  Begin
    ClickMouse(x, y, True);
  End;
End;

Begin
  // Clear report box.
  ClearReport;
  // Clear debug box.
  ClearDebug;
  // Activate the targetted client.
  ActivateClient;
  // Keeps repeating forever or until F12 is pressed.
  Repeat
    // I know I can make for statements for these, but I was too lazy.
    // Plus when I tried, it messed up bad. Whatever.
    
    // 6 Different Quadrants for Section 1
    FindTheColor(0, 0, 100, 100);
    FindTheColor(100, 0, 200, 100);
    FindTheColor(200, 0, 300, 100);
    FindTheColor(300, 0, 400, 100);
    FindTheColor(400, 0, 500, 100);
    FindTheColor(500, 0, 550, 100);
    // 6 Different Quadrants for Section 2
    FindTheColor(0, 100, 100, 200);
    FindTheColor(100, 100, 200, 200);
    FindTheColor(200, 100, 300, 200);
    FindTheColor(300, 100, 400, 200);
    FindTheColor(400, 100, 500, 200);
    FindTheColor(500, 100, 550, 200);
    // 6 Different Quadrants for Section 3
    FindTheColor(0, 200, 100, 300);
    FindTheColor(100, 200, 200, 300);
    FindTheColor(200, 200, 300, 300);
    FindTheColor(300, 200, 400, 300);
    FindTheColor(400, 200, 500, 300);
    FindTheColor(500, 200, 550, 300);
    // 6 Different Quadrants for Section 4
    FindTheColor(0, 300, 100, 400);
    FindTheColor(100, 300, 200, 400);
    FindTheColor(200, 300, 300, 400);
    FindTheColor(300, 300, 400, 400);
    FindTheColor(400, 300, 500, 400);
    FindTheColor(500, 300, 550, 400);
  Until(False)or(IsFKeyDown(12))
End.
