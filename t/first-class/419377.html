public static void make(final Player player, final int amount, final int slot) {
	player.getActionSender().sendCloseInterface();
	final int arrayIndex = (Integer)player.getTemporary("FLETCHING_INDEX");
	if (player.getSkills().getLevel(Skills.FLETCHING) < REQUIRED_LEVELS[arrayIndex][slot]) {
		player.getActionSender().sendMessage("You need a Fletching level of " + REQUIRED_LEVELS[arrayIndex][slot] + " to make a "+ ItemDefinition.forId(STRUNG_BOWS[arrayIndex][slot][0]).getName().toLowerCase() + ".");
		return;
	} else if (player.getTemporary("IS_FLETCHING") != null) {
		((ActionEvent)player.getTemporary("ACTION_EVENT")).stop();
		return;
	}
	player.addTemporary("IS_FLETCHING", true);
	player.playAnimation(Animation.create(1248));
	//save memory here by doing it via the player class (player.getInventory()) and then doing Inventory.refresh() at the very end?
	player.getInventory().remove(new Item(LOGS[arrayIndex], 1));
	player.getSkills().addExperience(Skills.FLETCHING, RECIEVED_UNSTRUNG_EXPERIENCE[arrayIndex][slot]);
	//save memory here by doing it via the player class (player.getInventory()) and then doing Inventory.refresh() at the very end?
	if (arrayIndex == 0) {
		player.getInventory().add(new Item(UNSTRUNG_BOWS[arrayIndex][slot], 15));
	} else {
		player.getInventory().add(new Item(UNSTRUNG_BOWS[arrayIndex][slot], 1));	
	}
	player.getActionSender().sendMessage("You carefully cut the wood into a " + ItemDefinition.forId(UNSTRUNG_BOWS[arrayIndex][slot]).getName().toLowerCase() + ".");
	ActionEvent event = new ActionEvent() {
		
		int amt = amount-1;
		private Event event;

		@Override
		public void run() {
			event = new Event(1200) {
				public void execute() {
					if(!player.getInventory().contains((LOGS[arrayIndex])) || amt < 1) {
						stop();
						player.playAnimation(Animation.create(-1));
						player.removeTemporary("IS_FLETCHING");
						return;
					}
					player.playAnimation(Animation.create(1248));
					//save memory here by doing it via the player class (player.getInventory()) and then doing Inventory.refresh() at the very end?
					player.getInventory().remove(new Item(LOGS[arrayIndex], 1));
					player.getSkills().addExperience(Skills.FLETCHING, RECIEVED_UNSTRUNG_EXPERIENCE[arrayIndex][slot]);
					//save memory here by doing it via the player class (player.getInventory()) and then doing Inventory.refresh() at the very end?
					if (arrayIndex == 0) {
						player.getInventory().add(new Item(UNSTRUNG_BOWS[arrayIndex][slot], 15));
					} else {
						player.getInventory().add(new Item(UNSTRUNG_BOWS[arrayIndex][slot], 1));	
					}
					player.getActionSender().sendMessage("You carefully cut the wood into a " + ItemDefinition.forId(UNSTRUNG_BOWS[arrayIndex][slot]).getName().toLowerCase().replaceAll(" (u)", ".") + ".");
					System.out.println("amt = "+amt);
					amt--;
				}
			};
			World.getWorld().submit(event);
		}

		@Override
		public void stop() {
			player.playAnimation(Animation.create(-1));
			player.removeTemporary("IS_FLETCHING");
			return;				
		}
	};
	player.addTemporary("ACTION_EVENT", event);
	event.run();
}
