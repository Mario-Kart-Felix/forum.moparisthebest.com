Weibs July 30, 2016, 1:44am #2

You could try something like this:

[php]procedure SpawnMobs;
begin
AddToReport(‘Walking around to find monsters’);
temp := 0;
repeat
KeyDown(VK_UP);
KeyDown(VK_LEFT);
Wait(1000);
KeyUp(VK_UP);
KeyUp(VK_Left);
Wait(50);
temp := temp + 1;
until(temp = 10);
temp := 0;
repeat
KeyDown(VK_DOWN);
KeyDown(VK_RIGHT);
Wait(1000);
KeyUp(VK_DOWN);
KeyUp(VK_RIGHT);
Wait(50);
temp := temp + 1;
until(temp = 10);
end;[/php]
