latifundio August 2, 2016, 3:29pm #4

Did you compiled from last revision? i think it is r53.

Also did you got a folder named “xml” with the GameObjectLoc.xml.gz file inside?

Anyway here is a build file so you can compile with ant.

[code]

<property name="src" location="src" />
<property name="build" location="build" />
<property name="lib" location="lib" />
<property name="jar" location="editor.jar" />

<target name="compile">
	<delete file="${jar}" />
	<mkdir dir="${build}" />
	<javac srcdir="${src}" destdir="${build}" debug="on" target="1.5">
	  <classpath>
		<pathelement location="${lib}/xpp3.jar" />
		<pathelement location="${lib}/xstream.jar"/>
		<pathelement path="${java.class.path}/"/>
	  </classpath>
	</javac>
	<jar basedir="${build}" destfile="${jar}" />
	<delete dir="${build}" />
</target>

<target name="run">
	<java classname="com.Gui" fork="true">
		<classpath>
			<pathelement location="${lib}/xpp3.jar" />
			<pathelement location="${lib}/xstream.jar" />
			<pathelement path="${jar}/" />
		</classpath>
	</java>
</target>

[/code]
