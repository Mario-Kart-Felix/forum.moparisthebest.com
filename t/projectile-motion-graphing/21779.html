package king;

import java.awt.Color;
import java.awt.Dimension;
import java.awt.Graphics;
import java.awt.Graphics2D;
import java.awt.Image;
import java.awt.Panel;
import java.awt.Point;
import java.awt.RenderingHints;
import java.awt.event.MouseEvent;
import java.awt.event.MouseListener;
import java.awt.event.MouseMotionListener;
import java.util.ArrayList;

public class MotionPanel extends Panel implements MouseListener, MouseMotionListener {
    private static final double ACC_GRAVITY = 9.8;
	private static final int STARTX = 10, STARTY = 490;
	
	private ArrayList<Point> points = new ArrayList<Point>();
	private double theta;
	private int relativeX = 110, relativeY = 390;
	private int calculationX, calculationY;
	private boolean isRunning = false;
	
	
	public MotionPanel() {
		setPreferredSize(new Dimension(500, 500));
		addMouseListener(this);
		addMouseMotionListener(this);
	}
	
	public void update(Graphics g) {
		paint(g);
	}

	public void paint(Graphics g) {
		Image offscreen = createImage(getWidth(), getHeight());
		Graphics2D buffer = (Graphics2D)offscreen.getGraphics();
		buffer.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
		
		buffer.setColor(Color.RED);
		buffer.drawLine(relativeX, STARTY, relativeX, relativeY);
		buffer.drawLine(STARTX, STARTY, relativeX, STARTY);
		
		buffer.setColor(Color.BLACK);
		buffer.drawString("X: " + calculationX, 10, 10);
		buffer.drawString("Y: " + calculationY, 10, 20);
		buffer.drawString(Math.round(theta) + "¬∫", STARTX + 20, STARTY - 5);
		
		buffer.drawLine(STARTX, STARTY, relativeX, relativeY);
		for(Point p : points)
			buffer.fillOval(p.x - 5, p.y - 5, 10, 10);
		
		g.drawImage(offscreen, 0, 0, this);
	}

	public void mouseClicked(MouseEvent me) {
		if(isRunning)
			return;
		
		new Thread(new Runnable() {
			public void run() {
				int time = 0;
				points.clear();
				isRunning = true;
				
				final double vi = Math.hypot(calculationX, calculationY) / 5;
				final double vxi = vi * Math.cos(Math.toRadians(theta));
				final double vyi = vi * Math.sin(Math.toRadians(theta));
				System.out.println(vxi +"::"+ vyi);
				double x = 0D, y = 0D;
				
				while(x < 500 && y < 500 && x >= 0 && y >= 0) {
				    x = vxi * time;
				    y = (vyi * time) - (0.5D * ACC_GRAVITY * Math.pow(time, 2));
				    
					System.out.println(time + ": " + x + ", " + y);
					points.add(new Point((int)Math.round(x) + STARTX, STARTY - (int)Math.round(y)));
					repaint();
					
					try {
						Thread.sleep(250);
					} catch(InterruptedException e) { }
					time++;
				}
				
				repaint();
				isRunning = false;
			}
		}).start();
	}

	public void mouseEntered(MouseEvent me) {}
	public void mouseExited(MouseEvent me) {}
	public void mousePressed(MouseEvent me) {}
	public void mouseReleased(MouseEvent me) {}
	public void mouseDragged(MouseEvent me) {
		mouseMoved(me);
	}

	public void mouseMoved(MouseEvent me) {
		relativeX = me.getX();
		relativeY = me.getY();
		
		calculationX = relativeX - STARTX;
		calculationY = STARTY - relativeY;
		
		if(calculationX < 0 || calculationY < 0)
		    theta = 0;
		else
		    theta = Math.abs(Math.toDegrees(Math.atan2(calculationY, calculationX)));
		repaint();
	}
}
