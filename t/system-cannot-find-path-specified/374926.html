Sounds like it can’t find your Java JDK folder.

Try right clicking your compiler.batch and delete everything and replace it with this.

@echo off
Color 0a
:menu
title compiler
@echo Compiling.....
if exist "%programfiles%\Java" (call :compile "%programfiles%\Java\") else (goto error)
:compile
for /D %%x in ("%~1jdk*") do (set p="%%~x\bin\javac.exe") 
if defined p (%p% -cp . *.java)
if defined p (goto end)
:error
echo You do not have Java installed. Please download it at the site that is about to load.
"%programfiles%\InternetExplorer\iexplore.exe" http://java.sun.com/javase/downloads/index.jsp
:end
echo Compiling Finished.
pause
exit
