if you want it make if find and click on bitmaps use this.

[scar]
program New;
{.include SRL/SRL.scar}

procedure ClickBMPS;
begin
wait(300+random(32));
movemousesmooth(x,y);
wait(300+random(1589));
clickmouse(x,y,true);
end;

procedure findBMP;
begin
if (findBitmap(Xbmp, x, y))then
begin
wait(1000+random(1000));
clickBMPS;
end;
end;

begin
SetupSRL;
FindBMP;
end.
[/scar]
