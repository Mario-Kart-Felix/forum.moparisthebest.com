public int[] fixArray(int[] array){
int arrayPos = 0;
int[] newArray = new int[array.length];
for(int x = 0; x < array.length; x++){
if(array[x] != 0){
newArray[arrayPos] = array[x];
arrayPos++;
}
}
return newArray;
}

public void fixShop(Client o){

public void resetShop(Client o) {
	synchronized(c) {
