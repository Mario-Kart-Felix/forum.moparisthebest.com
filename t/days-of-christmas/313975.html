public class Bass {

    public static void main(String[] args) {
        int pintsToDo = 1;
        int loops = Integer.parseInt(args[0]);
        int speed = 0;
        try {
            for(int l = 1; l <= loops; l++) {
                System.out.println("On the "+num2word(""+l+"")+" day of Christmas my true love sent to me...");
                if(l == 1) {
                    Thread.sleep(500-speed);
                    System.out.println(" A pint of bass in a straight glass.");
                } else if (l > 1) {
                    for(int i = pintsToDo; i > 0; i--) {
                        Thread.sleep(500-speed);
                        if(i == 1) {
                            System.out.println(makeSpaces(i, pintsToDo)+"and a pint of bass in a straight glass.");
                        } else if(i > 1) {
                            System.out.println(makeSpaces(i, pintsToDo)+""+i+" pints of bass");
                        }
                    }
                }
                System.out.println();
                Thread.sleep(1500-speed);
                pintsToDo++;
                if((500-speed) > 30)
                    speed = speed + 15;
            }
        } catch(InterruptedException ie) {
            System.out.println("Interrupted while sleeping: " + ie);
        }
    }
    public static String num2word(String number) {
        String[] suffixIndexes = { "3", "0", "1", "2", "3", "3", "3", "3", "3", "3" };
        String[] suffixes = { "st", "nd", "rd", "th" };
        return number+suffixes[Integer.parseInt(suffixIndexes[Integer.parseInt(number.substring(number.length()-1, number.length()))])];
    }
    public static String makeSpaces(int pintNumber, int pintsToDo) {
        int spaceCount = pintsToDo-pintNumber;
        String spaces = " ";
        for(int i = 0; i < spaceCount; i++) {
            spaces = spaces + " ";
        }
        return spaces;
    }
}
