Ron July 30, 2016, 4:06am #2

You forgot to SetupSRL;

program New;

{.include srl/srl.scar}

const
   object = 'trout'; //3-5 letters of the name (in order) excluding drop//
   itemcolor = 10526892;
   
Begin
  SetupSRL;
  repeat
    if InvFull then
      DropItems(itemcolor, 10, 28, object)
  until(false)
end.

~Ron
