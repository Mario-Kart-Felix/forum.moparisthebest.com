Features:

  • All axes
  • Correct stumps, timers, dying and respawning
  • All basic trees with correct exp rates, logs, whatever. xD

You’ll need to make the trees have health, and in which are depleted when they are being cut. Currently they are randomly cut down.

public static void cutTree(final Player player, int tree) {

        final int treeType = data[0];

                player.getActionSender().sendMessage("You need a woodcutting level of " + lvlReq + " to cut this tree.");
                return;

            final Item item = new Item(logReturn, 1);
    	checkForAxe(player);

	    if(player.getInventory().getItemContainer().freeSlot() == -1) {

public static void makeArrows(final Player player, int used) {
for (int[] data : ARROW_DATA) {
    int startId = data[0];
    int usedId = data[1];
    int endId = data[2];
    int req = data[3];
    int exp = data[4];
    if (used == usedId) {
	Item item1 = new Item(usedId, 1);
	Item item2 = new Item(startId, 1);
	player.getActionSender().removeInterfaces();
	if (player.getSkill().getLevelForXP(player.getSkill().getExp()[9]) < req) {
	    player.getActionSender().sendMessage("You need a Fletching level of " + req + " to make this.");
	    return;
	}
	if (isFletching == true)
	    return;
	player.getActionSender().sendMessage("You attach the " + item1.getDefinition().getName().toLowerCase() + " to the " + item2.getDefinition().getName().toLowerCase() + ".");
	if (player.getInventory().getItemContainer().getCount(usedId) > 15
	    && player.getInventory().getItemContainer().getCount(startId) > 15) {
	    player.getSkill().addExp(9, exp*15);
	    player.getInventory().removeItem(new Item(startId, 15));
	    player.getInventory().removeItem(new Item(usedId, 15));
	    player.getInventory().addItem(new Item(endId, 15));
	} else {
	    int amnt = 0;
	    if (player.getInventory().getItemContainer().getCount(startId) 
		> player.getInventory().getItemContainer().getCount(usedId)) {
		amnt = player.getInventory().getItemContainer().getCount(usedId);
	    } else {
		amnt = player.getInventory().getItemContainer().getCount(startId);
	    }
	    player.getSkill().addExp(9, exp*amnt);
	    player.getInventory().removeItem(new Item(startId, amnt));
	    player.getInventory().removeItem(new Item(usedId, amnt));
	    player.getInventory().addItem(new Item(endId, amnt));
	}
    }
}
}
