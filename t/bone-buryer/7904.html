[SCAR]program BoneBuryer; // By Adrix

// Bones: First bank slot.
// Withdraw and bury.

// Includes:

{.include SRL\SRL.scar}

// Variables and Constants:

var MouseX: integer;
var MouseY: integer;

// Procedures:

// Withdraw all stuff in first bank slot, little wait time.
procedure WithdrawUN;
var rx, ry : Integer;
begin

rx := 82;
ry := 65;
if (BankScreen) then
begin

Mouse(rx, ry, 0, 0, False);
wait(250 + random(250))
ChooseOption(rx, ry, 'Withdraw All');

end;
end;

// Find randoms:
procedure FindRandoms;
begin

FindNormalRandoms;

if(FindFight) then
begin

RunAwayDirection(‘N’);
wait(8000 + random(7000));
RunBack;

end;

if(WeAreDead) then
begin

Writeln(‘Sorry dude, i couldn’‘t save you.’)
LogOut;
TerminateScript;

end;
end;

// Open bank, withdraw bones, close bank.
procedure WithdrawBones;
begin

repeat
OpenBank;
until((BankScreen));

if(BankScreen)then
begin

wait(500 + random(500));
WithdrawUN;
wait(500 + random(500));
MMouse(486, 41, 5, 5);
wait(250 + random(250));
Mouse(486, 41, 2, 2, true);
wait(500 + random(500));

end;
end;

// Click every inventory slot.
procedure ClickAll;
var slotCount: integer;
begin

for slotCount := 1 to 28 do
begin

MMouseItem(slotCount);
wait(250 + random(250));
GetMousePos(mouseX, mouseY);
Mouse(mouseX, mouseY, 2, 2, true);
wait(100 + random(100));

if(not(slotCount = 28)) then
MMouseItem(slotCount + 1);

FindRandoms;
wait(100 + random(100));

end;
end;

// Main Loop
begin

SetupSRL;

repeat

WithdrawBones;
ClickAll;

until(false);

end.[/SCAR]
