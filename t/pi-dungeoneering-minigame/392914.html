Well this is sort-of a project type thing in development, but i decided to just let the idea out there for people to complete this as i am basically done with RSPS related shit.
The Idea: So basically this is just like castle wars and what not. There are 2 teams (red & blue). Each team scores a point when they kill a boss and each member of the team enters the portal (under a certain amount of time). So an example: Red team kills blue teams boss, but now everyone of the members on the red team must now enter the portal (the blue team can attack/stop them while they are trieng to enter the portal) and they have a set time limit of 60 seconds(1 minute). If all of the team enter’s the portal then they score a point. 10 points is auto-win, and if it is a auto-win then they earn double the points they normally would from winning a match normally.

public void joinWait(Client c, int team) {
	if (c.playerEquipment[c.playerHat] > 0 || c.playerEquipment[c.playerCape] > 0 || c.playerEquipment[c.playerWeapon] > 0) {
		c.sendMessage("You may not wear any armor into dungeoneering.");
		return;
	}
	if (team == 1) {
		if (blueWait.size() > redWait.size()) {
			c.sendMessage("This team is currently full.");
			return;
		} else {
			blueWait.add(c.playerId);
			c.castleWarsTeam = 1;
		}
	} else if (team == 2) {
		if (redWait.size() > redWait.size()) {
			c.sendMessage("This team is currently full.");
			return;
		} else {
			redWait.add(c.playerId);
			c.dungeoneeringTeam = 2;
		}
	} else {
		if (redWait.size() > redWait.size()) {
			redWait.add(c.playerId);
			c.sendMessage("You have been added to the Red team.");
			c.dungeoneeringTeam = 1;
		} else {
			redWait.add(c.playerId);
			c.sendMessage("You have been added to the red team.");
			c.dungeoneeringTeam = 2;
		}
	}
	toWaitingRoom(c, c.dungeoneeringTeam);
}

public void toWaitingRoom(Client c, int team) {
	if (team == 1) {
		c.getPA().movePlayer(2377 + Misc.random(10), 9485 + Misc.random(5),0);
	} else if (team == 2) {
		c.getPA().movePlayer(2417 + Misc.random(10), 9485 + Misc.random(7),0);
	}
	c.inDoWait = true;
}

public void leaveWaitingRoom(Client c) {
	c.inCwWait = false;
	c.getPA().movePlayer(2439 + Misc.random(4), 3085 + Misc.random(5), 0);
	if (c.dungeoneeringTeam == 1) {
		blueWait.remove(blue.indexOf(c.playerId));
		c.castleWarsTeam = -1;
	} else if (c.castleWarsTeam == 2) {
		redWait.remove(red.indexOf(c.playerId));
		c.dungeoneeringTeam = -1;
	}			
}


int properTimer = 0;
public void process() {
	if (properTimer > 0) {
		properTimer--;
		return;
	} else {
		properTimer = 4;
	}
		
	if (gameStartTimer > 0) {
		gameStartTimer--;
		updatePlayers();
	} else if (gameStartTimer == 0) {
		startGame();
	}
	if (timeRemaining > 0) {
		timeRemaining--;
		updateInGamePlayers();
	} else if (timeRemaining == 0) {
		endGame();
	}
	/** 
	* The process of the dungeoneering team - called every 3 seconds
	* Thank you sanity
	*/
}
