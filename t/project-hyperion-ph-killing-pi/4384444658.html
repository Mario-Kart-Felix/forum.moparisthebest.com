package org.hyperion.rs2.content;

import org.hyperion.rs2.model.Player;
import org.hyperion.rs2.model.combat.CombatCurses;

/**
 * A class that handles usage of Quick Curses.
 * 
 * @author Relex lawl / Relex
 *
 */

public class QuickCurses {

	private enum QuickCurseData {
		PROTECT_ITEM(17202, CombatCurses.PROTECT_ITEM, 630),
		SAP_WARRIOR(17203, CombatCurses.SAP_WARRIOR, 631),
		SAP_RANGER(17204, CombatCurses.SAP_RANGER, 632),
		SAP_MAGE(17205, CombatCurses.SAP_MAGE, 633),
		SAP_SPIRIT(17206, CombatCurses.SAP_SPIRIT, 634),
		BERSERKER(17207, CombatCurses.BERSERKER, 635),
		DEFLECT_SUMMONING(17208, CombatCurses.DEFLECT_SUMMONING, 636),
		DEFLECT_MAGIC(17209, CombatCurses.DEFLECT_MAGIC, 637),
		DEFLECT_MISSILES(17210, CombatCurses.DEFLECT_MISSILES, 638),
		DEFLECT_MELEE(17211, CombatCurses.DEFLECT_MELEE, 639),
		LEECH_ATTACK(17212, CombatCurses.LEECH_ATTACK, 640),
		LEECH_RANGED(17213, CombatCurses.LEECH_RANGED, 641),
		LEECH_MAGIC(17214, CombatCurses.LEECH_MAGIC, 642),
		LEECH_DEFENCE(17215, CombatCurses.LEECH_DEFENCE, 643),
		LEECH_STRENGTH(17216, CombatCurses.LEECH_STRENGTH, 644),
		LEECH_ENERGY(17217, CombatCurses.LEECH_ENERGY, 645),
		LEECH_SPECIAL_ATTACK(17218, CombatCurses.LEECH_SPECIAL_ATTACK, 646),
		WRATH(17219, CombatCurses.WRATH, 647),
		SOUL_SPLIT(17220, CombatCurses.SOUL_SPLIT, 648),
		TURMOIL(17221, CombatCurses.TURMOIL, 649);
		
		/**
		 * Defining enum's parameters
		 */
		private int actionButtonID, curseID, configButtonID;
		/**
		 * Creation an instance for QuickCurseData enum.
		 * 
		 * @param actionButtonID	Curse's action button ID.
		 * @param curseID			Curse's unique ID.
		 * @param configID			Curse's check mark config ID.
		 */
		private QuickCurseData(final int actionButtonID, final int curseID, final int configButtonID) {
			this.actionButtonID = actionButtonID;
			this.curseID = curseID;
			this.configButtonID = configButtonID;
		}
		/**
		 * Getting QuickCurseData's action button ID.
		 * 
		 * @return	actionButtonID.
		 */
		private final int getActionButtonID() {
			return actionButtonID;
		}
		/**
		 * Getting QuickCurseData's prayer ID.
		 * 
		 * @return	curseID.
		 */
		private final int getCurseID() {
			return curseID;
		}
		/**
		 * Getting QuickCurseData's config button ID.
		 * 
		 * @return	configButtonID.
		 */
		private final int getConfigButtonID() {
			return configButtonID;
		}
	}
	/**
	 * Handles clicking on prayer's orb.
	 * 
	 * @param actionButtonID	actionButton being hit.
	 */
	public void selectCurse(final int actionButtonID) {
		final int curseID = getCurseIDForActionButton(actionButtonID);
		if (curseID == -1)
			return;
		toggleCheckmark(actionButtonID, !quickCurseSelected[curseID]);
		switch (curseID) {
			case CombatCurses.SAP_WARRIOR:
				toggleMultipleCheckmarks(CombatCurses.ATTACK_CURSES, curseID);
				toggleMultipleCheckmarks(CombatCurses.STRENGTH_CURSES, curseID);
				toggleMultipleCheckmarks(CombatCurses.DEFENCE_CURSES, curseID);
				break;
			case CombatCurses.SAP_RANGER:
				toggleMultipleCheckmarks(CombatCurses.RANGED_CURSES, curseID);
				toggleMultipleCheckmarks(CombatCurses.DEFENCE_CURSES, curseID);
				break;
			case CombatCurses.SAP_MAGE:
				toggleMultipleCheckmarks(CombatCurses.MAGIC_CURSES, curseID);
				toggleMultipleCheckmarks(CombatCurses.DEFENCE_CURSES, curseID);
				break;
			case CombatCurses.SAP_SPIRIT:
				toggleMultipleCheckmarks(CombatCurses.SPIRIT_CURSES, curseID);
				break;
			case CombatCurses.DEFLECT_SUMMONING:
			case CombatCurses.DEFLECT_MAGIC:
			case CombatCurses.DEFLECT_MISSILES:
			case CombatCurses.DEFLECT_MELEE:
			case CombatCurses.WRATH:
			case CombatCurses.SOUL_SPLIT:
				toggleMultipleCheckmarks(CombatCurses.OVERHEAD_CURSES, curseID);
				break;
			case CombatCurses.LEECH_ATTACK:
				toggleMultipleCheckmarks(CombatCurses.ATTACK_CURSES, curseID);
				break;
			case CombatCurses.LEECH_RANGED:
				toggleMultipleCheckmarks(CombatCurses.RANGED_CURSES, curseID);
				break;
			case CombatCurses.LEECH_MAGIC:
				toggleMultipleCheckmarks(CombatCurses.MAGIC_CURSES, curseID);
				break;
			case CombatCurses.LEECH_STRENGTH:
				toggleMultipleCheckmarks(CombatCurses.STRENGTH_CURSES, curseID);
				break;
			case CombatCurses.LEECH_ENERGY:
				toggleCheckmark(CombatCurses.TURMOIL, false);
				break;
			case CombatCurses.LEECH_SPECIAL_ATTACK:
				toggleMultipleCheckmarks(CombatCurses.SPIRIT_CURSES, curseID);
				toggleCheckmark(CombatCurses.TURMOIL, false);
				break;
			case CombatCurses.TURMOIL:
				toggleMultipleCheckmarks(CombatCurses.LEECH_CURSES, curseID);
				break;
		}
	}
	/**
	 * Opens quick prayer's interface when
	 * choosing select quick prayer option
	 * in prayer orb.
	 */
	public void openQuickCurseInterface() {
		loadCheckmarks();
		p.getActionSender().sendSidebarInterface(6, 18200);
		p.getActionSender().setViewedTab(6);
	}
	/**
	 * Confirms the quick prayer selection.
	 * Only then will quick curses with checkmarks
	 * take effect.
	 */
	public void confirmSelection() {
		for (int i = 0; i < quickCurseSelected.length; i++)
			quickCurses[i] = quickCurseSelected[i];
		p.getActionSender().sendSidebarInterface(6, 25500);
	}
	/**
	 * Handles clicking on prayer orb
	 * to turn quick curses on.
	 */
	public void toggleQuickCurses() {
		quickCursesOn = !quickCursesOn;
		final String value = quickCursesOn ? "on" : "off";
		p.getActionSender().sendString(-1, ":quicks: " + value);
		p.getCombatCurses().deactivateAllCurses();
		if (quickCursesOn) {
			for (int i = 0; i < quickCurses.length; i++) {
				if (quickCurses[i])
					p.getCombatCurses().activateCurse(i);
			}
		}
	}
	/**
	 * Loads checkmarks for chosen quick curses
	 * when select quick curses option is chosen.
	 */
	private void loadCheckmarks() {
		for (int i = 0; i < quickCurseSelected.length; i++) {
			final QuickCurseData qpd = getQuickCurseDataForCurseID(i);
			try {
				toggleCheckmark(qpd.getConfigButtonID(), quickCurses[i]);
			} catch (Exception e) {
				System.out.println("error loading checkmark #" + i);
			}
		}
	}
	/**
	 * Handles checkmark configs,
	 * both adding and taking away.
	 * 
	 * @param actionButtonID	Curse's actionButtonID.
	 * @param addCheckmark		Whether or not to add a checkmark.
	 */
	private void toggleCheckmark(final int actionButtonID, boolean addCheckmark) {
		for (QuickCurseData qpd : QuickCurseData.values()) {
			if (qpd.getActionButtonID() == actionButtonID) {
				p.getActionSender().sendConfig(qpd.getConfigButtonID(), addCheckmark ? 1 : 0);
				quickCurseSelected[qpd.getCurseID()] = addCheckmark;
			}
		}
	}
	/**
	 * Handles checkmark configs,
	 * same as toggleCheckmark, but for multiple prayers.
	 * 
	 * @param curseIDs		Curse IDs' checkmarks to take away.
	 * @param curseID		Curse ID that's not affected.
	 */
	private void toggleMultipleCheckmarks(final int[] curseIDs, final int curseID) {
		for (int i = 0; i < curseIDs.length; i++) {
			if (curseIDs[i] != curseID) {
				QuickCurseData qpd = getQuickCurseDataForCurseID(curseIDs[i]);
				toggleCheckmark(qpd.getActionButtonID(), false);
			}
		}
	}
	/**
	 * Gets QuickCurseData for curse ID.
	 * 
	 * @param curseID		Curse ID's QuickCurseData to return.
	 * @return				Curse ID's QuickCurseData.
	 */
	private QuickCurseData getQuickCurseDataForCurseID(final int curseID) {
		for (QuickCurseData qpd : QuickCurseData.values()) {
			if (qpd.getCurseID() == curseID)
				return qpd;
		}
		return null;
	}
	/**
	 * Gets prayer ID for an action button.
	 * 
	 * @param actionButtonID	ActionButtonID's prayer to return. 
	 * @return					ActionButtonID's curseID.
	 */
	private static final int getCurseIDForActionButton(final int actionButtonID) {
		for (QuickCurseData qpd : QuickCurseData.values()) {
			if (qpd.getActionButtonID() == actionButtonID) {
				//System.out.println("CurseID: " + qpd.getCurseID());
				return qpd.getCurseID();
			}
		}
		return -1;
	}
	/**
	 * Checks if <code> actionButtonID </code>
	 * is a valid quick prayer button.
	 * 
	 * @param actionButtonID	ActionButtonID to check.
	 * @return					ActionButtonID is a valid quick prayer button.
	 */
	public static final boolean isQuickCurseButton(final int actionButtonID) {
		boolean value = false;
		for (QuickCurseData qpd : QuickCurseData.values()) {
			if (qpd.getActionButtonID() == actionButtonID) {
				value = true;
				break;
			}
		}
		return value;
	}
	/**
	 * Adding a Player instance.
	 */
	private Player p;
	/**
	 * Creating QuickCurses per Player
	 * 
	 * @param p		Single player to create an instance for.
	 */
	public QuickCurses(Player p) {
		this.p = p;
	}
	/**
	 * Used to add a temporary checkmark for curses.
	 */
	public boolean[] quickCurseSelected = new boolean[20];
	/**
	 * Used for actual quick curses Player p has chosen.
	 */
	public boolean[] quickCurses = new boolean[20];
	/**
	 * Checks if Player p already has quick curses on.
	 */
	public boolean quickCursesOn = false;
}
