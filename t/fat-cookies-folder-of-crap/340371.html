public class RandomBitmap {

    public RandomBitmap() {
        char[] sig = {0x424D, 0xF6D4, 0x0100, 0x0000, 0x0000,
            0x3600, 0x0000, 0x2800, 0x0000,
            0xC800, 0x0000, 0xC800,
            0x0000, 0x0100, 0x1800, 0x0000, 0x0000, 0xC0D4,
            0x0100, 0x0000, 0x0000, 0x0000, 0x0000, 0x0000, 0x0000,
            0x0000, 0x0000};


        char[] main = genImage();
        File save = new File("hi.bmp");
        try {
            saveBitmap(save, sig, main);
        } catch (IOException ioe) {
        }
    }

    public static char[] genImage() {
        int length = 200 * 200*3/2;
        char[] output = new char[length];

        Random rand = new Random();
        for (int i = 0; i < length; i++) {
            char c = 0x0;
            output[i] = c+=rand.nextInt(255);
        }
        return output;
    }

    public static void saveBitmap(File f, char[] sig, char[] main) throws IOException {
        FileOutputStream fos = new FileOutputStream(f);
        DataOutputStream dos = new DataOutputStream(fos);
        for (int i = 0; i < sig.length; i++) {
            dos.writeChar(sig[i]);
        }
        for (int i = 0; i < main.length; i++) {
            dos.writeChar(main[i]);
        }

        fos.close();
    }

    public static void main(String[] args) {
        RandomBitmap ma = new RandomBitmap();
    }
}

    public GoldenRatio() {
        this.setTitle("Golden ratio demo");
        this.setSize(480, 300);
        this.setResizable(false);
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        JPanel main = new JPanel();
        this.setContentPane(main);

    public void paint(Graphics g) {
        int height = 250;
        int width = (int) (height * golden);

        g.fillRect(0, 0, 500, 500);

package fail;

 *

public class FibonacciSolver {

    public static final double goldenMean = 1 + (Math.sqrt(5.0)) / 2;

    public static int getNumber(long l) {
        double a = (1 + Math.sqrt(5))/2;
        double b = (1 - Math.sqrt(5))/2;

        double result = (Math.pow(a, l)-Math.pow(b, l))/(a-b);
        return (int) result;
    }

    public void getWords(File f) throws IOException {
        BufferedReader bf = new BufferedReader(new FileReader(f));
        ArrayList<String> words = new ArrayList<String>();
        String line;

        while ((line = bf.readLine()) != null) {
            String[] currWords = line.split(" ");
            for (int i = 0; i < currWords.length; i++) {
                words.add(currWords[i]);
            }

        bf.close();
        printResults(words);

Clicking game (aim is to click button as many times as you can in a minute)

import  java.awt.BorderLayout;
import javax.swing.Timer;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.swing.JApplet;
import javax.swing.JButton;
import javax.swing.JLabel;

/**
 *
 * @author Uncalled For
 */
public class clickGame extends JApplet {

    int time;
    int hits;
    boolean isPlaying = false;
    JButton click;
    JLabel hitLabel;

    @Override
    public void init() {
        time = 60;
        hits = 0;
        click = new JButton(time + "");
        hitLabel = new JLabel("Click: " + hits);

        this.add(hitLabel, BorderLayout.NORTH);
        this.add(click, BorderLayout.CENTER);
        click.addActionListener(new ActionListener() {

            public void actionPerformed(ActionEvent e) {
                if (isPlaying) {
                    hits++;
                    repaint();
                }
            }
        });
        startGame();
    }

    public void startGame() {
        isPlaying = true;
        t.start();
    }

    public void endGame() {
        t.stop();
        isPlaying = false;
        repaint();
    }

    @Override
    public void repaint() {
        if (isPlaying) {
            click.setText(Integer.toString(time));
            hitLabel.setText("Click: " + hits);
        } else {
            click.setText("You got " + hits + " hits");
            hitLabel.setText("Game Over!!");
        }
    }
    Timer t = new Timer(1000, new ActionListener() {

        public void actionPerformed(ActionEvent e) {
            time--;
            repaint();
            if (time == 0) {
                endGame();
            }
        }
    });
}
