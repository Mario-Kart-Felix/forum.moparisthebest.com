moparisthebest July 30, 2016, 12:27am #39

here is my latest Tester class, it has worked great for me so far, very visual compared to a bunch of printlns, and it can be run as an applet or application:

import javax.swing.*;
import java.applet.Applet;
import java.awt.*;
import java.awt.event.*;

public class Tester extends Applet implements KeyListener, MouseListener, MouseMotionListener, FocusListener {

    public boolean mL, mML, kL, fL;

    public int width = 400;
    public int height = 400;
    public int x, y;
    String s = "";
    String focus = "";

    public static void main(String[] args) {
        JFrame.setDefaultLookAndFeelDecorated(true);
        JFrame frame = new JFrame("Tester");
        frame.setResizable(false);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        Tester tester = new Tester();
        tester.init();
        frame.add(tester);
        frame.setSize(400, 400);
        frame.setVisible(true);
    }

    public void init() {
        System.out.println("To toggle listeners on and off:
F1 : MouseListener
F2 : MouseMotionListener
F3 : KeyListener
F4 : FocusListener
");
        setSize(width, height);
        x = width / 2;
        y = height / 2;
        setBackground(Color.black);
        addMouseListener(this);
        addMouseMotionListener(this);
        addKeyListener(this);
        addFocusListener(this);

        //mL = true;
        //mML = true;
        //kL = true;
        //fL = true;
    }

    private void msg_mL(String s) {
        if (mL)
            System.out.println(s);
    }

    private void msg_mML(String s) {
        if (mML)
            System.out.println(s);
    }

    private void msg_kL(String s) {
        if (kL)
            System.out.println(s);
    }

    private void msg_fL(String s) {
        if (fL)
            System.out.println(s);
    }

    public void mousePressed(MouseEvent evt) {
        msg_mL("MouseEvent(Component source, MouseEvent.MOUSE_PRESSED , " + evt.getWhen() + ", " + evt.getModifiers() + "[" + MouseEvent.getMouseModifiersText(evt.getModifiers()) + "], " + evt.getX() + ", " + evt.getY() + ", " + evt.getClickCount() + ", " + evt.isPopupTrigger() + ", " + evt.getButton() + ");");
    }

    public void mouseReleased(MouseEvent evt) {
        msg_mL("MouseEvent(Component source, MouseEvent.MOUSE_RELEASED, " + evt.getWhen() + ", " + evt.getModifiers() + "[" + MouseEvent.getMouseModifiersText(evt.getModifiers()) + "], " + evt.getX() + ", " + evt.getY() + ", " + evt.getClickCount() + ", " + evt.isPopupTrigger() + ", " + evt.getButton() + ");");
    }

    public void mouseClicked(MouseEvent evt) {
        msg_mL("MouseEvent(Component source, MouseEvent.MOUSE_CLICKED , " + evt.getWhen() + ", " + evt.getModifiers() + "[" + MouseEvent.getMouseModifiersText(evt.getModifiers()) + "], " + evt.getX() + ", " + evt.getY() + ", " + evt.getClickCount() + ", " + evt.isPopupTrigger() + ", " + evt.getButton() + ");
");
        x = evt.getX();
        y = evt.getY();
        s = "";
        repaint();
        evt.consume();
    }

    public void mouseEntered(MouseEvent evt) {
        msg_mL("MouseEvent(Component source, MouseEvent.MOUSE_ENTERED, " + evt.getWhen() + ", " + evt.getModifiers() + "[" + MouseEvent.getMouseModifiersText(evt.getModifiers()) + "], " + evt.getX() + ", " + evt.getY() + ", " + evt.getClickCount() + ", " + evt.isPopupTrigger() + ", " + evt.getButton() + ");
");
    }

    public void mouseExited(MouseEvent evt) {
        msg_mL("MouseEvent(Component source, MouseEvent.MOUSE_EXITED, " + evt.getWhen() + ", " + evt.getModifiers() + "[" + MouseEvent.getMouseModifiersText(evt.getModifiers()) + "], " + evt.getX() + ", " + evt.getY() + ", " + evt.getClickCount() + ", " + evt.isPopupTrigger() + ", " + evt.getButton() + ");
");
    }

    public void mouseDragged(MouseEvent evt) {
        msg_mML("MouseEvent(Component source, MouseEvent.MOUSE_DRAGGED" + ", " + evt.getWhen() + ", " + evt.getModifiers() + "[" + MouseEvent.getMouseModifiersText(evt.getModifiers()) + "], " + evt.getX() + ", " + evt.getY() + ", " + evt.getClickCount() + ", " + evt.isPopupTrigger() + ", " + evt.getButton() + ");");
    }

    public void mouseMoved(MouseEvent evt) {
        msg_mML("MouseEvent(Component source, MouseEvent.MOUSE_MOVED" + ", " + evt.getWhen() + ", " + evt.getModifiers() + "[" + MouseEvent.getMouseModifiersText(evt.getModifiers()) + "], " + evt.getX() + ", " + evt.getY() + ", " + evt.getClickCount() + ", " + evt.isPopupTrigger() + ", " + evt.getButton() + ");");
        x = evt.getX();
        y = evt.getY();
        repaint();
        evt.consume();
    }

    public void keyPressed(KeyEvent evt) {
        msg_kL("KeyEvent(Component source, KeyEvent.KEY_PRESSED, " + evt.getWhen() + ", " + evt.getModifiers() + "[" + KeyEvent.getKeyModifiersText(evt.getModifiers()) + "], " + evt.getKeyCode() + "[" + KeyEvent.getKeyText(evt.getKeyCode()) + "], " + evt.getKeyChar() + ");");
        int i = evt.getKeyCode();
        if (i == KeyEvent.VK_F1)
            mL = !mL;
        else if (i == KeyEvent.VK_F2)
            mML = !mML;
        else if (i == KeyEvent.VK_F3)
            kL = !kL;
        else if (i == KeyEvent.VK_F4)
            fL = !fL;
        else if (i == KeyEvent.VK_ENTER) {
            s = "";
            repaint();
        }

        char c = evt.getKeyChar();
        if (c != KeyEvent.CHAR_UNDEFINED) {
            s = s + c;
            repaint();
            evt.consume();
        }
    }

    public void keyReleased(KeyEvent evt) {
        msg_kL("KeyEvent(Component source, KeyEvent.KEY_RELEASED, " + evt.getWhen() + ", " + evt.getModifiers() + "[" + KeyEvent.getKeyModifiersText(evt.getModifiers()) + "], " + evt.getKeyCode() + "[" + KeyEvent.getKeyText(evt.getKeyCode()) + "], " + evt.getKeyChar() + ");
");
    }

    public void keyTyped(KeyEvent evt) {
        msg_kL("KeyEvent(Component source, KeyEvent.KEY_TYPED, " + evt.getWhen() + ", " + evt.getModifiers() + "[" + KeyEvent.getKeyModifiersText(evt.getModifiers()) + "], " + evt.getKeyCode() + "[" + KeyEvent.getKeyText(evt.getKeyCode()) + "], " + evt.getKeyChar() + ");");
    }

    public void focusGained(FocusEvent evt) {
        msg_fL("FocusEvent(Component source, FocusEvent.FOCUS_GAINED, " + evt.isTemporary() + ");");
        focus = "GAINED, temp: " + evt.isTemporary();
        repaint();
    }

    public void focusLost(FocusEvent evt) {
        msg_fL("FocusEvent(Component source, FocusEvent.FOCUS_LOST, " + evt.isTemporary() + ");");
        focus = "LOST, temp: " + evt.isTemporary();
        repaint();
    }

    public void paint(Graphics g) {
        g.setColor(Color.gray);
        g.drawLine(x, y, x, y - 10);
        g.drawLine(x, y, x + 10, y);
        g.setColor(Color.green);
        g.drawString(x + "," + y, x - 20, y - 15);
        g.setColor(Color.orange);
        g.drawString(s, x + 2, y - 2);
        g.setColor(Color.yellow);
        g.drawString(focus, 1, 15);
    }
}

semaphore:

There is nothing wrong with editing the client, as long as you know what you are doing. Anyone that says otherwise, is just really saying that they don’t know how to do it right.

I cannot agree with that in either case.
Editing the client is possible and not too hard if you have enough experience and patience. I followed this sheme for a very long time, basically it was my only way of doing things. But one could aswell state that “There is nothing wrong with packet sending, as long as you know what you are doing”. Ok, not really a good comparison due to the greatly increased complexity of latter approach. I mean, your statement implies that there are no other ways, maybe even better. What happens if jagex decides one day to recode their protection or even just introduces a new trap. Do you go through the code every time to check what they have done? Maybe your system is good enough that there’s only very few situations where engine interferes with the client and in the worst case it will just not work anymore.
Wouldn’t it be great to have an universal engine that would plug into any client, insensitive to any protection mechanisms that are coded around the client core. Of course, such a system will not survive a (partial) rewrite of the client (like rs -> rs2). Still, refraining from direct interference with the client will avoid a lot of hassle with traps and other protections put into place. I’m not saying such a sheme is trivial or even realistic to implement, but if I ever will come back to this again my research is surely going into this direction.
Anyway, as mentioned I followed the former way for a long time and things started to get boring and cumbersome at one point. I had very robust routines and almost fully automated updating mechanism, basically I only had to manully go through the diff between clients and now and then required treatment of some “special packets”. My system required this due to the nice features it had. For example it had a no-graphics mode which lowered CPU usage to ~1% even on moderate machines (Athlon 700Mhz). Maybe you know the code good enough such that you know which trap in the rendering code I’m talking about.
Just saying, eventually it’s a good idea to try some new concepts. At least for me, I’m not doing this to get “useless stuff”, but in order to learn something, try “something”, just to see what happens. I always enjoyed using my engine to send out my creations into the virtual world in order to do something interesting and the excitement when checking your log files the next day and to see what funny situations they were involved in I even had a nice recording/playback system coded in (which actually is rather trivial to implement - just overide the stream class to dump the incoming data and also store some meta data for proper synchronisation and if you whish also the user input) and I still have GBs of records on some backup disks. Afaik 1Mb gives you about 2h 100% accurate playback with camera movement and mouse motion/clicks.
To summarize, this is (was) my way of playing runescape :)))
