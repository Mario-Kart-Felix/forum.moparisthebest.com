This is completely retarded and it’s not at all useful! But it’s my first ‘form’ script, so I felt like posting it.

program New;

var
  // Form variables.
  frmDesign : TForm;
  Label1 : TLabel;
  Edit1 : TEdit;
  Button1 : TButton;

procedure Test(Sender : TObject);
begin
  Label1.Left := Random(100);
  Label1.Top := Random(100);
  Label1.Caption := Edit1.Text;
end;

procedure InitForm;
begin
frmDesign := CreateForm;
frmDesign.Left := 366;
frmDesign.Top := 179;
frmDesign.Width := 239;
frmDesign.Height := 217;
frmDesign.Caption := 'Form Script';
frmDesign.Color := clBtnFace;
frmDesign.Font.Color := clWindowText;
frmDesign.Font.Height := -11;
frmDesign.Font.Name := 'MS Sans Serif';
frmDesign.Font.Style := [];
frmDesign.Visible := False;
frmDesign.PixelsPerInch := 96;
Label1 := TLabel.Create(frmDesign);
Label1.Parent := frmDesign;
Label1.Width := 82;
Label1.Height := 32;
Label1.Font.Color := clWindowText;
Label1.Font.Height := -28;
Label1.Font.Name := 'MS Sans Serif';
Label1.Font.Style := [];
Label1.ParentFont := False;
Edit1 := TEdit.Create(frmDesign);
Edit1.Parent := frmDesign;
Edit1.Left := 19;
Edit1.Top := 78;
Edit1.Width := 121;
Edit1.Height := 21;
Edit1.TabOrder := 8;
Edit1.Text := 'Type stuff!'
Button1 := TButton.Create(frmDesign);
Button1.Parent := frmDesign;
Button1.Left := 5;
Button1.Top := 150;
Button1.Width := 70;
Button1.Height := 25;
Button1.Caption := 'Test';
Button1.TabOrder := 9;
Button1.OnClick := @Test;
end;

procedure SafeInitForm;
var
  v : TVariantArray;
begin
  SetArrayLength(V, 0);
  ThreadSafeCall('InitForm', v);
end;

procedure ShowFormModal;
begin
  frmDesign.ShowModal;
end;

procedure SafeShowFormModal;
var
  v : TVariantArray;
begin
  SetArrayLength(V, 0);
  ThreadSafeCall('ShowFormModal', v);
end;

begin
  SafeInitForm;
  SafeShowFormModal;
  repeat
    GetApplication.MESSAGEBOX('Did you like my Script?','Bored...',1);
  until(IsFKeyDown(12))
end.
