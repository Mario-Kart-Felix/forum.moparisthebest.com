// ==UserScript==
// @name           runescape mess hider
// @namespace       http://schmo.org
// @description     hides all iframes on runescape.com
// @include         http://*.runescape.com*
// @include         http://runescape.com*
// ==/UserScript==

(function() {

    var HideIframes =
    {
        go: function()
        {
            this.injectCSS("IFRAME { display: none; }\n");
        },


        injectCSS: function(css)
        {
            head = document.getElementsByTagName("head")[0];
            style = document.createElement("style");
            style.setAttribute("type", 'text/css');
            style.innerHTML = css;
            head.appendChild(style);
        }
    }

    HideIframes.go();

})();
