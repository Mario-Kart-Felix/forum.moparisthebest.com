[size=28pt]by: Davidi2[/size][/center]

I am creating a game similar to the first Mario Bros games. The ones where you avoid the barrels and try to reach the top of the level by climbing ladders etc. Before I even started making the game, I decided that I should allow people to make their own maps for the game, other then the normal ones that it would come with. I was going to release the game and the level editor at the same time, but the game is taking me longer than expected and I’m not sure the codes all put together would be less then 20,000 characters.

So let me start explaining the program.
You start out with a blank black slate, and a Control Panel with some options on it. The levels are 33x33 grid based, and each grid space (I will use the term tile) can be taken up by almost any tile in the Control Panel. The tiles are:

[ul][li]Block - Walkable tile[/li]
[li]Water - Not walkable, kills on contact from top or sides[/li]
[li]Ladder - Climbable and walkable[/li]
[li]Spike - Not walkable, kills on contact from top[/li]
[li]Blank - Not walkable, player falls when in contact[/li]
[li]Player start - Represents the player start position[/li][/ul]

Just to understand more, you beat the level when you reach the door, which has yet to be added. I’m still deciding how that part will work with the game, so I haven’t put it in yet.

For MoparScapers: You place tiles by selecting the corresponding tile in the Control Panel, and clicking/dragging the mouse over the area you want the tile added.

Some more information:

[ul][li]The tiles are images in the format MapEditorImage, a custom file extension for this application. (.mei)[/li]
[li]This application also supporst saving and loading, map backup, and corrupt file detection. The saved MapEditorMapdata, a custom file extension for this application. (.mem)[/li]
[li]You can shrink the Control Panel by pressing hide, or clicking the exit button on it. The Control Panel will not close until the main panel closes however.[/li][/ul]

public String getRowStringRepresentation(int row) {
	String currentString = "";
	for(int i = 0; i < 33; i++) {
		currentString += grid[i][row];
	}
	return currentString;
}

public void addTile(int xSpot, int ySpot, int id) {
	if(gridL[xSpot][ySpot] != null && grid[xSpot][ySpot] != 2 && !overwrite && id != 2) {
		return;
	}
	if(grid[xSpot][ySpot] == 5 && id != 5) {
		playerStartActive = false;
	}
	if(id == 5 && grid[xSpot][ySpot] != 5 && playerStartActive) {
		JOptionPane.showMessageDialog(null, "You are attempting to add a player start when one already exists on the field!\nPlease erase the old one first!", "Warning", JOptionPane.WARNING_MESSAGE);
		return;
	}
	if(gridL[xSpot][ySpot] != null) {
		panel1.remove(gridL[xSpot][ySpot]);
	}
	grid[xSpot][ySpot] = id;
	switch (id) {
		case 0:
			gridL[xSpot][ySpot] = new JLabel(ladder);
		break;
		case 1:
			gridL[xSpot][ySpot] = new JLabel(block);
		break;
		case 2:
			gridL[xSpot][ySpot] = new JLabel(blank);
		break;
		case 3:
			gridL[xSpot][ySpot] = new JLabel(water);
		break;
		case 4:
			gridL[xSpot][ySpot] = new JLabel(spike);
		break;
		case 5:
			gridL[xSpot][ySpot] = new JLabel(playerStart);
			playerStartActive = true;
		break;
	}
	gridL[xSpot][ySpot].setBounds(xSpot*20, ySpot*20, 20, 20);
	panel1.add(gridL[xSpot][ySpot]);	
	panel1.repaint();
}

private void initializeComponent() {
	panel1.setLayout(null);

	this.add(panel1);
	for(int y = 0; y < 33; y++) {
		for(int x = 0; x < 33; x++) {
			addTile(x, y, 2);
		}
	}
	panel1.repaint();
	this.add(panel1);
	this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
	this.setSize(663,685);
	this.setLocation(50,25);
	this.setVisible(true);
	this.setTitle("Map Editor");
	this.setResizable(false);
    	this.addMouseListener(new MouseAdapter() {
		public void mouseReleased(MouseEvent e) { 
			int xSpot = (int)((e.getX()-3)/20);
			int ySpot = (int)((e.getY()-25)/20);
			if(xSpot > -1 && ySpot > -1 && xSpot < 33 && ySpot < 33) {
				addTile(xSpot, ySpot, tileID);
			}
		}

		public void mouseDragged(MouseEvent e) {

			if(xSpot > -1 && ySpot > -1 && xSpot < 33 && ySpot < 33) {
				addTile(xSpot, ySpot, tileID);
			}
		}

}

public void saveMap(File f) {
	try {
	File finalFile;
	if(f.toString().endsWith(".mem")) {
		finalFile = f;
	}
	else {
		finalFile = new File(f.toString() + ".mem");
	}
	if(finalFile.exists()) {
		JOptionPane.showMessageDialog(null, "You are attempting to overwrite an old file.\nA backup will be kept, and if saving fails the backup will be restored.", "Warning", JOptionPane.WARNING_MESSAGE);
		try{
			String line = "";
		BufferedReader br = new BufferedReader(new FileReader(finalFile));
		BufferedWriter bbw = new BufferedWriter(new FileWriter("Backup/"+finalFile.getName()));
		line = br.readLine();	
		while(line != null) {
			bbw.write(line);
			bbw.newLine();
			bbw.flush();
			line = br.readLine();
		}
		bbw.close();
		br.close();
	} catch(Exception e) {
			JOptionPane.showMessageDialog(null, "Error with backup!\nOverwrite aborted.", "Backup Error", JOptionPane.ERROR_MESSAGE);
			return;
	}
			if(!finalFile.delete()) {
			JOptionPane.showMessageDialog(null, "Error deleting old file for overwrite!\nOverwrite aborted.", "Save Error", JOptionPane.ERROR_MESSAGE);
			return;
		}
	}
	if(finalFile.createNewFile()) {
		BufferedWriter bw = new BufferedWriter(new FileWriter(finalFile));
		for(int i = 0; i < 33; i++) {	
			String s = mf.getRowStringRepresentation(i);
			bw.write(s);
			bw.newLine();
			bw.flush();
		}
		bw.flush();
		bw.close();
	}
	else {
		JOptionPane.showMessageDialog(null, "There was an error creating the new file!\nYou backup is in the \"Backup\" folder.", "Save Error", JOptionPane.ERROR_MESSAGE);
		return;
	}	
	} catch(Exception e) { 
		JOptionPane.showMessageDialog(null, "There was a saving error!\n" + e, "Save Error", JOptionPane.ERROR_MESSAGE);
		return;
	}
}

	public void loadMap(File f) {
	int counter = 0;
	String line = "";
	try {
		File finalFile;
		if(f.toString().endsWith(".mem")) {
			finalFile = f;
		}
		else {
			JOptionPane.showMessageDialog(null, "That's an invalid file! \nAll maps must be \".mem\"", "File Error", JOptionPane.ERROR_MESSAGE);
			return;
		}
		BufferedReader br = new BufferedReader(new FileReader(finalFile));
		line = br.readLine();	
		while(line != null) {
			String tempString = "";
			for(int x = 0; x < 33; x++) {
				try {
					tempString += line.charAt(x);
					int theInt = Integer.parseInt(tempString);
					tempString = "";
				}
				catch(Exception e) {
					JOptionPane.showMessageDialog(null, "That's a corrupt map file! \nPlease do not edit map files out of the map editor!", "File Error", JOptionPane.ERROR_MESSAGE);
					br.close();
					return;
				}
			}
			readLines.add(line);
			if(line.length() != 33) {
				JOptionPane.showMessageDialog(null, "That's a corrupt map file! \nPlease do not edit map files out of the map editor!", "File Error", JOptionPane.ERROR_MESSAGE);
				br.close();
				return;
			}
			counter++;
			line = br.readLine();
		}
		if(counter != 33) {
			JOptionPane.showMessageDialog(null, "That's a corrupt map file! \nPlease do not edit map files out of the map editor!", "File Error", JOptionPane.ERROR_MESSAGE);
			br.close();
			return;
		}
		writeToMainFrame();
		br.close();		
	} catch(Exception e) {
		JOptionPane.showMessageDialog(null, "That's a corrupt map file! \nPlease do not edit map files out of the map editor!", "File Error", JOptionPane.ERROR_MESSAGE);
		return;
	}
}

public void setOverwrite() {
	if(overwrite.isSelected() == true) {
		mf.overwrite = true;
	}
	else {
		mf.overwrite = false;
	}
}

public void writeToMainFrame() {
	mf.overwrite = true;
	int theInt = 0;
	String tempString = "";
	for (int i = 0; i < 33; i++) {
		String s = readLines.get(i);
		for(int x = 0; x < 33; x++) {
			tempString += s.charAt(x);
			theInt = Integer.parseInt(tempString);
			mf.addTile(x, i, theInt);
			tempString = "";
		}
	}
	setOverwrite();
}

	}
}
