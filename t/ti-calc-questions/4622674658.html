Define LibPub modexp(p,e,n)= Func : :Local re :Local r :Local rr :Local p :Local e : :rr:=1 :p:=p :e:=e : :While e≠0 :re:=mod(e,2) :r:=mod(p^(2),n) : :If re=1 Then : rr:=mod(rr*p,n) :EndIf : :p:=r :e:=floor(((e)/(2))) :EndWhile :Return rr :EndFuncDefine LibPub findprime(digits)= Func :Local f :Local i :f:=10^(digits) :i:=floor(rand(1)[1]*f) :While not isPrime(i) :i:=floor(rand(1)[1]*f) :EndWhile :Return i :EndFuncDefine LibPub eea(m,n)= Func :Local x :Local xx :Local y :Local yy :Local r :Local rr :Local q :Local temp :y:=0 :rr:=m :xx:=1 :yy:=0 :r:=n :x:=0 :While r≠0 :q:=floor(((rr)/(r))) :temp:=r :r:=mod(rr,r) :rr:=temp :temp:=x :x:=xx-q*x :xx:=temp :temp:=y :y:=yy-q*y :yy:=temp :EndWhile :Return {rr,abs(xx),abs(yy)} :EndFuncDefine LibPub genrsa(digits)= Func :Local p :Local q :Local n :Local b :Local e :Local d : :p:=findprime(digits) :q:=findprime(digits) : :n:=p*q :b:=(p-1)*(q-1) : :e:=randInt(3,n-1) :While gcd(e,b)≠1 :e:=randInt(3,n-1) :EndWhile :d:=eea(e,b)[2] :Return {p,q,n,b,e,d} :EndFunc
Example Usage:
k:=genrsa(3)
modexp(4,k[5],k[3])
modexp(283,k[6],k[3])
