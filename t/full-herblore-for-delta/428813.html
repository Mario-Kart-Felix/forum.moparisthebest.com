public class MakePotions {

/**
  * @author RandQm
  * Unfinished potion making
  */
	
    public static long actionInterval = 1250;
    public static long lastAction = System.currentTimeMillis();
	
	public int[][] unfPots = { 
			{249, 91, 3 , 227}, {251, 93, 5, 227},{253, 95, 12, 227}, {255, 97, 22, 227}, {257, 99, 30, 227}, {2998, 3002, 34, 227}, {259, 101, 45, 227}, 
			{261, 103, 50, 227}, {263, 105, 55, 227}, {3000, 3004, 63, 227}, {265, 107, 66, 227}, {2481, 2483, 69, 227}, {267, 109, 72, 227}, {269, 111, 78, 227},
			};
	
	public void makeUnfPot(final int playerId, final int id, final int id2) {
		final client c = (client) PlayerHandler.players[playerId];
		for (int j = 0; j < unfPots.length; j++) {
			 if (System.currentTimeMillis() - lastAction < actionInterval)
                 return;
			if (unfPots[j][0] == id && unfPots[j][3] == id2) {
				if (c.playerLevel[c.playerHerblore] >= unfPots[j][2]) {
					c.setAnimation(0x340);
					lastAction = System.currentTimeMillis();
					c.deleteItem(id, c.getItemSlot(id), 1);
					c.deleteItem(id2, c.getItemSlot(id2), 1);
					c.addItem(unfPots[j][1], 1);
					sendMsg(playerId, 0);
					break;
				} else if (c.playerLevel[c.playerHerblore] < unfPots[j][2]) {
					c.sM("You need a herblore level of at least "+unfPots[j][2]+" to make this potion.");
				}
			}
		}
	}
	
	public static String[] herbType = {"guam", "marrentil", "tarromin", "harralander", "ranarr", "toadflax", "irit", "avantoe", "kwuarm", "snapdragon", "cadantine", "lantadyme", "dwarf weed", "torstol"};
	
	public void sendMsg (final int playerId, final int ArrayIndex) {
		final client c = (client) PlayerHandler.players[playerId];
		String herbName = herbType[ArrayIndex];
		c.sM("You add the "+herbName+" to the vial of water.");
	}
}
