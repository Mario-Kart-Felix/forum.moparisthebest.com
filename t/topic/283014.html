public void attackNPCArea(Player p, int gfx, int damage) { if (p == null) { return; } for (NPC n : Engine.npcs) { if (n == null || p.distanceToPoint(n.absX, n.absY) > 8) { continue; } n.hitDiff = Misc.random(damage); n.npcAnim = n.getBlock(n.npcId); n.attackPlayer = p.playerId; if (gfx > 0) { n.npcGfx = gfx; n.gfxUpdate = true; } n.hitUpdate = true; n.animUpdate = true; n.npcUpdate = true; } }

Of course, you would have to edit the code to get it to work for the source you’re using.
