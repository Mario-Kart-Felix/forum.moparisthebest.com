private int pennies = 0;

	public int getPennies()
	{
		return pennies;
	}
	public void setPennies(int i)
	{
		pennies = i;
	}
	public void incPennies()
	{
		pennies++;
	}
	/**
	 * Updates Pennies

	public void sendPennies() {
		RSCPacketBuilder s = new RSCPacketBuilder();
