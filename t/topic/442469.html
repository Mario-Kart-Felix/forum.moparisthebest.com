Contains the player’s levels in all statistics as well the experience.
Here it is though;

[code=java]package runescape.players;

public class Statistics {

public static final int ATTACK = 0;
public static final int DEFENCE = 1;
public static final int STRENGTH = 2;
public static final int HITPOINTS = 3;
public static final int RANGED = 4;
public static final int PRAYER = 5;
public static final int MAGIC = 6;
public static final int COOKING = 7;
public static final int WOODCUTTING = 8;
public static final int FLETCHING = 9;
public static final int FISHING = 10;
public static final int FIREMAKING = 11;
public static final int CRAFTING = 12;
public static final int SMITHING = 13;
public static final int MINING = 14;
public static final int HERBLORE = 15;
public static final int AGILITY = 16;
public static final int THIEVING = 17;
public static final int SLAYER = 18;
public static final int FARMING = 19;
public static final int RUNECRAFTING = 20;

private int[] level = new int[20];
private double[] experience = new double[20];

public Statistics(int[] level, double[] exp) {
	this.level = level;
	this.experience = exp;
}

public void damageStat(int stat, int damage) {
	this.level[stat] -= damage;
	if (this.level[stat] < 0) {
		this.level[stat] = 0;
	}
}

public void improveStat(int stat, int bonus, int cap) {
	this.level[stat] += bonus;
	if (this.level[stat] > cap) {
		this.level[stat] = cap;
	}
}

public void neutralizeStats() {
	for (int i = 0; i < 20; i++) {
		if (i != HITPOINTS || i != PRAYER) {
			if (this.level[i] > getLevelByExp(i)) {
				this.level[i]--;
			} else if (this.level[i] < getLevelByExp(i)) {
				this.level[i]++;
			}
		}
	}
}

public int getLevelByExp(int stat) {
	double exp = this.experience[stat];
	if (exp >= 13034431) return 99;
	double e = 0;
	double b = 0;
	for (int i = 1; i < 100; i++) {
		e += i + 300.0 * Math.pow(2, i/7);
		b = e/4;
		if (b >= exp) return i;
	}
	return 0;
}

}
[/code]
