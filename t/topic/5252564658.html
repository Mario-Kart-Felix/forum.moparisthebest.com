in the mean time I started working on an ActorBitMap

[code=java]
public final class ActorBitMap {

private final Actor[] actors;

private int[] entries;

private int size;

public ActorBitMap(int capacity) {
	actors = new Actor[(capacity + 7) >> 3];
}

public void add(T actor) {
	int id = actor.getIndex();

	entries[id >> 3] |= 1 << (id & 7);
	actors[size++] = actor;
}

public void remove(T actor) {
	int id = actor.getIndex();

	entries[id >> 3] &= ~(1 << (id & 7));
}

public boolean contains(T actor) {
	int id = actor.getIndex();

	return (entries[id >> 3] &= ~(1 << (id & 7))) != 0;
}

public void clear() {
	size = 0;
}

public Actor[] getActors() {
	return actors;
}

public int getSize() {
	return size;
}

}[/code]
