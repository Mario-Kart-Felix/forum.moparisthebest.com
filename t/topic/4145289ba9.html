public class Firemaking {

public static boolean handleTinderbox(Player c, Item id4, Item id5) {
	int id1 = id4.getId();
	int id2 = id5.getId();
	int logId;
	if (id1 == 590) {
		logId = id2;
	} else {
		logId = id1;
	}
	if(makeFire(c, logId))
		return true;

		return false;

public static int[] logs = {1511,2862,1521,1519,6333,6332,1517,1515,1513};
public static int[] level = {1,1,15,30,35,50,45,60,75};
public static int[] XP = {40,40,60,90,105,157,135,202,303};
public static boolean makeFire(Player c, int id) {
	for (int j = 0; j < logs.length; j++) {
		if (logs[j] == id) {
			if (c.getSkill().getLevel()[11] >= level[j]) {
				c.getInventory().removeItem(new Item(id));
				c.getUpdateFlags().sendAnimation(733, 0);
				c.getSkill().addExp(11, (XP[j]));
				c.getActionSender().sendMessage("You make a fire with the tinderbox and logs");
				return true;
			} else {
				c.getActionSender().sendMessage("You need a firemaking level of " + level[j] + " to fire this.");
			}
		}
	}
	return false;
}
