I figured it out…

The question was how do you make the console close on the stroke of a certain key… in this case the escape key…

I found the solution of

if (Console.ReadKey(true).Key == ConsoleKey.Escape) { Console.Read(); }

In my case I was using it to assist in creating a simple console program where you could type something in and it would relay it to you…

[code]namespace randomProject
{
class Program
{

    static void Main(string[] args)
    {

        Console.WriteLine("Press Esc at any point to terminate program");

        Console.WriteLine("Please Type the name you wish to use:");

        string userName = Console.ReadLine();


        Console.WriteLine();

        while (Console.ReadKey(true).Key != ConsoleKey.Escape)
        {

            String ut = Console.ReadLine();

            Console.WriteLine(userName + " Says: " + ut);




            if (Console.ReadKey(true).Key == ConsoleKey.Escape)
            {
                Console.Read();
                break;
            }

        }
    }
}

}[/code]
