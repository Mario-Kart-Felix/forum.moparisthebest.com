public static String combatDiffColor(int i, int j) { int k = i - j; if(k < -9) return "@red@"; if(k < -6) return "@or3@"; if(k < -3) return "@or2@"; if(k < 0) return "@or1@"; if(k > 9) return "@gre@"; if(k > 6) return "@gr3@"; if(k > 3) return "@gr2@"; if(k > 0) return "@gr1@"; else return "@yel@"; }
