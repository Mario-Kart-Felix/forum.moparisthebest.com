import runescape.content.interfaces.ButtonHandler;
import runescape.content.interfaces.ItemOptionHandler;
import runescape.model.Item;
import runescape.model.Skills;
import runescape.model.player.Player;

/*

  • This whole thing could probably be reworked to work for every lamp.
  • Make into an interface Event
  • Finish this the correct way with interfaces. IE: The interface at the end for telling you how much xp you’ve received instead of sending a message.
    */

public class GenieLamp extends Content implements ItemOptionHandler, ButtonHandler {

private static final int[] GENIE_LAMP_SKILL_IDS = { 
	Skills.ATTACK, Skills.STRENGTH, Skills.RANGE, Skills.MAGIC, Skills.DEFENCE,
	Skills.HITPOINTS, Skills.PRAYER, Skills.AGILITY, Skills.HERBLORE,
	Skills.THIEVING, Skills.CRAFTING, Skills.RUNECRAFTING, Skills.MINING,
	Skills.SMITHING, Skills.FISHING, Skills.COOKING, Skills.FIREMAKING,
	Skills.WOODCUTTING,	Skills.FLETCHING, Skills.SLAYER, Skills.FARMING, 
	Skills.CONSTRUCTION 
};

private static int genieLampIndex = -1;

public void setGenieLampIndex(int genieLampIndex) {
	this.genieLampIndex = genieLampIndex;
}

public int getGenieLampIndex() {
	return genieLampIndex;
}

@Override
public boolean handleButton(Player player, int interfaceId, int buttonId, int buttonId2) {
	if (interfaceId == 134) {			
		if (buttonId == 26) {
			if (getGenieLampIndex() == -1) {
				player.getActionSender().sendMessage("You must first select a skill before confirming your wish.");
			}				
			final int skillId = GENIE_LAMP_SKILL_IDS[getGenieLampIndex()];
			player.getActionSender().sendMessage("The lamp grants you " + 10 * player.getSkills().getLevel(skillId) +" "+ Skills.SKILL_NAME[skillId] +" XP!");
			player.getSkills().addExperience(skillId, 10 * player.getSkills().getLevel(skillId));
			player.getInventory().remove(new Item(2528, 1));
			player.getInventory().refresh(player);
			player.getActionSender().sendCloseInterface();
			player.getActionSender().sendConfig(261, 0);
			setGenieLampIndex(-1);
		} else {
			setGenieLampIndex(buttonId - 4);
			player.getActionSender().sendConfig(261, buttonId - 3);			
		}
	}
	return false;
}

@Override
public boolean handleItemOption(Player player, int interfaceId, int index, int itemId) {
	if (interfaceId == 149 && itemId == 2528) {
		player.getActionSender().sendInterface(134);			
	}
	return false;
}

@Override
public Object getObject() {
	return this;
}

}
