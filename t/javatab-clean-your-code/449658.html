Good day all!

I’ve written a program that automatically tabs your Java code to make it easier to read and understand. The reason I did this is because when I look at java code written by lazy people, such as most RSPS server and client bases, they never add tabs, which makes it really hard to read and understand the process that the code takes.

It works with all java code, and doesn’t cause compiler errors, I’ve tested this on a 508 server source and compiled it no problem.

The program is attached, I’ve only made compiler and runner for Mac OS X and Linux, if you have Windows shouldn’t be that hard for you to fix it up.

DOWNLOAD!

HOW TO USE:

For entire directory

For a single java file

You can also specify the tab type using “-t” parameter before the list of files/folders you want to auto tab.
Valid options:
-t tab - Each tab will be one tab character “\t”
-t space - Each tab will be 4 spaces
-t spaceX - X is the number of spaces you want to put, instead of 4

[code]import java.io.*;

public class JavaTab {
public static void main(String args[]) {
String tab = “\t”;
for (int i = 0; i < args.length; i++) {
if (args[i].equals("-t")) {
i++;
if (args[i].equals(“tab”)) {
tab = “\t”;
} else if (args[i].equals(“space”)) {
tab = " “;
} else if (args[i].startsWith(“space”)) {
int spaceCount = Integer.parseInt(args[i].substring(5));
tab = “”;
for (int j = 0; j < spaceCount; j++) {
tab += " “;
}
}
}
File file = new File(args[i]);
if (file.isFile()) {
new JavaTab(file, tab).start();
} else if (file.isDirectory()) {
scanPath(file, tab);
}
}
}
private static void scanPath(File folder, String tab) {
File files[] = folder.listFiles();
for (int i = 0; i < files.length; i++) {
if (files[i].isDirectory()) {
scanPath(files[i], tab);
} else if (files[i].getName().toLowerCase().endsWith(”.java”)) {
new JavaTab(files[i], tab).start();
}
}
}
private final File file;
private final String tab;
public JavaTab(File file, String tab) {
this.file = file;
this.tab = tab;
}
private boolean started = false;
private final Object lockObject = new Object();
public void start() {
final JavaTab jt = this;
synchronized (lockObject) {
if (!started) {
started = true;
new Thread(new Runnable() {
public void run() {
jt.run();
}
}).start();
}
}
}
private void run() {
try {
BufferedReader reader = null;
{
CharArrayWriter caw = new CharArrayWriter();
FileReader fr = new FileReader(file);
char[] buffer = new char[4096];
int c = 0;
while ((c = fr.read(buffer, 0, buffer.length)) != -1) {
caw.write(buffer, 0, c);
}
reader = new BufferedReader(new CharArrayReader(caw.toCharArray()));
}
FileWriter fw = new FileWriter(file);
int tabCount = 0;
String line = null;
while ((line = reader.readLine()) != null) {
boolean inStringTypeA = false;
boolean inStringTypeB = false;
int tabsToPut = tabCount;
line = line.trim();
CharArrayReader car = new CharArrayReader(line.toCharArray());
int x = 0;
int lastChar = 0;
while ((x = car.read()) != -1) {
if (!inStringTypeA && !inStringTypeB) {
if (x == 123) tabCount += 1; // Open Bracket {
if (x == 125) tabCount -= 1; // Close Bracket }
}
if (lastChar != 92) { // If last character is not
if (x == 34 && !inStringTypeB) inStringTypeA = !inStringTypeA; // Open/Close Speech Quote Str
if (x == 39 && !inStringTypeA) inStringTypeB = !inStringTypeB; // Open/Close Apostrophe Str
}
if (tabCount < tabsToPut) tabsToPut = tabCount;
if (lastChar == 92)
lastChar = 0;
else
lastChar = x;
}
for (int i = 0; i < tabsToPut; i++) {
fw.write(tab);
}
fw.write(line);
fw.write("\n");
}
fw.close();
} catch (Exception e) {
e.printStackTrace();
}
}
