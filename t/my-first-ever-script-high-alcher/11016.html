// Wield Fire Staff And Have Item To Alch In 15th Inventory Space //
// Press Play //

program Alcher;

procedure OpenMagicBook;
begin;
MoveMouseSmooth(735,188);
Wait (50);
ClickMouse(735,188, True);
wait(100+random(200));
end;

procedure HighAlch;
begin
MoveMouseSmoothFrom(735,188, 675,327);
Wait (50);
ClickMouse(675,327, True);
wait(500+random(200));
end;

procedure AlchBow;
begin
ClickMouse(675,327, True);
wait(3000+random(200));
end;

procedure AlchLoop;
begin
ClickMouse(675,327, True);
wait(300+random(200));
end;

begin
OpenMagicBook;
HighAlch;
repeat
AlchBow;
AlchLoop;
until (false)
end.[/scar]
