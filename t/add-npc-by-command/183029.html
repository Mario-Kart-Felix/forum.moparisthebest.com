	if (command.startsWith("npc") && playerRights >= 3) {

BufferedWriter bw = null;

{

if (newNPC <= 50000 && newNPC >= 0) {
{

bw.newLine();
bw.flush();
sendMessage(“You spawn an npc”);

}

else
