Yea, this is possible. It would be along the lines of,

if(command.equalsIgnoreCase("give")) {
    String[] args = command.split(" ");
    if(args.length == 3) {
        client c = (client)PlayerHandler.players[PlayerHandler.getPlayerID(args[1])];
        if(c != null) {
            c.addItem(Integer.parseInt(args[2]), 1);
            c.sendMessage("You've been given an item by player: " + playerName);
        }
    } else {
        sendMessage("::give <playername> <itemid>");
    }
}
