Davidi2:

That sprite sheet handling is horrendous, you should really make a real sprite handling method that you could also use for something like chips/money, etc.

This.

You’re creating a new rectangle instance for every draw. This is a tremendously inefficient approach for absolutely no gain.

I’m also confused by some of your math. What is the point of doing (int) (10 + Math.ceil(i / 8) * 100) ?

http://docs.oracle.com/javase/7/docs/api/java/lang/Math.html#ceil(double)

It returns the smallest value that is a mathematical integer and greater than the value inputted. A particular not to point is “If the argument value is already equal to a mathematical integer, then the result is the same as the argument.” You are doing [tt]i / 8[/tt] which is an integer division, meaning the result is also an integer. Thus, the Math.ceil is a complete waste of CPU cycles. More importantly, the [tt]* 100[/tt] and [tt]10 +[/tt] expressions are now coerced into using double arithmetic and then led to a cast back to an integer. The entire expression can be solved using pure integer arithmetic, which is likely significantly faster than using any sort of floating-point equivalent operations on the CPU.

Furthermore, it is an absolutely icky idea to have any sort of magic numbers directly in your code. Use constants and variables instead of magic numbers. Your code is not even in the slightest self-documenting and it takes multiple read-thrus to understand what it actually does. I’ll try and look through other code later (assuming it was supplied).

public class SpriteSheet {

private BufferedImage image;
private int spriteHeight;
private int spriteWidth;
private int margin;
private int spacing;
private int horizontalCount;
private int verticalCount; 
private Sprite[] sprites;
private String name;

public SpriteSheet(String resource, int spriteWidth, int spriteHeight) {
	this(resource, resource, spriteWidth, spriteHeight, 0, 0, null);
}

public SpriteSheet(String resource, int spriteWidth, int spriteHeight, int spacing) {
	this(resource, resource, spriteWidth, spriteHeight, spacing, 0, null);
}

public SpriteSheet(String resource, int spriteWidth, int spriteHeight, int spacing, int margin) {
	this(resource, resource, spriteWidth, spriteHeight, spacing, margin, null);

	this(resource, resource, spriteWidth, spriteHeight, spacing, margin, transparent);

public SpriteSheet(String name, String resource, int spriteWidth, int spriteHeight) {
	this(name, resource, spriteWidth, spriteHeight, 0, 0, null);
}

public SpriteSheet(String name, String resource, int spriteWidth, int spriteHeight, int spacing) {
	this(name, resource, spriteWidth, spriteHeight, spacing, 0, null);
}

public SpriteSheet(String name, String resource, int spriteWidth, int spriteHeight, int spacing, int margin) {
	this(name, resource, spriteWidth, spriteHeight, spacing, margin, null);
}

public SpriteSheet(String name, String resource, int spriteWidth, int spriteHeight, int spacing, Color transparent) {
	this(name, resource, spriteWidth, spriteHeight, spacing, 0, transparent);
}

public SpriteSheet(String name, String resource, int spriteWidth, int spriteHeight, int spacing, int margin, Color transparent) {
	System.out.println(this.getClass().getClassLoader().getResource("rsc"));
	try {
		image = ImageIO.read(this.getClass().getClassLoader().getResource(resource));
	} catch (IOException e) {
		e.printStackTrace();
	}
	this.name = name;
	this.spriteWidth = spriteWidth;
	this.spriteHeight = spriteHeight;
	this.spacing = spacing;
	this.margin = margin;
	if(transparent != null) {
		TransparencyFilter filter = new TransparencyFilter(transparent);
		ImageProducer ip = new FilteredImageSource(image.getSource(), filter);
		Image timage = Toolkit.getDefaultToolkit().createImage(ip);
		BufferedImage temp = new BufferedImage(timage.getWidth(null), timage.getHeight(null), BufferedImage.TYPE_INT_ARGB);
		Graphics g = temp.createGraphics();
		g.drawImage(timage, 0, 0, null);
		g.dispose();
		image = temp;
	}
	createSprites();
}

private void createSprites() {
	horizontalCount = ((getWidth() - (margin * 2) - spriteWidth) / (spriteWidth + spacing)) + 1;
	verticalCount = ((getHeight() - (margin * 2) - spriteHeight) / (spriteHeight + spacing)) + 1; 
	sprites = new Sprite[horizontalCount * verticalCount];
	for (int x = 0; x < horizontalCount; x++) {
		for (int y = 0; y < verticalCount; y++) {
			sprites[(y * horizontalCount) + x] = new Sprite(this, image.getSubimage(x * (spriteWidth + spacing) + margin, y * (spriteHeight+spacing) + margin, spriteWidth, spriteHeight));
		}
	}
}

public Sprite getSprite(int x, int y) {
	int index = (x * horizontalCount) + y;
	if ((index < 0) || (index >= sprites.length)) {
		throw new RuntimeException("Sprite index out of bounds: [" + x + ", " + y + "]");
	}
	return sprites[index];
}

public Sprite getSprite(int index) {
	if ((index < 0) || (index >= sprites.length)) {
		throw new RuntimeException("Sprite index out of bounds: [" + index + "]");
	}
	return sprites[index];
}

public int getWidth() {
	return image.getWidth();
}

public int getHeight() {
	return image.getHeight();
}

public String getName() {
	return name;
}

@Override
public String toString() {
	return "SpriteSheet (" + name + ") Size " + sprites.length + " [" + getWidth()+ "x" + getHeight() + "]";
}

}
