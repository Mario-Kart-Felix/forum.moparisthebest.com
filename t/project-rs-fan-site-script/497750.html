<?php 
	session_start();
	require_once '/config/dataConfig.php';
	require_once '/config/siteConfig.php';
	$pageName = "Member";
	if($_SESSION['loggedIn'] === 1) {
		$result = mysql_query("SELECT id,email,title,avatar FROM users WHERE username ='$username'");
		$row = mysql_fetch_array($result);
		if (!$result) {
			echo 'Could not run query: ' . mysql_error();
			exit;

		$username = mysql_real_escape_string($_SESSION['username']);

		echo '<div id="memMessage">';
		echo '<p>Below are a few more options for you to customize your profile! Take a minute to update your profile!</p>';

		echo '</div>';
		echo '</div>';
		echo '<div id="sideBar">';
		echo '</div>';
		echo '<div class="clear"></div>';
		include '/tmpl/ovr/ovrFoot.php';
	}
	else {
		include '/tmpl/ovr/ovrHead.php';
		include '/tmpl/navAndHead.php';
		echo '<div id="content">';

		echo '<center><strong><p>You must login to view this page! <a href="login.php">Login</a> or <a href="register.php">register</a>.</p></strong></center>';

?>
