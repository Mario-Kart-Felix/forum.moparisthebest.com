[code=java5]package org.rscemulation.server;

import org.jibble.pircbot.PircBot;
import org.jibble.pircbot.User;
import org.jibble.pircbot.Colors;

import org.rscemulation.server.model.Player;
import org.rscemulation.server.model.World;

public class IRC extends PircBot implements Runnable {
long lastCommand = 0;

public void run() {
	if (Config.IRC) {
    	Thread.currentThread().setPriority(Thread.MIN_PRIORITY);
    	Thread.currentThread().setName("IRC");
    	
    	try {
    		if (Config.IRC_DEBUG)
    			setVerbose(true);
    	    setName(Config.IRC_USERNAME);
    	    setLogin(Config.IRC_USERNAME);
    		connect(Config.IRC_SERVER);
    		identify(Config.IRC_PASSWORD);
    		setVersion(Config.IRC_USERNAME);
    		setFinger(Config.IRC_USERNAME);
    		
    	    joinChannel(Config.IRC_CHANNEL);
    	} catch (Exception e) {
    	    e.printStackTrace();
    	}
	}
}

public String bold() {
    return Colors.BOLD;
}   

protected void onDisconnect() {
	run();
}

public void onMessage(String channel, String sender, String login, String hostname, String message) {
	if (message.startsWith("!"))
	    handleCommand(message, sender);    	
}

public boolean isOp(String sender) {
	User users[] = getUsers(Config.IRC_CHANNEL);

	for (User user : users) {
		if (sender.equals(user.getNick()) && user.isOp())
			return true;
	}	
	return false;   	
}

protected void onJoin(String channel, String sender, String login, String hostname)  {
	if (isOp(getName()))
		voice(Config.IRC_CHANNEL, sender);
	
	if (Config.IRC_GREET_1 != null)
		sendMessage(sender, Config.IRC_GREET_1);
	if (Config.IRC_GREET_2 != null)
		sendMessage(sender, Config.IRC_GREET_2);
	if (Config.IRC_GREET_3 != null)
		sendMessage(sender, Config.IRC_GREET_3);   	
}

public void onPrivateMessage(String sender, String login, String hostname, String message) {
	if (message.startsWith("!"))
	    handlePrivateCommand(message, sender);    	
}

public void sendNotify(String message) {
	if (Config.IRC_NOTIFY)
		sendMessage(Config.IRC_CHANNEL, message);
}  

public void sendMessage(String message) {
	sendMessage(Config.IRC_CHANNEL, message);
}  

public boolean isAdmin(String sender) {
	if ((sender.equalsIgnoreCase("Kryptix") || sender.equalsIgnoreCase("Pyru")) && isOp(sender))
		return true;
	return false;
}

public void handleCommand(String message, String sender) {
	if (System.currentTimeMillis() < lastCommand + 2000)
		return;
	lastCommand = System.currentTimeMillis();
	String[] args = message.split(" ");
	String command = args[0];
	message = message.replace(args[0], "").trim();

	if (command.equalsIgnoreCase("!time") || command.equalsIgnoreCase("!date")) {
		sendMessage(Config.IRC_CHANNEL, sender + ": RSCEmulation's time/date is: " + new java.util.Date().toString());
	} else if (command.equalsIgnoreCase("!toggle")) {
		Config.IRC_NOTIFY = !Config.IRC_NOTIFY;
		sendMessage(sender + ": Notify messages have been " + (Config.IRC_NOTIFY ? "enabled" : "disabled")); 
	} else if (command.equalsIgnoreCase("!debug")) {
		if (isAdmin(sender)) {
			Config.IRC_DEBUG = !Config.IRC_DEBUG;
    		sendMessage(sender + ": Live debugging has been " + (Config.IRC_DEBUG ? "enabled" : "disabled"));
    		setVerbose(Config.IRC_DEBUG);
		}
	} else if (command.equalsIgnoreCase("!online")) {
		int online = World.getPlayers().count();
		if (online > 0)
			sendMessage(Config.IRC_CHANNEL, sender + ": There's currently " + bold() + online + bold() + " player" + (online > 1 ? "s" : "") + " online RSCEmulation");    		
	} else if (command.equalsIgnoreCase("!say") || command.equalsIgnoreCase("!s")) {
		if (isAdmin(sender)) {
			synchronized (World.getPlayers()) {
				for (Player p : World.getPlayers())
					p.sendNotification("#adm#@yel@" + sender + " (IRC):@whi@ " + message);
			}
			sendMessage(bold() + sender + ":" + bold() + " " + message);
		}
	} else if (command.equalsIgnoreCase("!disconnect")) {
		if (isAdmin(sender)) {
			disconnect();
		}
	} else if (command.equalsIgnoreCase("!nick")) {
		if (isAdmin(sender)) {
			if (message.length() > 2)
				changeNick(message);
		}
	} else if (command.equalsIgnoreCase("!deop")) {
		if (isAdmin(sender)) {
			if (message.length() > 2)
				deOp(Config.IRC_CHANNEL, message);
		}
	} else if (command.equalsIgnoreCase("!devoice")) {
		if (isAdmin(sender)) {
			if (message.length() > 2)
				deVoice(Config.IRC_CHANNEL, message);
		}
	} else if (command.equalsIgnoreCase("!kick")) {
		if (isAdmin(sender)) {
			if (message.length() > 2)
				kick(Config.IRC_CHANNEL, message);
		}
	}
}	

public void handlePrivateCommand(String message, String sender) {
	if (System.currentTimeMillis() < lastCommand + 2000)
		return;
	lastCommand = System.currentTimeMillis();		
	if (isAdmin(sender)) {
		String[] args = message.split(" ");
		String command = args[0];
		message = message.replace(args[0], "").trim();

    	if (command.equalsIgnoreCase("!shutdown"))
    		World.startShutdown(message);
	}
}
