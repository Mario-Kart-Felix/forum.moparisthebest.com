import java.io.*;

public class GasMileage {

public static void main(String[] args) throws IOException {

	/* Declare and Construct Variables */
	String distance, price, gas;
	int miles, dollars, gallons;
	double fuel, cost, total;
	BufferedReader dataIn= new BufferedReader (new InputStreamReader(System.in));

	/* Print Prompts and get Input */
	System.out.println("\tZach's Gas Mileage Calculator");
	System.out.println();
	System.out.println("\t\tEnter your total distance you are traveling");
	distance = dataIn.readLine();
	miles = Integer.parseInt(distance);
	System.out.println("\t\tEnter the price of gas to the gallon");
	price = dataIn.readLine();
	dollars = Integer.parseInt(distance);
	System.out.println("\t\tEnter your car's gallons to the mile ratio");
	gas = dataIn.readLine();
	gallons = Integer.parseInt(gas);

	/* Calculations */
	fuel = miles / gallons;
	cost = fuel * dollars;
	total = cost;

	/* Output */
	System.out.println();
	System.out.println("\tThe cost of gas you will need for this trip is " + total + ".");
	System.out.println();
}
