I think you could accomplish that by using two for loops.
You loop through the values of the first array, and compare the current value of the loop against the other array. You use a second loop to compare the single value from the first array and add to an int for each match found.
An example is below.

[code=java]public class DiceTest {

public static void main(String[] args) {
	int[] set = { 3, 1, 5, 2, 4 };
	int[] otherSet = { 6, 2, 5, 5, 2 };

	System.out.println("Matches found: ");
	for (int i : set)
		checkMatches(i, otherSet);
}

public static void checkMatches(int value, int[] setTwo) {
	int matched = 0;
	for (int i : setTwo) {
		if (i == value)
			matched++;
	}
	System.out.println("Found " + matched + " matches for " + value);
}

}[/code]

Output:

Matches found: Found 0 matches for 3 Found 0 matches for 1 Found 2 matches for 5 Found 2 matches for 2 Found 0 matches for 4

Of course, you’d want to tweak it to what the assignment is.

Okay, how about this? [code=java]public class DiceGame {
public static void main(String[] args) {
int[] first_set = new int[]{ 1, 4, 5 };
int[] second_set = new int[]{ 1, 4, 5 };
int[] intersection = new int[3];

    for(int i = 0; i < first_set.length; i++) {
        for(int j = 0; j < first_set.length; j++) {
            if(first_set[i] == second_set[j])
                intersection[i] = first_set[i];
        }
    }

    int count = 0;
    for(int i = 0; i < intersection.length; i++) {
        if(intersection[i] != 0) count++;
    }

    System.out.println(count+" of a kind");
}

}[/code]

import java.util.ArrayList;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.concurrent.ThreadLocalRandom;

/**
 *
 * @author Silabsoft
 */
public class DiceGame {

    private final Player playerA;
    private final Player playerB;
    private int gameId = 0;
    private static final int NUMBER_OF_DICE = 5;

    public DiceGame() {
        playerA = new Player();
        playerB = new Player();
    }

    public static void main(String args[]) {
        DiceGame game = new DiceGame();
        for (int i = 0; i < 25; i++) { //play 25 games
            game.rollDice();
            game.findAndDelcareWinner();
        }
    }

    private void rollDice() {
        int[] playerARoll = new int[NUMBER_OF_DICE];
        int[] playerBRoll = new int[NUMBER_OF_DICE];
        for (int i = 0; i < NUMBER_OF_DICE; i++) {
            playerARoll[i] = ThreadLocalRandom.current().nextInt(1, 6 + 1);
            playerBRoll[i] = ThreadLocalRandom.current().nextInt(1, 6 + 1);
        }
        playerA.getRollResults().add(gameId, playerARoll);
        playerB.getRollResults().add(gameId, playerBRoll);
    }

    private void findAndDelcareWinner() {
        int playerAKindCount = getKindCount(playerA.getRollResults().get(gameId));
        int playerBKindCount = getKindCount(playerB.getRollResults().get(gameId));
        String winner = "Game[" + gameId + "]  ";
        if (playerAKindCount == playerBKindCount) {
            winner += "Resulted in a tie ";
        } else {
            winner += playerAKindCount > playerBKindCount ? "Player A won with with " + playerAKindCount : "Player B won with " + playerBKindCount;
            winner += " of a kind ";
        }
        winner += "PLAYER A:" + playerA.rollToString(gameId) + " PLAYER B: " + playerB.rollToString(gameId);

        System.out.println(winner);
        gameId++;
    }

    private int getKindCount(int[] roll) {
        Arrays.sort(roll);
        int lastValue = 0;
        int largestCount = 1;
        int currentCount = 1;
        for (int i = 0; i < roll.length; i++) {
            if (i > 0) {
                if (roll[i] == lastValue) {
                    currentCount++;
                } else {
                    if (largestCount < currentCount) {
                        largestCount = currentCount;
                    }
                    currentCount = 1;
                }
            }
            lastValue = roll[i];
        }
        if (currentCount > largestCount) {
            return currentCount;
        } else {
            return largestCount;
        }
    }

    private class Player {

        private final LinkedList<int[]> rollResults;

        public Player() {
            this.rollResults = new LinkedList<int[]>();
        }

        public LinkedList<int[]> getRollResults() {
            return rollResults;
        }

        public String rollToString(int gameId) {
            String rollString = "";
            int[] roll = rollResults.get(gameId);
            for (int i = 0; i < roll.length; i++) {
                rollString += "[" + roll[i] + "]";
            }
            return rollString;
        }
    }
}
