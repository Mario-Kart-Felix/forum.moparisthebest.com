		c.outStream.writeWord(id.length);
		for (int i = 0; i < id.length; i++) {
			c.outStream.writeByte(1);

				c.outStream.writeWordBigEndianA(id[i]+1);
			} else {
				c.outStream.writeWordBigEndianA(0);
			}
		}
		c.outStream.endFrameVarSizeWord();   
		c.flushOutStream();
