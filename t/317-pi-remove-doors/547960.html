Hella simple…
Client sided find your ObjectDef.java
Add this under the objectDef.readValues(stream);

if(objectDef != null && objectDef.name != null) { if(objectDef.name.toLowerCase().contains("door")) { for(int q = 0; q < objectDef.anIntArray773.length; q++) objectDef.anIntArray773[q] = 0; objectDef.hasActions = false; } }
