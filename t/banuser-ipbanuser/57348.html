Well, let’s see. For banning usernames:

if(command.startsWith("banuser")) { String name = command.substring(8); if(appendToBanned(name)) { sendMessage(name + " has been banned"); } }

And add this boolean:

public boolean appendToBanned(String name) try { File file = new File("./data/bannedusers.txt"); BufferedReader br = new BufferedReader(new FileReader(file)); BufferedWriter bw = new BufferedWriter(new FileWriter(file, true)); for(int i = 0; i < file.length(); i++) String line = null; while((line = br.readLine()) != null) { if(line.equalsIgnoreCase(name)) { return false; break; } } } bw.write(name); bw.newLine(); bw.flush(); bw.close(); return true; } catch(IOException e) { sendMessage("Error writing or reading the bans file"); } return false; }
