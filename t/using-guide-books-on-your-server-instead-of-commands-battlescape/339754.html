public void previousPage()
{
    if(page > 1)
    {
        page--;
        reloadPage();
    }
}

public void nextPage()
{
    if(page < book.length)
    {
        page++;
        reloadPage();
    }
}

public void reloadPage()
{
    for(int i = 0; i < book[page - 1].length; i++)
    {
        changeText126(book[page - 1][i], bookIDs[i]);
    }
    if(page > 1)
    {
    changeText126("@dre@Page " + (page - 1), 14165);
    } else
    {
    changeText126("", 14165);
    }
    if(page < book.length)
