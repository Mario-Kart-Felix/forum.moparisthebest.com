Yeah, it always clicks in the same spot, and it right-clicks, not left.

Try this:

program New;
 var x,y,XChange, YChange: integer;
begin
 repeat
  XChange:=random(3);
  YChange:=random(3);
  MoveMouseSmooth(353+XChange,97+YChange);
  GetMousePos(x,y);
  ClickMouse(x,y,True);
  Wait(1000+random(500));
until(false);
end.
