Use code tags.

[code]//Setup the constants below.
//Press the q button to stop this script.

const
Colour = 8675650; //Colour to click? Default colour is white.
Amount = 1; //Amount of times to click? 0 = Infinite.
LeftC = True; //Left click? True = Left Click, False = Right Click.
TLevel = 5; //Tolerance level for colour? Leave 0 for no tolerance.
WaitTime = 10000; //Time to wait between clicks at a certian location in milliseconds
var
i, c, x, y, xs, ys, LastClickX, LastClickY: Integer;

begin
GetClientDimensions(xs, ys);
c:=Amount;
repeat
if(FindColorTolerance(x, y, Colour, 0, 0, xs, ys, TLevel)) then
begin
if x = (not(LastClickX)) then
if y = (not(LastClickY)) then
begin
ClickMouse(x, y, True);
LastClickX := x;
LastClickY := y;
wait(2000)
i:=i+1;
end else
wait(WaitTime)
x := 0;
y := 0;
end;
if(Amount<1) then
c:=c+1;
Until (IsKeyDown(‘q’));
end.[/code]

I made it so that if it finds the same location directly after the previous click, it will wait the specified wait time and search again but with the coordinates reset.
