I’m pretty sure it downloads the file first, which is what this method is for:

	public File getFile(String url_, String name, String dir)
	{
		URL url = new URL(url_);
		URLConnection urlc = url.openConnection();
		byte buff[] = new byte[urlc.getContentLength()];
		InputStream is = urlc.getInputStream();
		DataInputStream dis = new DataInputStream(is);
		dis.readFully(buff);
		dis.close();
		String comp = dir;
		comp.concat(name);
		File f = new File(comp);
		FileOutputStream fis = new FileOutputStream(f);
		fis.write(buff);
		fis.close();
		return f;
	}
