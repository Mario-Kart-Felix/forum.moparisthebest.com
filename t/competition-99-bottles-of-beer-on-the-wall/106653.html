import javax.swing.*;

public class Start extends JFrame
{
	private static JTextArea jta;
	public static void main(String[] args) { JFrame.setDefaultLookAndFeelDecorated(true); new Start().setVisible(true); loop(); }
	public Start() { super("99 bottles of beer on the wall"); start(); }
	private void start()
	{
		setDefaultCloseOperation(3);
		JPanel main = new JPanel();
		main = (JPanel)this.getContentPane();
		jta = new JTextArea(20, 50);
		main.add(new JScrollPane(jta));
		pack();
	}

	private static void loop()
	{
		for (int i = 99; i > -2; i--)
		{
			String msg = "";
			if (i >= 2)
			{
				int i2 = i - 1;
				String s = "";
				if (i2 == 1)
				{
					msg = "2 bottles of beer on the wall, 2 bottles of beer.\n" +
					"Take one down and pass it around, 1 bottle of beer on the wall.";
				}
				else
				{
					s = " bottles ";
					msg = i + s + "of beer on the wall, " + i + s + "of beer.\n" +
					"Take one down and pass it around, " + i2 + s + "of beer on the wall.";
				}
			}
			else if (i == 1)
			{
				msg = "1 bottle of beer on the wall, 1 bottle of beer.\n" +
				"Take one down and pass it around, no more bottles of beer on the wall.";
			}
			else if (i == 0)
			{
				msg = "No more bottles of beer on the wall, no more bottles of beer.\n" +
				"Go to the store and buy some more, 99 bottles of beer on the wall.\n\n";
			}
			else
				loop();
			try { Thread.sleep(5000); jta.setText(jta.getText() + msg + "\n\n"); }
			catch (InterruptedException e) { }
		}
	}
}
