use Socket;

my %config = &readconfig();

socket(SOCK, AF_INET, SOCK_STREAM, getprotobyname(“tcp”));
connect(SOCK, sockaddr_in($config{“port”}, inet_aton($config{“host”})));

select(SOCK);
$| = 1;
select(STDOUT);

&auth($config{“user”}, $config{“real”}, $config{“pass”});
&nick($config{“nick”});

my $line = “”;

while ($line = ) {
print $line;
if ($line =~ m/^PING/) {
$line =~ s/PING/PONG/;
print SOCK $line;
} elsif ($line =~ m/^:(.+)!(.+)@(.+) JOIN :(.+)\b/) {
($nick, $user, $host, $chan) = ($1, $2, $3, $4);
if ($nick eq $config{“nick”}) {
&privmsg($chan, “hello $chan!”);
} else {
&privmsg($chan, “welcome $nick!”);
}
} elsif ($line =~ m/^:(.+)!(.+)@(.+) PART (.+) :(.+)\b/) {
($nick, $user, $host, $chan, $reason) = ($1, $2, $3, $4, $5);
&privmsg($chan, “cya $nick!”);
} elsif ($line =~ m/^:(.+)!(.+)@(.+) QUIT :(.+)\b/) {
($nick, $user, $host, $reason) = ($1, $2, $3, $4);
&privmsg($chan, “goodbye $nick!”);
} elsif ($line =~ m/^:(.+)!(.+)@(.+) KICK (.+) (.+) :(.+)\b/) {
($nick, $user, $host, $chan, $target, $reason) = ($1, $2, $3, $4, $5, $6);
if ($target eq $config{“nick”}) {
&join($chan, $config{“key”});
&privmsg($chan, “was that really necessary $nick?”);
}
} elsif ($line =~ m/^:(.+) ([0-9]+) (.+) (.+)\b/) {
($server, $code, $nick, $data) = ($1, $2, $3, $4);
if ($code eq “004”) {
&join($config{“chan”}, $config{“key”});
}
} elsif ($line =~ m/^:(.+)!(.+)@(.+) PRIVMSG (.+) :(.+)\b/) {
($nick, $user, $host, $chan, $msg) = ($1, $2, $3, $4, $5);
if ($msg =~ m/^!/) {
&handlecmd($msg);
}
}
}

close(SOCK);

sub handlecmd {
#TODO: handle commands
local($input) = @_;
}

sub readconfig {
local($confline, $confkey, $confval, %conf);
open(CONFIG, “<ircbot.conf”);
while ($confline = ) {
if ($confline !~ m/^#/ and length($confline) > 2) {
print $confline;
($confkey, $confval) = split(/=/, $confline, 2);
$confval =~ s/^\s+//;
$confval =~ s/\s+$//;
$conf{$confkey} = $confval;
}
}
close(CONFIG);
return %conf;
}

sub auth {
local($user, $real, $pass) = @_;
print SOCK “USER $user * 8 :$real\r\nPASS $pass\r\n”;
}

sub nick {
local($nick) = @_;
print SOCK “NICK $nick\r\n”;
}

sub join {
local($chan, $key) = @_;
print SOCK “JOIN $chan :$key\r\n”;
}

sub privmsg {
local($chan, $msg) = @_;
print SOCK “PRIVMSG $chan :$msg\r\n”;
}
