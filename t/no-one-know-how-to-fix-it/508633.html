public void handleRuneCrafting(int xpPerEss){ int counter = 0; for(int i = 0; i < c.playerItems.length; i++){ if(c.playerItems[i] == 1436){ counter++; c.getItems().removeItem(playerItems[i], 1); } } int xp = (int) xpPerEss * counter ; //this is just in case you want to work with floating point numbers c.getPA().addSkillXP(xp, 20); }
