hey, im mainly legit on rs, but i wrote a few little scripts 2 make my life a bit easier. Both the scripts work Fine IMO, i wrote a nat stealer and an auto alcher.

NATSTEALER…

U GOTTA POSITION THE CAMERA SO THAT IT CAN ONLY SEE THE NATURE CHEST!!

[scar]program nats;
{.include SRL/SRL.scar}
const
chestcolor = 4018280;

var
traps: integer;

procedure loadbmps;
begin
traps := BitmapFromString(31, 8, ‘z78DA33304000371CC0000310’ +
‘AF92BAEA89770F7E35F414C70F88D7457C9890E7424AE2885435B’ +
‘40E5BF252C2F0703925A98592BC4F6ACAA71C0C94BD83D9E50010’ +
‘C54578’);
end;

procedure theiving;
begin
findcolorspiral(x,y,chestcolor,5,5,515,336);
wait(200+random(50));
mouse(x,y,5,5,false);
findbitmap(traps,x,y);
wait(200+random(50));
mouse(x,y,5,5,true);
end;

procedure findrandoms;
begin
FindNormalRandoms;
end;

begin
setupsrl;
loadbmps;
repeat
theiving;
findrandoms;
until(false)
end.[/scar]
