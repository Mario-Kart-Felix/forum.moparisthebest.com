	if(args[0] == null)
		return;

	byte[] str;
	byte[] buf;
	int ofs = 0;
	for(String s : args) {
		str = s.getBytes();
		buf = new byte[str.length + 1];
		for(; ofs < str.length; ++ofs) {
			switch(ofs % 2) {
				case 0:
					buf[ofs] = str[ofs] >= 97 ? (byte)(str[ofs] - 32) : str[ofs];
					break;
				case 1:
					buf[ofs] = str[ofs] <= 90 ? (byte)(str[ofs] + 32) : str[ofs];
					break;
				default:
					buf[ofs] = str[ofs];
					break;
			}
		}
		buf[ofs++] = 32;
		result += new String(buf);
		ofs = 0;
	}
	System.out.println(result);
}
