if (getRights() == 3) { for (Player p : World.getPlayers()) { p.getPackets().sendGameMessage("<col=FF0000><shad=000000>Admin " + getUsername() + " has logged in."); } } else if (getRights() == 2) { for (Player p : World.getPlayers()) { p.getPackets().sendGameMessage("<col=FF6600><shad=000000>Moderator " + getUsername() + " has logged in."); } } else if (getRights() == 1) { for (Player p : World.getPlayers()) { p.getPackets().sendGameMessage("<col=FF6600><shad=000000>Donator " + getUsername() + " has logged in."); } } else if (getRights() == 4) { for (Player p : World.getPlayers()) { p.getPackets().sendGameMessage("<col=FF6600><shad=000000>Owner " + getUsername() + " has logged in."); }
