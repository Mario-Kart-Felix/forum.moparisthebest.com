private Entry entries[];

public void put(K key, V value) {
	int hash = key.hashCode() % entries.length;
	Entry entry = new Entry(key, value), old = entries[hash];
	if (old != null)
		entry.linked = old;
	entries[hash] = entry;
}

public V get(K key) {
	int hash = key.hashCode() % entries.length;
	Entry entry = entries[hash];
	while (entry != null) {
		if (key.equals(entry.key))
			return (V) entry.value;
		entry = entry.linked;
	}
	return null;
}

public void remove(K key) {
	int hash = key.hashCode() % entries.length;
	Entry entry = entries[hash], previous = entry;
	if (entry == null)
		return;
	if (entry.linked == null) {
		entries[hash] = null;
		return;
	}
	while (entry != null) {
		if (key.equals(entry.key)){
			previous.linked = entry;
			return;
		}
		previous = entry;
		entry = entry.linked;
	}
}

private class Entry<K, V> {

	public Entry(K key, V value) {
		this.key = key;
		this.value = value;
	}

	public K key;
	public V value;
	public Entry linked;

}
