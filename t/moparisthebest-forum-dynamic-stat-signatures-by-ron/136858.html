function str_between($str, $betw, $betw2)
{
  $str = " " . $str;
  $pos = strpos($str, $betw);
  if ($pos > 0)
  {
    $str2 = substr($str, $pos + strlen($betw));
    $pos2 = strpos($str2, $betw2);
    if ($pos2 == 0)
      return '';
    return substr($str2, 0, $pos2);
  }
  else
    return '';
}
