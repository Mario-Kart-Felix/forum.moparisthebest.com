program AutoFighter;
{.Include SRL/SRL.Scar.}
{.Include SRL/SRL/Skill/Fighting.Scar.}
Var
WX, WY, MonstersToKill : Integer;

//===============Setup==================\
Const
Tol = 1;//Tolerace if found the monster\
MonsterColor = 0;//The color of the monster\
WanaEat = True;//Do you want to eat\
FoodColor = 0;//color of your food\
MonsterColor1 = 0;//Get a different color of the monster\
EatHp = 50;//What do you want your hp to be before eating\
FightMode = 2;// 1=Attack 2=Strengh 3=Balanced 4=Defence\
MonsterColor2 = 0;//Try get a different color of the monster\
MonsterName = ‘Goblin’;//The name of the monster\
TurnOffChats = False;//Do you want to turn off chats?\
GenieSkill = ‘Strengh’;//What skill to use if a genie is found?\
MonsterssTokill = 20;//How many monster do you want to kill?\
DisguiseScarAs = ‘Google - FireFox’;//What do you want to Diguise scar as?\
//==============End Of Setup=============\

Procedure FindMonsterr;
Begin
Wait(1000+random(50));
KAttackMonster(MonsterColor, MonsterColor1, MonsterColor2, Tol, MonsterName);
Wait(5000+random(1000));
MonstersToKill := MonstersToKill + 1;
End;

Procedure FightingMode;
Begin
SetFightMode(FightMode);
End;

Procedure Chats;
Begin
If TurnOffChats = True Then
Begin
ChatsOff;
End;
End;

Procedure LetsEat;
Begin
If Wanaeat = True then
Begin
wait(200)random(100)
RealEatIfNeeded(FoodColor,EatHp,WanaEat);
End;
End;

Procedure Randoms;
Begin
FindDead;
FindMod;
FindMime;
SolveSandWich;
FindMaze;
FindTalk;
FindDemon;
FindScapeRune;
DwarfItem;
FindLamp(GenieSkill);
SolveFrog;
FindQuiz;
FindTalk;
If (FindNewBox) then
Begin
writeln(‘Found Strange Box Trying To Solve…’)
OpenBox;
SolveBox;
wait(1000);
End;
If (FindNewBox) then
Begin
GambleNewBox;
End;
If (FindNewBox) then
Begin
LogOut;
Writeln(‘Unable To Solve Box…Terminating’);
TerminateScript;
End;
End;

Procedure ProgressReport;
Begin
AddToReport(’==========Progress Report=======’)
AddToReport(’==Killed ’ + inttostr(MonstersToKill) + ’ monsters’);
AddtoReport(’==This Autofighter was created by Arr0w’)
AddToReport(’==From moparisthebest and villu-reborn’)
AddToReport(’==To check for updates check moapristhebest’)
End;

Procedure ModSetRun(Run: Boolean);
Begin
GameTab(11);
If Run Then
Begin
If (Not (FindColorTolerance(WX, WY, 1711220, 663, 431, 665, 433, 5))) Then
Begin
Mouse(648, 431, 20, 20, true);
Wait(100+Random(100));
End;
End Else
Begin
If FindColorTolerance(WX, WY, 1711220, 663, 431, 665, 433, 5) Then
Begin
Mouse(648, 431, 20, 20, true);
Wait(100+Random(100));
End;
End;
End;

Begin
ClearDebug;
DisguiseScar(DisguiseScarAs)
ProgressReport;
SetupSRl;
ActivateClient;
Chats;
ModSetRun(True);
Repeat
FindMonsterr;
Randoms;
LetsEat;
Until(Monstersstokill = MonstersToKill)
If MonsterssToKill = MonstersToKill Then
Begin
OpenWebPage(‘http://www.flamed-bigmac.co.nr/forums’)
End;
End.[/scar]
Thank you.
