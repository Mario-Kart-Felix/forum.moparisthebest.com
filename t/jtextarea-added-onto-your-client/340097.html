What this does is let’s you print out things in a box similar to the CMD box although it’s customizable.

Add this file into your client files as JTextAreaZ.java

[code]
import java.awt.FlowLayout;
import javax.swing.*;
import javax.swing.text.Document;

public class JTextAreaZ {

public JTextAreaZ() {

}

public static void start() {
	JFrame.setDefaultLookAndFeelDecorated(true);
	JFrame jframe = new JFrame("Client Info Box");
	jframe.setLayout(new FlowLayout());
	jframe.setDefaultCloseOperation(1);
	String s = "Welcome to CLIENT NAME HERE's client!";

	/*
	 *This sets up an icon instead of the default java one*
	 */
	/*jframe.setIconImage(new ImageIcon("cacheDirectory/clientIconName.png").getImage());*/

	JScrollPane jscrollpane = new JScrollPane(textArea2, 20, 30);
	textArea2.setEditable(false);
	textArea2.setLineWrap(true);
	jframe.setResizable(false);
	jframe.add(jscrollpane);
	jframe.pack();
	jframe.setVisible(true);
}

public static void append(String s) {
	textArea2.append((new StringBuilder()).append(s).append("\n").toString());
	textArea2.setCaretPosition(textArea2.getDocument().getLength());
}

public static void main(String args[]) {
	SwingUtilities.invokeLater(new Runnable() {

		public void run() {
			JTextAreaZ.append("run");
		}

	});
}

static JTextArea textArea2 = new JTextArea(10, 46);

}[/code]
