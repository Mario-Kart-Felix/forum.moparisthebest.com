Automatically sings a thousand bottles of beer on the wall - make sure your bank window is open because this will also take out and put back in stuff in your bank so that you dont get any randomevents ;D
[scar]program beer;
var v:integer;
procedure clack;
begin
wait(random(5000)+5000);
sendkeys(inttostr(v)+’ bottles of beer on the wall’+chr(13));
wait(random(5000)+5000);
movemousesmooth(94,77);
wait(random(500));
clickmouse(94,77,true);
wait(random(500));
movemousesmooth(584,230);
wait(random(500));
clickmouse(584,230,true);
wait(random(500));
sendkeys(inttostr(v)+’ bottles of beer’+chr(13));
wait(random(5000)+5000);
sendkeys(‘take one down, pass it around’+chr(13));
wait(random(5000)+5000);
x:=x-1;
sendkeys(inttostr(v)+’ bottles of beer on the wall’+chr(13));
wait(random(5000)+5000);
end;
procedure none;
begin
sendkeys(‘no bottles of beer on the wall’+chr(13));
wait(random(5000)+5000);
sendkeys(‘no bottles of beer’+chr(13));
wait(random(5000)+5000);
sendkeys(‘Everybody on rs2 lying on the ground and dead’+chr(13));
wait(random(5000)+5000);
end;

begin
repeat
v:=1000;
repeat
clack;
until(v=0);
none;
until(false);
end.[/scar]
