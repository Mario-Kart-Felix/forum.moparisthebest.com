LinkedList:

public class List<T extends Node> {

	public final Node head;

	public List() {
		this.head = new Node();
		head.previous = head.next = head;
	}

	@SuppressWarnings("unchecked")
	public T peek() {
		return (T) head.next;
	}

	@SuppressWarnings("unchecked")
	public T peekLast() {
		return (T) head.previous;
	}

	@SuppressWarnings("unchecked")
	public T pop() {
		Node node = head.next;
		if (node == head)
			return null;
		node.unlink();
		return (T) node;
	}

	@SuppressWarnings("unchecked")
	public T pollLast() {
		Node node = head.previous;
		if (node == head)
			return null;
		node.unlink();
		return (T) node;
	}

	public void push(Node node) {
		addAfter(head, node);
	}

	public void offer(Node node) {
		addBefore(head, node);
	}

	public boolean addAfter(Node node, Node add) {
		return addBefore(node.next, add);
	}

	public boolean addBefore(Node node, Node add) {
		if (contains(add))
			return false;
		(add.previous = node.previous).next = add;
		(add.next = node).previous = add;
		return true;
	}

	public boolean contains(Node node) {
		for (Node next = head.next; next != head; next = next.next)
			if (node == next || node.equals(next))
				return true;
		return false;
	}

	public boolean remove(Node node) {
		for (Node next = head.next; next != head; next = next.next)
			if (node == next) {
				node.unlink();
				return true;
			}
		return false;
	}

	public boolean isEmpty() {
		// return head.next == head;
		return size() == 0;
	}

	public void clear() {
		Node node = head.next;
		while (node != head) {
			Node next = node.next;
			node.next = node.previous = null;
			node = next;
		}
		head.next = head.previous = head;
	}

	public int size() {
		Node node = head.next;
		int count = 0;
		while (node != head) {
			node = node.next;
			count++;
		}
		return count;
	}
}

Node:

public class Node {

	public Node previous, next;
	public long identifier;

	public final void unlink() {
		if (next != null && previous != null) {
			previous.next = next;
			next.previous = previous;
		}
		this.next = this.previous = null;
	}
}

Edit: here’s a quick remove-by-index method, presuming you are iterating forward, and the node at index = 0 is the first non-head node in a nonempty list:

	public boolean remove(int index) {
		int count = 0;
		Node node = null;
		for (Node next = head.next; next != head; next = next.next) {
			if (count++ == index) {
				node = next;
				break;
			}
		}
		if (node == null)
			return false;
		node.unlink();
		return true;
	}
