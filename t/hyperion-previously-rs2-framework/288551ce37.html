Build.xml

<project name="hyperion" default="compile" basedir=".">
	<property name="src" location="src" />
	<property name="build" location="build" />
	<property name="lib" location="lib" />
	<property name="jar" location="hyperion.jar" />

	<target name="compile">
		<delete file="${jar}" />
		<delete dir="${build}" />
		<mkdir dir="${build}"/>
		<javac srcdir="${src}" destdir="${build}" debug="on" target="1.5">
		  <classpath>
			<pathelement location="${lib}/jython.jar" />
			<pathelement location="${lib}/commons-compress-1.0.jar" />
			<pathelement location="${lib}/slf4j-api-1.5.8.jar"/>
			<pathelement location="${lib}/mina-core-2.0.0-M6.jar"/>
			<pathelement location="${lib}/junit-4.6.jar" />
			<pathelement location="${lib}/slf4j-jdk14-1.5.8.jar" />
			<pathelement path="${java.class.path}/"/>
		  </classpath>
		</javac>
		<jar basedir="${build}" destfile="${jar}" />
	</target>
	
	<target name="runserver">
		<java classname="org.hyperion.Server" fork="true">
			<jvmarg value="-Xms512m" />
			<jvmarg value="-Xmx1024m" />
			<classpath>
			<pathelement location="${lib}/jython.jar" />
			<pathelement location="${lib}/commons-compress-1.0.jar" />
			<pathelement location="${lib}/slf4j-api-1.5.8.jar"/>
			<pathelement location="${lib}/mina-core-2.0.0-M6.jar"/>
			<pathelement location="${lib}/junit-4.6.jar" />
			<pathelement location="${lib}/slf4j-jdk14-1.5.8.jar" />
			<pathelement path="${jar}/" />
			</classpath>
		</java>
	</target>
</project>
