Here, I kinda fixed it up so you get the point.
I hope I helped.

[php]program traderdropper;
{.include srl/srl.scar}
{.include srl/srl/misc/trade.scar}

//const – didnt need this.
// x=‘true’;

//var; – if there are no variables, you do not need this.

procedure acceptit;
begin
AcceptTrade; // didnt need the true false, will accept a trade when
// it sees it.
end;

procedure drop;
begin
//use either of the 2:
//dropto(1,28); – will drop from the specified slot, to the other
//specified slot.
//dropall; – will drop everything.
end;

begin
setupsrl; //need this
repeat
acceptit;
until(tradescreen) or not(tradescreen) and not(loggedin)
if tradescreen then
begin
//blah blah what you want to go here
if not(tradescreen) and not(loggedin) then
begin
// blah blah what you want to go here
end;
end;
end.[/php]
