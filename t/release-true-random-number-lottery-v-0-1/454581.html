Hello,
As a fanatic of true random numbers I have noticed to my great surprise that previous RSC lotteries were Pseudorandom, meaning that they were generated by software algorithmic. Therefore, it would render the lotteries not random nor “real”. In a attempt facilitate this issue I have constructed (with minimal help from other sources) a new lottery system, with more advanced features and a True random number generator. I first discovered a library for interfacing with a free popular true random number generator service, Random.org, which utilizes atmospheric noise as the seed for producing random numbers. Below is the source code from this project, which obviously you are allowed to implement in future projects.

package org.rscdaemon.server.lotto;

import java.util.ArrayList;
import org.rscdaemon.server.model.Player;
import org.rscdaemon.server.model.World;
import org.rscdaemon.server.model.InvItem;
import java.io.IOException;

public class lottery{
	public static Player player;
	private static boolean lottoStatus = false;
	public static int lottoAmount = 0;
	public static int lottoCost = 0;
	public static int numOfPlayers = -1; //set to -1 to balance out arraylist
	public static ArrayList<Player> tickets = new ArrayList<Player>();
	public static ArrayList<String> tickets2 = new ArrayList<String>();
	public static final World world = World.getWorld();
	public static double lotteryRatio = 1; //edit this to number between 0 and 1, to only enter a certain amount of the fee to the lotto total, example 0.5 will add half to the total. To subsidize the cost use a number above 1

	public static void startLottery(int lottoGp, double lottonewratio){ //Start Lotto, set price
		lotteryRatio = lottonewratio;
		lottoCost = lottoGp;
		lottoStatus = true;
	}
	public static boolean checkStatus(){ //Is it running?
		return lottoStatus;
	}
	public static int checkAmount(){	//amount up for grabs
		return lottoAmount;
	}
	public static int checkCost(){ //lottery cost
		return lottoCost;
	}	
	public static void enterLottery(Player p){ //Gives a player a lottery number
		if(p.getInventory().countId(10) < lottoCost){
			p.getActionSender().sendMessage("@gre@You need@red@ " + lottoCost + " @gre@gold coins in you inventory to enter the lottery");
		}else{
			p.getInventory().remove(10, lottoCost);
			p.getActionSender().sendInventory();
			tickets.add(p);
			tickets2.add(p.getUsername());
			lottoAmount += Math.ceil(lottoCost * lotteryRatio);
			numOfPlayers++;
			p.getActionSender().sendMessage("@gre@You have entered the lottery");
		}
	}
	public static void drawLotto() throws IllegalArgumentException, IOException{ //Draws a winner using atmospheric noise via random.org
		random lottoGen = new random();
		int lotteryNumberWinner = lottoGen.getSingleRandomInt(0, tickets.size());
		Player p = tickets.get(lotteryNumberWinner);
		InvItem coins = new InvItem(10);
		coins.setAmount(lottoAmount);
		p.getInventory().add(coins);
		p.getActionSender().sendInventory();
		p.getActionSender().sendMessage("@gre@Congratulations, You have won the lottery, @red@ " + lottoAmount + " @gre@ gold pieces have been added to your inventory");
		world.sendToAll("@gre@The lottery has ended @red@" + p.getUsername() + " has won");
			tickets = new ArrayList<Player>();
			tickets2 = new ArrayList<String>();
			numOfPlayers = -1;
			lottoAmount = 0;
			lottoCost = 0;
			lottoStatus = false;
	}
}

/** Java Random Numbers - Random.Org API
Credits go to Pavelps Scheidl
Edited by X1f3nt to remove unneeded method
*/



package org.rscdaemon.server.lotto;

import java.io.IOException;
import java.net.*;
import java.util.Scanner;

public class random
{
    private int currentDefaultTimeout;
    public random()
    {
        currentDefaultTimeout = 10000;
    }

    public random(int timeoutMillis)
    {
        currentDefaultTimeout = 10000;
        currentDefaultTimeout = timeoutMillis;
    }
    public int getSingleRandomInt(int min, int max) throws IOException, IllegalArgumentException{
        int obtainedInteger = 0;
        try{
            URL url = new URL((new StringBuilder("http://www.random.org/integers/?num=1&min=")).append(min).append("&max=").append(max).append("&col=1&base=10&format=plain&rnd=new").toString());
            URLConnection randomOrgConnection = url.openConnection();
            randomOrgConnection.setConnectTimeout(currentDefaultTimeout);
            Scanner randomScan = new Scanner(randomOrgConnection.getInputStream());
            if(randomScan.hasNextInt()){
                obtainedInteger = randomScan.nextInt();
            } else {
                IOException exception = new IOException("Connection open, input obtained, no numbers found.");
                throw exception;
            }
        }
        catch(MalformedURLException e)
        {
            e.printStackTrace();
        }
        catch(IOException e)
        {
            IOException exception = new IOException("Connection open, no input obtained.");
            throw exception;
        }
        return obtainedInteger;
    }
}

[size=14pt]Commands explained[/size]
::startlottery (Admin) Used by Admins to start the lottery. Takes to arguments: 1 Gold piece cost 2 ratio (this is the ratio of the actual cost of the lottery and the amount inserted into the lottery prize amount, use a number between 0 and 1 to have a lower amount and a number above 1 to subsidize inserted amounts)

::stoplottery (Admin) Used by Admins to stop the lottery, interfaces with random.org to get a random number to chose a winner and give them the reward

::lottery Used by players to buy a ticket, removes the cost amount from the player’s inventory

::lotterycost Used by players to view cost of lottery

::lotteryamount Used by players to view lottery prize amount

::lottocp (Admin) Used by Admins to view stats on the lottery
