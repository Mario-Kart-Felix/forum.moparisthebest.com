package com.main;

import java.net.*;
import java.io.*;

/**
 * 
 * @author Emily
 * @author Java Network Programming
 */

public class Main {

	/*
	 * This program uses two threads, one to handle input from the client
	 * and the other to send output from the server.
	 */
	public static void main(String[] args) {
		int PORT;
		ServerSocket ss;
		Socket theConnection;
		try {
			PORT = Integer.parseInt(args[0]);
		}
		catch(Exception e) {
			PORT = 0;
		}
		try {
			ss = new ServerSocket(PORT);
			System.out.println("Listening on port "+ss.getLocalPort());
			while (true) {
				theConnection = ss.accept();
				System.out.println("Connection established with "+theConnection);
				InputThread it = new InputThread(theConnection.getInputStream());
				it.start();
				OutputThread ot = new OutputThread(theConnection.getOutputStream(), it);
				ot.start();
				//need to wait for ot and it to finish
				try {
					ot.join();
					it.join();
				}
				catch (InterruptedException e) {
					
				}
			}
		}
		catch (IOException e) {
			
		}
	}
}
/*
 * Contains a single filed, data will b read from one byte at a time
 * Each byte is cast to a char 'c', and printed on System.out
 */
class InputThread extends Thread {
	
	InputStream is;
	
	public InputThread(InputStream is) {
		this.is = is;
	}
	public void run() {
		try {
			while (true) {
				int i = is.read();
				if(i == -1)
					break;
				char c = (char) i;
				System.out.print(c);
			}
		}
		catch (IOException e) {
			System.err.println(e);
		}
	}

}
/*
 * Sends output to the client
 * Its constructor contains two arguments, and output stream for sending
 * data to the client, and the Input Thread
 */
class OutputThread extends Thread {

	PrintStream ps;
	DataInputStream is;
	InputThread it;
	
	public OutputThread(OutputStream os, InputThread it) {
		ps = new PrintStream(os);
		this.it = it;
		is = new DataInputStream(System.in);
	}
	public void run() {
		String line;
		try {
			while (true) {
				line = is.readLine();
				if(line.equals("."))
					break;
				ps.println(line);
			}
		}
		catch (IOException e) {
			
		}
		it.stop();
	}
}
