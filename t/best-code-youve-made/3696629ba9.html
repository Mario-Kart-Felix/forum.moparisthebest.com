this has no practical purpose that i can think of. this is an extension of a lab we did in class(web basics). we had to make a container(no object, it was just a table). and repeat the content and containers.

what this is/does.

it is a container object.

asks you how many containers and how many columns of containers you want.

li tags are hidden. li tags contain the content of the containers and how many time the content should repeat.

when the containers are to be wrote, a function stores the array of li tags’ info in to a 2d array of container objects.

the layout of the array represents the layout of how the containers will be arranged.

it has some checks in it to make the code perform like it was designed.

this isnt my best code but its some that i wrote recently.

<script type="text/javascript" src="container-object.js"></script>
<style type="text/css"> 
ul{
	visibility:hidden;
	position:absolute;
	left:0px;
	top:0px;
}
</style>

<body>
	<ul>
    <script type="text/javascript">
	var timesLooped = 1;
	var loops=prompt("how many containers do you want?","50");
	while(timesLooped<=loops){
		document.write('<li id="'+timesLooped+'">This is the content of the container of the first row, first column. </li>');
		timesLooped++;
	}
	
    </script>
	</ul>
 
	<table cellspacing="10px">
		<script type="text/javascript">
		
		var containers = new Array();
		
		function populateContainerArray(c){// c is how many columns or cells/containers per row
			var cC = document.getElementsByTagName('li');//makes an array of li tags
			var b = 0;//the current li tag
			var isCorrect = false;//should u continue making columns?
			
			//repeats until a "c"th(the # c is) of the array cC is reached
			for(var a=0;a<Math.ceil(cC.length/c);a++){
				if(c>cC.length){
					c = cC.length;
					//alert('The number of columns specified is more than how many containers there are... but i fixed it; there will be no error caused by this occurrence. Np.');
				}
				containers[a] = new Array(c);//makes a new array/row
				var e = 0;//current column
				while(!isCorrect){//when isCorrect is false loop
					if(b<cC.length){//does the next li tag exist? yes
						containers[a][e] = new container(cC[b].innerHTML,cC[b].id);//create a new container where it should be.
					} else {//no more li tags? end
						isCorrect = true;//exit loop
					}
					if(e==c-1){//is the row full? yes
						isCorrect = true;//exit loop
					}
					b++;
					e++;
				}
				isCorrect = false;//reset isCorrect
			}
			
		}
		
			//loop the rows of the table, for each row of containers
			var columns=prompt("how many columns do you want?","24");
			populateContainerArray(columns);
			
			for(var r =0;r<containers.length;r++){
			
				document.write('<tr>');
				
				//loop the cells of the table, for each of the containers
				for(var c = 0;c<containers[r].length;c++){
				
					document.write('<td>');
					document.write(containers[r][c].tableTop);
					
					//loop the content of the containers
					for(var rC = 0;rC<containers[r][c].repeats;rC++){
					document.write(containers[r][c].content);
					}
					
					document.write(containers[r][c].tableBottom);
					document.write('</td>');
					
				}//end of for(c <= ...
				
				document.write('</tr>');
				
			}//end of for(r <= ...
			
		</script>
	</table>
	
</body>

		this.content = content;
		this.tableBottom = '\
					</td>\
				</tr>\
				<tr>\
					<td style="background-image:url(\'container-bottom380x15.gif\'); height:15px; background-repeat: no-repeat"></td>\
				</tr>\
			</table>';
		this.repeats = Math.floor(repeatN);
	}//end of container object
