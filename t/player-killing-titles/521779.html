[code]package neitiznot.model.content.combat.wilderness;

import java.util.HashMap;
import java.util.Map.Entry;

import neitiznot.model.players.Client;

/**
*

  • @author Rhubarb/Int Bauk
  • Still doing.

*/
public class PlayerTitle {

/**
 * Gets the display interface.
 */
private static final int interface_id = 0;
public int getDisplayinterface() {
	return interface_id;
}

/**
 * Boolean for the players title.
 */
public static boolean hasTitle = false;

/**
 * Custom title input.
 */
private static String custom_title;
private static String getTitle() {
	return custom_title;
}

/**
 * Hashmap to store all our titles: ("title", pk points);
 */
private static HashMap<String, Integer> titles = new HashMap<String, Integer>();
static {
	titles.put("Newbie", 0);
	titles.put("Killer", 50);
	titles.put("Destroyer", 100);
	titles.put("Beast", 200);
	titles.put("Legendary", 300);
	titles.put("Veteran", 300);
	titles.put(getTitle(), 1000);
}

/**
 * Handles the players title.
 * @param player
 */
public void handleTitle(Client player) {
	for(Entry<String, Integer> data : titles.entrySet()) {
		String title = data.getKey();
		Integer points = data.getValue();
		if(player.pkPoints == points) {
			player.sendMessage("You've been rewarded the title: " + title + ".");
		} else if(hasTitle) {
			player.sendMessage("Your current title is: " + title + ".");
		}
		
	}
}

}
