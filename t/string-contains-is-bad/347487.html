public boolean endsWith(StringBuilder builder, String s) { int offset = s.length(); int typedOffset = builder.length() - offset; int sOffset = 0; if (typedOffset < 0) { return false; } while (--offset >= 0) { if (builder.charAt(typedOffset++) != s.charAt(sOffset++)) { return false; } } return true; }
