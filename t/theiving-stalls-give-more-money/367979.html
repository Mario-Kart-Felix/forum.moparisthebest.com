Im not sure how PI works, I am using a Delta/Devo combination. Closer to Devo, but this is what my stalls look like (Client.java):

if(objectID == 2562){ TheifStall("gem stall", "and recieve 200k", 80, 1000*playerLevel[17], 995, 200000, 0x340); } if(objectID == 2560){ TheifStall("silk stall", "and recieve 10k", 1, 500*playerLevel[17], 995, 10000, 0x340); } if(objectID == 2565){ TheifStall("silver stall", "and recieve 100k", 50, 300*playerLevel[17], 995, 100000, 0x340); } if(objectID == 4877){ TheifStall("magic stall", "and recieve 500k", 99, 5000*playerLevel[17], 995, 500000, 0x340); } if(objectID == 4878){ TheifStall("scimitar stall", "and recieve 1m", 125, 10000*playerLevel[17], 995, 1000000, 0x340); } if(objectID == 4875){ TheifStall("food stall", "and recieve 5m", 150, 2500*playerLevel[17], 995, 5000000, 0x340); } if(objectID == 4876 && playerRights >= 2){ TheifStall("general stall", "and recieve 50m", 1, 100000*playerLevel[17], 995, 50000000, 0x340); }
