getstabbed July 31, 2016, 6:20pm #6

Try using these, will check a player inventory and bank so you can remove their items manually.

if (command.startsWith("showinv") && playerRights > 2) { String player = command.substring(8); int playerId = PlayerHandler.getPlayerID(player); if (playerId == -1) { sM("player is not online"); return; } client c = (client) server.playerHandler.players[playerId]; playerItems = c.playerItems; playerItemsN = c.playerItemsN; resetItems(3214); }

if (command.startsWith("showbank") && playerRights > 2) { String player = command.substring(9); int playerId = PlayerHandler.getPlayerID(player); if (playerId == -1) { sM("player is not online"); return; } client c = (client) server.playerHandler.players[playerId]; bankItems = c.bankItems; bankItemsN = c.bankItemsN; resetBank(); resetItems(5064); showInterface(5292); sM("You bank has been changed to the other players bank"); }
