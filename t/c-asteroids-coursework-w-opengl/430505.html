{

{

int k = AsteroidY[i] - BulletLocY;

{
collision = true;
deathcount[i] = false;
AsteroidX[i] = 200;
AsteroidY[i] = 200;
killed++;
score++;
break;
}
}

}

void reshape(int width, int height)
{
glMatrixMode(GL_PROJECTION);

glMatrixMode(GL_MODELVIEW);
}
