Am currently writing for you

This is a base. You didn’t really specify how people won etc so I had to improvise. Currently the players get put in the game after 10 minutes, every ten minutes a new game starts whether or not old one was finished, only one game can be running at once. If two games run over no new juggernaut will be picked and the old one will be used.

There is no way to win except the juggernaut killing everyone.

Also if you want to use this you will have to work out how to implement it this is just the class and I am quite tired now so yeah 

import java.util.ArrayList;
/**
 * A class handling the juggernaut minigame
 * @author Tatemo
 */

public class Juggernaut {

	/**
	 * All of the players in the game.
	 */
	public ArrayList<Client> playersInGame = new ArrayList<Client>();

	/**
	 * Contains players in waiting room
	 */
	public ArrayList<Client> playersInWait = new ArrayList<Client>();

	/**
	 * Contains players which died
	 */
	public ArrayList<Client> playersWhichDied = new ArrayList<Client>();
	
	/**
	 * The juggernaut.
	 */
	public Client juggernaut;

	/**
 	 * The game start timer
	 */
	public int gameStartTimer = 600;

	/**
	 * Players remaining alive
	 */
	public int playersAlive;

	/**
	 * Contains all the players
	 */
	public Client allPlayersInWait() {
		for (Client player: playersInWait) {
			if(player != null)
				return player;
		}
	}
	
	/**
	 * Contains all dead players
	 */
	public Client allPlayersWhichDied() {
		for (Client player: playersWhichDied) {
			if(player != null)
				return player;
		}
	}

	/**
	 * Contains all players in game
	 */
	public Client allPlayersInGame() {
		for (Client player: playersInGame) {
			if(player != null)
				return player;
		}
	}
	
	/**
	 * Chooses the juggernaut
	 */
	public void chooseJuggernaut() {
		while(juggernaut == false) {
			for(Client player : playersInGame) {
				if(player != null && Misc.random(playersInGame.length) == 0) {
					juggernaut = player;
				}
			}
		}
	}


	/**
	 * Handles the process of the game
	 */
	public void process() {
		for(Client player : playersInGame) {
			if (player.isDead == true) {
				player.teleportOutOfGame();
				remove.playersInGame(player);
			}
		}
		gameStartTimer--;
		if (gameStartTimer == 300) {
			allPlayersInWait().sendMessage("The game will start in 5 minutes");
		} else if (gameStartTimer == 0) {
			allPlayersInWait().teleportToGame();
			add.playersInGame(allPlayersInWait());
			playersAlive = playersInWait.length;
			remove.playersInWait(allPlayersInWait());
			gameStartTimer = 600;
		}
		playersAlive = playersInGame.length;
		if (playersAlive == 1)
			endGame(false);
	}

	/**
	 * Ends the game when only juggernaut remains
	 */
	public void endGame(boolean juggernautLoggedOut) {
		if (juggernautLoggedOut) {
			allPlayersWhichDied().sendMessage("The juggernaut logged out");
			allPlayersInGame().sendMessage("The juggernaut logged out");
			allPlayersInGame().teleportOut();
			remove.playersInGame(allPlayersInGame());
			remove.playersWhichDied(allPlayersWhichDied());
		}
		if (!juggernautLoggedOut) {
			allPlayersWhichDied().sendMessage("The juggernaut won!");
			remove.playersWhichDied(allPlayersWhichDied());
		}	
		juggernaut = null;
	}
}
