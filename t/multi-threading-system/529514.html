package thread;

import java.util.LinkedList;
import java.util.Queue;
import java.util.concurrent.Executors;
import java.util.concurrent.ScheduledExecutorService;
import java.util.concurrent.TimeUnit;

public class ExecutionThreadListener {

	public enum ExecutionThreadState {
		ACTIVE, PENDING, INACTIVE
	}
	
	public static final ExecutionThreadListener instintation = new ExecutionThreadListener();

	public ScheduledExecutorService mainProcessor = Executors.newScheduledThreadPool(Runtime.getRuntime().availableProcessors());

	private Queue<ExecutionThread> executionThreadQueue = new LinkedList<ExecutionThread>();

	public Queue<ExecutionThread> getExecutionThreadQueue() {
		return executionThreadQueue;
	}

	public ScheduledExecutorService getMainProcessor() {
		return mainProcessor;
	}

	public void setExecutionThreadQueue(Queue<ExecutionThread> echoQueue) {
		this.executionThreadQueue = echoQueue;
	}

	public void setMainProcessor(ScheduledExecutorService mainProcessor) {
		this.mainProcessor = mainProcessor;
	}

	public synchronized void submitNewThread(final ExecutionThread thread) {
		if (thread.getState() == ExecutionThreadState.PENDING) {
			getExecutionThreadQueue().add(thread);
			thread.setState(ExecutionThreadState.ACTIVE);
		}
		thread.getExecutionService().scheduleAtFixedRate(new Runnable() {

			@Override
			public void run() {
				if (getExecutionThreadQueue().contains(thread)) {
					if (thread.getState() == ExecutionThreadState.ACTIVE) {
						thread.execute();
					} else {
						thread.destruct();
						getExecutionThreadQueue().remove(thread);
					}
				} else {
					thread.setState(ExecutionThreadState.INACTIVE);
				}
			}
		}, 0, thread.getExecutionDelay(), TimeUnit.MILLISECONDS);
	}
}

Looks pretty good, but the main processor was the default service. If a thread had a high memory requirement it would define a new service, if it had a low one it would specify the main processor. This way your could have a synchronization between multiple-threads with different services without having to either cram everything into one service or re-write code multiple times over to accommodate new additions.

PS. I actually wrote this for a 2D game but it could be applied anywhere attributes need to be processed.
