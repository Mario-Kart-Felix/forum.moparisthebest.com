You can easily use PHP to spoof headers.

$thesearemyheaders = array( 'http' => array('method' => "POST",
'header'=> "accept-language: en\r\n" .  "Host: $host\r\n" .
  "Referer: $moparisthebestreffererurl\r\n" .
  "Content-Type: application/x-www-form-urlencoded\r\n" . "Content-Length: 33\r\n\r\n" . "POSTshithere\r\n"));
$stream = stream_context_create($thesearemyheaders);
$f = fopen($statuspagevotesubmiturl, 'r', false, $stream);
fpassthru($f);
fclose($f);
