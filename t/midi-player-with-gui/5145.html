public int getFileNumber(File file) 
{
	File dir = new File("./MIDI");
	File files[] = dir.listFiles();
	for (int i = 0; i < files.length; i++) 
	if(files[i].equals(file))  
	return i;

return 0;
}

private void midiPlayer(String s)
{
    if(sequencer != null && sequencer.isRunning())
    {
        sequencer.stop();
        sequencer.close();
    }
    if(s.equals("Next")) {
        midiCount++;
	midiPlayer("Play");
    } else
  if(s.equals("Previous")) {
        midiCount--;
	midiPlayer("Play");
    } else
    if(!s.equals("Stop")) 
        try
        {
            String as[] = file.list();
            if(as.length > 0)
            {
                if(midiCount > as.length - 1)
                    midiCount = as.length - 1;
                if(midiCount < 0)
                    midiCount = 0;
                System.out.println("Playing " + as[midiCount]);
                file1 = new File("MIDI/" + as[midiCount]);
                try
                {
                    sequencer = MidiSystem.getSequencer();
                    sequencer.setSequence(MidiSystem.getSequence(file1));
                    sequencer.setLoopCount(-1);
                    sequencer.open();
                    sequencer.start();
                }
                catch(MidiUnavailableException midiunavailableexception)
                {
                    System.out.println("Midi device unavailable!");
                }
                catch(InvalidMidiDataException invalidmididataexception)
                {
                    System.out.println("Invalid Midi data!");
                }
                catch(IOException ioexception)
                {
                    System.out.println("I/O Error!");
                }
            }
        }
        catch(Exception exception)
        {
            System.out.println("MIDI file gone!");
        }
    else
        System.out.println("Stopped");
}

public void valueChanged(ListSelectionEvent evt) {
	value = list.getSelectedValue();
	file1 = new File("./MIDI/" + value);

public void actionPerformed(ActionEvent evt) {

	if (cmd != null) {

		}

			Box("About", "It plays Music!");
					 
		}
		if(cmd.equalsIgnoreCase("Next Midi"))
              midiPlayer("Next");
              if(cmd.equalsIgnoreCase("Previous Midi"))
              midiPlayer("Previous");
              if(cmd.equalsIgnoreCase("Play"))
              midiPlayer("Play");
              if(cmd.equalsIgnoreCase("Stop"))
              midiPlayer("Stop");
        }
  }
