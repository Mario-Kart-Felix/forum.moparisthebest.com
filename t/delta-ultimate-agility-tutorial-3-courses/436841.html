public class AgilityHandler { //2547 - 3569 | intrance barbarian

	Client c;
	
	public AgilityHandler (Client c) {
		this.c = c;
	}
	
	public int neededX, neededY;
	public double expAdded;
	public boolean isPassing = false;
	int crossEmote;
	int expRate = 8; //The experiences are real runescape ones, this is to control how many times you want to multiply them.
	long agilityTimer;

	/**
	 * @param walkX = How many xcoords the player must move
	 * @param walkY = How many ycoords the player must move
	 * @param needX = Where the player should be when done crossing the obstacle (xCoord)
	 * @param needY = Where the player should be when done crossing the obstacle (yCoord)
	 * @param Emote = The emote obviously the player will be doing when crossing the obstacle
	 * @param Exp = The amount of experience added when finishing an obstacle
	 * @Info: This method is to control when the player has to cross an obstacle but doesn't change height.
	 */
	void prepareCrossing (int walkX, int walkY, int needX, int needY, int Emote, double Exp) {
			if (System.currentTimeMillis() - c.lastAction > agilityTimer) {
				agilityTimer = 2000;
				crossEmote = Emote;
				neededX = needX;
				neededY = needY;
				expAdded = Exp;
				isPassing = true;
				crossingProcess(walkX, walkY);
		}
	}
	
	/**
	 * @param walkX = same as above
	 * @param walkY = same as above
	 * @Info: This is the crossing of the obstacle itself.
	 */
	void crossingProcess (int walkX, int walkY) {
		if (isPassing == true) {
			walkToSpot(walkX, walkY);
			c.WalkToEmote(crossEmote);
			c.lastAction = System.currentTimeMillis();
		}
	}
	
	/**
	 * @param neededX
	 * @param neededY
	 * @param getHeight = the height the player will be on after finishing the obstacle
	 * @param Emote
	 * @param Exp
	 * @Info: This is when the player does have to change height. 
	 */
	void prepareClimbing (int neededX, int neededY, int getHeight, int Emote, double Exp) {
			if (System.currentTimeMillis() - c.lastAction > agilityTimer && isPassing == false) {
				agilityTimer = 2000;
				c.setAnimation(Emote);
				expAdded = Exp;
				isPassing = true;
				climbingProcess(neededX, neededY, getHeight);
		}
	}
	
	/**
	 * @param neededX
	 * @param neededY
	 * @param getHeight
	 * @Info: The actual climbing/changing height. We use the task system to delay the changing of height since
	 * this causes that the player does his emote when already crossed the obstacle instead of while.
	 */
	void climbingProcess(final int neededX, final int neededY, final int getHeight) {
		if (isPassing) {
			ServerConstants.getTaskScheduler().schedule(new Task(1, false) {
				@Override
				protected void execute() {
					c.toX = neededX;
					c.toY = neededY;
					c.heightLevel = getHeight;
					isPassing = false;
					c.lastAction = System.currentTimeMillis();
					stop();
				}
			});
		}
	}
	
	/**
	 * Resetting the crossing obstacle action and adding the experience.
	 */
	public void resetAgility () {
		c.addSkillXP(((int)expAdded) * expRate, 16);
		isPassing = false;
		neededX = 0;
		neededY = 0;
		//c.forceReset();
		c.resetAnimation();
		c.StopWalkToEmote();
		c.updateRequired = true;
		c.appearanceUpdateRequired = true;
	}
	
	/**
	 * This is just custom by me, do with it what you want.
	 * I made it so player with a higher agility level also get more tickets.
	 */
	void calculateTickets () {
		int id = 0; //this is to control which type of tickets must be added. This checks in what agility area the player is.
		c.acc.barbarianProgress = 0;
		c.acc.gnomeProgress = 0;

		if (inGnomeAgility()) //check
			id = 2996;
		else if (inBarbarianAgility()) //check
			id = 7810;
		else if (inPyramidAgility()) //check
			id = 7811;
	
		if (c.playerLevel[16] < 10)
			c.addItem(id, 1);
		if (c.playerLevel[16] >= 10 && c.playerLevel[16] < 20)
			c.addItem(id, 2);
		if (c.playerLevel[16] >= 20 && c.playerLevel[16] < 30)
			c.addItem(id, 3);
		if (c.playerLevel[16] >= 30 && c.playerLevel[16] < 50)
			c.addItem(id, 5);
		if (c.playerLevel[16] >= 50 && c.playerLevel[16] < 65)
			c.addItem(id, 7);
		if (c.playerLevel[16] >= 65 && c.playerLevel[16] < 80)
			c.addItem(id, 10);
		if (c.playerLevel[16] >= 80 && c.playerLevel[16] < 92)
			c.addItem(id, 15);
		if (c.playerLevel[16] >= 92 && c.playerLevel[16] < 99)
			c.addItem(id, 20);
		if (c.playerLevel[16] >= 99)
			c.addItem(id, 25);
	}
	
	/**
	 * These give the area of an agility course
	 */
	public boolean inGnomeAgility() {
		return (c.absX >= 2468 && c.absX <= 2491  && c.absY >= 3413 && c.absY <= 3441);
	}
	public boolean inBarbarianAgility() {
		return (c.absX >= 2527 && c.absX <= 2556  && c.absY >= 3540 && c.absY <= 3574);
	}
	public boolean inPyramidAgility() {
		return (c.absX >= 3352 && c.absX <= 3377  && c.absY >= 2829 && c.absY <= 2854);
	}
	
	/**
	 * This is to control the walking when crossing an obstacle.
	 */
	public void walkToSpot(int i, int j) {
        Player.newWalkCmdSteps = 0;
        if(++Player.newWalkCmdSteps > 50)
        	Player.newWalkCmdSteps = 0;
        int k = c.absX + i;
        	k -= c.mapRegionX * 8;
        	Player.newWalkCmdX[0] = Player.newWalkCmdY[0] = Player.tmpNWCX[0] = Player.tmpNWCY[0] = 0;
        int l = c.absY + j;
        	l -= c.mapRegionY * 8;
        	c.isRunning2 = false;
        	c.isRunning = false;
        for(c.i = 0; c.i < Player.newWalkCmdSteps; c.i++) {
        	Player.newWalkCmdX[c.i] += k;
        	Player.newWalkCmdY[c.i] += l;
        }
        c.lastWalk = System.currentTimeMillis();
        c.walkDelay = 1;
        c.poimiY = l;
        c.poimiX = k;
    }
}

			followID = 0;
			if (handleagility.isPassing == true)

This checks all the time if the player is in one of the agility areas.
When he is it makes that you can’t follow players within that area.
When the player is busy with crossing an obstacle it makes that the player isn’t running, so the emote can be properly executed,
and he shouldn’t be able to run while crossing an object anyways.
Then when the player is done crossing the objects (when he’s on the supposed X and Y coordinate)
it resets agility.

Now we need to do one more thing, this is to make that the player can’t start running WHILE crossing the objects,
for the same reason as above.

		case 153:

				c.isRunning2 = true;
				c.isRunning = true;
			}
			break;

Replace it by this:

		case 153:
			if (c.handleagility.isPassing)
				return;
			if (c.playerEnergy > 0) {
				c.isRunning2 = true;
				c.isRunning = true;
			}
			break;

So when the player hits the button to toggle running,
it will check if he’s crossing an obstacle. If he is it will not toggle running.
