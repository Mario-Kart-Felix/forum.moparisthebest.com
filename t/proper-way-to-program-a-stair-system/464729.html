I suppose you could make it location based

public class ObjectEntity {
	private final int id;
	private final Location location;

	public ObjectEntity(final int id, final Location location) {
		this.id = id;
		this.location = location;
	}

	@Override
	public int hashCode() {
		final int prime = 31;
		int result = 1;
		result = prime * result + id;
		result = prime * result
			+ ((location == null) ? 0 : location.hashCode());
		return result;
	}

	@Override
	public boolean equals(final Object obj) {
		if (this == obj) {
			return true;
		}
		if (obj == null) {
			return false;
		}
		if (getClass() != obj.getClass()) {
			return false;
		}
		final ObjectEntity other = (ObjectEntity) obj;
		if (id != other.id) {
			return false;
		}
		if (location == null) {
			if (other.location != null) {
				return false;
			}
		} else if (!location.equals(other.location)) {
			return false;
		}
		return true;
	}
}
final HashMap<ObjectEntity, Location> stairs = new HashMap<ObjectEntity, Location>();
// ...
stairs.put(new ObjectEntity(stairId, someLocation), destination);
// when handling object packets
player.teleportTo(stairs.get(targetObject));
