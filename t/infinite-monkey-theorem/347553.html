[code]package com.anthony.test.infinitemonkey;

import java.text.DecimalFormat;
import java.text.DecimalFormatSymbols;
import java.util.*;
import java.util.concurrent.TimeUnit;

/**

  • Tests the infinite monkey theorem.

  • @author anthony
    */
    public class InfiniteMonkeyTheorem {

    private String string;
    private int monkeys;
    public static final char[] KEYBOARD = “abcdefghijklmnopqrstuvwxyz ABCDEFGHIJKLMNOPQRSTUVWXYZ”.toCharArray(); // TODO: more keys
    public static final int BUILDER_SIZE = 15000000;
    private DecimalFormat df = new DecimalFormat();
    private List keyPressResults = new ArrayList();
    private long maxKeyPresses = 0;
    private long minKeyPresses = Long.MAX_VALUE;

    public static void main(String[] args) {
    new InfiniteMonkeyTheorem(“hello”, 50).startTest();
    }

    public InfiniteMonkeyTheorem(String string, int monkeys) {
    this.string = string;
    this.monkeys = monkeys;
    DecimalFormatSymbols dfs = new DecimalFormatSymbols();
    dfs.setGroupingSeparator(’,’);
    df.setDecimalFormatSymbols(dfs);
    }

    public void startTest() {
    System.out.println(“Keyboard key count: " + KEYBOARD.length);
    System.out.println(“String length: " + string.length());
    System.out.println(“Monkey count: " + monkeys);
    for (int i = 0; i < monkeys; i++) {
    Monkey monkey = new Monkey(i + 1);
    Thread thread = new Thread(monkey);
    thread.start();
    try {
    thread.join();
    } catch (InterruptedException e) {
    e.printStackTrace();
    }
    }
    System.out.println(monkeys + " monkeys just typed “” + string + “”.”);
    System.out.println(””" + string + “” appeared as a substring after an average of " + df.format(avg(keyPressResults.toArray())) + " key presses per monkey.");
    System.out.println("Range of key presses per monkey: " + df.format(minKeyPresses) + “-” + df.format(maxKeyPresses) + “.”);
    }

    private long avg(Object[] objs) {
    long total = 0;
    for (Object obj : objs) {
    total += (Long) obj;
    }
    return total / objs.length;
    }

    class Monkey implements Runnable {

     private StringBuilder typed = new StringBuilder(BUILDER_SIZE);
 private Random generator = new Random(new Random().nextInt());
 private int id;
 private long keyPresses;

 public Monkey(int id) {
     this.id = id;
 }

 private boolean endsWith(StringBuilder builder, String s) {
     int offset = s.length();
     int typedOffset = builder.length() - offset;
     int sOffset = 0;
     if (typedOffset < 0) {
         return false;
     }
     while (offset-- > 0) {
         if (builder.charAt(typedOffset++) != s.charAt(sOffset++)) {
             return false;
         }
     }
     return true;
 }

 public void run() {
     Timer timer = new Timer();
     timer.schedule(new TimerTask() {

         @Override
         public void run() {
             //System.out.println("[Monkey-" + id + "] Current key presses: " + keyPresses);
             if (typed.length() >= BUILDER_SIZE) {
                 String lastChars = typed.substring((typed.length() - string.length()) + 1);
                 typed = new StringBuilder(BUILDER_SIZE);
                 typed.append(lastChars);
                 System.gc();
             }
         }
     }, 1000, 1000);
     long timeTaken;
     long startTime = System.nanoTime();
     while (true) {
         if (endsWith(typed, string)) {
             break;
         }
         keyPresses++;
         typed.append(KEYBOARD[generator.nextInt(KEYBOARD.length)]);
     }
     timeTaken = System.nanoTime() - startTime;
     long timeTakenSecs = TimeUnit.SECONDS.convert(timeTaken, TimeUnit.NANOSECONDS);
     System.out.println("[Monkey-" + id + "] The string \"" + string + "\" appeared as a substring after " + timeTakenSecs + " seconds.");
     System.out.println("[Monkey-" + id + "] The string \"" + string + "\" appeared as a substring after " + df.format(keyPresses) + " key presses.");
     keyPressResults.add(keyPresses);
     if (keyPresses > maxKeyPresses) {
         maxKeyPresses = keyPresses;
     }
     if (keyPresses < minKeyPresses) {
         minKeyPresses = keyPresses;
     }
     timer.cancel();
     System.gc();
 }

    }
    }[/code]
