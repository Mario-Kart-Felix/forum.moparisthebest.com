I use this on my webpage (www.newbiehacker.org) and figured I’d post it here.

It requires any element with the name as “clock” e.g.

<p name="clock" align="right"></p>

EDIT:
you have to edit your body tag (I completely forgot)

<body onload="StartClock()" onunload="KillClock()">

here it is:

<!--
  var clockID = 0;
  
  function UpdateClock() {
    if(clockID) {
      clearTimeout(clockID);
      clockID  = 0;
    }
    var tDate = new Date();
    var y = tDate.getYear();
    if(y < 1900)
      y += 1900;
    var mo = tDate.getMonth();
    var day = tDate.getDate();
    var end = '';
    if(day > 10 && day < 20)
      end = 'th';
    else
    {
      for(j = 0; j < 30; j+= 10){
        for(j1 = 0; j1 < 10; j1++){
          if(j1 + j == day){
            if(j1 == 1)
              end = 'st';
            else if(j1 == 2)
              end = 'nd';
            else if(j1 == 3)
              end = 'rd';
            else end = 'th';
          }
        }
      }
    }  
    var months = new Array('January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December');
    var date = day + end +  ' ' + months[mo] + ' ' + y;
    
    //get the 2 digits for the hour
    var h = tDate.getHours();
    var dayTime;
    if(h > 12){
      h -= 12;
      dayTime="PM";
    }else
      dayTime="AM";
    var h1 = 0;
    for(i = 0; i < 60; i += 10)
      for(i1 = 0; i1 < 10; i1++)
        if(i1 + i == h){
          h1 = i / 10;
          h -= i;
        }
    
    //get the 2 digits for the minute
    var m = tDate.getMinutes();
    var m1 = 0;
    for(i = 0; i < 60; i += 10)
      for(i1 = 0; i1 < 10; i1++)
        if(i1 + i == m){
          m1 = i / 10;
          m -= i;
        }
    
    //get the 2 digits for current second
    var s = tDate.getSeconds();
    var s1 = 0;
    for(i = 0; i < 60; i += 10)
      for(i1 = 0; i1 < 10; i1++)
        if(i1 + i == s){
          s1 = i / 10;
          s -= i;
        }
    
    var h = h1 + "" + h;
    var m = m1 + "" + m;
    var s = s1 + "" + s;
    clockID = setTimeout("UpdateClock()", 1000);
    clock.innerHTML = date + "    " + h + ":" + m + ":" + s + " " + dayTime;
  }

  function StartClock() {
    clockID = setTimeout("UpdateClock()", 500);
  }

  function KillClock() {
    if(clockID) {
      clearTimeout(clockID);
      clockID  = 0;
    }
  }
//-->
