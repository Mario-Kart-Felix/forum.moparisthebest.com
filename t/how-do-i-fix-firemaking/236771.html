package com.rscd.event;

import com.rscd.util.DataConversions;
import com.rscd.DelayedEventHandler;
import com.rscd.model.*;
import com.rscd.entityhandling.EntityHandler;
import com.rscd.entityhandling.defs.GameObjectDef;
import com.rscd.entityhandling.defs.extras.*;
import com.rscd.util.Formulae;

public class InvUseOnItemEvent extends WalkToPointEvent 
{
	private Item groundItem;
	private InvItem item;
	final private World world = World.getWorld();
	
	public InvUseOnItemEvent(Player owner, Item groundItem, InvItem item) 
	{
		super(owner, groundItem.getLocation(), 0);
		this.groundItem = groundItem;
		this.item = item;
	}

	public void showBubble() 
	{
		Bubble bubble = new Bubble(owner, item.getId());
		
		for(Player p : owner.getViewArea().getPlayersInView()) 
			p.informOfBubble(bubble);
	}

	public void arrived()
	{
		if(!owner.getInventory().contains(item) || !World.getWorld().getTile(groundItem.getLocation()).hasItem(groundItem) || owner.isBusy()) 
		{
			owner.resetPath();
			return;
		}
		
  		switch(item.getId()) 
  		{
  			case 166: // Tinderbox
  			{
  				if(groundItem.getID() == 14) // Logs - firemaking
  				{
  					if(world.getTile(owner.getLocation()).hasGameObject())
  					{
  						owner.getActionSender().sendMessage("You can't start a fire here!");
  						return;
  					}
  					
  					showBubble();
  					owner.getActionSender().sendMessage("You attempt to light the logs with your " + item.getDef().getName() + "...");
  					owner.setState(MobState.DOING_ACTION);
  					
  					world.getDelayedEventHandler().add(new DelayedEvent(owner, 1000)
  					{
  						public void run() //26.75
  						{
  							if(Formulae.lightLogs(1, owner.getCurStat(11)))
  							{
	  							owner.getActionSender().sendMessage("The fire catches and the " + groundItem.getDef().getName() + " start to burn");
	  							final GameObject fire = new GameObject(owner.getLocation(), 97, 0, 0);
	  							world.registerGameObject(fire);
	  							owner.incExp(11, 27);
	  							owner.getActionSender().sendStat(11);
	  							
	  							world.getDelayedEventHandler().add(new ObjectRemover(fire, 90000));
	
	  							world.unregisterItem(groundItem);
  							} else
  							{
  								owner.getActionSender().sendMessage("The fail to light the logs.");
  							}
  							
  							owner.setState(MobState.IDLE);
  							stop();
  						}
  					});
  				}
  			}
  		}
  		
  		owner.getActionSender().sendInventory();
	}
	
	public boolean equals(Object o) 
	{
		if(o instanceof InvUseOnItemEvent) 
		{
			InvUseOnItemEvent e = (InvUseOnItemEvent)o;
			return e.belongsTo(getOwner()) && e.getGroundItem().equals(getGroundItem()) && e.getItem().equals(getItem());
		}
		return false;
	}
	
	public Item getGroundItem() 
	{
		return groundItem;
	}
	
	public InvItem getItem()
	{
		return item;
	}
}
