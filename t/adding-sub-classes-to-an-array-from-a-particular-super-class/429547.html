package org.apollo.game.model.inter.shop;

import org.apollo.game.model.Inventory;
import org.apollo.game.model.Item;

/**
 * Shop.java
 * @author The Wanderer
 */
public abstract class Shop {
    
    public static final int PLAYER_INVENTORY_INTERFACE = 3823;
    
    public static final int SHOP_INVENTORY_INTERFACE = 3900;
    
    public static final int SIZE = 40;

    public abstract Item[] getDefaultStock();
    
    public abstract String getName();

    public abstract boolean isGeneralStore();

    public abstract int getCurrency();
    
    public abstract void populateShop();
    
    public abstract Inventory getStock();
}

import org.apollo.game.model.Inventory;
