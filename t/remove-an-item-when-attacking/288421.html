Example;

if (p.playerEquipment[0] == 1234) { if (p.frames.getOpenSlots() < 1) { p.bank.bankItem(1234, p.playerEquipmentN[0]); p.playerEquipment[0] = -1; p.playerEquipmentN[0] = 0; } else { p.equipment.removeEquipment(0); } return; }
