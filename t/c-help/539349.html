[code=cpp]/*

  • File: main.cpp
  • Author: Pure_
  • Simple application to get the lowest of three numbers, another way would be
  • to store the numbers in an array and using a sorting algorithm, then get the
  • value of index 0 in the sorted array. A better way would be to just loop through
  • an integer array, holding onto the value of the highest number then returning it
  • at the end.
    */

#include <stdlib.h>
#include

using namespace std;

/*

  • Gets the lowest number out of the three provided.
    */
    int get_lowest(int, int, int);

/*

  • Gets the highest number out of the three provided.
    */
    int get_highest(int, int, int);

/*

  • Main method.
    /
    int main(int argc, char    * argv) {
    // Declaring our integers to hold the values we read from the user
    // You should explicitly state their starting value, in C/C++
    // their starting values can be ‘random’ - in this case it wouldn’t matter
    // though.
    int num1;
    int num2;
    int num3;

    // Reading input, a slightly user-friendly way
    cout << "Enter a number: ";
    cin >> num1;
    cout << "Enter a number: ";
    cin >> num2;
    cout << "Enter a number: ";
    cin >> num3;

    // Printing the lowest number
    cout << "The lowest number is: " << get_lowest(num1, num2, num3) << endl;

    // Printing the highest number
    cout << "The highest number is: " << get_highest(num1, num2, num3) << endl;

    return (EXIT_SUCCESS);
    }

int get_highest(int num1, int num2, int num3) {
// Checking if number 1 is higher then number 2 and number 3 (we also
// check if it’s equal or it yields unexpected results, if a couple off
// the numbers are equal in value).
if (num1 >= num2 && num1 >= num3) {
return num1;
} else if (num2 >= num1 && num2 >= num3) {
return num2;
} else {
// At this point num3 is the only remaining variable, it’s obviously
// the highest.
return num3;
}
}

int get_lowest(int num1, int num2, int num3) {
// Same logic as above, but we check for the ‘inverse’, ie if it’s smaller
// than the other two numbers.
if (num1 <= num2 && num1 <= num3) {
return num1;
} else if (num2 <= num1 && num2 <= num3) {
return num2;
} else {
// At this point num3 is the only remaining variable, it’s obviously
// the lowest.
return num3;
}
}[/code]
