[code]import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;

public class PlayersOnline {
private static Connection connection;
private static Statement statement;

private String IP = "", database = "", username = "", password = "";

/**
 * Creates a connection to your MySQL database
 */
private void createConnection() {
	try {
		Class.forName("com.mysql.jdbc.Driver").newInstance();
		connection = DriverManager.getConnection("jdbc:mysql://" + IP + "/"
				+ database, username, password);
		statement = connection.createStatement();
	} catch (Exception e) {
		e.printStackTrace();
	}
}

/**
 * Save players to database
 */
public void updatePlayers(Client c) {
	createConnection();
	String username = c.getUsername(); // different through each server
	if (c.logout != true) {
		query("INSERT INTO `online` (`playerName`) VALUES ('" + username
				+ ");");
	} else {
		query("DELETE FROM `online` WHERE playerName = '" + username + "';");
	}

}

/**
 * 
 * @param s
 * @return
 * @throws SQLException
 */
public static ResultSet query(String s) throws SQLException {
	try {
		if (s.toLowerCase().startsWith("select")) {
			ResultSet rs = statement.executeQuery(s);
			return rs;
		} else {
			statement.executeUpdate(s);
		}
		return null;
	} catch (Exception e) {
		destroy();
	}
	return null;
}

/**
 * Destorys the connection
 */
public static void destroy() {
	try {
		statement.close();
		connection.close();
	} catch (Exception e) {
		e.printStackTrace();
	}
}

}

In reply to your pm, here’s some code for retrieving the number of rows in a table.

/**
* Define some database connection variables.
**/
$host ="localhost";
$user = "root";
$pass = "yourpass";
$db = "your_db";

/**
* Connect to the database.
**/
$con = mysqli_connect($host, $user, $pass, $db);

/**
* Create a query to manipulate the database.
**/
$sql = "SELECT * FROM tbl_name";
 
/**
* Query the database.
**/
$res = mysqli_query($con, $sql);

/**
* Count the number of rows.
**/
$online_players = mysqli_num_rows($res);

/**
* Display the amount of rows, or which is now the player count.
**/
echo "There are currently " . $online_players . " online!";

/**
* Close the connection to the database.
**/
mysqli_close($con);
