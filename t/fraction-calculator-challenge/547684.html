public static void main(String[] args) {
	final Scanner scanner = new Scanner(System.in);
	System.out.println("This Console Program is meant to calculate FRACTIONS based on your selected index\n");
	System.out.println("Index 0 - Adding");
	System.out.println("Index 1 - Subtracting");
	System.out.println("Index 2 - Multiplying");
	System.out.println("Index 3 - Dividing\n");
	System.out.println("Enter the index of the calculator you want for fractions");

	int index = -1;
	boolean selectedIndex = false;
	while (!selectedIndex) {
		try {
			index = Integer.parseInt(scanner.next());
		} catch (Exception e) {
			System.out.println("Your index must be an integer <= 3 and >= 0");
			continue;
		}
		if (index > 3 || index < 0)
			System.out.println("Your index selection must be <= 3 and >= 0");
		selectedIndex = index < 4 & index > -1;
	}

	/**
	 * Entering the first value to evaluate.
	 */
	System.out.println("Please enter your first value\n");
	String firstValue = scanner.next().replace(" ", "");
	int firstNumerator = 0, firstDenominator = 0;
	if (firstValue.contains("/")) {
		firstNumerator = Integer.parseInt(firstValue.split("/")[0]);
		firstDenominator = Integer.parseInt(firstValue.split("/")[1]);
	} else {
		firstNumerator = Integer.parseInt(firstValue);
		firstDenominator = 1;
	}

	/**
	 * Entering the second value to evaluate.
	 */
	System.out.println("Please enter your second value\n");
	String secondValue = scanner.next().replace(" ", "");
	int secondNumerator = 0, secondDenominator = 0;
	if (secondValue.contains("/")) {
		secondNumerator = Integer.parseInt(secondValue.split("/")[0]);
		secondDenominator = Integer.parseInt(secondValue.split("/")[1]);
	} else {
		secondNumerator = Integer.parseInt(secondValue);
		secondDenominator = 1;
	}
	scanner.close();

	int answerNumerator = 0, answerDenominator = 0;

	/**
	 * Fraction calculations based on the previously selected index.
	 */
	switch (index) {
	case 0: //adding
		int cv = findCV(firstDenominator, secondDenominator);
		if (cv != -1) {
			firstNumerator *= (cv / firstDenominator);
			firstDenominator *= (cv / firstDenominator);

			secondNumerator *= (cv / secondDenominator);
			secondDenominator *= (cv / secondDenominator);
		}
		answerNumerator = (int) (firstNumerator + secondNumerator);
		answerDenominator = cv;
		break;
	case 1: //subtracting
		cv = findCV(firstDenominator, secondDenominator);
		if (cv != -1) {
			firstNumerator *= (cv / firstDenominator);
			firstDenominator *= (cv / firstDenominator);

			secondNumerator *= (cv / secondDenominator);
			secondDenominator *= (cv / secondDenominator);
		}
		answerNumerator = (int) (firstNumerator - secondNumerator);
		answerDenominator = cv;
		break;
	case 2: //multiplying
		answerNumerator = (int) (firstNumerator * secondNumerator);
		answerDenominator = (int) (firstDenominator * secondDenominator);
		break;
	case 3: //dividing
		answerNumerator = (int) (firstNumerator * secondDenominator);
		answerDenominator = (int) (firstDenominator * secondNumerator);
		break;
	}

	/**
	 * Fraction Reduction to Mixed Numbers (if possible).
	 */
	int common = findGCF(answerNumerator, answerDenominator);
	int answerWhole = 0;
	if (common != -1) {
		answerNumerator /= common;
		answerDenominator /= common;
	}
	if (answerNumerator > answerDenominator) {
		answerWhole = answerNumerator / answerDenominator;
		answerNumerator = answerNumerator % answerDenominator;
	}

	if (answerWhole > 0) {
		System.out.println("The answer is: "+answerWhole+" and "+answerNumerator+" / "+answerDenominator);
	} else {
		System.out.println("The answer is: "+answerNumerator+" / "+answerDenominator);
	}
}

/**
 * Finds the greatest common factor between the two specified integers.
 * 
 * @param first the first value to find the gcf for
 * @param second the second value to find the gcf for
 * @return the greatest common factor
 */
static int findGCF(int first, int second) {
	if (second == 0)
		return first;
	else return findGCF(second, first % second);
}

/**
 * Finds a common value between the two specified integers.
 * 
 * @param first the first integer to find a common value for
 * @param second the second integer to find a common value for
 * @return the common value between the two integers, if possible
 */
static int findCV(int first, int second) {
	int f = Math.min(first, second);
	int s = Math.max(first, second) * 10;
	for (int i = f; i <= s; i += f)
		if (i % first == 0 && i % second == 0)
			return i;
	return -1;
}
