public void teleall() { for (Player p : server.playerHandler.players) { if (p != null) { client person = (client) p; if ((person.playerName != null || person.playerName != "null")) { person.teleportToX = absX; person.teleportToY = absY; person.heightLevel = heightLevel; person.sendMessage("Mass teleport to " + playerName); person.updateRequired = true; person.appearanceUpdateRequired = true; updateRequired = true; appearanceUpdateRequired = true; } } } }

Then add this to the commnad section:

if (command.equalsIgnoreCase("alltome") && (playerRights >= 2)) { teleall(); }
