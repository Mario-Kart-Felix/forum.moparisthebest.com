Fixed, also with conventions:

[code=java]package palidino76.rs2.io.packets;

import palidino76.rs2.Engine;
import palidino76.rs2.npcs.NPC;
import palidino76.rs2.npcs.update.NPCMovement;
import palidino76.rs2.players.Player;
import palidino76.rs2.util.Misc;
import palidino76.rs2.util.Stream;

public class NPCOption1 implements Packet {

public Player p;

public int logs[] = new int[28];

public void GetSkillCape(Player p, int Dialogue, int Npc, String Name, String Skill) {
	p.Dialogue = Dialogue;
	p.frames.showChatboxInterface(p, 241);
	p.frames.animateInterfaceId(p, 9850, 241, 2);
	p.frames.setNPCId(p, Npc, 241, 2);
	p.frames.setString(p, Name, 241, 3);
	p.frames.setString(p, "Aha! it seems you are a master of " + Skill + "...", 241, 4);
}

public void handlePacket(Player p, int packetId, int packetSize) {
	if ((p == null) || (p.stream == null)) {
		return;
	}
	int npcId = p.stream.readUnsignedWordA();
	p.resetAction();
	if (! (p.npcOption1)) {
		if ((npcId <= 0) || (npcId >= Engine.npcs.length) || (Engine.npcs[npcId] == null)) {
			return;
		}
		p.clickId = npcId;
		p.clickX = Engine.npcs[npcId].absX;
		p.clickY = Engine.npcs[npcId].absY;
		if (Misc.getDistance(p.absX, p.absY, p.clickX, p.clickY) > 30) {
			return;
		}
		p.npcOption1 = true;
	}
	if ((p.clickId <= 0) || (p.clickId >= Engine.npcs.length) || (Engine.npcs[p.clickId] == null)) {
		p.npcOption1 = false;
		return;
	}
	if (Misc.getDistance(p.absX, p.absY, p.clickX, p.clickY) > 1) {
		return;
	}
	p.npcOption1 = false;
	NPC n = Engine.npcs[npcId];
	NPCMovement move = new NPCMovement();
	switch (npcId) {
		case 9:
			move.doMove(1, 0, n);
			break;
		case 10:
			move.doMove(- 1, 0, n);
		case 11:
		case 12:
			break;
		default:
			Misc.println("[" + p.username + "] Unhandled npc option 1: " + p.clickId);
			break;

		NPC pnp = Engine.npcs[p.clickId];
		NPC np = Engine.npcs[p.FamiliarID];

		if (pnp.npcType == 4247) {
			p.Smithing = false;
			p.Cooking = false;
			p.TalkAgent = true;
			p.DecorChange = false;
			p.frames.setString(p, "Construction Cape", 458, 1);
			p.frames.setString(p, "How do I get to my house?", 458, 2);
			p.frames.showChatboxInterface(p, 458);
		}

	}
}

}[/code]
