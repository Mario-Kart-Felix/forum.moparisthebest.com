use #define and #ifdef/#ifndef to differentiate between the windows and unix sleep functions, that way the user only has to change one thing in the source to compile it himself… for example:

[code]#define WIN32

#ifndef WIN32
#include <time.h>
#else
#include <windows.h>
#endif[/code]

[code]#ifndef WIN32
void Sleep(unsigned int mseconds)
{
clock_t goal = mseconds + clock();
while (goal > clock());
}
#endif

void reDraw(){
display();
#ifndef WIN32
Sleep(50000);
#else
Sleep(50);
#endif
}[/code]
that way people only have to comment out #define WIN32 to compile for linux.
