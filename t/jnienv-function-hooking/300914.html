JNIEnv is basically a JNINativeInterface pointer. JNINativeInterface is a collection of pointers, the first 4 being reserved, and the rest being to functions. This makes it easy for assembly programmers to utilize JNI functions if they know the position of the function they want to call in the struct:

GetJNIFunc proc JniTable:DWORD, FuncIndex:DWORD
    mov eax, FuncIndex
    mov ebx, 4
    mul ebx
    mov eax, JniTable
    add eax, ebx
    mov eax,[eax]
    ret
GetJNIFunc endp

I quickly realized it would be just as easy to replace the function pointers with your own. And so I did.

jnihook.c:

#include <stdio.h>
#include <windows.h>

#define GETVERSION_INDEX	4

typedef int (__stdcall *fnGetVersion)(LPVOID **table);
fnGetVersion origGetVersion;

LPVOID *HookJNITable(LPVOID *table, int index, LPVOID newFunc) {
    LPVOID oldFunc = table[index];
    table[index] = newFunc;
    return oldFunc;
}

int __stdcall MyGetVersion(LPVOID **table) {
    printf("JNI hook! :)\n");
    return origGetVersion(table);
}

__declspec(dllexport) void __stdcall Java_Test_PrintVersion(LPVOID **table, LPVOID jclass) {
    fnGetVersion l_GetVersion = (fnGetVersion) (*table)[GETVERSION_INDEX];
    printf("version = %d\n", l_GetVersion(table));
}

__declspec(dllexport) void __stdcall Java_Test_Patch(LPVOID **table, LPVOID jclass) {
    origGetVersion = (fnGetVersion) HookJNITable(*table, GETVERSION_INDEX, &MyGetVersion);
}

BOOL APIENTRY DllMain(HANDLE hModule, DWORD dwReason, LPVOID lpReserved) {
    return TRUE;
}

Notice how it has no JNI-specific dependencies, like jni.h

Test.java:

public class Test {

    public static native void Patch();

    public static native void PrintVersion();

    public static void main(String[] argv) {
        PrintVersion();
        Patch();
        PrintVersion();
    }

    static {
        System.loadLibrary("jnihook");
    }
}
