[code]final class Deque {

Node head;
private Node current;

final void insertBack(Node node) {
	if (node.prev != null) {
		node.unlink();
	}
	node.next = head;
	node.prev = head.prev;
	node.prev.next = node;
	node.next.prev = node;
}

final void merge(Deque deque) {
	mergeInternal(deque, head.next);
}

final int size() {
	int i = 0;
	for (Node node = head.next; head != node;) {
		node = node.next;
		i++;
	}

	return i;
}

final void clear() {
	do {
		Node node = head.next;
		if (node == head) {
			break;
		}
		node.unlink();
	} while (true);
	current = null;
}

final void insertFront(Node node) {
	if (node.prev != null) {
		node.unlink();
	}
	node.next = head.next;
	node.prev = head;
	node.prev.next = node;
	node.next.prev = node;
}

private final void mergeInternal(Deque deque, Node front) {
	Node back = head.prev;
	head.prev = front.prev;
	front.prev.next = head;
	if (front != head) {
		front.prev = deque.head.prev;
		front.prev.next = front;
		deque.head.prev = back;
		back.next = deque.head;
	}
}

final Node getPrev() {
