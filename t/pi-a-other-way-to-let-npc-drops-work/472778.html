if(npcs[i].npcType > 0) { int random2 = Misc.random(8); if(random2 == 4) { Server.itemHandler.createGroundItem(c, 12158, npcs[i].absX, npcs[i].absY, 1, c.playerId); }

if(npcs[i].npcType > 8528) { int random3 = Misc.random(6); if (random3 == 6) { Server.itemHandler.createGroundItem(c, 18359, npcs[i].absX, npcs[i].absY, 1, c.playerId); }
