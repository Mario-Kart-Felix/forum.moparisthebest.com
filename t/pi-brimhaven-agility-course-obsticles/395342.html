	if (playerLevel[16] >= Req) {
		agilityEmote = true;

					stopEmote();
					c.stop(); 
				} 
			}, agilityTimer);
	} else { 
		sendMessage("You Need " + Req + " Agility To Do This Obsticle"); 
	}
}

		getPA().addSkillXP(amtEXP, playerAgility); 
			EventManager.getSingleton().addEvent(new Event() { 
				public void execute(EventContainer c) {
