program mopar;

const
 BackColor = 0;
 FrontColor = 12285440;
            //65280
            //16711680
 DrawSpeed = 0;
 TextHeight = 10;
 YStrech = 1.3;

 Width = 600;
 Height= 703;

var
 Message:array[1..50]of string;

procedure LoadArray;
begin
message[1] :='                ....................          ';
message[2] :='           ............................        ';
message[3] :='        ..........  ...........  ..........      ';
message[4] :='      ..........    ...........    ...........    ';
message[5] :='     .........      ...........      ..........   ';
message[6] :='    .........       ...........       ..........  ';
message[7] :=' ';
message[8] :=' ';
message[9] :='The Image above is the Mopar logo.';
message[10] :='in Ascii art.';
message[11] :='made by Mod Robby w/ a bit of editing by gokon';
end;

procedure DrawMopar;
var bmp,yPos,f:Integer; can:TCanvas;
begin
bmp:= BitmapFromString(Width,Height,'');
can:= GetBitmapCanvas(bmp);

can.brush.color:= BackColor;
can.brush.style:= bsSolid;
can.pen.color:= BackColor;
can.pen.width:= 1;
can.rectangle(0,0,Width,Height);

can.Font.Color:= FrontColor;
can.Font.Size:= TextHeight;

yPos:= 2;
for f:=1 to 50 do
 begin
 can.TextOut(50,yPos,message[f]);
 yPos:= Round(yPos + (TextHeight*YStrech));
 CopyCanvas(can,GetDebugCanvas,0,0,Width,Height,0,0,Width,Height);
 wait(DrawSpeed);
 end

end;


begin
LoadArray;
DisplayDebugImgWindow(Width,Height);
DrawMopar;
end.
