case 823: if (actionTimer == 0) { sendMessage("omg you root the doll."); setAnimation(1110); addSkillXP((370*playerLevel[0]), 0); addSkillXP((400*playerLevel[3]), 3); addSkillXP((400*playerLevel[18]), 18); actionTimer = 20; } break;

note: 400 doesn’t mean 400xp so just play around with it for a while.
