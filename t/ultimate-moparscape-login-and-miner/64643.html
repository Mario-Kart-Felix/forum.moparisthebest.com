program AutoLogin;

const
  Username = 'yes' ;  // MS Username
  Password = 'yes' ;  // MS Password

begin
  Wait(1000)
  MoveMouseSmooth(295, 358);
  Wait(1000)
  ClickMouse(295, 358, True);
end;

procedure EnterServer;
begin
  Wait(1000)
  MoveMouseSmooth(387, 374);
  Wait(1000)
  ClickMouse(387, 374, True);
end;

begin
  Wait(1000)
  MoveMouseSmooth(465, 347);
  Wait(1000)
  ClickMouse(465, 347, True);
end;

procedure SetupInfo;
begin
  Wait(1000)
  SendKeys(Username + Chr(13));
  Wait(1000)
  SendKeys(Password + Chr(13));
end;

procedure LoginEnter;
begin
  Wait(1000)
  MoveMouseSmooth(262, 374);
  Wait(1000)
  ClickMouse(262, 374, True);
end;

  Wait(1000)

  Wait(1000)
  MoveMouseSmooth(605, 302);
  ClickMouse(605, 302, True);

procedure MineIron;

 Wait(3000)
