// MARTIJNTJUHS powerminer!!!
// 4th script!

{.include srl\srl.scar}
{.include srl\srl\skill\mining.scar}

Const
  RockColor = 6461160;  // Color of the rock you wanna mine...
  LoadsToDo = 10000;    // Loads to mine and drop hehehe.

Var
  Loads, Ores : Integer;

Procedure DeclarePlayers;
begin
  HowManyPlayers := 2; // Set Number of Players here.
  NumberOfPlayers(HowManyPlayers); // Sets the Players Array Length;
  CurrentPlayer := 0; // CurrentPlayer = Array Index

  Players[0].Name := 'scheeraap'; // Username
  Players[0].Pass := 'mafkees';   // Password
  Players[0].Nick := 'eer';       // 3 or 4 letters username
  Players[0].Loc := '';           //Leave as blank
  Players[0].Skill := 'mining';   //The skill for when you get a lamp
  Players[0].Active := True;

  Players[0].Name := 'lalalala'; // Username
  Players[0].Pass := 'dfhgh';    // Password
  Players[0].Nick := 'ala';      // 3 or 4 letters username
  Players[0].Loc := '';          // Leave as blank
  Players[0].Skill := 'mining';  // The skill for when you get a lamp
  Players[0].Active := True;
  
  WriteLn(IntToStr(HowManyPlayers) + ' Players');
end;

Procedure PowerMiners2;
Var
  x, y, prevcount, a : Integer;
begin
  PrevCount := InvCount;
  if FindObj(x, y, 'ine', RockColor, 10) then
  begin
    Mouse(x, y, 10, 5, True);
    a := 0;
    Repeat
      Wait(1000);
      a := a + 1;
      if (a = 3) then break;
    Until(InvFull)
    Loads := Loads + 1;
    Ores := Ores + 27;
  end;
end;

  b, a : integer;

  if (FindFight) then

  begin
    OpenBox;
    Wait(1000);
    GambleNewBox;
    Wait(1000);
  end;
  if (FindNewBox) then
  begin
    OpenBox;
    Wait(1000);
    GambleNewBox;
  end;
  if (FindNewBox) then
  begin
    LogOut;

Procedure DropOre;
begin
  MouseSpeed:= 13+Random(7);
  DropTo(2, 14);
  DropTo(15, 28);
end;

Procedure ProgressReport;
begin
  ClearDebug;
  WriteLn('Progress Report for MARTIJNTJUHS Powerminer Miner');
  WriteLn(' ');
  WriteLn('Mined and Dropped ' + IntToStr(Loads) + ' Loads.');
  WriteLn(' ');
  WriteLn('Mined and Dropped ' + IntToStr(Ores) + ' Ores.');
end;

begin
  SetupSRL;
  SetupSRLMining;

      DropOre;
