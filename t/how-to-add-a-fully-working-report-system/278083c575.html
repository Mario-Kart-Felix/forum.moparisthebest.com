[code]package server.model.players.packets;

import server.util.;
import server.model.players.;
import server.;
import java.text.DateFormat;
import java.text.;
import java.util.*;

public class ReportAbuse implements PacketType {

@Override
public void processPacket(Client c, int packetType, int packetSize) {
	if(System.currentTimeMillis() - c.reportDelay > 60000) {
		String abuser = Misc.longToPlayerName(c.getInStream().readQWord());
		String Name = "";
		int rule = c.getInStream().readUnsignedByte();
		if(abuser.toLowerCase().equals(c.playerName)) {
			c.sendMessage("You cannot report yourself!");
			return;
		}
		if(Server.playerHandler.isPlayerOn(abuser.toLowerCase())) {
			switch(rule) {
				case 0:
					Name = "Offensive language";
					break;
				case 1:
					Name = "Item scamming";
					break;
				case 2:
					Name = "Password scamming";
					break;
				case 3:
					Name = "Bug abuse";
					break;
				case 4:
					Name = "Staff impersonation";
					break;
				case 5:
					Name = "Account sharing/trading";
					break;
				case 6:
					Name = "Macroing";
					break;
				case 7:
					Name = "Multiple logging in";
					break;
				case 8:
					Name = "Encouraging others to break rules";
					break;
				case 9:
					Name = "Misuse of report abuse";
					break;
				case 10:
					Name = "Advertising";
					break;
				case 11:
					Name = "Real world trading";
					break;
			}
			c.reportDelay = System.currentTimeMillis();
			//c.writeLog(""+c.playerName+" reported "+abuser+" for "+Name+". | Date: "+ date +"; Time: "+ time +";", "Reports");
			c.sendMessage("Your report has been stored and online moderators have also been notified.");
			for(Player p : Server.playerHandler.players) {
				if(p == null || p.playerRights == 0)
					continue;
				Client staff = (Client) Server.playerHandler.players[p.playerId];
				staff.sendMessage("[Report] "+c.playerName+" has reported "+abuser+" for: "+Name+".");
			}
		} else {
			c.sendMessage("That player is not online.");
		}
	} else {
		c.sendMessage("You can only send a report every 60 seconds!");
	}
}

public static String time = new SimpleDateFormat("hh:mm:ss").format(new Date());
