Windows Registry Editor Version 5.00

[HKEY_CLASSES_ROOT\ft000001]
"EditFlags"=dword:00000000
"BrowserFlags"=dword:00000008
@=""

[HKEY_CLASSES_ROOT\ft000001\shell]
@="Decompile"

[HKEY_CLASSES_ROOT\ft000001\shell\Decompile]

[HKEY_CLASSES_ROOT\ft000001\shell\Decompile\command]
@="\"C:\\WINDOWS\\jad.exe\" -sjava -nonlb \"%1\""

[HKEY_LOCAL_MACHINE\SOFTWARE\Classes\ft000001]
"EditFlags"=dword:00000000
"BrowserFlags"=dword:00000008
@=""

[HKEY_LOCAL_MACHINE\SOFTWARE\Classes\ft000001\shell]
@="Decompile"

[HKEY_LOCAL_MACHINE\SOFTWARE\Classes\ft000001\shell\Decompile]
