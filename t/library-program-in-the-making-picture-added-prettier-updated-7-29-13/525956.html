Here’s how I would do it(with a few more suggested things to make it easier to read):

import java.io.*;
import java.util.*;

class Book {
   
   public void createNewBook(int bookNumber, String bookCheck, String bookTitle, String bookAuthor, String bookType) {
      System.out.println("Book ID number: " + bookNumber + ".");
      System.out.println("Book title: " + bookTitle + ".");
      System.out.println("Book Author: " + bookAuthor + ".");
      System.out.println("Book type: " + bookType + ".");
      System.out.println();
   }

}

public class BookShow extends Book {
   
   public static void main (String args[]) {
      BookShow bookShow = new BookShow();
      bookshow.getBookDetails();
   }

   private void getBookDetails() {
      Book nB = new Book();
      int a;
      String b;
      String c;
      String d;
      String e;
      String correct;
      Scanner bG = new Scanner(System.in);
      System.out.println("What's the number of the book?");
      a = bG.nextInt();
      System.out.println("Is the book checked in currently?");
      b = bG.nextLine();
      System.out.println("what's the book's title?");
      c = bG.nextLine();
      System.out.println("Who is the book's author?");
      d = bG.nextLine();
      System.out.println("What's the book's genre?");
      e = bG.nextLine();
      System.out.println("Is this information correct?");
      correct = bG.nextLine();
      if (correct.equals("Yes")) {
         System.out.println("Saved book information.\n");
         BufferedWriter bookWrite = null;
         try {
            bookWrite = new BufferedWriter(new FileWriter(d + "-" + c + ".txt"));
            bookWrite.write("Book number: " + a);
            bookWrite.newLine();
            bookWrite.write("Book is in:" + b);
            bookWrite.newLine();
            bookWrite.write("Book title: " + c);
            bookWrite.newLine();
            bookWrite.write("Book author: " + d);
            bookWrite.newLine();
            bookWrite.write("Book genre: " + e);
            bookWrite.newLine();
            bookWrite.close();
            System.out.println("\n\nBook details:\n");
            nB.createNewBook(a, b, c, d, e);
         } catch (IOException g) {
            System.out.println("uh-ho!");
         }
      } else if (correct.equals("No")) {
         System.out.println("\nRestarting information.\n");
         getBookDetails();
      }
   }
      
}
