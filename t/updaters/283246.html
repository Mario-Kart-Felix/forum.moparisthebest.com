Unhandled July 31, 2016, 8:24am #2

They analyse the bytecode of the client, looking for patterns or “hooks” that can be consistently found between versions. Usually they will also inject accessor methods for fields throughout the client, allowing a bot to get information directly.
Bytecode libraries are commonly used to help with this process, a few examples:
BCEL - http://jakarta.apache.org/bcel/
ASM - http://asm.ow2.org/
Javassist - http://www.csg.is.titech.ac.jp/~chiba/javassist/
You could also write your own, following the JVM spec.

Now do some research for yourself, and revise your question a bit.

		initGUI();

		loader.setSize(dim);
		loader.setPreferredSize(dim);
		Dimension dim2 = new Dimension(768, 506);
		JScrollPane appletScroll = new JScrollPane(loader);
		appletScroll.getViewport().setSize(dim2);
		appletScroll.setPreferredSize(dim2);
		add(appletScroll, BorderLayout.CENTER);
		pack();
	}
}


class BotStub implements AppletStub, AppletContext {
	private static final Map<URL,WeakReference<Image>> imageCache;
	private final Map<String, InputStream> inputCache;
	private final Map<String, String> parameters;
	private final int world;
	private final URL codeBase;
	private final URL documentBase;
	private final Applet applet;
	private boolean isActive;

	static {
		imageCache = new HashMap<URL,WeakReference<Image>>();
	}

	public static void clearImageCache() {
		imageCache.clear();
	}

	public void clearInputCache() {
		inputCache.clear();
	}

	public final int getWorldNumber() {
		return world;
	}

	public BotStub(int world, Applet applet) {
		this.world = world;
		this.applet = applet;
		parameters = loadParameters();
		inputCache = Collections.synchronizedMap(new HashMap<String, InputStream>(2));
		try {
			codeBase = new URL("http://world" + world + ".runescape.com");
			documentBase = new URL("http://world" + world + ".runescape.com/m0");
		} catch (MalformedURLException e) {
			throw new RuntimeException(e);
		}
	}

	private HashMap<String, String> loadParameters() {
		HashMap<String, String> parameters = new HashMap<String, String>();
		parameters.put("java_arguments", "\"-Dsun.java2d.noddraw=true\"");
		parameters.put("worldid", "" + world);
		parameters.put("modewhere", "0");
		parameters.put("modewhat", "0");
		parameters.put("advertsuppressed", false ? "1" : "0");
		parameters.put("lang", "0");
		parameters.put("objecttag", "0");
		parameters.put("js", "1");
		parameters.put("game", "0");
		parameters.put("affid", "0");
		parameters.put("settings", "VPEWjAdl0TmCHVq*pm8v7k7lUtJhWpNG1*WoIL-Adjg");
		parameters.put("haveie6", "1");
		parameters.put("cookieprefix", "");
		parameters.put("cookiehost", ".runescape.com");
		parameters.put("crashurl", "http://www.runescape.com/slu.ws?j=1&m=0&crash=1");
		parameters.put("unsignedurl", "http://www.runescape.com/slu.ws?j=1&m=0");
		parameters.put("pre142url", "http://world" + world + ".runescape.com/m0,j1,a1,o0");

		return parameters;
	}

	public URL getCodeBase() {
		return codeBase;
	}

	public URL getDocumentBase() {
		return documentBase;
	}

	public String getParameter(String s) {
		String parameter = parameters.get(s);
		if (s != null) {
			return parameter;
		}
		return "";
	}

	public AppletContext getAppletContext() {
		return this;
	}

	public boolean isActive() {
		return isActive;
	}

	public void setActive(boolean isActive) {
		this.isActive = isActive;
	}

	public void appletResize(int x, int y) {
		//applet.setSize(x, y);
		//log("Attempting resize: " + x + "," + y);
		//Its okay to ignore it. Its is a notification to the browser.
	}

	public Applet getApplet(String name) {
		String thisName = parameters.get("name");
		if (thisName == null) return null;
		return thisName.equals(name) ? applet : null;
	}

	public Enumeration<Applet> getApplets() {
		Vector<Applet> apps = new Vector<Applet>();
		apps.add(applet);
		return apps.elements();
	}

	public AudioClip getAudioClip(URL url) {
		log("NOT YET IMPLEMENTED getAudioClip=" + url);
		return null;
	}

	public Image getImage(URL url) {
		synchronized (imageCache) {
			WeakReference<Image> ref = imageCache.get(url);
			Image img;
			if (ref == null || (img = ref.get()) == null ) {
				img = Toolkit.getDefaultToolkit().createImage(url);
				ref = new WeakReference<Image>(img);
				imageCache.put(url, ref);
			}
			return img;
		}
	}

	public InputStream getStream(String key) {
		return inputCache.get(key);
	}

	public Iterator<String> getStreamKeys() {
		return Collections.unmodifiableSet(inputCache.keySet()).iterator();
	}

	public void setStream(String key, InputStream stream) throws IOException {
		inputCache.put(key, stream);
	}

	public void showDocument(URL url) {
		if(url.toString().contains("outofdate")) {
			log("===============================================================================");
			log("RSBot is currently outdated, please wait patiently for an update.");
			log("Posting a duplicate thread or a thread somewhere other than ");
			log("the download section will result in an instant BAN.");
			log("===============================================================================");
		} else {
			log("Attempting to show: " + url.toString());
		}
	}

	public void showDocument(URL url, String target) {
		if(url.toString().contains("outofdate")){
			log("===============================================================================");
			log("RSBot is currently outdated, please wait patiently for an update.");
			log("Posting a duplicate thread or a thread somewhere other than ");
			log("the download section will result in an instant BAN.");
			log("===============================================================================");
		}else
			log("Attempting to show: " + url.toString() + " [" + target + "]");
	}
	
	public void showStatus(String status) {
		log("Status: " + status);
	}

	void log(String s)
	{
		System.out.println(s);
	}
}
