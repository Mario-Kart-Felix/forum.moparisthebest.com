[code=java] public void sendObject(int objectId, int objectType, int objectFace,
Position pos) {
// Sending obj coords
StreamBuffer.OutBuffer out = StreamBuffer.newOutBuffer(16);
out.writeHeader(getEncryptor(), 85);
out.writeByte(pos.getLocalY(player.getPosition()), StreamBuffer.ValueType.C);
out.writeByte(pos.getLocalX(player.getPosition()), StreamBuffer.ValueType.C);

    // Removing current object
    out.writeHeader(getEncryptor(), 101);
    out.writeByte((objectType << 2) + (objectFace & 3), StreamBuffer.ValueType.C);
    out.writeByte(0);
    
    // Sending new object
    if (objectId != -1) {
        out.writeHeader(getEncryptor(), 151);
        out.writeByte(0, StreamBuffer.ValueType.A);
        out.writeShort(objectId, StreamBuffer.ByteOrder.LITTLE);
        out.writeByte((objectType << 2) + (objectFace & 3), StreamBuffer.ValueType.S);
    }
    send(out.getBuffer());
