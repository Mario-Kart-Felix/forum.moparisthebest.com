Op here is my first draft, it’s really buggy but logging in and out seems to fix the object syncing. Perhaps try building off that - you aren’t going to get anymore from me. File format is: ID Pos (x y z) Open? (default value) Open face Close face rest not parsed. All split by single tabs.

[code=java]package net.pure2.hades.world;

import java.io.BufferedReader;
import java.io.File;
import java.io.FileReader;
import java.util.HashMap;
import net.pure2.hades.Player;
import net.pure2.hades.entity.Position;
import net.pure2.hades.entity.player.PlayerHandler;
import net.pure2.hades.util.Misc;

/**
*

  • @author Pure_
    */
    public final class DoorHandler {

    // TODO Fix up - sometimes opening/closing spawns doors in weird locations
    // they display well if i relog after doing actions
    // coords seem to be the same as the ones we send
    // - ones after the initial door are messed, maybe the pos region isnt changing?

    /**

    • A collection of our loaded doors, Position toString() against Door.
      */
      private static final HashMap<String, Door> DOORS_LIST = new HashMap<>();

    /**

    • Loads doors from the given file.

    • @param fileName file to load data from
      */
      public static void loadDoors(String fileName) {
      final long startTime = System.currentTimeMillis();

      File file = new File(fileName);

      if (!file.exists()) {
      throw new NullPointerException("Door file does not exist (in "
      + fileName + “)”);
      }

      DOORS_LIST.clear();

      // Loading door data
      try (BufferedReader br = new BufferedReader(new FileReader(file))) {
      String line = null;

       // Reading the file line by line
 while ((line = br.readLine()) != null) {
     // Checking if the line is valid
     if (!line.startsWith("#")) {
         String[] lineComponents = line.split("\t");

         // Door id
         int objId = Integer.parseInt(lineComponents[0]);

         // Position
         String[] positionSeg = lineComponents[1].split(" ");
         int x = Integer.parseInt(positionSeg[0]);
         int y = Integer.parseInt(positionSeg[1]);
         int z = Integer.parseInt(positionSeg[2]);
         Position pos = new Position(x, y, z);
         
         // Default door state
         final boolean defaultState = Boolean.parseBoolean(lineComponents[2]);

         // Door open face
         final int openFace = Misc.getFace(lineComponents[3]);

         // Door closed face
         final int closeFace = Misc.getFace(lineComponents[4]);

         // Adding the door to the doors list
         DOORS_LIST.put(pos.toString(),
                 new Door(objId, pos, defaultState, openFace, closeFace));
     }
 }

      } catch (Exception ex) {
      ex.printStackTrace();
      }

      // Load message
      System.out.println("Loaded " + DOORS_LIST.size() + " doors in “
      + (System.currentTimeMillis() - startTime) + " ms.”);
      }

    /**

    • Flips the state of the door (in the given position).

    • @param pos Position to check
      */
      public static void flipState(Position pos) {
      // Getting the door and flipping its state
      Door door = DOORS_LIST.get(pos.toString());

      if (door == null) {
      return;
      }
      door.flipState();

      // Updating the door for other players
      for (Player p : PlayerHandler.getPlayers()) {
      // Skipping over invalid players
      if (p == null) {
      continue;
      }

       // Sending the object packet
 p.sendObject(door.getObjectId(), door.getObjectType(),
         door.getFace(), door.getPosition());

      }
      }

    /**

    • Sends all door states to the given player.
    • @param player Player
      */
      public static void sendAllDoors(Player player) {
      for (Door door : DOORS_LIST.values()) {
      if (!door.isDefaultState()) {
      player.sendObject(door.getObjectId(), door.getObjectType(),
      door.getFace(), door.getPosition());
      }
      }
      }

    /**

    • Checks if the given coordinates house a valid (handled) door.
    • @param pos Position of
    • @return boolean, door?
      */
      public static boolean isDoor(Position pos) {
      return DOORS_LIST.containsKey(pos.toString());
      }
      }

/**

  • A door object.

  • @author Pure_
    */
    class Door {

    private final int objectId;
    private final Position pos;
    private final int openFace;
    private final int closeFace;
    private final boolean defaultState;
    private boolean open;

    public Door(int objectId, Position pos, boolean defaultState, int openFace, int closeFace) {
    this.objectId = objectId;
    this.pos = pos;
    this.defaultState = defaultState;
    this.open = defaultState;
    this.openFace = openFace;
    this.closeFace = closeFace;
    }

    public void flipState() {
    open = !open;
    }

    public boolean isDefaultState() {
    return open == defaultState;
    }

    public int getObjectId() {
    return objectId;
    }

    public int getObjectType() {
    return 0;
    }

    public int getFace() {
    return open ? openFace : closeFace;
    }

    public Position getPosition() {
    return pos;
    }
    }[/code]
