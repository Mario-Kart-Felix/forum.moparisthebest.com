public void process() { /** * Add all LOCAL NPCs into a list */ for (int i = 0; i < PlayerHandler.players.length; i++) { if (PlayerHandler.players[i] != null) localNPCs.addAll(RegionManager.getLocalNpcsById((Client)PlayerHandler.players[i])); } /** * So we don't double-process... */ for (int npc : localNPCs) { if (!finalNPCs.contains(npc) && !ForceProcessNpcs.forcedList.contains(npc)) { finalNPCs.add(npc); } } localNPCs.clear(); /** * Process Forced NPCs */ for (int npc : ForceProcessNpcs.forcedList) { if (npc != -1) { Server.npcHandler.subProcess(npc); } } /** * Process them */ for (int npc : finalNPCs) { if (npc != -1 && NPCHandler.npcs[npc] != null) { Server.npcHandler.subProcess(npc); } } finalNPCs.clear(); }
