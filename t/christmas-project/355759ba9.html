Keep things you didn’t write out of here (lawl and super). Anyway, here is my (old) entry:

I wrote this my second year in college in a computer graphics class, written in C++ using OpenGL, it’s cross-platform, if you want to compile it use the following command:

g++ -lglut threedscene.cpp shapes.cpp -o threedscene

(for debian you need freeglut3-dev)

It isn’t just a scene, you can move around in it and everything, change the fog color or anything else if you want, I like it looking just a little creepy.

Here is the code:

void SnowBall(void){

}

void Top(void){
IT(0.25, 0.25, 0.25, 1,1,1, 0,snowball.y*1.3,0, 0,0,0,0, Hat);

void Bottom(void){

IT(2,2,2, 0,snowball.y2.2,0, 0,0,0,0, SnowBall);
IT(3,3,3, 0,0,0, 0,0,0,0, SnowBall);
}

void Snowman(void){
glMaterialfv(GL_FRONT_AND_BACK, GL_EMISSION, snowemission);
IT(1,1,1, 0,snowball.z,0, 0,0,0,0, Bottom);
glMaterialfv(GL_FRONT_AND_BACK, GL_EMISSION, def);
}

//end snowman

//start tree

}

IT(1,1,1, 0,layer.y*1.6,0, 0,0,0,0, Layer);

{

return;
posx += cos(d2r(mvdir)) * (speedtimerInterval);
posz += sin(d2r(mvdir)) * (speedtimerInterval);

glutPostRedisplay();
}

void special(int key, int x, int y)
{
static const double inc = 10;
if (key == GLUT_KEY_LEFT) {
if((abs(mvdir) - abs(lkhdir)) == -90)
return;
lkhdir -= inc;
} else if (key == GLUT_KEY_RIGHT) {
if((abs(mvdir) - abs(lkhdir)) == 90)
return;
lkhdir += inc;
} else if (key == GLUT_KEY_UP) {
if(lkvdir == 90)
return;
lkvdir += inc;
} else if (key == GLUT_KEY_DOWN) {
if(lkvdir == -60)
return;
lkvdir -= inc;
}
if(!moving)
glutPostRedisplay();
}

[code]#include
using namespace std;

#include “shapes.h”
#include <GL/glut.h>
const int SLICES = 25;
const int STACKS = 20;
const int LOOPS = 20;

void quad()
{
glNormal3f(0,0,1);
glBegin(GL_POLYGON);
glVertex3f(-0.5, -0.5, 0);
glVertex3f(+0.5, -0.5, 0);
glVertex3f(+0.5, +0.5, 0);
glVertex3f(-0.5, +0.5, 0);
glEnd();
}
void sphere()
{ glutSolidSphere(0.5, SLICES, STACKS); }
void tetrahedron()
{
const double S = 1.0;
glMatrixMode(GL_MODELVIEW);
glPushMatrix();
glScaled(S, S, S);
glutSolidTetrahedron();
glPopMatrix();
}
void octahedron()
{
const double S = 0.68;
glMatrixMode(GL_MODELVIEW);
glPushMatrix();
glScaled(S, S, S);
glutSolidOctahedron();
glPopMatrix();
}
void teapot()
{ glutSolidTeapot(0.3); }
void openEndedCone()
{ glutSolidCone(0.5, 1.0, SLICES, STACKS); }
void closedCone()
{
glutSolidCone(0.5, 1.0, SLICES, STACKS);
IT(1,1,1, 0,0,0, 180,0,1,0, disk);
}
void torus()
{ glutSolidTorus(0.1, 0.4, SLICES, STACKS); }
void cube()
{ glutSolidCube(1.0); }

static GLUquadric *q = gluNewQuadric();
void disk()
{
gluDisk(q, 0, 0.5, SLICES, LOOPS);
}
void openCylinder()
{
gluCylinder(q, 0.5, 0.5, 1, SLICES, STACKS);
}
void closedCylinder()
{
gluCylinder(q, 0.5, 0.5, 1, SLICES, STACKS);
IT(1,1,1, 0,0,0, 180,0,1,0, disk);
IT(1,1,1, 0,0,1, 0,0,1,0, disk);
}

void IT(double sx, double sy, double sz,
double dx, double dy, double dz,
double deg, double vx, double vy, double vz,
ObjectFunc func)
{
glPushMatrix();

glTranslatef(dx, dy, dz);
glRotatef(deg, vx, vy, vz);
glScalef(sx, sy, sz);
(*func)();

glPopMatrix();

}

void IT(double r, double g, double b,
double sx, double sy, double sz,
double dx, double dy, double dz,
double deg, double vx, double vy, double vz,
ObjectFunc func)
{
glPushAttrib(GL_ALL_ATTRIB_BITS);
glColor3f(r, g, b);
IT(sx, sy, sz, dx, dy, dz, deg, vx, vy, vz, func);
glPopAttrib();
}
