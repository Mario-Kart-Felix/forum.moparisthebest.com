Only rule is it has to be drawn with java’s graphics objects only, no importing images you’ve gone and drawn in photoshop.

[code]/**

  • Christmas Applet with a snow-dude and snow flakes.
  • @author pplsuqbawlz
    */

import javax.swing.;
import java.awt.;
import java.awt.image.BufferedImage;

public class silly extends JFrame implements Runnable {
private final int SNOWFLAKES = 500;
private Canvas canvas;
private BufferedImage image;
private BufferedImage buffer;
private int width;
private int height;

int snowflakeX[];
int snowflakeY[];


public static void main(String args[]) {
    new silly();
}

public silly() {
    canvas = new Canvas();
    this.setTitle("Christmas Animation");
    this.setBackground(Color.blue);
    this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    this.add("Center", canvas);
    this.pack();
    this.setVisible(true);
    width = 800;
    height = 600;
    image = new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);
    buffer = new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);
    snowflakeX = new int[SNOWFLAKES];
    snowflakeY = new int[SNOWFLAKES];
    for (int l = 0; l < snowflakeX.length; l++) {
        snowflakeX[l] = (int) (width * Math.random());
        snowflakeY[l] = (int) (height * Math.random());
    }
    new Thread(this).start();
    this.setSize(width, height);

}

public void run() {
    while (true) {
        try {
            long time = System.currentTimeMillis();

            Insets i = this.getInsets();
            int w = this.getWidth() - (i.left + i.right);
            int h = this.getHeight() + i.top + i.bottom;
            if ((w != width || h != height) && (w > 1 && h > 1)) {
                width = w;
                height = h;
                for (int l = 0; l < snowflakeX.length; l++) {
                    snowflakeX[l] = (int) (width * Math.random());
                    snowflakeY[l] = (int) (height * Math.random());
                }
                image = new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);
                buffer = new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);


                Graphics2D g = (Graphics2D) image.getGraphics();
                //sky
                g.setPaint(new GradientPaint(0, 0, Color.blue.darker(), 0, height, Color.blue.darker().darker().darker().darker()));
                g.fillRect(0, 0, width, height);
                //ground
                g.setPaint(new GradientPaint(0, height - 200, Color.gray, 0, height, Color.lightGray));
                g.fillRect(0, height - 200, width, 200);
                // Arms
                g.setColor(Color.black);
                g.fillRect(width / 2 - 100, height - 305, 200, 10);
                // Body
                g.setColor(Color.white);
                g.fillOval(width / 2 - 30, height - 400, 60, 60);
                g.fillOval(width / 2 - 50, height - 350, 100, 100);
                g.fillOval(width / 2 - 70, height - 260, 140, 140);
                // eyes
                g.setColor(Color.black);
                g.fillOval(width / 2 - 14, height - 380, 9, 9);
                g.fillOval(width / 2 + 6, height - 380, 9, 9);
                // buttons
                g.fillOval(width / 2 - 5, height - 325, 11, 11);
                g.fillOval(width / 2 - 5, height - 300, 11, 11);
                g.fillOval(width / 2 - 5, height - 275, 11, 11);

            }

            Graphics gx = buffer.getGraphics();
            gx.drawImage(image, 0, 0, null);


            gx.setColor(Color.white);
            for (int l = 0; l < snowflakeY.length; l++) {
                int incr = 1 + (int) (3 * Math.random());
                snowflakeY[l] += incr;
                if (snowflakeY[l] > height) {
                    snowflakeY[l] = 2;
                }
                gx.drawLine(snowflakeX[l], snowflakeY[l], snowflakeX[l] + (int) ((Math.random() * 7) - 4), snowflakeY[l] + (int) ((Math.random() * 7) - 4));
            }

            canvas.getGraphics().drawImage(buffer, 0, 0, null);

            time = System.currentTimeMillis() - time;
            if(time < 10) {
                Thread.sleep(10 - time);
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

12/27/2006: Just made a couple mods to the snowglobe, and as someone requested it as an applet as well, I created a separate class to run it. I did a bit more commenting, but I got bored of it too quickly

ChristmasProject.java

import javax.swing.JApplet;
import javax.swing.JFrame;

/**
 * @(#)ChristmasProject.java
 * 
 *
 * @author S
 */
public final class ChristmasProject extends JApplet {
    private static SnowGlobe snowGlobe; 
    
    /**
     * @param args command line arguments
     */
    public final static void main(String[] args) {
        JFrame.setDefaultLookAndFeelDecorated(false);
        JFrame mainFrame = new JFrame("Christmas SnowGlobe");
        mainFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        
        snowGlobe = new SnowGlobe(mainFrame);
        mainFrame.getContentPane().add(snowGlobe);
        
        mainFrame.setVisible(true);
        mainFrame.pack();
        
        new Thread(snowGlobe).start();
    }

    public void init() {
        snowGlobe = new SnowGlobe(this);
        setContentPane(snowGlobe);
    }
    
    public void start() {
        new Thread(snowGlobe).start();
    }
}

SnowGlobe.java

import java.awt.Color;
import java.awt.Container;
import java.awt.Dimension;
import java.awt.GradientPaint;
import java.awt.Graphics;
import java.awt.Graphics2D;
import java.awt.Image;
import java.awt.Polygon;
import java.awt.image.BufferedImage;
import java.util.ArrayList;
import java.util.List;
import javax.swing.JPanel;

/**
 * @(#)SnowGlobe.java
 * 
 *
 * @author S
 */
public class SnowGlobe extends JPanel implements Runnable {
    private static final int WIDTH = 600, HEIGHT = 600;
    
    private BufferedImage sceneryBase;
    private Container parentContainer;
    private Image imageBuffer;
    private List<SnowFlake> snowFlakes;
    private static boolean isRunning = false;
    
    
    /**
     * Default Constructor
     * Performs initialization of display panel and image components, 
     * drawing default snowglobe and background scenery.  
     */
    public SnowGlobe(Container parentContainer) {
        super(null, true);
        this.parentContainer = parentContainer;
        setPreferredSize(new Dimension(WIDTH, HEIGHT));
        isRunning = true;
        
        sceneryBase = new BufferedImage(WIDTH, HEIGHT, BufferedImage.TYPE_INT_RGB);
        imageBuffer = new BufferedImage(WIDTH, HEIGHT, BufferedImage.TYPE_INT_RGB);
        Graphics2D g = sceneryBase.createGraphics();
        
        // Add gradient background
        g.setPaint(new GradientPaint(0, 0, new Color(100, 100, 255), 0, (HEIGHT / 2), Color.BLUE));
        g.fillRect(0, 0, WIDTH, (HEIGHT / 2));
        g.setPaint(new GradientPaint(0, (HEIGHT / 2), Color.BLUE, 0, HEIGHT, new Color(25, 25, 25)));
        g.fillRect(0, (HEIGHT / 2), WIDTH, (HEIGHT / 2));
        
        // Draw Christmas Tree
        g.setColor(new Color(100, 5, 10));
        g.fillRoundRect((WIDTH / 2) - 20, HEIGHT - 150, 40, 120, 20, 20);
        
        /*
         * Alternative smoother tree style
            treeLayer.addPoint((WIDTH / 2), HEIGHT - y);
            treeLayer.addPoint((WIDTH / 4) + x, HEIGHT - 125);
            treeLayer.addPoint((WIDTH / 2), HEIGHT - y - 75);
            treeLayer.addPoint((WIDTH / 4) * 3 - x, HEIGHT - 125);
         */
        for(int x = 0, y = 200; y < HEIGHT - 100; y += 45, x += 15) {
            // Draw tree tier
            Polygon treeLayer = new Polygon();
            treeLayer.addPoint((WIDTH / 2), HEIGHT - y);
            treeLayer.addPoint((WIDTH / 4) + x, HEIGHT - y + 125);
            treeLayer.addPoint((WIDTH / 2), HEIGHT - y + 75);
            treeLayer.addPoint((WIDTH / 4) * 3 - x, HEIGHT - y + 125);
            g.setColor(new Color(0, 128, 0));
            g.fillPolygon(treeLayer);
        }
        
        // Initializion of snowflake collection
        snowFlakes = new ArrayList<SnowFlake>();
    }
    
    
    public void run() {
        boolean createNewSnowFlakes = false;
        while(isRunning && parentContainer.isVisible()) {
            if((createNewSnowFlakes = !createNewSnowFlakes)) {
                int newSnowFlakes = 5 + (int)(Math.random() * 3);
                for(int i = 0; i < newSnowFlakes; i++)
                    snowFlakes.add(new SnowFlake(25 + (int)(Math.random() * (WIDTH - 50))));
            }
            
            // Reset buffer
            Graphics g = imageBuffer.getGraphics();
            g.drawImage(sceneryBase, 0, 0, this);
            
            // Draw and manage snow
            List<SnowFlake> snowFlakesToRemove = new ArrayList<SnowFlake>();
            for(SnowFlake s : snowFlakes)
                if(!s.paint(g))
                    snowFlakesToRemove.add(s);
            for(SnowFlake s : snowFlakesToRemove)
                snowFlakes.remove(s);
            
            // Draw snowglobe
            g.setColor(new Color(255, 255, 255, 75));
            g.fillOval(0, 0, WIDTH, HEIGHT);
            g.setColor(Color.BLACK);
            for(int i = 0; i < 15; i++)
                g.drawOval(i, i, WIDTH - 2 * i, HEIGHT - 2 * i);
            
            repaint();
            try {
                Thread.sleep(75);
            } catch(InterruptedException e) {}
        }
    }
    
    
    public class SnowFlake {
        public int x, y;
        private int finalX, finalY;
        private Color snowFlakeColor;
        private Polygon snowFlakeShape;
        
        public SnowFlake(int x) {
            this.x = finalX = x;
            this.y = (HEIGHT / 2) - (int)Math.sqrt(Math.pow((HEIGHT / 2D), 2) - Math.pow((x - (HEIGHT / 2D)), 2)) + 25;
            finalY = HEIGHT - 15 - y;
            
            generateSnowflake();
        }
        
        private void generateSnowflake() {
            snowFlakeColor = new Color(225 + (int)(Math.random() * 30),
                225 + (int)(Math.random() * 30), 225 + (int)(Math.random() * 30));
            
            snowFlakeShape = new Polygon();
            for(int i = 0; i < 3 + Math.ceil(Math.random()); i++)
                snowFlakeShape.addPoint(x + (int)(Math.random() * 6) - 3, y + (int)(Math.random() * 6) - 3);
        }
        
        public boolean paint(Graphics g) {
            int averageY = 0, 
                yOffset = 1 + (int)(Math.random() * 5), 
                xOffset = (int)(Math.random() * 2) - 1;
            for(int i = 0; i < snowFlakeShape.npoints; i++) {
                snowFlakeShape.ypoints[i] += yOffset;
                averageY += snowFlakeShape.ypoints[i];
                snowFlakeShape.ypoints[i] += xOffset;
            }
            
            if((averageY / snowFlakeShape.npoints) >= finalY)
                return false;
            
            g.setColor(snowFlakeColor);
            g.fillPolygon(snowFlakeShape);
            return true;
        }
    }
    
    
    public void paintComponent(Graphics g) {
        g.drawImage(imageBuffer, 0, 0, this);
    }
    
    public void update(Graphics g) {
        paint(g);
    }
}
