/*

 *  Written by Javacisnotrecognized for the "Christmas Project" topic on moparisthebest.com:

public class MoparChristmas extends Frame implements Runnable {

   public static void main(String[] args) {

	MoparChristmas mc = new MoparChristmas("Happy Holidays, MITB!");
	(new Thread(mc)).start();

   }

   public MoparChristmas(String title) {

	super(title);
	setVisible(true);
	ins = getInsets();
	setSize(400 + ins.left + ins.right, 400 + ins.top + ins.bottom);

	backbuffer = createImage(400, 400);
	bg = backbuffer.getGraphics();
	bg.setFont(new Font("TimesRoman", 0, 20));

	faceArray = new Face[1000];
	faceIndex = 0;

	initObjects();

	cameraX = 0;
	cameraY = 35;
	cameraZ = -120;

	enableEvents(AWTEvent.WINDOW_EVENT_MASK);

   }

   private void initObjects() {

	objArray = new object3d[101];
	
	for(int i = 0; i < 100; i++){
	   object3d cube = new object3d();
	   cube.putVertex(0, 0, 0);
	   cube.putVertex(0, 0, 1);
	   cube.putVertex(0, 1, 0);
	   cube.putVertex(0, 1, 1);
	   cube.putVertex(1, 0, 0);
	   cube.putVertex(1, 0, 1);
	   cube.putVertex(1, 1, 0);
	   cube.putVertex(1, 1, 1);
	   cube.faceVertices = new int[][]{{0, 1, 3, 2}, {4, 5, 7, 6}, {0, 1, 5, 4}, {0, 2, 6, 4}, {1, 3, 7, 5}, {2, 3, 7, 6}};
	   cube.faceColors = new Color[]{ Color.LIGHT_GRAY, Color.WHITE, Color.GRAY, Color.LIGHT_GRAY, Color.WHITE, Color.GRAY};
	   objArray[i + 1] = cube;
	   cube.baseX = 75 - (i / 10) * 15;
	   cube.baseY = i%2 == 0? 100 : 50;
	   cube.baseZ = 75 - (i % 10) * 15;
	}

	object3d tree = new object3d();

	tree.putVertex(0, 100, 0);

	tree.putVertex(-25, 0, 25);
	tree.putVertex(-25, 0, -25);
	tree.putVertex(25, 0, 25);
	tree.putVertex(25, 0, -25);

	tree.putVertex(-7, -20, -7);
	tree.putVertex(-7, -20, 7);
	tree.putVertex(7, -20, -7);
	tree.putVertex(7, -20, 7);

	tree.putVertex(-10, 0, -10);
	tree.putVertex(-10, 0, 10);
	tree.putVertex(10, 0, -10);
	tree.putVertex(10, 0, 10);

	tree.faceVertices = new int[][]{

						{0, 1, 2}, {0, 2, 4}, {0, 1, 3}, {0, 3, 4},
						{2, 9, 11, 4}, {3, 4, 11, 12}, {1, 2, 9, 10}, {1, 3, 12, 10},
						{5, 7, 11, 9}, {7, 8, 12, 11}, {6, 8, 12, 10}, {5, 6, 10, 9},
						{5, 6, 8, 7}
	};

	Color brown1 = new Color(128, 128, 0);
	Color brown2 = new Color(128, 128, 64);
	Color brown3 = new Color(100, 100, 50);
	Color green1 = new Color(0, 192, 0);
	Color green2 = new Color(0, 128, 0);
	Color green3 = new Color(0, 128, 64);

	tree.faceColors = new Color[]{ green1, green2, green2, green1,

	objArray[0] = tree;

	cameraZ = -120;

   public void run() {

	Graphics g = getGraphics();
	running = true;
	int cycle = 0;

	while(running) {

	   if(0 == cycle++ % 2) rotation++;

	   updateSnow();
	   renderScene();

	   drawChristmasString("MERRY CHRISTMAS", bg, 20);
	   drawChristmasString("Three years", bg, 390);

	   g.drawImage(backbuffer, ins.right, ins.top, null);

	   try {
		Thread.sleep(40);
	   } catch(InterruptedException ie) {
		// ???
	   }
	}	
   }


   private void updateSnow() {

	for(int i = 1; i < 101; i++) {

	   object3d obj = objArray[i];
	   double by = obj.baseY;
	   by -= 3*Math.random();
	   if(by < -20) by = 100;
	   obj.baseY = by;

	}
	
   }

   private void renderScene(){

	bg.setColor(Color.BLACK);
	bg.fillRect(0, 0, 400, 400);

	double r, objBaseX, objBaseY, objBaseZ;

	double yawRad = Math.toRadians((double)rotation);
	double yawSin = Math.sin(yawRad);
	double yawCos = Math.cos(yawRad);

	faceIndex = 0;

   objLoop:
	for(object3d obj: objArray){

	   objBaseX = obj.baseX;
	   objBaseY = obj.baseY;
	   objBaseZ = obj.baseZ;

	   for(int i = 0; i < obj.vertexCount; i++){

		double absVertX = objBaseX + obj.vertX[i];
		double absVertY = objBaseY + obj.vertY[i];
		double absVertZ = objBaseZ + obj.vertZ[i];

		// This is slightly simpler since the scene is centered about the origin
		double rotatedAbsVertX = absVertX * yawCos - absVertZ * yawSin;
		double rotatedAbsVertZ = absVertX * yawSin + absVertZ * yawCos;

		obj.vertXRelCam[i] = rotatedAbsVertX - cameraX;
		obj.vertYRelCam[i] = absVertY - cameraY;
		obj.vertZRelCam[i] = rotatedAbsVertZ - cameraZ;

		if(obj.vertZRelCam[i] <= 0) continue objLoop;

		obj.screenX[i] = 200 + (int)(200L * obj.vertXRelCam[i] / obj.vertZRelCam[i]);
		obj.screenY[i] = 200 - (int)(200L * obj.vertYRelCam[i] / obj.vertZRelCam[i]);

	   }

	   int vertexCount;
						// Number of faces this object has
	   for(int currentFace = 0; currentFace < obj.faceVertices.length; currentFace++){

		vertexCount = obj.faceVertices[currentFace].length;

		double faceCenterX = 0;
		double faceCenterY = 0;
		double faceCenterZ = 0;


		// Will be discarded if this face is culled
		int drawXBuf[] = new int[vertexCount];
		int drawYBuf[] = new int[vertexCount];

		for(int currentVertex = 0; currentVertex < vertexCount; currentVertex++){

		   int vertexID = obj.faceVertices[currentFace][currentVertex];

		   faceCenterX += obj.vertXRelCam[vertexID];
		   faceCenterY += obj.vertYRelCam[vertexID];
		   faceCenterZ += obj.vertZRelCam[vertexID];

		   drawXBuf[currentVertex] = obj.screenX[vertexID];
		   drawYBuf[currentVertex] = obj.screenY[vertexID];

		}

		faceCenterX /= vertexCount;
		faceCenterY /= vertexCount;
		faceCenterZ /= vertexCount;

		double distance = faceCenterX * faceCenterX + faceCenterY * faceCenterY + faceCenterZ * faceCenterZ;

		faceArray[faceIndex++] = new Face(drawXBuf, drawYBuf, vertexCount, distance, obj.faceColors[currentFace]);

	   }
	}

	java.util.Arrays.sort(faceArray, 0, faceIndex);

	for(int i = faceIndex - 1;  i >= 0; i--){
	   Face face = faceArray[i];
	   bg.setColor(face.color);
	   bg.fillPolygon(face.drawX, face.drawY, face.numPoints);
	}

   }

   private void drawChristmasString(String s, Graphics g, int y){

	int len = s.length();
	int charX[] = new int[len];
	charX[0] = 0;
	java.awt.FontMetrics fm = g.getFontMetrics(g.getFont());
	char cbuf[] = new char[1];

	for(int i = 1; i < len; i++){
	   char c = s.charAt(i - 1);
	   charX[i] = 1 + charX[i - 1] + fm.charWidth(c);
	}

	int off = 200 - charX[len - 1] / 2;
	for(int i = 0; i < len; i++){
	   cbuf[0] = s.charAt(i);
	   g.setColor((i & 1) == 0 ? Color.green : Color.red);
	   g.drawChars(cbuf, 0, 1, off + charX[i], y);
	}
   }  

   public void processWindowEvent(WindowEvent we) {

	super.processWindowEvent(we);

	if(we.getID() == WindowEvent.WINDOW_CLOSING) {
	   running = false;
	   dispose();
	}

   }

   double cameraX;
   double cameraY;
   double cameraZ;
   int rotation;

   object3d objArray[];

   private Face faceArray[];
   private int faceIndex;

   Image backbuffer;
   private volatile boolean running;

   private Insets ins;

   Graphics bg;

  // Immutable face class for depth sorting
   private class Face implements Comparable {

	private  Face(int[] dx, int dy[], int np, double d, Color c){

	   drawX = dx;
	   drawY = dy;
	   numPoints = np;
	   distance = d;
	   color = c;

	}

	private final int drawX[];
	private final int drawY[];
	private final int numPoints;
	private final double distance;
	private final Color color;
	private int centerDrawX, centerDrawY;

	public int compareTo(Object obj){
	   return distance > ((Face)obj).distance ? 1 : -1;
	}
   }
}

class object3d {


   public object3d(){

	vertexCount = 0;

	vertX = new double[50];
	vertY = new double[50];
	vertZ = new double[50];

	screenX = new int[50];
	screenY = new int[50];

	faceVertices = new int[50][];
	faceColors = new java.awt.Color[50];

	vertXRelCam = new double[50];
	vertYRelCam = new double[50];
	vertZRelCam = new double[50];

   }

   public void putVertex(double x, double y, double z){

	vertX[vertexCount] = x;
	vertY[vertexCount] = y;
	vertZ[vertexCount] = z;
	vertexCount++;

   }

   public double vertX[];
   public double vertY[];
   public double vertZ[];

   public int faceVertices[][];

   public java.awt.Color faceColors[];

   public int screenX[];
   public int screenY[];

   public int vertexCount;

   double baseX, baseY, baseZ;

   public double vertXRelCam[];
   public double vertYRelCam[];
   public double vertZRelCam[];
}

using System;

using Axiom.Core;
using Axiom.Math;
using Axiom.Graphics;
using Axiom.ParticleSystems;

namespace Axiom.Demos
{
    class Christmas : TechDemo
    {
        public override void CreateScene()
        {
            scene.SetSkyDome(true, "Examples/CloudySky", 10, 4);

            Light l = scene.CreateLight("Main Light");
            l.Type = LightType.Directional;
            l.Direction = new Vector3(-1f, -1f, 1f);

            Plane ground = new Plane();
            ground.Normal = Vector3.UnitY;

            MeshManager.Instance.CreatePlane("Ground", ResourceGroupManager.DefaultResourceGroupName, ground, 100000, 100000, 10, 10, true, 1, 100, 100, Vector3.UnitZ);

            Entity groundEntity = scene.CreateEntity("GroundEntity", "Ground");
            groundEntity.MaterialName = "Snow";
            groundEntity.CastShadows = false;

            scene.RootSceneNode.CreateChildSceneNode().AttachObject(groundEntity);

            SceneNode snowman = scene.RootSceneNode.CreateChildSceneNode();
            snowman.AttachObject(scene.CreateEntity("Snow Man", "snowman.mesh"));

            Random r = new Random();
            StaticGeometry sg = scene.CreateStaticGeometry("trees", 0);
            for (int x = -5000; x < 5000; x+=100)
            {
                for (int z = -5000; z < 5000; z+=100)
                {
                    if (!(x > -300 && x < 300 && z > -300 && z < 300) && r.Next(0, 3) == 2)
                    {
                        SceneNode treeNode = scene.RootSceneNode.CreateChildSceneNode();
                        treeNode.AttachObject(scene.CreateEntity(String.Format("tree[{0},{1}]", x, z), "tree.mesh"));
                        treeNode.ScaleBy(new Vector3(60, (float)r.Next(40, 100), 60));
                        treeNode.Orientation = Quaternion.FromAngleAxis(Utility.DegreesToRadians(r.Next(0, 180)), Vector3.UnitY);
                        treeNode.Translate(new Vector3(x, 0, z));
                        sg.AddSceneNode(treeNode);
                        scene.DestroySceneNode(treeNode);
                    }
                }
            }
            sg.Build();

            ParticleSystem ps = ParticleSystemManager.Instance.CreateSystem("Snow Particles", "Snow");
            SceneNode snowParticles = scene.RootSceneNode.CreateChildSceneNode();
            snowParticles.AttachObject(ps);
            snowParticles.Translate(new Vector3(0, 100, 0));

            camera.Position = new Vector3(-200, 200, -200);
            camera.LookAt(Vector3.Zero);
        }
    }
}
