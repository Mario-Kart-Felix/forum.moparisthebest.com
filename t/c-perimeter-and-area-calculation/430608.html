Rectangle.cs

[code]using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace Figuren.Components
{
class Rechthoek
{
public decimal omtrek(decimal hoogte, decimal breedte)
{
return 2 * hoogte + 2 * breedte;
}

    public decimal oppervlakte(decimal hoogte, decimal breedte)
    {
        return hoogte * breedte;
    }
}

}[/code]

Triangle.cs

[code]using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace Figuren.Components
{
class Driehoek
{
public decimal omtrek(decimal hoogte, decimal breedte)
{
return hoogte + breedte + Convert.ToDecimal(Math.Sqrt(System.Math.Pow(Convert.ToDouble(hoogte), 2) + System.Math.Pow(Convert.ToDouble(breedte), 2)));
}

    public decimal oppervlakte(decimal hoogte, decimal breedte)
    {
        return (hoogte * breedte) / 2;
    }
}

}[/code]

Circle.cs

[code]using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace Figuren.Components
{
class Cirkel
{
public decimal omtrek(decimal straal)
{
return 2 * Convert.ToDecimal(Math.PI) * straal;
}

    public decimal oppervlakte(decimal straal)
    {
        return Convert.ToDecimal(Math.Pow(Convert.ToDouble(straal), 2) * Math.PI);
    }
}

}[/code]

Form1.cs

[code]using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Windows.Forms;
using Figuren.Components;

namespace OmtrekOppervlakte
{
public partial class OmtrekOppervlakte : Form
{
// Deze 2 booleans zijn nodig om de status van de 2 checkboxen te representeren.
private Boolean rechthoekBoolean = false;
private Boolean driehoekBoolean = false;
private Boolean cirkelBoolean = false;

    public OmtrekOppervlakte()
    {
        InitializeComponent();
    }


    // Hier wordt de status van checkBoxRechthoek in een boolean gezet.
    private void checkBoxRechthoek_CheckedChanged(object sender, EventArgs e)
    {
        if (rechthoekBoolean == false)
        {
            rechthoekBoolean = true;
        }
        else
        {
            rechthoekBoolean = false;
        }
    }


    // Hier wordt de status van checkBoxDriehoek in een boolean gezet.
    private void checkBoxDriehoek_CheckedChanged(object sender, EventArgs e)
    {
        if (driehoekBoolean == false)
        {
            driehoekBoolean = true;
        }
        else
        {
            driehoekBoolean = false;
        }
    }


    // Hier wordt de status van checkBoxCirkel in een boolean gezet en de andere figuren worden uitgeschakeld.
    private void checkBoxCirkel_CheckedChanged(object sender, EventArgs e)
    {
        if (cirkelBoolean == false)
        {
            checkBoxRechthoek.Checked = false;
            checkBoxDriehoek.Checked = false;
            
            cirkelBoolean = true;


            checkBoxRechthoek.Enabled = false;
            checkBoxDriehoek.Enabled = false;
            textBoxBreedte.Enabled = false;


            labelHoogte.Text = "Straal";
        }
        else
        {
            cirkelBoolean = false;


            checkBoxRechthoek.Enabled = true;
            checkBoxDriehoek.Enabled = true;
            textBoxBreedte.Enabled = true;


            labelHoogte.Text = "Hoogte";
        }
    }


    private void buttonBereken_Click(object sender, EventArgs e)
    {
        textBoxResultaat.Text = "De omtrek en oppervlakte:";


        // Hier wordt gecontroleerd of de waarden van textBoxHoogte en textBoxBreedte getallen zijn.
        decimal number;
        Boolean hoogteBoolean = Decimal.TryParse(Convert.ToString(textBoxHoogte.Text), out number);
        Boolean breedteBoolean = Decimal.TryParse(Convert.ToString(textBoxBreedte.Text), out number);


        // [---RECHTHOEK & DRIEHOEK---]
        // Indien de ingevoerde hoogte en breedte getallen zijn, ga verder.
        if (hoogteBoolean == true && breedteBoolean == true)
        {
            if (rechthoekBoolean == true)
            {
                Rechthoek rechthoekInstance = new Rechthoek();
                textBoxResultaat.Text += Environment.NewLine + "Rechthoek Omtrek: " + Math.Round(rechthoekInstance.omtrek(Convert.ToDecimal(textBoxHoogte.Text), Convert.ToDecimal(textBoxBreedte.Text)), 2) + " cm";
                textBoxResultaat.Text += Environment.NewLine + "Rechthoek Oppervlakte: " + Math.Round(rechthoekInstance.oppervlakte(Convert.ToDecimal(textBoxHoogte.Text), Convert.ToDecimal(textBoxBreedte.Text)), 2) + " cm²";
            }


            if (driehoekBoolean == true)
            {
                if (rechthoekBoolean == true)
                {
                    textBoxResultaat.Text += Environment.NewLine;
                }


                Driehoek driehoekInstance = new Driehoek();
                textBoxResultaat.Text += Environment.NewLine + "Driehoek Omtrek: " + Math.Round(driehoekInstance.omtrek(Convert.ToDecimal(textBoxHoogte.Text), Convert.ToDecimal(textBoxBreedte.Text)), 2) + " cm";
                textBoxResultaat.Text += Environment.NewLine + "Driehoek Oppervlakte: " + Math.Round(driehoekInstance.oppervlakte(Convert.ToDecimal(textBoxHoogte.Text), Convert.ToDecimal(textBoxBreedte.Text)), 2) + " cm²";
            }
        }
        else if (cirkelBoolean == false)
        {
            if (hoogteBoolean == false || breedteBoolean == false)
            {
                voegNummerIn();
            }
        }


        // [---CIRKEL---]
        // Indien het ingevoerde straal een getal is en de cirkel checkbox geslecteerd is, ga verder.
        if (cirkelBoolean == true && hoogteBoolean == true)
        {
            Cirkel cirkelInstance = new Cirkel();
            textBoxResultaat.Text = "De omtrek en oppervlakte:";
            textBoxResultaat.Text += Environment.NewLine + "Cirkel Omtrek: " + Math.Round(cirkelInstance.omtrek(Convert.ToDecimal(textBoxHoogte.Text)), 2) + " cm";
            textBoxResultaat.Text += Environment.NewLine + "Cirkel Oppervlakte: " + Math.Round(cirkelInstance.oppervlakte(Convert.ToDecimal(textBoxHoogte.Text)), 2) + " cm²";
        }
        else if (cirkelBoolean == true && hoogteBoolean == false)
        {
            voegNummerIn();
        }


        // Indien geen opties geselecteerd zijn, ga verder.
        if (rechthoekBoolean == false && driehoekBoolean == false && cirkelBoolean == false)
        {
            selecteerOptie();
        }
    }


    private void voegNummerIn()
    {
        if (rechthoekBoolean == true || driehoekBoolean == true)
        {
            textBoxResultaat.Text += Environment.NewLine + Environment.NewLine + "Voer de hoogte en/of breedte in.";
        }
        else if (cirkelBoolean == true)
        {
            textBoxResultaat.Text += Environment.NewLine + Environment.NewLine + "Voer de straal in.";
        }
    }


    private void selecteerOptie()
    {
        textBoxResultaat.Text += Environment.NewLine + Environment.NewLine + "Gelieve een optie te selecteren.";
    }
}

}[/code]
