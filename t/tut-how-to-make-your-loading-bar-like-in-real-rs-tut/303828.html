First open up your client folder then client.java and search:

aClass30_Sub2_Sub1_Sub4_1272.method381

Then you should see something like this:

aClass30_Sub2_Sub1_Sub4_1272.method381(0xffffff , "*YourClient* is loading - Please wait...", 23693, c1 / 2 - 26 - byte1, c / 2); int j = c1 / 2 - 18 - byte1; // Colours Class30_Sub2_Sub1.method337(c / 2 - 152, 304, 34, 0x990000, j, true); Class30_Sub2_Sub1.method337(c / 2 - 151, 302, 32, 0, j + 1, true); Class30_Sub2_Sub1.method336(30, j + 2, c / 2 - 150, 0x990000, i * 3, 0); Class30_Sub2_Sub1.method336(30, j + 2, (c / 2 - 150) + i * 3, 0, 300 - i * 3, 0); aClass30_Sub2_Sub1_Sub4_1272.method381(0xffffff, s, 23693, (c1 / 2 + 5) - byte1, c / 2);
Replace that all with this:

aClass30_Sub2_Sub1_Sub4_1272.method381(0xffffff , "*YourClient* is loading - Please wait...", 23693, c1 / 2 - 26 - byte1, c / 2); int j = c1 / 2 - 18 - byte1; // Colours Class30_Sub2_Sub1.method337(c / 2 - 152, 304, 34, 0x990000, j, true); Class30_Sub2_Sub1.method337(c / 2 - 151, 302, 32, 0, j + 1, true); Class30_Sub2_Sub1.method336(30, j + 2, c / 2 - 150, 0x990000, i * 3, 0); Class30_Sub2_Sub1.method336(30, j + 2, (c / 2 - 150) + i * 3, 0, 300 - i * 3, 0); aClass30_Sub2_Sub1_Sub4_1272.method381(0xffffff, s, 23693, (c1 / 2 + 5) - byte1, c / 2);
