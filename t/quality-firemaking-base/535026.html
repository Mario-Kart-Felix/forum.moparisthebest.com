[code=java]package server.euthymia.entity.player.skill;

import server.Server;
import server.euthymia.entity.player.Player;
import server.task.Task;

/**
*

  • @author Strucker

*/
public class Firemaking {

/*
 * The constant for the lighting sound
 */
public static final int LIGHTING_FIRE_SOUND = 375;

/*
 * The constant for when the lighting has succeeded
 */
public static final int LIGHTING_FIRE_SUCCEED = 235;

/*
 * The tinderbox constant
 */
public static final int TINDERBOX = 590;

/*
 * The firemakings skill id
 */
public static final int FIREMAKING_SKILL_ID = 11;
/*
 * Holds the logs data
 */
private enum LogData {
	NORMAL(1511, 1, 40),

	MAGIC(1513, 75, 304),

	YEW(1515, 60, 203),

	MAPLE(1517, 40, 132),

	WILLOW(1519, 30, 90),

	OAK(1521, 15, 60);

	private int log, requiredLevel, experience;

	/*
	 * Sets the values to our ints
	 */
	LogData(int log, int requiredLevel, int experience) {
		this.log = log;
		this.requiredLevel = requiredLevel;
		this.experience = experience;
	}

	/*
	 * Returns the experience
	 */
	private int getExperience() {
		return experience;
	}

	/*
	 * Returns the required level to light the log
	 */
	private int getRequiredLevel() {
		return requiredLevel;
	}

	/*
	 * returns the log to light
	 */
	private int getLog() {
		return log;
	}
}

/*
 * Starts lighthing the logs
 * 
 * @param player the player that will light the log
 * 
 * @param itemId the log that will be used with the tinderbox
 * 
 * @param usedOn the item that will light our log
 */
public static void lightLog(final Player player, int itemId, int usedOn) {
	for (LogData e : LogData.values()) {
		
		/*
		 * If the item matches a log, and we used it on a tinderbox continue
		 */
		if (itemId == e.getLog() && usedOn == TINDERBOX) {
			player.deleteItem(e.getLog(), 1);
			
			/*
			 * Schedules a new task to be executed in two ticks
			 */
			Server.getTaskScheduler().schedule(new Task(2, true) {
				protected void execute() {
					
					/*
					 * Walks the player to the south
					 */
					player.playerWalk(0, -1);
					stop();
				}
			});
		}
	}
}

}
[/code]
