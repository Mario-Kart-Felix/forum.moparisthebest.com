I don’t think it would be a 9-dimensional array, would it? This is my take on it:

See, you’d set up the array like this:

Spaces := Array of Array of TEdit;

Then, the numbers on the left side ( take away one, because you’re dealing with arrays ) is the array index of the first Array, and the numbers on top ( again, minus one ) is the array index of the second array.

So to access that upper left corner box, you’d use this:

Spaces[0][0];

Its like a table. Get it? If you need any more help, just ask.

Here, I Cruel1zed your script.
Knocked it down to 145 lines.
And also made it easier to add more puzzles. You just have to add it as an array. First half of the array being the initial state (0 for blanks) and the second half being the puzzle answer.

[code]program Sudoku;
type TPuzzle = array of Byte;
var
frmDesign : TForm;
Edits: array[0…80] of TEdit;
Exitbtn, CheckAnswerebtn, SetupChallengebtn : TButton;
Puzzle1: TPuzzle;

procedure LoadPuzzle(puz:TPuzzle);
var i: Integer;
begin
for i:= 0 to 80 do
with Edits[i] do begin
Font.Color:= clBlack;
Text:= ‘’;
ReadOnly:= False;
if puz[i] <> 0 then begin
Font.Color:= clRed;
Text:= IntToStr(puz[i]);
ReadOnly:= True;
end;
end;
end;

procedure CheckPuzzle(puz:TPuzzle);
var i: Integer;
begin
for i:= 0 to 80 do
if Edits[i].Text = IntToStr(puz[81+i]) then begin
Edits[i].Font.Color:= clRed;
Edits[i].ReadOnly:= True;
end;
end;

procedure SetupChallengeClick(sender: TObject);
begin
LoadPuzzle(Puzzle1);
end;

procedure ExitClick(sender: TObject);
begin
frmDesign.ModalResult:= mrOk;
end;

procedure CheckAnswereClick(sender: TObject);
begin
CheckPuzzle(Puzzle1);
end;

procedure frmDesignPaint(Sender: TObject);
begin
with frmDesign.Canvas do begin
Rectangle(63,1,64,176);
Rectangle(122,1,123,176);
Rectangle(2,60,184,61);
Rectangle(2,117,184,118);
end;
end;

procedure AddButton(btn:TButton; event:TNotifyEvent; title:String; y:Integer);
begin
btn:= TButton.Create(frmDesign);
with btn do begin
Parent:= frmDesign;
OnClick:= event;
SetBounds(200,y,96,25);
Hint:= title;
Caption:= title;
ParentShowHint:= False;
ShowHint:= True;
end;
end;

procedure InitForm;
var i,j: Integer;
begin
frmDesign:= CreateForm;
with frmDesign do begin
SetBounds(250,114,310,210);
Caption := ‘Sudoku V1…By r!ch!e’;
OnPaint:= @frmDesignPaint;
end;
for i:= 0 to 80 do begin
Edits[i]:= TEdit.Create(frmDesign);
with Edits[i] do begin
MaxLength:= 1;
Parent:= frmDesign;
AutoSize:= False;
j:= Round(i/9);
if (i/9) < j then j:= j-1;
SetBounds(5+(i Mod 9)20, 5+j19, 17, 17);
end;
end;
AddButton(SetupChallengebtn, @SetupChallengeClick, ‘Setup Challenge!’, 20);
AddButton(CheckAnswerebtn, @CheckAnswereClick, ‘Check Answeres!’, 50);
AddButton(Exitbtn, @ExitClick, ‘Exit’, 80);
end;

procedure SafeInitForm;
var V: TVariantArray;
begin
setarraylength(V, 0);
ThreadSafeCall(‘InitForm’, v);
end;

procedure ShowFormModal;
begin
frmDesign.ShowModal;
end;

procedure SafeShowFormModal;
var V: TVariantArray;
begin
setarraylength(V, 0);
ThreadSafeCall(‘ShowFormModal’, v);
end;

procedure SetSelfWindowState(state: TWindowState);
begin
GetSelf.WindowState:= state;
end;

procedure SafeSetSelfWindowState(state: TWindowState);
var V: TVariantArray;
begin
SetArrayLength(V, 1);
V[0] := state;
ThreadSafeCall(‘SetSelfWindowState’, V);
end;

begin
