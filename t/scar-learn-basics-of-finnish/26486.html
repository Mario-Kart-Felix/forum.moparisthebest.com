program LearnBasicsOfFinnish;
//////////////////////
//Made by Jei jei KK /
//////////////////////
////////////////////////////////////////////////////////////////
// IF YOU WANT ANY WORDS TO BE TRANSLATED,//////////////////////
// please, make a list. It’s easier translate from there, then,/
// PM me on moparisthebest or sythe.org.////////////////////////
////////////////////////////////////////////////////////////////
//Current version is 0.1 //////
//Only present here, yet…//////
////////////////////////////////
{Fun Fact;
This script toke me time to make.}

/////////////////////////////////////////////////////////////////////////
//Don’t try to say these words yet, you don’t say them like in english.//
//Example. If I say word “Here” with english, and type it ‘finnish-like’/
//You would end writing “Hiö” or “Hiä”///////////////////////////////////
/////////////////////////////////////////////////////////////////////////
const

// What = 0 // Disabled, Until, I make a list of words i’m gonna translate.
WaitT = 3000; // Milliseconds to wait between each word.
Mwait = 7000; // Other Wait. This should be bigger than WaitT
//^^^^ Does not apply every word.
Procedure Basics;
begin
ClearDebug;
wait(WaitT);
Writeln(‘Firstly, basics.’)
Writeln(’“Minä” Means “Me, or I”’)
wait(waitT);
Writeln(’“Sinä” Means “You”(only one person)’)
Wait(waitT);
Writeln(’“Hän” Means “She, or He”, In finnish, you dont know, if person is female’)
Writeln(’ or male, in this case.’)
Wait(waitT);
Writeln(‘Ok, there was; I, You, She/He.’)
Writeln(‘Now, Multiple persons.’)
Writeln(‘Be sure to read everything, clearing debug.’)
Wait(Mwait);
ClearDebug;
Wait(waitT);
Writeln(’“Me” Means “We”, Simple, Huh?’)
Wait(waitT);
Writeln(’“Te” Means “You”(Multiple persons!)’)
Wait(waitT);
Writeln(’“He” Means “Them” or “They”’)
Wait(WaitT);
Writeln(‘Now you know “I”, “You,” “She/he”, “We”, “You”, “Them”’)
end;

Procedure Present;
begin
ClearDebug;
Writeln(‘What “I” do.’)
Writeln(‘Correct me if i spelt wrong the “present”.’)
Writeln(‘It should mean, thing what (I, or others) do RIGHT NOW.’)
Wait(WaitT);
Writeln(‘In finnish, you add things to end of the verb, depends on who does, and when.’)
Writeln(‘Be sure to read, Clearing debug.’)
Wait(Mwait);
ClearDebug;
Wait(WaitT);
Writeln(‘Example.’)
Wait(WaitT);
Writeln(‘minä kirjoitaN’)
Writeln(‘See the N? normally, passive, or whatever would be’)
Wait(waitT);
Writeln(‘KirjoitTaA’)
Writeln(‘In finnish, you strech(Whatever) the words.’)
Writeln(‘Kirjoittaa= Write.’)
Writeln(’(Minä)Kirjoitan= (I) Write, or (I Am Writing)’)
Wait(WaitT);

Wait(Mwait);

ClearDebug;
Writeln(‘Now, What “You” (one person) do’)
Wait(waitT);
Writeln(‘Now you know about streching words, so im not gonna explain again.’)
Writeln(‘Example.’)
Wait(waitT);
Writeln(‘sinä syöt’)
Writeln(‘Read upper part of streching.’)
Wait(WaitT);
Writeln(’(Sinä)Syöt= (You)Eat, or (You)(are) Eating’)
Writeln(‘Passive for “Syöt” would be “Syödä”.’)
Wait(WaitT);

Wait(Mwait);

ClearDebug;
Writeln(‘Now, What “She/He” does.’)
Wait(waitT);
Writeln(‘Streching words, you know that.’)
Writeln(‘Example of this.’)
Writeln(‘Hän lukee’)
Wait(WaitT);
Writeln(’(hän)lukee= (He/She)Reads, or (He/She) Is Reading’)
Writeln(‘Passive would be “lukea”’)
Wait(WaitT);

Wait(Mwait);
ClearDebug;
Writeln(‘Now, what “We” Do.’)
Wait(WaitT);
Writeln(‘Example.’)
Writeln(‘Me katsomme’)
Wait(WaitT);
Writeln(’(me)Katsomme =(we)(are)Watching’)
Writeln(‘Passive for “Katsomme” would be’)
Wait(WaitT);
Writeln(’“Katsotaan”’)
wait(WaitT);

Wait(Mwait);
ClearDebug;
Writeln(‘Now, what “You” Do(Multiple persons)’)
Wait(WaitT);
Writeln(‘Again, example…’)
Writeln(‘Te piirrätte’)
Wait(WaitT);
Writeln(’(te)Piirrätte =(You)(are)Drawing.’)
Writeln(‘Passive for “Piirrätte” Would be’)
Wait(WaitT);
Writeln(‘Piirretään’)
Wait(WaitT);

Wait(Mwait)
ClearDebug;
Writeln(‘Now, What “They” do.’)
Wait(WaitT);
Writeln(‘Example.’)
Writeln(‘He pelaavat.’)
Wait(WaitT);
Writeln(’(he)Pelaavat =(they)(Are)Playing’)
Writeln(‘Passive for “Pelaavat” would be’)
Wait(WaitT);
Writeln(‘Pelataan’)
Wait(WaitT);
Writeln(‘Ok, Now you know the basics.’)
Writeln(‘You understand finnish now good as…’)
Wait(WaitT);
Writeln(’…3-4 year old baby, maybe less.’)
end;

begin
ClearDebug;
Basics;
Present;

end.
[/scar]
