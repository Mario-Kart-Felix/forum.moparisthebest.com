public class RainbowText {
	
	/**
	 * This will keep a track of the next colour to add
	 */
	private static byte colorIndex = 0;
	
	/**
	 * A list of the all the possible colours
	 */
	private final static String[] COLOURS = { //Rainbow Colour scheme
		"#FF2B21","#FF7C21","#FFDD56",
		"#B4FF00","#00FF48","#00C2EF", "#0050EF",
		"#6F09EF", "#E909FF"
	};

	/**
	 * This will find the next colour to add to
	 * the character
	 * @return 
	 * 			- the colour string to add
	 */
	private static String applyColour(){
		colorIndex++;
		if(colorIndex >= COLOURS.length)
			colorIndex = 0;
		return COLOURS[colorIndex];
	}
	
	/**
	 * This will check the console for inputs and
	 * add the next colour in line to the characters
	 * then it will copy the text to your clipboard
	 * @param args
	 */

		String msg = "";
		Scanner s = new Scanner(System.in);
		while(s.hasNext()){
			System.out.print("[B][SIZE=32pt][FONT=Courier New]");
			msg += "[B][SIZE=4][FONT=Courier New]";
			String letter = s.next();
				System.out.print(" ");

	}

public final class RainbowText {

// Preset named colors for other functionality
private static final Set<NamedColor> COLORS = initColors();

/**
 * {@link ThreadLocalRandom} to prevent unnecessary wait on
 * {@link AtomicLong}
 */
private static final Random RANDOM = ThreadLocalRandom.current();

private static Set<NamedColor> initColors() {
	Set<NamedColor> colors = new HashSet<>();
	colors.add(new NamedColor("Black", 0x00, 0x00, 0x00));
	// Other colors...
	return Collections.unmodifiableSet(colors);
}

private static Color getRandomColor() {
	return new Color(RANDOM.nextFloat(), RANDOM.nextFloat(), RANDOM.nextFloat());
}

public static void main(String[] args) {
	StringBuilder bldr = new StringBuilder();

	bldr.append("[B][SIZE=4][FONT=Courier New] ");

	try (Scanner scanner = new Scanner(System.in)) {
		String string = scanner.nextLine();

		for (int index = 0; index < string.length(); index++) {
			Color color = getRandomColor();
			String hex = String.format("#%02x%02x%02x", color.getRed(), color.getGreen(), color.getBlue());
			bldr.append("" + string.substring(index, index + 1) + "");
		}

		bldr.append("[/FONT][/SIZE][/B]");
	}

	StringSelection selection = new StringSelection(bldr.toString());
	Clipboard clipboard = Toolkit.getDefaultToolkit().getSystemClipboard();
	clipboard.setContents(selection, selection);
}

private static final class NamedColor {
	private final String name;
	private final int red;
	private final int green;
	private final int blue;

	public NamedColor(String name, int red, int green, int blue) {
		this.name = name;
		this.red = red;
		this.green = green;
		this.blue = blue;
	}

	@Override
	public int hashCode() {
		int hashCode = name.hashCode();
		hashCode = 31 * hashCode + red >> 16;
		hashCode = 31 * hashCode + green >> 8;
		hashCode = 31 * hashCode + blue;
		return hashCode;
	}

	@Override
	public boolean equals(Object obj) {
		if (this == obj) {
			return true;
		}
		if (!(obj instanceof Color)) {
			return false;
		}
		NamedColor other = (NamedColor) obj;
		return other.red == red && other.green == other.green && other.blue == blue && other.name.equals(name);
	}
}

private RainbowText() {

}
