public int barsNeeded(int slot, int column) {
 if (column == 1119) {
  if (slot == 0 || slot == 1) return 1;
  if (slot == 2 || slot == 3) return 2;
  if (slot == 4) return 3;
 }
 if (column == 1120) {
  if (slot == 0 || slot == 1) return 1;
  if (slot == 2 || slot == 3) return 3;
  if (slot == 4) return 2;

  if (slot == 0 || slot == 1 || slot == 2) return 3;
  if (slot == 3) return 5;

  if (slot == 0 || slot == 4) return 1;

 return 0;
}
