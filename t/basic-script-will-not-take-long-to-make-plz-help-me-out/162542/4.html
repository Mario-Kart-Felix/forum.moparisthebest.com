The_Un-Named July 30, 2016, 5:25pm #2

const
Key = ‘3’;
Minutes = 10;

begin
repeat
KeyDown(GetKeyCode(Key));
Wait(Random(25));
KeyUp(GetKeyCode(Key));
Wait(Minutes * 60 * 60);
until False;
end.
