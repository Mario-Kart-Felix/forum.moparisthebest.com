public void moveSlow(int x2, int y2)
{
    int dx = x2-xpos;

    if(Math.abs(dx) > 20) {
        normal(dx, dy, x2, y2, 2000, 50);

        x2 += 4;

private void normal(int dx, int dy, int nx, int ny, int random, int chops)
{
    double upto = Math.abs(dx);
    double inc = upto / chops;
    double ntd = Math.PI / 2;
    boolean xp = dx > 0 ? true : false;
    boolean yp = dy > 0 ? true : false;
    int neg = dx > 0 ? 1 : -1;
    for(double i = 0; i < upto; i += inc + rand.nextInt(random) / 1000.)
    {
        int newX = 0, newY = 0;
        
        if(xp){
            newX = (int)(xpos + i);
        }else newX = (int)(xpos - i);
        
        if(yp && xp)
        {
            newY = (int)( ypos + dy * Math.sin(ntd * (i/upto)) );
        }else if(!yp && xp)
        {
            newY = (int)( ypos + dy * Math.sin(ntd * (i/upto)) );
        }else if(yp && !xp)
        {
            newY = (int)( ypos + dy * Math.sin(ntd * (i/upto)) );
        }else {
            newY = (int)( ypos + dy * Math.sin(ntd * (i/upto)) );
        }
        move(newX, newY, false);
        sleep(rand.nextInt(10) + 5);
    }
    xpos = nx;
    ypos = ny;
}

it is pretty simple, and follows a sin pattern from either 0 to pi/2 or 0 to -pi/2
