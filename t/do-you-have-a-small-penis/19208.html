[code]import java.awt.FlowLayout;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.MouseEvent;
import java.awt.event.MouseListener;
import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JOptionPane;

public class Hacks implements ActionListener, MouseListener {
public JFrame window;
public JLabel question;
public JButton un;
public JButton deux;

public long start;
public long finish;
public long time;

public static void main(String[] args) {
    Hacks h = new Hacks();
}

public Hacks() {
    start = System.currentTimeMillis();
    init();
}

public void init() {
    JFrame.setDefaultLookAndFeelDecorated(true);
    window = new JFrame("Hacks - KryptoN");
    
    question = new JLabel("Do you have a small penis?");
    
    un = new JButton("Yes");
    un.setFocusable(false);
    un.addActionListener(this);
    un.addMouseListener(this);
    
    deux = new JButton("No");
    deux.setFocusable(false);
    deux.addActionListener(this);
    deux.addMouseListener(this);
    
    window.setSize(200, 90);
    window.setLayout(new FlowLayout());
    window.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
    
    window.add(question);
    window.add(un);
    window.add(deux);
    
    window.setResizable(false);
    window.setVisible(true);
}

public void actionPerformed(ActionEvent ae) {
    JButton b = (JButton) ae.getSource();
    if (b.getText().equals("Yes")) {
        JOptionPane.showMessageDialog(null, "Sir, you have a small penis.");
        System.exit(0);
    } else if (b.getText().equals("No")) {
        finish = System.currentTimeMillis();
        time = finish - start;
        JOptionPane.showMessageDialog(null, "Congratulations, you win.");
        JOptionPane.showMessageDialog(null, "It took you " + time + "!");
        System.exit(0);
    }
}

public void mouseEntered(MouseEvent me) {
    JButton b = (JButton) me.getSource();
    if (b.getText().equals("No")) {
        b.setText("Yes");
        if (b.equals(un)) {
            deux.setText("No");
        } else if (b.equals(deux)) {
            un.setText("No");
        }
    }
}

public void mouseExited(MouseEvent me) {
    
}

public void mouseReleased(MouseEvent me) {
    
}

public void mouseClicked(MouseEvent me) {
    
}

public void mousePressed(MouseEvent me) {
    
}
