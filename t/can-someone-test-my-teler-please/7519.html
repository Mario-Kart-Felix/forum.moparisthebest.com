procedure DoTeleing;
var bmp, x, y, n, i: Integer;
Begin
  case Lowercase(teleto) of
    'camelot': bmp:= Cammy;
    'falador': bmp:= Fally;
    'varrock': bmp:= Varrock;
    'lumbridge': bmp:= Lumbridge;
  end;
  GameTab(7);
  repeat
    if (Findbitmap(bmp, x, y)) then
    begin
      MMouse(x, y, 0, 0);
      Wait(32 + Random(10));
      HoldMouse(x, y, True);
      for i:= 1 to 4 do
      begin
        Wait(5 + Random(5));
      end;
      ReleaseMouse(x, y, True);
      Timetele:= Timetele + 1;
      Wait((WaitSpeed/3) + Random(WaitSpeed/2));
    end
    else
    begin
      n:= n + 1;
      if (n >= 3) then GameTab(7);
      if (n >= 6) then
      begin
        Writeln('Lost Tele Buttons?');
        TerminateScript;
      end;
    end;
  until(Timetele >= Teletimes);
end;
