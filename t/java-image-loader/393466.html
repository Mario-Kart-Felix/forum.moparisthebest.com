Panel:

[code=java]//imports
import javax.swing.;
import java.awt.;

//Panel class
public class Panel extends JPanel {

Image img;
public static String imgPath;

public Panel() {

    ImageIcon i = new ImageIcon(imgPath);
    img = i.getImage();

}

public void paint(Graphics g) {

    super.paint(g);
    Graphics2D g2d = (Graphics2D) g;

    g2d.drawImage(img, 0, 0, null);

}

public int getImgX() {

    int x = img.getWidth(this);

    return x;

}

public int getImgY() {

    int y = img.getHeight(this);

    return y;

}

}[/code]
