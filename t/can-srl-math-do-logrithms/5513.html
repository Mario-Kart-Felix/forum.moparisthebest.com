function loge(i:extended) : extended;
var
count,exp:integer;
xval:extended;
begin
exp:= 1;
xval:= (i - 1) / (i + 1);
for count:= 1 to 5 do
begin
result:= result + pow(xval,exp)/exp;
exp:= exp + 2;
end;
result:= result * 2;
end;

function logb(base,i:extended) : extended;
begin
result:= loge(i)/loge(base);
end;

function arctan(e : extended) : extended;
var
count : integer;
res : extended;
inv : boolean;
begin
if(abs(e) > 1)then
begin
e := pow(e,-1);
inv := true;
end else
begin
inv := false;
end;
for count := 0 to 6 do
begin
res := res + ( ( pow(2,2 * count) * pow(fract(count),2) ) / fract(2 * count + 1) ) * ( ( pow(e,2*count+1) ) / ( pow(1 + pow(e,2),count + 1) ) );
end;
if(inv)then
begin
res := 1.5707963267 - res;
end;
result := res;
end;
