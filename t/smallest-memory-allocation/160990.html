I wrote you a working example

[code]/*
remove the “unnecessary” local variables to optimize the code.
I kept them in to keep it simple.
*/

public final class BitsetGrid {

private final int width;

private final int height;

private final int[] grid;

public BitsetGrid(int width, int height) {
this.width = width;
this.height = height;
int tmp = width * height;
int n = tmp / 32; //optimize performance with tmp >>> 5
int k = tmp % 32; //optimize performance with tmp & 0x1f
if(k > 0)
n++;
grid = new int[n];
}

public final int getWidth() {
return width;
}

public final int getHeight() {
return height;
}

public final boolean get(int idx) {
int n = idx / 32; //optimize performance with idx >>> 5
int k = idx % 32; //optimize performance with idx & 0x1f
return (grid[n] & (1 << k)) != 0;
}

public final void set(int idx, boolean flag) {
int n = idx / 32; //optimize performance with idx >>> 5
int k = idx % 32; //optimize performance with idx & 0x1f
if(flag)
grid[n] |= 1 << k;
else
grid[n] &= ~(1 << k);
}

public final boolean get(int x, int y) {
return get(y * width + x);
}

public final void set(int x, int y, boolean flag) {
set(y * width + x, flag);
}

}[/code]
