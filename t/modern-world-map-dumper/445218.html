When I mean random, a random color is selected for a floor.

        FLOOR_COLORS = new int[256];
        for(int i = 0; i < FLOOR_COLORS.length; i++) {
            int colorSeed = (int) (Math.random() * 10);
            int color = 0;
            if(colorSeed < 2)
                color = 0x0000FF;
            else if(colorSeed < 4)
                color = 0x00FF00;
            else if(colorSeed < 6)
                color = 0xFF0000;
            else if(colorSeed < 8)
                color = 0xFFFF00;
            else
                color = 0x00FFFF;
            FLOOR_COLORS[i] = color + (int) (Math.random() * 256);
        }
