package server.model.players.content;

import java.util.ArrayList;
import java.util.List;

import server.model.players.Client;

/**
 * 
 * 
 * @author Rodrigo(Imsuperman05)
 */
public class CashBag {

	static List<Integer> bag = new ArrayList<Integer>();
	
	public static List<Integer> getBag() {
		return bag;
	}
	
	public static void addToBag(Client c, int amount) {
		bag.add(amount);
		c.sendMessage("You add "+amount+" coins to your bag.");
	}
	
	public static void take(Client c, int amount) {
		bag.remove(amount);
		c.sendMessage("You take out "+amount+" coins from your bag.");
	}
	
	public static void clearbag(Client c) {
		bag.clear();
		c.sendMessage("You clear your bag.");
	}
}
