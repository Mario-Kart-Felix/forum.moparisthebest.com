package org.ventus.game.model.players.packets.commands;

import java.util.ArrayList;
import java.util.List;

import org.ventus.game.model.players.Client;
import org.ventus.game.model.players.PacketType;
import org.ventus.game.model.players.packets.commands.impl.AdministratorCommands;
import org.ventus.game.model.players.packets.commands.impl.DefaultCommands;
import org.ventus.game.model.players.packets.commands.impl.DeveloperCommands;
import org.ventus.game.model.players.packets.commands.impl.DonatorCommands;
import org.ventus.game.model.players.packets.commands.impl.ModeratorCommands;

/**
 * The command packet.
 * @author Play Boy & Frostbit3
 *
 */
public class CommandPacket implements PacketType {
	
	private List<CommandParent> commands = new ArrayList<CommandParent>();
	
	public CommandPacket() {
		commands.add(0, new DeveloperCommands());
		commands.add(1, new AdministratorCommands());
		commands.add(2, new ModeratorCommands());
		commands.add(3, new DonatorCommands());
		commands.add(4, new DefaultCommands());
	}

    @Override
    public void processPacket(Client c, int packetType, int packetSize) {
    	String command = c.getInStream().readString();
    	
    	switch (c.playerRights) {
    	case 3:
    		commands.get(0).handleCommand(c, command);
    	case 2:
    		commands.get(1).handleCommand(c, command);
    	case 1:
    		commands.get(2).handleCommand(c, command);
    	default:
    		commands.get(4).handleCommand(c, command);
    	}
    	
    }
}

package org.ventus.game.model.players.packets.commands.impl;

import org.ventus.game.Config;
import org.ventus.game.Server;
import org.ventus.game.model.players.Client;
import org.ventus.game.model.players.PlayerHandler;
import org.ventus.game.model.players.packets.commands.CommandParent;
import org.ventus.game.utility.Misc;

public class DefaultCommands implements CommandParent {

	@Override
	public void handleCommand(Client c, String cmd) {
        String[] args = cmd.split(" ");
		if (cmd.startsWith("/") && cmd.length() > 1) {
			if (c.clanId > 0) {
				cmd = cmd.substring(1);
				Server.clanChat.playerMessageToClan(c.playerId, cmd, c.clanId);
			} else {
				if (c.clanId != -1) {
					c.clanId = -1;
					c.sendMessage("You are not in a clan.");
				}
			}
		}
        
        if (cmd.equalsIgnoreCase("players")) {
            c.sendMessage("There are currently @blu@" + PlayerHandler.getPlayerCount() + "@bla@ players online!");
        }
        
        if (cmd.startsWith("changepassword") && cmd.length() > 15) {
            c.playerPass = cmd.substring(15);
            c.sendMessage("Success your new password is now: @red@" + c.playerPass);
        }        
        
        if (cmd.startsWith("yell") && c.playerRights >= 1) {
            String rank = "";
            String Message = cmd.substring(4).toLowerCase();
            if (c.playerRights >= 1) {
                rank = "@blu@[ Mod ] " + Misc.optimizeText(c.playerName) + ": ";
            }
            if (c.playerRights >= 2) {
                rank = "@lre@[ Admin ] " + Misc.optimizeText(c.playerName) + ": ";
            }
            if (c.playerRights >= 3) {
                rank = "@red@[ Developer ] " + Misc.optimizeText(c.playerName) + ": ";
            }
            if (c.playerRights >= 4) {
                rank = "@mon@[ Donator ] [" + Misc.optimizeText(c.playerName) + "]: ";
            }
            for (int j = 0; j < PlayerHandler.players.length; j++) {
                if (PlayerHandler.players[j] != null) {
                    Client c2 = (Client) PlayerHandler.players[j];
                    c2.sendMessage(rank + Misc.optimizeText(Message));
                }
            }
        }				
		
	}

}

package org.ventus.game.model.players.packets.commands.impl;

import org.ventus.game.Config;
import org.ventus.game.Connection;
import org.ventus.game.model.players.Client;
import org.ventus.game.model.players.PlayerHandler;
import org.ventus.game.model.players.packets.commands.CommandParent;

public class ModeratorCommands implements CommandParent {

	@Override
	public void handleCommand(Client c, String cmd) {
		
	       if (cmd.equals("ban") && cmd.charAt(3) == ' ') {
	            try {
	                String playerToBan = cmd.substring(4);
	                Connection.addNameToBanList(playerToBan);
	                Connection.addNameToFile(playerToBan);
	                for (int i = 0; i < Config.MAX_PLAYERS; i++) {
	                    if (PlayerHandler.players[i] != null) {
	                        if (PlayerHandler.players[i].playerName.equalsIgnoreCase(playerToBan)) {
	                            PlayerHandler.players[i].disconnected = true;
	                            Client c2 = (Client) PlayerHandler.players[i];
	                            c2.sendMessage(" " + c2.playerName + " Got Banned By " + c.playerName + ".");
	                        }
	                    }
	                }
	            } catch (Exception e) {
	                c.sendMessage("Player Must Be Offline.");
	            }
	        }
	        
	        if (cmd.equals("kick") && cmd.charAt(4) == ' ') {
	            try {
	                String playerToBan = cmd.substring(5);
	                for (int i = 0; i < Config.MAX_PLAYERS; i++) {
	                    if (PlayerHandler.players[i] != null) {
	                        if (PlayerHandler.players[i].playerName.equalsIgnoreCase(playerToBan)) {
	                            PlayerHandler.players[i].disconnected = true;
	                        }
	                    }
	                }
	            } catch (Exception e) {
	                c.sendMessage("Player Must Be Offline.");
	            }
	        }

	        if (cmd.equals("unban")) {
	            try {
	                String playerToBan = cmd.substring(6);
	                Connection.removeNameFromBanList(playerToBan);
	                c.sendMessage(playerToBan + " has been unbanned.");
	            } catch (Exception e) {
	                c.sendMessage("Player Must Be Offline.");
	            }
	        }		
	}

}
