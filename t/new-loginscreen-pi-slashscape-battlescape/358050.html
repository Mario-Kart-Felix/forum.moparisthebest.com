	private void processLoginScreenInput()
	{
		if(loginScreenState == 2)
		{
			int i = super.myWidth / 2 - 80;
			int l = super.myHeight / 2 + 20;
			l += 20;
			if(super.clickMode3 == 1 && super.saveClickX >= i - 75 && super.saveClickX <= i + 75 && super.saveClickY >= l - 20 && super.saveClickY <= l + 20)
			{
				loginMessage1 = "";
				loginMessage2 = "Enter your username & password.";
				loginScreenState = 2;
				loginScreenCursorPos = 0;
			}
			i = super.myWidth / 2 + 80;
			if(super.clickMode3 == 1 && super.saveClickX >= i - 75 && super.saveClickX <= i + 75 && super.saveClickY >= l - 20 && super.saveClickY <= l + 20)
			{
				loginMessage1 = "";
				loginMessage2 = "Enter your username & password.";
				loginScreenState = 2;
				loginScreenCursorPos = 0;
			}
		} else
		{
			if(loginScreenState == 0)
			{
				int j = super.myHeight / 2 - 40;
				j += 30;
				j += 25;
				if(super.clickMode3 == 1 && super.saveClickY >= j - 15 && super.saveClickY < j)
					loginScreenCursorPos = 0;
				j += 15;
				if(super.clickMode3 == 1 && super.saveClickY >= j - 15 && super.saveClickY < j)
					loginScreenCursorPos = 1;
				j += 15;
				int i1 = super.myWidth / 2 - 0;
				int k1 = super.myHeight / 2 + 30;
				k1 += 0;
				if(super.clickMode3 == 1 && super.saveClickX >= i1 - 200 && super.saveClickX <= i1 + 200 && super.saveClickY >= k1 - 25 && super.saveClickY <= k1 + 25)
				{
					loginFailures = 0;
					login(myUsername, myPassword, false);
					if(loggedIn)
						return;
				}
				k1 += 0;
				if(super.clickMode3 == 200 && super.saveClickX >= i1 - 200 && super.saveClickX <= i1 + 200 && super.saveClickY >= k1 - 50 && super.saveClickY <= k1 + 50)
				{
					launchURL("http://google.com");
				}
				do
				{
					int l1 = readChar(-796);
					if(l1 == -1)
						break;
					boolean flag1 = false;
					for(int i2 = 0; i2 < validUserPassChars.length(); i2++)
					{
						if(l1 != validUserPassChars.charAt(i2))
							continue;
						flag1 = true;
						break;
					}

					if(loginScreenCursorPos == 0)
					{
						if(l1 == 8 && myUsername.length() > 0)
							myUsername = myUsername.substring(0, myUsername.length() - 1);
						if(l1 == 9 || l1 == 10 || l1 == 13)
							loginScreenCursorPos = 1;
						if(flag1)
							myUsername += (char)l1;
						if(myUsername.length() > 12)
							myUsername = myUsername.substring(0, 12);
					} else
					if(loginScreenCursorPos == 1)
					{
						if(l1 == 8 && myPassword.length() > 0)
							myPassword = myPassword.substring(0, myPassword.length() - 1);
						if (l1 == 9 || l1 == 10 || l1 == 13) {
							if (myPassword.length() > 0) {
								loginFailures = 0;
								login(myUsername, myPassword, false);
							} else {
								loginScreenCursorPos = 0;
							}
						}
						if(flag1)
							myPassword += (char)l1;
						if(myPassword.length() > 20)
							myPassword = myPassword.substring(0, 20);
					}
				} while(true);
				return;
			}
			if(loginScreenState == 3)
			{
				int k = super.myWidth / 2;
				int j1 = super.myHeight / 2 + 50;
				j1 += 20;
				if(super.clickMode3 == 1 && super.saveClickX >= k - 75 && super.saveClickX <= k + 75 && super.saveClickY >= j1 - 20 && super.saveClickY <= j1 + 20)
					loginScreenState = 0;
			}
		}
	}
