import java.util.Scanner;

public class NineQuestions {
	public static String[] QUESTIONS = {"Full name:", "age:", "location:", "come from where?", "hometown:", "job:",
				      "hobbies:", "first language:", "second language:"};

	public static void main(String[] args) {
		String[] information = new String[9];
		Scanner s = new Scanner(System.in);
		for (int i = 0; i < QUESTIONS.length; i++) {
			System.out.println(QUESTIONS[i]);
			information[i] = s.nextLine();
		}
		System.out.println("----------------");
		System.out.println("Output:");
		for (int i = 0; i < information.length; i++) {
			System.out.println(information[i]);
		}
	}
}
