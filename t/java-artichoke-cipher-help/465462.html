We are using Thomas Jefferson's Artichoke CIPHER systemin the first programming assignment. It was notactually used in the Lewis and Clark expedition.

As you can see, where your instructor put things on different lines, they really shouldn’t be there at all. Your algorithm works perfectly, just his input and output were wrong.

this seems to encode and decode just fine based on the details you provided, I didn’t bother looking up the actual cipher you might need to.

public class Artichoke {

    private final String keyword;
    private static final String VALID_CHARACTERS = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz ,.!?\'\"";

    public Artichoke(String keyword) {
        this.keyword = keyword;

    }

    public static void main(String[] arg) throws IOException {

        String keyword = "artichoke";
        Artichoke a = new Artichoke(keyword);
        String test = "We are using Thomas Jefferson's Artichoke CIPHER system";
        test = a.encode(test);
        System.out.println("encoded: " + test);
        test = a.decode(test);
        System.out.println("d: " + test);
    }

    public String encode(String s) {
        String t = "";
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            char shift = keyword.charAt(i % keyword.length());

            int shiftAmount = VALID_CHARACTERS.indexOf(shift);

            int index = VALID_CHARACTERS.indexOf(ch);
            int shifted = (index + shiftAmount) % VALID_CHARACTERS.length();
            char newChar = VALID_CHARACTERS.charAt(shifted);
            t += newChar;
        }
        return t;
    }

    public String decode(String s) {
        String t = "";
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            char shift = keyword.charAt(i % keyword.length());
            int shiftAmount = VALID_CHARACTERS.indexOf(shift);
            int index = VALID_CHARACTERS.indexOf(ch);
            int x= (index - shiftAmount);
            if(x < 0){
               x = VALID_CHARACTERS.length()+x;
            }
            char newChar = VALID_CHARACTERS.charAt(x);
            t += newChar;
        }
        return t;
    }
}
