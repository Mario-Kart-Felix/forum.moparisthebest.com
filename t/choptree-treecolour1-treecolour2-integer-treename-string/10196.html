[scar]Program ChopTree;

{.include SRL\SRL.scar}

Procedure ChopTree(TreeColour1, TreeColour2: integer; TreeName: String);
begin
if FindColor(x, y, TreeColour1, 5, 5, 600, 600) then
begin
if FindColor(x, y, TreeColour2, x - 25, y - 25, x + 25, y + 25) then
begin
MMouse(x, y, 5, 5);
GetMousePos(x, y);
Wait(400);
if IsUpText('Chop down ’ + TreeName) then
begin
Mouse(x, y, 0, 0, True);
Wait(1000);
end;
end;
end;
end;

begin
ChopTree(123456, 123456, ‘Magic’);
end.[/scar]

Searches For one colour, Searches for the other in a 50x50 Box, Checks if ‘Chop down’ + TreeName is found then it clicks the tree.
