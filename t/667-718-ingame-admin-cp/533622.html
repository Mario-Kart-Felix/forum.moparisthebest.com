Create a new Dialogue called “Admincp” add this in it:

[code]
package com.rs.game.player.dialogues;

import com.rs.game.WorldTile;
import com.rs.game.player.content.Magic;
import com.rs.game.player.Skills;

public class Admincp extends Dialogue {

@Override
public void start() {
	stage = 1;
	if (stage == 1) {
		sendOptionsDialogue("Select an Option",
				"Slayer teleports",
				"Admin panel");
		stage = 1;
	}
}

public void run(int interfaceId, int componentId) {
	if (stage == 1) {
		int option;
		if (componentId == OPTION_1) {
       				player.getDialogueManager().startDialogue("SlayerTeleports", 230);
		}
		if (componentId == OPTION_2) {
       				player.getDialogueManager().startDialogue("Admincp", 230);
		}
	}
}

@Override
public void finish() {

}

[code]
package com.rs.game.player.dialogues;

import com.rs.game.WorldTile;
import com.rs.game.ForceTalk;
import com.rs.game.player.content.Magic;
import com.rs.game.player.Skills;

public class Admin extends Dialogue {

@Override
public void start() {
	stage = 1;
	if (stage == 1) {
		sendOptionsDialogue("<col=FF0000>Admin Panel",
				"Hide",
				"Staff Island",
				"Bot Island",
				"Bot Court");
		stage = 1;
	}
}

public void run(int interfaceId, int componentId) {
	if (stage == 1) {
		int option;
		if (componentId == OPTION_1) {
			player.getAppearence().switchHidden();
			player.getPackets().sendGameMessage("<img=6>Only Administrators can see you <col=FFF000>  " + player.getAppearence().isHidden());
			player.getInterfaceManager().closeChatBoxInterface();
		}
		if (componentId == OPTION_2) {
			Magic.sendNormalTeleportSpell(player, 0, 0, new WorldTile(3122,
					3114, 0));
			end();
		}
		if (componentId == OPTION_3) {
			Magic.sendNormalTeleportSpell(player, 0, 0, new WorldTile(3673,
					3616, 0));
			end();
		}
		if (componentId == OPTION_4) {
			player.setNextWorldTile(new WorldTile(3680, 3623, 0));
			player.getInterfaceManager().closeChatBoxInterface();
		}

	}
}

@Override
public void finish() {

}

}[/code]
