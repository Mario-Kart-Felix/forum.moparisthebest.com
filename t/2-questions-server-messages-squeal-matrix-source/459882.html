	EventManager.getSingleton().addEvent(new Event() {
		public void execute(EventContainer c) {
			if(Config.LAST_MESSAGE == 0) {
				sendMessage("[@blu@Server@bla@] Remember to ::vote @bla@for the server every day to help us grow.");
				Config.LAST_MESSAGE = 1;
			} else if(Config.LAST_MESSAGE == 1) {
				sendMessage("[@blu@Server@bla@] Donations are automated!");
				Config.LAST_MESSAGE = 2;
			}
	}, 300000);//the yell timer - 5 minutes
}[/code]
