Sigh, try this:

import javax.swing.*;
import java.awt.event.*;
import java.awt.*;

public class HelloFrame extends JFrame implements ActionListener
{
	public static void main(String[] args)
	{
		new HelloFrame();
	}

	public HelloFrame()
	{
		this.setSize(400, 300);
		this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		this.setTitle("Games!");

		JButton button1 = new JButton("Webkinz");
		JButton button2 = new JButton("Whyville");
		JButton button3 = new JButton("Zwinky");
		JButton button4 = new JButton("MillsBerry");
		JButton button5 = new JButton("ClubPenguin");
		button1.addActionListener(this);
		button2.addActionListener(this);
		button3.addActionListener(this);
		button4.addActionListener(this);
		button5.addActionListener(this);

		JLabel label1 = new JLabel("Choose a game you would like to go to. Then you will go there.");
		JPanel panel1 = new JPanel();
		panel1.setLayout(new FlowLayout(FlowLayout.CENTER));
		panel1.add(label1);
		this.add(panel1);
		this.add(button1);
		this.add(button2);
		this.add(button3);
		this.add(button4);
		this.add(button5);
		this.setLayout(null);
		button2.setBounds(5, 5, 150, 50);
		button1.setBounds((getWidth() - 100) / 2, (getHeight() - 30) / 2, 100, 30);
		setLayout(null);
		setLayout(new FlowLayout());
		add(button1, FlowLayout.CENTER);
		this.setVisible(true);
	}

	public void actionPerformed(ActionEvent e)
	{
		if (e.getSource().equals(button1))
		{
			try
			{
				Runtime.getRuntime().exec("rundll32 url.dll,FileProtocolHandler http://www.webkinz.com");
			}
			catch (Exception e) { }
		}
		else if (e.getSource().equals(button2))
			System.out.println("button 2 pressed");
		else if (e.getSource().equals(button3))
			System.out.println("button 3 pressed");
		else if (e.getSource().equals(button4))
			System.out.println("button 4 pressed");
		else if (e.getSource().equals(button5))
			System.out.println("button 5 pressed");
	}
}

Please read the edits I made and try to remember them.
