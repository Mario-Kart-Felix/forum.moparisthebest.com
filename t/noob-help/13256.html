Program AutoClicker;

Const
  Times = 19;

Var
  Said : Integer;

Procedure Watch;
Begin
  MoveMouse(391,129);
  ClickMouse(391,129, True);
  Wait(90)
  MoveMouse(290,233);
  ClickMouse(290,233, True);
  Wait(1000)
  MoveMouse(505,282);
  ClickMouse(505,282, True);
  Wait(90)
  Said := Said + 1;
End;

Procedure Watch2;
Begin
  MoveMouse(215,182);
  ClickMouse(215,182, True);
  Wait(90)
  MoveMouse(480,266);
  ClickMouse(480,266, True);
  Wait(90)
End;

Begin
  ClearDebug;
  Repeat
    Repeat
      Watch;
    Until(Said = Times)
    Watch2;
  Until(False)
End.

[scar]Program AutoClicker;

Const
Times = 19;

Var
Said : Integer;

Procedure Watch;
Begin
repeat
MoveMouse(391,129);
ClickMouse(391,129, True);
Wait(90)
MoveMouse(290,233);
ClickMouse(290,233, True);
Wait(1000)
MoveMouse(505,282);
ClickMouse(505,282, True);
Wait(90)
Said := Said + 1;
until(said = times)
Said := Said - 19;
End;

Procedure Watch2;
Begin
MoveMouse(215,182);
ClickMouse(215,182, True);
Wait(90)
MoveMouse(480,266);
ClickMouse(480,266, True);
Wait(90)
End;

Begin
ClearDebug;
Repeat
Watch;
Watch2;
Until(False)
End.[/scar]
