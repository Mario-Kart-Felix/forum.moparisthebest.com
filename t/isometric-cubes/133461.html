function Opp(X1, X2: Integer): Integer;
begin
Result := X2 - X1;
end;

function Adj(Y1, Y2: Integer): Integer;
begin
Result := Y2 - Y1;

function Hyp(X1, Y1, X2, Y2: Integer): Integer;
var
T, P: Extended;
begin
T := Opp(X1, X2);
P := Adj(Y1, Y2);
T := Sqr(T);
P := Sqr§;
Result := Round(Sqrt(T + P));
end;

procedure LineFromSlow(Sx, Sy, Ex, Ey: Integer; Angle: Degree; Canvas: TCanvas);
var
T, M, X, Y: Integer;
begin
Canvas.MoveTo(Sx, Sy);
M := Hyp(Sx, Sy, Ex, Ey);
for T := 1 to M do
with Canvas do
begin
Radial(X, Y, Sx, Sy, T, Angle);

end;
