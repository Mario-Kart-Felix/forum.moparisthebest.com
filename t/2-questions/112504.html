I made this up real quick. I tested it and it works. Use it as a reference:

import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
import java.io.*;

public class SwingTest extends JFrame
{
	public SwingTest()
	{
		super("This is a simple Javax Swing test");
		makeGUI();
	}

	private void makeGUI()
	{
		setLocation(250, 250);
		setDefaultCloseOperation(3);
		JPanel main = new JPanel();
		main = (JPanel)this.getContentPane();
		JPanel topPanel = new JPanel();
		topPanel.setLayout(new BorderLayout(0, 0));
		JPanel centerPanel = new JPanel();
		centerPanel.setLayout(new BorderLayout(0, 0));
		final JTextArea jta = new JTextArea(30, 50);
		JMenuBar jmb = new JMenuBar();
		JMenu file = new JMenu("File");
		JMenu test = new JMenu("JMenu Test");
		JMenuItem save = new JMenuItem("Save File");
		save.addActionListener(new ActionListener()
		{
			public void actionPerformed(ActionEvent e)
			{
				try
				{
					JFileChooser chooser = new JFileChooser();
					chooser.showSaveDialog(SwingTest.this);
					File savedFile = chooser.getSelectedFile();
					PrintWriter out = new PrintWriter(new BufferedWriter(new FileWriter(savedFile)));
					out.write(jta.getText());
					out.close();
				}
				catch (IOException ioe) { ioe.printStackTrace(); }
			}
		});
		JMenuItem exit = new JMenuItem("Exit");
		exit.addActionListener(new ActionListener()
		{
			public void actionPerformed(ActionEvent e)
			{
				System.exit(0);
			}
		});
		JMenuItem menuItemTest = new JMenuItem("JMenuItem test");
		menuItemTest.addActionListener(new ActionListener()
		{
			public void actionPerformed(ActionEvent e)
			{
				System.out.println("test");
			}
		});
		file.add(save);
		file.addSeparator();
		file.add(exit);
		test.add(menuItemTest);
		jmb.add(file);
		jmb.add(test);
		topPanel.add(jmb);
		centerPanel.add(new JScrollPane(jta));
		main.add(topPanel, BorderLayout.NORTH);
		main.add(centerPanel, BorderLayout.CENTER);
		pack();
	}

	public static void main(String[] args)
	{
		new SwingTest().setVisible(true);
	}
}

mod_taharok July 30, 2016, 11:09am #12 
         BufferedWriter out = new BufferedWriter("File.txt");
         out.write(textName.getText().getBytes());
         out.close();

Redo that to:

         PrintWriter out = new PrintWriter(new BufferedWriter(new FileWriter(new File("File.txt")))); //write the file
         out.write(textName.getText()); //write the javax.swing.JTextField.getText() string's value into the PrintWriter file
         out.close(); //close the PrintWriter stream

Also, change your IO imports from

import java.io.BufferedWriter;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;

to

import java.io.*;

Finally, you will need to add a try catch statement to the entire block of code in the IndentFrame() constructor because the PrintWriter will through an IOException.
