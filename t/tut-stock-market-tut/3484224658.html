import java.util.Scanner;

public class StockMarket {

/*
 * Main void
 */
public static void main(String args[]) {
	showStockInterface();
	buyStock(1, 10000);
	System.out.println("");
	System.out.println("");
	showStash();
	System.out.println("");
	System.out.println("");
	System.out.println("");
	showStockInterface();
	buyStock(2, 5000);
	System.out.println("");
	System.out.println("");
	showStash();
	System.out.println("");
	System.out.println("");
	System.out.println("");
	showStockInterface();
}

/*
 * Variables we need
 */

public static String stockNames[] = { "Facepalm", "Racism" };
public static String stockOwned = "";
public static double stockCost [] = { 2000, 2000 };
public static int amountOfStocksBought[] = { 0, 0 };
public static int amountOfStockOwned;


/*
 * Builds the stock interface
 */

public static void showStockInterface() {
	//showInterface($$$$);
	System.out.println("Stock Market");
	System.out.println("---------------------");
	System.out.println("1. " +stockNames[0]+ " Cost: " +stockCost[0]);
	System.out.println("2. " +stockNames[1]+ " Cost: " +stockCost[1]);
	System.out.println("---------------------");
	System.out.println("What would you like to buy?");
	
}

/*
 * Where we display how much stock is owned.
 */
public static void showStash() {
	System.out.println("Stocks owned --------");
	System.out.println("---------------------");
	System.out.println("Currently owned stock: " +stockOwned);
	System.out.println("Amount of Facepalm stock owned: " +amountOfStocksBought[0]);
	System.out.println("Amount of Racism stock owned: " +amountOfStocksBought[1]);
	System.out.println("---------------------");
}

/*
 * Where we create buying stocks
 */
public static void buyStock(int stockNumber, int amount) {
	if(stockNumber == 1) {
		amountOfStocksBought[0] = amount;
		//deleteItem(995, stockCost[0]*amount);
		amountOfStockOwned = amount;
		stockOwned = "Facepalm";
		stockCost[0] += stockCost[0] % 1.001;
	} else if(stockNumber == 2) {
		amountOfStocksBought[1] = amount;
		//deleteItem(995, stockCost[1]*amount);
		amountOfStockOwned = amount;
		stockOwned = "Racism";
		stockCost[1] += stockCost[1] % 1.001;
	}
}

/*
 * Where we create selling stocks
 */
public static void sellStock(int stockNumber, int amount) {
	if(stockNumber == 1) {
		amountOfStocksBought[0] -= amount;
		amountOfStockOwned -= amount;
		stockOwned = "None";
	} else if(stockNumber == 2) {
		amountOfStocksBought[1] -= amount;
		amountOfStockOwned -= amount;
		stockOwned = "None";
	}
}

/*
 * A simple shortcut method of using System.out.println
 */
public static void sMB(String s) {
	System.out.println(s);
}

/*
 * A simple shortcut method of using System.out.print
 */
public static void sM(String s) { 
	System.out.print(s);
}
