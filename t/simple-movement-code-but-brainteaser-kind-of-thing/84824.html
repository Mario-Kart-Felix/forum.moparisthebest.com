onClipEvent (load) {
    walkSpeed = 5;
}

onClipEvent (enterFrame) {
    if (Key.isDown(Key.UP)) {
        _y -= walkSpeed;
    }
}

onClipEvent (enterFrame) {
    if (Key.isDown(Key.DOWN)) {
        _y += walkSpeed;
    }
}

onClipEvent (enterFrame) {
    if (Key.isDown(Key.RIGHT)) {
        _x += walkSpeed;
    }
}

onClipEvent (enterFrame) {
    if (Key.isDown(Key.LEFT)) {
        _x -= walkSpeed;
    }
}
