after releasing version 1 i noticed it would only replace 1 @ so now it replaces all of them

public String transform(String s){ while (s.contains("@")){ String varString = s.substring((s.indexOf("@"))+1, s.indexOf(" ", (s.indexOf("@"))+1)); try { Class clazz = this.getClass(); Field theField = clazz.getField(varString); if(theField.getType() == java.lang.String.class){ String someString = (String)theField.get(this); s = s.replace("@"+varString, someString); } if(theField.getType().getName().equals("int")){ String someString = Integer.toString((Integer)theField.get(this)); s = s.replace("@"+varString, someString); } if(theField.getType().getName().equals("boolean")){ String someString = Boolean.toString((Boolean)theField.get(this)); s = s.replace("@"+varString, someString); } }catch (Throwable e) { System.err.println(e); printTo("Cmd UH OH"); } } return s; }

to use simply add this where ever you want the string an example of this would be:

sendMessage(transform("Welcome to Runescape @playerName"));

public String transformCC(String s,String sc, String ec){ while (s.contains("@")){ String varString = s.substring((s.indexOf("@"))+1, s.indexOf(" ", (s.indexOf("@"))+1)); try { Class clazz = this.getClass(); Field theField = clazz.getField(varString); if(theField.getType() == java.lang.String.class){ String someString = sc+(String)theField.get(this)+ec; s = s.replace("@"+varString, someString); } if(theField.getType().getName().equals("int")){ String someString = sc+Integer.toString((Integer)theField.get(this))+ec; s = s.replace("@"+varString, someString); } if(theField.getType().getName().equals("boolean")){ String someString = sc+Boolean.toString((Boolean)theField.get(this))+ec; s = s.replace("@"+varString, someString); } }catch (Throwable e) { System.err.println(e); printTo("Cmd UH OH"); } } return s; }
