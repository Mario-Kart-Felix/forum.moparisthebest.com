This is my ‘router_rules’ script that I whipped up, it is set to run in /etc/network/interfaces when the LAN interface comes up (put ‘up /path/to/router_rules.sh’ under it)

#!/bin/sh
# rules for this server as a router

WAN_IFACE=eth0
LAN_IFACE=br0

ROUTER_IP=192.168.1.1

LOCAL_DNS_IP=192.168.1.3

# format: public_port>lan_ip
# examples
# forward a single port: 8080>192.168.1.55:80
# forward a range: 12000:13000>192.168.1.55
PORT_FORWARDS="8181>192.168.1.1:80 8084>192.168.1.2:80 3890>192.168.1.44:22 8083>192.168.1.4:80"

UDP_PORT_FORWARDS="7>192.168.1.254:7 6060:6065>192.168.1.4 69>192.168.1.4:69 10000:20000>192.168.1.4"
#UDP_PORT_FORWARDS="7>192.168.1.254:7"



# remove 192.* as default gateway
route del default gw $ROUTER_IP

# clears all rules
iptables -X
iptables -F
iptables -Z

#Flushing the tables...
iptables -t filter -F
iptables -t nat -F
iptables -t mangle -F

# Flush existing firewall rules
iptables --flush

# Delete any extraneous chains which may exist from a previous script
iptables --delete-chain

# Change the default policy of all three chains to DROP
#iptables -P INPUT DROP
#iptables -P FORWARD DROP
#iptables -P OUTPUT DROP

# localhost
#iptables -A INPUT -i lo -j ACCEPT
#iptables -A OUTPUT -o lo -j ACCEPT


#Enable NAT (through the WAN interface)
iptables -t nat -A POSTROUTING -o $WAN_IFACE -j MASQUERADE
#then we need to enable routing:
echo "1" > /proc/sys/net/ipv4/ip_forward

# forces all local DNS requests to go through this local server, regardless of request
iptables -t nat -A PREROUTING -p udp -i $LAN_IFACE --dport 53 -j DNAT --to $LOCAL_DNS_IP
iptables -t nat -A PREROUTING -p tcp -i $LAN_IFACE --dport 53 -j DNAT --to $LOCAL_DNS_IP

# enable ping from only certain places, deny from rest
# 66.220.2.74 is HE tunnel (done in upipv6.sh now)
iptables -A INPUT -p icmp --icmp-type 8 -s 66.220.2.74 -m state --state NEW,ESTABLISHED,RELATED -j ACCEPT
iptables -A INPUT -p icmp -i $LAN_IFACE -m state --state NEW,ESTABLISHED,RELATED -j ACCEPT
iptables -A INPUT -p icmp -i $WAN_IFACE -j DROP

# supposedly this lets you port forward to client IPs from the host router
#iptables -t nat -R POSTROUTING 3 -s 192.168.0.0/16 -d 192.168.0.0/16 -j MASQUERADE

# forward ports
for port_forward in $PORT_FORWARDS
do
	public_port=$(echo $port_forward | cut -f1 -d'>')
	destination=$(echo $port_forward | cut -f2 -d'>')
	iptables -t nat -I PREROUTING -i $WAN_IFACE -p tcp --dport $public_port -j DNAT --to-destination $destination
	iptables -A FORWARD -i $WAN_IFACE  -p tcp --dport $public_port -j ACCEPT
done

# forward udp ports
for port_forward in $UDP_PORT_FORWARDS
do
	public_port=$(echo $port_forward | cut -f1 -d'>')
	destination=$(echo $port_forward | cut -f2 -d'>')
	iptables -t nat -I PREROUTING -i $WAN_IFACE -p udp --dport $public_port -j DNAT --to-destination $destination
	iptables -A FORWARD -i $WAN_IFACE  -p udp --dport $public_port -j ACCEPT
done

# enable multicast so WOL across the internet works
# http://wiki.archlinux.org/index.php/Wake-on-LAN#Across_the_internet
arp -s 192.168.1.254 FF:FF:FF:FF:FF:FF

# transparent squid proxy settings
#iptables -t nat -A PREROUTING -s 192.168.1.3 -p tcp --dport 80 -j ACCEPT
#iptables -t nat -A PREROUTING -p tcp --dport 80 -j DNAT --to-destination 192.168.1.3:3128
#iptables -t nat -A POSTROUTING -j MASQUERADE

You will certainly have to change some stuff to suit you in there, but it will enable ipv4 routing. I have another set of scripts to enable/disable IPv6 running through Hurricane Electric.
