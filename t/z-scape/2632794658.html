             }
         } else {
             cancel();
         }
     } else if(getOwner().isAttackingPlayer()){
         Player player = getOwner().getAttackingPlayer();

             }
         } else {
             cancel();
         }
     } else {
         cancel();
     }
 }

 @Override
 public boolean cancel() {
     if(getOwner().isAttackingPlayer()){
         getOwner().setAnimation(new PlayerAnimation(-1,0));
         getOwner().getAnimation().setChanged(true);
         getOwner().setAttackingPlayer(null);
         getOwner().resetCombatCycle();
         terminate();
     } else if(getOwner().isAttackingNPC()){

         getOwner().getAnimation().setChanged(true);

 }
