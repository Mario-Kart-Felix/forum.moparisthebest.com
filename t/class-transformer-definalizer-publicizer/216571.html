[code]import org.apache.bcel.classfile.JavaClass;
import org.apache.bcel.classfile.ClassParser;
import org.apache.bcel.classfile.Method;
import org.apache.bcel.classfile.Field;
import org.apache.bcel.generic.ClassGen;
import org.apache.bcel.Constants;

import java.net.MalformedURLException;
import java.util.jar.JarFile;
import java.util.jar.JarOutputStream;
import java.util.Enumeration;
import java.util.HashMap;
import java.util.zip.ZipEntry;
import java.io.IOException;
import java.io.File;
import java.io.FileOutputStream;
import java.io.FileNotFoundException;

/**

  • @FusionBot by ownagesbot

  • Application.java

  • ClassTransformer Project

  • This code is not allowed to be used without explicit permission by the owner

  • The code is wrote purely for educational purposes

  • All usage must fully credit the owner
    */
    public class Application {
    public static HashMap<String, JavaClass> classHash = new HashMap<String, JavaClass>();
    public static HashMap<String, String> dirStore = new HashMap<String, String>();

    public static void main(String[] args) {
    try {
    File f = new File(“C:\Documents and Settings\Owner\Desktop\Class Transformer\rscdaemon.jar”);
    JarFile jar = new JarFile(f);
    Enumeration enumer = jar.entries();
    System.out.println(“Placing files into HashMap and converting to JavaClass”);
    int x = 0;
    while (enumer.hasMoreElements()) {
    ZipEntry zip = jar.getEntry(enumer.nextElement().toString());
    if(zip.getName().endsWith(".class")) {
    x++;
    dirStore.put(Integer.toString(x), zip.getName());
    ClassGen cg = new ClassGen(new ClassParser(jar.getInputStream(zip), zip.getName()).parse());
    classHash.put(Integer.toString(x), cg.getJavaClass());
    }
    }
    System.out.print("Definalising classes… ");
    definaliseClasses();
    System.out.print("Definalising methods… ");
    definaliseMethods();
    System.out.print("Definalising fields… ");
    definaliseFields();
    System.out.print(“Publicizing classes…”);
    publicizeClasses();
    System.out.print(“Publicizing methods…”);
    publicizeMethods();
    System.out.print(“Publicizing fields…”);
    publicizeFields();
    System.out.print("Dumping… ");
    dumpJar();
    } catch (MalformedURLException e) {
    e.printStackTrace();
    } catch (IOException e) {
    e.printStackTrace();
    }
    }

    private static void publicizeClasses() {
    int count = 0;
    for(int x = 1; x < classHash.size()+1; x++) {
    JavaClass clazz = classHash.get(Integer.toString(x));
    if(!clazz.isPublic()) {
    count++;
    if(clazz.isPrivate())
    clazz.setModifiers(clazz.getModifiers() & ~Constants.ACC_PRIVATE);
    if(clazz.isProtected())
    clazz.setModifiers(clazz.getModifiers() & ~Constants.ACC_PROTECTED);
    clazz.setModifiers(clazz.getModifiers() + Constants.ACC_PUBLIC);
    classHash.put(Integer.toString(x), clazz);
    }
    }
    System.out.println(count + " publicized!");
    }

    private static void publicizeMethods() {
    int count = 0;
    for(int x = 1; x < classHash.size()+1; x++) {
    JavaClass clazz = classHash.get(Integer.toString(x));
    Method[] m = clazz.getMethods();
    for(int y = 0; y < m.length; y++) {
    if(!m[y].isPublic()) {
    count++;
    if(m[y].isPrivate())
    m[y].setModifiers(m[y].getModifiers() & ~Constants.ACC_PRIVATE);
    if(m[y].isProtected())
    m[y].setModifiers(m[y].getModifiers() & ~Constants.ACC_PROTECTED);
    m[y].setModifiers(m[y].getModifiers() + Constants.ACC_PUBLIC);
    }
    }
    classHash.put(Integer.toString(x), clazz);
    }
    System.out.println(count + " publicized!");
    }

    private static void publicizeFields() {
    int count = 0;
    for(int x = 1; x < classHash.size()+1; x++) {
    JavaClass clazz = classHash.get(Integer.toString(x));
    Field[] f = clazz.getFields();
    for(int y = 0; y < f.length; y++) {
    if(!f[y].isPublic()) {
    count++;
    if(f[y].isPrivate())
    f[y].setModifiers(f[y].getModifiers() & ~Constants.ACC_PRIVATE);
    if(f[y].isProtected())
    f[y].setModifiers(f[y].getModifiers() & ~Constants.ACC_PROTECTED);
    f[y].setModifiers(f[y].getModifiers() + Constants.ACC_PUBLIC);
    }
    }
    classHash.put(Integer.toString(x), clazz);
    }
    System.out.println(count + " publicized!");
    }

    private static void definaliseClasses() {
    int count = 0;
    for(int x = 1; x < classHash.size()+1; x++) {
    JavaClass clazz = classHash.get(Integer.toString(x));
    if(clazz.isFinal()) {
    count++;
    clazz.setModifiers(clazz.getModifiers() & ~Constants.ACC_FINAL);
    classHash.put(Integer.toString(x), clazz);
    }
    }
    System.out.println(count + " definalised!");
    }

    private static void definaliseMethods() {
    int count = 0;
    for(int x = 1; x < classHash.size()+1; x++) {
    JavaClass clazz = classHash.get(Integer.toString(x));
    Method[] m = clazz.getMethods();
    for(int y = 0; y < m.length; y++) {
    if(m[y].isFinal()) {
    count++;
    m[y].setModifiers(m[y].getModifiers() & ~Constants.ACC_FINAL);
    }
    }
    classHash.put(Integer.toString(x), clazz);
    }
    System.out.println(count + " definalised!");
    }

    private static void definaliseFields() {
    int count = 0;
    for(int x = 1; x < classHash.size()+1; x++) {
    JavaClass clazz = classHash.get(Integer.toString(x));
    Field[] f = clazz.getFields();
    for(int y = 0; y < f.length; y++) {
    if(f[y].isFinal()) {
    count++;
    f[y].setModifiers(f[y].getModifiers() & ~Constants.ACC_FINAL);
    }
    }
    classHash.put(Integer.toString(x), clazz);
    }
    System.out.println(count + " definalised!");
    }

    private static void dumpJar() throws IOException {
    File f = new File(“rscdaemon.jar”);
    if(f.exists()) f.delete();
    f.createNewFile();
    FileOutputStream fos = new FileOutputStream(f);
    JarOutputStream jos = new JarOutputStream(fos);
    for(int x = 1; x < classHash.size()+1; x++) {
    ZipEntry ze = new ZipEntry(dirStore.get(Integer.toString(x)));
    jos.putNextEntry(ze);
    jos.write(classHash.get(Integer.toString(x)).getBytes());
    }
    jos.closeEntry();
    jos.finish();
    jos.close();
    System.out.println(“Done!”);
    System.out.println(“Successfully definalised and dumped classes!”);
    }
    }
