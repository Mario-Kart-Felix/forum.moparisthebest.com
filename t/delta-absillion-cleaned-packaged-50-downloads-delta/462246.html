                Client c;

                        this.c = c; //Allows us to call Client
                }

                /**
                *Makes the dialogue go onwards.
                */

                public void n() {
                        c.NpcDialogue++; //Allows the dialogue to continue
                        c.NpcDialogueSend = true; //Don't know why but adding...

                *Makes the dialogue end.
                */

                public void f() {
                        c.NpcDialogue = 0;
                        c.NpcDialogueSend = false; //Made to make the boolean false so it doesn't continue any dialogue.

                /**
                *@npcId Gets the npc id so it can call the dialogue.
                *@NpcDialogue The integar for dialogues. NpcDialogue = 0;
                *All the chit chat is here...

                public void Conversations() {
                        switch (c.npcId) { //Effective easier way for getting dialogue ids rather than a whole new void calling a continueDialogue void.
