[code]import java.io.;
import java.net.;
import java.util.*;

public class ChatServer extends ConnectionManager {
public boolean serverRunning = true;
ServerSocket serverSocket = null;
public static User[] users = new User[500];
public String splitter = “\|\|\|\|”;
public ChatServer(int port) throws Exception {
try {
serverSocket = new ServerSocket(port);
System.out.println("ChatServer started on port: "+port);
}
catch (IOException e) {
System.out.println("Could not start server on port: " +port);
System.exit(-1);
}
}

public void run() {
	Socket socket = null;
	while (serverRunning) {
		try {
			socket = serverSocket.accept();
			socket.setTcpNoDelay(true);
			System.out.println("Connection Accepted from: " + getAddress(socket));
			int id = getFreeId();
			if (id != -1) {
				users[id] = new User(socket, id, this);
				sockets[id] = socket;
			}
			else {
				socket.close();
			}
			String filename = ("./BannedHosts/" + getAddress(socket) + ".dat");
			File banned = new File(filename);
			if (banned.exists()) {
				try {
					users[id].rwSocket.write("09You have been banned from this server.\nPlease refer to the hoster for more details.");
				}
				catch (Exception e) {
				}
				try {
					socket.close();
					users[id].destruct();
				}
				catch (IOException e) {
					e.printStackTrace();
				}
				sockets[id] = null;
				users[id] = null;
			}
			else {
				outputAll("14" + users[id].username + splitter + "11" + users[id].username + " has joined the channel!");
				setUserList();
			}
		}
		catch (IOException e) {
			System.out.println("Client accept failed.");
			System.exit(-1);
		}
	}
}

public int getFreeId() {
	for (int i = 0; i < sockets.length; i++) {
		if (sockets[i] != null) {
			continue;
		}
		return i;
	}
	return -1;
}

public void setUserList() {
	String userList = (splitter+"16");
	for(User u : users) {
		if(u != null && u.username.length() > 0) {
			userList += (u.username + "\"");
		} 
	}
	for(User u : users) {
		try {
			u.rwSocket.write(userList);
		}
		catch(Exception e) {
		}
	}
}

public void outputAll(String message) {
	for(User u : users) {
		if(u != null && getAddress(u.socket) != "0.0.0.0") {
			try {
				u.rwSocket.write(message);
			}
			catch(Exception e) {
			}
		}
	}
}

UserSocket.java (this is where the magic happens IMO)

[code]import java.net.;
import java.io.;
import java.lang.*;

public class UserSocket {

public User user;
public Socket socket;
public InputStream input;
public OutputStream output;
public ObjectOutputStream oos;
public String splitter = "\\|\\|\\|\\|";


public UserSocket(User p, Socket s) {
user = p;
    socket = s;
    try {
        input = socket.getInputStream();
        output = socket.getOutputStream();
} catch (Exception e) {
        e.printStackTrace();
    }
}

public int read() throws Exception {
if (input == null || input.available() < 1) {
        return -1;
    }
byte[] userInputB = new byte[input.available()];
input.read(userInputB);
String userInput = new String(userInputB);
String packetId = userInput.substring(0, 2);
handleInput(userInput.substring(2), packetId);
return 0;

}

public void write(byte[] b) throws Exception {
    if (socket == null) {
        return;
    }
    output.write(b);
}

public void write(String s) throws Exception {
    if (socket == null) {
        return;
    }
byte[] b = s.getBytes();
    output.write(b);
}

public void writeUsers(Object users) {
try {	
	oos.writeObject(users);
}
catch (IOException e) {
	e.printStackTrace();
}
}

public void handleInput(String input, String packetId) {
	int packet = Integer.parseInt(packetId);
	switch (packet) {
		case 10: //Normal Text
			if(input != "") {
				user.cs.outputAll("10[" + user.username + "]: "+input);
			}
		break;

		case 11: //Username Change
			boolean taken = false;
			for (User u : user.cs.users) {
				if(u != null) {
					if(input.equalsIgnoreCase(u.username)) {
						taken = true;
						try {
							write("09That username is in-use!");
						}
						catch (Exception e) {
							e.printStackTrace();
						}
						return;
					}
				}
			}
			if(taken == false && input.length() < 13 && !input.equalsIgnoreCase("server") && !input.contains(";") && !input.contains("\"") && !input.contains("\\") && !input.contains("}") 
                                 && !input.contains("{") && !input.contains("[") && !input.contains("]") && !input.contains(")") && !input.contains("(")) {
				user.cs.outputAll("11" + user.username + " is now known as " + input + splitter + "15" +user.username + "\"" +input);
				user.username = input;
			}
			else {
				try {
					write("09That username contains invalid characters or is too long! (Max 12)");
				}	
				catch (Exception e) {
					e.printStackTrace();
				}
			}
		break;

		case 12: //Command (starts with /)
			if(input.startsWith("ban")) {
				boolean ban = false;
				for (User u : user.cs.users) {
					if(u != null) {
						if(u.username.equalsIgnoreCase(input.substring(4))) {
							try {
								ban = new File("./BannedHosts/" + u.cs.getAddress(socket) + ".dat").createNewFile();
							}
							catch (IOException e) {
								e.printStackTrace();
							}
							if(ban) {
								user.cs.outputAll("11" + input.substring(4) + " has been ip-banned!");
								try {
									u.socket.close();
								}
								catch (IOException e) {
								}
								u.cs.sockets[u.userID] = null;
								u.cs.users[u.userID] = null;
								user.destruct();
							}
							return;
						 }	
					}
				}
			}	
		break;
		
		case 14: //Quit
			System.out.println("14");
			user.cs.outputAll("11" + user.username + " has quit.");
			user.cs.setUserList();
			user.destruct();
		break;
	}
}

}

public void parseInput(String input) { //All = 10, ServerAll = 11, thisOnly = 12 newUser = 14
	String finalInput = (input.substring(2));
	if(input.startsWith("12")) {
		thisOnly("\n" + finalInput);
	}
	if(input.startsWith("09")) {
		error("\n" + finalInput);
	}
	else if(input.startsWith("11")) {
		serverAll("\n" + finalInput);
	}
	else if(input.startsWith("10")) {
		all("\n" + finalInput);
	}
	else if(input.startsWith("14")) {
		addUser(finalInput);
	}
	else if(input.startsWith("15")) {
		String newUser = finalInput.substring(finalInput.indexOf("\"")+1);
		String oldUser = finalInput.substring(0, finalInput.indexOf("\""));
		changeUser(oldUser, newUser);
	}
	else if(input.startsWith("16")) {
		setUserList(finalInput);
	}
	ChatArea.selectAll();
}

[code]import java.io.;
import java.net.;
import java.lang.;
import java.util.;

public class ClientThread extends Thread {
public InputStream input;
public Socket socket = null;
public boolean reading = true;
public ChatClient cs;

public ClientThread(Socket socket, ChatClient chs) {
	this.socket = socket;
this.cs = chs;
}

public void run() {
	while (reading) {
		readData();
		try {
			Thread.sleep(1);
		}
		catch (InterruptedException e) {
  			e.printStackTrace();
   			}
	}
}

public void readData() {
	try { 
		input = socket.getInputStream();
		if (input == null || input.available() < 1) {
        			return;
    		}
		byte[] serverInputB = new byte[input.available()];
		input.read(serverInputB);
		String splitter = "\\|\\|\\|";
		String serverInput = new String(serverInputB);
		if(serverInput.contains(splitter)) {
			String[] finalInput = serverInput.split("\\|");
			for(int i = 0; i < finalInput.length; i++) {
				if(finalInput[i].length() > 2) {
					cs.parseInput(finalInput[i].replace("\\", ""));
				}
			}
		}
		else {
			cs.parseInput(serverInput);
		}

	}
	catch (IOException e) {
		//e.printStackTrace();
	}
}
