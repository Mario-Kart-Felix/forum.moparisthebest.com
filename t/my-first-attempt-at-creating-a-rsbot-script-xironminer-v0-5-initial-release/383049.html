public boolean onStart(){
	if (game.isLoggedIn());
	log("xIronMiner Initialised.");
	startTime = System.currentTimeMillis();
	startExp = skills.getCurrentExp(skills.MINING);
	startLevel = skills.getCurrentLevel(skills.MINING);

@Override
public int loop() {
	TryRun();
	switch(mode){
	case 0: //Powermining
		if(inventory.isFull()){
			DropOres();
		} else {
			MineOres();
		}
		break;
	case 1: //Mine 1 Drop 1
		if (inventory.containsOneOf(IRONORE_ID)){
			status = "Dropping the ore.";
			inventory.dropAllExcept(PICKAXE);
		} else {
			MineOres();
		}
		break;
	case 2: //Banking
		if (inventory.isFull()){
			if (atBank()){
				doBank();
			} else {
				walk();
			}
		} else {
			if (atMine()){
				MineOres();
			} else {
				walkR();
			}
		}
		break;
	}
	return random(250, 500);
}

public void TryRun(){
	if (Running != true){
		if(!walking.isRunEnabled() && walking.getEnergy() > random(23,69)){
			walking.setRun(true);
			Running = true;
			status = "Turning on running.";
		}
	}
}

public void DropOres(){
	status = "Dropping all ores.";
	inventory.dropAllExcept(PICKAXE);
	loadsDone++;
}

public void MineOres(){
	if (walking.getEnergy() == 0){
		Running = false;
	}
	status = "Finding rock.";
	RSObject iron = objects.getNearest(IRONROCKS);
		if (iron != null && getMyPlayer().getAnimation() == -1) {
			status = "Mining ores.";
			iron.doAction("Mine");
			if (random(1, 8) == 4){
				status = "Antiban.";
				Antiban();
			}
		}
	}

private boolean atBank(){
	RSObject bank = objects.getNearest(BOOTHID);
	if (bank != null){
		if (bank.isOnScreen()){
			return true;
		}
	}
	return false;
}

private boolean atMine(){
	RSObject rock = objects.getNearest(IRONROCKS);
	if (rock != null){
		if (rock.isOnScreen()){
			return true;
		}
	}
	return false;
}

private void doBank(){
	if(bank.isOpen()){
		bank.depositAllExcept(PICKAXE);
		sleep(800, 1350);
	} else {
		bank.open();
		sleep(1750, 2200);
	}
}

private void walk(){
	switch(Bank){
		case 0:
			status = "Walking to Draynor Bank.";
			pathToBankD.traverse();
			break;
		case 1:
			status = "Walking to Varrock East Bank.";
			pathToBankVE.traverse();
			break;
		case 2:
			status = "Walking to Varrock West Bank.";
			pathToBankVW.traverse();
	}
}

private void walkR(){
	switch(Bank){
		case 0:
			status = "Walking to the Rimmington Mine.";
			pathToBankD.reverse();
			pathToBankD.traverse();
			pathToBankD.reverse();
			break;
		case 1:
			status = "Walking to the Varrock East Mine.";
			pathToBankVE.reverse();
			pathToBankVE.traverse();
			pathToBankVE.reverse();
			break;
		case 2:
			status = "Walking to the Varrock West Mine.";
			pathToBankVW.reverse();
			pathToBankVW.traverse();
			pathToBankVW.reverse();
			break;
	}
}

public void messageReceived(MessageEvent e) {
    String serverMessage = e.getMessage().toLowerCase();
    if (serverMessage.contains("you manage to mine")){
    	oresMined++;
    }
}

private void Antiban(){

			sleep(1000, 2000);

		case 19:
			camera.moveRandomly(random(800, 1250));
			break;
	}
}

@Override
public void onRepaint(Graphics g) {
	//Run time counter
	long millis = System.currentTimeMillis() - startTime;
	long hours = millis / (1000 * 60 * 60);
	millis -= hours * (1000 * 60 * 60);
	long minutes = millis / (1000 * 60);
	millis -= minutes * (1000 * 60);
	long seconds = millis / 1000;
	
	//Exp gained
	expGained = skills.getCurrentExp(skills.MINING) - startExp;
	
	//Current Level
	currentLevel = skills.getCurrentLevel(skills.MINING);
	
	//Levels gained
	levelsGained = currentLevel - startLevel;
	
	//Exp / h
	float expSec = 0;
	if ((minutes > 0 || hours > 0 || seconds > 0) && expGained > 0){
		expSec = ((float) expGained)/(float)(seconds + (minutes*60) + (hours*60*60));
	}
	float expMin = expSec * 60;
	float expHour = expMin * 60;
	
	//Ores / h
	float oresSec = 0;
	if ((minutes > 0 || hours > 0 || seconds > 0) && oresMined > 0){
		oresSec = ((float) oresMined)/(float)(seconds + (minutes*60) + (hours*60*60));
	}
	float oresMin = oresSec * 60;
	float oresHour = oresMin * 60;
	
	//Profit / h
	float profSec = 0;
	if ((minutes > 0 || hours > 0 || seconds > 0) && oresMined > 0){
		profSec = ((float) oresMined)/(float)(seconds + (minutes*60) + (hours*60*60));
	}
	float profMin = profSec * 60;
	float profHour = profMin * 60;
	
	g.setColor(new Color(0, 0, 0, 150));
	g.fillRect(9, 347, 484, 109);
	g.setFont(new Font("Arial Black", 0, 18));
	g.setColor(new Color(0, 0, 0, 176));
	g.drawString("xIronMiner", 11, 363);
	g.setFont(new Font("Arial Black", 0, 14));
	g.setColor(new Color(0, 0, 0, 176));
	g.drawString("v" + PROPS.version(), 124, 363);
	g.setFont(new Font("Arial", 0, 12));
	g.setColor(new Color(0, 0, 0, 176));
	g.drawString("Runtime: " + hours + ":" + minutes + ":" + seconds, 11, 378);
	/*g.setFont(new Font("Arial", 0, 12));
	g.setColor(new Color(0, 0, 0, 176));
	g.drawString("Time to level: ", 11, 390);*/
	g.setFont(new Font("Arial", 0, 12));
	g.setColor(new Color(0, 0, 0, 176));
	g.drawString("Current Level: " + currentLevel + "(" + levelsGained + ")", 11, 402);
	g.setFont(new Font("Arial", 0, 12));
	g.setColor(new Color(0, 0, 0, 176));
	g.drawString("Status: " + status, 11, 415);
	g.setFont(new Font("Arial", 0, 12));
	g.setColor(new Color(0, 0, 0, 176));
	g.drawString("Mined ore: " + oresMined, 352, 378);
	g.setFont(new Font("Arial", 0, 12));
	g.setColor(new Color(0, 0, 0, 176));
	g.drawString("Mined ore / h: " + (int)oresHour, 352, 390);
	g.setFont(new Font("Arial", 0, 12));
	g.setColor(new Color(0, 0, 0, 176));
	g.drawString("EXP Gained: " + expGained, 352, 402);
	g.setFont(new Font("Arial", 0, 12));
