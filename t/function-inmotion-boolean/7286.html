Function InMotionMM : Boolean;
var
area : array [0…4] of Integer;
areax : array [0…4] of Integer;

area[0]:=GetColor(642, 13);
area[1]:=GetColor(576, 84);

area[4]:=GetColor(719, 74);

areax[2]:=GetColor(619, 134);
areax[3]:=GetColor(669, 136);
areax[4]:=GetColor(719, 74);
if (area[0] = areax[0]) and (area[1] = areax[1]) and (area[2] = areax[2]) and (area[3] = areax[3]) and (area[4] = areax[4]) then begin
Result:= False;
end
else
Result:= True;
end;
