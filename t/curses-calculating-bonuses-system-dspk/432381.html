public void turmBonus() {
	if (c.curseActive[19] == false) {
	c.curseDefence = 0;
	c.curseStrength = 0;
	c.curseAttack = 0;
	deactive = false;
	}else if (c.inCombat == true && deactive == false) {
	c.curseDefence = 30;
	c.curseStrength = 33;
	c.curseAttack = 30;
