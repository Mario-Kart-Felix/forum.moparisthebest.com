program HundredBottlesOfBeerByRon;

const
  HowManyBottles = 100; // Num of bottles to go through.
  WaitTime = 8000;      // The WaitTime in between each bottle in milliseconds.

var
  arrBottle, arrDrink, arrPlace : array of string;
  WhichBottle, WhichDrink, WhichPlace,
  NumOfBottles, WaitT, i : integer;

begin
  ClearDebug;
  arrBottle := ['bottle', 'cup', 'glass', 'mug', 'jug'];
  arrDrink := ['beer', 'soda', 'water', 'lemonade', 'ice tea'];
  arrPlace := ['wall', 'ground', 'ceiling', 'house', 'table'];
  WhichBottle := Random(GetArrayLength(arrBottle));
  WhichDrink := Random(GetArrayLength(arrDrink));
  WhichPlace := Random(GetArrayLength(arrPlace));
  NumOfBottles := HowManyBottles;
  if (NumOfBottles < 10) then
    NumOfBottles := 10;
  WaitT := WaitTime;
  if (WaitT < 10) then
    WaitT := 10;
  for i := NumOfBottles downto 1 do
  begin
    WriteLn(IntToStr(i) + ' ' + arrBottle[WhichBottle] + '(s) of ' + arrDrink[WhichDrink] + ' on the ' + arrPlace[WhichPlace] + '. ' + IntToStr(i) + ' ' + arrBottle[WhichBottle] + '(s) of ' + arrDrink[WhichDrink] + '.');
    WriteLn('  Take one down, pass it around.');
    if (i > 1) then
      WriteLn('  ' + IntToStr(i - 1) + ' ' + arrBottle[WhichBottle] + '(s) of ' + arrDrink[WhichDrink] + ' on the ' + arrPlace[WhichPlace] + '.');
    Wait(WaitT);
  end;
  WriteLn('');
  WriteLn('~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~');
  WriteLn(' No more ' + arrBottle[WhichBottle] + '(s) of ' + arrDrink[WhichDrink] + ' on the ' + arrPlace[WhichPlace] + '!');
  WriteLn('~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~');
end.

Just a few tips.
You don’t have to put

x:= 0;

The integer, if never used before, starts at 0.

And also, you could use the x integer in a for loop.

for x:= 99 downto 0 do Writeln(inttostr(x));

Counts from 99 down to 0 and Writes it or…

for x:= 0 to 99 do Writeln(inttostr(x));

Count from 0 to 99 and Writes it.
