Step 3:

  • Create another java file and call it ObjectManager.java
  • Add this code into it:
/**
 * @Author: RandQm
 */

import java.io.BufferedReader;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;

public class ObjectManager {
	
	private static final int MAX_OBJECTS = 1000;
	
	private Object objects[] = new Object[MAX_OBJECTS];
	
	
	/*
	 * Replacing an existing object by a new object.
	 */
	public void replaceObject (client c, int x, int y, int objId, int face, int type) {
		c.outStream.createFrame(85);
		c.outStream.writeByteC(y - (c.mapRegionY * 8));
		c.outStream.writeByteC(x - (c.mapRegionX * 8));
		c.outStream.createFrame(101);
		c.outStream.writeByteC((type << 2) + (face & 3));
		c.outStream.writeByte(0);
		if (objId != -1) {
			c.outStream.createFrame(151);
			c.outStream.writeByteS(0);
			c.outStream.writeWordBigEndian(objId);
			c.outStream.writeByteS((type << 2) + (face & 3));
		}
	}
	
	/*
	 * Create a new private object on the map.
	 */
	public void addObject (client c, int x, int y, int objId, int face, int type) {
		c.outStream.createFrame(85);
		c.outStream.writeByteC(y - (c.mapRegionY * 8));
		c.outStream.writeByteC(x - (c.mapRegionX * 8));
		c.outStream.createFrame(151);
		c.outStream.writeByteA(0);
		c.outStream.writeWordBigEndian(objId);
		c.outStream.writeByteS((type << 2) + (face & 3));
	}
	
	
	/*
	 * Removing all temporary objects.
	 */
	public ObjectManager() {
		for (int i = 0; i < MAX_OBJECTS; i++) {
			objects[i] = null;
		}
	}
	
	/*
	 * Create a new object on the map visible for all players in the region.
	 */
	public void addGlobalObject (int objType, int original, int objX, int objY, int objStandTime, int objCreator, boolean temp) {
		/*
		 * Picks a new value from (maxobjects) everytime a new object gets added, until there are no unused values anymore,
		 * this is to make the server recognize every object as unique, so if we spawn multiple of the same object,
		 * it still recognizes em as different ones
		 */
		int slot = -1;
		for (int i = 1; i < MAX_OBJECTS; i++) {
			if (objects[i] == null) {
				slot = i;
				break;
			}
		}
		if (slot == -1)
			return;
		/*
		 * Creating the new object via Object.java you just added
		 */
		Object newObjects = new Object(slot, objType, original, objX, objY, objStandTime, objCreator, temp);
		objects[slot] = newObjects;
	}
	
	/*
	 * Updating a temporary object.
	 */
	private void updateObjects (final int objIndex, final int objX, final int objY, final int objId) {
		/*
		 * Loops through all players
		 */
		for (Player p : PlayerHandler.players) {
			if (p != null) {
				client person = (client) p;
				if (person.playerName != null || person.playerName != "null") {
					/*
					 * If a player online has changed region, or logged in,
					 * objects will be updated for that person.
					 */
					if (person.changedRegion) {
						objects[objIndex].createObject();
						person.changedRegion = false;
					}
				}
			}
		}
	}
	
	/*
	 * Object processing
	 */
	public void process() {
		for (int i = 0; i < MAX_OBJECTS; i++) {
			if (objects[i] != null) { //If there are objects created then...
				if (objects[i].objectStandTime > 0) { //the standtimer of the object will run
					objects[i].objectStandTime --;
					//checks if objects have to be updated for a player
					updateObjects(i, objects[i].objectX, objects[i].objectY, objects[i].objectType);
				} else { //if the timer ran out it will delete or replace(by the original) the object
					objects[i].deleteTempObject(objects[i].objectX, objects[i].objectY, objects[i].objectCreator);
					objects[i] = null; //makes the used slot by the object available again
				}
			}
		}
	}
	
	/*
	 * Adding static, not originally existing objects to the players map.
	 * Deleting originally existing objects from the player map.
	 * This method is to read our cfg file we are adding for ^ dat
	 */
	public boolean loadWorldObjects (client c, String FileName) {
		String line = "";
		String token = "";
		String token2 = "";
		String token2_2 = "";
		String[] token3 = new String[10];
		boolean EndOfFile = false;
		BufferedReader characterfile = null;
		try {
			characterfile = new BufferedReader(new FileReader("config\\" + FileName));
		} catch (FileNotFoundException fileex) {
			System.out.println(FileName + ": file not found.");
			return false;
		}
		try {
			line = characterfile.readLine();
		} catch (IOException ioexception) {
			System.out.println(FileName + ": error loading file.");
			return false;
		}
		while ((EndOfFile == false) && (line != null)) {
			line = line.trim();
			int spot = line.indexOf("=");
			if (spot > -1) {
				token = line.substring(0, spot);
				token = token.trim();
				token2 = line.substring(spot + 1);
				token2 = token2.trim();
				token2_2 = token2.replaceAll("\t\t", "\t");
				token2_2 = token2_2.replaceAll("\t\t", "\t");
				token2_2 = token2_2.replaceAll("\t\t", "\t");
				token2_2 = token2_2.replaceAll("\t\t", "\t");
				token2_2 = token2_2.replaceAll("\t\t", "\t");
				token3 = token2_2.split("\t");
				if (token.equals("add")) {
					addObject(c, Integer.parseInt(token3[0]), Integer.parseInt(token3[1]), Integer.parseInt(token3[2]), 0, 10);
				}
				if (token.equals("delete")) {
					replaceObject(c, Integer.parseInt(token3[0]), Integer.parseInt(token3[1]), -1, 0, 10);
				}
			} else {
				if (line.equals("[ENDOFOBJECTLIST]")) {
					try {
						characterfile.close();
					} catch (IOException ioexception) {
					}
					return true;
				}
			}
			try {
				line = characterfile.readLine();
			} catch (IOException ioexception1) {
				EndOfFile = true;
			}
		}
		try {
			characterfile.close();
		} catch (IOException ioexception) {
		}
		return false;
	}

}

Step 4:

  • Go into the directory where your cfg files are located, this will most probably be the folder called “config”
    if it isn’t, don’t forget to change the path in the code to load the file from the previous step.
  • Add a new file and call it worldObjects.cfg
  • then add this code into it:
/////////////////////////////////////////////////////////////
//To create a new object on the map:                       //
//add = objectCoordinateX	objectCoordinateY	objectId  //
/////////////////////////////////////////////////////////////
add = 0000	0000	0

/////////////////////////////////////////////////////////////
//To delete an existing object on the map:                 //
//delete = objectCoordinateX	objectCoordinateY         //
/////////////////////////////////////////////////////////////
delete = 0000	0000

[ENDOFOBJECTLIST]

Step 5:

  • Open your server.java
  • add in there (you’ll see more lines similar to this)
public static ObjectManager objectManager = null;
  • And a bit lower inside ‘public static void main’, with the others similar to this:
objectManager = new ObjectManager();

Step 6:

  • Now open process.java and with the similar lines (e.g. “server.playerHandler.process();”)
    add this line: this is to keep our object process running
server.objectManager.process();

Step 7:

  • Now open your client.java and search for:
void initialize()[/code]
This is where your welcome message ect is located, if you scroll down till the end of the method you should see one or more of these:

[/code]
this is to load the objects from our cfg file when the player logs in.
