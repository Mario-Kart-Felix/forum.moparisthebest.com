package com;

import de.fernflower.main.extern.IIdentifierRenamer;

public class Rename implements IIdentifierRenamer {
  
  private static int countClass = 0;
  private static int countField = 0;
  private static int countMethod = 0;
  
  public boolean toBeRenamed(int elementType, String className, String element, String descriptor) {
    if(elementType == IIdentifierRenamer.ELEMENT_CLASS) {
      return !className.equals("client");
    }
    
    if(elementType == IIdentifierRenamer.ELEMENT_FIELD || elementType == IIdentifierRenamer.ELEMENT_METHOD) {
      return element.length() < 3;
    }
    
    return false;
  }

  public String getNextClassname(String fullName, String shortName) {
    System.out.println(fullName + " " + shortName);
    return "class" + (countClass++);
  }

  public String getNextFieldname(String className, String field, String descriptor) {
    System.out.println(className + "  field " + field + " " + descriptor);
    return "field" + (countField++);
  }

  public String getNextMethodname(String className, String method, String descriptor) {
    System.out.println(className + " method " + method + " " + descriptor);
    return "method" + (countMethod++);
  }

}
