I was thinking about charging for this. But I’m nice.

You park this script at the thugs, stand about here, using this camera angle:

Let the script run, and it’ll collect right click and pick up herbs, natures, chaos and laws. When your inventory is full, you drop the herbs you don’t want, and let it run again.

Collect 1k natures, and lots of rannar quick. Based off other scripts. It even simulates human actions very closely.

{
   # ThugFighter Professional
    By Bumis
}
program ThugFighter;
{.include SSI2.txt}
{.include BoxBreaker2.txt}
var
 ItemColor1,ItemColor2,ItemColor3,ItemTol: Integer;
 ItemGrabbed,DistortMouseCount,DMI: Integer;
 InvenCount,CheckFighting: Integer;
 DoingInven,DoingMain,CheckStats,CheckingStats: Integer;
 IAmFighting,DoingSpecial: Boolean;
Procedure Variables;
Begin
 //Set Item Name On Line-
 ItemColor1:=14080
 ItemColor2:=1477906
 ItemColor3:=1878496
 ItemTol:=10
 UpChars:= LoadChars2(AppPath + 'CharsRS22\');
 ChatChars:= LoadChars2(AppPath + 'CharsChat2\');
 InvenCount:=0
End;

Procedure Review;
Begin
 Writeln('')
 Writeln('')
 Writeln('::::::::::::::::::::::::::::::::::::::::::::::::')
 Writeln(': ThugFighter/Bumis')
 Writeln(': Picked up items ' + IntToStr(ItemGrabbed) + ' times.')
 If DoingInven=1 Then
  Begin
   Writeln(': Current Process: Doing Inventory')
  End;
 If DoingMain=1 Then
  Begin
   Writeln(': Current Process: Doing Main Procedure')
  End;
 If IAmFighting=True Then
  Begin
   Writeln(': Current Process: Doing Special')
  End;
 If CheckingStats=1 Then
  Begin
   Writeln(': Current Process: Checking Stats')
  End;
 Writeln('::::::::::::::::::::::::::::::::::::::::::::::::')
Writeln('')
Writeln('')
End;

Procedure DoInven;
Begin
 Status('Checking Stats')
 MMouse(650,187,7,7)
 Wait(50)
 GetMousePos(x,y)
 ClickMouse(x,y,true)
 Wait(100+Random(20))
  If Not FindColorTolerance(x,y,4345174,707,437,722,448,10) Then
   Begin
    playsound('C:\WINDOWS\Media\ringin.wav');
    MMouse(200,200,100,100)
    Wait(200+random(200))
    playsound('C:\WINDOWS\Media\ringin.wav');
    playsound('C:\WINDOWS\Media\ringin.wav');
   End;
  If FindColorTolerance(x,y,4345174,707,437,722,448,10) Then
  Begin
   MMouse(200,200,100,100)
   Wait(200+random(200))
   MMouse(200,200,100,100)
   Wait(200+random(200))
   MMouse(200,200,100,100)
   Wait(200+random(200))
  End;
End;

Procedure CheckSpecial;
Begin
Review;
IAmFighting:=True
Wait(1300+Random(400))
MMouse(200,200,75,75)
Wait(500+Random(100))
MMouse(559,186,5,5)
Wait(400+Random(200))
GetMousePos(x,y)
ClickMouse(x,y,true)
 If FindColorTolerance(x,y,1203866,615,440,641,452,10) Then
  Begin
   DoingSpecial:=True
   Review;
   MoveMouseSmooth(577+Random(110),442+Random(11))
   Wait(2000+Random(400))
   GetMousePos(x,y)
   ClickMouse(x,y,true)
   Wait(5000+random(1400))
   CheckFighting:=0
  End;
IAmFighting:=False
DoingSpecial:=False
Review;
End;

Procedure AmIFighting;
Begin
 If FindColorTolerance(x,y,65280,187,30,419,243,10) Then
  Begin
   IAmFighting:=True
   Review;
   CheckSpecial;
  End;
IAmFighting:=False
Review;
End;

Procedure CheckMyStats;
Begin
 Wait(500+Random(500))
 MMouse(583,186,7,7)
 Wait(200+Random(100))
 GetMousePos(x,y)
 MMouse(x,y,2,2)
 Wait(200+Random(200))
 MoveMouseSmooth(559+Random(50),212+Random(80))
 Wait(500+Random(300))
 GetMousePos(x,y)
 MMouse(x,y,2,2)
End;

Procedure DistortMouseNow;
Begin
  DMI:=0
  Status('Distort Mouse Activated')
  MMouse(260,160,130,80)
  Wait(300+Random(260))
   Repeat
    DMI:=DMI+1
    GetMousePos(x,y)
    MMouse(x,y,2,2)
    Wait(200+Random(200))
   Until(DMI=5+Random(22))
  DMI:=0
End;

Procedure FindItems;
Begin
 Status('Checking Items')
 If FindColorTolerance(x,y,ItemColor1,0,0,500,620,ItemTol) Then
  Begin
   MMouse(x,y,2,2)
   if FindColorTolerance(x,y,4289162,0,0,110,26,10) then
   begin
   Wait(300+Random(250))
   GetMousePos(X,Y)
   ClickMouse(x,y,false)
   Wait(800+Random(700))
   if IsTextInAreaEx(x-160,y-160,x+125,y+125,x,y,'Herb',100,UpChars,True,false,0,1,3637214) then
    Begin
     MMouse(x+Random(5),y+random(3),2,2)
     Wait(400+Random(700))
     GetMousePos(X,Y)
     ClickMouse(x,y,true)
    End;
   End;
  end;
 If FindColorTolerance(x,y,ItemColor2,0,0,500,620,ItemTol) Then
  Begin
   MMouse(x,y,2,2)
   if FindColorTolerance(x,y,4289162,0,0,110,26,10) then
   begin
   Wait(200+Random(150))
   GetMousePos(X,Y)
   ClickMouse(x,y,false)
   Wait(700+Random(700))
   if IsTextInAreaEx(x-160,y-160,x+125,y+125,x,y,'Nature rune',100,UpChars,True,false,0,1,3637214) then
    Begin
     MMouse(x+Random(5),y+random(3),2,2)
     Wait(400+Random(400))
     GetMousePos(X,Y)
     ClickMouse(x,y,true)
    End;
  End;
  End;
 If FindColorTolerance(x,y,ItemColor3,0,0,500,620,ItemTol) Then
  Begin
   MMouse(x,y,2,2)
   if FindColorTolerance(x,y,4289162,0,0,110,26,10) then
   begin
   Wait(200+Random(150))
   GetMousePos(X,Y)
   ClickMouse(x,y,false)
   Wait(700+Random(700))
   if IsTextInAreaEx(x-160,y-160,x+125,y+125,x,y,'Chaos rune',100,UpChars,True,false,0,1,3637214) then
    Begin
     MMouse(x+Random(5),y+random(3),2,2)
     Wait(400+Random(400))
     GetMousePos(X,Y)
     ClickMouse(x,y,true)
    End;
    end;
  End
End;

Procedure AntiLogOut;
Begin
FindItems;
DoingMain:=1
Status('Main Procedure')
Review;
Wait(500+random(2000))
GetMousePos(x,y)
MMouse(x,y,2,2)
Wait(1000+Random(3000))
 Begin
  MMouse(361,233,77,77)
  Wait(100+Random(400))
  GetMousePos(x,y)
  MMouse(x,y,5,5)
  Wait(100+Random(400))
  DistortMouseCount:=DistortMouseCount+1
  If DistortMouseCount=1+Random(4) Then
   Begin
    DistortMouseNow;
   End;
  GetMousePos(x,y)
  MMouse(x,y,3,3)
  Wait(100+Random(400))
  MMouse(250,150,250,150)
  If CheckStats=4+Random(2) Then
   Begin
    CheckingStats:=1
    Review;
    CheckMyStats;
    CheckStats:=0
   End;
  InvenCount:=InvenCount+1
  CheckFighting:=CheckFighting+1
  If CheckFighting=4+Random(3) Then
   Begin
    CheckFighting:=0
    DoingMain:=0
    Review;
    AmIFighting;
   End;
  If InvenCount=6+Random(4) Then
   Begin
    MMouse(650,183,4,4)
    DoingMain:=0
    DoInven;
   End;
 Review;
 End;
DoingMain:=0
End;

begin
Variables;
BoxOpenerSetup;
DistortMouseCount:=0
DisguiseScar('Bearshare Pro 4.5.1');
Writeln('')
Writeln('')
Writeln('')
Writeln('::::::::::::::::::::::::::::::::::::::::::::::::::::')
Writeln(': ThugFighter 1.0 /                   :')
Writeln(':  Written by Bumis, Based off of BumTokel Coding  :')
Writeln('::::::::::::::::::::::::::::::::::::::::::::::::::::')
Writeln('')
Writeln('')
Writeln('')
Wait(5000)
IAmFighting:=False
 Repeat
 Review;
 Wait(1000+Random(500))
 AntiLogout;
 Until(False)
end.
