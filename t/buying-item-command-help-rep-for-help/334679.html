void tradeIn(int i, int iAmount, int j, int jAmount)
{
	if(playerHasItem(i, iAmount)
	{
		deleteItem(i, iAmount);
		addItem(j, jAmount);
		sM("You trade " + iAmount.toString() + " " + getItemName(i) + " for " +getItemName(j);
	} else {
		sM("Insufficient " + getItemName(i) + " you need " + iAmount.toString() +".");
	}
}
