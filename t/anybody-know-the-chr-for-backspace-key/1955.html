Procedure CleanKeys;
var i: integer;
begin

KeyUp(i);
end;

{This procedure presses the key you input. It then waits for

save lines in your script Remember to use shift or control

begin
KeyDown(key);
Wait(WaitTime);
KeyUp(key);
end;
