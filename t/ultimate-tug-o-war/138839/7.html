This plays tug-o-war AND does the combos. Please post comments.

program ultimatetugowar;
{.include SRL/SRL.scar}
 
begin
SetupSRL;
 repeat
  KeyDown(GetKeyCode('v'));
  KeyDown(GetKeyCode('x'));
  KeyDown(GetKeyCode('n'));
  Sleep(65);
  KeyUp(GetKeyCode('v'));
  KeyUp(GetKeyCode('x'));
  KeyUp(GetKeyCode('n'))
  KeyDown(GetKeyCode('b'));
  KeyDown(GetKeyCode('c'));
  KeyDown(GetKeyCode('m'));
  Sleep(65);
  KeyUp(GetKeyCode('b'));
  KeyUp(GetKeyCode('m'));
  KeyUp(GetKeyCode('c'));
  If (FindColorTolerance(x, y, 3983594, 95, 10, 180, 45, 25))Then
   begin
    KeyDown(GetKeyCode('q'));
    KeyDown(GetKeyCode('w'));
    KeyDown(GetKeyCode('e'));
    Sleep(45);
    KeyUp(GetKeyCode('q'));
    KeyUp(GetKeyCode('w'));
    KeyUp(GetKeyCode('e'));
    KeyDown(GetKeyCode('q'));
    KeyDown(GetKeyCode('w'));
    KeyDown(GetKeyCode('e'));
    Sleep(65);
    KeyUp(GetKeyCode('q'));
    KeyUp(GetKeyCode('w'));
    KeyUp(GetKeyCode('e'));
    KeyDown(GetKeyCode('q'));
    KeyDown(GetKeyCode('w'));
    KeyDown(GetKeyCode('e'));
    Sleep(65);
    KeyUp(GetKeyCode('q'));
    KeyUp(GetKeyCode('w'));
    KeyUp(GetKeyCode('e'));
   end;
 until(False);
end.
