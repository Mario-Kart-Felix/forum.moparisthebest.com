[code]{Infantry001’s AutoTalker.
Fill in the setup. Run script!
Simple as that!

Thank you for using Infantry001’s AutoTalker
}
program AutoTalker;

{.include SRL\SRL.scar}

var
sentencestyped:integer;

//SETUP\

const
Message1=‘Buyin blah blah blah’; //Type your messages here.
Message2=‘watever watever’;
Message3=‘etc. etc.’;
Message4=‘just frickin hurry up’;
Message5=‘typing messages…’;
Seconds=1; //How many seconds between typing (Preferably left at one).
progress=2; //How often (in minutes) you want a progress report.

Procedure ProgressReport;
begin
ClearDebug;
Writeln(’[]—////////////////////\\\\\\\\\\—[]’);
Writeln(’ Thanks for using Infantry001s’‘s ‘);
Writeln(’ AutoTalker ‘);
Writeln(’ Worked for:’ +ScriptTime2(2));
Writeln(’ Sentences Typed:’ +IntToStr(sentencestyped));
Writeln(’[]—////////////////////\\\\\\\\\\—[]’);
end;

Procedure talk;
begin
TypeSend(Message1);
sentencestyped :=sentencestyped+1;
wait(Seconds * 1000+random(250));
TypeSend(Message2);
sentencestyped :=sentencestyped+1;
wait(Seconds * 1000+random(250));
TypeSend(Message3);
sentencestyped :=sentencestyped+1;
wait(Seconds * 1000+random(250));
TypeSend(Message4);
sentencestyped :=sentencestyped+1;
wait(Seconds * 1000+random(250));
TypeSend(Message5);
sentencestyped :=sentencestyped+1;
wait(Seconds * 1000+random(250));
end;

begin
SetupSRL;
ActivateClient;
SetTimeOut(progress * 60000,‘ProgressReport’);
repeat
talk;
until(false)
end.[/code]

There ya go! Have fun
