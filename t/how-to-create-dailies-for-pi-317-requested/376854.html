   public int GetLastLogin(int Date)
   {
      Calendar cal = new GregorianCalendar();
      int day = cal.get(Calendar.DAY_OF_MONTH);
      int month = cal.get(Calendar.MONTH);
      int year = cal.get(Calendar.YEAR);
      int calc = ((year * 10000) + (month * 100) + day);

      return (calc - Date);
   }
