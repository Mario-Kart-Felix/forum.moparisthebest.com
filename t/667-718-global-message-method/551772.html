World.java

public static void sendWorldMessage(string msg) {
	for (Player p: getPlayers()) {
		if (p == null) {
	             continue; 
		}
		p.getPackets().sendGameMessage(msg);
	}
}

public void Yell(string msg) {
	for (Player p: World.getPlayers()) {
		if (p == null) {
	             continue; 
		}
		p.getPackets().sendGameMessage(msg);
