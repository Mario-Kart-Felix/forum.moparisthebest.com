Var k : Integer;
Begin
// Get your clients dimensions.

// For k equals 1 to 10 do this…
For k := 1 to 10 do
Begin
// Move mouse randomly
MoveMouse(Random(x2), Random(y2));
// Add 1 to k each time until it equals 10.

End;
End;[/scar]
