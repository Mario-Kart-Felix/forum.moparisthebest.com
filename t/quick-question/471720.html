I’ve added boxing arena to my server and i need the chaotic altar Object id, Anyone have it?

Here’s the boxingarena class:

package server.model.minigames;
 
import server.model.players.Client;
import server.util.Misc;
 
 
public class BoxingArena {
 
    public static final int ARENA_X = 2098;
    public static final int ARENA_Y = 4430;
     
    public static final int BLUE_GLOVES = 7673;
    public static final int RED_GLOVES = 7671;
     
    public static final int RESPAWN_X = 2097;
    public static final int RESPAWN_Y = 4425;
 
 
    public static void attemptEnter(Client c) {
        if (c.altarPrayed == 1) {
            c.sendMessage("Please pray at chaotic altar");
            return;
        }
         
        if (c.getItems().freeSlots() < 27) {
            c.sendMessage("Please empty your inventory before entering the boxing arena.");
            return;
        }
         
        for (int i = 0; i < c.playerEquipment.length; i++) {
            if (c.playerEquipment[i] != -1) {
                c.sendMessage("Please remove all equipment before entering the boxing arena.");
                return;
            }
        }
         
        c.getPA().movePlayer(2098, 4430, 0);
        c.getCombat().resetPrayers();
        c.getItems().setEquipment(getRandomGloves(), 1, c.playerHands);
        c.getCombat().getPlayerAnimIndex(c.getItems().getItemName(c.playerEquipment[c.playerWeapon]));
        c.isRunning2 = false;
        c.getPA().sendFrame36(173, 0);
        c.sendMessage("Goodluck!");
        c.teleBlockLength = 9999;
        c.teleBlockDelay = 9999;    
 
    }
     
     
    public static void attemptLeave(Client c) {
         
    if(c.underAttackBy > 0) {
            c.sendMessage("You can not teleport in combat");
            return;
        }
         
        c.getItems().deleteEquipment(1, c.playerHands);
        c.playerStandIndex = 2061;
        c.playerWalkIndex = 2064;
        c.playerRunIndex = 2077;
        c.getPA().movePlayer(2098, 4423, 0);
        c.teleBlockLength = 9999;
        c.teleBlockDelay = 9999;    
 
    }
     
    public static int getRandomGloves() {
         
        int random = Misc.random(0, 1);
         
        if (random == 1) {
            return RED_GLOVES;
        } else {
            return BLUE_GLOVES;
        }
    }
     
}
