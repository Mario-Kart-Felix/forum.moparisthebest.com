package server.model.content.music;

import server.model.players.Client;

/**
 * 
 * @author Tom
 *
 */

public class Music {
	
	public Music(int music, int swX, int swY, int neX, int neY, String name, int tabId, int array) {
		this.music = music;
		this.swX = swX;
		this.swY = swY;
		this.neX = neX;
		this.neY = neY;
		this.name = name;
		this.tabId = tabId;
		this.array = array;
	}

	public int music, swX, swY, neX, neY, tabId, array;
	public String name;
	
	public static void playMusic(Client c) {
		Music song = getMusicId(c);
		if (song == null) {
			c.outStream.createFrame(74);
			c.outStream.writeWordBigEndian(2);
			c.getPA().sendFrame126("Autumn Voyage", 4439);
			return;
		}
		c.outStream.createFrame(74);
		c.outStream.writeWordBigEndian(song.music);
		for (int i = 0; i < songs.length; i++) {
			if (unlocked[songs[i].array] == false && which == songs[i]) {
				c.sendMessage("@red@You have unlocked the song: "+songs[i].name);
				unlocked[songs[i].array] = true;
				updateList(c, songs[i].tabId);
			}
			if (which == songs[i])
		c.getPA().sendFrame126(songs[i].name, 4439);
		}
	}
	
	 public static void updateList(Client c, int id) {
	        c.getPA().sendColor(id, 0 << 10 | 255 << 5 | 0);
	    }
	 
	 public static Music which;
	 
	private static Music getMusicId(Client c) {
		int x = c.absX, y = c.absY;
		for (int i = 0; i < songs.length; i++) {
			if (x >= songs[i].swX && x <= songs[i].neX && y >= songs[i].swY && y <= songs[i].neY) {
				which = songs[i];
				return songs[i];
			}
		}
		return null;
	}
	
	public static boolean[] unlocked = {false, false};
	
	
	
	// song id, swX, swY, neX, neY, name, childId, arraySlot
	public static final Music[] songs = {
		new Music(76, 3200, 3199, 3273, 3302, "Harmony", 4344, 0), // Harmony
		new Music(2, 3200, 3303, 3273, 3353, "Autumn Voyage", 4304, 1), // Autumn Voyage
	};
}

public class MusicTab {

	public MusicTab(int songId, int buttonId, String songName, int array) {
		this.songId = songId;
		this.buttonId = buttonId;
		this.songName = songName;
		this.array = array;
	}
	
	public int songId, buttonId, array;
	public String songName;
	
	public static final MusicTab[] music = {
		new MusicTab(76, 16248, "Harmony", 0),
		new MusicTab(2, 16208, "Autumn Voyage", 1),
		new MusicTab(111, 17051, "Still Night", 2),
		new MusicTab(123, 16196, "Arabian 2", 3),
		new MusicTab(36, 16195, "Arabian", 4),
		new MusicTab(50, 1692, "Alkhaird", 5),
		new MusicTab(47, 6298, "Duel Arena", 6),
		new MusicTab(122, 17044, "Shine", 7),
	
	};
	
	public static void loadMusicTab(Client c) {
		for (int j = 0; j < Music.unlocked.length; j++) {
			for (int i = 0; i < Music.songs.length; i++) {
			if (Music.unlocked[Music.songs[i].array] == true) {
			Music.updateList(c, Music.songs[i].tabId);
			} else {
			c.getPA().sendColor(Music.songs[i].tabId, 255 << 10 |0  << 5 | 0);
			}
		}
	}
} 
	
	public static void handleClick(Client c, int Id) {
		for (int i = 0; i < music.length; i++) {
			if (Id == music[i].buttonId) {
				for (int j = 0; j < Music.songs.length; j++) {
					if (Music.unlocked[music[i].array] == false) {
						c.sendMessage("You need to unlock this song first!");
						return;
					}
				}
			c.getPA().sendFrame126(music[i].songName, 4439);
			c.outStream.createFrame(74);
			c.outStream.writeWordBigEndian(music[i].songId);
			}
		}
	}
	
}

		for (int i = 0; i < music.length; i++) {

				for (int j = 0; j < Music.songs.length; j++) {
					if (Music.unlocked[music[i].array] == false) {
						c.sendMessage("You need to unlock this song first!");
						return;
					}
				}
			c.getPA().sendFrame126(music[i].songName, 4439);
			c.outStream.createFrame(74);
			c.outStream.writeWordBigEndian(music[i].songId);
			}
		}
