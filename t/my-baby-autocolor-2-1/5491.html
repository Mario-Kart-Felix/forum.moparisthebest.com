{*******************************************************************************
function IsColorInArray (var TheColor:integer ; TheArray: array of Integer; ArraySize:integer):boolean;
By: Roberty Bob
Description: Returns true if the specified color is found in the array.
*******************************************************************************}
function IsColorInArray (var TheColor:integer ; TheArray: array of Integer; ArraySize:integer):boolean;
var
  ColorFound: boolean;
var
  i: integer;
  
begin
  ColorFound:= false;
  i:= 0;

  repeat
    if TheArray[i]= TheColor then
    begin
      ColorFound:= true;
      break;
    end;
    i:= i+1;
  until i = (ArraySize)

result:= ColorFound;
end;

{*******************************************************************************
function FindAllColorsInBoxTolerance(var x,y: integer; ColorArray: array of Integer; BoxSize,Tol:integer):boolean;
By: Roberty Bob
Description: Pass in an array of object colors with principal reference color in Array[0]
Returns true if all the colors were found in a box surrounding the principal reference color.
*******************************************************************************}
function FindAllColorsInBoxTolerance(var x,y: integer; ColorArray: array of Integer; BoxSize,Tol:integer):boolean;

var
i,j,a1,b1,HBox,ArraySize,Sx,Sy,Ex,Ey: integer;
var
AllFound,ColorMissing: boolean;

begin
//******** Set these to your specific search area, to make function more efficient.
Sx:= MSX1; //currently main screen
Sy:= MSY1;
Ex:= MSX2;
Ey:= MSY2;
//********
HBox:= (BoxSize/2);
ColorMissing:= false;
AllFound:= false;
ArraySize:= GetArrayLength(ColorArray);
x:= MSCX-250+random(500);
y:= MSCY-200+random(400);

repeat
FindColorSpiralTolerance(x,y,ColorArray[0],Sx,Sy,Ex,Ey,Tol);
a1:= x;
b1:= y;
i:= 0;
repeat
if FindColorSpiralTolerance(a1,b1,ColorArray[i],a1-(HBox),b1-(HBox),a1+(HBox),b1+(HBox),Tol) then
AllFound:=true
else
begin
ColorMissing:= true;
AllFound:= false;
break;
end;

  i:= i+1;
until i= (ArraySize-1)
if ColorMissing then break;
j:= j+1;

until AllFound or (j=10)

result:= AllFound;
end;

{*******************************************************************************
function AutoColor;
By: Roberty Bob
Description: Creates an array of unique colors for an object using a specified reference
color array, returns true if the reference array is completely refreshed.
*******************************************************************************}
function AutoColor(ObjText:string; ColorArray: array of Integer; BlackListSize:integer):boolean;
var
A,B,C, A1,B1,C1,A2,B2,C2, incA,incB,incC, incA1,incB1,incC1,incA2,incB2,incC2,
i,j, DynCount, ToleranceLimit, SearchBoxSizeLimit, ColorArraySize,
ACtempx, ACtempy, TempColor, TempMS : integer;
var
TolGood, SearchBoxGood,RefColorGood,
ColorArrayFilled, Black, Debug, OnAWinner,
EnableDualLayerSearch,EnableScriptDebugging: boolean;
var
XferColorArray: array of integer;
BlackListColorArray: array of integer;
var
DynString: string;

begin
//+++++++++++ SETUP ++++++++++++++++
//================================//
EnableScriptDebugging:= True;
EnableDualLayerSearch:= True ;
//Dual Layer will normally be overkill but it if want gaurenteed success
//================================//

// Initialise misc. variables
TempMS:= mousespeed;
benmouse:= false;
mousespeed:= 20;

// Initial loop values | boxsize | tolerance | array[ref] |
A1:= 200; // First layer - cursory scan
B1:= 5;
C1:= 1;
A2:= 10; // Second layer- intensive failproof scan.
B2:= 5;
C2:= 0;
// Loop increment sizes
incA1:= 40; // First layer - cursory scan
incB1:= 4;
incC1:= 2
incA2:= 20; // Second- more detailed scan
incB2:= 5;
incC2:= 1;
// Limiting values for the loops.
ToleranceLimit:= 20;
SearchBoxSizeLimit:= 500; //maximum size of the rectangle formed around the player.

// Initialise booleans.
TolGood:= false;
SearchBoxGood:= false;
RefColorGood:= false;
ColorArrayFilled:= false;
OnAWinner:= false;
Black:= false;
debug:= EnableScriptDebugging;

ColorArraySize:= GetArrayLength(ColorArray);
SetArrayLength(XferColorArray,ColorArraySize);
SetArrayLength(BlackListColorArray,BlackListSize);

// Initialise Arrays
i:= 0;
for i:= 0 to ColorArraySize-1 do
begin
XferColorArray[i]:= 0;
i:= i+1;
end;

i:= 0
for i:= 0 to (BlackListSize-1) do
begin
BlackListColorArray[i]:= 0;
i:=i+1;
end;

if debug then writeln(‘Initialised’);
i:= 0;
mmouse(MSCX,MSCY,10,10);
wait(100+random(100));
//-----------The Triple Iteration-----------------------------------------------
if debug then writeln(‘Starting the multilayered triple loop’);
j:= 0;
if not EnableDualLayerSearch then j:=1;
repeat

if j=0 then
begin
if debug then writeLn(‘first tier search commencing’);
A:= A1;
B:= B1;
C:= C1;
incA:= incA1;
incB:= incB1;
incC:= incC1;
end

else
begin
if debug then writeLn(‘second tier search commencing’);
A:= A2;
B:= B2;
C:= C2;
incA:= incA2;
incB:= incB2;
incC:= incC2;
end;

repeat
  if debug then writeLn('incA ='+inttostr(incA));
  B:= 0;
  repeat
    if debug then writeLn('incB ='+inttostr(incB));
    C := 0;
    repeat
      if debug then writeLn('incC ='+inttostr(incC));

      black:= false;
      if not OnAWinner then
      begin
      ACtempX:= MSCX-250+random(500);
      ACtempY:= MSCY-160+random(320);
      FindAllColorsInBoxTolerance(ACtempX,ACtempY,ColorArray,40,20);
      end;

      if debug then writeln('searching for color within limits');
      if FindColorSpiralTolerance(ACtempX,ACtempY,ColorArray[C],(MSCX-A),(MSCY-((2/3)*A)),(MSCX+A),(MSCY+((2/3)*A)),(B))
      and not IsColorInArray(ColorArray[C],BlackListColorArray,BlackListSize) then
      begin

        if debug then writeln('I''ve found a potential color for further checking!');
        MMouse(ACtempX,ACtempY,0,0);
        wait(30+random(30));

        if IsUpText(ObjText) then
        begin
          OnAWinner:=true;
          if debug then writeln('I''ve now confirmed the color exactly.');
          TempColor:= getcolor(ACtempX,ACtempY);
          i:= 0;
          repeat
            if ColorArrayFilled then break;
            begin
              if (XferColorArray[i]=0) or (XferColorArray[i]=TempColor) then
              begin
                if debug then writeln('filling temporary color transfer array');
                begin
                  XferColorArray[i]:= TempColor;
                  break;
                end;
              end;
              i:= i+1
            end;
          until (i=ColorArraySize)

          if not (XferColorArray[ColorArraySize-1]=0) then
          begin
            ColorArrayFilled:= true;
            i:= 0;
            if debug then writeln('changing the array');
            repeat
              begin
                ColorArray[i]:= XferColorArray[i];
                i:= i+1;
              end;
            until (i=ColorArraySize)
          end;
          begin
            ACtempX:=ACtempX-7+random(14);
            ACtempY:=ACtempX-7+random(14);
          end;
        end

        else
        begin
          black:= true;
          repeat
          if ColorArrayFilled then break;
            begin
              if (BlackListColorArray[i]=0) or (BlackListColorArray[i]=TempColor) then
              begin
                if debug then writeln('blacklisting a wrong color');
                BlackListColorArray[i]:= TempColor;
                break;
              end;
              i:= i+1
            end;
          until (i=BlackListSize)
        end;
      end;


      C:= C+incC;
    until (C>=ColorArraySize-1) or RefColorGood or ColorArrayFilled

    B:= B+incB;
  until (B>=ToleranceLimit) or TolGood or ColorArrayFilled

  A:= A+incA;
until (A>=(SearchBoxSizeLimit/2)) or SearchBoxGood or ColorArrayFilled

j:= j+1;
until (j>=2)

DynCount:= 0;
DynString:= ‘( ‘;
WriteLn(’’);
WriteLn(‘The colors for your chosen object have been selected’);

i:= 0;
repeat
if not (XferColorArray[i]=0) then
begin
DynCount:= DynCount+1;
DynString:= DynString+inttostr(XferColorArray[i])+’ ‘;
end;
i:= i+1;
until (i=ColorArraySize)
writeLn(‘and ‘+inttostr(DynCount)+’/’+inttostr(ColorArraySize)+’ were dynamically changed from the reference array.’);
if not (XferColorArray[0]=0) then
begin
DynString:= DynString+’)’;
writeLn('to become '+DynString);
end;

if (XferColorArray[0]=0) then
  begin
  writeLn('You really need to change the parameters. Try increasing the tolerance?');
  writeLn('');
  mousespeed:= TempMS;
  end;
mousespeed:= TempMS;

result:= ColorArrayFilled;
end;

begin
SetArray;
AutoColor(‘Attack’,ChickenColor,10);
end.[/code]
