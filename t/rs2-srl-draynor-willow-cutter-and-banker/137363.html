function FindFastRandoms: Boolean; // By WT-Fakawi, Modded by Nashi, Used By Script_Owned4
var
i: Integer;
begin
for I := 1 to 12 do
begin
case I of
1: if FindDead then
Result := True;
2: if FindMod then
Result := True;
3: if FindMime then
Result := True;
4: if FindMaze then
Result := True;
5: if FindQuiz then
Result := True;
6: if FindDemon then
Result := True;
7: if InBlack then
begin
Result := True;
Exit;
end;
8: if NoGameTab then
begin
Result := True;
Exit;
end;
9 : if RC then
Result := True;
10: if FindTalk then
Result := True;
11: if FindFight then
begin
RunTo(‘W’, True);
end;
end;
Wait(1);
end;
end;

procedure chopdown;
begin
Wait(19000);
if(FindDTM(WillowTree2,x,y,x1,y1,x2,y2))or

Writeln(‘Cut Down A Tree’);
end;
