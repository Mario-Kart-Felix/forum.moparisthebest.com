class Main {

private static String getCookies() throws Exception {
	String cookies = "";
	URLConnection url = new URI("http://www.acagames.ulmb.com/register.php").toURL().openConnection();
	url.connect();

	Map<String, List<String>> headers = url.getHeaderFields();
	for (String s : headers.get("Set-Cookie")) {
		String[] split = s.split(";");
		for (int i = 0; i < (split.length - 1); i++) {
			cookies = split[i] + "; " + cookies;
		}
	}
	return cookies.substring(0, cookies.length() - 2);
}

private static String genUser(int length) {
	String user = "";
	for (int i = 0; i < length; i++) {
		user += (char) (97 + Math.random() * (122 - 97));
	}
	return user;
}

public static void main(String[] args) {
