	{
		Converter converter = new Converter();
		System.out.println( converter.toBinary( "a b c" ) );
		System.out.println( converter.toString( converter.toBinary( "a b c" ) ) );
	}


	private final String toBinary( String string )
	{
		byte[] bytes = string.getBytes( Charset.forName( "UTF-8" ) );
		StringBuilder bldr = new StringBuilder( string.length() );
		for( byte value: bytes ) {
			bldr.append( String.format( "%8s", Integer.toBinaryString( value & 0xFF ) ).replace( ' ', '0' ) );
		}
		return bldr.toString();
	}


	private final String toString( String string )
	{
		byte[] buffer = new byte[ string.length() / 8 ];
		for( int index = 0, pos = 0; index < buffer.length; index ++ ) {
			String temp = string.substring( pos, pos + 8 );
			buffer[ index ] = ( byte )Integer.parseInt( temp, 2 );
			pos += 8;
		}
		return new String( buffer, Charset.forName( "ISO-8859-1" ) );
	}


	private Converter()
	{

	}

}

import java.util.Vector;

public class Converter {

	/**
	 * @param args
	 */
	public static void main(String[] args) {

		String example = "This is a test.";
		char[] exampleChars = example.toCharArray();
		String binaryString = "";
		for (char c : exampleChars) {
			String charBits = Integer.toBinaryString((int)c);
			while(charBits.length() < 8)
				charBits = "0"+charBits;
			binaryString += charBits;
		}
		System.out.println(binaryString);
		
		Vector<String> splitBits = new Vector<String>();
		for(int i = 0; i < binaryString.length(); i+=8) {
			splitBits.add(binaryString.substring(i, i+8));
		}
		String completeString = "";
		for(Object o : splitBits.toArray()) {
			String p = (String)o;
			completeString += (char)Integer.parseInt(p,2);
		}
		System.out.println(completeString);
	}

}

Far from efficient but it works perfectly
