{*******************************************************************************
function GetTraderName: String;
By: masquerader
Description: Returns the name of the person you are trading with.
*******************************************************************************}

function GetTraderName: string;
var
colonx, colony: Integer;
begin
if (TradeScreen) then
begin
if (IsTextInAreaEx(2, 31, 477, 50, colonx, colony, ‘:’, 0, upchars, False,
True, 0, 0, 65535)) then
begin
Result := Trim(GetTextAtEx(colonx + 8, colony, 0, upchars, False, True, 0,
2, 65535, 12, False, tr_AlphaNumericChars))
end
else
WriteLn(‘Couldnt get trader name’);
end;
end;

{*******************************************************************************
function AcceptTrade: Boolean;
By: masquerader
Description: Accepts the trade on the last line in the chatbox.
*******************************************************************************}

function AcceptTrade: Boolean;
//only works with last line, turn public chat off
var
timeout: Integer;
fx, fy: Integer;
begin
if FindColor(Fx, Fy, 8388736, 21, 416, 29, 425) then
begin
Mouse(Fx + 2, Fy + 12, 2, 7, True);
MarkTime(timeout);
repeat
Wait(500 + Random(100))
if (TimeFromMark(timeout) > 10000) then
Exit;
until (TradeScreen);
Result := True;
end;
end;

{*******************************************************************************
procedure Accept;
By: masquerader
Description: Accepts the opened trade.
*******************************************************************************}

procedure Accept;
begin
if (TradeScreen) then
begin
Mouse(264, 194, 20, 10, True);
Wait(1000 + Random(500));
end else
if (TradeScreen2) then
begin
Mouse(228, 310, 20, 4, True);
Wait(1000 + Random(500));
end
end;

{*******************************************************************************
procedure Decline;
By: masquerader
Description: Declines the opened trade.
*******************************************************************************}

procedure Decline;
begin
if (TradeScreen) then
begin
Mouse(259, 264, 20, 10, True);
Wait(1000 + Random(500));
end else
if (TradeScreen2) then
begin
Mouse(296, 308, 20, 4, True);
Wait(1000 + Random(500));
end
end;
{*******************************************************************************
function Waiting: Boolean;
By: masquerader
Description: Returns true if you are waiting for the other player to accept the trade.

function Waiting: Boolean;

if (TradeScreen) then

else if (TradeScreen2) then

else if (TradeScreen2) then
Result := (GetColor(228, 45) = 16776960)

{*******************************************************************************
function CheckItem(them: Boolean; name: String): Integer;
By: masquerader modified by Pyro and Cheesehunk
Description: returns number item at second screen
it might mess up because it goes out of the box/off the screen,
but thats not my fault
make the first letter capital, and the rest lowercase
*******************************************************************************}

function CheckItem(them: Boolean; name: string): Integer;

//them, or you
var
number: string;
itemx, itemy, i: Integer;
add: Integer;
x, y: Integer;
x1, y1, x2, y2: Integer;
begin
if (not (TradeScreen2)) then
Exit;
if (them) then
begin
x1 := 262;
y1 := 85;
x2 := 514;
y2 := 337
end else
begin
x1 := 0;
y1 := 83;
x2 := 258;
y2 := 335
end;

while (IsTextInAreaEx(x1, y1, x2, y2, itemx, itemy, name, 0, upchars, True,
True, 0, 2, 16777215)) do
begin
if (IsTextInAreaEx(itemx, itemy, x2, itemy, x, y, ‘(’, 0, upchars, True,
True, 0, 2, 16777215)) then
begin
number := Trim(GetTextAtEx(x + 3, y, 0, upchars, True, True, 0, 2,
16777215, 12, False, tr_digits));
for i := 1 to Length(number) do
begin
if (Copy(number, i, 1) = ’ ') then Delete(number, i, 1)
end;
add := StrToInt(number);
Result := Result + add;
end else
if (IsTextInAreaEx(itemx, itemy, x2, itemy, x, y, ‘x’, 0, upchars, True,
True, 0, 2, 16777215)) then
begin
number := Trim(GetTextAtEx(x + 10, y, 0, upchars, True, True, 0, 2,
16777215, 12, False, tr_digits));
for i := 1 to Length(number) do
begin
if (Copy(number, i, 1) = ’ ') then Delete(number, i, 1)
end;
add := StrToInt(number);
Result := Result + add;
end else
Result := Result + 1;
y1 := itemy + 12;
end;
end;

{*******************************************************************************
function TrimTradeName(TheString: String): String;
By: Wizzup?
Description: Trims the string to trader name.
*******************************************************************************}

Function TrimTradeName(TheString:String):String;
Var
StrLength, I, E: Integer;
TempStr: String;

Begin
StrLength := Length(TheString);
WriteLn(TheString);
While I < StrLength Do
Begin
I := I+1;
E := I;
If StrGet(TheString, I) = ’ ’ Then
Begin
If I - StrLength >= 2 Then Break;
TempStr := StrGet(TheString, I+1) + StrGet(TheString, I+2);
If TempStr = ‘wi’ Then
Begin
SetLength(TheString,I-1);
Result := TheString;
WriteLn(Result);
Break;
End;
End;
End;
End;

{*******************************************************************************
function TrimTradeNameEx(TheString: String): String;
By: Wizzup?
Description: Trims the string to trader name.
*******************************************************************************}

Function TrimTradeNameEx(TheString:String):String;
Var
A, StrLength: Integer;

Begin
StrLength := Length(TheString);
WriteLn(TheString);
A := Pos(‘wis’, TheString);
If A = 0 Then A := Pos(‘shes’, TheString);
If A <> 0 Then
Begin
SetLength(TheString, A-2);
Result := TheString;
End;
WriteLn(TheString);

End;

Function SomeoneTrades: Boolean;

Var

Result := FindColor(FPoint.X, FPoint.Y, 8388736, 30, 418, 40, 427);

{*******************************************************************************
function HandleTrade: Boolean;
by: Wizzup?
Description: Handle’s a trade.
*******************************************************************************}

Function HandleTrade: Boolean;

Var
TheTrader : String;
FPoint : TPoint;
TimeMark, I: Integer;
ItemPuttedUp, HeDeclined, WeDeclined, DeclinedI, SameTrader:Boolean;

Begin
If SomeoneTrades Then
Begin
If Not (TimeFromMark(TradeMark) > 20000) Then
Begin
NOTrades := NOTrades+1;
Result := True;
Exit;
End;
SetArrayLength(Traders, NumberOfTraders+1);
TheTrader := LowerCase(Trim(GetTextAtEx(20, 415, 35, SmallChars, False, False,
0, 2, 8388736, 20, True, tr_NormalChars)));
TheTrader := TrimTradeNameEx(TheTrader);

For I:=0 To NumberOfTraders Do
Begin
 TheTrader := LowerCase(TheTrader);
 WriteLn(Traders[I]);
 If TheTrader = Traders[I] Then SameTrader := True;
 If SameTrader And Not TradeSameTrader Then
   Begin
     WriteLn('Same Trader');
     Case Random(4) Of
       0: Typesend(TheTrader+', do not trade again.');
       1: Typesend('Im not trading with you again '+TheTrader+'.');
       2: Typesend('Stop trading '+TheTrader+'.');
       3: Typesend('I remember you. '+TheTrader+', do not trade again.');
     End;
     NOTrades := NOtrades+1;
     Wait(1000+Random(1000));
     MarkTime(TradeMark);
     Exit;
   End;
End;

If FindColor(FPoint.X, FPoint.Y, 8388736, MCX1 , MCY1 , MCX2, MCY2) Then
Begin
  Mouse(FPoint.X+100, FPoint.Y+3, 80, 2, True);

  MarkTime(TimeMark);
  Repeat
    Wait(1000);
    If TimeFromMark(TimeMark) > 5000 Then
    Begin
      Decline;
      WeDeclined := True;
      Break;
    End;
  Until (TradeScreen);
  WriteLn('TradeScreen popped up.');

  MarkTime(TimeMark);
  Repeat
    If WeDeclined Then Break;
    If HeDeclined Then Break;
    Wait(1000);
    If TimeFromMark(TimeMark) > 20000 Then
    Begin
      Decline;
      WeDeclined := True;
      Break;
    End;
    If Not TradeScreen Then
      Begin
        HeDeclined := True
        Break;
      End;
  Until (PlayerAccepted);
  WriteLn(TheTrader+' Accepted, we might if he putted up an item...checking..');

  If FindColor(FPoint.X, FPoint.Y, 65536, 40, 75, 490, 310) Then
    ItemPuttedUp := True;
  If ItemPuttedUp Then WriteLn(TheTrader+' Putted up an item, or more..')
  Else WriteLn(TheTrader+' Did not put up an item.');

  If Not ItemPuttedUp Then
    Begin
      If Random(2) = 0 Then
      Begin
        WeDeclined := True;
        Decline;
        Wait(1000);
      End;
    End;
  Accept;

  MarkTime(TimeMark);
  Repeat
    If WeDeclined Then Break;
    If HeDeclined Then Break;
    Wait(1000);
    If TimeFromMark(TimeMark) > 20000 Then
    Begin
      Decline;
      WeDeclined := True;
      Break;
    End;
    If Not TradeScreen2 Then
      Begin
        HeDeclined := True
        Break;
      End;
  Until (PlayerAccepted);
  Accept;

  If WeDeclined Then WriteLn('We declined, it either took too long of he did not put up an item.');
  If HeDeclined Then WriteLn('He Declined. Asking him to stop trading.');
  If (HeDeclined or WeDeclined) Then DeclinedI := True;

  If Not DeclinedI Then WriteLn('Trade SuccesFull! Now responding with text.')
  Else WriteLn('Player Declined, now telling him to stop trading.');
  If Not DeclinedI Then
  Begin
    If ItemPuttedUp Then
    Begin
      Case Random(5) Of
        0: Typesend('Thanks '+TheTrader+'.');
        1: Typesend('Ty '+TheTrader+'.');
        2: Typesend('=p ty '+TheTrader+'.');
        3: Typesend('Thanks.');
        4: Typesend('=d Thanks '+TheTrader+'.');
        End;
    End
    Else
    Begin
      Case Random(4) Of
      0: Typesend(TheTrader+', do not trade again.');
      1: Typesend('Dont trade if you have nothing to offer '+TheTrader+'.');
      2: Typesend('Stop trading '+TheTrader+'.');
      3: Typesend('Dont trade if you got nothing for me '+TheTrader+'.');
      End;
    End;
  End
  Else
  Begin
    Case Random(4) Of
      0: Typesend(TheTrader+', do not trade again.');
      1: Typesend('Dont trade if you have nothing to offer '+TheTrader+'.');
      2: Typesend('Stop trading '+TheTrader+'.');
      3: Typesend('Dont trade if you got nothing for me '+TheTrader+'.');
    End;
  End;
End;
Result := True;
WriteLn('"Solved" a trade!');
Trades := Trades+1;
Traders[NumberOfTraders] := TheTrader;
NumberOfTraders := NumberOfTraders+1;
MarkTime(TradeMark);

End;
End;
