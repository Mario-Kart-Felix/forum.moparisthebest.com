Noun: Array [0…NN] of String;
Verb: Array [0…NV] of String;
PreVerb: Array [0…NPR] of String;
PostVerb: Array [0…NPV] of String;

Begin
PreAdj[0]:='very ’

Hey, to make your script a little better, try switching the words ‘is’ and ‘was’.

Here is program I made with a function called RandSwitch();

program New;

function RandSwitch(Text,Text2 : string) : string;
begin
  if(Random(2) = 0)then
    Result := Text
  else
    Result := Text2;
end;

begin
  ClearDebug;
  WriteLn(RandSwitch('is','was'));
end.

Also, you can make your arrays like this instead and you don’t have to set a value like 0…45 or something.

program New;

var
  i : integer;
  MyArray : array of string;

begin
  ClearDebug;
  MyArray := ['lol','rofl','haha','jaja','lmao'];
  for i := 0 to GetArrayLength(MyArray)-1 do
    WriteLn('I told him my dog died and he had the odacity to say, "'+MyArray[i]+'"!');
end.
