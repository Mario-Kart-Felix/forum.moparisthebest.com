Copy paste this into it:

[/size][/font][code]
package com.rs2.model.content.skills;

import com.rs2.model.World;
import com.rs2.model.npcs.Npc;
import com.rs2.model.players.Player;
import com.rs2.model.players.Item;

public class CuttingGems {

public static void handleChisel(Item id4, Item id5, Player c) {

int id1 = id4.getId();
int id2 = id5.getId();
if (id1 == 1755)
cutGem(id2, c);
else
cutGem(id1, c);
}

public static int[][] gems = {{1623,1607},{1621,1605},{1619,1603},{1617,1601},{1631,1615},{6571,6573}};
public static int[][] anims = {{888},{889},{887},{890},{885},{2717}};
public static int[][] level = {{20},{27},{34},{43},{55},{67}};
public static int[][] XP = {{70},{150},{260},{500},{700},{1500}};
public static void cutGem(int id, Player c) {
for (int j = 0; j < gems.length; j++) {
if (gems[j][0] == id) {
if (c.getSkill().getLevelForXP(c.getSkill().getExp()[12]) >= XP[j][0]) {
c.getInventory().removeItem(new Item(id));
c.getInventory().addItem1(new Item(gems[j][1]), 1);
c.getUpdateFlags().sendAnimation((anims[j][0]), 0);
c.getSkill().addExp((XP[j][0]), 12);
break;
} else {
c.getActionSender().sendMessage(“You need a crafting level of " + level[j][0] + " to craft this.”);
}
}
}
}

/**
 * Represents the different types of gems found in the game.
 * 
 * @author Zyle Moore
 * @version 0.1
 */
public enum Gem {
    OPAL(1609, 1625, 1, 15),
    JADE(1611, 1627, 13, 20),
    RED_TOPAZ(1613, 1629, 16, 25),
    SAPPHIRE(1607, 1623, 20, 50),
    EMERALD(1605, 1621, 27, 57.5F),
    RUBY(1603, 1619, 34, 85),
    DIAMOND(1601, 1617, 43, 107.5F),
    DRAGONSTONE(1615, 1631, 55, 137.5F),
    ONYX(6573, 6571, 67, 167.5F);

    private final int uncutId;
    private final int cutId;
    private final int levelReq;
    private final float exp;

    private Gem(final int uncutId, final int cutId, final int levelReq,
        final float exp) {
        this.uncutId = uncutId;
        this.cutId = cutId;
        this.levelReq = levelReq;
        this.exp = exp;
    }

    /**
     * Returns the Item id of the uncut gem.
     * 
     * @return the uncut gem id
     */
    public int getUncutId() {
        return uncutId;
    }

    /**
     * Returns the Item id of the cut gem.
     * 
     * @return the cut gem id
     */
    public int getCutId() {
        return cutId;
    }

    /**
     * Returns the Crafting level required to cut the gem.
     * 
     * @return the level requirement
     */
    public int getLevelReq() {
        return levelReq;
    }

    /**
     * Returns the experience gained from successfully cutting the gem.
     * 
     * @return the experience gained
     */
    public float getExp() {
        return exp;
    }
}
