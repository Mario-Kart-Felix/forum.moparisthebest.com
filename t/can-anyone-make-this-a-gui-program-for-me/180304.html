public void paint(Graphics g) {
	int	i,j;
	int	ix,iy;
	int	col;
	float	h,s,b;

public void paintValue(Graphics g) {

	String	wks,outs;

	h = (float)hValue;
	s = (float)sValue;
	b = (float)bValue;

	green = (col-red*256*256)/256;
	blue = col-red*256*256-green*256;
