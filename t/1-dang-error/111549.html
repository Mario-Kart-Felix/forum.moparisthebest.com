public int GetNPCID(int coordX, int coordY) {

		if (server.npcHandler.npcs[i] != null) {
			if (server.npcHandler.npcs[i].absX == coordX && server.npcHandler.npcs[i].absY == coordY) {
				return server.npcHandler.npcs[i].npcType;
			}

	}
