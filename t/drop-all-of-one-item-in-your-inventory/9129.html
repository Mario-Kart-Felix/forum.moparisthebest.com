[scar]program Test;
{.include SRL\SRL.scar}

Procedure DropAllOf(ItemColour: Integer; ItemName: String);
var
xv,yx: integer;
begin
Repeat
If (FindColor(xv, yx, ItemColour, MIX1, MIY1, MIX2, MIY2)) Then
begin
MMouse(xv, yx, 5, 5);
Wait(300);
GetMousePos(xv, yx);
If (IsUpText(ItemName)) then
begin
Mouse(xv, yx, 0, 0, False);
Wait(300);
ChooseOption(xv, yx, ‘Drop’);
Wait(1000);
end;
end;
Until(Not (FindColor(xv, yx, ItemColour, MIX1, MIY1, MIX2, MIY2)));
end;

begin
SetUpSRL;
ActivateClient;
DropAllOf(3636181, ‘Copper ore’);
end.[/scar]
