            {
                turn = 1;
                Console.WriteLine("Type(Executable, printIpInfo): ");
                Console.WriteLine();
                string Type = Console.ReadLine();
                if (Type == "Executable")
                {
                    Console.WriteLine("Enter Info: (Example: pathToExecutable args1 args2 args3) ");
                    string execute = Console.ReadLine();
                    string[] executedArgs = execute.Split(' ');
                    Process pro = new Process();
                    pro.StartInfo.FileName = @execute;
                    pro.StartInfo.Arguments = executedArgs[0];
                    pro.Start();

            }
        }
