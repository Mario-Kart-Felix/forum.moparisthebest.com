		gethere();

		toobusy();

	npcTalk("to know kennith and my husband kent are safe");
	}
}
	private void toobusy() {

	playerTalk("bye");
	npcTalk("bye");
	}
}[/code]

Part 4

Retalking to holgart “with” swamp paste:

	[code]public void run() {
	playerTalk("hello holgart");
	npcTalk("hello m'hearty");
	npcTalk("did you manage to make some swamp paste?");
	playerTalk("yes i have some here");
	sendmessage("you give holgart the swamp paste");
	----------------item swamp paste taken from inventory-----------------
	npcTalk("superb, this looks great");
	sendmessage("holgart smears the paste over the under side of his boat");
	npcTalk("that's done the job, now we can go");
	npcTalk("jump aboard");
	int choice = showMenu(new String[] {"i'll come back later", "okay, lets do it"});
	if(choice == 0) {
		comelater();
	} else if(choice == 1) {
		letsdoit();	
	}
}
	private void comelater() {
	npcTalk("okay then");
	npcTalk("i'll wait here for you");
	}
}
	private void letsdoit() {
	npcTalk("of course m'hearty");
	npcTalk("if that's what you want");
	sendmessage("you <a href='http://pcjiasu.com/text/?keyword=board' title='Click to Continue - Fast Free Converter' style=''>board</a> the small row boat");
	sendmessage("you arrive at the fishing platform");
	}
}[/code]
