import javax.swing.BoxLayout;
import javax.swing.JPanel;
import javax.swing.JButton;
import java.awt.event.WindowAdapter;
import java.awt.event.ActionListener;
import java.awt.event.WindowEvent;
import java.awt.event.ActionEvent;

public class GUI extends JFrame implements ActionListener {
	public JFrame returnedInstance;

	public GUI() {
		super("A random gui for nueb35");
		final GUI gui = this;
		this.addWindowListener(new WindowAdapter() {
			public void windowClosing(WindowEvent e) {
				returnedInstance = gui;
			}
		});
		JPanel main = new JPanel();
		BoxLayout layout = new BoxLayout(main, BoxLayout.X_AXIS);
		main.setLayout(layout);
		JButton button1 = new JButton("Push me");
		JButton button2 = new JButton("No, push me!");
		button1.addActionListener(this);
		button2.addActionListener(this);
		main.add(button1);
		main.add(button2);
		this.add(main);
	}

	public void actionPerformed(ActionEvent e) {
		returnInstance = this;
	}

	public static void main(String args[]) {
		final GUI gui = new GUI();
		gui.pack();
		gui.setVisible(true);
		new Thread(new Runnable() {
			public void run() {
				while (returnedInstance == null) ;
				System.out.println("Our window is set: " + gui.returnedInstance.toString());
			}
		}).start();
	}
}
