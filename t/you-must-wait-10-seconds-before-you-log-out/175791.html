public void entangle()

public void inCombat() { LogoutDelay = 30; }

You can change the logout delay to how you want it, 30 is about 20 seconds, which I feel is the best time.

Then search:

public int EntangleDelay = 0;

Above that, add this:

public int LogoutDelay = 0;

After you have added that int, search:

if(EntangleDelay > 0) EntangleDelay -= 1;

Above that, add this:

if(LogoutDelay > 0) LogoutDelay -= 1;

case 9154: //Log out { if ((absX >= 0000 && absX <= 0000 && absY >=0000 && absY <=0000 ) || (absX >= 0000 && absX <= 0000 && absY >=0000 && absY <=0000)) { sendMessage("Logout has been disabled for this area."); } else { if (LogoutDelay >= 1) { sendMessage("You must wait 20 seconds after combat to log out!"); } else if (LogoutDelay == 0) { logout(); savefile = true; } } } break;
